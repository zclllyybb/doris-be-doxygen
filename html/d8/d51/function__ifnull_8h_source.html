53 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
54 if (arguments[0]->is_nullable()) {
55 return arguments[1];
56 }
57 return arguments[0];
58 }
59
60 bool use_default_implementation_for_nulls() const override { return false; }
61
62 // ifnull(col_left, col_right) == if(isnull(col_left), col_right, col_left)
63 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
64 size_t result, size_t input_rows_count) override {
65 ColumnWithTypeAndName& col_left = block.get_by_position(arguments[0]);
66 if (col_left.column->only_null()) {
67 block.get_by_position(result).column = block.get_by_position(arguments[1]).column;
68 return Status::OK();
69 }
70
71 ColumnWithTypeAndName null_column_arg0 {nullptr, std::make_shared<DataTypeUInt8>(), ""};
72 ColumnWithTypeAndName nested_column_arg0 {nullptr, col_left.type, ""};
73
74 col_left.column = col_left.column->convert_to_full_column_if_const();
75
77 if (auto* nullable = check_and_get_column<ColumnNullable>(*col_left.column)) {
78 null_column_arg0.column = nullable->get_null_map_column_ptr();
79 nested_column_arg0.column = nullable->get_nested_column_ptr();
80 nested_column_arg0.type =
81 reinterpret_cast<const DataTypeNullable*>(nested_column_arg0.type.get())
82 ->get_nested_type();
83 } else {
84 block.get_by_position(result).column = col_left.column;
85 return Status::OK();
86 }
87 const ColumnsWithTypeAndName if_columns {
88 null_column_arg0, block.get_by_position(arguments[1]), nested_column_arg0};
89
90 // see get_return_type_impl
91 // if result is nullable, means both then and else column are nullable, we use original col_left to keep nullable info
92 // if result is not nullable, means both then and else column are not nullable, we use nested_column_arg0 to remove nullable info
93 bool result_nullable = block.get_by_position(result).type->is_nullable();
94 Block temporary_block({
95 null_column_arg0,
96 block.get_by_position(arguments[1]),
97 result_nullable
98 ? col_left
99 : nested_column_arg0, // if result is nullable, we need pass the original col_left else pass nested_column_arg0
100 block.get_by_position(result),
101 });
102
103 auto func_if = SimpleFunctionFactory::instance().get_function(
104 "if", if_columns, block.get_by_position(result).type);
105 func_if->execute(context, temporary_block, {0, 1, 2}, 3, input_rows_count);
106 block.get_by_position(result).column = temporary_block.get_by_position(3).column;
107 return Status::OK();
108 }
109};
