50template <typename ConcreteAction, typename Name>
51class FunctionArrayIndex : public IFunction {
52public:
53 using ResultType = typename ConcreteAction::ResultType;
54
55 static constexpr auto name = Name::name;
56 static FunctionPtr create() { return std::make_shared<FunctionArrayIndex>(); }
57
59 String get_name() const override { return name; }
60
61 bool is_variadic() const override { return false; }
62
63 size_t get_number_of_arguments() const override { return 2; }
64
65 bool use_default_implementation_for_nulls() const override { return false; }
66
67 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
68 // DCHECK(is_array(arguments[0]));
69 return make_nullable(std::make_shared<DataTypeNumber<ResultType>>());
70 }

72 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
73 size_t result, size_t input_rows_count) override {
74 return _execute_non_nullable(block, arguments, result, input_rows_count);

78 ColumnPtr _execute_string(const ColumnArray::Offsets64& offsets, const UInt8* nested_null_map,
79 const IColumn& nested_column, const IColumn& right_column,
80 const UInt8* right_nested_null_map, const UInt8* outer_null_map) {
81 // check array nested column type and get data
82 const auto& str_offs = reinterpret_cast<const ColumnString&>(nested_column).get_offsets();
83 const auto& str_chars = reinterpret_cast<const ColumnString&>(nested_column).get_chars();
84
85 // check right column type and get data
86 const auto& right_offs = reinterpret_cast<const ColumnString&>(right_column).get_offsets();
87 const auto& right_chars = reinterpret_cast<const ColumnString&>(right_column).get_chars();
88
89 // prepare return data
90 auto dst = ColumnVector<ResultType>::create(offsets.size());
91 auto& dst_data = dst->get_data();
92 auto dst_null_column = ColumnUInt8::create(offsets.size());
93 auto& dst_null_data = dst_null_column->get_data();
94
95 // process
96 for (size_t row = 0; row < offsets.size(); ++row) {
97 if (outer_null_map && outer_null_map[row]) {
98 dst_null_data[row] = true;
99 continue;
100 }
101 dst_null_data[row] = false;
102 ResultType res = 0;
103 size_t off = offsets[row - 1];
104 size_t len = offsets[row] - off;
105
106 size_t right_off = right_offs[row - 1];
107 size_t right_len = right_offs[row] - right_off;
108 for (size_t pos = 0; pos < len; ++pos) {
109 // match null value
110 if (right_nested_null_map && right_nested_null_map[row] && nested_null_map &&
111 nested_null_map[pos + off]) {
112 ConcreteAction::apply(res, pos);
113 if constexpr (!ConcreteAction::resume_execution) {
114 break;
115 }
116 }
117 // some is null while another is not
118 if (right_nested_null_map && nested_null_map &&
119 right_nested_null_map[row] != nested_null_map[pos + off]) {
120 continue;
121 }
122 if (nested_null_map && nested_null_map[pos + off]) {
123 continue;
124 }
125 size_t str_pos = str_offs[pos + off - 1];
126 size_t str_len = str_offs[pos + off] - str_pos;
127 const char* left_raw_v = reinterpret_cast<const char*>(&str_chars[str_pos]);
128 const char* right_raw_v = reinterpret_cast<const char*>(&right_chars[right_off]);
129 // StringRef operator == using vec impl
130 if (StringRef(left_raw_v, str_len) == StringRef(right_raw_v, right_len)) {
131 ConcreteAction::apply(res, pos);
132 if constexpr (!ConcreteAction::resume_execution) {
133 break;
134 }
135 }
136 }
137 dst_data[row] = res;
138 }
139 return ColumnNullable::create(std::move(dst), std::move(dst_null_column));
140 }

142 template <typename NestedColumnType, typename RightColumnType>
143 ColumnPtr _execute_number(const ColumnArray::Offsets64& offsets, const UInt8* nested_null_map,
144 const IColumn& nested_column, const IColumn& right_column,
145 const UInt8* right_nested_null_map, const UInt8* outer_null_map) {
146 // check array nested column type and get data
147 const auto& nested_data =
148 reinterpret_cast<const NestedColumnType&>(nested_column).get_data();
149
150 // check right column type and get data
151 const auto& right_data = reinterpret_cast<const RightColumnType&>(right_column).get_data();
152
153 // prepare return data
154 auto dst = ColumnVector<ResultType>::create(offsets.size());
155 auto& dst_data = dst->get_data();
156 auto dst_null_column = ColumnUInt8::create(offsets.size());
157 auto& dst_null_data = dst_null_column->get_data();
158
159 // process
160 for (size_t row = 0; row < offsets.size(); ++row) {
161 if (outer_null_map && outer_null_map[row]) {
162 dst_null_data[row] = true;
163 continue;
164 }
165 dst_null_data[row] = false;
166 ResultType res = 0;
167 size_t off = offsets[row - 1];
168 size_t len = offsets[row] - off;
169 for (size_t pos = 0; pos < len; ++pos) {
170 // match null value
171 if (right_nested_null_map && right_nested_null_map[row] && nested_null_map &&
172 nested_null_map[pos + off]) {
173 ConcreteAction::apply(res, pos);
174 if constexpr (!ConcreteAction::resume_execution) {
175 break;
176 }
177 }
178 // some is null while another is not
179 if (right_nested_null_map && nested_null_map &&
180 right_nested_null_map[row] != nested_null_map[pos + off]) {
181 continue;
182 }
183 if (nested_null_map && nested_null_map[pos + off]) {
184 continue;
185 }
186 if (nested_data[pos + off] == right_data[row]) {
187 ConcreteAction::apply(res, pos);
188 if constexpr (!ConcreteAction::resume_execution) {
189 break;
190 }
191 }
192 }
193 dst_data[row] = res;
194 }
195 return ColumnNullable::create(std::move(dst), std::move(dst_null_column));
196 }

272 Status _execute_non_nullable(Block& block, const ColumnNumbers& arguments, size_t result,
273 size_t input_rows_count) {
274 // extract array offsets and nested data
275 auto left_column =
276 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
277 const ColumnArray* array_column = nullptr;
278 const UInt8* array_null_map = nullptr;
279 if (left_column->is_nullable()) {
280 auto nullable_array = reinterpret_cast<const ColumnNullable*>(left_column.get());
281 array_column =
282 reinterpret_cast<const ColumnArray*>(&nullable_array->get_nested_column());
283 array_null_map = nullable_array->get_null_map_column().get_data().data();
284 } else {
285 array_column = reinterpret_cast<const ColumnArray*>(left_column.get());
286 }
287 const auto& offsets = array_column->get_offsets();
288 const UInt8* nested_null_map = nullptr;
289 ColumnPtr nested_column = nullptr;
290 if (array_column->get_data().is_nullable()) {
291 const auto& nested_null_column =
292 reinterpret_cast<const ColumnNullable&>(array_column->get_data());
293 nested_null_map = nested_null_column.get_null_map_column().get_data().data();
294 nested_column = nested_null_column.get_nested_column_ptr();
295 } else {
296 nested_column = array_column->get_data_ptr();
297 }
298
299 // get right column
300 ColumnPtr right_full_column =
301 block.get_by_position(arguments[1]).column->convert_to_full_column_if_const();
302 ColumnPtr right_column = right_full_column;
303 const UInt8* right_nested_null_map = nullptr;
304 if (right_column->is_nullable()) {
305 const auto& nested_null_column = assert_cast<const ColumnNullable&>(*right_full_column);
306 right_column = nested_null_column.get_nested_column_ptr();
307 right_nested_null_map = nested_null_column.get_null_map_column().get_data().data();
308 }
309 // execute
310 auto array_type = remove_nullable(block.get_by_position(arguments[0]).type);
311 auto left_element_type =
312 remove_nullable(assert_cast<const DataTypeArray&>(*array_type).get_nested_type());
313 auto right_type = remove_nullable(block.get_by_position(arguments[1]).type);
314
315 ColumnPtr return_column = nullptr;
316 if (is_string(right_type) && is_string(left_element_type)) {
317 return_column = _execute_string(offsets, nested_null_map, *nested_column, *right_column,
318 right_nested_null_map, array_null_map);
319 } else if (is_number(right_type) && is_number(left_element_type)) {
320 if (check_column<ColumnUInt8>(*nested_column)) {
321 return_column = _execute_number_expanded<ColumnUInt8>(
322 offsets, nested_null_map, *nested_column, *right_column,
323 right_nested_null_map, array_null_map);
324 } else if (check_column<ColumnInt8>(*nested_column)) {
325 return_column = _execute_number_expanded<ColumnInt8>(
326 offsets, nested_null_map, *nested_column, *right_column,
327 right_nested_null_map, array_null_map);
328 } else if (check_column<ColumnInt16>(*nested_column)) {
329 return_column = _execute_number_expanded<ColumnInt16>(
330 offsets, nested_null_map, *nested_column, *right_column,
331 right_nested_null_map, array_null_map);
332 } else if (check_column<ColumnInt32>(*nested_column)) {
333 return_column = _execute_number_expanded<ColumnInt32>(
334 offsets, nested_null_map, *nested_column, *right_column,
335 right_nested_null_map, array_null_map);
336 } else if (check_column<ColumnInt64>(*nested_column)) {
337 return_column = _execute_number_expanded<ColumnInt64>(
338 offsets, nested_null_map, *nested_column, *right_column,
339 right_nested_null_map, array_null_map);
340 } else if (check_column<ColumnInt128>(*nested_column)) {
341 return_column = _execute_number_expanded<ColumnInt128>(
342 offsets, nested_null_map, *nested_column, *right_column,
343 right_nested_null_map, array_null_map);
344 } else if (check_column<ColumnFloat32>(*nested_column)) {
345 return_column = _execute_number_expanded<ColumnFloat32>(
346 offsets, nested_null_map, *nested_column, *right_column,
347 right_nested_null_map, array_null_map);
348 } else if (check_column<ColumnFloat64>(*nested_column)) {
349 return_column = _execute_number_expanded<ColumnFloat64>(
350 offsets, nested_null_map, *nested_column, *right_column,
351 right_nested_null_map, array_null_map);
352 } else if (check_column<ColumnDecimal32>(*nested_column)) {
353 return_column = _execute_number_expanded<ColumnDecimal32>(
354 offsets, nested_null_map, *nested_column, *right_column,
355 right_nested_null_map, array_null_map);
356 } else if (check_column<ColumnDecimal64>(*nested_column)) {
357 return_column = _execute_number_expanded<ColumnDecimal64>(
358 offsets, nested_null_map, *nested_column, *right_column,
359 right_nested_null_map, array_null_map);
360 } else if (check_column<ColumnDecimal128I>(*nested_column)) {
361 return_column = _execute_number_expanded<ColumnDecimal128I>(
362 offsets, nested_null_map, *nested_column, *right_column,
363 right_nested_null_map, array_null_map);
364 } else if (check_column<ColumnDecimal128>(*nested_column)) {
365 return_column = _execute_number_expanded<ColumnDecimal128>(
366 offsets, nested_null_map, *nested_column, *right_column,
367 right_nested_null_map, array_null_map);
368 }
369 } else if ((is_date_or_datetime(right_type) || is_date_v2_or_datetime_v2(right_type)) &&
370 (is_date_or_datetime(left_element_type) ||
371 is_date_v2_or_datetime_v2(left_element_type))) {
372 if (nested_column->is_date_type()) {
373 return_column = _execute_number_expanded<ColumnDate>(
374 offsets, nested_null_map, *nested_column, *right_column,
375 right_nested_null_map, array_null_map);
376 } else if (nested_column->is_datetime_type()) {
377 return_column = _execute_number_expanded<ColumnDateTime>(
378 offsets, nested_null_map, *nested_column, *right_column,
379 right_nested_null_map, array_null_map);
380 } else if (check_column<ColumnDateV2>(*nested_column)) {
381 return_column = _execute_number_expanded<ColumnDateV2>(
382 offsets, nested_null_map, *nested_column, *right_column,
383 right_nested_null_map, array_null_map);
384 } else if (check_column<ColumnDateTimeV2>(*nested_column)) {
385 return_column = _execute_number_expanded<ColumnDateTimeV2>(
386 offsets, nested_null_map, *nested_column, *right_column,
387 right_nested_null_map, array_null_map);
388 }
389 }
390
391 if (return_column) {
392 block.replace_by_position(result, std::move(return_column));
393 return Status::OK();
394 }
395 return Status::RuntimeError("execute failed or unsupported types for function {}({}, {})",
396 get_name(),
397 block.get_by_position(arguments[0]).type->get_name(),
398 block.get_by_position(arguments[1]).type->get_name());
399 }
