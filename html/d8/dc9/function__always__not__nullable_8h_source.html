44 template <typename ColumnType, bool is_nullable>
45 Status execute_internal(const ColumnPtr& column, const DataTypePtr& data_type,
46 MutableColumnPtr& column_result) {
47 auto type_error = [&]() {
48 return Status::RuntimeError("Illegal column {} of argument of function {}",
49 column->get_name(), get_name());
50 };
51 if constexpr (is_nullable) {
52 const ColumnNullable* col_nullable = check_and_get_column<ColumnNullable>(column.get());
53 const ColumnType* col =
54 check_and_get_column<ColumnType>(col_nullable->get_nested_column_ptr().get());
55 const ColumnUInt8* col_nullmap = check_and_get_column<ColumnUInt8>(
56 col_nullable->get_null_map_column_ptr().get());
57
58 if (col != nullptr && col_nullmap != nullptr) {
59 if constexpr (WithReturn) {
60 RETURN_IF_ERROR(
61 Function::vector_nullable(col, col_nullmap->get_data(), column_result));
62 } else {
63 Function::vector_nullable(col, col_nullmap->get_data(), column_result);
64 }
65 } else {
66 return type_error();
67 }
68 } else {
69 const ColumnType* col = check_and_get_column<ColumnType>(column.get());
70 if constexpr (WithReturn) {
71 RETURN_IF_ERROR(Function::vector(col, column_result));
72 } else {
73 Function::vector(col, column_result);
74 }
75 }
76 return Status::OK();
77 }
78
79 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
80 size_t result, size_t input_rows_count) override {
81 const ColumnPtr& column = block.get_by_position(arguments[0]).column;
82 const DataTypePtr& data_type = block.get_by_position(arguments[0]).type;
83 WhichDataType which(data_type);
84
85 MutableColumnPtr column_result = get_return_type_impl({})->create_column();
86 column_result->resize(input_rows_count);
87
88 auto type_error = [&]() {
89 return Status::RuntimeError("Illegal column {} of argument of function {}",
90 block.get_by_position(arguments[0]).column->get_name(),
91 get_name());
92 };
93 Status status = Status::OK();
94 if (which.is_nullable()) {
95 const DataTypePtr& nested_data_type =
96 static_cast<const DataTypeNullable*>(data_type.get())->get_nested_type();
97 WhichDataType nested_which(nested_data_type);
98 if (nested_which.is_string_or_fixed_string()) {
99 status = execute_internal<ColumnString, true>(column, data_type, column_result);
100 } else if (nested_which.is_int64()) {
101 status = execute_internal<ColumnInt64, true>(column, data_type, column_result);
102 } else {
103 return type_error();
104 }
105 } else if (which.is_string_or_fixed_string()) {
106 status = execute_internal<ColumnString, false>(column, data_type, column_result);
107 } else if (which.is_int64()) {
108 status = execute_internal<ColumnInt64, false>(column, data_type, column_result);
109 } else {
110 return type_error();
111 }
112 if (status.ok()) {
113 block.replace_by_position(result, std::move(column_result));
114 }
115 return status;
116 }
117};
