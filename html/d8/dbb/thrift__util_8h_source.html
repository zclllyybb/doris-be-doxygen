37// Note: thrift will encode NULLs into the serialized buffer so it is not valid
38// to treat it as a string.

46 // Serializes obj into result. Result will contain a copy of the memory.
47 template <class T>
48 Status serialize(T* obj, std::vector<uint8_t>* result) {
49 uint32_t len = 0;
50 uint8_t* buffer = nullptr;
51 RETURN_IF_ERROR(serialize<T>(obj, &len, &buffer));
52 result->resize(len);
53 memcpy(&((*result)[0]), buffer, len);
54 return Status::OK();
55 }

57 // serialize obj into a memory buffer. The result is returned in buffer/len. The
58 // memory returned is owned by this object and will be invalid when another object
59 // is serialized.
60 template <class T>
61 Status serialize(T* obj, uint32_t* len, uint8_t** buffer) {
62 try {
63 _mem_buffer->resetBuffer();
64 obj->write(_protocol.get());
65 } catch (std::exception& e) {
66 return Status::InternalError("Couldn't serialize thrift object:\n{}", e.what());
67 }
68
69 _mem_buffer->getBuffer(buffer, len);
70 return Status::OK();
71 }

118// Deserialize a thrift message from buf/len. buf/len must at least contain
119// all the bytes needed to store the thrift message. On return, len will be
120// set to the actual length of the header.
121template <class T>
122Status deserialize_thrift_msg(const uint8_t* buf, uint32_t* len, bool compact,
123 T* deserialized_msg) {
124 // Deserialize msg bytes into c++ thrift msg using memory
125 // transport. TMemoryBuffer is not const-safe, although we use it in
126 // a const-safe way, so we have to explicitly cast away the const.
127 std::shared_ptr<apache::thrift::transport::TMemoryBuffer> tmem_transport(
128 new apache::thrift::transport::TMemoryBuffer(const_cast<uint8_t*>(buf), *len));
129 std::shared_ptr<apache::thrift::protocol::TProtocol> tproto =
130 create_deserialize_protocol(tmem_transport, compact);
131
132 try {
133 deserialized_msg->read(tproto.get());
134 } catch (std::exception& e) {
135 return Status::InternalError("Couldn't deserialize thrift msg:\n{}", e.what());
136 } catch (...) {
137 // TODO: Find the right exception for 0 bytes
138 return Status::InternalError("Unknown exception");
139 }
140
141 uint32_t bytes_left = tmem_transport->available_read();
142 *len = *len - bytes_left;
143 return Status::OK();
144}
