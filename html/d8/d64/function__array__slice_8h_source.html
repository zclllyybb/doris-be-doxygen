31class FunctionArraySlice : public IFunction {
32public:
33 static constexpr auto name = "array_slice";
34 static FunctionPtr create() { return std::make_shared<FunctionArraySlice>(); }
35
37 String get_name() const override { return name; }
38
39 bool is_variadic() const override { return true; }
40
41 size_t get_number_of_arguments() const override { return 0; }
42
43 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
44 DCHECK(is_array(arguments[0]))
45 << "First argument for function: " << name
46 << " should be DataTypeArray but it has type " << arguments[0]->get_name() << ".";
47 DCHECK(is_integer(arguments[1]))
48 << "Second argument for function: " << name << " should be Integer but it has type "
49 << arguments[1]->get_name() << ".";
50 if (arguments.size() > 2) {
51 DCHECK(is_integer(arguments[2]))
52 << "Third argument for function: " << name
53 << " should be Integer but it has type " << arguments[2]->get_name() << ".";
54 }
55 return arguments[0];
56 }
57
58 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
59 size_t result, size_t input_rows_count) override {
60 auto array_column =
61 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
62 auto offset_column =
63 block.get_by_position(arguments[1]).column->convert_to_full_column_if_const();
64 ColumnPtr length_column = nullptr;
65 if (arguments.size() > 2) {
66 length_column =
67 block.get_by_position(arguments[2]).column->convert_to_full_column_if_const();
68 }
69 // extract src array column
70 ColumnArrayExecutionData src;
71 if (!extract_column_array_info(*array_column, src)) {
72 return Status::RuntimeError(
73 fmt::format("execute failed, unsupported types for function {}({}, {})",
74 get_name(), block.get_by_position(arguments[0]).type->get_name(),
75 block.get_by_position(arguments[1]).type->get_name()));
76 }
77 // prepare dst array column
78 bool is_nullable = src.nested_nullmap_data ? true : false;
79 ColumnArrayMutableData dst = create_mutable_data(src.nested_col, is_nullable);
80 dst.offsets_ptr->reserve(input_rows_count);
81 // execute
82 slice_array(dst, src, *offset_column, length_column.get());
83 ColumnPtr res_column = assemble_column_array(dst);
84 block.replace_by_position(result, std::move(res_column));
85 return Status::OK();
86 }
87};
