78// Finds the first occurrence of a character in at most a given number of bytes
79// of a char* string. Returns a pointer to the first occurrence, or NULL if no
80// occurrence found in the first sz bytes.
81// Never searches past the first null character in the string; therefore, only
82// suitable for null-terminated strings.
83// WARNING: Removes const-ness of string argument!
84inline char* strnchr(const char* buf, char c, int sz) {
85 const char* end = buf + sz;
86 while (buf != end && *buf) {
87 if (*buf == c) return const_cast<char*>(buf);
88 ++buf;
89 }
90 return NULL;
91}

99// Matches a prefix (which must be a char* literal!) against the beginning of
100// str. Returns a pointer past the prefix, or NULL if the prefix wasn't matched.
101// (Like the standard strcasecmp(), but for efficiency doesn't call strlen() on
102// prefix, and returns a pointer rather than an int.)
103//
104// The ""'s catch people who don't pass in a literal for "prefix"
105#ifndef strprefix
106#define strprefix(str, prefix) \
107 (strncmp(str, prefix, sizeof("" prefix "") - 1) == 0 ? str + sizeof(prefix) - 1 : NULL)
108#endif

209// Returns the number of times a character occurs in a string for a null
210// terminated string.
211inline ptrdiff_t strcount(const char* buf, char c) {
212 if (buf == NULL) return 0;
213 ptrdiff_t num = 0;
214 for (const char* bp = buf; *bp != '\0'; bp++) {
215 if (*bp == c) num++;
216 }
217 return num;
218}
219// Returns the number of times a character occurs in a string for a string
220// defined by a pointer to the first character and a pointer just past the last
221// character.
222inline ptrdiff_t strcount(const char* buf_begin, const char* buf_end, char c) {
223 if (buf_begin == NULL) return 0;
224 if (buf_end <= buf_begin) return 0;
225 ptrdiff_t num = 0;
226 for (const char* bp = buf_begin; bp != buf_end; bp++) {
227 if (*bp == c) num++;
228 }
229 return num;
230}

251// STL-compatible function objects for char* string keys:

253// Compares two char* strings for equality. (Works with NULL, which compares
254// equal only to another NULL). Useful in hash tables:
255// hash_map<const char*, Value, hash<const char*>, streq> ht;
256struct streq : public binary_function<const char*, const char*, bool> {
257 bool operator()(const char* s1, const char* s2) const {
258 return ((s1 == 0 && s2 == 0) || (s1 && s2 && *s1 == *s2 && strcmp(s1, s2) == 0));
259 }
260};
261
262// Compares two char* strings. (Works with NULL, which compares greater than any
263// non-NULL). Useful in maps:
264// map<const char*, Value, strlt> m;
265struct strlt : public binary_function<const char*, const char*, bool> {
266 bool operator()(const char* s1, const char* s2) const {
267 return (s1 != s2) && (s2 == 0 || (s1 != 0 && strcmp(s1, s2) < 0));
268 }
269};

316// Copies at most n-1 bytes from src to dest, and returns dest. If n >=1, null
317// terminates dest; otherwise, returns dest unchanged. Unlike strncpy(), only
318// puts one null character at the end of dest.
319inline char* safestrncpy(char* dest, const char* src, size_t n) {
320 if (n < 1) return dest;
321
322 // Avoid using non-ANSI memccpy(), which is also deprecated in MSVC
323 for (size_t i = 0; i < n; ++i) {
324 if ((dest[i] = src[i]) == '\0') return dest;
325 }
326
327 dest[n - 1] = '\0';
328 return dest;
329}
