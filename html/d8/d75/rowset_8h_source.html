42// the rowset state transfer graph:
43// ROWSET_UNLOADED <--|
44// ↓ |
45// ROWSET_LOADED |
46// ↓ |
47// ROWSET_UNLOADING -->|
48enum RowsetState {
49 // state for new created rowset
50 ROWSET_UNLOADED,
51 // state after load() called
52 ROWSET_LOADED,
53 // state for closed() called but owned by some readers
54 ROWSET_UNLOADING
55};
56
57class RowsetStateMachine {
58public:
59 RowsetStateMachine() : _rowset_state(ROWSET_UNLOADED) {}
60
61 Status on_load() {
62 switch (_rowset_state) {
63 case ROWSET_UNLOADED:
64 _rowset_state = ROWSET_LOADED;
65 break;
66
67 default:
68 return Status::Error<ErrorCode::ROWSET_INVALID_STATE_TRANSITION>();
69 }
70 return Status::OK();
71 }
72
73 Status on_close(uint64_t refs_by_reader) {
74 switch (_rowset_state) {
75 case ROWSET_LOADED:
76 if (refs_by_reader == 0) {
77 _rowset_state = ROWSET_UNLOADED;
78 } else {
79 _rowset_state = ROWSET_UNLOADING;
80 }
81 break;
82
83 default:
84 return Status::Error<ErrorCode::ROWSET_INVALID_STATE_TRANSITION>();
85 }
86 return Status::OK();
87 }
88
89 Status on_release() {
90 switch (_rowset_state) {
91 case ROWSET_UNLOADING:
92 _rowset_state = ROWSET_UNLOADED;
93 break;
94
95 default:
96 return Status::Error<ErrorCode::ROWSET_INVALID_STATE_TRANSITION>();
97 }
98 return Status::OK();
99 }
100
101 RowsetState rowset_state() { return _rowset_state; }
102
103private:
104 RowsetState _rowset_state;
105};

159 // close to clear the resource owned by rowset
160 // including: open files, indexes and so on
161 // NOTICE: can not call this function in multithreads
162 void close() {
163 RowsetState old_state = _rowset_state_machine.rowset_state();
164 if (old_state != ROWSET_LOADED) {
165 return;
166 }
167 Status st = Status::OK();
168 {
169 std::lock_guard<std::mutex> close_lock(_lock);
170 uint64_t current_refs = _refs_by_reader;
171 old_state = _rowset_state_machine.rowset_state();
172 if (old_state != ROWSET_LOADED) {
173 return;
174 }
175 if (current_refs == 0) {
176 do_close();
177 }
178 st = _rowset_state_machine.on_close(current_refs);
179 }
180 if (!st.ok()) {
181 LOG(WARNING) << "state transition failed from:" << _rowset_state_machine.rowset_state();
182 return;
183 }
184 VLOG_NOTICE << "rowset is close. rowset state from:" << old_state << " to "
185 << _rowset_state_machine.rowset_state() << ", version:" << start_version()
186 << "-" << end_version() << ", tabletid:" << _rowset_meta->tablet_id();
187 }

231 void release() {
232 // if the refs by reader is 0 and the rowset is closed, should release the resouce
233 uint64_t current_refs = --_refs_by_reader;
234 if (current_refs == 0 && _rowset_state_machine.rowset_state() == ROWSET_UNLOADING) {
235 {
236 std::lock_guard<std::mutex> release_lock(_lock);
237 // rejudge _refs_by_reader because we do not add lock in create reader
238 if (_refs_by_reader == 0 &&
239 _rowset_state_machine.rowset_state() == ROWSET_UNLOADING) {
240 // first do close, then change state
241 do_close();
242 _rowset_state_machine.on_release();
243 }
244 }
245 if (_rowset_state_machine.rowset_state() == ROWSET_UNLOADED) {
246 VLOG_NOTICE
