127 template <typename KeyHolder>
128 void ALWAYS_INLINE emplace(KeyHolder&& key_holder, LookupResult& it, bool& inserted) {
129 const auto& key = key_holder_get_key(key_holder);
130 inserted = false;
131 it = &*_hash_map.lazy_emplace(key, [&](const auto& ctor) {
132 inserted = true;
133 key_holder_persist_key(key_holder);
134 ctor(key_holder_get_key(key_holder), nullptr);
135 });
136
137 if constexpr (PartitionedHashTable) {
138 _check_if_need_partition();
139 }
140 }

224 size_t get_buffer_size_in_bytes() const {
225 const auto capacity = _hash_map.capacity();
226 return capacity * sizeof(typename HashMapImpl::slot_type);
227 }
228
229 size_t get_buffer_size_in_cells() const { return _hash_map.capacity(); }
230
231 bool add_elem_size_overflow(size_t row) const {
232 const auto capacity = _hash_map.capacity();
233 // phmap use 7/8th as maximum load factor.
234 return (_hash_map.size() + row) > (capacity * 7 / 8);
235 }
236
237 size_t size() const { return _hash_map.size(); }
238
239 char* get_null_key_data() { return nullptr; }
240 bool has_null_key_data() const { return false; }
241
242 bool need_partition() { return _need_partition; }
243
244 void set_partitioned_threshold(int threshold) { _partitioned_threshold = threshold; }
245
246 bool check_if_need_partition(size_t bucket_count) {
247 if constexpr (PartitionedHashTable) {
248 return _partitioned_threshold > 0 && bucket_count >= _partitioned_threshold;
249 } else {
250 return false;
251 }
252 }
253
254 bool empty() const { return _hash_map.empty(); }
255
256 void clear_and_shrink() { _hash_map.clear(); }
257
258private:
259 void _check_if_need_partition() {
260 if (UNLIKELY(check_if_need_partition(_hash_map.size() + 1))) {
261 _need_partition = add_elem_size_overflow(1);
262 }
263 }
264
265 HashMapImpl _hash_map;
266 // the bucket count threshold above which it's converted to partioned hash table
267 // > 0: enable convert dynamically
268 // 0: convert is disabled
269 int _partitioned_threshold = 0;
270 // if need resize and bucket count after resize will be >= _partitioned_threshold,
271 // this flag is set to true, and resize does not actually happen,
272 // PartitionedHashTable will convert this hash table to partitioned hash table
273 bool _need_partition;
274};
