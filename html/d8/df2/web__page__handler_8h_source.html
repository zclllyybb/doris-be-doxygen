49 // Register a route 'path' to be rendered via template.

82 // Container class for a list of path handler callbacks for a single URL.
83 class PathHandler {

86 PageHandlerCallback callback)
87 : is_styled_(is_styled),
88 is_on_nav_bar_(is_on_nav_bar),
89 alias_(std::move(alias)),
90 callback_(std::move(callback)) {}
91
92 bool is_styled() const { return is_styled_; }
93 bool is_on_nav_bar() const { return is_on_nav_bar_; }
94 const std::string& alias() const { return alias_; }
95 const PageHandlerCallback& callback() const { return callback_; }
96
97 private:
98 // If true, the page appears is rendered styled.
99 bool is_styled_;
100
101 // If true, the page appears in the navigation bar.
102 bool is_on_nav_bar_;
103
104 // Alias used when displaying this link on the nav bar.
105 std::string alias_;
106
107 // Callback to render output for this page.
108 PageHandlerCallback callback_;
109 };
