32// The RecordBatchQueue is created and managed by the ResultQueueMgr to
33// cache external query results, as well as query status. Where both
34// BlockingGet and BlockingPut operations block if the queue is empty or
35// full, respectively.
36class RecordBatchQueue {
37public:
38 RecordBatchQueue(u_int32_t max_elements) : _queue(max_elements) {}
39
40 Status status() {
41 std::lock_guard<SpinLock> l(_status_lock);
42 return _status;
43 }
44
45 void update_status(const Status& status);
46
47 bool blocking_get(std::shared_ptr<arrow::RecordBatch>* result) {
48 return _queue.blocking_get(result);
49 }
50
51 bool blocking_put(const std::shared_ptr<arrow::RecordBatch>& val) {
52 return _queue.blocking_put(val);
53 }
54
55 // Shut down the queue. Wakes up all threads waiting on blocking_get or blocking_put.
56 void shutdown();
57
58private:
59 BlockingQueue<std::shared_ptr<arrow::RecordBatch>> _queue;
60 SpinLock _status_lock;
61 Status _status;
62};

64} // namespace doris
