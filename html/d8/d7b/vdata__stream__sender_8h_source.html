65// We use BroadcastPBlockHolder to hold a broadcasted PBlock. For broadcast shuffle, one PBlock
66// will be shared between different channel, so we have to use a ref count to mark if this
67// PBlock is available for next serialization.
68class BroadcastPBlockHolder {
69public:
70 BroadcastPBlockHolder() : _ref_count(0) {}
71 ~BroadcastPBlockHolder() noexcept = default;
72
73 void unref() noexcept {
74 DCHECK_GT(_ref_count._value, 0);
75 _ref_count._value.fetch_sub(1);
76 }
77 void ref() noexcept { _ref_count._value.fetch_add(1); }
78
79 bool available() { return _ref_count._value == 0; }
80
81 PBlock* get_block() { return &pblock; }
82
83private:
84 AtomicWrapper<uint32_t> _ref_count;
85 PBlock pblock;
86};

378template <typename Channels>
379Status VDataStreamSender::channel_add_rows(Channels& channels, int num_channels,
380 const uint64_t* __restrict channel_ids, int rows,
381 Block* block) {
382 std::vector<int> channel2rows[num_channels];
383
384 for (int i = 0; i < rows; i++) {
385 channel2rows[channel_ids[i]].emplace_back(i);
386 }
387
388 for (int i = 0; i < num_channels; ++i) {
389 if (!channel2rows[i].empty()) {
390 RETURN_IF_ERROR(channels[i]->add_rows(block, channel2rows[i]));
391 }
392 }
393
394 return Status::OK();
395}

397class PipChannel final : public Channel {
398public:
399 PipChannel(VDataStreamSender* parent, const RowDescriptor& row_desc,
400 const TNetworkAddress& brpc_dest, const TUniqueId& fragment_instance_id,
401 PlanNodeId dest_node_id, int buffer_size, bool is_transfer_chain,
402 bool send_query_statistics_with_every_batch)
403 : Channel(parent, row_desc, brpc_dest, fragment_instance_id, dest_node_id, buffer_size,
404 is_transfer_chain, send_query_statistics_with_every_batch) {
405 ch_roll_pb_block();
406 }
407
408 ~PipChannel() override {
409 if (_ch_cur_pb_block) {
410 delete _ch_cur_pb_block;
411 }
412 }
413
414 void ch_roll_pb_block() override {
415 // We have two choices here.
416 // 1. Use a PBlock pool and fetch an available PBlock if we need one. In this way, we can
417 // reuse the memory, but we have to use a lock to synchronize.
418 // 2. Create a new PBlock every time. In this way we don't need a lock but have to allocate
419 // new memory.
420 // Now we use the second way.
421 _ch_cur_pb_block = new PBlock();
422 }
423
424 // Asynchronously sends a block
425 // Returns the status of the most recently finished transmit_data
426 // rpc (or OK if there wasn't one that hasn't been reported yet).
427 // if batch is nullptr, send the eof packet
428 Status send_block(PBlock* block, bool eos = false) override {
429 if (eos) {
430 if (_eos_send) {
431 return Status::OK();
432 } else {
433 _eos_send = true;
434 }
435 }
436 if (eos || block->column_metas_size()) {
437 RETURN_IF_ERROR(_buffer->add_block(
438 {this, block ? std::make_unique<PBlock>(*block) : nullptr, eos}));
439 }
440 return Status::OK();
441 }
442
443 Status send_block(BroadcastPBlockHolder* block, bool eos = false) override {
444 if (eos) {
445 if (_eos_send) {
446 return Status::OK();
447 } else {
448 _eos_send = true;
449 }
450 }
451 if (eos || block->get_block()->column_metas_size()) {
452 RETURN_IF_ERROR(_buffer->add_block({this, block, eos}));
453 }
454 return Status::OK();
455 }
456
457 // send _mutable_block
458 Status send_current_block(bool eos) override {
459 if (is_local()) {
460 return send_local_block(eos);
461 }
462
463 PBlock* block_ptr = nullptr;
464 if (_mutable_block) {
465 block_ptr = new PBlock(); // TODO: need a pool of PBlock()
466 auto block = _mutable_block->to_block();
467 RETURN_IF_ERROR(_parent->serialize_block(&block, block_ptr));
468 block.clear_column_data();
469 _mutable_block->set_muatable_columns(block.mutate_columns());
470 }
471 RETURN_IF_ERROR(send_block(block_ptr, eos));
472 return Status::OK();
473 }

477 _buffer->register_sink(_fragment_instance_id);
478 }

480private:
481 friend class VDataStreamSender;
482
483 pipeline::ExchangeSinkBuffer* _buffer = nullptr;
484 bool _eos_send = false;
485};
