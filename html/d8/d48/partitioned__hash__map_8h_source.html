26template <typename ImplTable>
27class PartitionedHashMapTable : public PartitionedHashTable<ImplTable> {
28public:
29 using Impl = ImplTable;
30 using Base = PartitionedHashTable<ImplTable>;
31 using Key = typename ImplTable::key_type;
32 using LookupResult = typename Impl::LookupResult;
33
34 using Base::Base;
35 using Base::prefetch;
36
37 using mapped_type = typename Base::mapped_type;
38
39 auto& ALWAYS_INLINE operator[](const Key& x) {
40 LookupResult it;
41 bool inserted;
42 this->emplace(x, it, inserted);
43
44 if (inserted) {
45 new (lookup_result_get_mapped(it)) mapped_type();
46 }
47
48 return *lookup_result_get_mapped(it);
49 }
50
51 template <typename Func>
52 void for_each_mapped(Func&& func) {
53 for (auto& v : *this) {
54 func(v.get_second());
55 }
56 }
57};
