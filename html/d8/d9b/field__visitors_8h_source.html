43template <typename Visitor, typename F>
44typename std::decay_t<Visitor>::ResultType apply_visitor(Visitor&& visitor, F&& field) {
45 switch (field.get_type()) {
46 case Field::Types::Null:
47 return visitor(field.template get<Null>());
48 case Field::Types::UInt64:
49 return visitor(field.template get<UInt64>());
50 case Field::Types::UInt128:
51 return visitor(field.template get<UInt128>());
52 case Field::Types::Int64:
53 return visitor(field.template get<Int64>());
54 case Field::Types::Float64:
55 return visitor(field.template get<Float64>());
56 case Field::Types::String:
57 return visitor(field.template get<String>());
58 case Field::Types::Array:
59 return visitor(field.template get<Array>());
60 case Field::Types::Tuple:
61 return visitor(field.template get<Tuple>());
62 case Field::Types::Decimal32:
63 return visitor(field.template get<DecimalField<Decimal32>>());
64 case Field::Types::Decimal64:
65 return visitor(field.template get<DecimalField<Decimal64>>());
66 case Field::Types::Decimal128:
67 return visitor(field.template get<DecimalField<Decimal128>>());
68 case Field::Types::Decimal128I:
69 return visitor(field.template get<DecimalField<Decimal128I>>());
70 case Field::Types::AggregateFunctionState:
71 return visitor(field.template get<AggregateFunctionStateData>());
72
73 default:
74 LOG(FATAL) << "Bad type of Field";
75 return {};
76 }
77}

79template <typename Visitor, typename F1, typename F2>
80typename std::decay_t<Visitor>::ResultType apply_visitor(Visitor&& visitor, F1&& field1,
81 F2&& field2) {
82 switch (field1.getType()) {
83 case Field::Types::Null:
84 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),

86 case Field::Types::UInt64:
87 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
88 field1.template get<UInt64>(), std::forward<F2>(field2));
89 case Field::Types::UInt128:
90 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
91 field1.template get<UInt128>(), std::forward<F2>(field2));
92 case Field::Types::Int64:
93 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
94 field1.template get<Int64>(), std::forward<F2>(field2));
95 case Field::Types::Float64:
96 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
97 field1.template get<Float64>(), std::forward<F2>(field2));
98 case Field::Types::String:
99 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
100 field1.template get<String>(), std::forward<F2>(field2));
101 case Field::Types::Array:
102 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
103 field1.template get<Array>(), std::forward<F2>(field2));
104 case Field::Types::Tuple:
105 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
106 field1.template get<Tuple>(), std::forward<F2>(field2));
107 case Field::Types::Decimal32:
108 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
109 field1.template get<DecimalField<Decimal32>>(),
110 std::forward<F2>(field2));
111 case Field::Types::Decimal64:
112 return apply_binary_visitor_impl(std::forward<Visitor>(visitor),
113 field1.template get<DecimalField<Decimal64>>(),

115 case Field::Types::Decimal128:
