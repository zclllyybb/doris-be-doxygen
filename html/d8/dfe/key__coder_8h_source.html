33using FullEncodeAscendingFunc = void (*)(const void* value, std::string* buf);
34using EncodeAscendingFunc = void (*)(const void* value, size_t index_size, std::string* buf);
35using DecodeAscendingFunc = Status (*)(Slice* encoded_key, size_t index_size, uint8_t* cell_ptr,
36 MemPool* pool);

38// Order-preserving binary encoding for values of a particular type so that
39// those values can be compared by memcpy their encoded bytes.
40//
41// To obtain instance of this class, use the `get_key_coder(FieldType)` method.
42class KeyCoder {
43public:
44 template <typename TraitsType>
45 KeyCoder(TraitsType traits);
46
47 // encode the provided `value` into `buf`.
48 void full_encode_ascending(const void* value, std::string* buf) const {
49 _full_encode_ascending(value, buf);
50 }
51
52 // similar to `full_encode_ascending`, but only encode part (the first `index_size` bytes) of the value.
53 // only applicable to string type
54 void encode_ascending(const void* value, size_t index_size, std::string* buf) const {
55 _encode_ascending(value, index_size, buf);
56 }
57
58 Status decode_ascending(Slice* encoded_key, size_t index_size, uint8_t* cell_ptr,
59 MemPool* pool) const {
60 return _decode_ascending(encoded_key, index_size, cell_ptr, pool);
61 }
62
63private:
64 FullEncodeAscendingFunc _full_encode_ascending;
65 EncodeAscendingFunc _encode_ascending;
66 DecodeAscendingFunc _decode_ascending;
67};

74template <FieldType field_type>
75class KeyCoderTraits<field_type, typename std::enable_if<std::is_integral<typename CppTypeTraits<
76 field_type>::CppType>::value>::type> {
77public:
78 using CppType = typename CppTypeTraits<field_type>::CppType;
79 using UnsignedCppType = typename CppTypeTraits<field_type>::UnsignedCppType;
80
81private:
82 // Swap value's endian from/to big endian
83 static UnsignedCppType swap_big_endian(UnsignedCppType val) {
84 switch (sizeof(UnsignedCppType)) {
85 case 1:
86 return val;
87 case 2:
88 return BigEndian::FromHost16(val);
89 case 4:
90 return BigEndian::FromHost32(val);
91 case 8:
92 return BigEndian::FromHost64(val);
93 case 16:
94 return BigEndian::FromHost128(val);
95 default:
96 LOG(FATAL) << "Invalid type to big endian, type=" << field_type
97 << ", size=" << sizeof(UnsignedCppType);
98 }
99 }
100
101public:
102 static void full_encode_ascending(const void* value, std::string* buf) {
103 UnsignedCppType unsigned_val;
104 memcpy(&unsigned_val, value, sizeof(unsigned_val));
105 // swap MSB to encode integer
106 if (std::is_signed<CppType>::value) {
107 unsigned_val ^=
108 (static_cast<UnsignedCppType>(1) << (sizeof(UnsignedCppType) * CHAR_BIT - 1));
109 }
110 // make it bigendian
111 unsigned_val = swap_big_endian(unsigned_val);
112
113 buf->append((char*)&unsigned_val, sizeof(unsigned_val));
114 }
115
116 static void encode_ascending(const void* value, size_t index_size, std::string* buf) {
117 full_encode_ascending(value, buf);
118 }
119
120 static Status decode_ascending(Slice* encoded_key, size_t index_size, uint8_t* cell_ptr,
121 MemPool* pool) {
122 if (encoded_key->size < sizeof(UnsignedCppType)) {
123 return Status::InvalidArgument("Key too short, need={} vs real={}",
124 sizeof(UnsignedCppType), encoded_key->size);
125 }
126 UnsignedCppType unsigned_val;
127 memcpy(&unsigned_val, encoded_key->data, sizeof(UnsignedCppType));
128 unsigned_val = swap_big_endian(unsigned_val);
129 if (std::is_signed<CppType>::value) {
130 unsigned_val ^=
131 (static_cast<UnsignedCppType>(1) << (sizeof(UnsignedCppType) * CHAR_BIT - 1));
132 }
133 memcpy(cell_ptr, &unsigned_val, sizeof(UnsignedCppType));
134 encoded_key->remove_prefix(sizeof(UnsignedCppType));
135 return Status::OK();
136 }
137};

268class KeyCoderTraits<OLAP_FIELD_TYPE_CHAR> {
269public:
270 static void full_encode_ascending(const void* value, std::string* buf) {
271 auto slice = reinterpret_cast<const Slice*>(value);
272 buf->append(slice->get_data(), slice->get_size());

274
275 static void encode_ascending(const void* value, size_t index_size, std::string* buf) {
276 const Slice* slice = (const Slice*)value;
277 CHECK(index_size <= slice->size)
278 << "index size is larger than char size, index=" << index_size
279 << ", char=" << slice->size;
280 buf->append(slice->data, index_size);
281 }
282
283 static Status decode_ascending(Slice* encoded_key, size_t index_size, uint8_t* cell_ptr,
284 MemPool* pool) {
285 if (encoded_key->size < index_size) {
286 return Status::InvalidArgument("Key too short, need={} vs real={}", index_size,
287 encoded_key->size);
288 }
289 Slice* slice = (Slice*)cell_ptr;
290 slice->data = (char*)pool->allocate(index_size);
291 slice->size = index_size;
292 memcpy(slice->data, encoded_key->data, index_size);
293 encoded_key->remove_prefix(index_size);
294 return Status::OK();
295 }
