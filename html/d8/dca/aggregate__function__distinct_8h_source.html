35template <typename T>
36struct AggregateFunctionDistinctSingleNumericData {
38 using Set = HashSetWithStackMemory<T, DefaultHash<T>, 4>;
39 using Self = AggregateFunctionDistinctSingleNumericData<T>;
40 Set set;
41
42 void add(const IColumn** columns, size_t /* columns_num */, size_t row_num, Arena*) {
43 const auto& vec = assert_cast<const ColumnVector<T>&>(*columns[0]).get_data();
44 set.insert(vec[row_num]);
45 }
46
47 void merge(const Self& rhs, Arena*) { set.merge(rhs.set); }
48
49 void serialize(BufferWritable& buf) const { set.write(buf); }
50
51 void deserialize(BufferReadable& buf, Arena*) { set.read(buf); }
52
53 MutableColumns get_arguments(const DataTypes& argument_types) const {
54 MutableColumns argument_columns;
55 argument_columns.emplace_back(argument_types[0]->create_column());
56 for (const auto& elem : set) {
57 argument_columns[0]->insert(elem.get_value());
58 }
59
60 return argument_columns;
61 }
62};

64struct AggregateFunctionDistinctGenericData {
66 using Set = HashSetWithSavedHashWithStackMemory<StringRef, StringRefHash, 4>;
67 using Self = AggregateFunctionDistinctGenericData;
68 Set set;
69
70 void merge(const Self& rhs, Arena* arena) {
71 Set::LookupResult it;
72 bool inserted;
73 for (const auto& elem : rhs.set) {
74 set.emplace(ArenaKeyHolder {elem.get_value(), *arena}, it, inserted);
75 }
76 }
77
78 void serialize(BufferWritable& buf) const {
79 write_var_uint(set.size(), buf);
80 for (const auto& elem : set) {
81 write_string_binary(elem.get_value(), buf);
82 }
83 }
84
85 void deserialize(BufferReadable& buf, Arena* arena) {
86 UInt64 size;
87 read_var_uint(size, buf);
88
89 StringRef ref;
90 for (size_t i = 0; i < size; ++i) {
91 read_string_binary(ref, buf);
92 set.insert(ref);
93 }
94 }
95};

97template <bool is_plain_column>
98struct AggregateFunctionDistinctSingleGenericData : public AggregateFunctionDistinctGenericData {
99 void add(const IColumn** columns, size_t /* columns_num */, size_t row_num, Arena* arena) {
100 Set::LookupResult it;
101 bool inserted;
102 auto key_holder = get_key_holder<is_plain_column>(*columns[0], row_num, *arena);
103 set.emplace(key_holder, it, inserted);
104 }
105
106 MutableColumns get_arguments(const DataTypes& argument_types) const {
107 MutableColumns argument_columns;
108 argument_columns.emplace_back(argument_types[0]->create_column());
109 for (const auto& elem : set) {
110 deserialize_and_insert<is_plain_column>(elem.get_value(), *argument_columns[0]);
111 }
112
113 return argument_columns;
114 }

117struct AggregateFunctionDistinctMultipleGenericData : public AggregateFunctionDistinctGenericData {
118 void add(const IColumn** columns, size_t columns_num, size_t row_num, Arena* arena) {
119 const char* begin = nullptr;
120 StringRef value(begin, 0);
121 for (size_t i = 0; i < columns_num; ++i) {
122 auto cur_ref = columns[i]->serialize_value_into_arena(row_num, *arena, begin);
123 value.data = cur_ref.data - value.size;
124 value.size += cur_ref.size;
125 }
126
127 Set::LookupResult it;
128 bool inserted;
129 auto key_holder = SerializedKeyHolder {value, *arena};
130 set.emplace(key_holder, it, inserted);
131 }
132
133 MutableColumns get_arguments(const DataTypes& argument_types) const {
134 MutableColumns argument_columns(argument_types.size());
135 for (size_t i = 0; i < argument_types.size(); ++i) {
136 argument_columns[i] = argument_types[i]->create_column();
137 }
138
139 for (const auto& elem : set) {
140 const char* begin = elem.get_value().data;
141 for (auto& column : argument_columns) {
142 begin = column->deserialize_and_insert_from_arena(begin);
143 }
144 }
145
146 return argument_columns;
147 }
148};

153template <typename Data>
154class AggregateFunctionDistinct
155 : public IAggregateFunctionDataHelper<Data, AggregateFunctionDistinct<Data>> {
156private:
157 size_t prefix_size;
158 AggregateFunctionPtr nested_func;
159 size_t arguments_num;
160
161 AggregateDataPtr get_nested_place(AggregateDataPtr __restrict place) const noexcept {
162 return place + prefix_size;
163 }
164
165 ConstAggregateDataPtr get_nested_place(ConstAggregateDataPtr __restrict place) const noexcept {
166 return place + prefix_size;
167 }
168
169public:
170 AggregateFunctionDistinct(AggregateFunctionPtr nested_func_, const DataTypes& arguments)
171 : IAggregateFunctionDataHelper<Data, AggregateFunctionDistinct>(
172 arguments, nested_func_->get_parameters()),
173 nested_func(nested_func_),
174 arguments_num(arguments.size()) {
175 size_t nested_size = nested_func->align_of_data();
176 CHECK_GT(nested_size, 0);
177 prefix_size = (sizeof(Data) + nested_size - 1) / nested_size * nested_size;
178 }
179
180 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
181 Arena* arena) const override {
182 this->data(place).add(columns, arguments_num, row_num, arena);
183 }
184
185 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
186 Arena* arena) const override {
187 this->data(place).merge(this->data(rhs), arena);
188 }
189
190 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
191 this->data(place).serialize(buf);
192 }
193
194 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
195 Arena* arena) const override {
196 this->data(place).deserialize(buf, arena);
197 }
198
199 // void insert_result_into(AggregateDataPtr place, IColumn & to, Arena * arena) const override
200 void insert_result_into(ConstAggregateDataPtr targetplace, IColumn& to) const override {
201 auto place = const_cast<AggregateDataPtr>(targetplace);
202 auto arguments = this->data(place).get_arguments(this->argument_types);
203 ColumnRawPtrs arguments_raw(arguments.size());
204 for (size_t i = 0; i < arguments.size(); ++i) {
205 arguments_raw[i] = arguments[i].get();
206 }
207
208 assert(!arguments.empty());
209 // nested_func->add_batch_single_place(arguments[0]->size(), get_nested_place(place), arguments_raw.data(), arena);
210 // nested_func->insert_result_into(get_nested_place(place), to, arena);
211
212 nested_func->add_batch_single_place(arguments[0]->size(), get_nested_place(place),
213 arguments_raw.data(), nullptr);
214 nested_func->insert_result_into(get_nested_place(place), to);
215 }
216
217 size_t size_of_data() const override { return prefix_size + nested_func->size_of_data(); }
218
219 size_t align_of_data() const override { return nested_func->align_of_data(); }
220
221 void create(AggregateDataPtr __restrict place) const override {
222 new (place) Data;
223 nested_func->create(get_nested_place(place));
224 }
225
226 void destroy(AggregateDataPtr __restrict place) const noexcept override {
227 this->data(place).~Data();
228 nested_func->destroy(get_nested_place(place));
229 }
230
231 String get_name() const override { return nested_func->get_name() + "Distinct"; }
232
233 DataTypePtr get_return_type() const override { return nested_func->get_return_type(); }
234
235 bool allocates_memory_in_arena() const override { return true; }
236};
