31// RowRange stands for range[From, To), From is inclusive,
32// To is exclusive. It is used for row id range calculation.
33class RowRange {
34public:
35 // Returns true if two ranges are overlapped or false.
36 // The union range will be returned through range.
37 static bool range_union(const RowRange& left, const RowRange& right, RowRange* range) {
38 if (left._from <= right._from) {
39 if (left._to >= right._from) {
40 range->_from = left._from;
41 range->_to = std::max(left._to, right._to);
42 return true;
43 }
44 } else if (right._to >= left._from) {
45 range->_from = right._from;
46 range->_to = std::max(left._to, right._to);
47 return true;
48 }
49 // return a invalid range
50 range->_from = 0;
51 range->_to = 0;
52 return false;
53 }
54
55 // Returns true if the two ranges are intersected or false.
56 // The intersection of the two ranges is returned through range.
57 static bool range_intersection(const RowRange& left, const RowRange& right, RowRange* range) {
58 if (left._from <= right._from) {
59 if (left._to > right._from) {
60 range->_from = right._from;
61 range->_to = std::min(left._to, right._to);
62 return true;
63 }
64 } else if (right._to > left._from) {
65 range->_from = left._from;
66 range->_to = std::min(left._to, right._to);
67 return true;
68 }
69 // return a invalid range
70 range->_from = 0;
71 range->_to = 0;
72 return false;
73 }
74
75 RowRange() : _from(0), _to(0) {}
76
77 // Creates a range of [from, to) (from inclusive and to exclusive; empty ranges are invalid)
78 RowRange(int64_t from, int64_t to) : _from(from), _to(to) {}
79
80 bool is_valid() const { return _from < _to; }
81
82 size_t count() const { return _to - _from; }
83
84 bool is_before(const RowRange& other) const { return _to <= other._from; }
85
86 bool is_after(const RowRange& other) const { return _from >= other._to; }
87
88 int64_t from() const { return _from; }
89
90 int64_t to() const { return _to; }
91
92 std::string to_string() const { return strings::Substitute("[$0-$1)", _from, _to); }
93
94private:
95 int64_t _from;
96 int64_t _to;
97};

123 // Calculates the union of the two specified RowRanges object. The union of two range is calculated if there are
124 // elements between them. Otherwise, the two disjunct ranges are stored separately.
125 // For example:
126 // [113, 241) ∪ [221, 340) = [113, 340)
127 // [113, 230) ∪ [230, 340) = [113, 340]
128 // while
129 // [113, 230) ∪ [231, 340) = [113, 230), [231, 340)
130 static void ranges_union(const RowRanges& left, const RowRanges& right, RowRanges* result) {
131 RowRanges tmp_range;
132 auto it1 = left._ranges.begin();
133 auto it2 = right._ranges.begin();
134 // merge and add
135 while (it1 != left._ranges.end() && it2 != right._ranges.end()) {
136 if (it1->is_after(*it2)) {
137 tmp_range.add(*it2);
138 ++it2;
139 } else {
140 tmp_range.add(*it1);
141 ++it1;
142 }
143 }
144 while (it1 != left._ranges.end()) {
145 tmp_range.add(*it1);
146 ++it1;
147 }
148 while (it2 != right._ranges.end()) {
149 tmp_range.add(*it2);
150 ++it2;
151 }
152 *result = std::move(tmp_range);
153 }
154
155 // Calculates the intersection of the two specified RowRanges object. Two ranges intersect if they have common
156 // elements otherwise the result is empty.
157 // For example:
158 // [113, 241) ∩ [221, 340) = [221, 241)
159 // while
160 // [113, 230) ∩ [230, 340) = <EMPTY>
161 //
162 // The result RowRanges object will contain all the row indexes there were contained in both of the specified objects
163 static void ranges_intersection(const RowRanges& left, const RowRanges& right,
164 RowRanges* result) {
165 RowRanges tmp_range;
166 int right_index = 0;
167 for (auto it1 = left._ranges.begin(); it1 != left._ranges.end(); ++it1) {
168 const RowRange& range1 = *it1;
169 for (int i = right_index; i < right._ranges.size(); ++i) {
170 const RowRange& range2 = right._ranges[i];
171 if (range1.is_before(range2)) {
172 break;
173 } else if (range1.is_after(range2)) {
174 right_index = i + 1;
175 continue;
176 }
177 RowRange merge_range;
178 bool ret = RowRange::range_intersection(range1, range2, &merge_range);
179 DCHECK(ret);
180 tmp_range.add(merge_range);
181 }
182 }
183 *result = std::move(tmp_range);
184 }

186 static roaring::Roaring ranges_to_roaring(const RowRanges& ranges) {
187 roaring::Roaring result;
188 for (auto it = ranges._ranges.begin(); it != ranges._ranges.end(); ++it) {
189 result.addRange(it->from(), it->to());
190 }
191 return result;
192 }

194 size_t count() { return _count; }
195
196 bool is_empty() { return _count == 0; }

198 bool contain(rowid_t from, rowid_t to) {
199 // binary search
200 RowRange tmp_range = RowRange(from, to);
201 int32_t start = 0;
202 int32_t end = _ranges.size();
203 while (start <= end) {
204 int32_t mid = (start + end) / 2;
205 if (_ranges[mid].is_before(tmp_range)) {
206 start = mid;
207 } else if (_ranges[mid].is_after(tmp_range)) {
208 end = mid - 1;
209 } else {
210 return true;
211 }
212 }
213 return false;
214 }

239 return result;

242 // Adds a range to the end of the list of ranges. It maintains the disjunct ascending order(*) of the ranges by
243 // trying to union the specified range to the last ranges in the list. The specified range shall be larger(*) than
244 // the last one or might be overlapped with some of the last ones.
245 void add(const RowRange& range) {
246 if (range.count() == 0) {
247 return;
248 }
249 RowRange range_to_add = range;
250 for (int i = _ranges.size() - 1; i >= 0; --i) {
251 const RowRange last = _ranges[i];
252 DCHECK(!last.is_after(range));
253 RowRange u;
254 bool ret = RowRange::range_union(last, range_to_add, &u);
255 if (!ret) {
256 // range do not intersect with the last
257 break;
258 }
259 range_to_add = u;
260 _ranges.erase(_ranges.begin() + i);
261 _count -= last.count();
262 }
263 _ranges.emplace_back(range_to_add);
264 _count += range_to_add.count();
265 }
266
267private:
268 std::vector<RowRange> _ranges;
269 size_t _count;
270};
