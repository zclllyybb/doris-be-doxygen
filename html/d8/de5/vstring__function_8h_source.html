59 static StringVal rtrim(const StringVal& str) {
60 if (str.is_null || str.len == 0) {
61 return str;
62 }
63 auto begin = 0;
64 auto end = str.len - 1;
65#if defined(__SSE2__) || defined(__aarch64__)
66 char blank = ' ';
67 const auto pattern = _mm_set1_epi8(blank);
68 while (end - begin + 1 >= REGISTER_SIZE) {
69 const auto v_haystack = _mm_loadu_si128(
70 reinterpret_cast<const __m128i*>(str.ptr + end + 1 - REGISTER_SIZE));
71 const auto v_against_pattern = _mm_cmpeq_epi8(v_haystack, pattern);
72 const auto mask = _mm_movemask_epi8(v_against_pattern);
73 int offset = __builtin_clz(~(mask << REGISTER_SIZE));
75 if (offset == 0) {
76 return StringVal(str.ptr + begin, end - begin + 1);
77 } else {
78 end -= offset;
79 }
80 }
81#endif
82 while (end >= begin && str.ptr[end] == ' ') {
83 --end;
84 }
85 if (end < 0) {
86 return StringVal("");
87 }
88 return StringVal(str.ptr + begin, end - begin + 1);
89 }

91 static StringVal ltrim(const StringVal& str) {
92 if (str.is_null || str.len == 0) {
93 return str;
94 }
95 auto begin = 0;
96 auto end = str.len - 1;
97#if defined(__SSE2__) || defined(__aarch64__)
98 char blank = ' ';
99 const auto pattern = _mm_set1_epi8(blank);
100 while (end - begin + 1 >= REGISTER_SIZE) {
101 const auto v_haystack =
102 _mm_loadu_si128(reinterpret_cast<const __m128i*>(str.ptr + begin));
103 const auto v_against_pattern = _mm_cmpeq_epi8(v_haystack, pattern);
104 const auto mask = _mm_movemask_epi8(v_against_pattern) ^ 0xffff;
106 if (mask == 0) {
107 begin += REGISTER_SIZE;
108 } else {
109 const auto offset = __builtin_ctz(mask);
110 begin += offset;
111 return StringVal(str.ptr + begin, end - begin + 1);
112 }
113 }
114#endif
115 while (begin <= end && str.ptr[begin] == ' ') {
116 ++begin;
117 }
118 return StringVal(str.ptr + begin, end - begin + 1);
119 }

129 static bool is_ascii(const StringVal& str) {
130 char or_code = 0;
131 for (size_t i = 0; i < str.len; i++) {
132 or_code |= str.ptr[i];
133 }
134 return !(or_code & 0x80);
135 }

137 static void reverse(const StringVal& str, StringVal dst) {
138 if (is_ascii(str)) {
139 int64_t begin = 0;
140 int64_t end = str.len;
141 int64_t result_end = dst.len - 1;
142
143 // auto SIMD here
144 auto* __restrict l = dst.ptr;
145 auto* __restrict r = str.ptr;
146 for (; begin < end; ++begin, --result_end) {
147 l[result_end] = r[begin];
148 }
149 } else {
150 for (size_t i = 0, char_size = 0; i < str.len; i += char_size) {
151 char_size = UTF8_BYTE_LENGTH[(unsigned char)(str.ptr)[i]];
152 // there exists occasion where the last character is an illegal UTF-8 one which returns
153 // a char_size larger than the actual space, which would cause offset execeeding the buffer range
154 // for example, consider str.len=4, i = 3, then the last char returns char_size 2, then
155 // the str.ptr + offset would exceed the buffer range
156 size_t offset = i + char_size;
157 if (offset > str.len) {
158 offset = str.len;
159 }
160 std::copy(str.ptr + i, str.ptr + offset, dst.ptr + str.len - offset);
161 }
162 }
163 }

165 static void hex_encode(const unsigned char* src_str, size_t length, char* dst_str) {
166 static constexpr auto hex_table = "0123456789ABCDEF";
167 auto src_str_end = src_str + length;
168
169#if defined(__SSE2__) || defined(__aarch64__)
170 constexpr auto step = sizeof(uint64);
171 if (src_str + step < src_str_end) {
172 const auto hex_map = _mm_loadu_si128(reinterpret_cast<const __m128i*>(hex_table));
173 const auto mask_map = _mm_set1_epi8(0x0F);
174
175 do {
176 auto data = _mm_loadu_si64(src_str);
177 auto hex_loc =
178 _mm_and_si128(_mm_unpacklo_epi8(_mm_srli_epi64(data, 4), data), mask_map);
179 _mm_storeu_si128(reinterpret_cast<__m128i*>(dst_str),
180 _mm_shuffle_epi8(hex_map, hex_loc));
181
182 src_str += step;
183 dst_str += step * 2;
184 } while (src_str + step < src_str_end);
185 }
186#endif
187 char res[2];
188 // hex(str) str length is n, result must be 2 * n length
189 for (; src_str < src_str_end; src_str += 1, dst_str += 2) {
190 // low 4 bits
191 *(res + 1) = hex_table[src_str[0] & 0x0F];
192 // high 4 bits
193 *res = hex_table[(src_str[0] >> 4)];
194 std::copy(res, res + 2, dst_str);
195 }
196 }
