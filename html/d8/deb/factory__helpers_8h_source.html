43template <std::size_t maximal_arity>
44inline void assert_arity_at_most(const std::string& name, const DataTypes& argument_types) {
45 if (argument_types.size() <= maximal_arity) {
46 return;
47 }
48
49 if constexpr (maximal_arity == 0) {
50 LOG(FATAL) << fmt::format("Aggregate function {} cannot have arguments", name);
51 }
52
53 if constexpr (maximal_arity == 1) {
54 LOG(FATAL) << fmt::format("Aggregate function {} requires zero or one argument", name);
55 }
56
57 LOG(FATAL) << fmt::format("Aggregate function {} requires at most {} arguments", name,
58 maximal_arity);
59}
