16// The AlphaNum type was designed to be used as the parameter type for StrCat().
17// I suppose that any routine accepting either a string or a number could accept
18// it. The basic idea is that by accepting a "const AlphaNum &" as an argument
19// to your function, your callers will automagically convert bools, integers,
20// and floating point values to strings for you.
21//
22// Conversion from 8-bit values is not accepted because if it were, then an
23// attempt to pass ':' instead of ":" might result in a 58 ending up in your
24// result.
25//
26// Bools convert to "0" or "1".
27//
28// Floating point values are converted to a string which, if passed to strtod(),
29// would produce the exact same original double (except in case of NaN; all NaNs
30// are considered the same value). We try to keep the string short but it's not
31// guaranteed to be as short as possible.
32//
33// This class has implicit constructors.
34// Style guide exception granted:
35// http://goto/style-guide-exception-20978288
36//
37struct AlphaNum {
38 StringPiece piece;
39 char digits[kFastToBufferSize];
40
41 // No bool ctor -- bools convert to an integral type.
42 // A bool ctor would also convert incoming pointers (bletch).
43
44 AlphaNum(int32 i32) // NOLINT(runtime/explicit)
45 : piece(digits, FastInt32ToBufferLeft(i32, digits) - &digits[0]) {}
46 AlphaNum(uint32 u32) // NOLINT(runtime/explicit)
47 : piece(digits, FastUInt32ToBufferLeft(u32, digits) - &digits[0]) {}
48 AlphaNum(int64 i64) // NOLINT(runtime/explicit)
49 : piece(digits, FastInt64ToBufferLeft(i64, digits) - &digits[0]) {}
50 AlphaNum(uint64 u64) // NOLINT(runtime/explicit)
51 : piece(digits, FastUInt64ToBufferLeft(u64, digits) - &digits[0]) {}
52
53#if defined(__APPLE__)
54 AlphaNum(size_t size) // NOLINT(runtime/explicit)
55 : piece(digits, FastUInt64ToBufferLeft(size, digits) - &digits[0]) {}
56#endif
57
58 AlphaNum(float f) // NOLINT(runtime/explicit)
59 : piece(digits, strlen(FloatToBuffer(f, digits))) {}
60 AlphaNum(double f) // NOLINT(runtime/explicit)
61 : piece(digits, strlen(DoubleToBuffer(f, digits))) {}
62
63 AlphaNum(const char* c_str) : piece(c_str) {} // NOLINT(runtime/explicit)
64 AlphaNum(StringPiece pc) : piece(std::move(pc)) {} // NOLINT(runtime/explicit)
65 AlphaNum(const string& s) : piece(s) {} // NOLINT(runtime/explicit)
66
67 StringPiece::size_type size() const { return piece.size(); }
68 const char* data() const { return piece.data(); }
69
70private:
71 // Use ":" not ':'
72 AlphaNum(char c); // NOLINT(runtime/explicit)
73};

77// ----------------------------------------------------------------------
78// StrCat()
79// This merges the given strings or numbers, with no delimiter. This
80// is designed to be the fastest possible way to construct a string out
81// of a mix of raw C strings, StringPieces, strings, bool values,
82// and numeric values.
83//
84// Don't use this for user-visible strings. The localization process
85// works poorly on strings built up out of fragments.
86//
87// For clarity and performance, don't use StrCat when appending to a
88// string. In particular, avoid using any of these (anti-)patterns:
89// str.append(StrCat(...)
90// str += StrCat(...)
91// str = StrCat(str, ...)
92// where the last is the worse, with the potential to change a loop
93// from a linear time operation with O(1) dynamic allocations into a
94// quadratic time operation with O(n) dynamic allocations. StrAppend
95// is a better choice than any of the above, subject to the restriction
96// of StrAppend(&str, a, b, c, ...) that none of the a, b, c, ... may
97// be a reference into str.
98// ----------------------------------------------------------------------

303// StrAppend()
304// Same as above, but adds the output to the given string.
305// WARNING: For speed, StrAppend does not try to check each of its input
306// arguments to be sure that they are not a subset of the string being
307// appended to. That is, while this will work:
308//
309// string s = "foo";
310// s += s;
311//
312// This will not (necessarily) work:
313//
314// string s = "foo";
315// StrAppend(&s, s);
316//
317// Note: while StrCat supports appending up to 12 arguments, StrAppend
