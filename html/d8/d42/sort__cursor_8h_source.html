30struct HeapSortCursorBlockView {
31public:
32 Block block;
33 ColumnRawPtrs sort_columns;
34 SortDescription& desc;
35
36 HeapSortCursorBlockView(Block&& cur_block, SortDescription& sort_desc)
37 : block(cur_block), desc(sort_desc) {
38 _reset();
39 }
40
41 void filter_block(IColumn::Filter& filter) {
42 Block::filter_block_internal(&block, filter, block.columns());
43 _reset();
44 }
45
46private:
47 void _reset() {
48 sort_columns.clear();
49 auto columns = block.get_columns();
50 for (size_t j = 0, size = desc.size(); j < size; ++j) {
51 auto& column_desc = desc[j];
52 size_t column_number = !column_desc.column_name.empty()
53 ? block.get_position_by_name(column_desc.column_name)
54 : column_desc.column_number;
55 sort_columns.push_back(columns[column_number].get());
56 }
57 }
58};

62class SharedHeapSortCursorBlockView {
63public:
64 SharedHeapSortCursorBlockView(HeapSortCursorBlockView&& reference)
65 : _ref_count(0), _reference(std::move(reference)) {}
66 SharedHeapSortCursorBlockView(const SharedHeapSortCursorBlockView&) = delete;
67 void unref() noexcept {
68 DCHECK_GT(_ref_count, 0);
69 _ref_count--;
70 if (_ref_count == 0) {
71 delete this;
72 }
73 }
74 void ref() noexcept { _ref_count++; }
75
76 HeapSortCursorBlockView& value() { return _reference; }
77
78 int ref_count() const { return _ref_count; }
79
80private:
81 ~SharedHeapSortCursorBlockView() noexcept = default;
82 int _ref_count;
83 HeapSortCursorBlockView _reference;

99 HeapSortCursorImpl(HeapSortCursorImpl&& other) {
100 _row_id = other._row_id;
101 _block_view = other._block_view;
102 other._block_view = nullptr;
103 }
104
105 HeapSortCursorImpl& operator=(HeapSortCursorImpl&& other) {
106 std::swap(_row_id, other._row_id);
107 std::swap(_block_view, other._block_view);
108 return *this;
109 }
110
111 ~HeapSortCursorImpl() {
112 if (_block_view) {
113 _block_view->unref();
114 }
115 }
116
117 size_t row_id() const { return _row_id; }
118
119 const ColumnRawPtrs& sort_columns() const { return _block_view->value().sort_columns; }
120
121 const Block* block() const { return &_block_view->value().block; }
122
123 const SortDescription& sort_desc() const { return _block_view->value().desc; }
124
125 bool operator<(const HeapSortCursorImpl& rhs) const {
126 for (size_t i = 0; i < sort_desc().size(); ++i) {
127 int direction = sort_desc()[i].direction;
128 int nulls_direction = sort_desc()[i].nulls_direction;
129 int res = direction * sort_columns()[i]->compare_at(row_id(), rhs.row_id(),
130 *(rhs.sort_columns()[i]),
131 nulls_direction);
132 // ASC: direction == 1. If bigger, res > 0. So we return true.
133 if (res < 0) {
134 return true;
135 }
136 if (res > 0) {
137 return false;
138 }
139 }
140 return false;
141 }
142
143private:
144 size_t _row_id;
145 SharedHeapSortCursorBlockView* _block_view;
146};

152struct MergeSortCursorImpl {
153 ColumnRawPtrs all_columns;
154 ColumnRawPtrs sort_columns;
155 SortDescription desc;
156 size_t sort_columns_size = 0;
157 size_t pos = 0;
158 size_t rows = 0;
159
160 MergeSortCursorImpl() = default;
161 virtual ~MergeSortCursorImpl() = default;
162
163 MergeSortCursorImpl(const Block& block, const SortDescription& desc_)
164 : desc(desc_), sort_columns_size(desc.size()) {
165 reset(block);
166 }
167
168 MergeSortCursorImpl(const SortDescription& desc_)
169 : desc(desc_), sort_columns_size(desc.size()) {}
170 bool empty() const { return rows == 0; }
171
173 void reset(const Block& block) { reset(block.get_columns(), block); }
174
176 void reset(const Columns& columns, const Block& block) {
177 all_columns.clear();
178 sort_columns.clear();
179
180 size_t num_columns = columns.size();
181
182 for (size_t j = 0; j < num_columns; ++j) {
183 all_columns.push_back(columns[j].get());
184 }
185
186 for (size_t j = 0, size = desc.size(); j < size; ++j) {
187 auto& column_desc = desc[j];
188 size_t column_number = !column_desc.column_name.empty()
189 ? block.get_position_by_name(column_desc.column_name)
190 : column_desc.column_number;
191 sort_columns.push_back(columns[column_number].get());
192 }
193
194 pos = 0;
195 rows = all_columns[0]->size();
196 }
197
198 bool isFirst() const { return pos == 0; }
199 bool isLast() const { return pos + 1 >= rows; }
200 void next() { ++pos; }
201
202 virtual bool has_next_block() { return false; }
203 virtual Block* block_ptr() { return nullptr; }
204};

206using BlockSupplier = std::function<Status(Block*, bool* eos)>;

208struct BlockSupplierSortCursorImpl : public MergeSortCursorImpl {
209 BlockSupplierSortCursorImpl(const BlockSupplier& block_supplier,
210 const std::vector<VExprContext*>& ordering_expr,
211 const std::vector<bool>& is_asc_order,
212 const std::vector<bool>& nulls_first)
213 : _ordering_expr(ordering_expr), _block_supplier(block_supplier) {
214 sort_columns_size = ordering_expr.size();
215
216 desc.resize(ordering_expr.size());
217 for (int i = 0; i < desc.size(); i++) {
218 desc[i].direction = is_asc_order[i] ? 1 : -1;
219 desc[i].nulls_direction = nulls_first[i] ? -desc[i].direction : desc[i].direction;
220 }
221 _is_eof = !has_next_block();
222 }
223
224 BlockSupplierSortCursorImpl(const BlockSupplier& block_supplier, const SortDescription& desc_)
225 : MergeSortCursorImpl(desc_), _block_supplier(block_supplier) {
226 _is_eof = !has_next_block();
227 }
228
229 bool has_next_block() override {
230 _block.clear();
231 auto status = _block_supplier(&_block, &_is_eof);
232 // If status not ok, upper callers could not detect whether it is eof or error.
233 // So that fatal here, and should throw exception in the future.
234 if (status.ok() && !_is_eof) {
235 if (_ordering_expr.size() > 0) {
236 for (int i = 0; status.ok() && i < desc.size(); ++i) {
237 // TODO yiguolei: throw exception if status not ok in the future
238 status = _ordering_expr[i]->execute(&_block, &desc[i].column_number);
239 }
240 }
241 MergeSortCursorImpl::reset(_block);
242 return status.ok();
243 }
244 return false;
245 }
246
247 Block* block_ptr() override {
248 if (_is_eof) {
249 return nullptr;
250 }
251 return &_block;
252 }
253
254 size_t columns_num() const { return all_columns.size(); }
255
256 Block create_empty_blocks() const {
257 size_t num_columns = columns_num();
258 MutableColumns columns(num_columns);
259 for (size_t i = 0; i < num_columns; ++i) {
260 columns[i] = all_columns[i]->clone_empty();
261 }
262 return _block.clone_with_columns(std::move(columns));
263 }
264
265 std::vector<VExprContext*> _ordering_expr;
266 Block _block;
267 BlockSupplier _block_supplier {};
268 bool _is_eof = false;
269};

272struct MergeSortCursor {
273 MergeSortCursorImpl* impl;
274
275 MergeSortCursor(MergeSortCursorImpl* impl_) : impl(impl_) {}
276 MergeSortCursorImpl* operator->() const { return impl; }
277
279 int8_t greater_at(const MergeSortCursor& rhs, size_t lhs_pos, size_t rhs_pos) const {
280 for (size_t i = 0; i < impl->sort_columns_size; ++i) {
281 int direction = impl->desc[i].direction;
282 int nulls_direction = impl->desc[i].nulls_direction;
283 int res = direction * impl->sort_columns[i]->compare_at(lhs_pos, rhs_pos,
284 *(rhs.impl->sort_columns[i]),
285 nulls_direction);
286 if (res > 0) {
287 return 1;
288 }
289 if (res < 0) {
290 return -1;
291 }
292 }
293 return 0;
294 }
295
297 bool totally_less(const MergeSortCursor& rhs) const {
298 if (impl->rows == 0 || rhs.impl->rows == 0) {
299 return false;
300 }
301
303 return greater_at(rhs, impl->rows - 1, 0) == -1;
304 }
305
306 bool greater(const MergeSortCursor& rhs) const {
307 return !impl->empty() && greater_at(rhs, impl->pos, rhs.impl->pos) > 0;
308 }
309
311 bool operator<(const MergeSortCursor& rhs) const { return greater(rhs); }
312};
