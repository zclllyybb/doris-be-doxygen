193 // Converted from Olap Date or Datetime
194 bool from_olap_datetime(uint64_t datetime) {
195 _neg = 0;
196 _type = TIME_DATETIME;
197 uint64_t date = datetime / 1000000;
198 uint64_t time = datetime % 1000000;
199
200 auto [year, month, day, hour, minute, second, microsecond] =
201 std::tuple {0, 0, 0, 0, 0, 0, 0};
202 year = date / 10000;
203 date %= 10000;
204 month = date / 100;
205 day = date % 100;
206 hour = time / 10000;
207 time %= 10000;
208 minute = time / 100;
209 second = time % 100;
210 microsecond = 0;
211
212 return check_range_and_set_time(year, month, day, hour, minute, second, microsecond, _type);
213 }

215 uint64_t to_olap_datetime() const {
216 uint64_t date_val = _year * 10000 + _month * 100 + _day;
217 uint64_t time_val = _hour * 10000 + _minute * 100 + _second;
218 return date_val * 1000000 + time_val;
219 }

221 bool from_olap_date(uint64_t date) {
222 _neg = 0;
223 _type = TIME_DATE;
224
225 auto [year, month, day, hour, minute, second, microsecond] =
226 std::tuple {0, 0, 0, 0, 0, 0, 0};
227
228 day = date & 0x1f;
229 date >>= 5;
230 month = date & 0x0f;
231 date >>= 4;
232 year = date;
233
234 return check_range_and_set_time(year, month, day, hour, minute, second, microsecond, _type);
235 }

244 return val;

294 // compute the diff between two datetime value
295 template <TimeUnit unit>
296 static int64_t datetime_diff(const DateTimeValue& ts_value1, const DateTimeValue& ts_value2) {
297 switch (unit) {
298 case YEAR: {
299 int year = (ts_value2.year() - ts_value1.year());
300 if (year > 0) {
301 year -= (ts_value2.to_datetime_int64() % 10000000000 -
302 ts_value1.to_datetime_int64() % 10000000000) < 0;
303 } else if (year < 0) {
304 year += (ts_value2.to_datetime_int64() % 10000000000 -
305 ts_value1.to_datetime_int64() % 10000000000) > 0;
306 }
307 return year;
308 }
309 case MONTH: {
310 int month = (ts_value2.year() - ts_value1.year()) * 12 +
311 (ts_value2.month() - ts_value1.month());
312 if (month > 0) {
313 month -= (ts_value2.to_datetime_int64() % 100000000 -
314 ts_value1.to_datetime_int64() % 100000000) < 0;
315 } else if (month < 0) {
316 month += (ts_value2.to_datetime_int64() % 100000000 -
317 ts_value1.to_datetime_int64() % 100000000) > 0;
318 }
319 return month;
320 }
321 case WEEK: {
322 int day = ts_value2.daynr() - ts_value1.daynr();
323 if (day > 0) {
324 day -= ts_value2.time_part_diff(ts_value1) < 0;
325 } else if (day < 0) {
326 day += ts_value2.time_part_diff(ts_value1) > 0;
327 }
328 return day / 7;
329 }
330 case DAY: {
331 int day = ts_value2.daynr() - ts_value1.daynr();
332 if (day > 0) {
333 day -= ts_value2.time_part_diff(ts_value1) < 0;
334 } else if (day < 0) {
335 day += ts_value2.time_part_diff(ts_value1) > 0;
336 }
337 return day;
338 }
339 case HOUR: {
340 int64_t second = ts_value2.second_diff(ts_value1);
341 int64_t hour = second / 60 / 60;
342 return hour;
343 }
344 case MINUTE: {
345 int64_t second = ts_value2.second_diff(ts_value1);
346 int64_t minute = second / 60;
347 return minute;
348 }
349 case SECOND: {
350 int64_t second = ts_value2.second_diff(ts_value1);
351 return second;
352 }
353 }
354 // Rethink the default return value
355 return 0;
356 }

417 // The bits in week_format has the following meaning:
418 // WEEK_MONDAY_FIRST (0)
419 // If not set:
420 // Sunday is first day of week
421 // If set:
422 // Monday is first day of week
423 //
424 // WEEK_YEAR (1)
425 // If not set:
426 // Week is in range 0-53
427 // Week 0 is returned for the last week of the previous year (for
428 // a date at start of january) In this case one can get 53 for the
429 // first week of next year. This flag ensures that the week is
430 // relevant for the given year. Note that this flag is only
431 // relevant if WEEK_JANUARY is not set.
432 // If set:
433 // Week is in range 1-53.
434 // In this case one may get week 53 for a date in January (when
435 // the week is that last week of previous year) and week 1 for a
436 // date in December.
437 //
438 // WEEK_FIRST_WEEKDAY (2)
439 // If not set
440 // Weeks are numbered according to ISO 8601:1988
441 // If set
442 // The week that contains the first 'first-day-of-week' is week 1.
443 //
444 // ISO 8601:1988 means that
445 // if the week containing January 1 has
446 // four or more days in the new year, then it is week 1;
447 // Otherwise it is the last week of the previous year, and the
448 // next week is week 1.

621 void from_packed_time(int64_t packed_time) {
622 _microsecond = packed_time % (1LL << 24);
623 int64_t ymdhms = packed_time >> 24;
624 int64_t ymd = ymdhms >> 17;
625 int64_t hms = ymdhms % (1 << 17);
626
627 _day = ymd % (1 << 5);
628 int64_t ym = ymd >> 5;
629 _month = ym % 13;
630 _year = ym / 13;
631 _year %= 10000;
632 _second = hms % (1 << 6);
633 _minute = (hms >> 6) % (1 << 6);
634 _hour = (hms >> 12);
635 _neg = 0;
636 _type = TIME_DATETIME;
637 }
638
639 int64_t make_packed_time(int64_t time, int64_t second_part) const {
640 return (time << 24) + second_part;
641 }

643 // To compatible with MySQL
644 int64_t to_int64_datetime_packed() const {
645 int64_t ymd = ((_year * 13 + _month) << 5) | _day;
646 int64_t hms = (_hour << 12) | (_minute << 6) | _second;
647 int64_t tmp = make_packed_time(((ymd << 17) | hms), _microsecond);
648 return _neg ? -tmp : tmp;
649 }

654 return _neg ? -tmp : tmp;
