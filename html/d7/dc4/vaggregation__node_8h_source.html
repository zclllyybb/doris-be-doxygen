75 size_t serialize_keys(const ColumnRawPtrs& key_columns, size_t num_rows) {
76 if (keys.size() < num_rows) {
77 keys.resize(num_rows);
78 }
79
80 size_t max_one_row_byte_size = 0;
81 for (const auto& column : key_columns) {
82 max_one_row_byte_size += column->get_max_row_byte_size();
83 }
84 size_t total_bytes = max_one_row_byte_size * num_rows;
85
86 if (total_bytes > SERIALIZE_KEYS_MEM_LIMIT_IN_BYTES) {
87 // reach mem limit, don't serialize in batch
88 // for simplicity, we just create a new arena here.
89 _arena.reset(new Arena());
90 size_t keys_size = key_columns.size();
91 for (size_t i = 0; i < num_rows; ++i) {
92 keys[i] = serialize_keys_to_pool_contiguous(i, keys_size, key_columns, *_arena);
93 }
94 keys_memory_usage = _arena->size();
95 } else {
96 _arena.reset();
97 if (total_bytes > _serialized_key_buffer_size) {
98 _serialized_key_buffer_size = total_bytes;
99 _mem_pool->clear();
100 _serialized_key_buffer = _mem_pool->allocate(_serialized_key_buffer_size, true);
101 }
102
103 for (size_t i = 0; i < num_rows; ++i) {
104 keys[i].data =
105 reinterpret_cast<char*>(_serialized_key_buffer + i * max_one_row_byte_size);
106 keys[i].size = 0;
107 }
108
109 for (const auto& column : key_columns) {
110 column->serialize_vec(keys, num_rows, max_one_row_byte_size);
111 }
112 keys_memory_usage = _serialized_key_buffer_size;
113 }
114 return max_one_row_byte_size;
115 }

270 using Data = TData;
271 using Key = typename Data::key_type;

282 template <typename Other>
283 AggregationMethodKeysFixed(const Other& other) : data(other.data) {}

285 using State = ColumnsHashing::HashMethodKeysFixed<typename Data::value_type, Key, Mapped,
286 has_nullable_keys, false>;
287
288 static void insert_key_into_columns(const Key& key, MutableColumns& key_columns,
289 const Sizes& key_sizes) {
290 size_t keys_size = key_columns.size();
291
292 static constexpr auto bitmap_size =
293 has_nullable_keys ? std::tuple_size<KeysNullMap<Key>>::value : 0;
295 size_t pos = bitmap_size;
296
297 for (size_t i = 0; i < keys_size; ++i) {
298 IColumn* observed_column;
299 ColumnUInt8* null_map;
300
301 bool column_nullable = false;
302 if constexpr (has_nullable_keys) {
303 column_nullable = is_column_nullable(*key_columns[i]);
304 }
305
307 if (column_nullable) {
308 ColumnNullable& nullable_col = assert_cast<ColumnNullable&>(*key_columns[i]);
309 observed_column = &nullable_col.get_nested_column();
310 null_map = assert_cast<ColumnUInt8*>(&nullable_col.get_null_map_column());
311 } else {
312 observed_column = key_columns[i].get();
313 null_map = nullptr;
314 }
315
316 bool is_null = false;
317 if (column_nullable) {
320 size_t bucket = i / 8;
321 size_t offset = i % 8;
322 UInt8 val = (reinterpret_cast<const UInt8*>(&key)[bucket] >> offset) & 1;
323 null_map->insert_value(val);
324 is_null = val == 1;
325 }
326
327 if (has_nullable_keys && is_null) {
328 observed_column->insert_default();
329 } else {
330 size_t size = key_sizes[i];
331 observed_column->insert_data(reinterpret_cast<const char*>(&key) + pos, size);
332 pos += size;
333 }
334 }
335 }
336
337 static void insert_keys_into_columns(std::vector<Key>& keys, MutableColumns& key_columns,
338 const size_t num_rows, const Sizes& key_sizes) {
339 for (size_t i = 0; i != num_rows; ++i) {
340 insert_key_into_columns(keys[i], key_columns, key_sizes);
341 }
342 }
343
344 void init_once() {
345 if (!inited) {
346 inited = true;
347 iterator = data.begin();
348 }
349 }
