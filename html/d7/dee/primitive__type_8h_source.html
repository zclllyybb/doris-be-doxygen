37constexpr bool is_enumeration_type(PrimitiveType type) {
38 switch (type) {
39 case TYPE_FLOAT:
40 case TYPE_DOUBLE:
41 case TYPE_NULL:
42 case TYPE_CHAR:
43 case TYPE_VARCHAR:
44 case TYPE_STRING:
45 case TYPE_DATETIME:
46 case TYPE_DATETIMEV2:
47 case TYPE_TIMEV2:
48 case TYPE_DECIMALV2:
49 case TYPE_DECIMAL32:
50 case TYPE_DECIMAL64:
51 case TYPE_DECIMAL128I:
52 case TYPE_BOOLEAN:
53 case TYPE_ARRAY:
54 case TYPE_STRUCT:
55 case TYPE_MAP:
56 case TYPE_HLL:
57 return false;
58 case TYPE_TINYINT:
59 case TYPE_SMALLINT:
60 case TYPE_INT:
61 case TYPE_BIGINT:
62 case TYPE_LARGEINT:
63 case TYPE_DATE:
64 case TYPE_DATEV2:
65 return true;
66
67 case INVALID_TYPE:
68 default:
69 DCHECK(false);
70 }
71
72 return false;
73}
