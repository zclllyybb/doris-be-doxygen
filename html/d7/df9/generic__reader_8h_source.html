36 }

38 std::unordered_set<std::string>* missing_cols) {
39 return Status::NotSupported("get_columns is not implemented");
40 }
41
42 virtual Status get_parsed_schema(std::vector<std::string>* col_names,

44 return Status::NotSupported("get_parsed_schema is not implemented for this reader.");

46 virtual ~GenericReader() = default;
47
50 virtual bool fill_all_columns() const { return _fill_all_columns; }

57 partition_columns,

59 return Status::OK();
60 }

67};
