31// The class is used to represent row's format in memory. Each row contains
32// multiple columns, some of which are key-columns (the rest are value-columns).
33// NOTE: If both key-columns and value-columns exist, then the key-columns
34// must be placed before value-columns.
35//
36// To compare two rows whose schemas are different, but they are from the same origin
37// we store all column schema maybe accessed here. And default access through column id
38class Schema {
39public:
40 Schema(TabletSchemaSPtr tablet_schema) {
41 size_t num_columns = tablet_schema->num_columns();
42 // ignore this column
43 if (tablet_schema->columns().back().name() == BeConsts::ROW_STORE_COL) {
44 --num_columns;
45 }
46 std::vector<ColumnId> col_ids(num_columns);
47 _unique_ids.resize(num_columns);
48 std::vector<TabletColumn> columns;
49 columns.reserve(num_columns);
50
51 size_t num_key_columns = 0;
52 for (uint32_t cid = 0; cid < num_columns; ++cid) {
53 col_ids[cid] = cid;
54 const TabletColumn& column = tablet_schema->column(cid);
55 _unique_ids[cid] = column.unique_id();
56 if (column.is_key()) {
57 ++num_key_columns;
58 }
59 if (column.name() == BeConsts::ROWID_COL) {
60 _rowid_col_idx = cid;
61 }
62 if (column.name() == VERSION_COL) {
63 _version_col_idx = cid;
64 }
65 columns.push_back(column);
66 }
67 _delete_sign_idx = tablet_schema->delete_sign_idx();
68 if (tablet_schema->has_sequence_col()) {
69 _has_sequence_col = true;
70 }
71 _init(columns, col_ids, num_key_columns);
72 }
73
74 // All the columns of one table may exist in the columns param, but col_ids is only a subset.
75 Schema(const std::vector<TabletColumn>& columns, const std::vector<ColumnId>& col_ids) {
76 size_t num_key_columns = 0;
77 _unique_ids.resize(columns.size());
78 for (size_t i = 0; i < columns.size(); ++i) {
79 if (columns[i].is_key()) {
80 ++num_key_columns;
81 }
82 if (columns[i].name() == DELETE_SIGN) {
83 _delete_sign_idx = i;
84 }
85 if (columns[i].name() == BeConsts::ROWID_COL) {
86 _rowid_col_idx = i;
87 }
88 if (columns[i].name() == VERSION_COL) {
89 _version_col_idx = i;
90 }
91 _unique_ids[i] = columns[i].unique_id();
92 }
93 _init(columns, col_ids, num_key_columns);
94 }
95
96 // Only for UT
97 Schema(const std::vector<TabletColumn>& columns, size_t num_key_columns) {
98 std::vector<ColumnId> col_ids(columns.size());
99 _unique_ids.resize(columns.size());
100 for (uint32_t cid = 0; cid < columns.size(); ++cid) {
101 col_ids[cid] = cid;
102 _unique_ids[cid] = columns[cid].unique_id();
103 }
104
105 _init(columns, col_ids, num_key_columns);
106 }
107
108 Schema(const std::vector<const Field*>& cols, size_t num_key_columns) {
109 std::vector<ColumnId> col_ids(cols.size());
110 _unique_ids.resize(cols.size());
111 for (uint32_t cid = 0; cid < cols.size(); ++cid) {
112 col_ids[cid] = cid;
113 if (cols.at(cid)->name() == DELETE_SIGN) {
114 _delete_sign_idx = cid;
115 }
116 if (cols.at(cid)->name() == VERSION_COL) {
117 _version_col_idx = cid;
118 }
119 _unique_ids[cid] = cols[cid]->unique_id();
120 }
121
122 _init(cols, col_ids, num_key_columns);
123 }
