1// Copyright 2007 Google Inc. All Rights Reserved.
2//
3// Character classification functions similar to standard <ctype.h>.
4// Some C++ implementations provide locale-sensitive implementations
5// of some <ctype.h> functions. These ascii_* functions are
6// hard-wired for ASCII. Hard-wired for ASCII is much faster.
7//
8// ascii_isalnum, ascii_isalpha, ascii_isascii, ascii_isblank,
9// ascii_iscntrl, ascii_isdigit, ascii_isgraph, ascii_islower,
10// ascii_isprint, ascii_ispunct, ascii_isspace, ascii_isupper,
11// ascii_isxdigit
12// Similar to the <ctype.h> functions with similar names.
13// Input parameter is an unsigned char. Return value is a bool.
14// If the input has a numerical value greater than 127
15// then the output is "false".
16//
17// ascii_tolower, ascii_toupper
18// Similar to the <ctype.h> functions with similar names.
19// Input parameter is an unsigned char. Return value is a char.
20// If the input is not an ascii {lower,upper}-case letter
21// (including numerical values greater than 127)
22// then the output is the same as the input.
23

26// Array of character information. This is an implementation detail.
27// The individual bits do not have names because the array definition is
28// already tightly coupled to these functions. Names would just make it
29// harder to read and debug.
