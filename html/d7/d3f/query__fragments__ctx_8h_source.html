39// Save the common components of fragments in a query.
40// Some components like DescriptorTbl may be very large
41// that will slow down each execution of fragments when DeSer them every time.
42class DescriptorTbl;
43class QueryFragmentsCtx {
44public:
45 QueryFragmentsCtx(int total_fragment_num, ExecEnv* exec_env)
46 : fragment_num(total_fragment_num), timeout_second(-1), _exec_env(exec_env) {
47 _start_time = DateTimeValue::local_time();
48 _shared_hash_table_controller.reset(new vectorized::SharedHashTableController());
49 }
50
51 ~QueryFragmentsCtx() {
52 // query mem tracker consumption is equal to 0, it means that after QueryFragmentsCtx is created,
53 // it is found that query already exists in _fragments_ctx_map, and query mem tracker is not used.
54 // query mem tracker consumption is not equal to 0 after use, because there is memory consumed
55 // on query mem tracker, released on other trackers.
56 if (query_mem_tracker->consumption() != 0) {
57 LOG(INFO) << fmt::format(
58 "Deregister query/load memory tracker, queryId={}, Limit={}, CurrUsed={}, "
59 "PeakUsed={}",
60 print_id(query_id), MemTracker::print_bytes(query_mem_tracker->limit()),
61 MemTracker::print_bytes(query_mem_tracker->consumption()),
62 MemTracker::print_bytes(query_mem_tracker->peak_consumption()));
63 }
64 }
65
66 // Notice. For load fragments, the fragment_num sent by FE has a small probability of 0.
67 // this may be a bug, bug <= 1 in theory it shouldn't cause any problems at this stage.
68 bool countdown() { return fragment_num.fetch_sub(1) <= 1; }
69
70 bool is_timeout(const DateTimeValue& now) const {
71 if (timeout_second <= 0) {
72 return false;
73 }
74 if (now.second_diff(_start_time) > timeout_second) {
75 return true;
76 }
77 return false;
78 }
79
80 void set_thread_token(int concurrency, bool is_serial) {
81 _thread_token = _exec_env->scanner_scheduler()->new_limited_scan_pool_token(
82 is_serial ? ThreadPool::ExecutionMode::SERIAL
83 : ThreadPool::ExecutionMode::CONCURRENT,
84 concurrency);
85 }
86
87 ThreadPoolToken* get_token() { return _thread_token.get(); }
88
89 void set_ready_to_execute(bool is_cancelled) {
90 {
91 std::lock_guard<std::mutex> l(_start_lock);
92 _is_cancelled = is_cancelled;
93 _ready_to_execute = true;
94 }
95 _start_cond.notify_all();
96 }
97 void set_ready_to_execute_only() {
98 {
99 std::lock_guard<std::mutex> l(_start_lock);
100 _ready_to_execute = true;
101 }
102 _start_cond.notify_all();
103 }
104
105 bool is_ready_to_execute() {
106 std::lock_guard<std::mutex> l(_start_lock);
107 return _ready_to_execute;
108 }
109
110 bool wait_for_start() {
111 int wait_time = config::max_fragment_start_wait_time_seconds;
112 std::unique_lock<std::mutex> l(_start_lock);
113 while (!_ready_to_execute.load() && !_is_cancelled.load() && --wait_time > 0) {
114 _start_cond.wait_for(l, std::chrono::seconds(1));
115 }
116 return _ready_to_execute.load() && !_is_cancelled.load();
117 }
118
119 std::shared_ptr<vectorized::SharedHashTableController> get_shared_hash_table_controller() {
120 return _shared_hash_table_controller;
121 }
122
123 vectorized::RuntimePredicate& get_runtime_predicate() { return _runtime_predicate; }
124
125public:
126 TUniqueId query_id;
127 DescriptorTbl* desc_tbl;
128 bool set_rsc_info = false;
129 std::string user;
130 std::string group;
131 TNetworkAddress coord_addr;
132 TQueryGlobals query_globals;
133
141 std::atomic<int> fragment_num;
142 int timeout_second;
143 ObjectPool obj_pool;
144 // MemTracker that is shared by all fragment instances running on this host.
145 std::shared_ptr<MemTrackerLimiter> query_mem_tracker;
146
147 std::vector<TUniqueId> fragment_ids;
148
149private:
150 ExecEnv* _exec_env;
151 DateTimeValue _start_time;
152
153 // A token used to submit olap scanner to the "_limited_scan_thread_pool",
154 // This thread pool token is created from "_limited_scan_thread_pool" from exec env.
155 // And will be shared by all instances of this query.
156 // So that we can control the max thread that a query can be used to execute.
157 // If this token is not set, the scanner will be executed in "_scan_thread_pool" in exec env.
158 std::unique_ptr<ThreadPoolToken> _thread_token;
159
160 std::mutex _start_lock;
161 std::condition_variable _start_cond;
162 // Only valid when _need_wait_execution_trigger is set to true in FragmentExecState.
163 // And all fragments of this query will start execution when this is set to true.
164 std::atomic<bool> _ready_to_execute {false};
165 std::atomic<bool> _is_cancelled {false};
166
167 std::shared_ptr<vectorized::SharedHashTableController> _shared_hash_table_controller;
168 vectorized::RuntimePredicate _runtime_predicate;
169};
