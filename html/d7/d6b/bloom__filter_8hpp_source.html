40// Bare metal bit set implementation. For performance reasons, this implementation does not
41// check for index bounds nor expand the bit set if the specified index is greater than the size.
42class BitSet {
43public:
44 BitSet() : _data(nullptr), _data_len(0) {}
45
46 ~BitSet() { SAFE_DELETE_ARRAY(_data); }
47
48 // Init BitSet with given bit_num, which will align up to uint64_t
49 bool init(uint32_t bit_num) {
50 if (bit_num <= 0) {
51 return false;
52 }
53
54 _data_len = (bit_num + sizeof(uint64_t) * 8 - 1) / (sizeof(uint64_t) * 8);
55 _data = new (std::nothrow) uint64_t[_data_len];
56 if (_data == nullptr) {
57 return false;
58 }
59
60 memset(_data, 0, _data_len * sizeof(uint64_t));
61 return true;
62 }
63
64 // Init BitSet with given buffer
65 bool init(uint64_t* data, uint32_t data_len) {
66 _data = data;
67 _data_len = data_len;
68 return true;
69 }
70
71 // Set the bit specified by param, note that uint64_t type contains 2^6 bits
72 void set(uint32_t index) { _data[index >> 6] |= 1L << (index % 64); }
73
74 // Return true if the bit specified by param is set
75 bool get(uint32_t index) const { return (_data[index >> 6] & (1L << (index % 64))) != 0; }
76
77 // Merge with another BitSet by byte, return false when the length is not equal
78 bool merge(const BitSet& set) {
79 if (_data_len != set.data_len()) {
80 return false;
81 }
82
83 for (uint32_t i = 0; i < _data_len; ++i) {
84 _data[i] |= set.data()[i];
85 }
86
87 return true;
88 }
89
90 // Convert BitSet to string to convenient debug and test
91 std::string to_string() const {
92 uint32_t bit_num = _data_len * sizeof(uint64_t) * 8;
93 std::string str(bit_num, '0');
94 for (uint32_t i = 0; i < bit_num; ++i) {
95 if ((_data[i >> 6] & (1L << i)) != 0) {
96 str[i] = '1';
97 }
98 }
99
100 return str;
101 }
102
103 uint64_t* data() const { return _data; }
104
105 uint32_t data_len() const { return _data_len; }
106
107 uint32_t bit_num() const { return _data_len * sizeof(uint64_t) * 8; }
108
109 void clear() { memset(_data, 0, _data_len * sizeof(uint64_t)); }
110
111 void reset() {
112 _data = NULL;
113 _data_len = 0;
114 }
115
116private:
117 uint64_t* _data;
118 uint32_t _data_len;
119};

121class BloomFilter {
122public:
123 BloomFilter() : _bit_num(0), _hash_function_num(0) {}
124 ~BloomFilter() {}
125
126 // Create BloomFilter with given entry num and fpp, which is used for loading data
127 bool init(int64_t expected_entries, double fpp) {
128 uint32_t bit_num = _optimal_bit_num(expected_entries, fpp);
129 if (!_bit_set.init(bit_num)) {
130 return false;
131 }
132
133 _bit_num = _bit_set.bit_num();
134 _hash_function_num = _optimal_hash_function_num(expected_entries, _bit_num);
135 return true;
136 }
137
138 // Create BloomFilter with given entry num and default fpp
139 bool init(int64_t expected_entries) {
140 return this->init(expected_entries, BLOOM_FILTER_DEFAULT_FPP);
141 }
142
143 // Init BloomFilter with given buffer, which is used for query
144 bool init(uint64_t* data, uint32_t len, uint32_t hash_function_num) {
145 _bit_num = sizeof(uint64_t) * 8 * len;
146 _hash_function_num = hash_function_num;
147 return _bit_set.init(data, len);
148 }
149
150 // Compute hash value of given buffer and add to BloomFilter
151 void add_bytes(const char* buf, uint32_t len) {
152 uint64_t hash = buf == nullptr ? BLOOM_FILTER_NULL_HASHCODE
153 : HashUtil::hash64(buf, len, DEFAULT_SEED);
154 add_hash(hash);
155 }
156
157 // Generate multiple hash value according to following rule:
158 // new_hash_value = hash_high_part + (i * hash_low_part)
159 void add_hash(uint64_t hash) {
160 uint32_t hash1 = (uint32_t)hash;
161 uint32_t hash2 = (uint32_t)(hash >> 32);
162
163 for (uint32_t i = 0; i < _hash_function_num; ++i) {
164 uint64_t combine_hash = hash1 + hash2 * i;
165 uint32_t index = combine_hash % _bit_num;
166 _bit_set.set(index);
167 }
168 }
169
170 // Compute hash value of given buffer and verify whether exist in BloomFilter
171 bool test_bytes(const char* buf, uint32_t len) const {
172 uint64_t hash = buf == nullptr ? BLOOM_FILTER_NULL_HASHCODE
173 : HashUtil::hash64(buf, len, DEFAULT_SEED);
174 return test_hash(hash);
175 }
176
177 // Verify whether hash value in BloomFilter
178 bool test_hash(uint64_t hash) const {
179 uint32_t hash1 = (uint32_t)hash;
180 uint32_t hash2 = (uint32_t)(hash >> 32);
181
182 for (uint32_t i = 0; i < _hash_function_num; ++i) {
183 uint64_t combine_hash = hash1 + hash2 * i;
184 uint32_t index = combine_hash % _bit_num;
185 if (!_bit_set.get(index)) {
186 return false;
187 }
188 }
189
190 return true;
191 }
192
193 // Merge with another BloomFilter, return false when the length
194 // and hash function number is not equal
195 bool merge(const BloomFilter& that) {
196 if (_bit_num == that.bit_num() && _hash_function_num == that.hash_function_num()) {
197 _bit_set.merge(that.bit_set());
198 return true;
199 }
200
201 return false;
202 }
203
204 void clear() { _bit_set.clear(); }
205
206 void reset() {
207 _bit_num = 0;
208 _hash_function_num = 0;
209 _bit_set.reset();
210 }
211
212 uint32_t bit_num() const { return _bit_num; }
213
214 uint32_t hash_function_num() const { return _hash_function_num; }
215
216 const BitSet& bit_set() const { return _bit_set; }
217
218 uint64_t* bit_set_data() const { return _bit_set.data(); }
219
220 uint32_t bit_set_data_len() const { return _bit_set.data_len(); }
221
222 // Convert BloomFilter to string to convenient debug and test
223 std::string to_string() const {
224 std::stringstream bf_stream;
225 bf_stream << "bit_num:" << _bit_num << " hash_function_num:" << _hash_function_num
226 << " bit_set:" << _bit_set.to_string();
227 return bf_stream.str();
228 }
229
230 // Get points which set by given buffer in the BitSet
231 std::string get_bytes_points_string(const char* buf, uint32_t len) const {
232 uint64_t hash = buf == nullptr ? BLOOM_FILTER_NULL_HASHCODE
233 : HashUtil::hash64(buf, len, DEFAULT_SEED);
234 uint32_t hash1 = (uint32_t)hash;
235 uint32_t hash2 = (uint32_t)(hash >> 32);
236
237 std::stringstream stream;
238 for (uint32_t i = 0; i < _hash_function_num; ++i) {
239 if (i != 0) {
240 stream << "-";
241 }
242
243 uint32_t combine_hash = hash1 + hash2 * i;
244 uint32_t index = combine_hash % _bit_num;
245 stream << index;
246 }
247
248 return stream.str();
249 }
250
251private:
252 // Compute the optimal bit number according to the following rule:
253 // m = -n * ln(fpp) / (ln(2) ^ 2)
254 uint32_t _optimal_bit_num(int64_t n, double fpp) {
255 return (uint32_t)(-n * log(fpp) / (log(2) * log(2)));
256 }
257
258 // Compute the optimal hash function number according to the following rule:
259 // k = round(m * ln(2) / n)
260 uint32_t _optimal_hash_function_num(int64_t n, uint32_t m) {
261 uint32_t k = (uint32_t)round(m * log(2) / n);
262 return k > 1 ? k : 1;
263 }
264
265 BitSet _bit_set;
266 uint32_t _bit_num;
267 uint32_t _hash_function_num;
268};
