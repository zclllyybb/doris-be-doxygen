67 std::shared_ptr<T> get_client(const std::string& host_port) {
68 std::shared_ptr<T> stub_ptr;
69 auto get_value = [&stub_ptr](const auto& v) {
70 // remove those compatibility codes when we finish upgrade phmap.
71 if constexpr (std::is_same_v<const typename StubMap<T>::mapped_type&, decltype(v)>) {
72 stub_ptr = v;
73 } else {
74 stub_ptr = v.second;
75 }
76 };
77 if (LIKELY(_stub_map.if_contains(host_port, get_value))) {
78 return stub_ptr;
79 }
80
81 // new one stub and insert into map
82 auto stub = get_new_client_no_cache(host_port);
83 _stub_map.try_emplace_l(
84 host_port,
85 [&stub](const auto& v) {
86 // remove those compatibility codes when we finish upgrade phmap.
87 if constexpr (std::is_same_v<const typename StubMap<T>::mapped_type&,
88 decltype(v)>) {
89 stub = v;
90 } else {
91 stub = v.second;
92 }
93 },
94 stub);
95 return stub;
96 }

98 std::shared_ptr<T> get_new_client_no_cache(const std::string& host_port,
99 const std::string& protocol = "baidu_std",
100 const std::string& connect_type = "") {
101 brpc::ChannelOptions options;
102 if constexpr (std::is_same_v<T, PFunctionService_Stub>) {
103 options.protocol = config::function_service_protocol;
104 } else {
105 options.protocol = protocol;
106 }
107 if (connect_type != "") {
108 options.connection_type = connect_type;
109 }
110 options.connect_timeout_ms = 2000;
111 options.max_retry = 10;
112
113 std::unique_ptr<brpc::Channel> channel(new brpc::Channel());
114 int ret_code = 0;
115 if (host_port.find("://") == std::string::npos) {
116 ret_code = channel->Init(host_port.c_str(), &options);
117 } else {
118 ret_code =
119 channel->Init(host_port.c_str(), config::rpc_load_balancer.c_str(), &options);
120 }
121 if (ret_code) {
122 return nullptr;
123 }
124 return std::make_shared<T>(channel.release(), google::protobuf::Service::STUB_OWNS_CHANNEL);
125 }

156 bool available(std::shared_ptr<T> stub, const std::string& host_port) {
157 if (!stub) {
158 LOG(WARNING) << "stub is null to: " << host_port;
159 return false;
160 }
161 std::string message = "hello doris!";
162 PHandShakeRequest request;
163 request.set_hello(message);
164 PHandShakeResponse response;
165 brpc::Controller cntl;
166 stub->hand_shake(&cntl, &request, &response, nullptr);
167 if (cntl.Failed()) {
168 LOG(WARNING) << "open brpc connection to " << host_port
169 << " failed: " << cntl.ErrorText();
170 return false;
171 } else if (response.has_status() && response.has_hello() && response.hello() == message &&
172 response.status().status_code() == 0) {
173 return true;
174 } else {
175 LOG(WARNING) << "open brpc connection to " << host_port
176 << " failed: " << response.DebugString();
177 return false;
178 }
179 }

182 std::string host_port = fmt::format("{}:{}", host, port);
183 return available(stub, host_port);
