53 // Deserialize field value from incoming string.
54 //
55 // NOTE: the parameter must be a '\0' terminated string. It do not include the null flag.
56 Status from_string(const std::string& value_string, const int precision = 0,
57 const int scale = 0) {
58 if (_is_string_type) {
59 if (value_string.size() > _var_length) {
60 Slice* slice = reinterpret_cast<Slice*>(cell_ptr());
61 slice->size = value_string.size();
62 _var_length = slice->size;
63 _string_content.reset(new char[slice->size]);
64 slice->data = _string_content.get();
65 }
66 }
67 return _rep->from_string(_field_buf + 1, value_string, precision, scale);
68 }
69

73 size_t field_size() const { return _rep->field_size(); }
74 bool is_null() const { return *reinterpret_cast<bool*>(_field_buf); }

76 void set_null() { *reinterpret_cast<bool*>(_field_buf) = true; }
77 void set_not_null() { *reinterpret_cast<bool*>(_field_buf) = false; }
78 char* nullable_cell_ptr() const { return _field_buf; }
79 void set_to_max() { _rep->set_to_max(_field_buf + 1); }
80 void set_to_min() { _rep->set_to_min(_field_buf + 1); }
81 void* cell_ptr() const { return _field_buf + 1; }
82 void* mutable_cell_ptr() const { return _field_buf + 1; }
83 const Field* field() const { return _rep; }
84
85 int cmp(const WrapperField* field) const { return _rep->compare_cell(*this, *field); }
86
87 void copy(const WrapperField* field) { _rep->direct_copy(this, *field); }
88
89private:

99 // Memory for string type field.
100 std::unique_ptr<char[]> _string_content;
