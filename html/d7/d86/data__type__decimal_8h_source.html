201 T scale_factor_for(const DataTypeDecimal<U>& x, bool) const {
202 if (get_scale() < x.get_scale()) {

225template <typename T, typename U>
226DataTypePtr decimal_result_type(const DataTypeDecimal<T>& tx, const DataTypeDecimal<U>& ty,
227 bool is_multiply, bool is_divide, bool is_plus_minus) {
228 using Type = std::conditional_t<sizeof(T) >= sizeof(U), T, U>;
229 if constexpr (IsDecimalV2<T> && IsDecimalV2<U>) {
230 return std::make_shared<DataTypeDecimal<Type>>((max_decimal_precision<T>(), 9));
231 } else {
232 UInt32 scale = std::max(tx.get_scale(), ty.get_scale());
233 auto precision = max_decimal_precision<Type>();
234
235 size_t multiply_precision = tx.get_precision() + ty.get_precision();
236 size_t divide_precision = tx.get_precision() + ty.get_scale();
237 size_t plus_minus_precision =
238 std::max(tx.get_precision() - tx.get_scale(), ty.get_precision() - ty.get_scale()) +
239 scale;
240 if (is_multiply) {
241 scale = tx.get_scale() + ty.get_scale();
242 precision = std::min(multiply_precision, max_decimal_precision<Decimal128I>());
243 } else if (is_divide) {
244 scale = tx.get_scale();
245 precision = std::min(divide_precision, max_decimal_precision<Decimal128I>());
246 } else if (is_plus_minus) {
247 precision = std::min(plus_minus_precision, max_decimal_precision<Decimal128I>());
248 }
249 return create_decimal(precision, scale, false);
250 }
251}

295template <typename FromDataType, typename ToDataType>
296inline std::enable_if_t<IsDataTypeDecimal<FromDataType> && IsDataTypeDecimal<ToDataType>,
297 typename ToDataType::FieldType>
298convert_decimals(const typename FromDataType::FieldType& value, UInt32 scale_from, UInt32 scale_to,
299 UInt8* overflow_flag = nullptr) {
300 using FromFieldType = typename FromDataType::FieldType;
301 using ToFieldType = typename ToDataType::FieldType;
302 using MaxFieldType =
303 std::conditional_t<(sizeof(FromFieldType) == sizeof(ToFieldType)) &&
304 (std::is_same_v<ToFieldType, Decimal128I> ||
305 std::is_same_v<FromFieldType, Decimal128I>),
306 Decimal128I,
307 std::conditional_t<(sizeof(FromFieldType) > sizeof(ToFieldType)),
308 FromFieldType, ToFieldType>>;
309 using MaxNativeType = typename MaxFieldType::NativeType;
310
311 MaxNativeType converted_value;
312 if (scale_to > scale_from) {
313 converted_value =
314 DataTypeDecimal<MaxFieldType>::get_scale_multiplier(scale_to - scale_from);
315 if (common::mul_overflow(static_cast<MaxNativeType>(value), converted_value,
316 converted_value)) {
317 if (overflow_flag) {
318 *overflow_flag = 1;
319 }
320 VLOG_DEBUG << "Decimal convert overflow";
321 return converted_value < 0
322 ? std::numeric_limits<typename ToFieldType::NativeType>::min()
323 : std::numeric_limits<typename ToFieldType::NativeType>::max();
324 }
325 } else {
326 converted_value =
327 value / DataTypeDecimal<MaxFieldType>::get_scale_multiplier(scale_from - scale_to);
328 }
329
330 if constexpr (sizeof(FromFieldType) > sizeof(ToFieldType)) {
331 if (converted_value < std::numeric_limits<typename ToFieldType::NativeType>::min()) {
332 if (overflow_flag) {
333 *overflow_flag = 1;
334 }
335 VLOG_DEBUG << "Decimal convert overflow";
336 return std::numeric_limits<typename ToFieldType::NativeType>::min();
337 }
338 if (converted_value > std::numeric_limits<typename ToFieldType::NativeType>::max()) {
339 if (overflow_flag) {
340 *overflow_flag = 1;
341 }
342 VLOG_DEBUG << "Decimal convert overflow";
343 return std::numeric_limits<typename ToFieldType::NativeType>::max();
344 }
345 }
346
347 return converted_value;
348}

350template <typename FromDataType, typename ToDataType>
351inline std::enable_if_t<IsDataTypeDecimal<FromDataType> && IsDataTypeNumber<ToDataType>,
352 typename ToDataType::FieldType>
353convert_from_decimal(const typename FromDataType::FieldType& value, UInt32 scale) {
354 using FromFieldType = typename FromDataType::FieldType;
355 using ToFieldType = typename ToDataType::FieldType;
356
357 if constexpr (std::is_floating_point_v<ToFieldType>) {
358 if constexpr (IsDecimalV2<FromFieldType>) {
359 return binary_cast<int128_t, DecimalV2Value>(value);
360 } else {
361 return static_cast<ToFieldType>(value) / FromDataType::get_scale_multiplier(scale);
362 }
363 } else {
364 FromFieldType converted_value =
365 convert_decimals<FromDataType, FromDataType>(value, scale, 0);
366
367 if constexpr (sizeof(FromFieldType) > sizeof(ToFieldType) ||
368 !std::numeric_limits<ToFieldType>::is_signed) {
369 if constexpr (std::numeric_limits<ToFieldType>::is_signed) {
370 if (converted_value < std::numeric_limits<ToFieldType>::min()) {
371 VLOG_DEBUG << "Decimal convert overflow";
372 return std::numeric_limits<ToFieldType>::min();
373 }
374 if (converted_value > std::numeric_limits<ToFieldType>::max()) {
375 VLOG_DEBUG << "Decimal convert overflow";
376 return std::numeric_limits<ToFieldType>::max();
377 }
378 } else {
379 using CastIntType =
380 std::conditional_t<std::is_same_v<ToFieldType, UInt64>, Int128, Int64>;
381
382 if (converted_value < 0 ||
383 converted_value >
384 static_cast<CastIntType>(std::numeric_limits<ToFieldType>::max())) {
385 VLOG_DEBUG << "Decimal convert overflow";
386 return std::numeric_limits<ToFieldType>::max();
387 }
388 }
389 }
390 return converted_value;
391 }

394template <typename FromDataType, typename ToDataType>
395inline std::enable_if_t<IsDataTypeNumber<FromDataType> && IsDataTypeDecimal<ToDataType>,
396 typename ToDataType::FieldType>
397convert_to_decimal(const typename FromDataType::FieldType& value, UInt32 scale,
398 UInt8* overflow_flag) {
399 using FromFieldType = typename FromDataType::FieldType;
400 using ToNativeType = typename ToDataType::FieldType::NativeType;
401
402 if constexpr (std::is_floating_point_v<FromFieldType>) {
403 if (!std::isfinite(value)) {
404 if (overflow_flag) {
405 *overflow_flag = 1;
406 }
407 VLOG_DEBUG << "Decimal convert overflow. Cannot convert infinity or NaN to decimal";
408 return value < 0 ? std::numeric_limits<ToNativeType>::min()
409 : std::numeric_limits<ToNativeType>::max();
410 }
411
412 FromFieldType out;
413 out = value * ToDataType::get_scale_multiplier(scale);
414 if (out <= static_cast<FromFieldType>(std::numeric_limits<ToNativeType>::min())) {
415 if (overflow_flag) {
416 *overflow_flag = 1;
417 }
418 VLOG_DEBUG << "Decimal convert overflow. Float is out of Decimal range";
419 return std::numeric_limits<ToNativeType>::min();
420 }
421 if (out >= static_cast<FromFieldType>(std::numeric_limits<ToNativeType>::max())) {
422 if (overflow_flag) {
423 *overflow_flag = 1;
424 }
425 VLOG_DEBUG << "Decimal convert overflow. Float is out of Decimal range";
426 return std::numeric_limits<ToNativeType>::max();
427 }
428 return out;
429 } else {
430 if constexpr (std::is_same_v<FromFieldType, UInt64>) {
431 if (value > static_cast<UInt64>(std::numeric_limits<Int64>::max())) {
432 return convert_decimals<DataTypeDecimal<Decimal128>, ToDataType>(value, 0, scale);
433 }
434 }
435 return convert_decimals<DataTypeDecimal<Decimal64>, ToDataType>(value, 0, scale);
436 }
437}
