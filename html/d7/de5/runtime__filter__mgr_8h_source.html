159// RuntimeFilterMergeController has a map query-id -> entity
160class RuntimeFilterMergeController {
161public:

163 ~RuntimeFilterMergeController() = default;
164
165 // thread safe
166 // add a query-id -> entity
167 // If a query-id -> entity already exists
168 // add_entity will return a exists entity
169 Status add_entity(const TExecPlanFragmentParams& params,
170 std::shared_ptr<RuntimeFilterMergeControllerEntity>* handle,
171 RuntimeState* state);
172 Status add_entity(const TPipelineFragmentParams& params,
173 const TPipelineInstanceParams& local_params,
174 std::shared_ptr<RuntimeFilterMergeControllerEntity>* handle,
175 RuntimeState* state);
176 // thread safe
177 // increase a reference count
178 // if a query-id is not exist
179 // Status.not_ok will be returned and a empty ptr will returned by *handle
180 Status acquire(UniqueId query_id, std::shared_ptr<RuntimeFilterMergeControllerEntity>* handle);
181
182 // thread safe
183 // remove a entity by query-id
184 // remove_entity will be called automatically by entity when entity is destroyed
185 Status remove_entity(UniqueId query_id);
186
187 static const int kShardNum = 128;
188
189private:
190 uint32_t _get_controller_shard_idx(UniqueId& query_id) {
191 return (uint32_t)query_id.hi % kShardNum;
192 }
193
194 std::mutex _controller_mutex[kShardNum];
195 // We store the weak pointer here.
196 // When the external object is destroyed, we need to clear this record
197 using FilterControllerMap =
198 std::unordered_map<std::string, std::weak_ptr<RuntimeFilterMergeControllerEntity>>;
199 // str(query-id) -> entity
200 FilterControllerMap _filter_controller_map[kShardNum];
201};
