35// Implementation of a free pool to recycle allocations. The pool is broken
36// up into 64 lists, one for each power of 2. Each allocation is rounded up
37// to the next power of 2. When the allocation is freed, it is added to the
38// corresponding free list.
39// Each allocation has an 8 byte header that immediately precedes the actual
40// allocation. If the allocation is owned by the user, the header contains
41// the ptr to the list that it should be added to on Free().
42// When the allocation is in the pool (i.e. available to be handed out), it
43// contains the link to the next allocation.
44// This has O(1) Allocate() and Free().
45// This is not thread safe.
46// TODO(zxy): consider integrating this with MemPool.
47// TODO: consider changing to something more granular than doubling.
48class FreePool {
49public:
50 // C'tor, initializes the FreePool to be empty. All allocations come from the
51 // 'mem_pool'.
52 FreePool(MemPool* mem_pool) : _mem_pool(mem_pool) { memset(&_lists, 0, sizeof(_lists)); }
53
54 virtual ~FreePool() {}
55
56 // Allocates a buffer of size.
57 uint8_t* allocate(int64_t size) {
58 // This is the typical malloc behavior. NULL is reserved for failures.
59 if (size == 0) {
60 return reinterpret_cast<uint8_t*>(0x1);
61 }
62
63 // Do ceil(log_2(size))
64 int free_list_idx = BitUtil::log2(size);
65 DCHECK_LT(free_list_idx, NUM_LISTS);
66
67 FreeListNode* allocation = _lists[free_list_idx].next;
68
69 if (allocation == NULL) {
70 // There wasn't an existing allocation of the right size, allocate a new one.
71 size = 1L << free_list_idx;
72 allocation = reinterpret_cast<FreeListNode*>(
73 _mem_pool->allocate(size + sizeof(FreeListNode)));
74 } else {
75 // Remove this allocation from the list.
76 _lists[free_list_idx].next = allocation->next;
77 }
78
79 DCHECK(allocation != NULL);
80 // Set the back node to point back to the list it came from so know where
81 // to add it on free().
82 allocation->list = &_lists[free_list_idx];
83 return reinterpret_cast<uint8_t*>(allocation) + sizeof(FreeListNode);
84 }
85
86 // Allocates a buffer of size.
87 uint8_t* aligned_allocate(int alignment, int64_t size) {
88 // The alignment should be a power of 2.
89 DCHECK(alignment > 0 && ((alignment - 1) & alignment) == 0);
90
91 // This is the typical malloc behavior. NULL is reserved for failures.
92 if (size == 0) {
93 return reinterpret_cast<uint8_t*>(alignment);
94 }
95
96 int padding = sizeof(FreeListNode) >= alignment ? 0 : (alignment - sizeof(FreeListNode));
97 size += padding;
98
99 // Do ceil(log_2(size))
100 int free_list_idx = BitUtil::log2(size);
101 DCHECK_LT(free_list_idx, NUM_LISTS);
102
103 FreeListNode* allocation = _lists[free_list_idx].next;
104
105 if (allocation == nullptr) {
106 // There wasn't an existing allocation of the right size, allocate a new one.
107 size = 1L << free_list_idx;
108 allocation = reinterpret_cast<FreeListNode*>(
109 _mem_pool->allocate_aligned(size + sizeof(FreeListNode), alignment));
110 } else {
111 // Remove this allocation from the list.
112 _lists[free_list_idx].next = allocation->next;
113 }
114
115 DCHECK(allocation != nullptr);
116 // Set the back node to point back to the list it came from so know where
117 // to add it on free().
118 allocation->list = &_lists[free_list_idx];
119 return reinterpret_cast<uint8_t*>(allocation) + sizeof(FreeListNode) + padding;
120 }

122 void free(uint8_t* ptr) {
123 if (ptr == NULL || reinterpret_cast<int64_t>(ptr) == 0x1) {
124 return;
125 }
126
127 FreeListNode* node = reinterpret_cast<FreeListNode*>(ptr - sizeof(FreeListNode));
128 FreeListNode* list = node->list;
129#ifndef NDEBUG
130 check_valid_allocation(list);
131#endif
132 if (UNLIKELY(nullptr == list)) {
133 // free memory directly if the pointer to free list is null
134 LOG(ERROR) << "The free list was released, and this may cause memory leak.";
135 free(ptr);
136 } else {
137 // Add node to front of list.
138 node->next = list->next;
139 list->next = node;
140 }
141 }
142
143 // Returns an allocation that is at least 'size'. If the current allocation
144 // backing 'ptr' is big enough, 'ptr' is returned. Otherwise a new one is
145 // made and the contents of ptr are copied into it.
146 uint8_t* reallocate(uint8_t* ptr, int64_t size) {
147 if (ptr == NULL || reinterpret_cast<int64_t>(ptr) == 0x1) {
148 return allocate(size);
149 }
150
151 FreeListNode* node = reinterpret_cast<FreeListNode*>(ptr - sizeof(FreeListNode));
152 FreeListNode* list = node->list;
153#ifndef NDEBUG
154 check_valid_allocation(list);
155#endif
156 int bucket_idx = (list - &_lists[0]);
157 // This is the actual size of ptr.
158 int allocation_size = 1 << bucket_idx;
159
160 // If it's already big enough, just return the ptr.
161 if (allocation_size >= size) {
162 return ptr;
163 }
164
165 // Make a new one. Since allocate() already rounds up to powers of 2, this
166 // effectively doubles for the caller.
167 uint8_t* new_ptr = allocate(size);
168 memcpy(new_ptr, ptr, allocation_size);
169 free(ptr);
170 return new_ptr;
171 }
172
173private:
174 static const int NUM_LISTS = 64;
175
176 struct FreeListNode {
177 // Union for clarity when manipulating the node.
178 union {
179 FreeListNode* next; // Used when it is in the free list
180 FreeListNode* list; // Used when it is being used by the caller.
181 };
182 };
183
184 void check_valid_allocation(FreeListNode* computed_list_ptr) {
185 // On debug, check that list is valid.
186 bool found = false;
187
188 for (int i = 0; i < NUM_LISTS && !found; ++i) {
189 if (computed_list_ptr == &_lists[i]) {
190 found = true;
191 }
192 }
193
194 DCHECK(found);
195 }
196
197 // MemPool to allocate from. Unowned.
198 MemPool* _mem_pool;
199
200 // One list head for each allocation size indexed by the LOG_2 of the allocation size.
201 // While it doesn't make too much sense to use this for very small (e.g. 8 byte)
202 // allocations, it makes the indexing easy.
203 FreeListNode _lists[NUM_LISTS];
204};
