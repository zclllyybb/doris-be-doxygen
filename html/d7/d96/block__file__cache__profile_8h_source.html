89 // avoid performance impact, use https to control
90 inline static std::atomic<bool> s_enable_profile = true;
91
92 static void set_enable_profile(bool flag) {
93 // if enable_profile = false originally, set true, it will clear the count
94 if (!s_enable_profile && flag) {
95 std::lock_guard lock(instance()._mtx);
96 instance()._profile.clear();
97 }
98 s_enable_profile.store(flag, std::memory_order_release);
99 }

115struct FileCacheProfileReporter {
116 RuntimeProfile::Counter* num_io_total;
117 RuntimeProfile::Counter* num_io_hit_cache;
118 RuntimeProfile::Counter* num_io_bytes_read_total;
119 RuntimeProfile::Counter* num_io_bytes_read_from_file_cache;
120 RuntimeProfile::Counter* num_io_bytes_read_from_write_cache;
121 RuntimeProfile::Counter* num_io_written_in_file_cache;
122 RuntimeProfile::Counter* num_io_bytes_written_in_file_cache;
123 RuntimeProfile::Counter* num_io_bytes_skip_cache;
124
125 FileCacheProfileReporter(RuntimeProfile* profile) {
126 static const char* cache_profile = "FileCache";
127 ADD_TIMER(profile, cache_profile);
128 num_io_total = ADD_CHILD_COUNTER(profile, "IOTotalNum", TUnit::UNIT, cache_profile);
129 num_io_hit_cache = ADD_CHILD_COUNTER(profile, "IOHitCacheNum", TUnit::UNIT, cache_profile);
130 num_io_bytes_read_total =
131 ADD_CHILD_COUNTER(profile, "ReadTotalBytes", TUnit::BYTES, cache_profile);
132 num_io_bytes_read_from_file_cache =
133 ADD_CHILD_COUNTER(profile, "ReadFromFileCacheBytes", TUnit::BYTES, cache_profile);
134 num_io_bytes_read_from_write_cache =
135 ADD_CHILD_COUNTER(profile, "ReadFromWriteCacheBytes", TUnit::BYTES, cache_profile);
136 num_io_written_in_file_cache =
137 ADD_CHILD_COUNTER(profile, "WriteInFileCacheNum", TUnit::UNIT, cache_profile);
138 num_io_bytes_written_in_file_cache =
139 ADD_CHILD_COUNTER(profile, "WriteInFileCacheBytes", TUnit::BYTES, cache_profile);
140 num_io_bytes_skip_cache =
141 ADD_CHILD_COUNTER(profile, "SkipCacheBytes", TUnit::BYTES, cache_profile);
142 }
143
144 void update(FileCacheStatistics* statistics) {
145 COUNTER_UPDATE(num_io_total, statistics->num_io_total);
146 COUNTER_UPDATE(num_io_hit_cache, statistics->num_io_hit_cache);
147 COUNTER_UPDATE(num_io_bytes_read_total, statistics->num_io_bytes_read_total);
148 COUNTER_UPDATE(num_io_bytes_read_from_file_cache,
149 statistics->num_io_bytes_read_from_file_cache);
150 COUNTER_UPDATE(num_io_bytes_read_from_write_cache,
151 statistics->num_io_bytes_read_from_write_cache);
152 COUNTER_UPDATE(num_io_written_in_file_cache, statistics->num_io_written_in_file_cache);
153 COUNTER_UPDATE(num_io_bytes_written_in_file_cache,
154 statistics->num_io_bytes_written_in_file_cache);
155 COUNTER_UPDATE(num_io_bytes_skip_cache, statistics->num_io_bytes_skip_cache);
156 }
157};
