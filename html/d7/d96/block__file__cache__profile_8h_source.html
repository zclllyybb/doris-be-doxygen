106struct FileCacheProfileReporter {
107 RuntimeProfile::Counter* num_local_io_total = nullptr;
108 RuntimeProfile::Counter* num_remote_io_total = nullptr;
109 RuntimeProfile::Counter* local_io_timer = nullptr;
110 RuntimeProfile::Counter* bytes_scanned_from_cache = nullptr;
111 RuntimeProfile::Counter* bytes_scanned_from_remote = nullptr;
112 RuntimeProfile::Counter* remote_io_timer = nullptr;
113 RuntimeProfile::Counter* write_cache_io_timer = nullptr;
114 RuntimeProfile::Counter* bytes_write_into_cache = nullptr;
115 RuntimeProfile::Counter* num_skip_cache_io_total = nullptr;
116
117 FileCacheProfileReporter(RuntimeProfile* profile) {
118 static const char* cache_profile = "FileCache";
119 ADD_TIMER(profile, cache_profile);
120 num_local_io_total =
121 ADD_CHILD_COUNTER(profile, "NumLocalIOTotal", TUnit::UNIT, cache_profile);
122 num_remote_io_total =
123 ADD_CHILD_COUNTER(profile, "NumRemoteIOTotal", TUnit::UNIT, cache_profile);
124 local_io_timer = ADD_CHILD_TIMER(profile, "LocalIOUseTimer", cache_profile);
125 remote_io_timer = ADD_CHILD_TIMER(profile, "RemoteIOUseTimer", cache_profile);
126 write_cache_io_timer = ADD_CHILD_TIMER(profile, "WriteCacheIOUseTimer", cache_profile);
127 bytes_write_into_cache =
128 ADD_CHILD_COUNTER(profile, "BytesWriteIntoCache", TUnit::BYTES, cache_profile);
129 num_skip_cache_io_total =
130 ADD_CHILD_COUNTER(profile, "NumSkipCacheIOTotal", TUnit::UNIT, cache_profile);
131 bytes_scanned_from_cache =
132 ADD_CHILD_COUNTER(profile, "BytesScannedFromCache", TUnit::BYTES, cache_profile);
133 bytes_scanned_from_remote =
134 ADD_CHILD_COUNTER(profile, "BytesScannedFromRemote", TUnit::BYTES, cache_profile);
135 }
136
137 void update(const FileCacheStatistics* statistics) {
138 COUNTER_UPDATE(num_local_io_total, statistics->num_local_io_total);
139 COUNTER_UPDATE(num_remote_io_total, statistics->num_remote_io_total);
140 COUNTER_UPDATE(local_io_timer, statistics->local_io_timer);
141 COUNTER_UPDATE(remote_io_timer, statistics->remote_io_timer);
142 COUNTER_UPDATE(write_cache_io_timer, statistics->write_cache_io_timer);
143 COUNTER_UPDATE(bytes_write_into_cache, statistics->bytes_write_into_cache);
144 COUNTER_UPDATE(num_skip_cache_io_total, statistics->num_skip_cache_io_total);
145 COUNTER_UPDATE(bytes_scanned_from_cache, statistics->bytes_read_from_local);
146 COUNTER_UPDATE(bytes_scanned_from_remote, statistics->bytes_read_from_remote);
147 }
148};
