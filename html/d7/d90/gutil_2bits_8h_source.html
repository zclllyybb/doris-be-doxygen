18 static int CountOnes(uint32 n) {
19 n -= ((n >> 1) & 0x55555555);
20 n = ((n >> 2) & 0x33333333) + (n & 0x33333333);
21 return (((n + (n >> 4)) & 0xF0F0F0F) * 0x1010101) >> 24;
22 }

24 // Count bits using sideways addition [WWG'57]. See Knuth TAOCP v4 7.1.3(59)
25 static inline int CountOnes64(uint64 n) {
26#if defined(__x86_64__)
27 n -= (n >> 1) & 0x5555555555555555ULL;
28 n = ((n >> 2) & 0x3333333333333333ULL) + (n & 0x3333333333333333ULL);
29 return (((n + (n >> 4)) & 0xF0F0F0F0F0F0F0FULL) * 0x101010101010101ULL) >> 56;
30#else
31 return CountOnes(n >> 32) + CountOnes(n & 0xffffffff);
32#endif
33 }

117// A utility class for some handy bit patterns. The names l and h
118// were chosen to match Knuth Volume 4: l is 0x010101... and h is 0x808080...;
119// half_ones is ones in the lower half only. We assume sizeof(T) is 1 or even.
120template <class T>
121struct BitPattern {
122 static const T half_ones = (static_cast<T>(1) << (sizeof(T) * 4)) - 1;
123 static const T l = (sizeof(T) == 1) ? 1 : (half_ones / 0xff * (half_ones + 2));
124 static const T h = ~(l * 0x7f);
125};

166inline uint8 Bits::ReverseBits8(unsigned char n) {
167 n = ((n >> 1) & 0x55) | ((n & 0x55) << 1);
168 n = ((n >> 2) & 0x33) | ((n & 0x33) << 2);
169 return ((n >> 4) & 0x0f) | ((n & 0x0f) << 4);
170}

172inline uint32 Bits::ReverseBits32(uint32 n) {
173 n = ((n >> 1) & 0x55555555) | ((n & 0x55555555) << 1);
174 n = ((n >> 2) & 0x33333333) | ((n & 0x33333333) << 2);
175 n = ((n >> 4) & 0x0F0F0F0F) | ((n & 0x0F0F0F0F) << 4);
176 n = ((n >> 8) & 0x00FF00FF) | ((n & 0x00FF00FF) << 8);
177 return (n >> 16) | (n << 16);
178}

180inline uint64 Bits::ReverseBits64(uint64 n) {
181#if defined(__x86_64__)
182 n = ((n >> 1) & 0x5555555555555555ULL) | ((n & 0x5555555555555555ULL) << 1);
183 n = ((n >> 2) & 0x3333333333333333ULL) | ((n & 0x3333333333333333ULL) << 2);
184 n = ((n >> 4) & 0x0F0F0F0F0F0F0F0FULL) | ((n & 0x0F0F0F0F0F0F0F0FULL) << 4);
185 n = ((n >> 8) & 0x00FF00FF00FF00FFULL) | ((n & 0x00FF00FF00FF00FFULL) << 8);
186 n = ((n >> 16) & 0x0000FFFF0000FFFFULL) | ((n & 0x0000FFFF0000FFFFULL) << 16);
187 return (n >> 32) | (n << 32);
188#else
189 return ReverseBits32(n >> 32) | (static_cast<uint64>(ReverseBits32(n & 0xffffffff)) << 32);
190#endif
191}

198// Log2Floor64() is defined in terms of Log2Floor32(), Log2FloorNonZero32()
199inline int Bits::Log2Floor64_Portable(uint64 n) {
200 const uint32 topbits = static_cast<uint32>(n >> 32);
201 if (topbits == 0) {
202 // Top bits are zero, so scan in bottom bits
203 return Log2Floor(static_cast<uint32>(n));
204 } else {
205 return 32 + Log2FloorNonZero(topbits);
206 }
207}

231template <class T>
232inline bool Bits::BytesContainByteLessThan(T bytes, uint8 c) {
233 T l = BitPattern<T>::l;
234 T h = BitPattern<T>::h;
235 // The c <= 0x80 code is straight out of Knuth Volume 4.
236 // Usually c will be manifestly constant.
237 return c <= 0x80 ? ((h & (bytes - l * c) & ~bytes) != 0)
238 : ((((bytes - l * c) | (bytes ^ h)) & h) != 0);
239}
240
241template <class T>
242inline bool Bits::BytesContainByte(T bytes, uint8 c) {
243 // Usually c will be manifestly constant.
244 return Bits::BytesContainByteLessThan<T>(bytes ^ (c * BitPattern<T>::l), 1);
245}

247template <class T>
248inline bool Bits::BytesAllInRange(T bytes, uint8 lo, uint8 hi) {
249 T l = BitPattern<T>::l;
250 T h = BitPattern<T>::h;
251 // In the common case, lo and hi are manifest constants.
252 if (lo > hi) {
253 return false;
254 }
255 if (hi - lo < 128) {
256 T x = bytes - l * lo;
257 T y = bytes + l * (127 - hi);
258 return ((x | y) & h) == 0;
259 }
260 return !Bits::BytesContainByteLessThan(bytes + (255 - hi) * l, lo + (255 - hi));
261}
