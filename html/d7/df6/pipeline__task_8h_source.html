51enum PipelineTaskState : uint8_t {
52 NOT_READY = 0, // do not prepare
53 BLOCKED_FOR_DEPENDENCY = 1,
54 BLOCKED_FOR_SOURCE = 2,
55 BLOCKED_FOR_SINK = 3,
56 RUNNABLE = 4, // can execute
57 PENDING_FINISH =
58 5, // compute task is over, but still hold resource. like some scan and sink task
59 FINISHED = 6,
60 CANCELED = 7,
61 BLOCKED_FOR_RF = 8,
62};

64inline const char* get_state_name(PipelineTaskState idx) {
65 switch (idx) {
66 case PipelineTaskState::NOT_READY:
67 return "NOT_READY";
68 case PipelineTaskState::BLOCKED_FOR_DEPENDENCY:
69 return "BLOCKED_FOR_DEPENDENCY";
70 case PipelineTaskState::BLOCKED_FOR_SOURCE:
71 return "BLOCKED_FOR_SOURCE";
72 case PipelineTaskState::BLOCKED_FOR_SINK:
73 return "BLOCKED_FOR_SINK";
74 case PipelineTaskState::RUNNABLE:
75 return "RUNNABLE";
76 case PipelineTaskState::PENDING_FINISH:
77 return "PENDING_FINISH";
78 case PipelineTaskState::FINISHED:
79 return "FINISHED";
80 case PipelineTaskState::CANCELED:
81 return "CANCELED";
82 case PipelineTaskState::BLOCKED_FOR_RF:
83 return "BLOCKED_FOR_RF";
84 }
85 __builtin_unreachable();
86}
