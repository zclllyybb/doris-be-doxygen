50enum class SourceState : uint8_t {
51 DEPEND_ON_SOURCE = 0, // Need more data from source.
52 MORE_DATA = 1, // Has more data to output. (e.g. RepeatNode)

249template <typename OperatorBuilderType>
250class DataSinkOperator : public OperatorBase {
251public:
252 using NodeType =
253 std::remove_pointer_t<decltype(std::declval<OperatorBuilderType>().exec_node())>;
254
255 DataSinkOperator(OperatorBuilderBase* builder, DataSink* sink)
256 : OperatorBase(builder), _sink(reinterpret_cast<NodeType*>(sink)) {}
257
258 ~DataSinkOperator() override = default;
259
260 Status prepare(RuntimeState* state) override {
261 RETURN_IF_ERROR(_sink->prepare(state));
262 _runtime_profile.reset(new RuntimeProfile(
263 fmt::format("{} (id={})", _operator_builder->get_name(), _operator_builder->id())));
264 _sink->profile()->insert_child_head(_runtime_profile.get(), true);
265 _mem_tracker =
266 std::make_unique<MemTracker>("DataSinkOperator:" + _runtime_profile->name(),
267 _runtime_profile.get(), nullptr, "PeakMemoryUsage");
268 return Status::OK();
269 }
270
271 Status open(RuntimeState* state) override {
272 SCOPED_TIMER(_runtime_profile->total_time_counter());
273 return _sink->open(state);
274 }
275
276 Status sink(RuntimeState* state, vectorized::Block* in_block,
277 SourceState source_state) override {
278 if (in_block->rows() > 0) {
279 return _sink->send(state, in_block, source_state == SourceState::FINISHED);
280 }
281 return Status::OK();
282 }
283
284 Status close(RuntimeState* state) override {
285 if (is_closed()) {
286 return Status::OK();
287 }
288 _fresh_exec_timer(_sink);
289 RETURN_IF_ERROR(_sink->close(state, Status::OK()));
290 _is_closed = true;
291 return Status::OK();
292 }
293
294 Status finalize(RuntimeState* state) override { return Status::OK(); }
295
296protected:
297 void _fresh_exec_timer(NodeType* node) {
298 node->profile()->total_time_counter()->update(
299 _runtime_profile->total_time_counter()->value());
300 }
301
302 NodeType* _sink;
303};

308template <typename OperatorBuilderType>
309class StreamingOperator : public OperatorBase {
310public:
311 using NodeType =
312 std::remove_pointer_t<decltype(std::declval<OperatorBuilderType>().exec_node())>;
313
314 StreamingOperator(OperatorBuilderBase* builder, ExecNode* node)
315 : OperatorBase(builder), _node(reinterpret_cast<NodeType*>(node)) {}
316
317 ~StreamingOperator() override = default;
318
319 Status prepare(RuntimeState* state) override {
320 _runtime_profile.reset(new RuntimeProfile(
321 fmt::format("{} (id={})", _operator_builder->get_name(), _operator_builder->id())));
322 _node->runtime_profile()->insert_child_head(_runtime_profile.get(), true);
323 _mem_tracker =
324 std::make_unique<MemTracker>(get_name() + ": " + _runtime_profile->name(),
325 _runtime_profile.get(), nullptr, "PeakMemoryUsage");
326 _node->increase_ref();
327 return Status::OK();
328 }
329
330 Status open(RuntimeState* state) override {
331 SCOPED_TIMER(_runtime_profile->total_time_counter());
332 RETURN_IF_ERROR(_node->alloc_resource(state));
333 return Status::OK();
334 }
335
336 Status sink(RuntimeState* state, vectorized::Block* in_block,
337 SourceState source_state) override {
338 SCOPED_TIMER(_runtime_profile->total_time_counter());
339 return _node->sink(state, in_block, source_state == SourceState::FINISHED);
340 }
341
342 Status close(RuntimeState* state) override {
343 if (is_closed()) {
344 return Status::OK();
345 }
346 _fresh_exec_timer(_node);
347 if (!_node->decrease_ref()) {
348 _node->release_resource(state);
349 }
350 _is_closed = true;
351 return Status::OK();
352 }
353
354 Status get_block(RuntimeState* state, vectorized::Block* block,
355 SourceState& source_state) override {
356 SCOPED_TIMER(_runtime_profile->total_time_counter());
357 DCHECK(_child);
358 RETURN_IF_ERROR(_child->get_block(state, block, source_state));
359 bool eos = false;
360 RETURN_IF_ERROR(_node->get_next_after_projects(
361 state, block, &eos,
362 std::bind(&ExecNode::pull, _node, std::placeholders::_1, std::placeholders::_2,
363 std::placeholders::_3)));
364 return Status::OK();
365 }
366
367 Status finalize(RuntimeState* state) override { return Status::OK(); }
368
369 bool can_read() override { return _node->can_read(); }
370
371protected:
372 void _fresh_exec_timer(NodeType* node) {
373 node->runtime_profile()->total_time_counter()->update(
374 _runtime_profile->total_time_counter()->value());
375 }
376
377 NodeType* _node;
378};

413template <typename OperatorBuilderType>
414class StatefulOperator : public StreamingOperator<OperatorBuilderType> {
415public:
416 using NodeType =
417 std::remove_pointer_t<decltype(std::declval<OperatorBuilderType>().exec_node())>;
418
419 StatefulOperator(OperatorBuilderBase* builder, ExecNode* node)
420 : StreamingOperator<OperatorBuilderType>(builder, node),
421 _child_block(new vectorized::Block),
422 _child_source_state(SourceState::DEPEND_ON_SOURCE) {}
423
424 virtual ~StatefulOperator() = default;
425
426 Status get_block(RuntimeState* state, vectorized::Block* block,
427 SourceState& source_state) override {
428 SCOPED_TIMER(this->_runtime_profile->total_time_counter());
429 auto& node = StreamingOperator<OperatorBuilderType>::_node;
430 auto& child = StreamingOperator<OperatorBuilderType>::_child;
431
432 if (node->need_more_input_data()) {
433 _child_block->clear_column_data();
434 RETURN_IF_ERROR(child->get_block(state, _child_block.get(), _child_source_state));
435 source_state = _child_source_state;
436 if (_child_block->rows() == 0 && _child_source_state != SourceState::FINISHED) {
437 return Status::OK();
438 }
439 node->prepare_for_next();
440 node->push(state, _child_block.get(), _child_source_state == SourceState::FINISHED);
441 }
442
443 if (!node->need_more_input_data()) {
444 bool eos = false;
445 RETURN_IF_ERROR(node->get_next_after_projects(
446 state, block, &eos,
447 std::bind(&ExecNode::pull, node, std::placeholders::_1, std::placeholders::_2,
448 std::placeholders::_3)));
449 if (eos) {
450 source_state = SourceState::FINISHED;
451 } else if (!node->need_more_input_data()) {
452 source_state = SourceState::MORE_DATA;
453 } else if (source_state == SourceState::MORE_DATA) {
454 source_state = _child_source_state;
455 }
456 }
457 return Status::OK();
458 }
459
460protected:
461 std::unique_ptr<vectorized::Block> _child_block;
462 SourceState _child_source_state;
463};
