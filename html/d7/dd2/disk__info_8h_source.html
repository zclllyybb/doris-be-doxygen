29// DiskInfo is an interface to query for the disk information at runtime. This
30// contains information about the system as well as the specific data node
31// configuration.
32// This information is pulled from /proc/partitions.
33// TODO: datanode information not implemented
34class DiskInfo {
35public:
36 // Initialize DiskInfo. Just be called before any other functions.
37 static void init();
38
39 // Returns the number of (logical) disks on the system
40 static int num_disks() {
41 DCHECK(_s_initialized);
42 return _s_disks.size();
43 }
44
45 // Returns the 0-based disk index for 'path' (path must be a FS path, not
46 // hdfs path).
47 static int disk_id(const char* path);
48
49 // Returns the device name (e.g. sda) for disk_id
50 static const std::string& device_name(int disk_id) {
51 DCHECK_GE(disk_id, 0);
52 DCHECK_LT(disk_id, _s_disks.size());
53 return _s_disks[disk_id].name;
54 }
55
56 static bool is_rotational(int disk_id) {
57 DCHECK_GE(disk_id, 0);
58 DCHECK_LT(disk_id, _s_disks.size());
59 return _s_disks[disk_id].is_rotational;
60 }
61
62 static std::string debug_string();
63
64 // get disk devices of given path
65 static Status get_disk_devices(const std::vector<std::string>& paths,
66 std::set<std::string>* devices);
67
68private:
69 static bool _s_initialized;
70
71 struct Disk {
72 // Name of the disk (e.g. sda)
73 std::string name;
74
75 // 0 based index. Does not map to anything in the system, useful to index into
76 // our structures
77 int id;
78
79 bool is_rotational;
80
81 Disk() : name(""), id(0) {}
82 Disk(const std::string& name) : name(name), id(0), is_rotational(true) {}
83 Disk(const std::string& name, int id) : name(name), id(id), is_rotational(true) {}
84 Disk(const std::string& name, int id, bool is_rotational)
85 : name(name), id(id), is_rotational(is_rotational) {}
86 };
87
88 // All disks
89 static std::vector<Disk> _s_disks;
90
91 // mapping of dev_ts to disk ids
92 static std::map<dev_t, int> _s_device_id_to_disk_id;
93
94 // mapping of devices names to disk ids
95 static std::map<std::string, int> _s_disk_name_to_disk_id;
96
97 static int _s_num_datanode_dirs;
98
99 static void get_device_names();
100};
101
102} // namespace doris
