53template <typename T>
54std::string get_ordinal_suffix(T n) {
55 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
56 "Unsigned integer value required");
57
58 const auto last_digit = n % 10;
59
60 if ((last_digit < 1 || last_digit > 3) || ((n > 10) && (((n / 10) % 10) == 1))) return "th";
61
62 switch (last_digit) {
63 case 1:
64 return "st";
65 case 2:
66 return "nd";
67 case 3:
68 return "rd";
69 default:
70 return "th";
71 }
72}

132template <typename F>
133std::string trim(const std::string& str, F&& predicate) {
134 size_t cut_front = 0;
135 size_t cut_back = 0;
136 size_t size = str.size();
137
138 for (size_t i = 0; i < size; ++i) {
139 if (predicate(str[i]))
140 ++cut_front;
141 else
142 break;
143 }
144
145 if (cut_front == size) return {};
146
147 for (auto it = str.rbegin(); it != str.rend(); ++it) {
148 if (predicate(*it))
149 ++cut_back;
150 else
151 break;
152 }
153
154 return str.substr(cut_front, size - cut_front - cut_back);
155}
