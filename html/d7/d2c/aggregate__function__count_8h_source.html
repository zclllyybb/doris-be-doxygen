34struct AggregateFunctionCountData {
35 UInt64 count = 0;
36};
37
39class AggregateFunctionCount final
40 : public IAggregateFunctionDataHelper<AggregateFunctionCountData, AggregateFunctionCount> {
41public:
42 AggregateFunctionCount(const DataTypes& argument_types_)
43 : IAggregateFunctionDataHelper(argument_types_, {}) {}
44
45 String get_name() const override { return "count"; }
46
47 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeInt64>(); }
48
49 void add(AggregateDataPtr __restrict place, const IColumn**, size_t, Arena*) const override {
50 ++data(place).count;
51 }
52
53 void reset(AggregateDataPtr place) const override {
54 AggregateFunctionCount::data(place).count = 0;
55 }
56
57 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
58 Arena*) const override {
59 data(place).count += data(rhs).count;
60 }
61
62 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
63 write_var_uint(data(place).count, buf);
64 }
65
66 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
67 Arena*) const override {
68 read_var_uint(data(place).count, buf);
69 }
70
71 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
72 assert_cast<ColumnInt64&>(to).get_data().push_back(data(place).count);
73 }
74
75 void deserialize_from_column(AggregateDataPtr places, const IColumn& column, Arena* arena,
76 size_t num_rows) const override {
77 auto data = assert_cast<const ColumnUInt64&>(column).get_data().data();
78 auto* dst_data = reinterpret_cast<Data*>(places);
79 for (size_t i = 0; i != num_rows; ++i) {
80 dst_data[i].count = data[i];
81 }
82 }
83
84 void serialize_to_column(const std::vector<AggregateDataPtr>& places, size_t offset,
85 MutableColumnPtr& dst, const size_t num_rows) const override {
86 auto& col = assert_cast<ColumnUInt64&>(*dst);
87 col.resize(num_rows);
88 auto* data = col.get_data().data();
89 for (size_t i = 0; i != num_rows; ++i) {
90 data[i] = this->data(places[i] + offset).count;
91 }
92 }
93
94 void streaming_agg_serialize_to_column(const IColumn** columns, MutableColumnPtr& dst,
95 const size_t num_rows, Arena* arena) const override {
96 auto& col = assert_cast<ColumnUInt64&>(*dst);
97 col.resize(num_rows);
98 col.get_data().assign(num_rows, static_cast<UInt64>(1UL));
99 }
100
101 void deserialize_and_merge_from_column(AggregateDataPtr __restrict place, const IColumn& column,
102 Arena* arena) const override {
103 auto data = assert_cast<const ColumnUInt64&>(column).get_data().data();
104 const size_t num_rows = column.size();
105 for (size_t i = 0; i != num_rows; ++i) {
106 this->data(place).count += data[i];
107 }
108 }
109
110 void serialize_without_key_to_column(ConstAggregateDataPtr __restrict place,
111 MutableColumnPtr& dst) const override {
112 auto& col = assert_cast<ColumnUInt64&>(*dst);
113 col.resize(1);
114 reinterpret_cast<Data*>(col.get_data().data())->count = this->data(place).count;
115 }
116
117 MutableColumnPtr create_serialize_column() const override {
118 return ColumnVector<UInt64>::create();
119 }
120
121 DataTypePtr get_serialized_type() const override { return std::make_shared<DataTypeUInt64>(); }
122};

125class AggregateFunctionCountNotNullUnary final
126 : public IAggregateFunctionDataHelper<AggregateFunctionCountData,
127 AggregateFunctionCountNotNullUnary> {
128public:
129 AggregateFunctionCountNotNullUnary(const DataTypes& argument_types_)
130 : IAggregateFunctionDataHelper(argument_types_, {}) {}
131
132 String get_name() const override { return "count"; }
133
134 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeInt64>(); }
135
136 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
137 Arena*) const override {
138 data(place).count += !assert_cast<const ColumnNullable&>(*columns[0]).is_null_at(row_num);
139 }
140
141 void reset(AggregateDataPtr place) const override { data(place).count = 0; }
142
143 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
144 Arena*) const override {
145 data(place).count += data(rhs).count;
146 }
147
148 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
149 write_var_uint(data(place).count, buf);
150 }
151
152 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
153 Arena*) const override {
154 read_var_uint(data(place).count, buf);
155 }
156
157 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
158 if (to.is_nullable()) {
159 auto& null_column = assert_cast<ColumnNullable&>(to);
160 null_column.get_null_map_data().push_back(0);
161 assert_cast<ColumnInt64&>(null_column.get_nested_column())
162 .get_data()
163 .push_back(data(place).count);
164 } else {
165 assert_cast<ColumnInt64&>(to).get_data().push_back(data(place).count);
166 }
167 }
168
169 void deserialize_from_column(AggregateDataPtr places, const IColumn& column, Arena* arena,
170 size_t num_rows) const override {
171 auto data = assert_cast<const ColumnUInt64&>(column).get_data().data();
172 auto* dst_data = reinterpret_cast<Data*>(places);
173 for (size_t i = 0; i != num_rows; ++i) {
174 dst_data[i].count = data[i];
175 }
176 }
177
178 void serialize_to_column(const std::vector<AggregateDataPtr>& places, size_t offset,
179 MutableColumnPtr& dst, const size_t num_rows) const override {
180 auto& col = assert_cast<ColumnUInt64&>(*dst);
181 col.resize(num_rows);
182 auto* data = col.get_data().data();
183 for (size_t i = 0; i != num_rows; ++i) {
184 data[i] = this->data(places[i] + offset).count;
185 }
186 }
187
188 void streaming_agg_serialize_to_column(const IColumn** columns, MutableColumnPtr& dst,
189 const size_t num_rows, Arena* arena) const override {
190 auto& col = assert_cast<ColumnUInt64&>(*dst);
191 col.resize(num_rows);
192 auto& data = col.get_data();
193 const ColumnNullable& input_col = assert_cast<const ColumnNullable&>(*columns[0]);
194 for (size_t i = 0; i < num_rows; i++) {
195 data[i] = !input_col.is_null_at(i);
196 }
197 }
198
199 void deserialize_and_merge_from_column(AggregateDataPtr __restrict place, const IColumn& column,
200 Arena* arena) const override {
201 auto data = assert_cast<const ColumnUInt64&>(column).get_data().data();
202 const size_t num_rows = column.size();
203 for (size_t i = 0; i != num_rows; ++i) {
204 this->data(place).count += data[i];
205 }
206 }
207
208 void serialize_without_key_to_column(ConstAggregateDataPtr __restrict place,
209 MutableColumnPtr& dst) const override {
210 auto& col = assert_cast<ColumnUInt64&>(*dst);
211 col.resize(1);
212 reinterpret_cast<Data*>(col.get_data().data())->count = this->data(place).count;
213 }
214
215 MutableColumnPtr create_serialize_column() const override {
216 return ColumnVector<UInt64>::create();
217 }
218
219 DataTypePtr get_serialized_type() const override { return std::make_shared<DataTypeUInt64>(); }
220};
