9// Hash 128 input bits down to 64 bits of output.
10// This is intended to be a reasonably good hash function.
11// It may change from time to time.
12inline uint64 Hash128to64(const uint128& x) {
13 // Murmur-inspired hashing.
14 const uint64 kMul = 0xc6a4a7935bd1e995ULL;
15 uint64 a = (Uint128Low64(x) ^ Uint128High64(x)) * kMul;
16 a ^= (a >> 47);
17 uint64 b = (Uint128High64(x) ^ a) * kMul;
18 b ^= (b >> 47);
19 b *= kMul;
20 return b;
21}
