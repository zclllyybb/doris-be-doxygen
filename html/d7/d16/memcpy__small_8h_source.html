54inline void memcpy_small_allow_read_write_overflow15_impl(char* __restrict dst,
55 const char* __restrict src, ssize_t n) {
56 while (n > 0) {
57 _mm_storeu_si128(reinterpret_cast<__m128i*>(dst),
58 _mm_loadu_si128(reinterpret_cast<const __m128i*>(src)));
59
60 dst += 16;
61 src += 16;
62 n -= 16;
63 }
64}
65} // namespace detail
