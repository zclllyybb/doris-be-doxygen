48// Return the enclosing directory of path.
49// This is like dirname(3) but for C++ strings.
50// The following list of examples shows the strings returned by dirname() and basename():
51// path dirname basename
52// "/usr/lib" "/usr" "lib"
53// "/usr/" "/" "usr"
54// "usr" "." "usr"
55// "/" "/" "/"
56// "." "." "."
57// ".." "." ".."
58std::string dir_name(const std::string& path);

62std::string base_name(const std::string& path);

64// It is used to replace std::filesystem::path::extension().
65// If the filename contains a dot but does not consist solely of one or to two dots,
66// returns the substring of file_name starting at the rightmost dot and ending at
67// the path's end. Otherwise, returns an empty string.
68// The dot is included in the return value so that it is possible to distinguish
69// between no extension and an empty extension.
70// NOTE: path can be either one file's full path or only file name
71std::string file_extension(const std::string& path);
