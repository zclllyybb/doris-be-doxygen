101struct PredicateTypeTraits {
102 static constexpr bool is_range(PredicateType type) {
103 return (type == PredicateType::LT || type == PredicateType::LE ||
104 type == PredicateType::GT || type == PredicateType::GE);
105 }
106
107 static constexpr bool is_bloom_filter(PredicateType type) { return type == PredicateType::BF; }
108
109 static constexpr bool is_list(PredicateType type) {
110 return (type == PredicateType::IN_LIST || type == PredicateType::NOT_IN_LIST);
111 }
112
113 static constexpr bool is_equal_or_list(PredicateType type) {
114 return (type == PredicateType::EQ || type == PredicateType::IN_LIST);
115 }
116
117 static constexpr bool is_comparison(PredicateType type) {
118 return (type == PredicateType::EQ || type == PredicateType::NE ||
119 type == PredicateType::LT || type == PredicateType::LE ||
120 type == PredicateType::GT || type == PredicateType::GE);
121 }
