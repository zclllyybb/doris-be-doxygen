100struct PredicateTypeTraits {
101 static constexpr bool is_range(PredicateType type) {
102 return (type == PredicateType::LT || type == PredicateType::LE ||
103 type == PredicateType::GT || type == PredicateType::GE);
104 }
105
106 static constexpr bool is_bloom_filter(PredicateType type) { return type == PredicateType::BF; }
107
108 static constexpr bool is_list(PredicateType type) {
109 return (type == PredicateType::IN_LIST || type == PredicateType::NOT_IN_LIST);
110 }
111
112 static constexpr bool is_comparison(PredicateType type) {
113 return (type == PredicateType::EQ || type == PredicateType::NE ||
114 type == PredicateType::LT || type == PredicateType::LE ||
115 type == PredicateType::GT || type == PredicateType::GE);
116 }
117};
