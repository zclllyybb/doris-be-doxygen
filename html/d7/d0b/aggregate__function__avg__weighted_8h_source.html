29template <typename T>
30struct AggregateFunctionAvgWeightedData {
31 void add(const T& data_val, double weight_val) {
32 if constexpr (IsDecimalV2<T>) {
33 DecimalV2Value value = binary_cast<Int128, DecimalV2Value>(data_val);
34 data_sum = data_sum + (static_cast<double>(value) * weight_val);
35 } else {
36 data_sum = data_sum + (data_val * weight_val);
37 }
38 weight_sum = weight_sum + weight_val;
39 }
40
41 void write(BufferWritable& buf) const {
42 write_binary(data_sum, buf);
43 write_binary(weight_sum, buf);
44 }
45
46 void read(BufferReadable& buf) {
47 read_binary(data_sum, buf);
48 read_binary(weight_sum, buf);
49 }
50
51 void merge(const AggregateFunctionAvgWeightedData& rhs) {
52 data_sum = data_sum + rhs.data_sum;
53 weight_sum = weight_sum + rhs.weight_sum;
54 }
55
56 void reset() {
57 data_sum = 0.0;
58 weight_sum = 0.0;
59 }
60
61 double get() const { return weight_sum ? data_sum / weight_sum : std::nan(""); }
62
63 double data_sum = 0.0;
64 double weight_sum = 0.0;
65};

67template <typename T>
68class AggregateFunctionAvgWeight final
69 : public IAggregateFunctionDataHelper<AggregateFunctionAvgWeightedData<T>,
70 AggregateFunctionAvgWeight<T>> {
71public:
72 using ColVecType = std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<T>, ColumnVector<T>>;
73
74 AggregateFunctionAvgWeight(const DataTypes& argument_types_)
75 : IAggregateFunctionDataHelper<AggregateFunctionAvgWeightedData<T>,
76 AggregateFunctionAvgWeight<T>>(argument_types_, {}) {}
77
78 String get_name() const override { return "avg_weighted"; }
79
80 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeFloat64>(); }
81
82 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
83 Arena*) const override {
84 const auto& column = static_cast<const ColVecType&>(*columns[0]);
85 const auto& weight = static_cast<const ColumnVector<Float64>&>(*columns[1]);
86 this->data(place).add(column.get_data()[row_num], weight.get_element(row_num));
87 }
88
89 void reset(AggregateDataPtr place) const override { this->data(place).reset(); }
90
91 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
92 Arena*) const override {
93 this->data(place).merge(this->data(rhs));
94 }
95
96 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
97 this->data(place).write(buf);
98 }
99
100 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
101 Arena*) const override {
102 this->data(place).read(buf);
103 }
104
105 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
106 auto& column = static_cast<ColumnVector<Float64>&>(to);
107 column.get_data().push_back(this->data(place).get());
108 }
109};
