33template <PrimitiveType T>
34class BloomFilterColumnPredicate : public ColumnPredicate {
35public:
36 using SpecificFilter = BloomFilterFunc<T>;
37
38 BloomFilterColumnPredicate(uint32_t column_id,
39 const std::shared_ptr<BloomFilterFuncBase>& filter,
40 int be_exec_version)
41 : ColumnPredicate(column_id),
42 _filter(filter),
43 _specific_filter(static_cast<SpecificFilter*>(_filter.get())),
44 _be_exec_version(be_exec_version) {}
45 ~BloomFilterColumnPredicate() override = default;
46
47 PredicateType type() const override { return PredicateType::BF; }
48
49 Status evaluate(BitmapIndexIterator* iterators, uint32_t num_rows,
50 roaring::Roaring* roaring) const override {
51 return Status::OK();
52 }
53
54 uint16_t evaluate(const vectorized::IColumn& column, uint16_t* sel,
55 uint16_t size) const override;
56
57private:
58 template <bool is_nullable>
59 uint16_t evaluate(const vectorized::IColumn& column, const uint8_t* null_map, uint16_t* sel,
60 uint16_t size) const {
61 if constexpr (is_nullable) {
62 DCHECK(null_map);
63 }
64
65 uint16_t new_size = 0;
66 if (column.is_column_dictionary()) {
67 auto* dict_col = reinterpret_cast<const vectorized::ColumnDictI32*>(&column);
68 for (uint16_t i = 0; i < size; i++) {
69 uint16_t idx = sel[i];
70 sel[new_size] = idx;
71 if constexpr (is_nullable) {
72 new_size += !null_map[idx] &&
73 _specific_filter->find_uint32_t(dict_col->get_hash_value(idx));
74 } else {
75 new_size += _specific_filter->find_uint32_t(dict_col->get_hash_value(idx));
76 }
77 }
78 } else if (IRuntimeFilter::enable_use_batch(_be_exec_version, T)) {
79 const auto& data =
80 reinterpret_cast<
81 const vectorized::PredicateColumnType<PredicateEvaluateType<T>>*>(
82 &column)
83 ->get_data();
84 new_size = _specific_filter->find_fixed_len_olap_engine((char*)data.data(), null_map,
85 sel, size, data.size() != size);
86 } else {
87 uint24_t tmp_uint24_value;
88 auto get_cell_value = [&tmp_uint24_value](auto& data) {
89 if constexpr (std::is_same_v<std::decay_t<decltype(data)>, uint32_t> &&
90 T == PrimitiveType::TYPE_DATE) {
91 memcpy((char*)(&tmp_uint24_value), (char*)(&data), sizeof(uint24_t));
92 return (const char*)&tmp_uint24_value;
93 } else {
94 return (const char*)&data;
95 }
96 };
97
98 auto& pred_col =
99 reinterpret_cast<
100 const vectorized::PredicateColumnType<PredicateEvaluateType<T>>*>(
101 &column)
102 ->get_data();
103
104 auto pred_col_data = pred_col.data();
105#define EVALUATE_WITH_NULL_IMPL(IDX) \
106 !null_map[IDX] && _specific_filter->find_olap_engine(get_cell_value(pred_col_data[IDX]))
107#define EVALUATE_WITHOUT_NULL_IMPL(IDX) \
108 _specific_filter->find_olap_engine(get_cell_value(pred_col_data[IDX]))
109 EVALUATE_BY_SELECTOR(EVALUATE_WITH_NULL_IMPL, EVALUATE_WITHOUT_NULL_IMPL)
110#undef EVALUATE_WITH_NULL_IMPL
111#undef EVALUATE_WITHOUT_NULL_IMPL
112 }
113 return new_size;
114 }
115
116 std::string _debug_string() const override {
117 std::string info = "BloomFilterColumnPredicate(" + type_to_string(T) + ")";
118 return info;
119 }
120
121 std::shared_ptr<BloomFilterFuncBase> _filter;
122 SpecificFilter* _specific_filter; // owned by _filter
123 mutable uint64_t _evaluated_rows = 1;
124 mutable uint64_t _passed_rows = 0;
125 mutable bool _always_true = false;
126 mutable bool _has_calculate_filter = false;
127 int _be_exec_version;
128};

130template <PrimitiveType T>
131uint16_t BloomFilterColumnPredicate<T>::evaluate(const vectorized::IColumn& column, uint16_t* sel,
132 uint16_t size) const {
133 uint16_t new_size = 0;
134 if (_always_true) {
135 return size;
136 }
137 if (column.is_nullable()) {
138 auto* nullable_col = reinterpret_cast<const vectorized::ColumnNullable*>(&column);
139 auto& null_map_data = nullable_col->get_null_map_column().get_data();
140 new_size =
141 evaluate<true>(nullable_col->get_nested_column(), null_map_data.data(), sel, size);
142 } else {
143 new_size = evaluate<false>(column, nullptr, sel, size);
144 }
145 // If the pass rate is very high, for example > 50%, then the bloomfilter is useless.
146 // Some bloomfilter is useless, for example ssb 4.3, it consumes a lot of cpu but it is
147 // useless.
148 _evaluated_rows += size;
149 _passed_rows += new_size;
150 vectorized::VRuntimeFilterWrapper::calculate_filter(
151 _evaluated_rows - _passed_rows, _evaluated_rows, _has_calculate_filter, _always_true);
152 return new_size;
153}
