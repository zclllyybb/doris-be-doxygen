43 void increment(int64_t key, uint32_t i) {
44 auto item = _counts.find(key);

57 void serialize(uint8_t* writer) const {

60 writer += sizeof(uint32_t);

69 void unserialize(const uint8_t* type_reader) {
70 uint32_t size;
71 memcpy(&size, type_reader, sizeof(uint32_t));
72 type_reader += sizeof(uint32_t);
73 for (uint32_t i = 0; i < size; ++i) {
74 int64_t key;
75 uint32_t count;
76 memcpy(&key, type_reader, sizeof(int64_t));
77 type_reader += sizeof(int64_t);
78 memcpy(&count, type_reader, sizeof(uint32_t));
79 type_reader += sizeof(uint32_t);
80 _counts.emplace(std::make_pair(key, count));
81 }

84 double get_percentile(std::vector<std::pair<int64_t, uint32_t>>& counts,
85 double position) const {
86 long lower = std::floor(position);
87 long higher = std::ceil(position);
88
89 auto iter = counts.begin();
90 for (; iter != counts.end() && iter->second < lower + 1; ++iter)
91 ;
92
93 int64_t lower_key = iter->first;
94 if (higher == lower) {
95 return lower_key;
96 }
97
98 if (iter->second < higher + 1) {
99 iter++;
100 }
101
102 int64_t higher_key = iter->first;
103 if (lower_key == higher_key) {
104 return lower_key;
105 }
106
107 return (higher - position) * lower_key + (position - lower) * higher_key;
108 }

110 doris_udf::DoubleVal terminate(double quantile) const {
111 if (_counts.empty()) {
112 return doris_udf::DoubleVal::null();
113 }
114
115 std::vector<std::pair<int64_t, uint32_t>> elems(_counts.begin(), _counts.end());
116 sort(elems.begin(), elems.end(),
117 [](const std::pair<int64_t, uint32_t> l, const std::pair<int64_t, uint32_t> r) {
118 return l.first < r.first;
119 });
120
121 long total = 0;
122 for (auto& cell : elems) {
123 total += cell.second;
124 cell.second = total;
125 }
126
127 long max_position = total - 1;
128 double position = max_position * quantile;
129 return doris_udf::DoubleVal(get_percentile(elems, position));
130 }
