67 // Compute the Crc32 hash for data using SSE4 instructions. The input hash parameter is
68 // the current hash/seed value.
69 // This should only be called if SSE is supported.
70 // This is ~4x faster than Fnv/Boost Hash.
71 // NOTE: DO NOT use this method for checksum! This does not generate the standard CRC32 checksum!
72 // For checksum, use CRC-32C algorithm from crc32c.h
73 // NOTE: Any changes made to this function need to be reflected in Codegen::GetHashFn.
74 // TODO: crc32 hashes with different seeds do not result in different hash functions.
75 // The resulting hashes are correlated.
76 static uint32_t crc_hash(const void* data, int32_t bytes, uint32_t hash) {
77 if (!CpuInfo::is_supported(CpuInfo::SSE4_2)) {
78 return zlib_crc_hash(data, bytes, hash);
79 }
80 uint32_t words = bytes / sizeof(uint32_t);
81 bytes = bytes % sizeof(uint32_t);
82
83 const uint32_t* p = reinterpret_cast<const uint32_t*>(data);
84
85 while (words--) {
86 hash = _mm_crc32_u32(hash, *p);
87 ++p;
88 }
89
90 const uint8_t* s = reinterpret_cast<const uint8_t*>(p);
91
92 while (bytes--) {
93 hash = _mm_crc32_u8(hash, *s);
94 ++s;
95 }
96
97 // The lower half of the CRC hash has has poor uniformity, so swap the halves
98 // for anyone who only uses the first several bits of the hash.
99 hash = (hash << 16) | (hash >> 16);
100 return hash;
101 }
102
103 static uint64_t crc_hash64(const void* data, int32_t bytes, uint64_t hash) {
104 uint32_t words = bytes / sizeof(uint32_t);
105 bytes = bytes % sizeof(uint32_t);
106
107 uint32_t h1 = hash >> 32;
108 uint32_t h2 = (hash << 32) >> 32;
109
110 const uint32_t* p = reinterpret_cast<const uint32_t*>(data);
111 while (words--) {
112 (words & 1) ? (h1 = _mm_crc32_u32(h1, *p)) : (h2 = _mm_crc32_u32(h2, *p));
113 ++p;
114 }
115
116 const uint8_t* s = reinterpret_cast<const uint8_t*>(p);
117 while (bytes--) {
118 (bytes & 1) ? (h1 = _mm_crc32_u8(h1, *s)) : (h2 = _mm_crc32_u8(h2, *s));
119 ++s;
120 }
121 union {
122 uint64_t u64;
123 uint32_t u32[2];
124 } converter;
125 converter.u64 = hash;
126
127 h1 = (h1 << 16) | (h1 >> 16);
128 h2 = (h2 << 16) | (h2 >> 16);
129 converter.u32[0] = h1;
130 converter.u32[1] = h2;
131
132 return converter.u64;
133 }
134#else
135 static uint32_t crc_hash(const void* data, int32_t bytes, uint32_t hash) {
136 return zlib_crc_hash(data, bytes, hash);
137 }
138#endif

152 // Murmur2 hash implementation returning 64-bit hashes.
153 static uint64_t murmur_hash2_64(const void* input, int len, uint64_t seed) {
154 uint64_t h = seed ^ (len * MURMUR_PRIME);
155
156 const uint64_t* data = reinterpret_cast<const uint64_t*>(input);
157 const uint64_t* end = data + (len / sizeof(uint64_t));
158
159 while (data != end) {
160 uint64_t k = *data++;
161 k *= MURMUR_PRIME;
162 k ^= k >> MURMUR_R;
163 k *= MURMUR_PRIME;
164 h ^= k;
165 h *= MURMUR_PRIME;
166 }
167
168 const uint8_t* data2 = reinterpret_cast<const uint8_t*>(data);
169 switch (len & 7) {
170 case 7:
171 h ^= uint64_t(data2[6]) << 48;
172 case 6:
173 h ^= uint64_t(data2[5]) << 40;
174 case 5:
175 h ^= uint64_t(data2[4]) << 32;
176 case 4:
177 h ^= uint64_t(data2[3]) << 24;
178 case 3:
179 h ^= uint64_t(data2[2]) << 16;
180 case 2:
181 h ^= uint64_t(data2[1]) << 8;
182 case 1:
183 h ^= uint64_t(data2[0]);
184 h *= MURMUR_PRIME;
185 }
186
187 h ^= h >> MURMUR_R;
188 h *= MURMUR_PRIME;
189 h ^= h >> MURMUR_R;
190 return h;
191 }

193 // default values recommended by http://isthe.com/chongo/tech/comp/fnv/
194 static const uint32_t FNV_PRIME = 0x01000193; // 16777619
195 static const uint32_t FNV_SEED = 0x811C9DC5; // 2166136261
196 static const uint64_t FNV64_PRIME = 1099511628211UL;
197 static const uint64_t FNV64_SEED = 14695981039346656037UL;
198 static const uint64_t MURMUR_PRIME = 0xc6a4a7935bd1e995ULL;
199 static const uint32_t MURMUR_SEED = 0xadc83b19ULL;
200 // Implementation of the Fowler–Noll–Vo hash function. This is not as performant
201 // as boost's hash on int types (2x slower) but has bit entropy.
202 // For ints, boost just returns the value of the int which can be pathological.
203 // For example, if the data is <1000, 2000, 3000, 4000, ..> and then the mod of 1000
204 // is taken on the hash, all values will collide to the same bucket.
205 // For string values, Fnv is slightly faster than boost.
206 static uint32_t fnv_hash(const void* data, int32_t bytes, uint32_t hash) {
207 const uint8_t* ptr = reinterpret_cast<const uint8_t*>(data);
208
209 while (bytes--) {
210 hash = (*ptr ^ hash) * FNV_PRIME;
211 ++ptr;
212 }
213
214 return hash;
215 }

217 static uint64_t fnv_hash64(const void* data, int32_t bytes, uint64_t hash) {
218 const uint8_t* ptr = reinterpret_cast<const uint8_t*>(data);
219
220 while (bytes--) {
221 hash = (*ptr ^ hash) * FNV64_PRIME;
222 ++ptr;
223 }

225 return hash;
226 }

228 // Our hash function is MurmurHash2, 64 bit version.
229 // It was modified in order to provide the same result in
230 // big and little endian archs (endian neutral).
231 static uint64_t murmur_hash64A(const void* key, int32_t len, unsigned int seed) {
232 const uint64_t m = MURMUR_PRIME;
233 const int r = 47;
234 uint64_t h = seed ^ (len * m);
235 const uint8_t* data = (const uint8_t*)key;
236 const uint8_t* end = data + (len - (len & 7));
237
238 while (data != end) {
239 uint64_t k;
240#if (BYTE_ORDER == BIG_ENDIAN)
241 k = (uint64_t)data[0];
242 k |= (uint64_t)data[1] << 8;
243 k |= (uint64_t)data[2] << 16;
244 k |= (uint64_t)data[3] << 24;
245 k |= (uint64_t)data[4] << 32;
246 k |= (uint64_t)data[5] << 40;
247 k |= (uint64_t)data[6] << 48;
248 k |= (uint64_t)data[7] << 56;
249#else
250 k = *((uint64_t*)data);
251#endif
252
253 k *= m;
254 k ^= k >> r;
255 k *= m;
256 h ^= k;
257 h *= m;
258 data += 8;
259 }
260
261 switch (len & 7) {
262 case 7:
263 h ^= (uint64_t)data[6] << 48;
264 case 6:
265 h ^= (uint64_t)data[5] << 40;
266 case 5:
267 h ^= (uint64_t)data[4] << 32;
268 case 4:
269 h ^= (uint64_t)data[3] << 24;
270 case 3:
271 h ^= (uint64_t)data[2] << 16;
272 case 2:
273 h ^= (uint64_t)data[1] << 8;
274 case 1:
275 h ^= (uint64_t)data[0];
276 h *= m;
277 }
278
279 h ^= h >> r;
280 h *= m;
281 h ^= h >> r;
282 return h;
283 }

285 // Computes the hash value for data. Will call either CrcHash or FnvHash
286 // depending on hardware capabilities.
287 // Seed values for different steps of the query execution should use different seeds
288 // to prevent accidental key collisions. (See IMPALA-219 for more details).
289 static uint32_t hash(const void* data, int32_t bytes, uint32_t seed) {
290#ifdef __SSE4_2__
291
292 if (LIKELY(CpuInfo::is_supported(CpuInfo::SSE4_2))) {
293 return crc_hash(data, bytes, seed);
294 } else {
295 return fnv_hash(data, bytes, seed);
296 }
297
298#else
299 return fnv_hash(data, bytes, seed);
300#endif
301 }
302
303 static uint64_t hash64(const void* data, int32_t bytes, uint64_t seed) {
304#ifdef _SSE4_2_
305 if (LIKELY(CpuInfo::is_supported(CpuInfo::SSE4_2))) {
306 return crc_hash64(data, bytes, seed);
307
308 } else {
309 uint64_t hash = 0;
310 murmur_hash3_x64_64(data, bytes, seed, &hash);
311 return hash;
312 }
313#else
314 uint64_t hash = 0;
315 murmur_hash3_x64_64(data, bytes, seed, &hash);
316 return hash;
317#endif
318 }
319 // hash_combine is the same with boost hash_combine,
320 // except replace boost::hash with std::hash
321 template <class T>
322 static inline void hash_combine(std::size_t& seed, const T& v) {
323 std::hash<T> hasher;
324 seed ^= hasher(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
325 }
