26// Bit-vector representing the selection status of each row in a row block.
27//
28// It is used to scan data with column predicate.
29// the bit in the selection vector is used to indicate whether the given row is live.
30class SelectionVector {
31public:
32 // Construct a new vector. The bits are initially in an indeterminate state.
33 // Call set_all_true() if you require all rows to be initially selected.
34 explicit SelectionVector(size_t row_capacity)
35 : _n_rows(row_capacity),
36 _n_bytes(BitmapSize(row_capacity)),
37 _bitmap(new uint8_t[_n_bytes]) {
38 CHECK_GT(_n_bytes, 0);
39 set_all_false();
40 }
41
42 // return the number of selected rows.
43 size_t count_selected() const { return Bits::Count(_bitmap.get(), _n_bytes); }
44
45 // Return true if any rows are selected, or false
46 // This is equivalent to (count_selected() > 0), but faster.
47 bool any_selected() const;
48
49 bool is_row_selected(size_t row) const {
50 DCHECK_LT(row, _n_rows);
51 return BitmapTest(_bitmap.get(), row);
52 }
53
54 void set_row_selected(size_t row) {
55 DCHECK_LT(row, _n_rows);
56 BitmapSet(_bitmap.get(), row);
57 }
58
59 void set_all_true() {
60 memset(_bitmap.get(), 0xff, _n_bytes);
61 pad_extra_bits_wit_zeroes();
62 }
63
64 void set_all_false() { memset(_bitmap.get(), 0, _n_bytes); }
65
66 void clear_bit(size_t row) {
67 DCHECK_LT(row, _n_rows);
68 return BitmapClear(_bitmap.get(), row);
69 }
70
71 uint8_t* mutable_bitmap() { return _bitmap.get(); }
72
73 const uint8_t* bitmap() const { return _bitmap.get(); }
74
75 size_t nrows() const { return _n_rows; }
76
77 std::string to_string() const { return BitmapToString(_bitmap.get(), _n_rows); }
78
79private:
80 // Pads any non-byte-aligned bits at the end of the SelectionVector with
81 // zeroes.
82 //
83 // To improve performance, CountSelected() and AnySelected() evaluate the
84 // SelectionVector's bitmap in terms of bytes. As such, they consider all of
85 // the trailing bits, even if the bitmap's bit length is not byte-aligned and
86 // some trailing bits aren't part of the bitmap.
87 //
88 // To address this without sacrificing performance, we need to zero out all
89 // trailing bits at construction time, or after any operation that sets all
90 // bytes in bulk.
91
92 void pad_extra_bits_wit_zeroes() {
93 size_t bits_in_last_byte = _n_rows & 7;
94 if (bits_in_last_byte > 0) {
95 BitmapChangeBits(_bitmap.get(), _n_rows, 8 - bits_in_last_byte, false);
96 }
97 }
98
99private:
100 // row capacity
101 size_t _n_rows;
102 size_t _n_bytes;
103 std::unique_ptr<uint8_t[]> _bitmap;
104 DISALLOW_COPY_AND_ASSIGN(SelectionVector);
105};
106
107inline bool SelectionVector::any_selected() const {
108 size_t rem = _n_bytes;
109 const uint32_t* p32 = reinterpret_cast<const uint32_t*>(_bitmap.get());
110 while (rem >= 4) {
111 if (*p32 != 0) {
112 return true;
113 }
114 ++p32;
115 rem -= 4;
116 }
117
118 const uint8_t* p8 = reinterpret_cast<const uint8_t*>(p32);
119 while (rem > 0) {
120 if (*p8 != 0) {
121 return true;
122 }
123 ++p8;
124 --rem;
125 }
126 return false;
127}

129// A SelectionVectorView keeps track of where in the selection vector a given
130// batch will start from. After processing a batch, Advance() should be called
131// and the view will move forward by the appropriate amount. In this way, the
132// underlying selection vector can easily be updated batch-by-batch.
133class SelectionVectorView {
134public:
135 // Constructs a new SelectionVectorView.
136 //
137 // The 'sel_vec' object must outlive this SelectionVectorView.
138 explicit SelectionVectorView(SelectionVector* sel_vec) : _sel_vec(sel_vec), _row_offset(0) {}
139 void advance(size_t skip) {
140 DCHECK_LE(skip, _sel_vec->nrows() - _row_offset);
141 _row_offset += skip;
142 }
143 void set_bit(size_t row_idx) {
144 DCHECK_LE(row_idx, _sel_vec->nrows() - _row_offset);
145 BitmapSet(_sel_vec->mutable_bitmap(), _row_offset + row_idx);
146 }
147 void clear_bit(size_t row_idx) {
148 DCHECK_LE(row_idx, _sel_vec->nrows() - _row_offset);
149 BitmapClear(_sel_vec->mutable_bitmap(), _row_offset + row_idx);
150 }
151 bool test_bit(size_t row_idx) {
152 DCHECK_LE(row_idx, _sel_vec->nrows() - _row_offset);
153 return BitmapTest(_sel_vec->bitmap(), _row_offset + row_idx);
154 }
155 void clear_bits(size_t nrows) {
156 DCHECK_LE(nrows, _sel_vec->nrows() - _row_offset);
157 BitmapChangeBits(_sel_vec->mutable_bitmap(), _row_offset, nrows, false);
158 }
159
160private:
161 SelectionVector* _sel_vec;
162 size_t _row_offset;
163};
