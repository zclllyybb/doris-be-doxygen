29// compile-time exp(v, n) by linear recursion
30template <typename T, T v, std::size_t n>
31constexpr inline const T exp = T(v) * exp<T, v, n - 1>;
32
33template <typename T, T v>
34constexpr inline const T exp<T, v, 0> = 1;
35
36// compile-time exponentiation table { exp(v, I) ... }
37template <typename T, T v, std::size_t... I>
38constexpr inline const T exp_table[] = {exp<T, v, I>...};
39
40// get value from compile-time exponentiation table by a (maybe) runtime offset
41template <typename T, T v, std::size_t... I>
42constexpr T get_exp_helper(std::size_t x, std::index_sequence<I...>) {
43 return exp_table<T, v, I...>[x];
44}
45
46// get_exp_helper with table { exp(v, 0), exp(v, 1) ... exp(v, N - 1) }
47template <typename T, T v, std::size_t N>
48constexpr T get_exp(std::size_t x) {
49 return get_exp_helper<T, v>(x, std::make_index_sequence<N> {});
50}
51
52} // namespace exp_details
