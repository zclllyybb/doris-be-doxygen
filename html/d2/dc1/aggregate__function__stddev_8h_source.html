29template <typename T, bool is_stddev>
30struct BaseData {
31 BaseData() : mean(0.0), m2(0.0), count(0) {}
32 virtual ~BaseData() = default;
33
34 void write(BufferWritable& buf) const {
35 write_binary(mean, buf);
36 write_binary(m2, buf);
37 write_binary(count, buf);
38 }
39
40 void read(BufferReadable& buf) {
41 read_binary(mean, buf);
42 read_binary(m2, buf);
43 read_binary(count, buf);
44 }
45
46 void reset() {
47 mean = 0.0;
48 m2 = 0.0;
49 count = 0;
50 }
51
52 double get_result(double res) const {
53 if constexpr (is_stddev) {
54 return std::sqrt(res);
55 } else {
56 return res;
57 }
58 }
59
60 double get_pop_result() const {
61 if (count == 1) {
62 return 0.0;
63 }
64 double res = m2 / count;
65 return get_result(res);
66 }
67
68 double get_samp_result() const {
69 double res = m2 / (count - 1);
70 return get_result(res);
71 }
72
73 void merge(const BaseData& rhs) {
74 if (rhs.count == 0) {
75 return;
76 }
77 double delta = mean - rhs.mean;
78 double sum_count = count + rhs.count;
79 mean = rhs.mean + delta * count / sum_count;
80 m2 = rhs.m2 + m2 + (delta * delta) * rhs.count * count / sum_count;
81 count = sum_count;
82 }
83
84 void add(const IColumn* column, size_t row_num) {
85 const auto& sources = static_cast<const ColumnVector<T>&>(*column);
86 double source_data = sources.get_data()[row_num];
87
88 double delta = source_data - mean;
89 double r = delta / (1 + count);
90 mean += r;
91 m2 += count * delta * r;
92 count += 1;
93 }
94
95 static DataTypePtr get_return_type() { return std::make_shared<DataTypeNumber<Float64>>(); }
96
97 double mean;
98 double m2;
99 int64_t count;
100};

102template <typename T, bool is_stddev>
103struct BaseDatadecimal {
104 BaseDatadecimal() : mean(0), m2(0), count(0) {}
105 virtual ~BaseDatadecimal() = default;
106
107 void write(BufferWritable& buf) const {
108 write_binary(mean, buf);
109 write_binary(m2, buf);
110 write_binary(count, buf);
111 }
112
113 void read(BufferReadable& buf) {
114 read_binary(mean, buf);
115 read_binary(m2, buf);
116 read_binary(count, buf);
117 }
118
119 void reset() {
120 mean = DecimalV2Value();
121 m2 = DecimalV2Value();
122 count = {};
123 }
124
125 DecimalV2Value get_result(DecimalV2Value res) const {
126 if constexpr (is_stddev) {
127 return DecimalV2Value::sqrt(res);
128 } else {
129 return res;
130 }
131 }
132
133 DecimalV2Value get_pop_result() const {
134 DecimalV2Value new_count = DecimalV2Value();
135 if (count == 1) {
136 return new_count;
137 }
138 DecimalV2Value res = m2 / new_count.assign_from_double(count);
139 return get_result(res);
140 }
141
142 DecimalV2Value get_samp_result() const {
143 DecimalV2Value new_count = DecimalV2Value();
144 DecimalV2Value res = m2 / new_count.assign_from_double(count - 1);
145 return get_result(res);
146 }
147
148 void merge(const BaseDatadecimal& rhs) {
149 if (rhs.count == 0) {
150 return;
151 }
152 DecimalV2Value new_count = DecimalV2Value();
153 new_count.assign_from_double(count);
154 DecimalV2Value rhs_count = DecimalV2Value();
155 rhs_count.assign_from_double(rhs.count);
156
157 DecimalV2Value delta = mean - rhs.mean;
158 DecimalV2Value sum_count = new_count + rhs_count;
159 mean = rhs.mean + delta * (new_count / sum_count);
160 m2 = rhs.m2 + m2 + (delta * delta) * (rhs_count * new_count / sum_count);
161 count += rhs.count;
162 }
163
164 void add(const IColumn* column, size_t row_num) {
165 const auto& sources = static_cast<const ColumnDecimal<T>&>(*column);
166 Field field = sources[row_num];
167 auto decimal_field = field.template get<DecimalField<T>>();
168 int128_t value;
169 if (decimal_field.get_scale() > DecimalV2Value::SCALE) {
170 value = static_cast<int128_t>(decimal_field.get_value()) /
171 (decimal_field.get_scale_multiplier() / DecimalV2Value::ONE_BILLION);
172 } else {
173 value = static_cast<int128_t>(decimal_field.get_value()) *
174 (DecimalV2Value::ONE_BILLION / decimal_field.get_scale_multiplier());
175 }
176 DecimalV2Value source_data = DecimalV2Value(value);
177
178 DecimalV2Value new_count = DecimalV2Value();
179 new_count.assign_from_double(count);
180 DecimalV2Value increase_count = DecimalV2Value();
181 increase_count.assign_from_double(1 + count);
182
183 DecimalV2Value delta = source_data - mean;
184 DecimalV2Value r = delta / increase_count;
185 mean += r;
186 m2 += new_count * delta * r;
187 count += 1;
188 }
189
190 static DataTypePtr get_return_type() {
191 return std::make_shared<DataTypeDecimal<Decimal128>>(27, 9);
192 }
193
194 DecimalV2Value mean;
195 DecimalV2Value m2;
196 int64_t count;
197};

233template <typename T, typename Data>
234struct SampData : Data {
235 using ColVecResult = std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<Decimal128>,
236 ColumnVector<Float64>>;
237 void insert_result_into(IColumn& to) const {
238 ColumnNullable& nullable_column = assert_cast<ColumnNullable&>(to);
239 if (this->count == 1 || this->count == 0) {
240 nullable_column.insert_default();
241 } else {
242 auto& col = static_cast<ColVecResult&>(nullable_column.get_nested_column());
243 if constexpr (IsDecimalNumber<T>) {
244 col.get_data().push_back(this->get_samp_result().value());
245 } else {
246 col.get_data().push_back(this->get_samp_result());
247 }
248 nullable_column.get_null_map_data().push_back(0);
249 }
250 }

253template <bool is_pop, typename Data, bool is_nullable>
254class AggregateFunctionSampVariance
255 : public IAggregateFunctionDataHelper<
256 Data, AggregateFunctionSampVariance<is_pop, Data, is_nullable>> {
257public:
258 AggregateFunctionSampVariance(const DataTypes& argument_types_)
259 : IAggregateFunctionDataHelper<
260 Data, AggregateFunctionSampVariance<is_pop, Data, is_nullable>>(
261 argument_types_, {}) {}
262
263 String get_name() const override { return Data::name(); }
264
265 DataTypePtr get_return_type() const override {
266 if constexpr (is_pop) {
267 return Data::get_return_type();
268 } else {
269 return make_nullable(Data::get_return_type());
270 }
271 }
272
273 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
274 Arena*) const override {
275 if constexpr (is_pop) {
276 this->data(place).add(columns[0], row_num);
277 } else {
278 if constexpr (is_nullable) {
279 const auto* nullable_column = check_and_get_column<ColumnNullable>(columns[0]);
280 if (!nullable_column->is_null_at(row_num)) {
281 this->data(place).add(&nullable_column->get_nested_column(), row_num);
282 }
283 } else {
284 this->data(place).add(columns[0], row_num);
285 }
286 }
287 }
288
289 void reset(AggregateDataPtr __restrict place) const override { this->data(place).reset(); }
290
291 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
292 Arena*) const override {
293 this->data(place).merge(this->data(rhs));
294 }
295
296 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
297 this->data(place).write(buf);
298 }
299
300 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
301 Arena*) const override {
302 this->data(place).read(buf);
303 }
304
305 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
306 this->data(place).insert_result_into(to);
307 }
308};
