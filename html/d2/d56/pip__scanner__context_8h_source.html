36 Status get_block_from_queue(RuntimeState* state, vectorized::BlockUPtr* block, bool* eos,
37 int id, bool wait = false) override {
38 {
39 std::unique_lock<std::mutex> l(_transfer_lock);
40 if (state->is_cancelled()) {
41 _process_status = Status::Cancelled("cancelled");
42 }
43
44 if (!_process_status.ok()) {
45 return _process_status;
46 }
47 }
48
49 {
50 std::unique_lock<std::mutex> l(*_queue_mutexs[id]);
51 if (!_blocks_queues[id].empty()) {
52 *block = std::move(_blocks_queues[id].front());
53 _blocks_queues[id].pop_front();
54 } else {
55 *eos = _is_finished || _should_stop;
56 return Status::OK();
57 }
58 }
59 _current_used_bytes -= (*block)->allocated_bytes();
60 return Status::OK();
61 }
62
63 // We should make those method lock free.
64 bool done() override { return _is_finished || _should_stop || _status_error; }
65
66 void append_blocks_to_queue(std::vector<vectorized::BlockUPtr>& blocks) override {
67 const int queue_size = _queue_mutexs.size();
68 const int block_size = blocks.size();
69 int64_t local_bytes = 0;
70 for (const auto& block : blocks) {
71 local_bytes += block->allocated_bytes();
72 }
73
74 for (int i = 0; i < queue_size && i < block_size; ++i) {
75 int queue = _next_queue_to_feed;
76 {
77 std::lock_guard<std::mutex> l(*_queue_mutexs[queue]);
78 for (int j = i; j < block_size; j += queue_size) {
79 _blocks_queues[queue].emplace_back(std::move(blocks[j]));
80 }
81 }
82 _next_queue_to_feed = queue + 1 < queue_size ? queue + 1 : 0;
83 }
84 _current_used_bytes += local_bytes;
85 }
86
87 bool empty_in_queue(int id) override {
88 std::unique_lock<std::mutex> l(*_queue_mutexs[id]);
89 return _blocks_queues[id].empty();
90 }
91
92 void set_max_queue_size(const int max_queue_size) override {
93 _max_queue_size = max_queue_size;
94 for (int i = 0; i < max_queue_size; ++i) {
95 _queue_mutexs.emplace_back(new std::mutex);
96 _blocks_queues.emplace_back(std::list<vectorized::BlockUPtr>());
97 }
98 }
99
100 bool has_enough_space_in_blocks_queue() const override {
101 return _current_used_bytes < _max_bytes_in_queue / 2 * _max_queue_size;
102 }
103
104private:
105 int _max_queue_size = 1;
106 int _next_queue_to_feed = 0;
107 std::vector<std::unique_ptr<std::mutex>> _queue_mutexs;
108 std::vector<std::list<vectorized::BlockUPtr>> _blocks_queues;
109 std::atomic_int64_t _current_used_bytes = 0;
110};
111} // namespace pipeline
112} // namespace doris
