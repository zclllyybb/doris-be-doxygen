29// TmpFileMgr creates and manages temporary files and directories on the local
30// filesystem. It can manage multiple temporary directories across multiple devices.
31// TmpFileMgr ensures that at most one directory per device is used unless overridden
32// for testing. GetFile() returns a File handle with a unique filename on a device. The
33// client owns the File handle and can use it to expand the file.
34// TODO: we could notify block managers about the failure so they can more take
35// proactive action to avoid using the device.
36class TmpFileMgr {
37public:
38 // DeviceId is a unique identifier for a temporary device managed by TmpFileMgr.
39 // It is used as a handle for external classes to identify devices.
40 typedef int DeviceId;
41
42 // File is a handle to a physical file in a temporary directory. Clients
43 // can allocate file space and remove files using AllocateSpace() and Remove().
44 // Creation of the file is deferred until the first call to AllocateSpace().
45 class File {
46 public:
47 ~File() {
48 // do nothing
49 }
50
51 // Allocates 'write_size' bytes in this file for a new block of data.
52 // The file size is increased by a call to truncate() if necessary.
53 // The physical file is created on the first call to AllocateSpace().
54 // Returns Status::OK()() and sets offset on success.
55 // Returns an error status if an unexpected error occurs.
56 // If an error status is returned, the caller can try a different temporary file.
57 Status allocate_space(int64_t write_size, int64_t* offset);
58
59 // Called to notify TmpFileMgr that an IO error was encountered for this file
60 void report_io_error(const std::string& error_msg);
61
62 // Delete the physical file on disk, if one was created.
63 // It is not valid to read or write to a file after calling Remove().
64 Status remove();
65
66 const std::string& path() const { return _path; }
67 int disk_id() const { return _disk_id; }
68 bool is_blacklisted() const { return _blacklisted; }
69
70 private:
71 friend class TmpFileMgr;
72
73 // The name of the sub-directory that Impala created within each configured scratch
74 // directory.
75 const static std::string _s_tmp_sub_dir_name;
76
77 // Space (in MB) that must ideally be available for writing on a scratch
78 // directory. A warning is issued if available space is less than this threshold.
79 const static uint64_t _s_available_space_threshold_mb;
80
81 File(TmpFileMgr* mgr, DeviceId device_id, const std::string& path);
82
83 // TmpFileMgr this belongs to.
84 TmpFileMgr* _mgr;
85
86 // Path of the physical file in the filesystem.
87 std::string _path;
88
89 // The temporary device this file is stored on.
90 DeviceId _device_id;
91
92 // The id of the disk on which the physical file lies.
93 int _disk_id;
94
95 // Current file size. Modified by AllocateSpace(). Size is 0 before file creation.
96 int64_t _current_size;
97
98 // Set to true to indicate that file can't be expanded. This is useful to keep here
99 // even though it is redundant with the global per-device blacklisting in TmpFileMgr
100 // because it can be checked without acquiring a global lock. If a file is
101 // blacklisted, the corresponding device will always be blacklisted.
102 bool _blacklisted;
103 };
104
105 TmpFileMgr(ExecEnv* exec_env);
106 TmpFileMgr();
107
108 ~TmpFileMgr();
109
110 // Creates the configured tmp directories. If multiple directories are specified per
111 // disk, only one is created and used. Must be called after DiskInfo::Init().
112 Status init();
113
114 // Custom initialization - initializes with the provided list of directories.
115 // If one_dir_per_device is true, only use one temporary directory per device.
116 // This interface is intended for testing purposes.
117 Status init_custom(const std::vector<std::string>& tmp_dirs, bool one_dir_per_device);
118
119 // Return a new File handle with a unique path for a query instance. The file path
120 // is within the (single) tmp directory on the specified device id. The caller owns
121 // the returned handle and is responsible for deleting it. The file is not created -
122 // creation is deferred until the first call to File::AllocateSpace().
123 Status get_file(const DeviceId& device_id, const TUniqueId& query_id, File** new_file);
124
125 // Return the scratch directory path for the device.
126 std::string get_tmp_dir_path(DeviceId device_id) const;
127
128 // Return a random scratch directory path from the devices.
129 std::string get_tmp_dir_path();
130
131 // Total number of devices with tmp directories that are active. There is one tmp
132 // directory per device.
133 int num_active_tmp_devices();
134
135 // Return vector with device ids of all tmp devices being actively used.
136 // I.e. those that haven't been blacklisted.
137 std::vector<DeviceId> active_tmp_devices();
138
139private:
140 // Dir stores information about a temporary directory.
141 class Dir {
142 public:
143 const std::string& path() const { return _path; }
144
145 // Return true if it was newly added to blacklist.
146 bool blacklist() {
147 bool was_blacklisted = _blacklisted;
148 _blacklisted = true;
149 return !was_blacklisted;
150 }
151 bool is_blacklisted() const { return _blacklisted; }
152
153 private:
154 friend class TmpFileMgr;
155
156 // path should be a absolute path to a writable scratch directory.
157 Dir(const std::string& path, bool blacklisted) : _path(path), _blacklisted(blacklisted) {}
158
159 std::string _path;
160
161 bool _blacklisted;
162 };
163
164 // Remove a device from the rotation. Subsequent attempts to allocate a file on that
165 // device will fail and the device will not be included in active tmp devices.
166 void blacklist_device(DeviceId device_id);
167
168 bool is_blacklisted(DeviceId device_id);
169
170 ExecEnv* _exec_env;
171 bool _initialized = false;
172
173 // Protects the status of tmp dirs (i.e. whether they're blacklisted).
174 SpinLock _dir_status_lock;
175
176 // The created tmp directories.
177 std::vector<Dir> _tmp_dirs;
178
179 // Metric to track active scratch directories.
180 IntGauge* active_scratch_dirs;
181};
