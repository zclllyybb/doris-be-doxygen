27class ParquetColumnReader {
28public:
29 struct Statistics {
30 Statistics()
31 : read_time(0),
32 read_calls(0),
33 read_bytes(0),
34 decompress_time(0),
35 decompress_cnt(0),
36 decode_header_time(0),
37 decode_value_time(0),
38 decode_dict_time(0),
39 decode_level_time(0),
40 decode_null_map_time(0) {}
41
42 Statistics(io::BufferedStreamReader::Statistics& fs, ColumnChunkReader::Statistics& cs,
43 int64_t null_map_time)
44 : read_time(fs.read_time),
45 read_calls(fs.read_calls),
46 read_bytes(fs.read_bytes),
47 decompress_time(cs.decompress_time),
48 decompress_cnt(cs.decompress_cnt),
49 decode_header_time(cs.decode_header_time),
50 decode_value_time(cs.decode_value_time),
51 decode_dict_time(cs.decode_dict_time),
52 decode_level_time(cs.decode_level_time),
53 decode_null_map_time(null_map_time) {}
54
55 int64_t read_time;
56 int64_t read_calls;
57 int64_t read_bytes;
58 int64_t decompress_time;
59 int64_t decompress_cnt;
60 int64_t decode_header_time;
61 int64_t decode_value_time;
62 int64_t decode_dict_time;
63 int64_t decode_level_time;
64 int64_t decode_null_map_time;
65
66 void merge(Statistics& statistics) {
67 read_time += statistics.read_time;
68 read_calls += statistics.read_calls;
69 read_bytes += statistics.read_bytes;
70 decompress_time += statistics.decompress_time;
71 decompress_cnt += statistics.decompress_cnt;
72 decode_header_time += statistics.decode_header_time;
73 decode_value_time += statistics.decode_value_time;
74 decode_dict_time += statistics.decode_dict_time;
75 decode_level_time += statistics.decode_level_time;
76 decode_null_map_time += statistics.decode_null_map_time;
77 }
78 };

80 ParquetColumnReader(const std::vector<RowRange>& row_ranges, cctz::time_zone* ctz,
81 io::IOContext* io_ctx)
82 : _row_ranges(row_ranges), _ctz(ctz), _io_ctx(io_ctx) {}
83 virtual ~ParquetColumnReader() = default;
84 virtual Status read_column_data(ColumnPtr& doris_column, DataTypePtr& type,
85 ColumnSelectVector& select_vector, size_t batch_size,
86 size_t* read_rows, bool* eof, bool is_dict_filter) = 0;
87
88 virtual Status read_dict_values_to_column(MutableColumnPtr& doris_column, bool* has_dict) {
89 return Status::NotSupported("read_dict_values_to_column is not supported");
90 }
91
92 virtual Status get_dict_codes(const ColumnString* column_string,
93 std::vector<int32_t>* dict_codes) {
94 return Status::NotSupported("get_dict_codes is not supported");
95 }
96
97 virtual MutableColumnPtr convert_dict_column_to_string_column(const ColumnInt32* dict_column) {
98 LOG(FATAL) << "Method convert_dict_column_to_string_column is not supported";
99 }
100
101 static Status create(io::FileReaderSPtr file, FieldSchema* field,
102 const tparquet::RowGroup& row_group,
103 const std::vector<RowRange>& row_ranges, cctz::time_zone* ctz,
104 io::IOContext* io_ctx, std::unique_ptr<ParquetColumnReader>& reader,
105 size_t max_buf_size);
106 void add_offset_index(tparquet::OffsetIndex* offset_index) { _offset_index = offset_index; }
107 void set_nested_column() { _nested_column = true; }
108 virtual const std::vector<level_t>& get_rep_level() const = 0;
109 virtual const std::vector<level_t>& get_def_level() const = 0;
110 virtual Statistics statistics() = 0;
111 virtual void close() = 0;
112
113protected:
114 void _generate_read_ranges(int64_t start_index, int64_t end_index,
115 std::list<RowRange>& read_ranges);
116
117 FieldSchema* _field_schema;
118 // When scalar column is the child of nested column, we should turn off the filtering by page index and lazy read.
119 bool _nested_column = false;
120 const std::vector<RowRange>& _row_ranges;
121 cctz::time_zone* _ctz;
122 io::IOContext* _io_ctx;
123 tparquet::OffsetIndex* _offset_index;
124 int64_t _current_row_index = 0;
125 int _row_range_index = 0;
126 int64_t _decode_null_map_time = 0;
127};

225class StructColumnReader : public ParquetColumnReader {
226public:
227 StructColumnReader(const std::vector<RowRange>& row_ranges, cctz::time_zone* ctz,
228 io::IOContext* io_ctx)
229 : ParquetColumnReader(row_ranges, ctz, io_ctx) {}
230 ~StructColumnReader() override { close(); }
231
232 Status init(std::vector<std::unique_ptr<ParquetColumnReader>>&& child_readers,
233 FieldSchema* field);
234 Status read_column_data(ColumnPtr& doris_column, DataTypePtr& type,
235 ColumnSelectVector& select_vector, size_t batch_size, size_t* read_rows,
236 bool* eof, bool is_dict_filter) override;
237
238 const std::vector<level_t>& get_rep_level() const override {
239 return _child_readers[0]->get_rep_level();
240 }
241 const std::vector<level_t>& get_def_level() const override {
242 return _child_readers[0]->get_def_level();
243 }
244
245 Statistics statistics() override {
246 Statistics st;
247 for (const auto& reader : _child_readers) {
248 Statistics cst = reader->statistics();
249 st.merge(cst);
250 }
251 return st;
252 }
253
254 void close() override {}
255
256private:
257 std::vector<std::unique_ptr<ParquetColumnReader>> _child_readers;
258};
