99 ParquetColumnReader(const std::vector<RowRange>& row_ranges, cctz::time_zone* ctz)
100 : _row_ranges(row_ranges), _ctz(ctz) {}
101 virtual ~ParquetColumnReader() {
102 if (_stream_reader != nullptr) {
103 delete _stream_reader;
104 _stream_reader = nullptr;
105 }
106 }
107 virtual Status read_column_data(ColumnPtr& doris_column, DataTypePtr& type,
108 ColumnSelectVector& select_vector, size_t batch_size,
109 size_t* read_rows, bool* eof) = 0;
110 static Status create(io::FileReaderSPtr file, FieldSchema* field,
111 const tparquet::RowGroup& row_group,
112 const std::vector<RowRange>& row_ranges, cctz::time_zone* ctz,
113 std::unique_ptr<ParquetColumnReader>& reader, size_t max_buf_size);
114 void init_column_metadata(const tparquet::ColumnChunk& chunk);
115 void add_offset_index(tparquet::OffsetIndex* offset_index) { _offset_index = offset_index; }
116 Statistics statistics() {
117 return Statistics(_stream_reader->statistics(), _chunk_reader->statistics(),
118 _decode_null_map_time);
119 }
120 virtual void close() = 0;
121
122protected:
123 void _generate_read_ranges(int64_t start_index, int64_t end_index,
124 std::list<RowRange>& read_ranges);
125
126 BufferedFileStreamReader* _stream_reader;
127 std::unique_ptr<ParquetColumnMetadata> _metadata;
128 const std::vector<RowRange>& _row_ranges;
129 cctz::time_zone* _ctz;
130 std::unique_ptr<ColumnChunkReader> _chunk_reader;
131 tparquet::OffsetIndex* _offset_index;
132 int64_t _current_row_index = 0;
133 int _row_range_index = 0;
134 int64_t _decode_null_map_time = 0;
