56// Location information for null indicator bit for particular slot.
57// For non-nullable slots, the byte_offset will be 0 and the bit_mask will be 0.
58// This allows us to do the NullIndicatorOffset operations (tuple + byte_offset &/|
59// bit_mask) regardless of whether the slot is nullable or not.
60// This is more efficient than branching to check if the slot is non-nullable.
61struct NullIndicatorOffset {
62 int byte_offset;
63 uint8_t bit_mask; // to extract null indicator
64 int8_t bit_offset; // only used to serialize, from 1 to 8, invalid null value
65 // bit_offset is -1.
66
67 NullIndicatorOffset(int byte_offset, int bit_offset_)
68 : byte_offset(byte_offset),
69 bit_mask(bit_offset_ == -1 ? 0 : 1 << (7 - bit_offset_)),
70 bit_offset(bit_offset_) {
71 DCHECK_LE(bit_offset_, 8);
72 }
73
74 bool equals(const NullIndicatorOffset& o) const {
75 return this->byte_offset == o.byte_offset && this->bit_mask == o.bit_mask;
76 }
77
78 std::string debug_string() const;
79};
