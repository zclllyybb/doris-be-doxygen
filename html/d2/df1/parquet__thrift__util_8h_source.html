34constexpr uint8_t PARQUET_VERSION_NUMBER[4] = {'P', 'A', 'R', '1'};
35constexpr uint32_t PARQUET_FOOTER_SIZE = 8;
36
37static Status parse_thrift_footer(io::FileReaderSPtr file,
38 std::shared_ptr<FileMetaData>& file_metadata) {
39 uint8_t footer[PARQUET_FOOTER_SIZE];
40 int64_t file_size = file->size();
41 size_t bytes_read = 0;
42 Slice result(footer, PARQUET_FOOTER_SIZE);
43 IOContext io_ctx;
44 RETURN_IF_ERROR(file->read_at(file_size - PARQUET_FOOTER_SIZE, result, io_ctx, &bytes_read));
45 DCHECK_EQ(bytes_read, PARQUET_FOOTER_SIZE);
46
47 // validate magic
48 uint8_t* magic_ptr = footer + PARQUET_FOOTER_SIZE - sizeof(PARQUET_VERSION_NUMBER);
49 if (memcmp(magic_ptr, PARQUET_VERSION_NUMBER, sizeof(PARQUET_VERSION_NUMBER)) != 0) {
50 return Status::Corruption("Invalid magic number in parquet file");
51 }
52
53 // get metadata_size
54 uint32_t metadata_size = decode_fixed32_le(footer);
55 if (metadata_size > file_size - PARQUET_FOOTER_SIZE) {
56 Status::Corruption("Parquet file size is ", file_size,
57 " bytes, smaller than the size reported by footer's (", metadata_size,
58 "bytes)");
59 }
60 tparquet::FileMetaData t_metadata;
61 // deserialize footer
62 uint8_t meta_buff[metadata_size];
63 Slice res(meta_buff, metadata_size);
64 RETURN_IF_ERROR(file->read_at(file_size - PARQUET_FOOTER_SIZE - metadata_size, res, io_ctx,
65 &bytes_read));
66 DCHECK_EQ(bytes_read, metadata_size);
67 RETURN_IF_ERROR(deserialize_thrift_msg(meta_buff, &metadata_size, true, &t_metadata));
68 file_metadata.reset(new FileMetaData(t_metadata));
69 RETURN_IF_ERROR(file_metadata->init_schema());
70 return Status::OK();
71}
72} // namespace doris::vectorized
