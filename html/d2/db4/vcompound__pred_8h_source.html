31inline std::string compound_operator_to_string(TExprOpcode::type op) {
32 if (op == TExprOpcode::COMPOUND_AND) {
33 return "and";
34 } else if (op == TExprOpcode::COMPOUND_OR) {
35 return "or";
36 } else {
37 return "not";
38 }
39}

41class VcompoundPred : public VectorizedFnCall {
42public:
43 VcompoundPred(const TExprNode& node) : VectorizedFnCall(node) {
44 _op = node.opcode;
45 _fn.name.function_name = compound_operator_to_string(_op);
46 _expr_name = "VCompoundPredicate (" + _fn.name.function_name + ")";
47 }

49 VExpr* clone(ObjectPool* pool) const override { return pool->add(new VcompoundPred(*this)); }
50
51 const std::string& expr_name() const override { return _expr_name; }

53 Status execute(VExprContext* context, doris::vectorized::Block* block,
54 int* result_column_id) override {
55 if (children().size() == 1 || !_all_child_is_compound_and_not_const()) {
56 return VectorizedFnCall::execute(context, block, result_column_id);
57 }
58
59 int lhs_id = -1;
60 int rhs_id = -1;
61 RETURN_IF_ERROR(_children[0]->execute(context, block, &lhs_id));
62 ColumnPtr lhs_column = block->get_by_position(lhs_id).column;
63
64 ColumnPtr rhs_column = nullptr;
65
66 size_t size = lhs_column->size();
67 uint8* __restrict data = _get_raw_data(lhs_column);
68 int filted = simd::count_zero_num((int8_t*)data, size);
69 bool full = filted == 0;
70 bool empty = filted == size;
71
72 const uint8* __restrict data_rhs = nullptr;
73 bool full_rhs = false;
74 bool empty_rhs = false;
75
76 auto get_rhs_colum = [&]() {
77 if (rhs_id == -1) {
78 RETURN_IF_ERROR(_children[1]->execute(context, block, &rhs_id));
79 rhs_column = block->get_by_position(rhs_id).column;
80 data_rhs = _get_raw_data(rhs_column);
81 if (!empty) {
82 if (const uint8* null_map =
83 _get_null_map(block->get_by_position(rhs_id).column);
84 null_map != nullptr) {
85 for (size_t i = 0; i < size; i++) {
86 data[i] &= !null_map[i];
87 }
88 }
89 }
90 int filted = simd::count_zero_num((int8_t*)data_rhs, size);
91 full_rhs = filted == 0;
92 empty_rhs = filted == size;
93 }
94 return Status::OK();
95 };
96
97 if (_op == TExprOpcode::COMPOUND_AND) {
98 if (!empty) { // empty and any = empty, so lhs should not empty
99 RETURN_IF_ERROR(get_rhs_colum());
100 if (empty_rhs) { // any and empty = empty
101 *result_column_id = rhs_id;
102 return Status::OK();
103 } else if (!full_rhs) { // any and full = any, so rhs should not full.
104 for (size_t i = 0; i < size; i++) {
105 data[i] &= data_rhs[i];
106 }
107 }
108 }
109 } else if (_op == TExprOpcode::COMPOUND_OR) {
110 if (!full) { // full or any = full, so lhs should not full
111 RETURN_IF_ERROR(get_rhs_colum());
112 if (full_rhs) { // any or full = full
113 *result_column_id = rhs_id;
114 return Status::OK();
115 } else if (!empty_rhs) { // any or empty = any, so rhs should not empty
116 for (size_t i = 0; i < size; i++) {
117 data[i] |= data_rhs[i];
118 }
119 }
120 }
121 } else {
122 LOG(FATAL) << "Compound operator must be AND or OR.";
123 }
124
125 *result_column_id = lhs_id;
126 return Status::OK();
127 }

142private:
143 bool _all_child_is_compound_and_not_const() const {
144 for (auto child : _children) {
145 // we can make sure non const compound predicate's return column is allow modifyied locally.
146 if (child->is_constant() || !child->is_compound_predicate()) {
147 return false;
148 }
149 }
150 return false;
151 }

153 uint8* _get_raw_data(ColumnPtr column) const {
154 if (column->is_nullable()) {
155 return reinterpret_cast<ColumnUInt8*>(
156 assert_cast<ColumnNullable*>(column->assume_mutable().get())
157 ->get_nested_column_ptr()
158 .get())
159 ->get_data()
160 .data();
161 } else {
162 return reinterpret_cast<ColumnUInt8*>(column->assume_mutable().get())
163 ->get_data()
164 .data();
165 }
166 }

169 if (column->is_nullable()) {
170 return reinterpret_cast<ColumnUInt8*>(
171 reinterpret_cast<ColumnNullable*>(column->assume_mutable().get())
172 ->get_null_map_column_ptr()
173 .get())
174 ->get_data()
175 .data();
176 } else {
177 return nullptr;
178 }
179 }
