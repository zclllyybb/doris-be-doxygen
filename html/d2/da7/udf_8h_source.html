166 // Allocates memory for UDAs. All UDA allocations should use this if possible instead of

180 // Reallocates 'ptr' to the new byte_size. If the currently underlying allocation
181 // is big enough, the original ptr will be returned. If the allocation needs to

184 // This should be used for buffers that constantly get appended to.

186

390// TODO: need to set explicit align?
391// Note: there is a difference between a nullptr string (is_null == true) and an
392// empty string (len == 0).
393struct StringVal : public AnyVal {
394 static const int MAX_LENGTH = (1 << 30);
395
396 int64_t len;
397 uint8_t* ptr;
398
399 // Construct a StringVal from ptr/len. Note: this does not make a copy of ptr
400 // so the buffer must exist as long as this StringVal does.
401 StringVal() : len(0), ptr(nullptr) {}
402
403 // Construct a StringVal from ptr/len. Note: this does not make a copy of ptr
404 // so the buffer must exist as long as this StringVal does.
405 StringVal(uint8_t* ptr, int64_t len) : len(len), ptr(ptr) {}
406
407 // Construct a StringVal from nullptr-terminated c-string. Note: this does not make a
408 // copy of ptr so the underlying string must exist as long as this StringVal does.
409 StringVal(const char* ptr) : len(strlen(ptr)), ptr((uint8_t*)ptr) {}
410
411 StringVal(const char* ptr, int64_t len) : len(len), ptr((uint8_t*)ptr) {}
412
413 static StringVal null() {
414 StringVal sv;
415 sv.is_null = true;
416 return sv;
417 }
418
419 // Creates a StringVal, allocating a new buffer with 'len'. This should
420 // be used to return StringVal objects in UDF/UDAs that need to allocate new
421 // string memory.
422 StringVal(FunctionContext* context, int64_t len);
423
424 // Creates a StringVal, which memory is available when this function context is used next time
425 static StringVal create_temp_string_val(FunctionContext* ctx, int64_t len);
426
427 bool resize(FunctionContext* context, int64_t len);
428
429 bool operator==(const StringVal& other) const {
430 if (is_null != other.is_null) {
431 return false;
432 }
433
434 if (is_null) {
435 return true;
436 }
437
438 if (len != other.len) {
439 return false;
440 }
441
442 return len == 0 || ptr == other.ptr || memcmp(ptr, other.ptr, len) == 0;
443 }
444
445 bool operator!=(const StringVal& other) const { return !(*this == other); }
446
450 static StringVal copy_from(FunctionContext* ctx, const uint8_t* buf, int64_t len);
451
455 void append(FunctionContext* ctx, const uint8_t* buf, int64_t len);
456
457 void append(FunctionContext* ctx, const uint8_t* buf, int64_t len, const uint8_t* buf2,
458 int64_t buf2_len);
459
460 std::string to_string() const { return std::string((char*)ptr, len); }
461};
