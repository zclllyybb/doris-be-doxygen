68template <typename Column>
69ColumnPtr select_index_impl(const Column& column, const IColumn& indexes, size_t limit) {
70 if (limit == 0) {
71 limit = indexes.size();
72 }
73
74 if (indexes.size() < limit) {
75 LOG(FATAL) << "Size of indexes is less than required.";
76 }
77
78 if (auto* data_uint8 = detail::get_indexes_data<UInt8>(indexes)) {
79 return column.template index_impl<UInt8>(*data_uint8, limit);
80 } else if (auto* data_uint16 = detail::get_indexes_data<UInt16>(indexes)) {
81 return column.template index_impl<UInt16>(*data_uint16, limit);
82 } else if (auto* data_uint32 = detail::get_indexes_data<UInt32>(indexes)) {
83 return column.template index_impl<UInt32>(*data_uint32, limit);
84 } else if (auto* data_uint64 = detail::get_indexes_data<UInt64>(indexes)) {
85 return column.template index_impl<UInt64>(*data_uint64, limit);
86 } else {
87 LOG(FATAL) << "Indexes column for IColumn::select must be ColumnUInt, got"
88 << indexes.get_name();
89 return nullptr;
90 }
91}
