30namespace doris {
31// Utility class for all Thrift servers. Runs a TNonblockingServer(default) or a
32// TThreadPoolServer with, by default, 2 worker threads, that exposes the interface
33// described by a user-supplied TProcessor object.
34// If TNonblockingServer is used, client must use TFramedTransport.
35// If TThreadPoolServer is used, client must use TSocket as transport.
36class ThriftServer {
37public:
38 // An opaque identifier for the current session, which identifies a client connection.
39 typedef std::string SessionKey;
40
41 // Interface class for receiving session creation / termination events.
42 class SessionHandlerIf {
43 public:
44 // Called when a session is established (when a client connects).
45 virtual void session_start(const SessionKey& session_key) = 0;
46
47 // Called when a session is terminated (when a client closes the connection).
48 // After this callback returns, the memory session_key references is no longer valid
49 // and clients must not refer to it again.
50 virtual void session_end(const SessionKey& session_key) = 0;
51 };
52
53 static const int DEFAULT_WORKER_THREADS = 2;
54
55 // There are 3 servers supported by Thrift with different threading models.
56 // THREAD_POOL -- Allocates a fixed number of threads. A thread is used by a
57 // connection until it closes.
58 // THREADED -- Allocates 1 thread per connection, as needed.
59 // NON_BLOCKING -- Threads are allocated to a connection only when the server
60 // is working on behalf of the connection.
61 enum ServerType { THREAD_POOL = 0, THREADED, NON_BLOCKING };
62
63 // Creates, but does not start, a new server on the specified port
64 // that exports the supplied interface.
65 // - name: human-readable name of this server. Should not contain spaces
66 // - processor: Thrift processor to handle RPCs
67 // - port: The port the server will listen for connections on
68 // - num_worker_threads: the number of worker threads to use in any thread pool
69 // - server_type: the type of IO strategy this server should employ
70 ThriftServer(const std::string& name,
71 const std::shared_ptr<apache::thrift::TProcessor>& processor, int port,
72 int num_worker_threads = DEFAULT_WORKER_THREADS,
73 ServerType server_type = THREADED);
74
75 ~ThriftServer() {}
76
77 int port() const { return _port; }
78
79 void stop();
80 // Blocks until the server stops and exits its main thread.
81 void join();
82
83 // FOR TESTING ONLY; stop the server and block until the server is stopped; use it
84 // only if it is a Threaded server.
85 void stop_for_testing();
86
87 // Starts the main server thread. Once this call returns, clients
88 // may connect to this server and issue RPCs. May not be called more
89 // than once.
90 Status start();
91
92 // Sets the session handler which receives events when sessions are created or closed.
93 void set_session_handler(SessionHandlerIf* session) { _session_handler = session; }
94
95 // Returns a unique identifier for the current session. A session is
96 // identified with the lifetime of a socket connection to this server.
97 // It is only safe to call this method during a Thrift processor RPC
98 // implementation. Otherwise, the result of calling this method is undefined.
99 // It is also only safe to reference the returned value during an RPC method.
100 static SessionKey* get_thread_session_key();
101
102private:
103 // True if the server has been successfully started, for internal use only
104 bool _started;
105
106 // The port on which the server interface is exposed
107 int _port;
108
109 // How many worker threads to use to serve incoming requests
110 // (requests are queued if no thread is immediately available)
111 int _num_worker_threads;
112
113 // ThreadPool or NonBlocking server
114 ServerType _server_type;
115
116 // User-specified identifier that shows up in logs
117 const std::string _name;
118
119 // Thread that runs the TNonblockingServer::serve loop
120 std::unique_ptr<std::thread> _server_thread;
121
122 // Thrift housekeeping
123 std::unique_ptr<apache::thrift::server::TServer> _server;
124 std::shared_ptr<apache::thrift::TProcessor> _processor;
125
126 // If not nullptr, called when session events happen. Not owned by us.
127 SessionHandlerIf* _session_handler;
128
129 // Protects _session_keys
130 std::mutex _session_keys_lock;
131
132 // Map of active session keys to shared_ptr containing that key; when a key is
133 // removed it is automatically freed.
134 typedef std::unordered_map<SessionKey*, std::shared_ptr<SessionKey>> SessionKeySet;
135 SessionKeySet _session_keys;
136
137 // Helper class which monitors starting servers. Needs access to internal members, and
138 // is not used outside of this class.
139 class ThriftServerEventProcessor;
140 friend class ThriftServerEventProcessor;
141
142 std::shared_ptr<MetricEntity> _thrift_server_metric_entity;
143 // Number of currently active connections
144 IntGauge* thrift_current_connections;
145 // Total connections made over the lifetime of this server
146 IntCounter* thrift_connections_total;
147};
148
149} // namespace doris
