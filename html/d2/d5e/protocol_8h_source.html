29inline StringRef get_url_scheme(const char* data, size_t size) {
30 // scheme = ALPHA *( ALPHA / DIGIT / "+" / "-" / "." )
31 const char* pos = data;
32 const char* end = data + size;
33
34 if (is_alpha_ascii(*pos)) {
35 for (++pos; pos < end; ++pos) {
36 if (!(is_alpha_numeric_ascii(*pos) || *pos == '+' || *pos == '-' || *pos == '.')) {
37 break;
38 }
39 }
40
41 return StringRef(data, pos - data);
42 }
43
44 return {};
45}

47struct ExtractProtocol {
48 static size_t get_reserve_length_for_element() { return strlen("https") + 1; }

50 static void execute(Pos data, size_t size, Pos& res_data, size_t& res_size) {
51 res_data = data;
52 res_size = 0;
53
54 StringRef scheme = get_url_scheme(data, size);
55 Pos pos = data + scheme.size;
56
57 if (scheme.size == 0 || (data + size) - pos < 4) return;
58
59 if (pos[0] == ':') res_size = pos - data;
60 }
61};
