42struct UInt128 {
44 UInt64 low;
45 UInt64 high;
46
47 UInt128() = default;
48 explicit UInt128(const UInt64 low_, const UInt64 high_) : low(low_), high(high_) {}
49 explicit UInt128(const UInt64 rhs) : low(rhs), high() {}
50
51 auto tuple() const { return std::tie(high, low); }
52
53 String to_hex_string() const {
54 std::ostringstream os;
55 os << std::setw(16) << std::setfill('0') << std::hex << high << low;
56 return String(os.str());
57 }

59 bool operator==(const UInt128 rhs) const { return tuple() == rhs.tuple(); }
60 bool operator!=(const UInt128 rhs) const { return tuple() != rhs.tuple(); }
61 bool operator<(const UInt128 rhs) const { return tuple() < rhs.tuple(); }
62 bool operator<=(const UInt128 rhs) const { return tuple() <= rhs.tuple(); }
63 bool operator>(const UInt128 rhs) const { return tuple() > rhs.tuple(); }
64 bool operator>=(const UInt128 rhs) const { return tuple() >= rhs.tuple(); }
65
66 UInt128 operator<<(const UInt128& rhs) const {
67 const uint64_t shift = rhs.low;
68 if (((bool)rhs.high) || (shift >= 128)) {
69 return UInt128(0);
70 } else if (shift == 64) {
71 return UInt128(0, low);
72 } else if (shift == 0) {
73 return *this;
74 } else if (shift < 64) {
75 return UInt128(low << shift, (high << shift) + (low >> (64 - shift)));
76 } else if ((128 > shift) && (shift > 64)) {
77 return UInt128(0, low << (shift - 64));
78 } else {
79 return UInt128(0);
80 }
81 }
82
83 UInt128& operator<<=(const UInt128& rhs) {
84 *this = *this << rhs;
85 return *this;
86 }
87
88 UInt128 operator+(const UInt128& rhs) const {
89 return UInt128(low + rhs.low, high + rhs.high + ((low + rhs.low) < low));
90 }
91
92 UInt128& operator+=(const UInt128& rhs) {
93 high += rhs.high + ((low + rhs.low) < low);
94 low += rhs.low;
95 return *this;
96 }
