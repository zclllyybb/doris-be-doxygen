26inline int64_t BitPacking::NumValuesToUnpack(int bit_width, int64_t in_bytes, int64_t num_values) {
27 // Check if we have enough input bytes to decode 'num_values'.
28 if (bit_width == 0 || BitUtil::RoundUpNumBytes(num_values * bit_width) <= in_bytes) {
29 // Limited by output space.
30 return num_values;
31 } else {
32 // Limited by the number of input bytes. Compute the number of values that can be
33 // unpacked from the input.
34 return (in_bytes * CHAR_BIT) / bit_width;
35 }
36}

38constexpr uint64_t GetMask(int num_bits) {
39 if (num_bits >= 64) {
40 return ~0L;
41 }
42 return (1ULL << num_bits) - 1;
43}

51template <typename OutType>
52std::pair<const uint8_t*, int64_t> BitPacking::UnpackValues(int bit_width,
53 const uint8_t* __restrict__ in,
54 int64_t in_bytes, int64_t num_values,
55 OutType* __restrict__ out) {
56 static_assert(IsSupportedUnpackingType<OutType>(), "Only unsigned integers are supported.");
57
58#pragma push_macro("UNPACK_VALUES_CASE")
59#define UNPACK_VALUES_CASE(ignore1, i, ignore2) \
60 case i: \
61 return UnpackValues<OutType, i>(in, in_bytes, num_values, out);
62
63 switch (bit_width) {
64 // Expand cases from 0 to 64.
65 BOOST_PP_REPEAT_FROM_TO(0, 65, UNPACK_VALUES_CASE, ignore);
66 default:
67 DCHECK(false);
68 return std::make_pair(nullptr, -1);
69 }
70#pragma pop_macro("UNPACK_VALUES_CASE")
71}

73template <typename OutType, int BIT_WIDTH>
74std::pair<const uint8_t*, int64_t> BitPacking::UnpackValues(const uint8_t* __restrict__ in,
75 int64_t in_bytes, int64_t num_values,
76 OutType* __restrict__ out) {
77 static_assert(IsSupportedUnpackingType<OutType>(), "Only unsigned integers are supported.");
78
79 constexpr int BATCH_SIZE = 32;
80 const int64_t values_to_read = NumValuesToUnpack(BIT_WIDTH, in_bytes, num_values);
81 const int64_t batches_to_read = values_to_read / BATCH_SIZE;
82 const int64_t remainder_values = values_to_read % BATCH_SIZE;
83 const uint8_t* in_pos = in;
84 OutType* out_pos = out;
85
86 // First unpack as many full batches as possible.
87 for (int64_t i = 0; i < batches_to_read; ++i) {
88 in_pos = Unpack32Values<OutType, BIT_WIDTH>(in_pos, in_bytes, out_pos);
89 out_pos += BATCH_SIZE;
90 in_bytes -= (BATCH_SIZE * BIT_WIDTH) / CHAR_BIT;
91 }
92
93 // Then unpack the final partial batch.
94 if (remainder_values > 0) {
95 in_pos =
96 UnpackUpTo31Values<OutType, BIT_WIDTH>(in_pos, in_bytes, remainder_values, out_pos);
97 }
98 return std::make_pair(in_pos, values_to_read);
99}

121template <typename OutType, int BIT_WIDTH>
122std::pair<const uint8_t*, int64_t> BitPacking::UnpackAndDecodeValues(
123 const uint8_t* __restrict__ in, int64_t in_bytes, OutType* __restrict__ dict,
124 int64_t dict_len, int64_t num_values, OutType* __restrict__ out, int64_t stride,
125 bool* __restrict__ decode_error) {
126 constexpr int BATCH_SIZE = 32;
127 const int64_t values_to_read = NumValuesToUnpack(BIT_WIDTH, in_bytes, num_values);
128 const int64_t batches_to_read = values_to_read / BATCH_SIZE;
129 const int64_t remainder_values = values_to_read % BATCH_SIZE;
130 const uint8_t* in_pos = in;
131 uint8_t* out_pos = reinterpret_cast<uint8_t*>(out);
132 // First unpack as many full batches as possible.
133 for (int64_t i = 0; i < batches_to_read; ++i) {
134 in_pos = UnpackAndDecode32Values<OutType, BIT_WIDTH>(in_pos, in_bytes, dict, dict_len,
135 reinterpret_cast<OutType*>(out_pos),
136 stride, decode_error);
137 out_pos += stride * BATCH_SIZE;
138 in_bytes -= (BATCH_SIZE * BIT_WIDTH) / CHAR_BIT;
139 }
140 // Then unpack the final partial batch.
141 if (remainder_values > 0) {
142 in_pos = UnpackAndDecodeUpTo31Values<OutType, BIT_WIDTH>(
143 in_pos, in_bytes, dict, dict_len, remainder_values,
144 reinterpret_cast<OutType*>(out_pos), stride, decode_error);
145 }
146 return std::make_pair(in_pos, values_to_read);
147}

149// Loop body of unrolled loop that unpacks the value. BIT_WIDTH is the bit width of
150// the packed values. 'in_buf' is the start of the input buffer and 'out_vals' is the
151// start of the output values array. This function unpacks the VALUE_IDX'th packed value
152// from 'in_buf'.
153//
154// This implements essentially the same algorithm as the (Apache-licensed) code in
155// bpacking.c at https://github.com/lemire/FrameOfReference/, but is much more compact
156// because it uses templates rather than source-level unrolling of all combinations.
157//
158// After the template parameters are expanded and constants are propagated, all branches
159// and offset/shift calculations should be optimized out, leaving only shifts by constants
160// and bitmasks by constants. Calls to this must be stamped out manually or with
161// BOOST_PP_REPEAT_FROM_TO: experimentation revealed that the GCC 4.9.2 optimiser was
162// not able to fully propagate constants and remove branches when this was called from
163// inside a for loop with constant bounds with VALUE_IDX changed to a function argument.
164//
165// We compute how many 32 bit words we have to read, which is either 1, 2 or 3. If it is
166// at least 2, the first two 32 bit words are read as one 64 bit word. Even if only one
167// word needs to be read, we try to read 64 bits if it does not lead to buffer overflow
168// because benchmarks show that it has a positive effect on performance.
169//
170// If 'FULL_BATCH' is true, this function call is part of unpacking 32 values, otherwise
171// up to 31 values. This is needed to optimise the length of the reads (32 or 64 bits) and
172// avoid buffer overflow (if we are unpacking 32 values, we can safely assume an input
173// buffer of length 32 * BIT_WIDTH).
174template <int BIT_WIDTH, int VALUE_IDX, bool FULL_BATCH>
175uint64_t UnpackValue(const uint8_t* __restrict__ in_buf) {
176 if (BIT_WIDTH == 0) return 0;
177
178 constexpr int FIRST_BIT_IDX = VALUE_IDX * BIT_WIDTH;
179 constexpr int FIRST_WORD_IDX = FIRST_BIT_IDX / 32;
180 constexpr int LAST_BIT_IDX = FIRST_BIT_IDX + BIT_WIDTH;
181 constexpr int LAST_WORD_IDX = BitUtil::round_up_numi32(LAST_BIT_IDX);
182 constexpr int WORDS_TO_READ = LAST_WORD_IDX - FIRST_WORD_IDX;
183 static_assert(WORDS_TO_READ <= 3, "At most three 32-bit words need to be loaded.");
184
185 constexpr int FIRST_BIT_OFFSET = FIRST_BIT_IDX - FIRST_WORD_IDX * 32;
186 constexpr uint64_t mask = GetMask(BIT_WIDTH);
187 const uint32_t* const in = reinterpret_cast<const uint32_t*>(in_buf);
188
189 // Avoid reading past the end of the buffer. We can safely read 64 bits if we know that
190 // this is a full batch read (so the input buffer is 32 * BIT_WIDTH long) and there is
191 // enough space in the buffer from the current reading point.
192 // We try to read 64 bits even when it is not necessary because the benchmarks show it
193 // is faster.
194 constexpr bool CAN_SAFELY_READ_64_BITS =
195 FULL_BATCH && FIRST_BIT_IDX - FIRST_BIT_OFFSET + 64 <= BIT_WIDTH * 32;
196
197 // We do not try to read 64 bits when the bit width is a power of two (unless it is
198 // necessary) because performance benchmarks show that it is better this way. This seems
199 // to be due to compiler optimisation issues, so we can revisit it when we update the
200 // compiler version.
201 constexpr bool READ_32_BITS =
202 WORDS_TO_READ == 1 && (!CAN_SAFELY_READ_64_BITS || BitUtil::IsPowerOf2(BIT_WIDTH));
203
204 if (READ_32_BITS) {
205 uint32_t word = in[FIRST_WORD_IDX];
206 word >>= FIRST_BIT_OFFSET < 32 ? FIRST_BIT_OFFSET : 0;
207 return word & mask;
208 }
209
210 uint64_t word = *reinterpret_cast<const uint64_t*>(in + FIRST_WORD_IDX);
211 word >>= FIRST_BIT_OFFSET;
212
213 if (WORDS_TO_READ > 2) {
214 constexpr int USEFUL_BITS = FIRST_BIT_OFFSET == 0 ? 0 : 64 - FIRST_BIT_OFFSET;
215 uint64_t extra_word = in[FIRST_WORD_IDX + 2];
216 word |= extra_word << USEFUL_BITS;
217 }
218
219 return word & mask;
220}

222template <typename OutType>
223void DecodeValue(OutType* __restrict__ dict, int64_t dict_len, uint32_t idx,
224 OutType* __restrict__ out_val, bool* __restrict__ decode_error) {
225 if (UNLIKELY(idx >= dict_len)) {
226 *decode_error = true;
227 } else {
228 // Use memcpy() because we can't assume sufficient alignment in some cases (e.g.
229 // 16 byte decimals).
230 memcpy(out_val, &dict[idx], sizeof(OutType));
231 }

234template <typename OutType, int BIT_WIDTH>
235const uint8_t* BitPacking::Unpack32Values(const uint8_t* __restrict__ in, int64_t in_bytes,
236 OutType* __restrict__ out) {
237 static_assert(BIT_WIDTH >= 0, "BIT_WIDTH too low");
238 static_assert(BIT_WIDTH <= MAX_BITWIDTH, "BIT_WIDTH too high");
239 DCHECK_LE(BIT_WIDTH, sizeof(OutType) * CHAR_BIT) << "BIT_WIDTH too high for output";
240 constexpr int BYTES_TO_READ = BitUtil::RoundUpNumBytes(32 * BIT_WIDTH);
241 DCHECK_GE(in_bytes, BYTES_TO_READ);
242
243 // Call UnpackValue for 0 <= i < 32.
244#pragma push_macro("UNPACK_VALUE_CALL")
245#define UNPACK_VALUE_CALL(ignore1, i, ignore2) \
246 out[i] = static_cast<OutType>(UnpackValue<BIT_WIDTH, i, true>(in));
247
248 BOOST_PP_REPEAT_FROM_TO(0, 32, UNPACK_VALUE_CALL, ignore);
249 return in + BYTES_TO_READ;
250#pragma pop_macro("UNPACK_VALUE_CALL")
251}

271template <typename OutType, int BIT_WIDTH>
272const uint8_t* BitPacking::UnpackAndDecode32Values(const uint8_t* __restrict__ in, int64_t in_bytes,
273 OutType* __restrict__ dict, int64_t dict_len,
274 OutType* __restrict__ out, int64_t stride,
275 bool* __restrict__ decode_error) {
276 static_assert(BIT_WIDTH >= 0, "BIT_WIDTH too low");
277 static_assert(BIT_WIDTH <= MAX_BITWIDTH, "BIT_WIDTH too high");
278 constexpr int BYTES_TO_READ = BitUtil::RoundUpNumBytes(32 * BIT_WIDTH);
279 DCHECK_GE(in_bytes, BYTES_TO_READ);
280 // TODO: this could be optimised further by using SIMD instructions.
281 // https://lemire.me/blog/2016/08/25/faster-dictionary-decoding-with-simd-instructions/
282
283 static_assert(BIT_WIDTH <= MAX_DICT_BITWIDTH, "Too high bit width for dictionary index.");
284
285 // Call UnpackValue() and DecodeValue() for 0 <= i < 32.
286#pragma push_macro("DECODE_VALUE_CALL")
287#define DECODE_VALUE_CALL(ignore1, i, ignore2) \
288 { \
289 uint32_t idx = UnpackValue<BIT_WIDTH, i, true>(in); \
290 uint8_t* out_pos = reinterpret_cast<uint8_t*>(out) + i * stride; \
291 DecodeValue(dict, dict_len, idx, reinterpret_cast<OutType*>(out_pos), decode_error); \
292 }
293
294 BOOST_PP_REPEAT_FROM_TO(0, 32, DECODE_VALUE_CALL, ignore);
295 return in + BYTES_TO_READ;
296#pragma pop_macro("DECODE_VALUE_CALL")

299template <typename OutType, int BIT_WIDTH>
300const uint8_t* BitPacking::UnpackUpTo31Values(const uint8_t* __restrict__ in, int64_t in_bytes,
301 int num_values, OutType* __restrict__ out) {
302 static_assert(BIT_WIDTH >= 0, "BIT_WIDTH too low");
303 static_assert(BIT_WIDTH <= MAX_BITWIDTH, "BIT_WIDTH too high");
304 DCHECK_LE(BIT_WIDTH, sizeof(OutType) * CHAR_BIT) << "BIT_WIDTH too high for output";
305 constexpr int MAX_BATCH_SIZE = 31;
306 const int BYTES_TO_READ = BitUtil::RoundUpNumBytes(num_values * BIT_WIDTH);
307 DCHECK_GE(in_bytes, BYTES_TO_READ);
308 DCHECK_LE(num_values, MAX_BATCH_SIZE);
309
310 // Make sure the buffer is at least 1 byte.
311 constexpr int TMP_BUFFER_SIZE = BIT_WIDTH ? (BIT_WIDTH * (MAX_BATCH_SIZE + 1)) / CHAR_BIT : 1;
312 uint8_t tmp_buffer[TMP_BUFFER_SIZE];
313
314 const uint8_t* in_buffer = in;
315 // Copy into padded temporary buffer to avoid reading past the end of 'in' if the
316 // last 32-bit load would go past the end of the buffer.
317 if (BitUtil::round_up(BYTES_TO_READ, sizeof(uint32_t)) > in_bytes) {
318 memcpy(tmp_buffer, in, BYTES_TO_READ);
319 in_buffer = tmp_buffer;
320 }
321
322#pragma push_macro("UNPACK_VALUES_CASE")
323#define UNPACK_VALUES_CASE(ignore1, i, ignore2) \
324 case 31 - i: \
325 out[30 - i] = static_cast<OutType>(UnpackValue<BIT_WIDTH, 30 - i, false>(in_buffer)); \
326 [[fallthrough]];
327
328 // Use switch with fall-through cases to minimise branching.
329 switch (num_values) {
330 // Expand cases from 31 down to 1.
331 BOOST_PP_REPEAT_FROM_TO(0, 31, UNPACK_VALUES_CASE, ignore);
332 case 0:
333 break;
334 default:
335 DCHECK(false);
336 }
337 return in + BYTES_TO_READ;
338#pragma pop_macro("UNPACK_VALUES_CASE")
339}
