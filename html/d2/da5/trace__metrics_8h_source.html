30// A simple map of constant string names to integer counters.
31//
32// Typically, the TRACE_COUNTER_INCREMENT(...) macro defined in
33// trace.h is used to increment a counter within this map.
34//
35// This currently is just a thin wrapper around a spinlocked map,
36// but if it becomes noticeable in the CPU profile, various optimizations
37// are plausible.
38class TraceMetrics {
39public:
40 TraceMetrics() {}
41 ~TraceMetrics() {}
42
43 // Internalize the given string by duplicating it into a process-wide
44 // pool. If this string has already been interned, returns a pointer
45 // to a previous instance. Otherwise, copies it into the pool.
46 //
47 // The resulting strings are purposefully leaked, so this should only
48 // be used in cases where the number of unique strings that will be
49 // passed is relatively low (i.e. not user-specified).
50 //
51 // Because 'name' is exposed back to operators, it must be a printable
52 // ASCII string.
53 static const char* InternName(const std::string& name);
54
55 // Increment the given counter.
56 void Increment(const char* name, int64_t amount);
57
58 // Return a copy of the current counter map.
59 std::map<const char*, int64_t> Get() const;
60
61 // Return metric's current value.
62 //
63 // NOTE: the 'name' MUST be the same const char* which is used for
64 // insertion. This is because we do pointer-wise comparison internally.
65 int64_t GetMetric(const char* name) const;
66
67private:
68 mutable SpinLock lock_;
69 std::map<const char*, int64_t> counters_;
70
71 DISALLOW_COPY_AND_ASSIGN(TraceMetrics);
72};
73
74inline void TraceMetrics::Increment(const char* name, int64_t amount) {
75 std::lock_guard<SpinLock> l(lock_);
76 counters_[name] += amount;
77}
78
79inline std::map<const char*, int64_t> TraceMetrics::Get() const {
80 std::unique_lock<SpinLock> l(lock_);
81 return counters_;
82}
83
84inline int64_t TraceMetrics::GetMetric(const char* name) const {
85 std::lock_guard<SpinLock> l(lock_);
86 return FindWithDefault(counters_, name, 0);
87}

89} // namespace doris
