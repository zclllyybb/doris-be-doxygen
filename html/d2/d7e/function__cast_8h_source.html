50inline UInt32 extract_to_decimal_scale(const ColumnWithTypeAndName& named_column) {
51 const auto* arg_type = named_column.type.get();
52 bool ok = check_and_get_data_type<DataTypeUInt64>(arg_type) ||
53 check_and_get_data_type<DataTypeUInt32>(arg_type) ||
54 check_and_get_data_type<DataTypeUInt16>(arg_type) ||
55 check_and_get_data_type<DataTypeUInt8>(arg_type);
56 if (!ok) {
57 LOG(FATAL) << fmt::format("Illegal type of toDecimal() scale {}",
58 named_column.type->get_name());
59 }
60
61 Field field;
62 named_column.column->get(0, field);
63 return field.get<UInt32>();

69template <typename FromDataType, typename ToDataType, typename Name>
70struct ConvertImpl {
71 using FromFieldType = typename FromDataType::FieldType;
72 using ToFieldType = typename ToDataType::FieldType;
73
74 template <typename Additions = void*>
75 static Status execute(Block& block, const ColumnNumbers& arguments, size_t result,
76 size_t /*input_rows_count*/, bool check_overflow [[maybe_unused]] = false,
77 Additions additions [[maybe_unused]] = Additions()) {
78 const ColumnWithTypeAndName& named_from = block.get_by_position(arguments[0]);
79
80 using ColVecFrom =
81 std::conditional_t<IsDecimalNumber<FromFieldType>, ColumnDecimal<FromFieldType>,
82 ColumnVector<FromFieldType>>;
83 using ColVecTo = std::conditional_t<IsDecimalNumber<ToFieldType>,
84 ColumnDecimal<ToFieldType>, ColumnVector<ToFieldType>>;
85
86 if constexpr (IsDataTypeDecimal<FromDataType> || IsDataTypeDecimal<ToDataType>) {
87 if constexpr (!(IsDataTypeDecimalOrNumber<FromDataType> || IsTimeType<FromDataType> ||
88 IsTimeV2Type<FromDataType>) ||
89 !IsDataTypeDecimalOrNumber<ToDataType>)
90 return Status::RuntimeError("Illegal column {} of first argument of function {}",
91 named_from.column->get_name(), Name::name);
92 }
93
94 if (const ColVecFrom* col_from =
95 check_and_get_column<ColVecFrom>(named_from.column.get())) {
96 typename ColVecTo::MutablePtr col_to = nullptr;
97 if constexpr (IsDataTypeDecimal<ToDataType>) {
98 UInt32 scale = additions;
99 col_to = ColVecTo::create(0, scale);
100 } else {
101 col_to = ColVecTo::create();
102 }
103
104 const auto& vec_from = col_from->get_data();
105 auto& vec_to = col_to->get_data();
106 size_t size = vec_from.size();
107 vec_to.resize(size);
108
109 if constexpr (IsDataTypeDecimal<FromDataType> || IsDataTypeDecimal<ToDataType>) {
110 ColumnUInt8::MutablePtr col_null_map_to = nullptr;
111 UInt8* vec_null_map_to = nullptr;
112 if (check_overflow) {
113 col_null_map_to = ColumnUInt8::create(size, 0);
114 vec_null_map_to = col_null_map_to->get_data().data();
115 }
116 for (size_t i = 0; i < size; ++i) {
117 if constexpr (IsDataTypeDecimal<FromDataType> &&
118 IsDataTypeDecimal<ToDataType>) {
119 vec_to[i] = convert_decimals<FromDataType, ToDataType>(
120 vec_from[i], vec_from.get_scale(), vec_to.get_scale(),
121 vec_null_map_to ? &vec_null_map_to[i] : vec_null_map_to);
122 } else if constexpr (IsDataTypeDecimal<FromDataType> &&
123 IsDataTypeNumber<ToDataType>) {
124 vec_to[i] = convert_from_decimal<FromDataType, ToDataType>(
125 vec_from[i], vec_from.get_scale());
126 } else if constexpr (IsDataTypeNumber<FromDataType> &&
127 IsDataTypeDecimal<ToDataType>) {
128 vec_to[i] = convert_to_decimal<FromDataType, ToDataType>(
129 vec_from[i], vec_to.get_scale(),
130 vec_null_map_to ? &vec_null_map_to[i] : vec_null_map_to);
131 } else if constexpr (IsTimeType<FromDataType> &&
132 IsDataTypeDecimal<ToDataType>) {
133 vec_to[i] = convert_to_decimal<DataTypeInt64, ToDataType>(
134 reinterpret_cast<const VecDateTimeValue&>(vec_from[i]).to_int64(),
135 vec_to.get_scale(),
136 vec_null_map_to ? &vec_null_map_to[i] : vec_null_map_to);
137 } else if constexpr (IsDateV2Type<FromDataType> &&
138 IsDataTypeDecimal<ToDataType>) {
139 vec_to[i] = convert_to_decimal<DataTypeUInt32, ToDataType>(
140 reinterpret_cast<const DateV2Value<DateV2ValueType>&>(vec_from[i])
141 .to_date_int_val(),
142 vec_to.get_scale(),
143 vec_null_map_to ? &vec_null_map_to[i] : vec_null_map_to);
144 } else if constexpr (IsDateTimeV2Type<FromDataType> &&
145 IsDataTypeDecimal<ToDataType>) {
146 // TODO: should we consider the scale of datetimev2?
147 vec_to[i] = convert_to_decimal<DataTypeUInt64, ToDataType>(
148 reinterpret_cast<const DateV2Value<DateTimeV2ValueType>&>(
149 vec_from[i])
150 .to_date_int_val(),
151 vec_to.get_scale(),
152 vec_null_map_to ? &vec_null_map_to[i] : vec_null_map_to);
153 }
154 }
155 if (check_overflow) {
156 block.replace_by_position(
157 result,
158 ColumnNullable::create(std::move(col_to), std::move(col_null_map_to)));
159 } else {
160 block.replace_by_position(result, std::move(col_to));
161 }
162
163 return Status::OK();
164 } else if constexpr (IsTimeType<FromDataType>) {
165 for (size_t i = 0; i < size; ++i) {
166 if constexpr (IsTimeType<ToDataType>) {
167 vec_to[i] = static_cast<ToFieldType>(vec_from[i]);
168 if constexpr (IsDateTimeType<ToDataType>) {
169 DataTypeDateTime::cast_to_date_time(vec_to[i]);
170 } else {
171 DataTypeDate::cast_to_date(vec_to[i]);
172 }
173 } else if constexpr (IsDateV2Type<ToDataType>) {
174 DataTypeDateV2::cast_from_date(vec_from[i], vec_to[i]);
175 } else if constexpr (IsDateTimeV2Type<ToDataType>) {
176 DataTypeDateTimeV2::cast_from_date(vec_from[i], vec_to[i]);
177 } else {
178 vec_to[i] =
179 reinterpret_cast<const VecDateTimeValue&>(vec_from[i]).to_int64();
180 }
181 }
182 } else if constexpr (IsTimeV2Type<FromDataType>) {
183 for (size_t i = 0; i < size; ++i) {
184 if constexpr (IsTimeV2Type<ToDataType>) {
185 if constexpr (IsDateTimeV2Type<ToDataType> && IsDateV2Type<FromDataType>) {
186 DataTypeDateV2::cast_to_date_time_v2(vec_from[i], vec_to[i]);
187 } else if constexpr (IsDateTimeV2Type<FromDataType> &&
188 IsDateV2Type<ToDataType>) {
189 DataTypeDateTimeV2::cast_to_date_v2(vec_from[i], vec_to[i]);
190 } else {
191 UInt32 scale = additions;
192 vec_to[i] = vec_from[i] / std::pow(10, 6 - scale);
193 }
194 } else if constexpr (IsTimeType<ToDataType>) {
195 if constexpr (IsDateTimeType<ToDataType> && IsDateV2Type<FromDataType>) {
196 DataTypeDateV2::cast_to_date_time(vec_from[i], vec_to[i]);
197 } else if constexpr (IsDateType<ToDataType> && IsDateV2Type<FromDataType>) {
198 DataTypeDateV2::cast_to_date(vec_from[i], vec_to[i]);
199 } else if constexpr (IsDateTimeType<ToDataType> &&
200 IsDateTimeV2Type<FromDataType>) {
201 DataTypeDateTimeV2::cast_to_date_time(vec_from[i], vec_to[i]);
202 } else if constexpr (IsDateType<ToDataType> &&
203 IsDateTimeV2Type<FromDataType>) {
204 DataTypeDateTimeV2::cast_to_date(vec_from[i], vec_to[i]);
205 } else {
206 return Status::InvalidArgument("Wrong cast expression!");
207 }
208 } else {
209 if constexpr (IsDateTimeV2Type<FromDataType>) {
210 vec_to[i] = reinterpret_cast<const DateV2Value<DateTimeV2ValueType>&>(
211 vec_from[i])
212 .to_int64();
213 } else {
214 vec_to[i] = reinterpret_cast<const DateV2Value<DateV2ValueType>&>(
215 vec_from[i])
216 .to_int64();
217 }
218 }
219 }
220 } else {
221 for (size_t i = 0; i < size; ++i) {
222 vec_to[i] = static_cast<ToFieldType>(vec_from[i]);
223 }
224 }
225
226 // TODO: support boolean cast more reasonable
227 if constexpr (std::is_same_v<uint8_t, ToFieldType>) {
228 for (int i = 0; i < size; ++i) {
229 vec_to[i] = static_cast<bool>(vec_to[i]);
230 }
231 }
232
233 block.replace_by_position(result, std::move(col_to));
234 } else {
235 return Status::RuntimeError("Illegal column {} of first argument of function {}",
236 named_from.column->get_name(), Name::name);
237 }
238 return Status::OK();
239 }

256struct ConvertImplToTimeType {

258 using ToFieldType = typename ToDataType::FieldType;

260 static Status execute(Block& block, const ColumnNumbers& arguments, size_t result,
261 size_t /*input_rows_count*/) {
262 const ColumnWithTypeAndName& named_from = block.get_by_position(arguments[0]);
263
264 using ColVecFrom =
265 std::conditional_t<IsDecimalNumber<FromFieldType>, ColumnDecimal<FromFieldType>,
266 ColumnVector<FromFieldType>>;
267
268 using DateValueType = std::conditional_t<
269 IsTimeV2Type<ToDataType>,
270 std::conditional_t<IsDateV2Type<ToDataType>, DateV2Value<DateV2ValueType>,
271 DateV2Value<DateTimeV2ValueType>>,
272 VecDateTimeValue>;
273 using ColVecTo = ColumnVector<ToFieldType>;
274
275 if (const ColVecFrom* col_from =
276 check_and_get_column<ColVecFrom>(named_from.column.get())) {
277 const auto& vec_from = col_from->get_data();
278 size_t size = vec_from.size();
279
280 // create nested column
281 auto col_to = ColVecTo::create(size);
282 auto& vec_to = col_to->get_data();
283
284 // create null column
285 ColumnUInt8::MutablePtr col_null_map_to;
286 col_null_map_to = ColumnUInt8::create(size);
287 auto& vec_null_map_to = col_null_map_to->get_data();
288
289 for (size_t i = 0; i < size; ++i) {
290 auto& date_value = reinterpret_cast<DateValueType&>(vec_to[i]);
291 if constexpr (IsDecimalNumber<FromFieldType>) {
292 // TODO: should we consider the scale of datetimev2?
293 vec_null_map_to[i] = !date_value.from_date_int64(
294 convert_from_decimal<FromDataType, DataTypeInt64>(
295 vec_from[i], vec_from.get_scale()));
296 } else {
297 vec_null_map_to[i] = !date_value.from_date_int64(vec_from[i]);
298 }
299 // DateType of VecDateTimeValue should cast to date
300 if constexpr (IsDateType<ToDataType>) {
301 date_value.cast_to_date();
302 } else if constexpr (IsDateTimeType<ToDataType>) {
303 date_value.to_datetime();
304 }
305 }
306 block.get_by_position(result).column =
307 ColumnNullable::create(std::move(col_to), std::move(col_null_map_to));
308 } else {
309 return Status::RuntimeError("Illegal column {} of first argument of function {}",
310 named_from.column->get_name(), Name::name);
311 }
312
313 return Status::OK();
314 }

344struct ConvertImplGenericFromString {
345 static Status execute(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
346 const size_t result, size_t input_rows_count) {
347 static_assert(std::is_same_v<StringColumnType, ColumnString>,
348 "Can be used only to parse from ColumnString");
349 const auto& col_with_type_and_name = block.get_by_position(arguments[0]);
350 const IColumn& col_from = *col_with_type_and_name.column;
351 // result column must set type

353 auto data_type_to = block.get_by_position(result).type;
354 if (const StringColumnType* col_from_string =
355 check_and_get_column<StringColumnType>(&col_from)) {
356 auto col_to = data_type_to->create_column();
357
358 size_t size = col_from.size();
359 col_to->reserve(size);
360
361 ColumnUInt8::MutablePtr col_null_map_to = ColumnUInt8::create(size);
362 ColumnUInt8::Container* vec_null_map_to = &col_null_map_to->get_data();
363
364 for (size_t i = 0; i < size; ++i) {
365 const auto& val = col_from_string->get_data_at(i);
366 // Note: here we should handle the null element
367 if (val.size == 0) {
368 col_to->insert_default();
369 continue;
370 }
371 ReadBuffer read_buffer((char*)(val.data), val.size);
372 Status st = data_type_to->from_string(read_buffer, col_to);
373 // if parsing failed, will return null
374 (*vec_null_map_to)[i] = !st.ok();
375 if (!st.ok()) {
376 col_to->insert_default();
377 }
378 }
379 block.get_by_position(result).column =
380 ColumnNullable::create(std::move(col_to), std::move(col_null_map_to));
381 } else {
382 return Status::RuntimeError(
383 "Illegal column {} of first argument of conversion function from string",
384 col_from.get_name());
385 }
386 return Status::OK();
387 }

463template <TypeIndex type_index, typename ColumnType>
464struct ConvertImplFromJsonb {
465 static Status execute(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
466 const size_t result, size_t input_rows_count) {
467 const auto& col_with_type_and_name = block.get_by_position(arguments[0]);
468 const IColumn& col_from = *col_with_type_and_name.column;
469 // result column must set type
470 DCHECK(block.get_by_position(result).type != nullptr);
471 auto data_type_to = block.get_by_position(result).type;
472 if (const ColumnString* column_string = check_and_get_column<ColumnString>(&col_from)) {
473 auto null_map_col = ColumnUInt8::create(input_rows_count, 0);
474 auto& null_map = null_map_col->get_data();
475 auto col_to = ColumnType::create();
476
477 //IColumn & col_to = *res;
478 // size_t size = col_from.size();
479 col_to->reserve(input_rows_count);
480 auto& res = col_to->get_data();
481 res.resize(input_rows_count);
482
483 for (size_t i = 0; i < input_rows_count; ++i) {
484 const auto& val = column_string->get_data_at(i);
485 // ReadBuffer read_buffer((char*)(val.data), val.size);
486 // RETURN_IF_ERROR(data_type_to->from_string(read_buffer, col_to));
487
488 if (val.size == 0) {
489 null_map[i] = 1;
490 res[i] = 0;
491 continue;
492 }
493
494 // doc is NOT necessary to be deleted since JsonbDocument will not allocate memory
495 JsonbDocument* doc = JsonbDocument::createDocument(val.data, val.size);
496 if (UNLIKELY(!doc || !doc->getValue())) {
497 null_map[i] = 1;
498 res[i] = 0;
499 continue;
500 }
501
502 // value is NOT necessary to be deleted since JsonbValue will not allocate memory
503 JsonbValue* value = doc->getValue();
504 if (UNLIKELY(!value)) {
505 null_map[i] = 1;
506 res[i] = 0;
507 continue;
508 }
509
510 if constexpr (type_index == TypeIndex::UInt8) {
511 if (value->isTrue()) {
512 res[i] = 1;
513 } else if (value->isFalse()) {
514 res[i] = 0;
515 } else {
516 null_map[i] = 1;
517 res[i] = 0;
518 }
519 } else if constexpr (type_index == TypeIndex::Int8) {
520 if (value->isInt8()) {
521 res[i] = ((const JsonbIntVal*)value)->val();
522 } else {
523 null_map[i] = 1;
524 res[i] = 0;
525 }
526 } else if constexpr (type_index == TypeIndex::Int16) {
527 if (value->isInt8() || value->isInt16()) {
528 res[i] = (int16_t)((const JsonbIntVal*)value)->val();
529 } else {
530 null_map[i] = 1;
531 res[i] = 0;
532 }
533 } else if constexpr (type_index == TypeIndex::Int32) {
534 if (value->isInt8() || value->isInt16() || value->isInt32()) {
535 res[i] = (int32_t)((const JsonbIntVal*)value)->val();
536 } else {
537 null_map[i] = 1;
538 res[i] = 0;
539 }
540 } else if constexpr (type_index == TypeIndex::Int64) {
541 if (value->isInt8() || value->isInt16() || value->isInt32() ||
542 value->isInt64()) {
543 res[i] = ((const JsonbIntVal*)value)->val();
544 } else {
545 null_map[i] = 1;
546 res[i] = 0;
547 }
548 } else if constexpr (type_index == TypeIndex::Float64) {
549 if (value->isDouble()) {
550 res[i] = ((const JsonbDoubleVal*)value)->val();
551 } else if (value->isInt8() || value->isInt16() || value->isInt32() ||
552 value->isInt64()) {
553 res[i] = ((const JsonbIntVal*)value)->val();
554 } else {
555 null_map[i] = 1;
556 res[i] = 0;
557 }
558 } else {
559 LOG(FATAL) << "unsupported type ";
560 }
561 }
562
563 block.replace_by_position(
564 result, ColumnNullable::create(std::move(col_to), std::move(null_map_col)));
565 } else {
566 return Status::RuntimeError(
567 "Illegal column {} of first argument of conversion function from string",
568 col_from.get_name());
569 }
570 return Status::OK();
571 }

695template <typename T>
696struct ToNumberMonotonicity {
697 static bool has() { return true; }
698
699 static UInt64 divide_by_range_of_type(UInt64 x) {
700 if constexpr (sizeof(T) < sizeof(UInt64))
701 return x >> (sizeof(T) * 8);
702 else
703 return 0;
704 }
705
706 static IFunction::Monotonicity get(const IDataType& type, const Field& left,
707 const Field& right) {
708 if (!type.is_value_represented_by_number()) return {};
709
712 if (check_and_get_data_type<DataTypeNumber<T>>(
713 &type) /*|| check_and_get_data_type<DataTypeEnum<T>>(&type)*/)
714 return {true, true, true};
715
717
719 if (std::is_floating_point_v<T>) return {true, true, true};
720
722 if (WhichDataType(type).is_float()) {
723 if (left.is_null() || right.is_null()) return {};
724
725 Float64 left_float = left.get<Float64>();
726 Float64 right_float = right.get<Float64>();
727
728 if (left_float >= std::numeric_limits<T>::min() &&
729 left_float <= static_cast<Float64>(std::numeric_limits<T>::max()) &&
730 right_float >= std::numeric_limits<T>::min() &&
731 right_float <= static_cast<Float64>(std::numeric_limits<T>::max()))
732 return {true};
733
734 return {};
735 }
736
738
739 const bool from_is_unsigned = type.is_value_represented_by_unsigned_integer();
740 const bool to_is_unsigned = std::is_unsigned_v<T>;
741
742 const size_t size_of_from = type.get_size_of_value_in_memory();
743 const size_t size_of_to = sizeof(T);
744
745 const bool left_in_first_half =
746 left.is_null() ? from_is_unsigned : (left.get<Int64>() >= 0);
747
748 const bool right_in_first_half =
749 right.is_null() ? !from_is_unsigned : (right.get<Int64>() >= 0);
750
752 if (size_of_from == size_of_to) {
753 if (from_is_unsigned == to_is_unsigned) return {true, true, true};
754
755 if (left_in_first_half == right_in_first_half) return {true};
756
757 return {};
758 }
759
761 if (size_of_from < size_of_to) {
762 if (from_is_unsigned == to_is_unsigned) return {true, true, true};
763
764 if (!to_is_unsigned) return {true, true, true};
765
767 if (left_in_first_half == right_in_first_half) return {true};
768
769 return {};
770 }
771
773 if (size_of_from > size_of_to) {
775 if (left.is_null() || right.is_null()) return {};
776
777 if (from_is_unsigned == to_is_unsigned) {
779 if (divide_by_range_of_type(left.get<UInt64>()) ==
780 divide_by_range_of_type(right.get<UInt64>()))
781 return {true};
782
783 return {};
784 } else {
787 if (left_in_first_half == right_in_first_half &&
788 (T(left.get<Int64>()) >= 0) == (T(right.get<Int64>()) >= 0) &&
789 divide_by_range_of_type(left.get<UInt64>()) ==
790 divide_by_range_of_type(right.get<UInt64>()))
791 return {true};
792
793 return {};
794 }
795 }
796
797 __builtin_unreachable();
798 }
799};

804struct ToStringMonotonicity {
805 static bool has() { return true; }
806
807 static IFunction::Monotonicity get(const IDataType& type, const Field& left,
808 const Field& right) {
809 IFunction::Monotonicity positive(true, true);
810 IFunction::Monotonicity not_monotonic;
811
812 if (left.is_null() || right.is_null()) return {};
813
814 if (left.get_type() == Field::Types::UInt64 && right.get_type() == Field::Types::UInt64) {
815 return (left.get<Int64>() == 0 && right.get<Int64>() == 0) ||
816 (floor(log10(left.get<UInt64>())) ==
817 floor(log10(right.get<UInt64>())))
818 ? positive
819 : not_monotonic;
820 }
821
822 if (left.get_type() == Field::Types::Int64 && right.get_type() == Field::Types::Int64) {
823 return (left.get<Int64>() == 0 && right.get<Int64>() == 0) ||
824 (left.get<Int64>() > 0 && right.get<Int64>() > 0 &&
825 floor(log10(left.get<Int64>())) ==
826 floor(log10(right.get<Int64>())))
827 ? positive
828 : not_monotonic;
829 }
830
831 return not_monotonic;
832 }

1072template <typename FromDataType, typename ToDataType, typename Name>
1073struct ConvertThroughParsing {
1074 static_assert(std::is_same_v<FromDataType, DataTypeString>,
1075 "ConvertThroughParsing is only applicable for String or FixedString data types");
1076
1077 using ToFieldType = typename ToDataType::FieldType;
1078
1079 static bool is_all_read(ReadBuffer& in) { return in.eof(); }
1080
1081 template <typename Additions = void*>
1082 static Status execute(Block& block, const ColumnNumbers& arguments, size_t result,
1083 size_t input_rows_count, bool check_overflow [[maybe_unused]] = false,
1084 Additions additions [[maybe_unused]] = Additions()) {
1085 using ColVecTo = std::conditional_t<IsDecimalNumber<ToFieldType>,
1086 ColumnDecimal<ToFieldType>, ColumnVector<ToFieldType>>;
1087
1088 const DateLUTImpl* local_time_zone [[maybe_unused]] = nullptr;
1089 const DateLUTImpl* utc_time_zone [[maybe_unused]] = nullptr;
1090
1091 const IColumn* col_from = block.get_by_position(arguments[0]).column.get();
1092 const ColumnString* col_from_string = check_and_get_column<ColumnString>(col_from);
1093
1094 if (std::is_same_v<FromDataType, DataTypeString> && !col_from_string) {
1095 return Status::RuntimeError("Illegal column {} of first argument of function {}",
1096 col_from->get_name(), Name::name);
1097 }
1098
1099 size_t size = input_rows_count;
1100 typename ColVecTo::MutablePtr col_to = nullptr;
1101
1102 if constexpr (IsDataTypeDecimal<ToDataType>) {
1103 UInt32 scale = additions;
1104 col_to = ColVecTo::create(size, scale);
1105 } else {
1106 col_to = ColVecTo::create(size);
1107 }
1108
1109 typename ColVecTo::Container& vec_to = col_to->get_data();
1110
1111 ColumnUInt8::MutablePtr col_null_map_to;
1112 ColumnUInt8::Container* vec_null_map_to [[maybe_unused]] = nullptr;
1113 col_null_map_to = ColumnUInt8::create(size);
1114 vec_null_map_to = &col_null_map_to->get_data();
1115
1116 const ColumnString::Chars* chars = nullptr;
1117 const IColumn::Offsets* offsets = nullptr;
1118 size_t fixed_string_size = 0;
1119
1120 if constexpr (std::is_same_v<FromDataType, DataTypeString>) {
1121 chars = &col_from_string->get_chars();
1122 offsets = &col_from_string->get_offsets();
1123 }
1124
1125 size_t current_offset = 0;
1126
1127 for (size_t i = 0; i < size; ++i) {
1128 size_t next_offset = std::is_same_v<FromDataType, DataTypeString>
1129 ? (*offsets)[i]
1130 : (current_offset + fixed_string_size);
1131 size_t string_size = std::is_same_v<FromDataType, DataTypeString>
1132 ? next_offset - current_offset
1133 : fixed_string_size;
1134
1135 ReadBuffer read_buffer(&(*chars)[current_offset], string_size);
1136
1137 bool parsed;
1138 if constexpr (IsDataTypeDecimal<ToDataType>) {
1139 parsed = try_parse_impl<ToDataType>(vec_to[i], read_buffer, local_time_zone,
1140 vec_to.get_scale());
1141 } else if constexpr (IsDataTypeDateTimeV2<ToDataType>) {
1142 auto type = check_and_get_data_type<DataTypeDateTimeV2>(
1143 block.get_by_position(result).type.get());
1144 parsed = try_parse_impl<ToDataType>(vec_to[i], read_buffer, local_time_zone,
1145 type->get_scale());
1146 } else {
1147 parsed = try_parse_impl<ToDataType>(vec_to[i], read_buffer, local_time_zone);
1148 }
1149 (*vec_null_map_to)[i] = !parsed || !is_all_read(read_buffer);
1150
1151 current_offset = next_offset;
1152 }
1153
1154 block.get_by_position(result).column =
1155 ColumnNullable::create(std::move(col_to), std::move(col_null_map_to));
1156 return Status::OK();
1157 }
