40template <typename Impl, typename Name>
41class FunctionUnaryToType : public IFunction {
42public:
43 static constexpr auto name = Name::name;
44 static constexpr bool has_variadic_argument =
45 !std::is_void_v<decltype(has_variadic_argument_types(std::declval<Impl>()))>;
46
47 static FunctionPtr create() { return std::make_shared<FunctionUnaryToType>(); }
48 String get_name() const override { return name; }
49 size_t get_number_of_arguments() const override { return 1; }
50 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
51 return std::make_shared<typename Impl::ReturnType>();
52 }
53
54 bool use_default_implementation_for_constants() const override { return true; }
55
56 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
57 size_t result, size_t input_rows_count) override {
58 return execute_impl<typename Impl::ReturnType>(block, arguments, result, input_rows_count);
59 }
60
61 DataTypes get_variadic_argument_types_impl() const override {
62 if constexpr (has_variadic_argument) return Impl::get_variadic_argument_types();
63 return {};
64 }

66private:
67 // handle result == DataTypeString
68 template <typename T, std::enable_if_t<std::is_same_v<T, DataTypeString>, T>* = nullptr>
69 Status execute_impl(Block& block, const ColumnNumbers& arguments, size_t result,
70 size_t input_rows_count) {
71 const ColumnPtr column = block.get_by_position(arguments[0]).column;
72 if constexpr (typeindex_is_int(Impl::TYPE_INDEX)) {
73 if (auto* col = check_and_get_column<ColumnVector<typename Impl::Type>>(column.get())) {
74 auto col_res = Impl::ReturnColumnType::create();
75 RETURN_IF_ERROR(Impl::vector(col->get_data(), col_res->get_chars(),
76 col_res->get_offsets()));
77 block.replace_by_position(result, std::move(col_res));
78 return Status::OK();
79 }
80 } else if constexpr (is_complex_v<typename Impl::Type>) {
81 if (const auto* col = check_and_get_column<ColumnComplexType<typename Impl::Type>>(
82 column.get())) {
83 auto col_res = Impl::ReturnColumnType::create();
84 RETURN_IF_ERROR(Impl::vector(col->get_data(), col_res->get_chars(),
85 col_res->get_offsets()));
86 block.replace_by_position(result, std::move(col_res));
87 return Status::OK();
88 }
89 }
90
91 return Status::RuntimeError("Illegal column {} of argument of function {}",
92 block.get_by_position(arguments[0]).column->get_name(),
93 get_name());
94 }

96 Status execute_impl(Block& block, const ColumnNumbers& arguments, size_t result,
97 size_t input_rows_count) {
98 const ColumnPtr column = block.get_by_position(arguments[0]).column;
99 if constexpr (Impl::TYPE_INDEX == TypeIndex::String) {

104 block.replace_by_position(result, std::move(col_res));
105 return Status::OK();
106 }
107 } else if constexpr (typeindex_is_int(Impl::TYPE_INDEX)) {
108 if (const auto* col =

110 auto col_res = Impl::ReturnColumnType::create();

130template <typename LeftDataType, typename RightDataType,
131 template <typename, typename> typename Impl, typename Name>
132class FunctionBinaryToType : public IFunction {
133public:
134 static constexpr auto name = Name::name;
135 static FunctionPtr create() { return std::make_shared<FunctionBinaryToType>(); }
136 String get_name() const override { return name; }
137 size_t get_number_of_arguments() const override { return 2; }
138 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
139 using ResultDataType = typename Impl<LeftDataType, RightDataType>::ResultDataType;
140 return std::make_shared<ResultDataType>();
141 }
142
143 bool use_default_implementation_for_constants() const override { return true; }
144
145 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
146 size_t result, size_t /*input_rows_count*/) override {
147 DCHECK_EQ(arguments.size(), 2);
148 const auto& [lcol, left_const] =
149 unpack_if_const(block.get_by_position(arguments[0]).column);
150 const auto& [rcol, right_const] =
151 unpack_if_const(block.get_by_position(arguments[1]).column);
152
153 using ResultDataType = typename Impl<LeftDataType, RightDataType>::ResultDataType;
154
155 using T0 = typename LeftDataType::FieldType;
156 using T1 = typename RightDataType::FieldType;
157 using ResultType = typename ResultDataType::FieldType;
158
159 using ColVecLeft =
160 std::conditional_t<is_complex_v<T0>, ColumnComplexType<T0>, ColumnVector<T0>>;
161 using ColVecRight =
162 std::conditional_t<is_complex_v<T1>, ColumnComplexType<T1>, ColumnVector<T1>>;
163
164 using ColVecResult =
165 std::conditional_t<is_complex_v<ResultType>, ColumnComplexType<ResultType>,
166 ColumnVector<ResultType>>;
167
168 typename ColVecResult::MutablePtr col_res = nullptr;
169
170 col_res = ColVecResult::create();
171 auto& vec_res = col_res->get_data();
172 vec_res.resize(block.rows());
173
174 if (auto col_left = check_and_get_column<ColVecLeft>(lcol.get())) {
175 if (auto col_right = check_and_get_column<ColVecRight>(rcol.get())) {
176 if (left_const) {
177 Impl<LeftDataType, RightDataType>::scalar_vector(
178 col_left->get_data()[0], col_right->get_data(), vec_res);
179 } else if (right_const) {
180 Impl<LeftDataType, RightDataType>::vector_scalar(
181 col_left->get_data(), col_right->get_data()[0], vec_res);
182 } else {
183 Impl<LeftDataType, RightDataType>::vector_vector(
184 col_left->get_data(), col_right->get_data(), vec_res);
185 }
186
187 block.replace_by_position(result, std::move(col_res));
188 return Status::OK();
189 }
190 }
191 return Status::RuntimeError("unimplements function {}", get_name());
192 }

244

246 if (auto col_right = check_and_get_column<ColVecRight>(rcol.get())) {
247 Impl<LeftDataType, RightDataType>::vector_vector(
248 col_left->get_chars(), col_left->get_offsets(), col_right->get_chars(),
249 col_right->get_offsets(), vec_res);
250 block.replace_by_position(result, std::move(col_res));
251 return Status::OK();
252 }
253 }
254 return Status::RuntimeError("unimplements function {}", get_name());
255 }

284class FunctionBinaryToNullType : public IFunction {
285public:
286 static constexpr auto name = Name::name;

288 String get_name() const override { return name; }
289 size_t get_number_of_arguments() const override { return 2; }
290 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {

295
296 bool use_default_implementation_for_constants() const override { return true; }
297
298 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
299 size_t result, size_t input_rows_count) override {
300 auto null_map = ColumnUInt8::create(input_rows_count, 0);
301 DCHECK_EQ(arguments.size(), 2);
302 ColumnPtr argument_columns[2];
303 for (int i = 0; i < 2; ++i) {
304 argument_columns[i] =
305 block.get_by_position(arguments[i]).column->convert_to_full_column_if_const();
306 if (auto* nullable = check_and_get_column<ColumnNullable>(*argument_columns[i])) {
307 // Danger: Here must dispose the null map data first! Because
308 // argument_columns[i]=nullable->get_nested_column_ptr(); will release the mem
309 // of column nullable mem of null map
310 VectorizedUtils::update_null_map(null_map->get_data(),
311 nullable->get_null_map_data());
312 argument_columns[i] = nullable->get_nested_column_ptr();
313 }
314 }
315
316 using ResultDataType = typename Impl<LeftDataType, RightDataType, ResultDateType,

318

321 using ResultType = typename ResultDataType::FieldType;
322
323 using ColVecLeft =
324 std::conditional_t<is_complex_v<T0>, ColumnComplexType<T0>, ColumnVector<T0>>;
325 using ColVecRight =
326 std::conditional_t<is_complex_v<T1>, ColumnComplexType<T1>, ColumnVector<T1>>;

328 using ColVecResult =
329 std::conditional_t<is_complex_v<ResultType>, ColumnComplexType<ResultType>,
330 ColumnVector<ResultType>>;
331
332 typename ColVecResult::MutablePtr col_res = nullptr;
333
334 col_res = ColVecResult::create();
335 auto& vec_res = col_res->get_data();
336 vec_res.resize(block.rows());
337
338 if (auto col_left = check_and_get_column<ColVecLeft>(argument_columns[0].get())) {
339 if (auto col_right = check_and_get_column<ColVecRight>(argument_columns[1].get())) {
340 Impl<LeftDataType, RightDataType, ResultDateType, ReturnType>::vector_vector(
341 col_left->get_data(), col_right->get_data(), vec_res, null_map->get_data());
342 block.get_by_position(result).column =
343 ColumnNullable::create(std::move(col_res), std::move(null_map));
344 return Status::OK();
345 }
346 }
347 return Status::RuntimeError("unimplements function {}", get_name());

369
370 // focus convert const to full column to simply execute logic

382 }
383 }
384
385 auto res = Impl::ColumnType::create();
386
387 auto specific_str_column = assert_cast<const ColumnString*>(argument_columns[0].get());
388 auto specific_char_column = assert_cast<const ColumnString*>(argument_columns[1].get());

392
393 auto& rdata = specific_char_column->get_chars();
394 auto& roffsets = specific_char_column->get_offsets();

401 null_map->get_data());
402 } else {
403 Impl::vector_vector(context, ldata, loffsets, rdata, roffsets, res->get_data(),
404 null_map->get_data());
405 }
406

408 ColumnNullable::create(std::move(res), std::move(null_map));
409 return Status::OK();
410 }
