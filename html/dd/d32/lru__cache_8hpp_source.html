26template <typename Key, typename Value>
27class LruCache {
28public:
29 typedef typename std::pair<Key, Value> KeyValuePair;
30 typedef typename std::list<KeyValuePair>::iterator ListIterator;
31
32 class Iterator {
33 public:
34 using iterator_category = std::input_iterator_tag;
35 using value_type = KeyValuePair;
36 using difference_type = ptrdiff_t;
37 using pointer = KeyValuePair*;
38 using reference = KeyValuePair&;
39
40 Iterator(typename std::unordered_map<Key, ListIterator>::iterator it) : _it(it) {}
41
42 Iterator& operator++() {
43 ++_it;
44 return *this;
45 }
46
47 bool operator==(const Iterator& rhs) const { return _it == rhs._it; }
48
49 bool operator!=(const Iterator& rhs) const { return _it != rhs._it; }
50
51 KeyValuePair* operator->() { return _it->second.operator->(); }
52
53 KeyValuePair& operator*() { return *_it->second; }
54
55 private:
56 typename std::unordered_map<Key, ListIterator>::iterator _it;
57 };
58
59 LruCache(size_t max_size) : _max_size(max_size) {}
60
61 void put(const Key& key, const Value& value) {
62 auto it = _cache_items_map.find(key);
63 if (it != _cache_items_map.end()) {
64 _cache_items_list.erase(it->second);
65 _cache_items_map.erase(it);
66 }
67
68 _cache_items_list.push_front(KeyValuePair(key, value));
69 _cache_items_map[key] = _cache_items_list.begin();
70
71 if (_cache_items_map.size() > _max_size) {
72 auto last = _cache_items_list.end();
73 last--;
74 _cache_items_map.erase(last->first);
75 _cache_items_list.pop_back();
76 }
77 }
78
79 void erase(const Key& key) {
80 auto it = _cache_items_map.find(key);
81 if (it != _cache_items_map.end()) {
82 _cache_items_list.erase(it->second);
83 _cache_items_map.erase(it);
84 }
85 }
86
87 // Must copy value, because value maybe released when caller used
88 bool get(const Key& key, Value* value) {
89 auto it = _cache_items_map.find(key);
90 if (it == _cache_items_map.end()) {
91 return false;
92 }
93 _cache_items_list.splice(_cache_items_list.begin(), _cache_items_list, it->second);
94 *value = it->second->second;
95 return true;
96 }
97
98 bool exists(const Key& key) const {
99 return _cache_items_map.find(key) != _cache_items_map.end();
100 }
101
102 size_t size() const { return _cache_items_map.size(); }
103
104 Iterator begin() { return Iterator(_cache_items_map.begin()); }
105
106 Iterator end() { return Iterator(_cache_items_map.end()); }
107
108private:
109 std::list<KeyValuePair> _cache_items_list;
110 std::unordered_map<Key, ListIterator> _cache_items_map;
111 size_t _max_size;
112};
