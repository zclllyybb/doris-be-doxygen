35// Fixed capacity FIFO queue, where both BlockingGet and BlockingPut operations block
36// if the queue is empty or full, respectively.
37template <typename T>
38class BlockingQueue {
39public:
40 BlockingQueue(size_t max_elements)
41 : _shutdown(false),
42 _max_elements(max_elements),
43 _total_get_wait_time(0),
44 _total_put_wait_time(0) {}
45
46 // Get an element from the queue, waiting indefinitely for one to become available.
47 // Returns false if we were shut down prior to getting the element, and there
48 // are no more elements available.
49 bool blocking_get(T* out) {
50 MonotonicStopWatch timer;
51 timer.start();
52 std::unique_lock<std::mutex> unique_lock(_lock);
53 _get_cv.wait(unique_lock, [this] { return _shutdown || !_list.empty(); });
54 _total_get_wait_time += timer.elapsed_time();
55
56 if (!_list.empty()) {
57 *out = _list.front();
58 _list.pop_front();
59 _put_cv.notify_one();
60 return true;
61 } else {
62 assert(_shutdown);
63 return false;
64 }
65 }
66
67 // Puts an element into the queue, waiting indefinitely until there is space.
68 // If the queue is shut down, returns false.
69 bool blocking_put(const T& val) {
70 MonotonicStopWatch timer;
71 timer.start();
72 std::unique_lock<std::mutex> unique_lock(_lock);
73 _put_cv.wait(unique_lock, [this] { return _shutdown || _list.size() < _max_elements; });
74 _total_put_wait_time += timer.elapsed_time();
75
76 if (_shutdown) {
77 return false;
78 }
79
80 _list.push_back(val);
81 _get_cv.notify_one();
82 return true;
83 }
84
85 // Shut down the queue. Wakes up all threads waiting on BlockingGet or BlockingPut.
86 void shutdown() {
87 {
88 std::lock_guard<std::mutex> guard(_lock);
89 _shutdown = true;
90 }
91
92 _get_cv.notify_all();
93 _put_cv.notify_all();
94 }
95
96 uint32_t get_size() const {
97 std::lock_guard<std::mutex> l(_lock);
98 return _list.size();
99 }
100
101 // Returns the total amount of time threads have blocked in BlockingGet.
102 uint64_t total_get_wait_time() const { return _total_get_wait_time; }
103
104 // Returns the total amount of time threads have blocked in BlockingPut.
105 uint64_t total_put_wait_time() const { return _total_put_wait_time; }
106
107private:
108 uint32_t SizeLocked(const std::unique_lock<std::mutex>& lock) const {
109 // The size of 'get_list_' is read racily to avoid getting 'get_lock_' in write path.
110 DCHECK(lock.owns_lock());
111 return _list.size();
112 }
113
114 bool _shutdown;
115 const int _max_elements;
116 std::condition_variable _get_cv; // 'get' callers wait on this
117 std::condition_variable _put_cv; // 'put' callers wait on this
118 // _lock guards access to _list, total_get_wait_time, and total_put_wait_time
119 mutable std::mutex _lock;
120 std::list<T> _list;
121 std::atomic<uint64_t> _total_get_wait_time;
122 std::atomic<uint64_t> _total_put_wait_time;
123};
