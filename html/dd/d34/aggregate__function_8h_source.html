213 void destroy_vec(AggregateDataPtr __restrict place,
214 const size_t num_rows) const noexcept override {
215 const size_t size_of_data_ = size_of_data();
216 for (size_t i = 0; i != num_rows; ++i) {
217 static_cast<const Derived*>(this)->destroy(place + size_of_data_ * i);
218 }
219 }

221 void add_batch(size_t batch_size, AggregateDataPtr* places, size_t place_offset,
222 const IColumn** columns, Arena* arena, bool agg_many) const override {
223 if constexpr (std::is_same_v<Derived, AggregateFunctionBitmapCount<false, ColumnBitmap>> ||
224 std::is_same_v<Derived, AggregateFunctionBitmapCount<true, ColumnBitmap>> ||
225 std::is_same_v<Derived,
226 AggregateFunctionBitmapOp<AggregateFunctionBitmapUnionOp>>) {
227 if (agg_many) {
228 phmap::flat_hash_map<AggregateDataPtr, std::vector<int>> place_rows;
229 for (int i = 0; i < batch_size; ++i) {
230 auto iter = place_rows.find(places[i] + place_offset);
231 if (iter == place_rows.end()) {
232 std::vector<int> rows;
233 rows.push_back(i);
234 place_rows.emplace(places[i] + place_offset, rows);
235 } else {
236 iter->second.push_back(i);
237 }
238 }
239 auto iter = place_rows.begin();
240 while (iter != place_rows.end()) {
241 static_cast<const Derived*>(this)->add_many(iter->first, columns, iter->second,
242 arena);
243 iter++;
244 }
245 return;
246 }
247 }
248
249 for (size_t i = 0; i < batch_size; ++i) {
250 static_cast<const Derived*>(this)->add(places[i] + place_offset, columns, i, arena);
251 }
252 }
253
254 void add_batch_selected(size_t batch_size, AggregateDataPtr* places, size_t place_offset,
255 const IColumn** columns, Arena* arena) const override {
256 for (size_t i = 0; i < batch_size; ++i) {
257 if (places[i]) {
258 static_cast<const Derived*>(this)->add(places[i] + place_offset, columns, i, arena);
259 }
260 }
261 }
262
263 void add_batch_single_place(size_t batch_size, AggregateDataPtr place, const IColumn** columns,
264 Arena* arena) const override {
265 for (size_t i = 0; i < batch_size; ++i) {
266 static_cast<const Derived*>(this)->add(place, columns, i, arena);
267 }
268 }
269 //now this is use for sum/count/avg/min/max win function, other win function should override this function in class
270 //stddev_pop/stddev_samp/variance_pop/variance_samp/hll_union_agg/group_concat
271 void add_range_single_place(int64_t partition_start, int64_t partition_end, int64_t frame_start,
272 int64_t frame_end, AggregateDataPtr place, const IColumn** columns,
273 Arena* arena) const override {
274 frame_start = std::max<int64_t>(frame_start, partition_start);
275 frame_end = std::min<int64_t>(frame_end, partition_end);
276 for (int64_t i = frame_start; i < frame_end; ++i) {
277 static_cast<const Derived*>(this)->add(place, columns, i, arena);
278 }
279 }

282 const IColumn** columns, Arena* arena, bool has_null) override {
283 for (size_t i = batch_begin; i <= batch_end; ++i) {
284 static_cast<const Derived*>(this)->add(place, columns, i, arena);
285 }
286 }

309 void streaming_agg_serialize(const IColumn** columns, BufferWritable& buf,
310 const size_t num_rows, Arena* arena) const override {
311 char place[size_of_data()];
312 for (size_t i = 0; i != num_rows; ++i) {
313 static_cast<const Derived*>(this)->create(place);
314 static_cast<const Derived*>(this)->add(place, columns, i, arena);
315 static_cast<const Derived*>(this)->serialize(place, buf);
316 buf.commit();
317 static_cast<const Derived*>(this)->destroy(place);
318 }

335 size_t num_rows) const override {

339 VectorBufferReader buffer_reader(column->get_data_at(i));
340 static_cast<const Derived*>(this)->create(place);
341 static_cast<const Derived*>(this)->deserialize(place, buffer_reader, arena);
342 }
343 }

373template <typename T, typename Derived>
374class IAggregateFunctionDataHelper : public IAggregateFunctionHelper<Derived> {
375protected:
376 using Data = T;
377
378 static Data& data(AggregateDataPtr __restrict place) { return *reinterpret_cast<Data*>(place); }
379 static const Data& data(ConstAggregateDataPtr __restrict place) {
380 return *reinterpret_cast<const Data*>(place);
381 }
382
383public:
384 IAggregateFunctionDataHelper(const DataTypes& argument_types_, const Array& parameters_)
385 : IAggregateFunctionHelper<Derived>(argument_types_, parameters_) {}
386
387 void create(AggregateDataPtr __restrict place) const override { new (place) Data; }
388
389 void destroy(AggregateDataPtr __restrict place) const noexcept override { data(place).~Data(); }
390
391 bool has_trivial_destructor() const override { return std::is_trivially_destructible_v<Data>; }
392
393 size_t size_of_data() const override { return sizeof(Data); }
394
396 size_t align_of_data() const override { return alignof(Data); }
397
398 void reset(AggregateDataPtr place) const override {}
399
400 void deserialize_and_merge(AggregateDataPtr __restrict place, BufferReadable& buf,
401 Arena* arena) const override {
402 char deserialized_data[size_of_data()];
403 AggregateDataPtr deserialized_place = (AggregateDataPtr)deserialized_data;
404
405 auto derived = static_cast<const Derived*>(this);
406 derived->create(deserialized_place);
407 derived->deserialize(deserialized_place, buf, arena);
408 derived->merge(place, deserialized_place, arena);
409 derived->destroy(deserialized_place);
410 }
411
412 void deserialize_and_merge_from_column(AggregateDataPtr __restrict place, const IColumn& column,
413 Arena* arena) const override {
414 size_t num_rows = column.size();
415 for (size_t i = 0; i != num_rows; ++i) {
416 VectorBufferReader buffer_reader(
417 (assert_cast<const ColumnString&>(column)).get_data_at(i));
418 deserialize_and_merge(place, buffer_reader, arena);
419 }
420 }
