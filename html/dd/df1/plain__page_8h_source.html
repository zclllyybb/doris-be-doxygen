31static const size_t PLAIN_PAGE_HEADER_SIZE = sizeof(uint32_t);
32
33template <FieldType Type>
34class PlainPageBuilder : public PageBuilder {
35public:
36 PlainPageBuilder(const PageBuilderOptions& options) : _options(options) {
37 // Reserve enough space for the page, plus a bit of slop since
38 // we often overrun the page by a few values.
39 _buffer.reserve(_options.data_page_size + 1024);
40 reset();
41 }
42
43 bool is_page_full() override { return _buffer.size() > _options.data_page_size; }
44
45 Status add(const uint8_t* vals, size_t* count) override {
46 if (is_page_full()) {
47 *count = 0;
48 return Status::OK();
49 }
50 size_t old_size = _buffer.size();
51 _buffer.resize(old_size + *count * SIZE_OF_TYPE);
52 memcpy(&_buffer[old_size], vals, *count * SIZE_OF_TYPE);
53 _count += *count;
54 return Status::OK();
55 }
56
57 OwnedSlice finish() override {
58 encode_fixed32_le((uint8_t*)&_buffer[0], _count);
59 if (_count > 0) {
60 _first_value.assign_copy(&_buffer[PLAIN_PAGE_HEADER_SIZE], SIZE_OF_TYPE);
61 _last_value.assign_copy(&_buffer[PLAIN_PAGE_HEADER_SIZE + (_count - 1) * SIZE_OF_TYPE],
62 SIZE_OF_TYPE);
63 }
64 return _buffer.build();
65 }
66
67 void reset() override {
68 _buffer.reserve(_options.data_page_size + 1024);
69 _count = 0;
70 _buffer.clear();
71 _buffer.resize(PLAIN_PAGE_HEADER_SIZE);
72 }
73
74 size_t count() const override { return _count; }
75
76 uint64_t size() const override { return _buffer.size(); }
77
78 Status get_first_value(void* value) const override {
79 if (_count == 0) {
80 return Status::NotFound("page is empty");
81 }
82 memcpy(value, _first_value.data(), SIZE_OF_TYPE);
83 return Status::OK();
84 }
85
86 Status get_last_value(void* value) const override {
87 if (_count == 0) {
88 return Status::NotFound("page is empty");
89 }
90 memcpy(value, _last_value.data(), SIZE_OF_TYPE);
91 return Status::OK();
92 }
93
94private:
95 faststring _buffer;
96 PageBuilderOptions _options;
97 size_t _count;
98 typedef typename TypeTraits<Type>::CppType CppType;
99 enum { SIZE_OF_TYPE = TypeTraits<Type>::size };
100 faststring _first_value;
101 faststring _last_value;
102};

104template <FieldType Type>
105class PlainPageDecoder : public PageDecoder {
106public:
107 PlainPageDecoder(Slice data, const PageDecoderOptions& options)
108 : _data(data), _options(options), _parsed(false), _num_elems(0), _cur_idx(0) {}
109
110 Status init() override {
111 CHECK(!_parsed);
112
113 if (_data.size < PLAIN_PAGE_HEADER_SIZE) {
114 return Status::InternalError(
115 "file corruption: not enough bytes for header in PlainPageDecoder ."
116 "invalid data size:{}, header size:{}",
117 _data.size, PLAIN_PAGE_HEADER_SIZE);
118 }
119
120 _num_elems = decode_fixed32_le((const uint8_t*)&_data[0]);
121
122 if (_data.size != PLAIN_PAGE_HEADER_SIZE + _num_elems * SIZE_OF_TYPE) {
123 return Status::InternalError("file corruption: unexpected data size.");
124 }
125
126 _parsed = true;
127
128 seek_to_position_in_page(0);
129 return Status::OK();
130 }
131
132 Status seek_to_position_in_page(size_t pos) override {
133 CHECK(_parsed) << "Must call init()";
134
135 if (PREDICT_FALSE(_num_elems == 0)) {
136 DCHECK_EQ(0, pos);
137 return Status::InternalError("invalid pos");
138 }
139
140 DCHECK_LE(pos, _num_elems);
141
142 _cur_idx = pos;
143 return Status::OK();
144 }
145
146 Status seek_at_or_after_value(const void* value, bool* exact_match) override {
147 DCHECK(_parsed) << "Must call init() firstly";
148
149 if (_num_elems == 0) {
150 return Status::NotFound("page is empty");
151 }
152
153 size_t left = 0;
154 size_t right = _num_elems;
155
156 const void* mid_value = nullptr;
157
158 // find the first value >= target. after loop,
159 // - left == index of first value >= target when found
160 // - left == _num_elems when not found (all values < target)
161 while (left < right) {
162 size_t mid = left + (right - left) / 2;
163 mid_value = &_data[PLAIN_PAGE_HEADER_SIZE + mid * SIZE_OF_TYPE];
164 if (TypeTraits<Type>::cmp(mid_value, value) < 0) {
165 left = mid + 1;
166 } else {
167 right = mid;
168 }
169 }
170 if (left >= _num_elems) {
171 return Status::NotFound("all value small than the value");
172 }
173 const void* find_value = &_data[PLAIN_PAGE_HEADER_SIZE + left * SIZE_OF_TYPE];
174 if (TypeTraits<Type>::cmp(find_value, value) == 0) {
175 *exact_match = true;
176 } else {
177 *exact_match = false;
178 }
179
180 _cur_idx = left;
181 return Status::OK();
182 }
183
184 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override {
185 return Status::NotSupported("plain page not implement vec op now");
186 }
187
188 size_t count() const override {
189 DCHECK(_parsed);
190 return _num_elems;
191 }
192
193 size_t current_index() const override {
194 DCHECK(_parsed);
195 return _cur_idx;
196 }
197
198private:
199 Slice _data;
200 PageDecoderOptions _options;
201 bool _parsed;
202 uint32_t _num_elems;
203 uint32_t _cur_idx;
204 typedef typename TypeTraits<Type>::CppType CppType;
205 enum { SIZE_OF_TYPE = TypeTraits<Type>::size };
206};
