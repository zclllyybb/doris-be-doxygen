119struct SubstringUtil {
120 static constexpr auto name = "substring";
121
122 static void substring_execute(Block& block, const ColumnNumbers& arguments, size_t result,
123 size_t input_rows_count) {
124 DCHECK_EQ(arguments.size(), 3);
125 auto null_map = ColumnUInt8::create(input_rows_count, 0);
126
127 ColumnPtr argument_columns[3];
128
129 for (int i = 0; i < 3; ++i) {
130 argument_columns[i] =
131 block.get_by_position(arguments[i]).column->convert_to_full_column_if_const();
132 if (auto* nullable = check_and_get_column<ColumnNullable>(*argument_columns[i])) {
133 // Danger: Here must dispose the null map data first! Because
134 // argument_columns[i]=nullable->get_nested_column_ptr(); will release the mem
135 // of column nullable mem of null map
136 VectorizedUtils::update_null_map(null_map->get_data(),
137 nullable->get_null_map_data());
138 argument_columns[i] = nullable->get_nested_column_ptr();
139 }
140 }
141
142 auto res = ColumnString::create();
143
144 auto specific_str_column = assert_cast<const ColumnString*>(argument_columns[0].get());
145 auto specific_start_column =
146 assert_cast<const ColumnVector<Int32>*>(argument_columns[1].get());
147 auto specific_len_column =
148 assert_cast<const ColumnVector<Int32>*>(argument_columns[2].get());
149
150 vector(specific_str_column->get_chars(), specific_str_column->get_offsets(),
151 specific_start_column->get_data(), specific_len_column->get_data(),
152 null_map->get_data(), res->get_chars(), res->get_offsets());
153
154 block.get_by_position(result).column =
155 ColumnNullable::create(std::move(res), std::move(null_map));
156 }
157
158private:
159 static void vector(const ColumnString::Chars& chars, const ColumnString::Offsets& offsets,
160 const PaddedPODArray<Int32>& start, const PaddedPODArray<Int32>& len,
161 NullMap& null_map, ColumnString::Chars& res_chars,
162 ColumnString::Offsets& res_offsets) {
163 int size = offsets.size();
164 res_offsets.resize(size);
165 res_chars.reserve(chars.size());
166
167 std::array<std::byte, 128 * 1024> buf;
168 PMR::monotonic_buffer_resource pool {buf.data(), buf.size()};
169 PMR::vector<size_t> index {&pool};
170
171 PMR::vector<std::pair<const unsigned char*, int>> strs(&pool);
172 strs.resize(size);
173 auto* __restrict data_ptr = chars.data();
174 auto* __restrict offset_ptr = offsets.data();
175 for (int i = 0; i < size; ++i) {
176 strs[i].first = data_ptr + offset_ptr[i - 1];
177 strs[i].second = offset_ptr[i] - offset_ptr[i - 1];
178 }
179
180 for (int i = 0; i < size; ++i) {
181 auto [raw_str, str_size] = strs[i];
182 // return empty string if start > src.length
183 if (start[i] > str_size || str_size == 0 || start[i] == 0 || len[i] <= 0) {
184 StringOP::push_empty_string(i, res_chars, res_offsets);
185 continue;
186 }
187 // reference to string_function.cpp: substring
188 size_t byte_pos = 0;
189 index.clear();
190 for (size_t j = 0, char_size = 0; j < str_size; j += char_size) {
191 char_size = UTF8_BYTE_LENGTH[(unsigned char)(raw_str)[j]];
192 index.push_back(j);
193 if (start[i] > 0 && index.size() > start[i] + len[i]) {
194 break;
195 }
196 }
197
198 int fixed_pos = start[i];
199 if (fixed_pos < -(int)index.size()) {
200 StringOP::push_empty_string(i, res_chars, res_offsets);
201 continue;
202 }
203 if (fixed_pos < 0) {
204 fixed_pos = index.size() + fixed_pos + 1;
205 }
206 if (fixed_pos > index.size()) {
207 StringOP::push_null_string(i, res_chars, res_offsets, null_map);
208 continue;
209 }
210
211 byte_pos = index[fixed_pos - 1];
212 int fixed_len = str_size - byte_pos;
213 if (fixed_pos + len[i] <= index.size()) {
214 fixed_len = index[fixed_pos + len[i] - 1] - byte_pos;
215 }
216
217 if (byte_pos <= str_size && fixed_len > 0) {
218 // return StringRef(str.data + byte_pos, fixed_len);
219 StringOP::push_value_string(
220 std::string_view {reinterpret_cast<const char*>(raw_str + byte_pos),
221 (size_t)fixed_len},
222 i, res_chars, res_offsets);
223 } else {
224 StringOP::push_empty_string(i, res_chars, res_offsets);
225 }
226 }
227 }

303class FunctionMask : public IFunction {
304public:
305 static constexpr auto name = "mask";
306 static constexpr unsigned char DEFAULT_UPPER_MASK = 'X';
307 static constexpr unsigned char DEFAULT_LOWER_MASK = 'x';
308 static constexpr unsigned char DEFAULT_NUMBER_MASK = 'n';
309 String get_name() const override { return name; }
310 static FunctionPtr create() { return std::make_shared<FunctionMask>(); }
311
312 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
313 return std::make_shared<DataTypeString>();
314 }
315
316 size_t get_number_of_arguments() const override { return 0; }
317
318 bool is_variadic() const override { return true; }
319
320 bool use_default_implementation_for_nulls() const override { return true; }
321 bool use_default_implementation_for_constants() const override { return false; }
322
323 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
324 size_t result, size_t input_rows_count) override {
325 DCHECK_GE(arguments.size(), 1);
326 DCHECK_LE(arguments.size(), 4);
327
328 char upper = DEFAULT_UPPER_MASK, lower = DEFAULT_LOWER_MASK, number = DEFAULT_NUMBER_MASK;
329
330 auto res = ColumnString::create();
331 const auto& source_column =
332 assert_cast<const ColumnString&>(*block.get_by_position(arguments[0]).column);
333
334 if (arguments.size() > 1) {
335 auto& col = *block.get_by_position(arguments[1]).column;
336 auto string_ref = col.get_data_at(0);
337 if (string_ref.size > 0) upper = *string_ref.data;
338 }
339
340 if (arguments.size() > 2) {
341 auto& col = *block.get_by_position(arguments[2]).column;
342 auto string_ref = col.get_data_at(0);
343 if (string_ref.size > 0) lower = *string_ref.data;
344 }
345
346 if (arguments.size() > 3) {
347 auto& col = *block.get_by_position(arguments[3]).column;
348 auto string_ref = col.get_data_at(0);
349 if (string_ref.size > 0) number = *string_ref.data;
350 }
351
352 if (arguments.size() > 4) {
353 return Status::InvalidArgument(
354 fmt::format("too many arguments for function {}", get_name()));
355 }
356
357 vector_mask(source_column, *res, upper, lower, number);
358
359 block.get_by_position(result).column = std::move(res);
360
361 return Status::OK();
362 }
363 friend class FunctionMaskPartial<true>;
364 friend class FunctionMaskPartial<false>;
365
366private:
367 static void vector_mask(const ColumnString& source, ColumnString& result, const char upper,
368 const char lower, const char number) {
369 result.get_chars().resize(source.get_chars().size());
370 result.get_offsets().resize(source.get_offsets().size());
371 memcpy_small_allow_read_write_overflow15(
372 result.get_offsets().data(), source.get_offsets().data(),
373 source.get_offsets().size() * sizeof(ColumnString::Offset));
374
375 const unsigned char* src = source.get_chars().data();
376 const size_t size = source.get_chars().size();
377 unsigned char* res = result.get_chars().data();
378 mask(src, size, upper, lower, number, res);
379 }
380
381 static void mask(const unsigned char* __restrict src, const size_t size,
382 const unsigned char upper, const unsigned char lower,
383 const unsigned char number, unsigned char* __restrict res) {
384 for (size_t i = 0; i != size; ++i) {
385 auto c = src[i];
386 if (c >= 'A' && c <= 'Z') {
387 res[i] = upper;
388 } else if (c >= 'a' && c <= 'z') {
389 res[i] = lower;
390 } else if (c >= '0' && c <= '9') {
391 res[i] = number;
392 } else {
393 res[i] = c;
394 }
395 }
396 }

399template <bool Reverse>
400class FunctionMaskPartial : public IFunction {
401public:
402 static constexpr auto name = Reverse ? "mask_last_n" : "mask_first_n";
403 String get_name() const override { return name; }
404 static FunctionPtr create() { return std::make_shared<FunctionMaskPartial>(); }
405
406 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
407 return std::make_shared<DataTypeString>();
408 }
409
410 size_t get_number_of_arguments() const override { return 0; }
411
412 bool is_variadic() const override { return true; }
413
414 bool use_default_implementation_for_nulls() const override { return true; }
415 bool use_default_implementation_for_constants() const override { return false; }
416
417 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
418 size_t result, size_t input_rows_count) override {
419 DCHECK_GE(arguments.size(), 1);
420 DCHECK_LE(arguments.size(), 2);
421
422 int n = -1;
423
424 auto res = ColumnString::create();
425 auto col = block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
426 const ColumnString& source_column = assert_cast<const ColumnString&>(*col);
427
428 if (arguments.size() == 2) {
429 auto& col = *block.get_by_position(arguments[1]).column;
430 n = col.get_int(0);
431 } else if (arguments.size() > 2) {
432 return Status::InvalidArgument(
433 fmt::format("too many arguments for function {}", get_name()));
434 }
435
436 if (n == -1) {
437 FunctionMask::vector_mask(source_column, *res, FunctionMask::DEFAULT_UPPER_MASK,
438 FunctionMask::DEFAULT_LOWER_MASK,
439 FunctionMask::DEFAULT_NUMBER_MASK);
440 } else if (n >= 0) {
441 vector(source_column, n, *res);
442 }
443
444 block.get_by_position(result).column = std::move(res);
445
446 return Status::OK();
447 }
448
449private:
450 static void vector(const ColumnString& src, int n, ColumnString& result) {
451 const auto num_rows = src.size();
452 auto* chars = src.get_chars().data();
453 auto* offsets = src.get_offsets().data();
454 result.get_chars().resize(src.get_chars().size());
455 result.get_offsets().resize(src.get_offsets().size());
456 memcpy_small_allow_read_write_overflow15(
457 result.get_offsets().data(), src.get_offsets().data(),
458 src.get_offsets().size() * sizeof(ColumnString::Offset));
459 auto* res = result.get_chars().data();
460
461 for (ssize_t i = 0; i != num_rows; ++i) {
462 auto offset = offsets[i - 1];
463 int len = offsets[i] - offset;
464 if constexpr (Reverse) {
465 auto start = std::max(len - n, 0);
466 if (start > 0) memcpy(&res[offset], &chars[offset], start);
467 offset += start;
468 } else {
469 if (n < len) memcpy(&res[offset + n], &chars[offset + n], len - n);
470 }
471
472 len = std::min(n, len);
473 FunctionMask::mask(&chars[offset], len, FunctionMask::DEFAULT_UPPER_MASK,
474 FunctionMask::DEFAULT_LOWER_MASK, FunctionMask::DEFAULT_NUMBER_MASK,
475 &res[offset]);
476 }
477 }

669 std::vector<const ColumnString::Offsets*> offsets_list(argument_size);
670 std::vector<const ColumnString::Chars*> chars_list(argument_size);

674 block.get_by_position(arguments[i]).column->convert_to_full_column_if_const();
675 auto col_str = assert_cast<const ColumnString*>(argument_columns[i].get());
676 offsets_list[i] = &col_str->get_offsets();
677 chars_list[i] = &col_str->get_chars();

685

690 for (size_t j = 0; j < input_rows_count; ++j) {
691 size_t append = (*offsets_list[i])[j] - (*offsets_list[i])[j - 1];

693 if (UNLIKELY(UINT_MAX - append < res_reserve_size)) {
694 return Status::BufferAllocFailed("concat output is too large to allocate");
695 }
696 res_reserve_size += append;
697 }

700 return Status::BufferAllocFailed("concat output is too large to allocate");
701 }
702 // for each terminal zero
703 res_reserve_size += input_rows_count;
704
705 res_data.resize(res_reserve_size);

708 int current_length = 0;
709 for (size_t j = 0; j < offsets_list.size(); ++j) {

717 &current_chars[current_offsets[i - 1]], size);
718 current_length += size;
719 }
720 }
721 res_offset[i] = res_offset[i - 1] + current_length;
722 }
723
724 block.get_by_position(result).column = std::move(res);
725 return Status::OK();

971 for (size_t i = 0; i < input_rows_count; ++i) {

973 auto& sep_chars = *chars_list[0];
974 auto& sep_nullmap = *null_list[0];

979 continue;

983 const char* sep_data = reinterpret_cast<const char*>(&sep_chars[sep_offsets[i - 1]]);

986 buffer.clear();
987 views.clear();
988
989 for (auto next_src_array_offset = src_array_offsets[i];
990 current_src_array_offset < next_src_array_offset; ++current_src_array_offset) {
991 const auto current_src_string_offset =
992 current_src_array_offset ? src_string_offsets[current_src_array_offset - 1]
993 : 0;

1003 }
1004
1005 fmt::format_to(buffer, "{}", fmt::join(views, sep));
