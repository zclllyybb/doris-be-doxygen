27// Stop watch for reporting elapsed time in nanosec based on CLOCK_MONOTONIC.
28// It is as fast as Rdtsc.
29// It is also accurate because it not affected by cpu frequency changes and
30// it is not affected by user setting the system clock.
31// CLOCK_MONOTONIC represents monotonic time since some unspecified starting point.
32// It is good for computing elapsed time.
33template <clockid_t Clock>
34class CustomStopWatch {
35public:
36 CustomStopWatch() {
37 _total_time = 0;
38 _running = false;
39 }
40
41 void start() {
42 if (!_running) {
43 clock_gettime(Clock, &_start);
44 _running = true;
45 }
46 }
47
48 void stop() {
49 if (_running) {
50 _total_time += elapsed_time();
51 _running = false;
52 }
53 }
54
55 // Restarts the timer. Returns the elapsed time until this point.
56 uint64_t reset() {
57 uint64_t ret = elapsed_time();
58
59 if (_running) {
60 clock_gettime(Clock, &_start);
61 }
62
63 return ret;
64 }
65
66 // Returns time in nanosecond.
67 uint64_t elapsed_time() const {
68 if (!_running) {
69 return _total_time;
70 }
71
72 timespec end;
73 clock_gettime(Clock, &end);
74 return (end.tv_sec - _start.tv_sec) * 1000L * 1000L * 1000L +
75 (end.tv_nsec - _start.tv_nsec);
76 }
77
78private:
79 timespec _start;
80 uint64_t _total_time; // in nanosec
81 bool _running;
82};
