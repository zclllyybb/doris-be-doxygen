33// Build index for primary key.
34// The primary key index is designed in a similar way like RocksDB
35// Partitioned Index, which is created in the segment file when MemTable flushes.
36// Index is stored in multiple pages to leverage the IndexedColumnWriter.
37//
38// NOTE: for now, it's only used when unique key merge-on-write property enabled.
39class PrimaryKeyIndexBuilder {
40public:
41 PrimaryKeyIndexBuilder(io::FileWriter* file_writer, size_t seq_col_length)
42 : _file_writer(file_writer), _num_rows(0), _size(0), _seq_col_length(seq_col_length) {}
43
44 Status init();
45
46 Status add_item(const Slice& key);
47
48 uint32_t num_rows() const { return _num_rows; }
49
50 uint64_t size() const { return _size; }
51
52 Slice min_key() { return Slice(_min_key.data(), _min_key.size() - _seq_col_length); }
53 Slice max_key() { return Slice(_max_key.data(), _max_key.size() - _seq_col_length); }
54
55 Status finalize(segment_v2::PrimaryKeyIndexMetaPB* meta);
56
57private:
58 io::FileWriter* _file_writer = nullptr;
59 uint32_t _num_rows;
60 uint64_t _size;
61 size_t _seq_col_length;
62
63 faststring _min_key;
64 faststring _max_key;
65 std::unique_ptr<segment_v2::IndexedColumnWriter> _primary_key_index_builder;
66 std::unique_ptr<segment_v2::BloomFilterIndexWriter> _bloom_filter_index_builder;
67};
68
69class PrimaryKeyIndexReader {
70public:
71 PrimaryKeyIndexReader() : _index_parsed(false), _bf_parsed(false) {}
72
73 Status parse_index(io::FileReaderSPtr file_reader,
74 const segment_v2::PrimaryKeyIndexMetaPB& meta);
75
76 Status parse_bf(io::FileReaderSPtr file_reader, const segment_v2::PrimaryKeyIndexMetaPB& meta);
77
78 Status new_iterator(std::unique_ptr<segment_v2::IndexedColumnIterator>* index_iterator) const {
79 DCHECK(_index_parsed);
80 index_iterator->reset(new segment_v2::IndexedColumnIterator(_index_reader.get()));
81 return Status::OK();
82 }
83
84 const TypeInfo* type_info() const {
85 DCHECK(_index_parsed);
86 return _index_reader->type_info();
87 }
88
89 // verify whether exist in BloomFilter
90 bool check_present(const Slice& key) {
91 DCHECK(_bf_parsed);
92 return _bf->test_bytes(key.data, key.size);
93 }
94
95 uint32_t num_rows() const {
96 DCHECK(_index_parsed);
97 return _index_reader->num_values();
98 }
99
100 uint64_t get_bf_memory_size() {
101 DCHECK(_bf_parsed);
102 return _bf->size();
103 }
104
105 uint64_t get_memory_size() {
106 DCHECK(_index_parsed);
107 return _index_reader->get_memory_size();
108 }
109
110private:
111 bool _index_parsed;
112 bool _bf_parsed;
113 std::unique_ptr<segment_v2::IndexedColumnReader> _index_reader;
114 std::unique_ptr<segment_v2::BloomFilter> _bf;
115};
