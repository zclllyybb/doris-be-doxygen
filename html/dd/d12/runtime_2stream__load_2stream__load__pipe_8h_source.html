33const size_t kMaxPipeBufferedBytes = 4 * 1024 * 1024;
34// StreamLoadPipe use to transfer data from producer to consumer
35// Data in pip is stored in chunks.
36
37class StreamLoadPipe : public MessageBodySink, public FileReader {
38public:
39 StreamLoadPipe(size_t max_buffered_bytes = kMaxPipeBufferedBytes,
40 size_t min_chunk_size = 64 * 1024, int64_t total_length = -1,
41 bool use_proto = false)
42 : _buffered_bytes(0),
43 _proto_buffered_bytes(0),
44 _max_buffered_bytes(max_buffered_bytes),
45 _min_chunk_size(min_chunk_size),
46 _total_length(total_length),
47 _use_proto(use_proto) {}
48
49 ~StreamLoadPipe() override {
50 SCOPED_SWITCH_THREAD_MEM_TRACKER_LIMITER(ExecEnv::GetInstance()->orphan_mem_tracker());
51 while (!_buf_queue.empty()) {
52 _buf_queue.pop_front();
53 }
54 }
55
56 Status open() override { return Status::OK(); }
57
58 Status append_and_flush(const char* data, size_t size, size_t proto_byte_size = 0) {
59 ByteBufferPtr buf = ByteBuffer::allocate(BitUtil::RoundUpToPowerOfTwo(size + 1));
60 buf->put_bytes(data, size);
61 buf->flip();
62 return _append(buf, proto_byte_size);
63 }
64
65 Status append(const char* data, size_t size) override {
66 size_t pos = 0;
67 if (_write_buf != nullptr) {
68 if (size < _write_buf->remaining()) {
69 _write_buf->put_bytes(data, size);
70 return Status::OK();
71 } else {
72 pos = _write_buf->remaining();
73 _write_buf->put_bytes(data, pos);
74
75 _write_buf->flip();
76 RETURN_IF_ERROR(_append(_write_buf));
77 _write_buf.reset();
78 }
79 }
80 // need to allocate a new chunk, min chunk is 64k
81 size_t chunk_size = std::max(_min_chunk_size, size - pos);
82 chunk_size = BitUtil::RoundUpToPowerOfTwo(chunk_size);
83 _write_buf = ByteBuffer::allocate(chunk_size);
84 _write_buf->put_bytes(data + pos, size - pos);
85 return Status::OK();
86 }
87
88 Status append(const ByteBufferPtr& buf) override {
89 if (_write_buf != nullptr) {
90 _write_buf->flip();
91 RETURN_IF_ERROR(_append(_write_buf));
92 _write_buf.reset();
93 }
94 return _append(buf);
95 }
96
97 // If _total_length == -1, this should be a Kafka routine load task,
98 // just get the next buffer directly from the buffer queue, because one buffer contains a complete piece of data.
99 // Otherwise, this should be a stream load task that needs to read the specified amount of data.
100 Status read_one_message(std::unique_ptr<uint8_t[]>* data, int64_t* length) override {
101 if (_total_length < -1) {
102 return Status::InternalError("invalid, _total_length is: {}", _total_length);
103 } else if (_total_length == 0) {
104 // no data
105 *length = 0;
106 return Status::OK();
107 }
108
109 if (_total_length == -1) {
110 return _read_next_buffer(data, length);
111 }
112
113 // _total_length > 0, read the entire data
114 data->reset(new uint8_t[_total_length]);
115 bool eof = false;
116 Status st = read(data->get(), _total_length, length, &eof);
117 if (eof) {
118 *length = 0;
119 }
120 return st;
121 }
122
123 Status read(uint8_t* data, int64_t data_size, int64_t* bytes_read, bool* eof) override {
124 SCOPED_SWITCH_THREAD_MEM_TRACKER_LIMITER(ExecEnv::GetInstance()->orphan_mem_tracker());
125 *bytes_read = 0;
126 while (*bytes_read < data_size) {
127 std::unique_lock<std::mutex> l(_lock);
128 while (!_cancelled && !_finished && _buf_queue.empty()) {
129 _get_cond.wait(l);
130 }
131 // cancelled
132 if (_cancelled) {
133 return Status::InternalError("cancelled: {}", _cancelled_reason);
134 }
135 // finished
136 if (_buf_queue.empty()) {
137 DCHECK(_finished);
138 data_size = *bytes_read;
139 *eof = (*bytes_read == 0);
140 return Status::OK();
141 }
142 auto buf = _buf_queue.front();
143 int64_t copy_size = std::min(data_size - *bytes_read, (int64_t)buf->remaining());
144 buf->get_bytes((char*)data + *bytes_read, copy_size);
145 *bytes_read += copy_size;
146 if (!buf->has_remaining()) {
147 _buf_queue.pop_front();
148 _buffered_bytes -= buf->limit;
149 _put_cond.notify_one();
150 }
151 }
152 DCHECK(*bytes_read == data_size)
153 << "*bytes_read=" << *bytes_read << ", data_size=" << data_size;
154 *eof = false;
155 return Status::OK();
156 }
157
158 Status readat(int64_t position, int64_t nbytes, int64_t* bytes_read, void* out) override {
159 return Status::InternalError("Not implemented");
160 }

162 int64_t size() override { return 0; }

164 Status seek(int64_t position) override { return Status::InternalError("Not implemented"); }

168 // called when consumer finished
169 void close() override { cancel("closed"); }
170
171 bool closed() override { return _cancelled; }
172
173 // called when producer finished
174 Status finish() override {
175 if (_write_buf != nullptr) {
176 _write_buf->flip();
177 _append(_write_buf);
178 _write_buf.reset();
179 }
180 {
181 std::lock_guard<std::mutex> l(_lock);
182 _finished = true;
183 }
184 _get_cond.notify_all();
185 return Status::OK();
186 }
187
188 // called when producer/consumer failed
189 void cancel(const std::string& reason) override {
190 {
191 std::lock_guard<std::mutex> l(_lock);
192 _cancelled = true;
193 _cancelled_reason = reason;
194 }
195 _get_cond.notify_all();
196 _put_cond.notify_all();
197 }
198
199private:
200 // read the next buffer from _buf_queue
201 Status _read_next_buffer(std::unique_ptr<uint8_t[]>* data, int64_t* length) {
202 std::unique_lock<std::mutex> l(_lock);
203 while (!_cancelled && !_finished && _buf_queue.empty()) {
204 _get_cond.wait(l);
205 }
206 // cancelled
207 if (_cancelled) {
208 return Status::InternalError("cancelled: {}", _cancelled_reason);
209 }
210 // finished
211 if (_buf_queue.empty()) {
212 DCHECK(_finished);
213 data->reset();
214 *length = 0;
215 return Status::OK();
216 }
217 auto buf = _buf_queue.front();
218 *length = buf->remaining();
219 data->reset(new uint8_t[*length]);
220 buf->get_bytes((char*)(data->get()), *length);
221 _buf_queue.pop_front();
222 _buffered_bytes -= buf->limit;
223 if (_use_proto) {
224 PDataRow** ptr = reinterpret_cast<PDataRow**>(data->get());
225 _proto_buffered_bytes -= (sizeof(PDataRow*) + (*ptr)->GetCachedSize());
226 }
227 _put_cond.notify_one();
228 return Status::OK();
229 }
230
231 Status _append(const ByteBufferPtr& buf, size_t proto_byte_size = 0) {
232 {
233 std::unique_lock<std::mutex> l(_lock);
234 // if _buf_queue is empty, we append this buf without size check
235 if (_use_proto) {
236 while (!_cancelled && !_buf_queue.empty() &&
237 (_proto_buffered_bytes + proto_byte_size > _max_buffered_bytes)) {
238 _put_cond.wait(l);
239 }
240 } else {
241 while (!_cancelled && !_buf_queue.empty() &&
242 _buffered_bytes + buf->remaining() > _max_buffered_bytes) {
243 _put_cond.wait(l);
244 }
245 }
246 if (_cancelled) {
247 return Status::InternalError("cancelled: {}", _cancelled_reason);
248 }
249 _buf_queue.push_back(buf);
250 if (_use_proto) {
251 _proto_buffered_bytes += proto_byte_size;
252 } else {
253 _buffered_bytes += buf->remaining();
254 }
255 }
256 _get_cond.notify_one();
257 return Status::OK();
258 }
259
260 // Blocking queue
261 std::mutex _lock;
262 size_t _buffered_bytes;
263 size_t _proto_buffered_bytes;
264 size_t _max_buffered_bytes;
265 size_t _min_chunk_size;
266 // The total amount of data expected to be read.
267 // In some scenarios, such as loading json format data through stream load,
268 // the data needs to be completely read before it can be parsed,
269 // so the total size of the data needs to be known.
270 // The default is -1, which means that the data arrives in a stream
271 // and the length is unknown.
272 // size_t is unsigned, so use int64_t
273 int64_t _total_length = -1;
274 bool _use_proto = false;
275 std::deque<ByteBufferPtr> _buf_queue;
276 std::condition_variable _put_cond;
277 std::condition_variable _get_cond;
278
279 ByteBufferPtr _write_buf;
280};
