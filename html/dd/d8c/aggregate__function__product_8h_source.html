104template <typename T, typename TResult, typename Data>
105class AggregateFunctionProduct final
106 : public IAggregateFunctionDataHelper<Data, AggregateFunctionProduct<T, TResult, Data>> {
107public:
108 using ResultDataType = std::conditional_t<IsDecimalNumber<T>, DataTypeDecimal<TResult>,
109 DataTypeNumber<TResult>>;
110 using ColVecType = std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<T>, ColumnVector<T>>;
111 using ColVecResult =
112 std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<TResult>, ColumnVector<TResult>>;
113
114 std::string get_name() const override { return "product"; }
115
116 AggregateFunctionProduct(const DataTypes& argument_types_)
117 : IAggregateFunctionDataHelper<Data, AggregateFunctionProduct<T, TResult, Data>>(
118 argument_types_),
119 scale(get_decimal_scale(*argument_types_[0])) {
120 if constexpr (IsDecimalNumber<T>) {
121 multiplier =
122 ResultDataType::get_scale_multiplier(get_decimal_scale(*argument_types_[0]));
123 }
124 }
125
126 DataTypePtr get_return_type() const override {
127 if constexpr (IsDecimalNumber<T>) {
128 return std::make_shared<ResultDataType>(ResultDataType::max_precision(), scale);
129 } else {
130 return std::make_shared<ResultDataType>();
131 }
132 }
133
134 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
135 Arena*) const override {
136 const auto& column = static_cast<const ColVecType&>(*columns[0]);
137 this->data(place).add(column.get_data()[row_num], multiplier);
138 }
139
140 void reset(AggregateDataPtr place) const override {
141 if constexpr (IsDecimalNumber<T>) {
142 this->data(place).reset(T(1 * multiplier));
143 } else {
144 this->data(place).reset(1);
145 }
146 }
147
148 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
149 Arena*) const override {
150 this->data(place).merge(this->data(rhs), multiplier);
151 }
152
153 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
154 this->data(place).write(buf);
155 }
156
157 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
158 Arena*) const override {
159 this->data(place).read(buf);
160 }
161
162 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
163 auto& column = static_cast<ColVecResult&>(to);
164 column.get_data().push_back(this->data(place).get());
165 }
166
167private:
168 UInt32 scale;
169 TResult multiplier;
170};
