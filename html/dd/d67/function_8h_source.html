32#define RETURN_REAL_TYPE_FOR_DATEV2_FUNCTION(TYPE) \
33 bool is_nullable = false; \
34 bool is_datev2 = false; \
35 for (auto it : arguments) { \
36 is_nullable = is_nullable || it.type->is_nullable(); \
37 is_datev2 = is_datev2 || WhichDataType(remove_nullable(it.type)).is_date_v2() || \
38 WhichDataType(remove_nullable(it.type)).is_date_time_v2(); \
39 } \
40 return is_nullable || !is_datev2 ? make_nullable(std::make_shared<TYPE>()) \
41 : std::make_shared<TYPE>();

303class FunctionBuilderImpl : public IFunctionBuilder {
304public:
305 FunctionBasePtr build(const ColumnsWithTypeAndName& arguments,
306 const DataTypePtr& return_type) const final {
307 const DataTypePtr& func_return_type = get_return_type(arguments);
308 DCHECK(return_type->equals(*func_return_type) ||
309 // For null constant argument, `get_return_type` would return
310 // Nullable<DataTypeNothing> when `use_default_implementation_for_nulls` is true.
311 (return_type->is_nullable() && func_return_type->is_nullable() &&
312 is_nothing(((DataTypeNullable*)func_return_type.get())->get_nested_type())) ||
313 is_date_or_datetime_or_decimal(return_type, func_return_type) ||
314 is_array_nested_type_date_or_datetime_or_decimal(return_type, func_return_type))
315 << " for function '" << this->get_name() << "' with " << return_type->get_name()
316 << " and " << func_return_type->get_name();
317
318 return build_impl(arguments, return_type);
319 }
