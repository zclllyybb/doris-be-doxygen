32#define RETURN_REAL_TYPE_FOR_DATEV2_FUNCTION(TYPE) \
33 bool is_nullable = false; \
34 bool is_datev2 = false; \
35 for (auto it : arguments) { \
36 is_nullable = is_nullable || it.type->is_nullable(); \
37 is_datev2 = is_datev2 || WhichDataType(remove_nullable(it.type)).is_date_v2() || \
38 WhichDataType(remove_nullable(it.type)).is_date_time_v2(); \
39 } \
40 return is_nullable || !is_datev2 ? make_nullable(std::make_shared<TYPE>()) \
41 : std::make_shared<TYPE>();

291class FunctionBuilderImpl : public IFunctionBuilder {
292public:
293 FunctionBasePtr build(const ColumnsWithTypeAndName& arguments,
294 const DataTypePtr& return_type) const final {
295 const DataTypePtr& func_return_type = get_return_type(arguments);
296 DCHECK(return_type->equals(*func_return_type) ||
297 // For null constant argument, `get_return_type` would return
298 // Nullable<DataTypeNothing> when `use_default_implementation_for_nulls` is true.
299 (return_type->is_nullable() && func_return_type->is_nullable() &&
300 is_nothing(((DataTypeNullable*)func_return_type.get())->get_nested_type())) ||
301 (is_date_or_datetime(
302 return_type->is_nullable()
303 ? ((DataTypeNullable*)return_type.get())->get_nested_type()
304 : return_type) &&
305 is_date_or_datetime(get_return_type(arguments)->is_nullable()
306 ? ((DataTypeNullable*)get_return_type(arguments).get())
307 ->get_nested_type()
308 : get_return_type(arguments))) ||
309 (is_date_v2_or_datetime_v2(
310 return_type->is_nullable()
311 ? ((DataTypeNullable*)return_type.get())->get_nested_type()
312 : return_type) &&
313 is_date_v2_or_datetime_v2(
314 get_return_type(arguments)->is_nullable()
315 ? ((DataTypeNullable*)get_return_type(arguments).get())
316 ->get_nested_type()
317 : get_return_type(arguments))) ||
318 // For some date functions such as str_to_date(string, string), return_type will
319 // be datetimev2 if users enable datev2 but get_return_type(arguments) will still
320 // return datetime. We need keep backward compatibility here.
321 (is_date_v2_or_datetime_v2(
322 return_type->is_nullable()
323 ? ((DataTypeNullable*)return_type.get())->get_nested_type()
324 : return_type) &&
325 is_date_or_datetime(get_return_type(arguments)->is_nullable()
326 ? ((DataTypeNullable*)get_return_type(arguments).get())
327 ->get_nested_type()
328 : get_return_type(arguments))) ||
329 (is_decimal(return_type->is_nullable()
330 ? ((DataTypeNullable*)return_type.get())->get_nested_type()
331 : return_type) &&
332 is_decimal(get_return_type(arguments)->is_nullable()
333 ? ((DataTypeNullable*)get_return_type(arguments).get())
334 ->get_nested_type()
335 : get_return_type(arguments))))
336 << " for function '" << this->get_name() << "' with " << return_type->get_name()
337 << " and " << func_return_type->get_name();
338
339 return build_impl(arguments, return_type);
340 }

343 bool is_deterministic_in_scope_of_query() const override { return true; }
344 bool is_stateful() const override { return false; }
345 bool is_variadic() const override { return false; }

348 void check_number_of_arguments(size_t number_of_arguments) const override;

350 DataTypePtr get_return_type(const ColumnsWithTypeAndName& arguments) const;

352 DataTypes get_variadic_argument_types() const override {
353 return get_variadic_argument_types_impl();
354 }

356 ColumnNumbers get_arguments_that_are_always_constant() const override { return {}; }
357 ColumnNumbers get_arguments_that_dont_imply_nullable_return_type(
358 size_t /*number_of_arguments*/) const override {
359 return {};
360 }
361
362protected:
364 virtual DataTypePtr get_return_type_impl(const ColumnsWithTypeAndName& arguments) const {
