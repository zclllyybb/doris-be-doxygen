41struct ZoneMap {
42 // min value of zone
43 char* min_value = nullptr;
44 // max value of zone
45 char* max_value = nullptr;
46
47 // if both has_null and has_not_null is false, means no rows.
48 // if has_null is true and has_not_null is false, means all rows is null.
49 // if has_null is false and has_not_null is true, means all rows is not null.
50 // if has_null is true and has_not_null is true, means some rows is null and others are not.
51 // has_null means whether zone has null value
52 bool has_null = false;
53 // has_not_null means whether zone has none-null value
54 bool has_not_null = false;
55
56 bool pass_all = false;
57
58 void to_proto(ZoneMapPB* dst, Field* field) const {
59 if (pass_all) {
60 dst->set_min("");
61 dst->set_max("");
62 } else {
63 dst->set_min(field->to_string(min_value));
64 dst->set_max(field->to_string(max_value));
65 }
66 dst->set_has_null(has_null);
67 dst->set_has_not_null(has_not_null);
68 dst->set_pass_all(pass_all);
69 }
70};

97// Zone map index is represented by an IndexedColumn with ordinal index.
98// The IndexedColumn stores serialized ZoneMapPB for each data page.
99// It also create and store the segment-level zone map in the index meta so that
100// reader can prune an entire segment without reading pages.
101template <PrimitiveType Type>
102class TypedZoneMapIndexWriter final : public ZoneMapIndexWriter {
103public:
104 explicit TypedZoneMapIndexWriter(Field* field);
105
106 void add_values(const void* values, size_t count) override;
107
108 void add_nulls(uint32_t count) override { _page_zone_map.has_null = true; }
109
110 // mark the end of one data page so that we can finalize the corresponding zone map
111 Status flush() override;
112
113 Status finish(io::FileWriter* file_writer, ColumnIndexMetaPB* index_meta) override;
114
115 void moidfy_index_before_flush(ZoneMap& zone_map) override;
116
117 uint64_t size() const override { return _estimated_size; }
118
119 void reset_page_zone_map() override;
120 void reset_segment_zone_map() override;
121
122private:
123 void _reset_zone_map(ZoneMap* zone_map) {
124 // we should allocate max varchar length and set to max for min value
125 _field->set_to_zone_map_max(zone_map->min_value);
126 _field->set_to_zone_map_min(zone_map->max_value);
127 zone_map->has_null = false;
128 zone_map->has_not_null = false;
129 zone_map->pass_all = false;
130 }
131
132 Field* _field;
133 // memory will be managed by MemPool
134 ZoneMap _page_zone_map;
135 ZoneMap _segment_zone_map;
136 // TODO(zc): we should replace this memory pool later, we only allocate min/max
137 // for field. But MemPool allocate 4KB least, it will a waste for most cases.
138 MemPool _pool;
139
140 // serialized ZoneMapPB for each data page
141 std::vector<std::string> _values;
142 uint64_t _estimated_size = 0;
143};
