141inline uint8_t* encode_varint64(uint8_t* dst, uint64_t v) {
142 static const unsigned int B = 128;
143 while (v >= B) {
144 // Fetch low seven bits from current v, and the eight bit is marked as compression mark.
145 // v | B is optimised from (v & (B-1)) | B, because result is assigned to uint8_t and other bits
146 // is cleared by implicit conversion.
147 *(dst++) = v | B;
148 v >>= 7;
149 }
150 *(dst++) = static_cast<unsigned char>(v);
151 return dst;
152}

157inline const uint8_t* decode_varint32_ptr(const uint8_t* ptr, const uint8_t* limit,
158 uint32_t* value) {
159 if (ptr < limit) {
160 uint32_t result = *ptr;
161 if ((result & 128) == 0) {
162 *value = result;
163 return ptr + 1;
164 }
165 }
166 return decode_varint32_ptr_fallback(ptr, limit, value);
167}

199// parse a varint32 from the start of `input` into `val`.
200// on success, return true and advance `input` past the parsed value.
201// on failure, return false and `input` is not modified.
202inline bool get_varint32(Slice* input, uint32_t* val) {
203 const uint8_t* p = (const uint8_t*)input->data;
204 const uint8_t* limit = p + input->size;
205 const uint8_t* q = decode_varint32_ptr(p, limit, val);
206 if (q == nullptr) {
207 return false;
208 } else {
209 *input = Slice(q, limit - q);
210 return true;
211 }
212}

229// parse a length-prefixed-slice from the start of `input` into `val`.
230// on success, return true and advance `input` past the parsed value.
231// on failure, return false and `input` may or may not be modified.
232inline bool get_length_prefixed_slice(Slice* input, Slice* val) {
233 uint32_t len;
234 if (get_varint32(input, &len) && input->get_size() >= len) {
235 *val = Slice(input->get_data(), len);
236 input->remove_prefix(len);
237 return true;
238 } else {
239 return false;
240 }
