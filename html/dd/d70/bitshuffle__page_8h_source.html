50// BitshufflePageBuilder bitshuffles and compresses the bits of fixed
51// size type blocks with lz4.
52//
53// The page format is as follows:
54//
55// 1. Header: (16 bytes total)
56//
57// <num_elements> [32-bit]
58// The number of elements encoded in the page.
59//
60// <compressed_size> [32-bit]
61// The post-compression size of the page, including this header.
62//
63// <padded_num_elements> [32-bit]
64// Padding is needed to meet the requirements of the bitshuffle
65// library such that the input/output is a multiple of 8. Some
66// ignored elements are appended to the end of the page if necessary
67// to meet this requirement.
68//
69// This header field is the post-padding element count.
70//
71// <elem_size_bytes> [32-bit]
72// The size of the elements, in bytes, as actually encoded. In the
73// case that all of the data in a page can fit into a smaller
74// integer type, then we may choose to encode that smaller type
75// to save CPU costs.
76//
77// This is currently only implemented in the UINT32 page type.
78//
79// NOTE: all on-disk ints are encoded little-endian
80//
81// 2. Element data
82//
83// The header is followed by the bitshuffle-compressed element data.
84//
85template <FieldType Type>
86class BitshufflePageBuilder : public PageBuilder {
87public:
88 BitshufflePageBuilder(const PageBuilderOptions& options)
89 : _options(options), _count(0), _remain_element_capacity(0), _finished(false) {
90 reset();
91 }

93 bool is_page_full() override { return _remain_element_capacity == 0; }

103 template <bool single>
104 inline Status add_internal(const uint8_t* vals, size_t* count) {
105 DCHECK(!_finished);
106 if (_remain_element_capacity <= 0) {
107 *count = 0;
108 return Status::OK();
109 }
110 int to_add = std::min<int>(_remain_element_capacity, *count);
111 int to_add_size = to_add * SIZE_OF_TYPE;
112 size_t orig_size = _data.size();
113 _data.resize(orig_size + to_add_size);
114 _count += to_add;
115 _remain_element_capacity -= to_add;
116 // return added number through count
117 *count = to_add;
118 if constexpr (single) {
119 if constexpr (SIZE_OF_TYPE == 1) {
120 *reinterpret_cast<uint8_t*>(&_data[orig_size]) = *vals;
121 return Status::OK();
122 } else if constexpr (SIZE_OF_TYPE == 2) {
123 *reinterpret_cast<uint16_t*>(&_data[orig_size]) =
124 *reinterpret_cast<const uint16_t*>(vals);
125 return Status::OK();
126 } else if constexpr (SIZE_OF_TYPE == 4) {
127 *reinterpret_cast<uint32_t*>(&_data[orig_size]) =
128 *reinterpret_cast<const uint32_t*>(vals);
129 return Status::OK();
130 } else if constexpr (SIZE_OF_TYPE == 8) {
131 *reinterpret_cast<uint64_t*>(&_data[orig_size]) =
132 *reinterpret_cast<const uint64_t*>(vals);
133 return Status::OK();
134 }
135 }
136 // when single is true and SIZE_OF_TYPE > 8 or single is false
137 memcpy(&_data[orig_size], vals, to_add_size);
138 return Status::OK();
139 }
140
141 OwnedSlice finish() override {
142 if (_count > 0) {
143 _first_value = cell(0);
144 _last_value = cell(_count - 1);
145 }
146 return _finish(SIZE_OF_TYPE);
147 }
148
149 void reset() override {
150 auto block_size = _options.data_page_size;
151 _count = 0;
152 _data.clear();
153 _data.reserve(block_size);
154 DCHECK_EQ(reinterpret_cast<uintptr_t>(_data.data()) & (alignof(CppType) - 1), 0)
155 << "buffer must be naturally-aligned";
156 _buffer.clear();
157 _buffer.resize(BITSHUFFLE_PAGE_HEADER_SIZE);
158 _finished = false;
159 _remain_element_capacity = block_size / SIZE_OF_TYPE;
160 }
161
162 size_t count() const override { return _count; }
163
164 uint64_t size() const override { return _buffer.size(); }
165
166 Status get_first_value(void* value) const override {
167 DCHECK(_finished);
168 if (_count == 0) {
169 return Status::NotFound("page is empty");
170 }
171 memcpy(value, &_first_value, SIZE_OF_TYPE);
172 return Status::OK();
173 }
174 Status get_last_value(void* value) const override {
175 DCHECK(_finished);
176 if (_count == 0) {
177 return Status::NotFound("page is empty");
178 }
179 memcpy(value, &_last_value, SIZE_OF_TYPE);
180 return Status::OK();
181 }
182
183private:
184 OwnedSlice _finish(int final_size_of_type) {
185 _data.resize(final_size_of_type * _count);
186
187 // Do padding so that the input num of element is multiple of 8.
188 int num_elems_after_padding = ALIGN_UP(_count, 8);
189 int padding_elems = num_elems_after_padding - _count;
190 int padding_bytes = padding_elems * final_size_of_type;
191 for (int i = 0; i < padding_bytes; i++) {
192 _data.push_back(0);
193 }
194
195 // reserve enough place for compression
196 _buffer.resize(
197 BITSHUFFLE_PAGE_HEADER_SIZE +
198 bitshuffle::compress_lz4_bound(num_elems_after_padding, final_size_of_type, 0));
199
200 int64_t bytes =
201 bitshuffle::compress_lz4(_data.data(), &_buffer[BITSHUFFLE_PAGE_HEADER_SIZE],
202 num_elems_after_padding, final_size_of_type, 0);
203 if (PREDICT_FALSE(bytes < 0)) {
204 // This means the bitshuffle function fails.
205 // Ideally, this should not happen.
206 warn_with_bitshuffle_error(bytes);
207 // It does not matter what will be returned here,
208 // since we have logged fatal in warn_with_bitshuffle_error().
209 return OwnedSlice();
210 }
211 // update header
212 encode_fixed32_le(&_buffer[0], _count);
213 encode_fixed32_le(&_buffer[4], BITSHUFFLE_PAGE_HEADER_SIZE + bytes);
214 encode_fixed32_le(&_buffer[8], num_elems_after_padding);
215 encode_fixed32_le(&_buffer[12], final_size_of_type);
216 _finished = true;
217 // before build(), update buffer length to the actual compressed size
218 _buffer.resize(BITSHUFFLE_PAGE_HEADER_SIZE + bytes);
219 return _buffer.build();
220 }
221
222 using CppType = typename TypeTraits<Type>::CppType;
223
224 CppType cell(int idx) const {
225 DCHECK_GE(idx, 0);
226 CppType ret;
227 memcpy(&ret, &_data[idx * SIZE_OF_TYPE], SIZE_OF_TYPE);
228 return ret;
229 }

232 PageBuilderOptions _options;

234 int _remain_element_capacity;
235 bool _finished;
236 faststring _data;
237 faststring _buffer;
238 CppType _first_value;
239 CppType _last_value;

242inline Status parse_bit_shuffle_header(const Slice& data, size_t& num_elements,
243 size_t& compressed_size, size_t& num_element_after_padding,
244 int& size_of_element) {
245 if (data.size < BITSHUFFLE_PAGE_HEADER_SIZE) {
246 return Status::InternalError("file corruption: invalid data size:{}, header size:{}",
247 data.size, BITSHUFFLE_PAGE_HEADER_SIZE);
248 }
249
250 num_elements = decode_fixed32_le((const uint8_t*)&data[0]);
251 compressed_size = decode_fixed32_le((const uint8_t*)&data[4]);
252 num_element_after_padding = decode_fixed32_le((const uint8_t*)&data[8]);
253 if (num_element_after_padding != ALIGN_UP(num_elements, 8)) {
254 return Status::InternalError(
255 "num of element information corrupted,"
256 " _num_element_after_padding:{}, _num_elements:{}",
257 num_element_after_padding, num_elements);
258 }
259 size_of_element = decode_fixed32_le((const uint8_t*)&data[12]);
260 switch (size_of_element) {
261 case 1:
262 case 2:
263 case 3:
264 case 4:
265 case 8:
266 case 12:
267 case 16:
268 break;
269 default:
270 return Status::InternalError("invalid size_of_elem:{}", size_of_element);
271 }
272 return Status::OK();
273}

275template <FieldType Type>
276class BitShufflePageDecoder : public PageDecoder {
277public:
278 BitShufflePageDecoder(Slice data, const PageDecoderOptions& options)
279 : _data(data),
280 _options(options),
281 _parsed(false),
282 _num_elements(0),
283 _num_element_after_padding(0),
284 _size_of_element(0),
285 _cur_index(0) {}
286
287 Status init() override {
288 CHECK(!_parsed);
289 size_t unused;
290 RETURN_IF_ERROR(parse_bit_shuffle_header(_data, _num_elements, unused,
291 _num_element_after_padding, _size_of_element));
292
293 if (_data.size !=
294 _num_element_after_padding * _size_of_element + BITSHUFFLE_PAGE_HEADER_SIZE) {
295 std::stringstream ss;
296 ss << "Size information unmatched, _data.size:" << _data.size
297 << ", _num_elements:" << _num_elements << ", expected size is "
298 << _num_element_after_padding * _size_of_element + BITSHUFFLE_PAGE_HEADER_SIZE;
299 return Status::InternalError(ss.str());
300 }
301
302 // Currently, only the UINT32 block encoder supports expanding size:
303 if (UNLIKELY(Type != OLAP_FIELD_TYPE_UNSIGNED_INT && _size_of_element != SIZE_OF_TYPE)) {
304 return Status::InternalError(
305 "invalid size info. size of element:{}, SIZE_OF_TYPE:{}, type:{}",
306 _size_of_element, SIZE_OF_TYPE, Type);
307 }
308 if (UNLIKELY(_size_of_element > SIZE_OF_TYPE)) {
309 return Status::InternalError("invalid size info. size of element:{}, SIZE_OF_TYPE:{}",
310 _size_of_element, SIZE_OF_TYPE);
311 }
312 _parsed = true;
313 return Status::OK();
314 }
315
316 Status seek_to_position_in_page(size_t pos) override {
317 DCHECK(_parsed) << "Must call init()";
318 if (PREDICT_FALSE(_num_elements == 0)) {
319 DCHECK_EQ(0, pos);
320 return Status::InvalidArgument("invalid pos");
321 }
322
323 DCHECK_LE(pos, _num_elements);
324 _cur_index = pos;
325 return Status::OK();
326 }
327
328 Status seek_at_or_after_value(const void* value, bool* exact_match) override {
329 DCHECK(_parsed) << "Must call init() firstly";
330
331 if (_num_elements == 0) {
332 return Status::NotFound("page is empty");
333 }
334
335 size_t left = 0;
336 size_t right = _num_elements;
337
338 void* mid_value = nullptr;
339
340 // find the first value >= target. after loop,
341 // - left == index of first value >= target when found
342 // - left == _num_elements when not found (all values < target)
343 while (left < right) {
344 size_t mid = left + (right - left) / 2;
345 mid_value = get_data(mid);
346 if (TypeTraits<Type>::cmp(mid_value, value) < 0) {
347 left = mid + 1;
348 } else {
349 right = mid;
350 }
351 }
352 if (left >= _num_elements) {
353 return Status::NotFound("all value small than the value");
354 }
355 void* find_value = get_data(left);
356 if (TypeTraits<Type>::cmp(find_value, value) == 0) {
357 *exact_match = true;
358 } else {
359 *exact_match = false;
360 }
361
362 _cur_index = left;
363 return Status::OK();
364 }

366 template <bool forward_index = true>
367 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) {
368 DCHECK(_parsed);
369 if (PREDICT_FALSE(*n == 0 || _cur_index >= _num_elements)) {
370 *n = 0;
371 return Status::OK();
372 }
373
374 size_t max_fetch = std::min(*n, static_cast<size_t>(_num_elements - _cur_index));
375
376 dst->insert_many_fix_len_data(get_data(_cur_index), max_fetch);
377 *n = max_fetch;
378 if constexpr (forward_index) {
379 _cur_index += max_fetch;
380 }
381
382 return Status::OK();
383 }
384
385 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override {
386 return next_batch<>(n, dst);
387 }
388
389 Status read_by_rowids(const rowid_t* rowids, ordinal_t page_first_ordinal, size_t* n,
390 vectorized::MutableColumnPtr& dst) override {
391 DCHECK(_parsed);
392 if (PREDICT_FALSE(*n == 0)) {
393 *n = 0;
394 return Status::OK();
395 }
396
397 auto total = *n;
398 auto read_count = 0;
399 CppType data[total];
400 for (size_t i = 0; i < total; ++i) {
401 ordinal_t ord = rowids[i] - page_first_ordinal;
402 if (UNLIKELY(ord >= _num_elements)) {
403 break;
404 }
405
406 data[read_count++] = *reinterpret_cast<CppType*>(get_data(ord));
407 }
408
409 if (LIKELY(read_count > 0)) dst->insert_many_fix_len_data((const char*)data, read_count);
410
411 *n = read_count;
412 return Status::OK();
413 }
414
415 Status peek_next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override {
416 return next_batch<false>(n, dst);
417 }
418
419 size_t count() const override { return _num_elements; }
420
421 size_t current_index() const override { return _cur_index; }
422
423 char* get_data(size_t index) const {
424 return &_data.data[BITSHUFFLE_PAGE_HEADER_SIZE + index * SIZE_OF_TYPE];
425 }
426
427private:
428 void _copy_next_values(size_t n, void* data) {
429 memcpy(data, get_data(_cur_index), n * SIZE_OF_TYPE);
430 }

443 size_t _cur_index;
444
445 friend class BinaryDictPageDecoder;
446};
