68inline std::vector<std::string> split(const std::string& s, const std::string& delim) {
69 std::vector<std::string> out;
70 size_t pos {};
71
72 for (size_t find = 0; (find = s.find(delim, pos)) != std::string::npos;
73 pos = find + delim.size()) {
74 out.emplace_back(s.data() + pos, s.data() + find);
75 }
76
77 out.emplace_back(s.data() + pos, s.data() + s.size());
78 return out;
79}

101struct StringCaseEqual {

103 bool operator()(const std::string& lhs, const std::string& rhs) const {
104 if (lhs.size() != rhs.size()) {
105 return false;
106 }

111struct StringCaseLess {
112public:
113 bool operator()(const std::string& lhs, const std::string& rhs) const {
114 size_t common_size = std::min(lhs.size(), rhs.size());
115 auto cmp = strncasecmp(lhs.c_str(), rhs.c_str(), common_size);
116 if (cmp == 0) {
117 return lhs.size() < rhs.size();
118 }
119 return cmp < 0;
120 }
121};
