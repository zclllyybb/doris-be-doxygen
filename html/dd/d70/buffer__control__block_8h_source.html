60// buffer used for result customer and producer
61class BufferControlBlock {
62public:
63 BufferControlBlock(const TUniqueId& id, int buffer_size);
64 virtual ~BufferControlBlock();
65
66 Status init();
67 virtual bool can_sink(); // 只有一个fragment写入，因此can_sink返回true，则一定可以执行sink
68 Status add_batch(std::unique_ptr<TFetchDataResult>& result);
69
70 // get result from batch, use timeout?
71 Status get_batch(TFetchDataResult* result);
72
73 void get_batch(GetResultBatchCtx* ctx);
74
75 // close buffer block, set _status to exec_status and set _is_close to true;
76 // called because data has been read or error happened.
77 Status close(Status exec_status);
78 // this is called by RPC, called from coordinator
79 Status cancel();
80
81 const TUniqueId& fragment_id() const { return _fragment_id; }
82
83 void set_query_statistics(std::shared_ptr<QueryStatistics> statistics) {
84 _query_statistics = statistics;
85 }
86
87 void update_num_written_rows(int64_t num_rows) {
88 // _query_statistics may be null when the result sink init failed
89 // or some other failure.
90 // and the number of written rows is only needed when all things go well.
91 if (_query_statistics != nullptr) {
92 _query_statistics->set_returned_rows(num_rows);
93 }
94 }
95
96 // TODO: The value of query peak mem usage in fe.audit.log comes from a random BE,
97 // not the BE with the largest peak mem usage
98 void update_max_peak_memory_bytes() {
99 if (_query_statistics != nullptr) {
100 int64_t max_peak_memory_bytes = _query_statistics->calculate_max_peak_memory_bytes();
101 _query_statistics->set_max_peak_memory_bytes(max_peak_memory_bytes);
102 }
103 }
104
105protected:
106 virtual bool _get_batch_queue_empty() { return _batch_queue.empty(); }
107 virtual void _update_batch_queue_empty() {}
108
109 using ResultQueue = std::list<std::unique_ptr<TFetchDataResult>>;
110
111 // result's query id
112 TUniqueId _fragment_id;
113 bool _is_close;
114 std::atomic_bool _is_cancelled;
115 Status _status;
116 std::atomic_int _buffer_rows;
117 const int _buffer_limit;
118 int64_t _packet_num;
119
120 // blocking queue for batch
121 ResultQueue _batch_queue;
122 // protects all subsequent data in this block
123 std::mutex _lock;
124 // signal arrival of new batch or the eos/cancelled condition
125 std::condition_variable _data_arrival;
126 // signal removal of data by stream consumer
127 std::condition_variable _data_removal;
128
129 std::deque<GetResultBatchCtx*> _waiting_rpc;
130
131 // It is shared with PlanFragmentExecutor and will be called in two different
132 // threads. But their calls are all at different time, there is no problem of
133 // multithreading access.
134 std::shared_ptr<QueryStatistics> _query_statistics;
135};

144 }
145
146private:

148 void _update_batch_queue_empty() override { _batch_queue_empty = _batch_queue.empty(); }
149
150 std::atomic_bool _batch_queue_empty = false;
151};
