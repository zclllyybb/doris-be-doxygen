85template <size_t ELEMENT_SIZE, size_t initial_bytes, typename TAllocator, size_t pad_right_,
86 size_t pad_left_>
87class PODArrayBase : private boost::noncopyable,
88 private TAllocator
89{
90protected:
92 static constexpr size_t pad_right = integerRoundUp(pad_right_, ELEMENT_SIZE);
94 static constexpr size_t pad_left = integerRoundUp(integerRoundUp(pad_left_, ELEMENT_SIZE), 16);
96 static constexpr char* null =
97 pad_left ? const_cast<char*>(empty_pod_array) + EmptyPODArraySize : nullptr;
98
99 static_assert(pad_left <= EmptyPODArraySize &&
100 "Left Padding exceeds EmptyPODArraySize. Is the element size too large?");
101
102 char* c_start = null;
103 char* c_end = null;
104 char* c_end_of_storage = null;
105 char* c_end_peak = null;
106
108 static size_t byte_size(size_t num_elements) { return num_elements * ELEMENT_SIZE; }
109
111 static size_t minimum_memory_for_elements(size_t num_elements) {
112 return byte_size(num_elements) + pad_right + pad_left;
113 }
114
115 inline void reset_peak() {
116 if (UNLIKELY(c_end - c_end_peak > 65536)) {
117 THREAD_MEM_TRACKER_TRANSFER_FROM(c_end - c_end_peak,
118 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
119 c_end_peak = c_end;
120 }
121 }
122
123 void alloc_for_num_elements(size_t num_elements) {
124 alloc(round_up_to_power_of_two_or_zero(minimum_memory_for_elements(num_elements)));
125 }
126
127 template <typename... TAllocatorParams>
128 void alloc(size_t bytes, TAllocatorParams&&... allocator_params) {
129 THREAD_MEM_TRACKER_TRANSFER_TO(bytes - pad_right - pad_left,
130 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
131 c_start = c_end = c_end_peak =
132 reinterpret_cast<char*>(TAllocator::alloc(
133 bytes, std::forward<TAllocatorParams>(allocator_params)...)) +
134 pad_left;
135 c_end_of_storage = c_start + bytes - pad_right - pad_left;
136
137 if (pad_left) memset(c_start - ELEMENT_SIZE, 0, ELEMENT_SIZE);
138 }
139
140 void dealloc() {
141 if (c_start == null) return;
142
143 unprotect();
144
145 TAllocator::free(c_start - pad_left, allocated_bytes());
146 THREAD_MEM_TRACKER_TRANSFER_FROM(c_end_of_storage - c_end_peak,
147 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
148 }
149
150 template <typename... TAllocatorParams>
151 void realloc(size_t bytes, TAllocatorParams&&... allocator_params) {
152 if (c_start == null) {
153 alloc(bytes, std::forward<TAllocatorParams>(allocator_params)...);
154 return;
155 }
156
157 unprotect();
158
159 THREAD_MEM_TRACKER_TRANSFER_TO(bytes - allocated_bytes(),
160 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
161
162 ptrdiff_t end_diff = c_end - c_start;
163 ptrdiff_t peak_diff = c_end_peak - c_start;
164
165 c_start = reinterpret_cast<char*>(TAllocator::realloc(
166 c_start - pad_left, allocated_bytes(), bytes,
167 std::forward<TAllocatorParams>(allocator_params)...)) +
168 pad_left;
169
170 c_end = c_start + end_diff;
171 c_end_peak = c_start + peak_diff;
172 c_end_of_storage = c_start + bytes - pad_right - pad_left;
173 }
174
175 bool is_initialized() const {
176 return (c_start != null) && (c_end != null) && (c_end_of_storage != null);
177 }
178
179 bool is_allocated_from_stack() const {
180 constexpr size_t stack_threshold = TAllocator::get_stack_threshold();
181 return (stack_threshold > 0) && (allocated_bytes() <= stack_threshold);
182 }
183
184 template <typename... TAllocatorParams>
185 void reserve_for_next_size(TAllocatorParams&&... allocator_params) {
186 if (size() == 0) {
187 // The allocated memory should be multiplication of ELEMENT_SIZE to hold the element, otherwise,
188 // memory issue such as corruption could appear in edge case.
189 realloc(std::max(integerRoundUp(initial_bytes, ELEMENT_SIZE),
190 minimum_memory_for_elements(1)),
191 std::forward<TAllocatorParams>(allocator_params)...);
192 } else
193 realloc(allocated_bytes() * 2, std::forward<TAllocatorParams>(allocator_params)...);
194 }
195
196#ifndef NDEBUG
199 void protect_impl(int prot) {
200 static constexpr size_t PROTECT_PAGE_SIZE = 4096;
201
202 char* left_rounded_up = reinterpret_cast<char*>(
203 (reinterpret_cast<intptr_t>(c_start) - pad_left + PROTECT_PAGE_SIZE - 1) /
204 PROTECT_PAGE_SIZE * PROTECT_PAGE_SIZE);
205 char* right_rounded_down =
206 reinterpret_cast<char*>((reinterpret_cast<intptr_t>(c_end_of_storage) + pad_right) /
207 PROTECT_PAGE_SIZE * PROTECT_PAGE_SIZE);
208
209 if (right_rounded_down > left_rounded_up) {
210 size_t length = right_rounded_down - left_rounded_up;
211 if (0 != mprotect(left_rounded_up, length, prot)) throw std::exception();
212 }
213 }
214
216 bool mprotected = false;
217#endif
218
219public:
220 bool empty() const { return c_end == c_start; }
221 size_t size() const { return (c_end - c_start) / ELEMENT_SIZE; }
222 size_t capacity() const { return (c_end_of_storage - c_start) / ELEMENT_SIZE; }
223
225 size_t allocated_bytes() const { return c_end_of_storage - c_start + pad_right + pad_left; }
226
227 void clear() { c_end = c_start; }
228
229 template <typename... TAllocatorParams>
230 void reserve(size_t n, TAllocatorParams&&... allocator_params) {
231 if (n > capacity())
232 realloc(round_up_to_power_of_two_or_zero(minimum_memory_for_elements(n)),
233 std::forward<TAllocatorParams>(allocator_params)...);
234 }
235
236 template <typename... TAllocatorParams>
237 void resize(size_t n, TAllocatorParams&&... allocator_params) {
238 reserve(n, std::forward<TAllocatorParams>(allocator_params)...);
239 resize_assume_reserved(n);
240 }
241
242 void resize_assume_reserved(const size_t n) {
243 c_end = c_start + byte_size(n);
244 reset_peak();
245 }
246
247 const char* raw_data() const { return c_start; }
248
249 template <typename... TAllocatorParams>
250 void push_back_raw(const char* ptr, TAllocatorParams&&... allocator_params) {
251 if (UNLIKELY(c_end == c_end_of_storage))
252 reserve_for_next_size(std::forward<TAllocatorParams>(allocator_params)...);
253
254 memcpy(c_end, ptr, ELEMENT_SIZE);
255 c_end += byte_size(1);
256 reset_peak();
257 }
258
259 void protect() {
260#ifndef NDEBUG
261 protect_impl(PROT_READ);
262 mprotected = true;
263#endif
264 }
265
266 void unprotect() {
267#ifndef NDEBUG
268 if (mprotected) protect_impl(PROT_WRITE);
269 mprotected = false;
270#endif
271 }
272
273 ~PODArrayBase() { dealloc(); }
274};

473 template <typename It1, typename It2, typename... TAllocatorParams>

485 template <typename It1, typename It2>
486 void insert(iterator it, It1 from_begin, It2 from_end) {
487 insert_prepare(from_begin, from_end);
488
489 size_t bytes_to_copy = this->byte_size(from_end - from_begin);
490 size_t bytes_to_move = (end() - it) * sizeof(T);
491
492 if (UNLIKELY(bytes_to_move))
493 memcpy(this->c_end + bytes_to_copy - bytes_to_move, this->c_end - bytes_to_move,
494 bytes_to_move);
495
496 memcpy(this->c_end - bytes_to_move, reinterpret_cast<const void*>(&*from_begin),
497 bytes_to_copy);
498 this->c_end += bytes_to_copy;
499 this->reset_peak();
500 }

510 void swap(PODArray& rhs) {
511#ifndef NDEBUG
512 this->unprotect();
513 rhs.unprotect();
514#endif
515
519 auto swap_stack_heap = [this](PODArray& arr1, PODArray& arr2) {
520 size_t stack_size = arr1.size();
521 size_t stack_allocated = arr1.allocated_bytes();
522
523 size_t heap_size = arr2.size();
524 size_t heap_allocated = arr2.allocated_bytes();
525
526#ifdef USE_MEM_TRACKER
527 size_t stack_peak_used = arr1.c_end_peak - arr1.c_start;
528 size_t heap_peak_used = arr2.c_end_peak - arr2.c_start;
529#endif
530
532 char* stack_c_start = arr1.c_start;
533
535 arr1.c_start = arr2.c_start;
536 arr1.c_end_of_storage = arr1.c_start + heap_allocated - arr1.pad_right;
537 arr1.c_end = arr1.c_start + this->byte_size(heap_size);
538 arr1.c_end_peak = arr1.c_end;
539 THREAD_MEM_TRACKER_TRANSFER_FROM(arr1.c_end_peak - arr1.c_start - heap_peak_used,
540 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
541
543 arr2.alloc(stack_allocated);
545 memcpy(arr2.c_start, stack_c_start, this->byte_size(stack_size));
546 arr2.c_end = arr2.c_end_peak = arr2.c_start + this->byte_size(stack_size);
547 THREAD_MEM_TRACKER_TRANSFER_FROM(arr2.c_end_peak - arr2.c_start - stack_peak_used,
548 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
549 };
550
551 auto do_move = [this](PODArray& src, PODArray& dest) {
552 if (src.is_allocated_from_stack()) {
553 dest.dealloc();
554 dest.alloc(src.allocated_bytes());
555 memcpy(dest.c_start, src.c_start, this->byte_size(src.size()));
556 dest.c_end = dest.c_end_peak = dest.c_start + (src.c_end - src.c_start);
557 THREAD_MEM_TRACKER_TRANSFER_FROM(dest.c_end_peak - dest.c_start,
558 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
559
560 THREAD_MEM_TRACKER_TRANSFER_FROM(src.c_end_of_storage - src.c_end_peak,
561 ExecEnv::GetInstance()->orphan_mem_tracker_raw());
562 src.c_start = Base::null;
563 src.c_end = Base::null;
564 src.c_end_of_storage = Base::null;
565 src.c_end_peak = Base::null;
566 } else {
567 std::swap(dest.c_start, src.c_start);
568 std::swap(dest.c_end, src.c_end);
569 std::swap(dest.c_end_of_storage, src.c_end_of_storage);
570 std::swap(dest.c_end_peak, src.c_end_peak);
571 }
572 };
573
574 if (!this->is_initialized() && !rhs.is_initialized()) {
575 return;
576 } else if (!this->is_initialized() && rhs.is_initialized()) {
577 do_move(rhs, *this);
578 return;
579 } else if (this->is_initialized() && !rhs.is_initialized()) {
580 do_move(*this, rhs);
581 return;
582 }
583
584 if (this->is_allocated_from_stack() && rhs.is_allocated_from_stack()) {
585 size_t min_size = std::min(this->size(), rhs.size());
586 size_t max_size = std::max(this->size(), rhs.size());
587
588 for (size_t i = 0; i < min_size; ++i) std::swap(this->operator[](i), rhs[i]);
589
590 if (this->size() == max_size) {
591 for (size_t i = min_size; i < max_size; ++i) rhs[i] = this->operator[](i);
592 } else {
593 for (size_t i = min_size; i < max_size; ++i) this->operator[](i) = rhs[i];
594 }
595
596 size_t lhs_size = this->size();
597 size_t lhs_allocated = this->allocated_bytes();
598
599 size_t rhs_size = rhs.size();
600 size_t rhs_allocated = rhs.allocated_bytes();
601
602 this->c_end_of_storage = this->c_start + rhs_allocated - Base::pad_right;
603 rhs.c_end_of_storage = rhs.c_start + lhs_allocated - Base::pad_right;
604
605 this->c_end = this->c_start + this->byte_size(rhs_size);
606 rhs.c_end = rhs.c_start + this->byte_size(lhs_size);
607 this->reset_peak();
608 rhs.reset_peak();
609 } else if (this->is_allocated_from_stack() && !rhs.is_allocated_from_stack()) {
610 swap_stack_heap(*this, rhs);
611 } else if (!this->is_allocated_from_stack() && rhs.is_allocated_from_stack()) {
612 swap_stack_heap(rhs, *this);
613 } else {
614 std::swap(this->c_start, rhs.c_start);
615 std::swap(this->c_end, rhs.c_end);
616 std::swap(this->c_end_of_storage, rhs.c_end_of_storage);
617 std::swap(this->c_end_peak, rhs.c_end_peak);
618 }
619 }

626 template <typename It1, typename It2>
627 void assign(It1 from_begin, It2 from_end) {
628 size_t required_capacity = from_end - from_begin;
629 if (required_capacity > this->capacity())
630 this->reserve(round_up_to_power_of_two_or_zero(required_capacity));
631
632 size_t bytes_to_copy = this->byte_size(required_capacity);
633 memcpy(this->c_start, reinterpret_cast<const void*>(&*from_begin), bytes_to_copy);
634 this->c_end = this->c_start + bytes_to_copy;
635 this->reset_peak();
636 }

640 bool operator==(const PODArray& other) const {
641 if (this->size() != other.size()) return false;
642
643 const_iterator this_it = begin();
644 const_iterator that_it = other.begin();
645
646 while (this_it != end()) {
647 if (*this_it != *that_it) return false;
648
649 ++this_it;
650 ++that_it;
651 }
652
653 return true;
654 }
