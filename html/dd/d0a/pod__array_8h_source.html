88template <size_t ELEMENT_SIZE, size_t initial_bytes, typename TAllocator, size_t pad_right_,
89 size_t pad_left_>
90class PODArrayBase : private boost::noncopyable,
91 private TAllocator
92{
93protected:
95 static constexpr size_t pad_right = integerRoundUp(pad_right_, ELEMENT_SIZE);
97 static constexpr size_t pad_left = integerRoundUp(integerRoundUp(pad_left_, ELEMENT_SIZE), 16);
99 static constexpr char* null =
100 pad_left ? const_cast<char*>(empty_pod_array) + EmptyPODArraySize : nullptr;
101
102 static_assert(pad_left <= EmptyPODArraySize &&
103 "Left Padding exceeds EmptyPODArraySize. Is the element size too large?");
104
105 char* c_start = null;
106 char* c_end = null;
107 char* c_end_of_storage = null;
108
110 static size_t byte_size(size_t num_elements) { return num_elements * ELEMENT_SIZE; }
111
113 static size_t minimum_memory_for_elements(size_t num_elements) {
114 return byte_size(num_elements) + pad_right + pad_left;
115 }
116
117 void alloc_for_num_elements(size_t num_elements) {
118 alloc(round_up_to_power_of_two_or_zero(minimum_memory_for_elements(num_elements)));
119 }

121 template <typename... TAllocatorParams>
122 void alloc(size_t bytes, TAllocatorParams&&... allocator_params) {
123 c_start = c_end = reinterpret_cast<char*>(TAllocator::alloc(
124 bytes, std::forward<TAllocatorParams>(allocator_params)...)) +
125 pad_left;
126 c_end_of_storage = c_start + bytes - pad_right - pad_left;
127
128 if (pad_left) memset(c_start - ELEMENT_SIZE, 0, ELEMENT_SIZE);
129 }
130
131 void dealloc() {
132 if (c_start == null) return;
133
134 unprotect();
135
136 TAllocator::free(c_start - pad_left, allocated_bytes());
137 }
138
139 template <typename... TAllocatorParams>
140 void realloc(size_t bytes, TAllocatorParams&&... allocator_params) {
141 if (c_start == null) {
142 alloc(bytes, std::forward<TAllocatorParams>(allocator_params)...);
143 return;
144 }
145
146 unprotect();
147
148 ptrdiff_t end_diff = c_end - c_start;
149
150 c_start = reinterpret_cast<char*>(TAllocator::realloc(
151 c_start - pad_left, allocated_bytes(), bytes,
152 std::forward<TAllocatorParams>(allocator_params)...)) +
153 pad_left;
154
155 c_end = c_start + end_diff;
156 c_end_of_storage = c_start + bytes - pad_right - pad_left;
157 }
158
159 bool is_initialized() const {
160 return (c_start != null) && (c_end != null) && (c_end_of_storage != null);
161 }
162
163 bool is_allocated_from_stack() const {
164 constexpr size_t stack_threshold = TAllocator::get_stack_threshold();
165 return (stack_threshold > 0) && (allocated_bytes() <= stack_threshold);
166 }

168 template <typename... TAllocatorParams>
169 void reserve_for_next_size(TAllocatorParams&&... allocator_params) {
170 if (size() == 0) {
171 // The allocated memory should be multiplication of ELEMENT_SIZE to hold the element, otherwise,
172 // memory issue such as corruption could appear in edge case.
173 realloc(std::max(integerRoundUp(initial_bytes, ELEMENT_SIZE),
174 minimum_memory_for_elements(1)),
175 std::forward<TAllocatorParams>(allocator_params)...);
176 } else
177 realloc(allocated_bytes() * 2, std::forward<TAllocatorParams>(allocator_params)...);
178 }

180#ifndef NDEBUG
183 void protect_impl(int prot) {
184 static constexpr size_t PROTECT_PAGE_SIZE = 4096;
185
186 char* left_rounded_up = reinterpret_cast<char*>(
187 (reinterpret_cast<intptr_t>(c_start) - pad_left + PROTECT_PAGE_SIZE - 1) /
188 PROTECT_PAGE_SIZE * PROTECT_PAGE_SIZE);
189 char* right_rounded_down =
190 reinterpret_cast<char*>((reinterpret_cast<intptr_t>(c_end_of_storage) + pad_right) /
191 PROTECT_PAGE_SIZE * PROTECT_PAGE_SIZE);
192
193 if (right_rounded_down > left_rounded_up) {
194 size_t length = right_rounded_down - left_rounded_up;
195 if (0 != mprotect(left_rounded_up, length, prot)) throw std::exception();
196 }
197 }
198
200 bool mprotected = false;
201#endif
202
203public:
204 bool empty() const { return c_end == c_start; }
205 size_t size() const { return (c_end - c_start) / ELEMENT_SIZE; }
206 size_t capacity() const { return (c_end_of_storage - c_start) / ELEMENT_SIZE; }
207
209 size_t allocated_bytes() const { return c_end_of_storage - c_start + pad_right + pad_left; }
210
211 void clear() { c_end = c_start; }
212
213 template <typename... TAllocatorParams>
214 void reserve(size_t n, TAllocatorParams&&... allocator_params) {
215 if (n > capacity())
216 realloc(round_up_to_power_of_two_or_zero(minimum_memory_for_elements(n)),
217 std::forward<TAllocatorParams>(allocator_params)...);
218 }
219
220 template <typename... TAllocatorParams>
221 void resize(size_t n, TAllocatorParams&&... allocator_params) {
222 reserve(n, std::forward<TAllocatorParams>(allocator_params)...);
223 resize_assume_reserved(n);
224 }
225
226 void resize_assume_reserved(const size_t n) { c_end = c_start + byte_size(n); }
227
228 const char* raw_data() const { return c_start; }
229
230 template <typename... TAllocatorParams>
231 void push_back_raw(const char* ptr, TAllocatorParams&&... allocator_params) {
232 if (UNLIKELY(c_end == c_end_of_storage))
233 reserve_for_next_size(std::forward<TAllocatorParams>(allocator_params)...);
234
235 memcpy(c_end, ptr, ELEMENT_SIZE);
236 c_end += byte_size(1);
237 }

239 void protect() {
240#ifndef NDEBUG
241 protect_impl(PROT_READ);
242 mprotected = true;
243#endif
244 }
245
246 void unprotect() {
247#ifndef NDEBUG
248 if (mprotected) protect_impl(PROT_WRITE);
249 mprotected = false;
250#endif
251 }

253 ~PODArrayBase() { dealloc(); }
254};

453 template <typename It1, typename It2>
454 void insert(iterator it, It1 from_begin, It2 from_end) {
455 insert_prepare(from_begin, from_end);
456
457 size_t bytes_to_copy = this->byte_size(from_end - from_begin);
458 size_t bytes_to_move = (end() - it) * sizeof(T);
459
460 if (UNLIKELY(bytes_to_move))
461 memcpy(this->c_end + bytes_to_copy - bytes_to_move, this->c_end - bytes_to_move,
462 bytes_to_move);
463
464 memcpy(this->c_end - bytes_to_move, reinterpret_cast<const void*>(&*from_begin),
465 bytes_to_copy);
466 this->c_end += bytes_to_copy;
467 }

476 void swap(PODArray& rhs) {
477#ifndef NDEBUG
478 this->unprotect();
479 rhs.unprotect();
480#endif
481
485 auto swap_stack_heap = [this](PODArray& arr1, PODArray& arr2) {
486 size_t stack_size = arr1.size();
487 size_t stack_allocated = arr1.allocated_bytes();
488
489 size_t heap_size = arr2.size();
490 size_t heap_allocated = arr2.allocated_bytes();
491
493 char* stack_c_start = arr1.c_start;
494
496 arr1.c_start = arr2.c_start;
497 arr1.c_end_of_storage = arr1.c_start + heap_allocated - arr1.pad_right;
498 arr1.c_end = arr1.c_start + this->byte_size(heap_size);
499
501 arr2.alloc(stack_allocated);
503 memcpy(arr2.c_start, stack_c_start, this->byte_size(stack_size));
504 arr2.c_end = arr2.c_start + this->byte_size(stack_size);
505 };
506
507 auto do_move = [this](PODArray& src, PODArray& dest) {
508 if (src.is_allocated_from_stack()) {
509 dest.dealloc();
510 dest.alloc(src.allocated_bytes());
511 memcpy(dest.c_start, src.c_start, this->byte_size(src.size()));
512 dest.c_end = dest.c_start + (src.c_end - src.c_start);
513
514 src.c_start = Base::null;
515 src.c_end = Base::null;
516 src.c_end_of_storage = Base::null;
517 } else {
518 std::swap(dest.c_start, src.c_start);
519 std::swap(dest.c_end, src.c_end);
520 std::swap(dest.c_end_of_storage, src.c_end_of_storage);
521 }
522 };
523
524 if (!this->is_initialized() && !rhs.is_initialized()) {
525 return;
526 } else if (!this->is_initialized() && rhs.is_initialized()) {
527 do_move(rhs, *this);
528 return;
529 } else if (this->is_initialized() && !rhs.is_initialized()) {
530 do_move(*this, rhs);
531 return;
532 }
533
534 if (this->is_allocated_from_stack() && rhs.is_allocated_from_stack()) {
535 size_t min_size = std::min(this->size(), rhs.size());
536 size_t max_size = std::max(this->size(), rhs.size());
537
538 for (size_t i = 0; i < min_size; ++i) std::swap(this->operator[](i), rhs[i]);
539
540 if (this->size() == max_size) {
541 for (size_t i = min_size; i < max_size; ++i) rhs[i] = this->operator[](i);
542 } else {
543 for (size_t i = min_size; i < max_size; ++i) this->operator[](i) = rhs[i];
544 }
545
546 size_t lhs_size = this->size();
547 size_t lhs_allocated = this->allocated_bytes();
548
549 size_t rhs_size = rhs.size();
550 size_t rhs_allocated = rhs.allocated_bytes();
551
552 this->c_end_of_storage = this->c_start + rhs_allocated - Base::pad_right;
553 rhs.c_end_of_storage = rhs.c_start + lhs_allocated - Base::pad_right;
554
555 this->c_end = this->c_start + this->byte_size(rhs_size);
556 rhs.c_end = rhs.c_start + this->byte_size(lhs_size);
557 } else if (this->is_allocated_from_stack() && !rhs.is_allocated_from_stack()) {
558 swap_stack_heap(*this, rhs);
559 } else if (!this->is_allocated_from_stack() && rhs.is_allocated_from_stack()) {
560 swap_stack_heap(rhs, *this);
561 } else {
562 std::swap(this->c_start, rhs.c_start);
563 std::swap(this->c_end, rhs.c_end);
564 std::swap(this->c_end_of_storage, rhs.c_end_of_storage);
565 }
566 }

573 template <typename It1, typename It2>
574 void assign(It1 from_begin, It2 from_end) {
575 size_t required_capacity = from_end - from_begin;
576 if (required_capacity > this->capacity())
577 this->reserve(round_up_to_power_of_two_or_zero(required_capacity));
578
579 size_t bytes_to_copy = this->byte_size(required_capacity);
580 memcpy(this->c_start, reinterpret_cast<const void*>(&*from_begin), bytes_to_copy);
581 this->c_end = this->c_start + bytes_to_copy;
582 }

586 bool operator==(const PODArray& other) const {
587 if (this->size() != other.size()) return false;
588
589 const_iterator this_it = begin();
590 const_iterator that_it = other.begin();
591
592 while (this_it != end()) {
593 if (*this_it != *that_it) return false;

597 }
598
