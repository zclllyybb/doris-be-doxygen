94class SimpleFunctionFactory {
95 using Creator = std::function<FunctionBuilderPtr()>;
96 using FunctionCreators = phmap::flat_hash_map<std::string, Creator>;
97 using FunctionIsVariadic = phmap::flat_hash_set<std::string>;
98
99public:
100 void register_function(const std::string& name, const Creator& ptr) {
101 DataTypes types = ptr()->get_variadic_argument_types();
102 // types.empty() means function is not variadic
103 if (!types.empty()) {
104 function_variadic_set.insert(name);
105 }
106
107 std::string key_str = name;
108 if (!types.empty()) {
109 for (const auto& type : types) {
110 key_str.append(type->get_family_name());
111 }
112 }
113 function_creators[key_str] = ptr;
114 }
115
116 template <class Function>
117 void register_function() {
118 if constexpr (std::is_base_of<IFunction, Function>::value) {
119 register_function(Function::name, &createDefaultFunction<Function>);
120 } else {
121 register_function(Function::name, &Function::create);
122 }
123 }
124
125 template <class Function>
126 void register_function(std::string name) {
127 function_creators[name] = &createDefaultFunction<Function>;
128 }
129
130 void register_alias(const std::string& name, const std::string& alias) {
131 function_alias[alias] = name;
132 }
133
134 FunctionBasePtr get_function(const std::string& name, const ColumnsWithTypeAndName& arguments,
135 const DataTypePtr& return_type) {
136 std::string key_str = name;
137
138 if (function_alias.count(name)) {
139 key_str = function_alias[name];
140 }
141
142 // if function is variadic, added types_str as key
143 if (function_variadic_set.count(key_str)) {
144 for (auto& arg : arguments) {
145 key_str.append(arg.type->is_nullable()
146 ? reinterpret_cast<const DataTypeNullable*>(arg.type.get())
147 ->get_nested_type()
148 ->get_family_name()
149 : arg.type->get_family_name());
150 }
151 }
152
153 auto iter = function_creators.find(key_str);
154 if (iter != function_creators.end()) {
155 return iter->second()->build(arguments, return_type);
156 }
157
158 LOG(WARNING) << fmt::format("Function signature {} is not found", key_str);
159 return nullptr;
160 }
161
162private:
163 FunctionCreators function_creators;
164 FunctionIsVariadic function_variadic_set;
165 std::unordered_map<std::string, std::string> function_alias;

167 template <typename Function>

169 return std::make_shared<DefaultFunctionBuilder>(Function::create());
170 }
