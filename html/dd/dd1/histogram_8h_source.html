29// Histogram data structure implementation:
30//
31// After construction, the 'value_index_map' will be set to:
32//
33// BucketValue: | 1 | 2 | 2*1.5 |2*1.5^2|2*1.5^3| ... |2*1.5^n| ... |UINT64MAX|
34// Index: | 0 | 1 | 2 | 3 | 4 | ... | n-1 | ... | 108 |
35//
36// The width of bucket is growing by 1.5 times and its initial values are 1 and 2.
37// The input value will be add to the bucket which lower bound is just greater than
38// input value. For example, input value A > 2*1.5^n and A <= 2*1.5^(n+1), A will be added
39// to the latter bucket.
40class HistogramBucketMapper {
41public:
42 HistogramBucketMapper();
43
44 // converts a value to the bucket index.
45 size_t index_for_value(const uint64_t& value) const;
46 // number of buckets required.
47
48 size_t bucket_count() const { return _bucket_values.size(); }
49
50 uint64_t last_value() const { return _max_bucket_value; }
51
52 uint64_t first_value() const { return _min_bucket_value; }
53
54 uint64_t bucket_limit(const size_t bucket_number) const {
55 DCHECK(bucket_number < bucket_count());
56 return _bucket_values[bucket_number];
57 }
58
59private:
60 std::vector<uint64_t> _bucket_values;
61 uint64_t _max_bucket_value;
62 uint64_t _min_bucket_value;
63 std::map<uint64_t, uint64_t> _value_index_map;
64};
