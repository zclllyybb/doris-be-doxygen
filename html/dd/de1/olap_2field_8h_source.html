98 // Compare two types of cell.
99 // This function differs compare in that this function compare cell which
100 // will consider the condition which cell may be nullptr. While compare only
101 // compare column content without considering nullptr condition.
102 // Only compare column content, without considering nullptr condition.
103 // RETURNS:
104 // 0 means equal,
105 // -1 means left less than right,
106 // 1 means left bigger than right
107 template <typename LhsCellType, typename RhsCellType>
108 int compare_cell(const LhsCellType& lhs, const RhsCellType& rhs) const {
109 bool l_null = lhs.is_null();
110 bool r_null = rhs.is_null();
111 if (l_null != r_null) {
112 return l_null ? -1 : 1;
113 }
114 return l_null ? 0 : _type_info->cmp(lhs.cell_ptr(), rhs.cell_ptr());
115 }

117 // Copy source cell's content to destination cell directly.
118 // For string type, this function assume that destination has
119 // enough space and copy source content into destination without
120 // memory allocation.
121 template <typename DstCellType, typename SrcCellType>
122 void direct_copy(DstCellType* dst, const SrcCellType& src) const {
123 bool is_null = src.is_null();
124 dst->set_is_null(is_null);
125 if (is_null) {
126 return;
127 }
128 if (type() == OLAP_FIELD_TYPE_STRING) {
129 auto dst_slice = reinterpret_cast<Slice*>(dst->mutable_cell_ptr());
130 auto src_slice = reinterpret_cast<const Slice*>(src.cell_ptr());
131 if (dst_slice->size < src_slice->size) {
132 *_long_text_buf = static_cast<char*>(realloc(*_long_text_buf, src_slice->size));
133 dst_slice->data = *_long_text_buf;
134 dst_slice->size = src_slice->size;
135 }
136 }
137 return _type_info->direct_copy(dst->mutable_cell_ptr(), src.cell_ptr());
138 }

140 // deep copy source cell' content to destination cell.
141 // For string type, this will allocate data form arena,
142 // and copy source's content.
143 template <typename DstCellType, typename SrcCellType>
144 void deep_copy(DstCellType* dst, const SrcCellType& src, vectorized::Arena* arena) const {
145 bool is_null = src.is_null();
146 dst->set_is_null(is_null);
147 if (is_null) {
148 return;
149 }
150 _type_info->deep_copy(dst->mutable_cell_ptr(), src.cell_ptr(), arena);
151 }

153 // used by init scan key stored in string format
154 // value_string should end with '\0'
155 Status from_string(char* buf, const std::string& value_string, const int precision = 0,
156 const int scale = 0) const {
157 if (type() == OLAP_FIELD_TYPE_STRING && !value_string.empty()) {
158 auto slice = reinterpret_cast<Slice*>(buf);
159 if (slice->size < value_string.size()) {
160 *_long_text_buf = static_cast<char*>(realloc(*_long_text_buf, value_string.size()));
161 slice->data = *_long_text_buf;
162 slice->size = value_string.size();
163 }
164 }
165 return _type_info->from_string(buf, value_string, precision, scale);
166 }

208protected:
209 TypeInfoPtr _type_info;
210 // unit : byte
211 // except for strings, other types have fixed lengths
212 // Note that, the struct type itself has fixed length, but due to
213 // its number of subfields is a variable, so the actual length of
214 // a struct field is not fixed.
215 uint32_t _length;
216 // Since the length of the STRING type cannot be determined,
217 // only dynamic memory can be used. Arena cannot realize realloc.
218 // The schema information is shared globally. Therefore,
219 // dynamic memory can only be managed in thread local mode.
220 // The memory will be created and released in rowcursor.
221 char** _long_text_buf = nullptr;
222
223 char* allocate_string_value(vectorized::Arena* arena) const {
224 char* type_value = arena->alloc(sizeof(Slice));
225 auto slice = reinterpret_cast<Slice*>(type_value);
226 slice->size = _length;
227 slice->data = arena->alloc(slice->size);
228 return type_value;
229 }
230
231 void clone(Field* other) const {
232 other->_type_info = clone_type_info(this->_type_info.get());
233 other->_key_coder = this->_key_coder;
234 other->_name = this->_name;
235 other->_index_size = this->_index_size;
236 other->_is_nullable = this->_is_nullable;
237 other->_sub_fields.clear();
238 other->_precision = this->_precision;
239 other->_scale = this->_scale;
240 other->_unique_id = this->_unique_id;
241 for (const auto& f : _sub_fields) {
242 Field* item = f->clone();
243 other->add_sub_field(std::unique_ptr<Field>(item));
244 }
245 }
246
247private:
248 // maximum length of Field, unit : bytes
249 // usually equal to length, except for variable-length strings
250 const KeyCoder* _key_coder;
251 std::string _name;
252 uint16_t _index_size;
253 bool _is_nullable;
254 std::vector<std::unique_ptr<Field>> _sub_fields;
255 int32_t _precision;
256 int32_t _scale;
257 int32_t _unique_id;
258};

338 // To prevent zone map cost too many memory, if varchar length
339 // longer than `MAX_ZONE_MAP_INDEX_SIZE`. we just allocate
340 // `MAX_ZONE_MAP_INDEX_SIZE` of memory
341 char* allocate_zone_map_value(vectorized::Arena* arena) const override {
342 char* type_value = arena->alloc(sizeof(Slice));
343 auto slice = reinterpret_cast<Slice*>(type_value);
344 slice->size = MAX_ZONE_MAP_INDEX_SIZE > _length ? _length : MAX_ZONE_MAP_INDEX_SIZE;
345 slice->data = arena->alloc(slice->size);
346 return type_value;
347 }
348
349 // only varchar filed need modify zone map index when zone map max_value
350 // index longer than `MAX_ZONE_MAP_INDEX_SIZE`. so here we add one
351 // for the last byte
352 // In UTF8 encoding, here do not appear 0xff in last byte
353 void modify_zone_map_index(char* src) const override {
354 auto slice = reinterpret_cast<Slice*>(src);
355 if (slice->size == MAX_ZONE_MAP_INDEX_SIZE) {
356 slice->mutable_data()[slice->size - 1] += 1;
357 }
358 }
359
360 void set_to_zone_map_max(char* ch) const override {
361 auto slice = reinterpret_cast<Slice*>(ch);
362 int length = _length < MAX_ZONE_MAP_INDEX_SIZE ? _length : MAX_ZONE_MAP_INDEX_SIZE;
363 slice->size = length;
364 memset(slice->data, 0xFF, slice->size);
365 }

535class FieldFactory {
536public:
537 static Field* create(const TabletColumn& column) {
538 // for key column
539 if (column.is_key()) {
540 switch (column.type()) {
541 case OLAP_FIELD_TYPE_CHAR:
542 return new CharField(column);
543 case OLAP_FIELD_TYPE_VARCHAR:
544 return new VarcharField(column);
545 case OLAP_FIELD_TYPE_STRING:
546 return new StringField(column);
547 case OLAP_FIELD_TYPE_STRUCT: {
548 auto* local = new StructField(column);
549 for (uint32_t i = 0; i < column.get_subtype_count(); i++) {
550 std::unique_ptr<Field> sub_field(
551 FieldFactory::create(column.get_sub_column(i)));
552 local->add_sub_field(std::move(sub_field));
553 }
554 return local;
555 }
556 case OLAP_FIELD_TYPE_ARRAY: {
557 std::unique_ptr<Field> item_field(FieldFactory::create(column.get_sub_column(0)));
558 auto* local = new ArrayField(column);
559 local->add_sub_field(std::move(item_field));
560 return local;
561 }
562 case OLAP_FIELD_TYPE_MAP: {
563 std::unique_ptr<Field> key_field(FieldFactory::create(column.get_sub_column(0)));
564 std::unique_ptr<Field> val_field(FieldFactory::create(column.get_sub_column(1)));
565 auto* local = new MapField(column);
566 local->add_sub_field(std::move(key_field));
567 local->add_sub_field(std::move(val_field));
568 return local;
569 }
570 case OLAP_FIELD_TYPE_DECIMAL:
571 [[fallthrough]];
572 case OLAP_FIELD_TYPE_DECIMAL32:
573 [[fallthrough]];
574 case OLAP_FIELD_TYPE_DECIMAL64:
575 [[fallthrough]];
576 case OLAP_FIELD_TYPE_DECIMAL128I:
577 [[fallthrough]];
578 case OLAP_FIELD_TYPE_DATETIMEV2: {
579 Field* field = new Field(column);
580 field->set_precision(column.precision());
581 field->set_scale(column.frac());
582 return field;
583 }
584 default:
585 return new Field(column);
586 }
587 }
588
589 // for value column
590 switch (column.aggregation()) {
591 case OLAP_FIELD_AGGREGATION_NONE:
592 case OLAP_FIELD_AGGREGATION_SUM:
593 case OLAP_FIELD_AGGREGATION_MIN:
594 case OLAP_FIELD_AGGREGATION_MAX:
595 case OLAP_FIELD_AGGREGATION_REPLACE:
596 case OLAP_FIELD_AGGREGATION_REPLACE_IF_NOT_NULL:
597 switch (column.type()) {
598 case OLAP_FIELD_TYPE_CHAR:
599 return new CharField(column);
600 case OLAP_FIELD_TYPE_VARCHAR:
601 return new VarcharField(column);
602 case OLAP_FIELD_TYPE_STRING:
603 return new StringField(column);
604 case OLAP_FIELD_TYPE_STRUCT: {
605 auto* local = new StructField(column);
606 for (uint32_t i = 0; i < column.get_subtype_count(); i++) {
607 std::unique_ptr<Field> sub_field(
608 FieldFactory::create(column.get_sub_column(i)));
609 local->add_sub_field(std::move(sub_field));
610 }
611 return local;
612 }
613 case OLAP_FIELD_TYPE_ARRAY: {
614 std::unique_ptr<Field> item_field(FieldFactory::create(column.get_sub_column(0)));
615 auto* local = new ArrayField(column);
616 local->add_sub_field(std::move(item_field));
617 return local;
618 }
619 case OLAP_FIELD_TYPE_MAP: {
620 DCHECK(column.get_subtype_count() == 2);
621 auto* local = new MapField(column);
622 std::unique_ptr<Field> key_field(FieldFactory::create(column.get_sub_column(0)));
623 std::unique_ptr<Field> value_field(FieldFactory::create(column.get_sub_column(1)));
624 local->add_sub_field(std::move(key_field));
625 local->add_sub_field(std::move(value_field));
626 return local;
627 }
628 case OLAP_FIELD_TYPE_DECIMAL:
629 [[fallthrough]];
630 case OLAP_FIELD_TYPE_DECIMAL32:
631 [[fallthrough]];
632 case OLAP_FIELD_TYPE_DECIMAL64:
633 [[fallthrough]];
634 case OLAP_FIELD_TYPE_DECIMAL128I:
635 [[fallthrough]];
636 case OLAP_FIELD_TYPE_DATETIMEV2: {
637 Field* field = new Field(column);
638 field->set_precision(column.precision());
639 field->set_scale(column.frac());
640 return field;
641 }
642 default:
643 return new Field(column);
644 }
645 case OLAP_FIELD_AGGREGATION_HLL_UNION:
646 return new HllAggField(column);
647 case OLAP_FIELD_AGGREGATION_BITMAP_UNION:
648 return new BitmapAggField(column);
