37// A LoadChannel manages tablets channels for all indexes
38// corresponding to a certain load job
39class LoadChannel {
40public:
41 LoadChannel(const UniqueId& load_id, std::unique_ptr<MemTracker> mem_tracker, int64_t timeout_s,
42 bool is_high_priority, const std::string& sender_ip);

46 Status open(const PTabletWriterOpenRequest& request);

48 // this batch must belong to a index in one transaction
49 template <typename TabletWriterAddRequest, typename TabletWriterAddResult>

51
52 // return true if this load channel has been opened and all tablets channels are closed then.
53 bool is_finished();
54

57 time_t last_updated_time() const { return _last_updated_time.load(); }
58
59 const UniqueId& load_id() const { return _load_id; }

61 int64_t mem_consumption() {
62 int64_t mem_usage = 0;
63 {
64 std::lock_guard<SpinLock> l(_tablets_channels_lock);
65 for (auto& it : _tablets_channels) {
66 mem_usage += it.second->mem_consumption();
67 }
68 }
69 _mem_tracker->set_consumption(mem_usage);
70 return mem_usage;
71 }
72
73 void get_writers_mem_consumption_snapshot(
74 std::vector<std::pair<int64_t, std::multimap<int64_t, int64_t, std::greater<int64_t>>>>*
75 writers_mem_snap) {
76 std::lock_guard<SpinLock> l(_tablets_channels_lock);
77 for (auto& it : _tablets_channels) {
78 std::multimap<int64_t, int64_t, std::greater<int64_t>> tablets_channel_mem;
79 it.second->get_writers_mem_consumption_snapshot(&tablets_channel_mem);
80 writers_mem_snap->emplace_back(it.first, std::move(tablets_channel_mem));
81 }
82 }
83
84 int64_t timeout() const { return _timeout_s; }
85
86 bool is_high_priority() const { return _is_high_priority; }
87
88 void flush_memtable_async(int64_t index_id, int64_t tablet_id) {
89 std::lock_guard<std::mutex> l(_lock);
90 auto it = _tablets_channels.find(index_id);
91 if (it != _tablets_channels.end()) {
92 it->second->flush_memtable_async(tablet_id);
93 }
94 }
95
96 void wait_flush(int64_t index_id, int64_t tablet_id) {
97 std::lock_guard<std::mutex> l(_lock);
98 auto it = _tablets_channels.find(index_id);
99 if (it != _tablets_channels.end()) {
100 it->second->wait_flush(tablet_id);
101 }
102 }
103
104protected:
105 Status _get_tablets_channel(std::shared_ptr<TabletsChannel>& channel, bool& is_finished,
106 const int64_t index_id);
107
108 template <typename Request, typename Response>
109 Status _handle_eos(std::shared_ptr<TabletsChannel>& channel, const Request& request,
110 Response* response) {
111 bool finished = false;
112 auto index_id = request.index_id();
113 RETURN_IF_ERROR(channel->close(
114 this, request.sender_id(), request.backend_id(), &finished, request.partition_ids(),
115 response->mutable_tablet_vec(), response->mutable_tablet_errors(),
116 request.slave_tablet_nodes(), response->mutable_success_slave_tablet_node_ids(),
117 request.write_single_replica()));
118 if (finished) {
119 std::lock_guard<std::mutex> l(_lock);
120 {
121 std::lock_guard<SpinLock> l(_tablets_channels_lock);
122 _tablets_channels.erase(index_id);
123 }
124 _finished_channel_ids.emplace(index_id);
125 }
126 return Status::OK();
127 }
128
129private:
130 UniqueId _load_id;
131 // Tracks the total memory consumed by current load job on this BE
132 std::unique_ptr<MemTracker> _mem_tracker;
133
134 // lock protect the tablets channel map
135 std::mutex _lock;
136 // index id -> tablets channel
137 std::unordered_map<int64_t, std::shared_ptr<TabletsChannel>> _tablets_channels;
138 SpinLock _tablets_channels_lock;
139 // This is to save finished channels id, to handle the retry request.
140 std::unordered_set<int64_t> _finished_channel_ids;
141 // set to true if at least one tablets channel has been opened
142 bool _opened = false;
143
144 std::atomic<time_t> _last_updated_time;
145
146 // the timeout of this load job.
147 // Timed out channels will be periodically deleted by LoadChannelMgr.
148 int64_t _timeout_s;
149
150 // true if this is a high priority load task
151 bool _is_high_priority = false;
152
153 // the ip where tablet sink locate
154 std::string _sender_ip = "";
155};

157template <typename TabletWriterAddRequest, typename TabletWriterAddResult>
158Status LoadChannel::add_batch(const TabletWriterAddRequest& request,
159 TabletWriterAddResult* response) {
160 int64_t index_id = request.index_id();
161 // 1. get tablets channel
162 std::shared_ptr<TabletsChannel> channel;
163 bool is_finished;
164 Status st = _get_tablets_channel(channel, is_finished, index_id);
165 if (!st.ok() || is_finished) {
166 return st;
167 }
168
169 // 2. add block to tablets channel
170 if (request.has_block()) {
171 RETURN_IF_ERROR(channel->add_batch(request, response));
172 }
173
174 // 3. handle eos
175 if (request.has_eos() && request.eos()) {
176 st = _handle_eos(channel, request, response);
177 if (!st.ok()) {
178 return st;
179 }
180 }
181 _last_updated_time.store(time(nullptr));
182 return st;
183}
