44// This class is used for checking whether a row should be deleted.
45// It is used in the following processes：
46// 1. Create and initialize a DeleteHandler object:
47// Status res;
48// DeleteHandler delete_handler;
49// res = delete_handler.init(tablet, condition_version);
50// 2. After all rows have been checked, you should release this object by calling:
51// delete_handler.finalize();
52//
53// NOTE：
54// * In the first step, before calling delete_handler.init(), you should lock the tablet's header file.
55class DeleteHandler {
56 // These static method is used to generate delete predicate pb during write or push handler
57public:
58 // generated DeletePredicatePB by TCondition
59 static Status generate_delete_predicate(const TabletSchema& schema,
60 const std::vector<TCondition>& conditions,
61 DeletePredicatePB* del_pred);
62
63 // construct sub condition from TCondition
64 static std::string construct_sub_predicates(const TCondition& condition);
65
66private:
67 // Validate the condition on the schema.
68 static Status check_condition_valid(const TabletSchema& tablet_schema, const TCondition& cond);
69
70 // Check whether the condition value is valid according to its type.
71 // 1. For integers(int8,int16,in32,int64,uint8,uint16,uint32,uint64), check whether they are overflow
72 // 2. For decimal, check whether precision or scale is overflow
73 // 3. For date and datetime, check format and value
74 // 4. For char and varchar, check length
75 static bool is_condition_value_valid(const TabletColumn& column,
76 const std::string& condition_op,
77 const std::string& value_str);
78
79public:
80 DeleteHandler() = default;
81 ~DeleteHandler() { finalize(); }
82
83 // Initialize DeleteHandler, use the delete conditions of this tablet whose version less than or equal to
84 // 'version' to fill '_del_conds'.
85 // NOTE: You should lock the tablet's header file before calling this function.
86 //
87 // input:
88 // * schema: tablet's schema, the delete conditions and data rows are in this schema
89 // * version: maximum version
90 // return:
91 // * Status::Error<DELETE_INVALID_PARAMETERS>(): input parameters are not valid
92 // * Status::Error<MEM_ALLOC_FAILED>(): alloc memory failed
93 Status init(TabletSchemaSPtr tablet_schema,
94 const std::vector<RowsetMetaSharedPtr>& delete_conditions, int64_t version);
95
96 bool empty() const { return _del_conds.empty(); }
97
98 // Release an instance of this class.
99 void finalize();
100
101 void get_delete_conditions_after_version(
102 int64_t version, AndBlockColumnPredicate* and_block_column_predicate_ptr,
103 std::unordered_map<int32_t, std::vector<const ColumnPredicate*>>*
104 col_id_to_del_predicates) const;
105
106private:
107 // Use regular expression to extract 'column_name', 'op' and 'operands'
108 bool _parse_condition(const std::string& condition_str, TCondition* condition);
109
110 bool _is_inited = false;
111 // DeleteConditions in _del_conds are in 'OR' relationship
112 std::vector<DeleteConditions> _del_conds;
113 std::unique_ptr<MemPool> _predicate_mem_pool;
114
115 DISALLOW_COPY_AND_ASSIGN(DeleteHandler);
116};
