28struct AuthInfo {
29 std::string user;
30 std::string passwd;
31 std::string cluster;

34 int64_t auth_code = -1;

38template <class T>
39void set_request_auth(T* req, const AuthInfo& auth) {
40 if (auth.auth_code != -1) {
41 // if auth_code is set, no need to set other info
42 req->__set_auth_code(auth.auth_code);
43 // user name and passwd is unused, but they are required field.
44 // so they have to be set.
45 req->user = "";
46 req->passwd = "";
47 } else if (auth.auth_code_uuid != "") {
48 req->__isset.auth_code_uuid = true;
49 req->auth_code_uuid = auth.auth_code_uuid;
50 } else {
51 req->user = auth.user;
52 req->passwd = auth.passwd;
53 if (!auth.cluster.empty()) {
54 req->__set_cluster(auth.cluster);
55 }
56 req->__set_user_ip(auth.user_ip);
57 }
58}
