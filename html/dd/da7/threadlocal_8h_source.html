18// Reference from kudu, Solve the problem of gcc11 compiling
19// non-trivial thread_local variables on lower versions of GLIBC.
20// see https://github.com/apache/doris/pull/7911
21//
22// Block-scoped static thread local implementation.
23//
24// Usage is similar to a C++11 thread_local. The BLOCK_STATIC_THREAD_LOCAL macro
25// defines a thread-local pointer to the specified type, which is lazily
26// instantiated by any thread entering the block for the first time. The
27// constructor for the type T is invoked at macro execution time, as expected,
28// and its destructor is invoked when the corresponding thread's Runnable
29// returns, or when the thread exits.
30//
31// Inspired by Poco <http://pocoproject.org/docs/Poco.ThreadLocal.html>,
32// Andrew Tomazos <http://stackoverflow.com/questions/12049684/>, and
33// the C++11 thread_local API.
34//
35// Example usage:
36//
37// // Invokes a 3-arg constructor on SomeClass:
38// BLOCK_STATIC_THREAD_LOCAL(SomeClass, instance, arg1, arg2, arg3);
39// instance->DoSomething();
40//
41
42#pragma once
43
44#include "gutil/port.h"
45
46#define BLOCK_STATIC_THREAD_LOCAL(T, t, ...) \
47 static __thread T* t; \
48 do { \
49 if (PREDICT_FALSE(t == NULL)) { \
50 t = new T(__VA_ARGS__); \
51 add_destructor(destroy<T>, t); \
52 } \
53 } while (false)

55// Class-scoped static thread local implementation.
56//
57// Very similar in implementation to the above block-scoped version, but
58// requires a bit more syntax and vigilance to use properly.
59//
60// DECLARE_STATIC_THREAD_LOCAL(Type, instance_var_) must be placed in the
61// class header, as usual for variable declarations.
62//
63// Because these variables are static, they must also be defined in the impl
64// file with DEFINE_STATIC_THREAD_LOCAL(Type, Classname, instance_var_),
65// which is very much like defining any static member, i.e. int Foo::member_.
66//
67// Finally, each thread must initialize the instance before using it by calling
68// INIT_STATIC_THREAD_LOCAL(Type, instance_var_, ...). This is a cheap
69// call, and may be invoked at the top of any method which may reference a
70// thread-local variable.
71//
72// Due to all of these requirements, you should probably declare TLS members
73// as private.
74//
75// Example usage:
76//
77// // foo.h
78// #include "kudu/utils/file.h"
79// class Foo {
80// public:
81// void DoSomething(std::string s);
82// private:
83// DECLARE_STATIC_THREAD_LOCAL(utils::File, file_);
84// };
85//
86// // foo.cc
87// #include "kudu/foo.h"
88// DEFINE_STATIC_THREAD_LOCAL(utils::File, Foo, file_);
89// void Foo::WriteToFile(std::string s) {
90// // Call constructor if necessary.
91// INIT_STATIC_THREAD_LOCAL(utils::File, file_, "/tmp/file_location.txt");
92// file_->Write(s);
93// }
94
95// Goes in the class declaration (usually in a header file).
96// dtor must be destructed _after_ t, so it gets defined first.
97// Uses a mangled variable name for dtor since it must also be a member of the
98// class.
99#define DECLARE_STATIC_THREAD_LOCAL(T, t) static __thread T* t
100
101// You must also define the instance in the .cc file.
102#define DEFINE_STATIC_THREAD_LOCAL(T, Class, t) __thread T* Class::t
103
104// Must be invoked at least once by each thread that will access t.
105#define INIT_STATIC_THREAD_LOCAL(T, t, ...) \
106 do { \
107 if (PREDICT_FALSE(t == NULL)) { \
108 t = new T(__VA_ARGS__); \
109 add_destructor(destroy<T>, t); \
110 } \
111 } while (false)
