169template <typename Request>
170Status TabletsChannel::_get_current_seq(int64_t& cur_seq, const Request& request) {
171 std::lock_guard<std::mutex> l(_lock);
172 if (_state != kOpened) {
173 return _state == kFinished ? _close_status
174 : Status::InternalError("TabletsChannel {} state: {}",
175 _key.to_string(), _state);
176 }
177 cur_seq = _next_seqs[request.sender_id()];
178 // check packet
179 if (request.packet_seq() > cur_seq) {
180 LOG(WARNING) << "lost data packet, expect_seq=" << cur_seq
181 << ", recept_seq=" << request.packet_seq();
182 return Status::InternalError("lost data packet");
183 }
184 return Status::OK();
185}
