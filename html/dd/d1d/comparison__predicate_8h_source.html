39 Status evaluate(BitmapIndexIterator* iterator, uint32_t num_rows,
40 roaring::Roaring* bitmap) const override {
41 if (iterator == nullptr) {
42 return Status::OK();
43 }
44
45 rowid_t ordinal_limit = iterator->bitmap_nums();
46 if (iterator->has_null_bitmap()) {
47 ordinal_limit--;
48 roaring::Roaring null_bitmap;
49 RETURN_IF_ERROR(iterator->read_null_bitmap(&null_bitmap));
50 *bitmap -= null_bitmap;
51 }

53 roaring::Roaring roaring;
54 bool exact_match;

59 bitmap);

62 Status evaluate(const Schema& schema, InvertedIndexIterator* iterator, uint32_t num_rows,
63 roaring::Roaring* bitmap) const override {
64 if (iterator == nullptr) {
65 return Status::OK();
66 }
67 auto column_desc = schema.column(_column_id);
68 std::string column_name = column_desc->name();
69
70 InvertedIndexQueryType query_type;
71 switch (PT) {
72 case PredicateType::EQ:
73 query_type = InvertedIndexQueryType::EQUAL_QUERY;
74 break;
75 case PredicateType::NE:
76 query_type = InvertedIndexQueryType::EQUAL_QUERY;
77 break;
78 case PredicateType::LT:
79 query_type = InvertedIndexQueryType::LESS_THAN_QUERY;
80 break;
81 case PredicateType::LE:
82 query_type = InvertedIndexQueryType::LESS_EQUAL_QUERY;
83 break;
84 case PredicateType::GT:
85 query_type = InvertedIndexQueryType::GREATER_THAN_QUERY;
86 break;
87 case PredicateType::GE:
88 query_type = InvertedIndexQueryType::GREATER_EQUAL_QUERY;
89 break;
90 default:
91 return Status::InvalidArgument("invalid comparison predicate type {}", PT);
92 }
93
94 roaring::Roaring roaring;
95 RETURN_IF_ERROR(iterator->read_from_inverted_index(column_name, &_value, query_type,
96 num_rows, &roaring));
97
98 if constexpr (PT == PredicateType::NE) {
99 *bitmap -= roaring;
100 } else {
101 *bitmap &= roaring;
102 }
103
104 return Status::OK();
105 }

137 bool evaluate_and(const std::pair<WrapperField*, WrapperField*>& statistic) const override {
138 if (statistic.first->is_null()) {
139 return true;
140 }
141 if constexpr (PT == PredicateType::EQ) {
142 if constexpr (Type == TYPE_DATE) {
143 T tmp_min_uint32_value = 0;
144 memcpy((char*)(&tmp_min_uint32_value), statistic.first->cell_ptr(),
145 sizeof(uint24_t));
146 T tmp_max_uint32_value = 0;
147 memcpy((char*)(&tmp_max_uint32_value), statistic.second->cell_ptr(),
148 sizeof(uint24_t));
149 return _operator(tmp_min_uint32_value <= _value && tmp_max_uint32_value >= _value,
150 true);
151 } else {
152 return _operator(
153 _get_zone_map_value<T>(statistic.first->cell_ptr()) <= _value &&
154 _get_zone_map_value<T>(statistic.second->cell_ptr()) >= _value,
155 true);
156 }
157 } else if constexpr (PT == PredicateType::NE) {
158 if constexpr (Type == TYPE_DATE) {
159 T tmp_min_uint32_value = 0;
160 memcpy((char*)(&tmp_min_uint32_value), statistic.first->cell_ptr(),
161 sizeof(uint24_t));
162 T tmp_max_uint32_value = 0;
163 memcpy((char*)(&tmp_max_uint32_value), statistic.second->cell_ptr(),
164 sizeof(uint24_t));
165 return _operator(tmp_min_uint32_value == _value && tmp_max_uint32_value == _value,
166 true);
167 } else {
168 return _operator(
169 _get_zone_map_value<T>(statistic.first->cell_ptr()) == _value &&
170 _get_zone_map_value<T>(statistic.second->cell_ptr()) == _value,
171 true);
172 }
173 } else if constexpr (PT == PredicateType::LT || PT == PredicateType::LE) {
174 COMPARE_TO_MIN_OR_MAX(first)
175 } else {
176 static_assert(PT == PredicateType::GT || PT == PredicateType::GE);
177 COMPARE_TO_MIN_OR_MAX(second)
178 }
179 }

245 template <bool is_and>
246 __attribute__((flatten)) void _evaluate_vec_internal(const vectorized::IColumn& column,
247 uint16_t size, bool* flags) const {
248 if (column.is_nullable()) {
249 auto* nullable_column_ptr =
250 vectorized::check_and_get_column<vectorized::ColumnNullable>(column);
251 auto& nested_column = nullable_column_ptr->get_nested_column();
252 auto& null_map = reinterpret_cast<const vectorized::ColumnUInt8&>(
253 nullable_column_ptr->get_null_map_column())
254 .get_data();
255
256 if (nested_column.is_column_dictionary()) {
257 if constexpr (std::is_same_v<T, StringRef>) {
258 auto* dict_column_ptr =
259 vectorized::check_and_get_column<vectorized::ColumnDictI32>(
260 nested_column);
261 auto dict_code = _is_range() ? dict_column_ptr->find_code_by_bound(
262 _value, _is_greater(), _is_eq())
263 : dict_column_ptr->find_code(_value);
264 auto* data_array = dict_column_ptr->get_data().data();
265
266 _base_loop_vec<true, is_and>(size, flags, null_map.data(), data_array,
267 dict_code);
268 } else {
269 LOG(FATAL) << "column_dictionary must use StringRef predicate.";
270 }
271 } else {
272 auto* data_array = reinterpret_cast<const vectorized::PredicateColumnType<
273 PredicateEvaluateType<Type>>&>(nested_column)
274 .get_data()
275 .data();
276
277 _base_loop_vec<true, is_and>(size, flags, null_map.data(), data_array, _value);
278 }
279 } else {
280 if (column.is_column_dictionary()) {
281 if constexpr (std::is_same_v<T, StringRef>) {
282 auto* dict_column_ptr =
283 vectorized::check_and_get_column<vectorized::ColumnDictI32>(column);
284 auto dict_code = _is_range() ? dict_column_ptr->find_code_by_bound(
285 _value, _is_greater(), _is_eq())
286 : dict_column_ptr->find_code(_value);
287 auto* data_array = dict_column_ptr->get_data().data();
288
289 _base_loop_vec<false, is_and>(size, flags, nullptr, data_array, dict_code);
290 } else {
291 LOG(FATAL) << "column_dictionary must use StringRef predicate.";
292 }
293 } else {
294 auto* data_array =
295 vectorized::check_and_get_column<
296 vectorized::PredicateColumnType<PredicateEvaluateType<Type>>>(
297 column)
298 ->get_data()
299 .data();
300
301 _base_loop_vec<false, is_and>(size, flags, nullptr, data_array, _value);
302 }
303 }
304
305 if (_opposite) {
306 for (uint16_t i = 0; i < size; i++) {
307 flags[i] = !flags[i];
308 }
309 }
310 }

323 template <typename LeftT, typename RightT>
324 bool _operator(const LeftT& lhs, const RightT& rhs) const {
325 if constexpr (PT == PredicateType::EQ) {
326 return lhs == rhs;
327 } else if constexpr (PT == PredicateType::NE) {
328 return lhs != rhs;
329 } else if constexpr (PT == PredicateType::LT) {
330 return lhs < rhs;
331 } else if constexpr (PT == PredicateType::LE) {
332 return lhs <= rhs;
333 } else if constexpr (PT == PredicateType::GT) {
334 return lhs > rhs;
335 } else if constexpr (PT == PredicateType::GE) {
336 return lhs >= rhs;
337 }
338 }
339
340 constexpr bool _is_range() const { return PredicateTypeTraits::is_range(PT); }
341
342 constexpr bool _is_greater() const { return _operator(1, 0); }
343
344 constexpr bool _is_eq() const { return _operator(1, 1); }
345
346 Status _bitmap_compare(Status status, bool exact_match, rowid_t ordinal_limit,
347 rowid_t& seeked_ordinal, BitmapIndexIterator* iterator,
348 roaring::Roaring* bitmap) const {
349 roaring::Roaring roaring;
350
351 if (status.is<ErrorCode::NOT_FOUND>()) {
352 if constexpr (PT == PredicateType::EQ || PT == PredicateType::GT ||
353 PT == PredicateType::GE) {
354 *bitmap &= roaring; // set bitmap to empty
355 }
356 return Status::OK();
357 }
358
359 if (!status.ok()) {
360 return status;
361 }
362
363 if constexpr (PT == PredicateType::EQ || PT == PredicateType::NE) {
364 if (exact_match) {
365 RETURN_IF_ERROR(iterator->read_bitmap(seeked_ordinal, &roaring));
366 }
367 } else if constexpr (PredicateTypeTraits::is_range(PT)) {
368 rowid_t from = 0;
369 rowid_t to = ordinal_limit;
370 if constexpr (PT == PredicateType::LT) {
371 to = seeked_ordinal;
372 } else if constexpr (PT == PredicateType::LE) {
373 to = seeked_ordinal + exact_match;
374 } else if constexpr (PT == PredicateType::GT) {
375 from = seeked_ordinal + exact_match;
376 } else if constexpr (PT == PredicateType::GE) {
377 from = seeked_ordinal;
378 }
379
380 RETURN_IF_ERROR(iterator->read_union_bitmap(from, to, &roaring));
381 }
382
383 if constexpr (PT == PredicateType::NE) {
384 *bitmap -= roaring;
385 } else {
386 *bitmap &= roaring;
387 }
388
389 return Status::OK();
390 }

409 template <bool is_nullable, bool is_and, typename TArray, typename TValue>
410 __attribute__((flatten)) void _base_loop_vec(uint16_t size, bool* __restrict bflags,
411 const uint8_t* __restrict null_map,
412 const TArray* __restrict data_array,
413 const TValue& value) const {
414 //uint8_t helps compiler to generate vectorized code
415 uint8_t* flags = reinterpret_cast<uint8_t*>(bflags);
416 if constexpr (is_and) {
417 for (uint16_t i = 0; i < size; i++) {
418 if constexpr (is_nullable) {
419 flags[i] &= (uint8_t)(!null_map[i] && _operator(data_array[i], value));
420 } else {
421 flags[i] &= (uint8_t)_operator(data_array[i], value);
422 }
423 }
424 } else {
425 for (uint16_t i = 0; i < size; i++) {
426 if constexpr (is_nullable) {
427 flags[i] = !null_map[i] && _operator(data_array[i], value);
428 } else {
429 flags[i] = _operator(data_array[i], value);
430 }
431 }
432 }
433 }

435 template <bool is_nullable, bool is_and, typename TArray, typename TValue>
436 void _base_loop_bit(const uint16_t* sel, uint16_t size, bool* flags,
437 const uint8_t* __restrict null_map, const TArray* __restrict data_array,
438 const TValue& value) const {
439 for (uint16_t i = 0; i < size; i++) {
440 if (is_and ^ flags[i]) {
441 continue;
442 }
443 if constexpr (is_nullable) {
444 if (_opposite ^ is_and ^
445 (!null_map[sel[i]] && _operator(data_array[sel[i]], value))) {
446 flags[i] = !is_and;
447 }
448 } else {
449 if (_opposite ^ is_and ^ _operator(data_array[sel[i]], value)) {
450 flags[i] = !is_and;
451 }
452 }
453 }
454 }

456 template <bool is_nullable, bool is_and>
457 void _base_evaluate_bit(const vectorized::IColumn* column, const uint8_t* null_map,
458 const uint16_t* sel, uint16_t size, bool* flags) const {
459 if (column->is_column_dictionary()) {
460 if constexpr (std::is_same_v<T, StringRef>) {
461 auto* dict_column_ptr =
462 vectorized::check_and_get_column<vectorized::ColumnDictI32>(column);
463 auto* data_array = dict_column_ptr->get_data().data();
464 auto dict_code = _is_range() ? dict_column_ptr->find_code_by_bound(
465 _value, _operator(1, 0), _operator(1, 1))
466 : dict_column_ptr->find_code(_value);
467 _base_loop_bit<is_nullable, is_and>(sel, size, flags, null_map, data_array,
468 dict_code);
469 } else {
470 LOG(FATAL) << "column_dictionary must use StringRef predicate.";
471 }
472 } else {
473 auto* data_array =
474 vectorized::check_and_get_column<
475 vectorized::PredicateColumnType<PredicateEvaluateType<Type>>>(column)
476 ->get_data()
477 .data();
478
479 _base_loop_bit<is_nullable, is_and>(sel, size, flags, null_map, data_array, _value);
480 }

483 template <bool is_nullable, typename TArray, typename TValue>
484 uint16_t _base_loop(uint16_t* sel, uint16_t size, const uint8_t* __restrict null_map,
485 const TArray* __restrict data_array, const TValue& value) const {
486 uint16_t new_size = 0;
487 for (uint16_t i = 0; i < size; ++i) {
488 uint16_t idx = sel[i];
489 if constexpr (is_nullable) {
490 if (_opposite ^ (!null_map[idx] && _operator(data_array[idx], value))) {
491 sel[new_size++] = idx;
492 }
493 } else {
494 if (_opposite ^ _operator(data_array[idx], value)) {
495 sel[new_size++] = idx;
496 }
497 }
498 }
499 return new_size;
500 }

502 template <bool is_nullable>
503 uint16_t _base_evaluate(const vectorized::IColumn* column, const uint8_t* null_map,
504 uint16_t* sel, uint16_t size) const {
505 if (column->is_column_dictionary()) {
506 if constexpr (std::is_same_v<T, StringRef>) {
507 auto* dict_column_ptr =
508 vectorized::check_and_get_column<vectorized::ColumnDictI32>(column);
509 auto* data_array = dict_column_ptr->get_data().data();
510 auto dict_code = _is_range() ? dict_column_ptr->find_code_by_bound(
511 _value, _is_greater(), _is_eq())
512 : dict_column_ptr->find_code(_value);
513
514 return _base_loop<is_nullable>(sel, size, null_map, data_array, dict_code);
515 } else {
516 LOG(FATAL) << "column_dictionary must use StringRef predicate.";
517 return 0;
518 }
519 } else {
520 auto* data_array =
521 vectorized::check_and_get_column<
522 vectorized::PredicateColumnType<PredicateEvaluateType<Type>>>(column)
523 ->get_data()
524 .data();
525
526 return _base_loop<is_nullable>(sel, size, null_map, data_array, _value);
527 }
