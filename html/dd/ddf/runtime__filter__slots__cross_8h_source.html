32namespace doris {
33// this class used in cross join node
34template <typename ExprCtxType = vectorized::VExprContext>
35class RuntimeFilterSlotsCross {
36public:
37 RuntimeFilterSlotsCross(const std::vector<TRuntimeFilterDesc>& runtime_filter_descs,
38 const std::vector<vectorized::VExprContext*>& src_expr_ctxs)
39 : _runtime_filter_descs(runtime_filter_descs), filter_src_expr_ctxs(src_expr_ctxs) {}
40
41 ~RuntimeFilterSlotsCross() = default;
42
43 Status init(RuntimeState* state) {
44 for (auto& filter_desc : _runtime_filter_descs) {
45 IRuntimeFilter* runtime_filter = nullptr;
46 RETURN_IF_ERROR(state->runtime_filter_mgr()->get_producer_filter(filter_desc.filter_id,
47 &runtime_filter));
48 DCHECK(runtime_filter != nullptr);
49 // cross join has not remote filter
50 DCHECK(!runtime_filter->has_remote_target());
51 _runtime_filters.push_back(runtime_filter);
52 }
53 return Status::OK();
54 }
55
56 Status insert(vectorized::Block* block) {
57 for (int i = 0; i < _runtime_filters.size(); ++i) {
58 auto* filter = _runtime_filters[i];
59 auto* vexpr_ctx = filter_src_expr_ctxs[i];
60
61 int result_column_id = -1;
62 RETURN_IF_ERROR(vexpr_ctx->execute(block, &result_column_id));
63 DCHECK(result_column_id != -1);
64 block->get_by_position(result_column_id).column =
65 block->get_by_position(result_column_id)
66 .column->convert_to_full_column_if_const();
67
68 auto& column = block->get_by_position(result_column_id).column;
69 if (auto* nullable =
70 vectorized::check_and_get_column<vectorized::ColumnNullable>(*column)) {
71 auto& column_nested = nullable->get_nested_column_ptr();
72 auto& column_nullmap = nullable->get_null_map_column_ptr();
73 std::vector<int> indexs;
74 for (int row_index = 0; row_index < column->size(); ++row_index) {
75 if (assert_cast<const vectorized::ColumnUInt8*>(column_nullmap.get())
76 ->get_bool(row_index)) {
77 continue;
78 }
79 indexs.push_back(row_index);
80 }
81 filter->insert_batch(column_nested, indexs);
82 } else {
83 std::vector<int> rows(column->size());
84 std::iota(rows.begin(), rows.end(), 0);
85 filter->insert_batch(column, rows);
86 }
87 }
88 return Status::OK();
89 }
90
91 void publish() {
92 for (auto& filter : _runtime_filters) {
93 filter->publish();
94 }
95 for (auto& filter : _runtime_filters) {
96 // todo: cross join may not need publish_finally()
97 filter->publish_finally();
98 }
99 }
100
101 bool empty() { return !_runtime_filters.size(); }
102
103private:
104 const std::vector<TRuntimeFilterDesc>& _runtime_filter_descs;
105 const std::vector<vectorized::VExprContext*> filter_src_expr_ctxs;
106 std::vector<IRuntimeFilter*> _runtime_filters;
107};
108
109using VRuntimeFilterSlotsCross = RuntimeFilterSlotsCross<vectorized::VExprContext>;
110} // namespace doris
