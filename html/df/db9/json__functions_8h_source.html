39struct JsonPath {
40 std::string key; // key of a json object
41 int idx; // array index of a json array, -1 means not set, -2 means *
42 bool is_valid; // true if the path is successfully parsed
43
44 JsonPath(const std::string& key_, int idx_, bool is_valid_)
45 : key(key_), idx(idx_), is_valid(is_valid_) {}
46
47 JsonPath(std::string&& key_, int idx_, bool is_valid_)
48 : key(std::move(key_)), idx(idx_), is_valid(is_valid_) {}
49
50 std::string to_string() const {
51 std::stringstream ss;
52 if (!is_valid) {
53 return "INVALID";
54 }
55 if (!key.empty()) {
56 ss << key;
57 }
58 if (idx == -2) {
59 ss << "[*]";
60 } else if (idx > -1) {
61 ss << "[" << idx << "]";
62 }
63 return ss.str();
64 }
65
66 std::string to_simdjson_pointer(bool* valid) const {
67 std::stringstream ss;
68 if (!is_valid) {
69 *valid = false;
70 return "";
71 }
72 ss << "/";
73 if (!key.empty()) {
74 ss << key;
75 }
76 if (idx == -2) {
77 // not support [*]
78 *valid = false;
79 return "";
80 } else if (idx > -1) {
81 ss << "/" << idx;
82 }
83 return ss.str();
84 }
85
86 std::string debug_string() const {
87 return fmt::format("key:{}, idx:{}, valid:{}", key, idx, is_valid);
88 }
89};
