41struct JsonPath {
42 std::string key; // key of a json object
43 int idx; // array index of a json array, -1 means not set, -2 means *
44 bool is_valid; // true if the path is successfully parsed
45
46 JsonPath(const std::string& key_, int idx_, bool is_valid_)
47 : key(key_), idx(idx_), is_valid(is_valid_) {}
48
49 JsonPath(std::string&& key_, int idx_, bool is_valid_)
50 : key(std::move(key_)), idx(idx_), is_valid(is_valid_) {}
51
52 std::string to_string() const {
53 std::stringstream ss;
54 if (!is_valid) {
55 return "INVALID";
56 }
57 if (!key.empty()) {
58 ss << key;
59 }
60 if (idx == -2) {
61 ss << "[*]";
62 } else if (idx > -1) {
63 ss << "[" << idx << "]";
64 }
65 return ss.str();
66 }
67
68 std::string debug_string() const {
69 return fmt::format("key:{}, idx:{}, valid:{}", key, idx, is_valid);
70 }
71};
