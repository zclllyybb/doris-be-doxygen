91inline bool memequalSSE2Wide(const char* p1, const char* p2, size_t size) {
98 if (size <= 16) {
99 if (size >= 8) {
101 return unaligned_load<uint64_t>(p1) == unaligned_load<uint64_t>(p2) &&
102 unaligned_load<uint64_t>(p1 + size - 8) ==
103 unaligned_load<uint64_t>(p2 + size - 8);
104 } else if (size >= 4) {
106 return unaligned_load<uint32_t>(p1) == unaligned_load<uint32_t>(p2) &&
107 unaligned_load<uint32_t>(p1 + size - 4) ==
108 unaligned_load<uint32_t>(p2 + size - 4);
109 } else if (size >= 2) {
111 return unaligned_load<uint16_t>(p1) == unaligned_load<uint16_t>(p2) &&
112 unaligned_load<uint16_t>(p1 + size - 2) ==
113 unaligned_load<uint16_t>(p2 + size - 2);
114 } else if (size >= 1) {
116 return *p1 == *p2;
117 }
118 return true;
119 }
120
121 while (size >= 64) {
122 if (compareSSE2x4(p1, p2)) {
123 p1 += 64;
124 p2 += 64;
125 size -= 64;
126 } else {
127 return false;
128 }
129 }
130
131 switch (size / 16) {
132 case 3:
133 if (!compareSSE2(p1 + 32, p2 + 32)) {
134 return false;
135 }
136 [[fallthrough]];
137 case 2:
138 if (!compareSSE2(p1 + 16, p2 + 16)) {
139 return false;
140 }
141 [[fallthrough]];
142 case 1:
143 if (!compareSSE2(p1, p2)) {
144 return false;
145 }
146 }
147
148 return compareSSE2(p1 + size - 16, p2 + size - 16);
149}

153// Compare two strings using sse4.2 intrinsics if they are available. This code assumes
154// that the trivial cases are already handled (i.e. one string is empty).
155// Returns:
156// < 0 if s1 < s2
157// 0 if s1 == s2
158// > 0 if s1 > s2
159// The SSE code path is just under 2x faster than the non-sse code path.
160// - s1/n1: ptr/len for the first string
161// - s2/n2: ptr/len for the second string
162// - len: min(n1, n2) - this can be more cheaply passed in by the caller
163inline int string_compare(const char* s1, int64_t n1, const char* s2, int64_t n2, int64_t len) {
164 DCHECK_EQ(len, std::min(n1, n2));
165#ifdef __SSE4_2__
166 while (len >= sse_util::CHARS_PER_128_BIT_REGISTER) {
167 __m128i xmm0 = _mm_loadu_si128(reinterpret_cast<const __m128i*>(s1));
168 __m128i xmm1 = _mm_loadu_si128(reinterpret_cast<const __m128i*>(s2));
169 int chars_match = _mm_cmpestri(xmm0, sse_util::CHARS_PER_128_BIT_REGISTER, xmm1,
170 sse_util::CHARS_PER_128_BIT_REGISTER, sse_util::STRCMP_MODE);
171 if (chars_match != sse_util::CHARS_PER_128_BIT_REGISTER) {
172 return (unsigned char)s1[chars_match] - (unsigned char)s2[chars_match];
173 }
174 len -= sse_util::CHARS_PER_128_BIT_REGISTER;
175 s1 += sse_util::CHARS_PER_128_BIT_REGISTER;
176 s2 += sse_util::CHARS_PER_128_BIT_REGISTER;
177 }
178#endif
179 unsigned char u1, u2;
180 while (len-- > 0) {
181 u1 = (unsigned char)*s1++;
182 u2 = (unsigned char)*s2++;
183 if (u1 != u2) {
184 return u1 - u2;
185 }
186 if (u1 == '\0') {
187 return n1 - n2;
188 }
189 }
190
191 return n1 - n2;
192}

255 // Byte-by-byte comparison. Returns:
256 // this < other: -1
257 // this == other: 0
258 // this > other: 1
259 int compare(const StringRef& other) const {
260 int l = std::min(size, other.size);
261
262 if (l == 0) {
263 if (size == other.size) {
264 return 0;
265 } else if (size == 0) {
266 return -1;
267 } else {
268 DCHECK_EQ(other.size, 0);
269 return 1;
270 }
271 }
272
273 return string_compare(this->data, this->size, other.data, other.size, l);
274 }

358inline size_t hash_less_than8(const char* data, size_t size) {
359 static constexpr doris::vectorized::UInt64 k2 = 0x9ae16a3b2f90404fULL;
360 static constexpr doris::vectorized::UInt64 k3 = 0xc949d7c7509e6557ULL;
361
362 if (size >= 4) {
363 doris::vectorized::UInt64 a = unaligned_load<uint32_t>(data);
364 return hash_len16(size + (a << 3), unaligned_load<uint32_t>(data + size - 4));
365 }
366
367 if (size > 0) {
368 uint8_t a = data[0];
369 uint8_t b = data[size >> 1];
370 uint8_t c = data[size - 1];
371 uint32_t y = static_cast<uint32_t>(a) + (static_cast<uint32_t>(b) << 8);
372 uint32_t z = size + (static_cast<uint32_t>(c) << 2);
373 return shift_mix(y * k2 ^ z * k3) * k2;
374 }
375
376 return k2;
377}

379inline size_t hash_less_than16(const char* data, size_t size) {
380 if (size > 8) {
381 doris::vectorized::UInt64 a = unaligned_load<doris::vectorized::UInt64>(data);
382 doris::vectorized::UInt64 b = unaligned_load<doris::vectorized::UInt64>(data + size - 8);
383 return hash_len16(a, rotate_by_at_least1(b + size, size)) ^ b;
384 }
385
386 return hash_less_than8(data, size);
387}

389struct CRC32Hash {
390 size_t operator()(const StringRef& x) const {
391 const char* pos = x.data;
392 size_t size = x.size;
393
394 if (size == 0) {
395 return 0;
396 }
397
398 if (size < 8) {
399 return hash_less_than8(x.data, x.size);
400 }
401
402 const char* end = pos + size;
403 size_t res = -1ULL;
404
405 do {
406 doris::vectorized::UInt64 word = unaligned_load<doris::vectorized::UInt64>(pos);
407 res = _mm_crc32_u64(res, word);
408
409 pos += 8;
410 } while (pos + 8 < end);
411
412 doris::vectorized::UInt64 word = unaligned_load<doris::vectorized::UInt64>(
413 end - 8);
414 res = _mm_crc32_u64(res, word);
415
416 return res;
417 }
418};
