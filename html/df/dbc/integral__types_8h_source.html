15// These typedefs are also defined in base/google.swig. In the
16// SWIG environment, we use those definitions and avoid duplicate
17// definitions here with an ifdef. The definitions should be the
18// same in both files, and ideally be only defined in this file.

20// Standard typedefs
21// All Google2 code is compiled with -funsigned-char to make "char"
22// unsigned. Google2 code therefore doesn't need a "uchar" type.
23typedef int8_t schar;

33// NOTE: unsigned types are DANGEROUS in loops and other arithmetical
34// places. Use the signed types unless your variable represents a bit
35// pattern (eg a hash value) or you really need the extra bit. Do NOT
36// use 'unsigned' to express "this value should always be positive";
37// use assertions for this.
38
39typedef uint8_t uint8;
40typedef uint16_t uint16;
41typedef uint32_t uint32;
42#ifdef _MSC_VER
43typedef unsigned __int64 uint64;
44#else

48// A type to represent a Unicode code-point value. As of Unicode 4.0,
49// such values require up to 21 bits.
50// (For type-checking on pointers, make this explicitly signed,
51// and it should always be the signed version of whatever int32 is.)
52typedef signed int char32;
53
54// A type to represent a natural machine word (for e.g. efficiently
55// scanning through memory for checksums or index searching). Don't use
56// this for storing normal integers. Ideally this would be just
57// unsigned int, but our 64-bit architectures use the LP64 model
58// (http://www.opengroup.org/public/tech/aspen/lp64_wp.htm), hence
59// their ints are only 32 bits. We want to use the same fundamental
60// type on all archs if possible to preserve *printf() compatability.
61typedef unsigned long uword_t;

63#endif /* SWIG */
