69// similar to reinterpret_cast but won't break strict-aliasing rules
70template <typename From, typename To>
71To binary_cast(From from) {
72 constexpr bool from_u64_to_db = match_v<From, uint64_t, To, double>;
73 constexpr bool from_i64_to_db = match_v<From, int64_t, To, double>;
74 constexpr bool from_db_to_i64 = match_v<From, double, To, int64_t>;
75 constexpr bool from_db_to_u64 = match_v<From, double, To, uint64_t>;
76 constexpr bool from_i64_to_vec_dt =
77 match_v<From, __int64_t, To, doris::vectorized::VecDateTimeValue>;
78 constexpr bool from_vec_dt_to_i64 =
79 match_v<From, doris::vectorized::VecDateTimeValue, To, __int64_t>;
80 constexpr bool from_i128_to_decv2 = match_v<From, __int128_t, To, DecimalV2Value>;
81 constexpr bool from_decv2_to_i128 = match_v<From, DecimalV2Value, To, __int128_t>;
82
83 constexpr bool from_ui32_to_date_v2 =
84 match_v<From, uint32_t, To,
85 doris::vectorized::DateV2Value<doris::vectorized::DateV2ValueType>>;
86
87 constexpr bool from_date_v2_to_ui32 =
88 match_v<From, doris::vectorized::DateV2Value<doris::vectorized::DateV2ValueType>, To,
89 uint32_t>;
90
91 constexpr bool from_ui64_to_datetime_v2 =
92 match_v<From, uint64_t, To,
93 doris::vectorized::DateV2Value<doris::vectorized::DateTimeV2ValueType>>;
94
95 constexpr bool from_datetime_v2_to_ui64 =
96 match_v<From, doris::vectorized::DateV2Value<doris::vectorized::DateTimeV2ValueType>,
97 To, uint64_t>;
98
99 static_assert(from_u64_to_db || from_i64_to_db || from_db_to_i64 || from_db_to_u64 ||
100 from_i64_to_vec_dt || from_vec_dt_to_i64 || from_i128_to_decv2 ||
101 from_decv2_to_i128 || from_ui32_to_date_v2 || from_date_v2_to_ui32 ||
102 from_ui64_to_datetime_v2 || from_datetime_v2_to_ui64);
103
104 if constexpr (from_u64_to_db) {
105 TypeConverter conv;
106 conv.u64 = from;
107 return conv.dbl;
108 } else if constexpr (from_i64_to_db) {
109 TypeConverter conv;
110 conv.i64 = from;
111 return conv.dbl;
112 } else if constexpr (from_db_to_i64) {
113 TypeConverter conv;
114 conv.dbl = from;
115 return conv.i64;
116 } else if constexpr (from_db_to_u64) {
117 TypeConverter conv;
118 conv.dbl = from;
119 return conv.u64;
120 } else if constexpr (from_i64_to_vec_dt) {
121 VecDateTimeInt64Union conv = {.i64 = from};
122 return conv.dt;
123 } else if constexpr (from_ui32_to_date_v2) {
124 DateV2UInt32Union conv = {.ui32 = from};
125 return conv.dt;
126 } else if constexpr (from_date_v2_to_ui32) {
127 DateV2UInt32Union conv = {.dt = from};
128 return conv.ui32;
129 } else if constexpr (from_ui64_to_datetime_v2) {
130 DateTimeV2UInt64Union conv = {.ui64 = from};
131 return conv.dt;
132 } else if constexpr (from_datetime_v2_to_ui64) {
133 DateTimeV2UInt64Union conv = {.dt = from};
134 return conv.ui64;
135 } else if constexpr (from_vec_dt_to_i64) {
136 VecDateTimeInt64Union conv = {.dt = from};
137 return conv.i64;
138 } else if constexpr (from_i128_to_decv2) {
139 DecimalInt128Union conv;
140 conv.i128 = from;
141 return conv.decimal;
142 } else if constexpr (from_decv2_to_i128) {
143 DecimalInt128Union conv;
144 conv.decimal = from;
145 return conv.i128;
146 } else {
147 __builtin_unreachable();
148 }
149}
