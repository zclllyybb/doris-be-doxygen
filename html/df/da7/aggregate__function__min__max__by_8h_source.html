54template <typename VT, typename KT>
55struct AggregateFunctionMaxByData : public AggregateFunctionMinMaxByBaseData<VT, KT> {
56 using Self = AggregateFunctionMaxByData;
57 void change_if_better(const IColumn& value_column, const IColumn& key_column, size_t row_num,
58 Arena* arena) {
59 if (this->key.change_if_greater(key_column, row_num, arena)) {
60 this->value.change(value_column, row_num, arena);
61 }
62 }
63
64 void change_if_better(const Self& to, Arena* arena) {
65 if (this->key.change_if_greater(to.key, arena)) {
66 this->value.change(to.value, arena);
67 }
68 }
69
70 static const char* name() { return "max_by"; }
71};

92template <typename Data>
93class AggregateFunctionsMinMaxBy final
94 : public IAggregateFunctionDataHelper<Data, AggregateFunctionsMinMaxBy<Data>> {
95private:
96 DataTypePtr& value_type;
97 DataTypePtr& key_type;
98
99public:
100 AggregateFunctionsMinMaxBy(const DataTypes& arguments)
101 : IAggregateFunctionDataHelper<Data, AggregateFunctionsMinMaxBy<Data>>(
102 {arguments[0], arguments[1]}),
103 value_type(this->argument_types[0]),
104 key_type(this->argument_types[1]) {
105 if (StringRef(Data::name()) == StringRef("min_by") ||
106 StringRef(Data::name()) == StringRef("max_by")) {
107 CHECK(key_type->is_comparable());
108 }
109 }
110
111 String get_name() const override { return Data::name(); }
112
113 DataTypePtr get_return_type() const override { return value_type; }
114
115 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
116 Arena* arena) const override {
117 this->data(place).change_if_better(*columns[0], *columns[1], row_num, arena);
118 }
119
120 void reset(AggregateDataPtr place) const override { this->data(place).reset(); }
121
122 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
123 Arena* arena) const override {
124 this->data(place).change_if_better(this->data(rhs), arena);
125 }
126
127 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
128 this->data(place).write(buf);
129 }
130
131 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
132 Arena*) const override {
133 this->data(place).read(buf);
134 }
135
136 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
137 this->data(place).insert_result_into(to);
138 }
139};
