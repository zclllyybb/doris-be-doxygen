52template <typename VT, typename KT>
53struct AggregateFunctionMaxByData : public AggregateFunctionMinMaxByBaseData<VT, KT> {
54 using Self = AggregateFunctionMaxByData;
55 bool change_if_better(const IColumn& value_column, const IColumn& key_column, size_t row_num,
56 Arena* arena) {
57 if (this->key.change_if_greater(key_column, row_num, arena)) {
58 this->value.change(value_column, row_num, arena);
59 return true;
60 }
61 return false;
62 }
63
64 bool change_if_better(const Self& to, Arena* arena) {
65 if (this->key.change_if_greater(to.key, arena)) {
66 this->value.change(to.value, arena);
67 return true;
68 }
69 return false;
70 }
71
72 static const char* name() { return "max_by"; }
73};

98template <typename Data, bool AllocatesMemoryInArena>
99class AggregateFunctionsMinMaxBy final
100 : public IAggregateFunctionDataHelper<
101 Data, AggregateFunctionsMinMaxBy<Data, AllocatesMemoryInArena>> {
102private:
103 DataTypePtr& value_type;
104 DataTypePtr& key_type;
105
106public:
107 AggregateFunctionsMinMaxBy(const DataTypePtr& value_type_, const DataTypePtr& key_type_)
108 : IAggregateFunctionDataHelper<
109 Data, AggregateFunctionsMinMaxBy<Data, AllocatesMemoryInArena>>(
110 {value_type_, key_type_}, {}),
111 value_type(this->argument_types[0]),
112 key_type(this->argument_types[1]) {
113 if (StringRef(Data::name()) == StringRef("min_by") ||
114 StringRef(Data::name()) == StringRef("max_by")) {
115 CHECK(key_type_->is_comparable());
116 }
117 }
118
119 String get_name() const override { return Data::name(); }
120
121 DataTypePtr get_return_type() const override { return value_type; }
122
123 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
124 Arena* arena) const override {
125 this->data(place).change_if_better(*columns[0], *columns[1], row_num, arena);
126 }
127
128 void reset(AggregateDataPtr place) const override { this->data(place).reset(); }
129
130 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
131 Arena* arena) const override {
132 this->data(place).change_if_better(this->data(rhs), arena);
133 }
134
135 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
136 this->data(place).write(buf);
137 }
138
139 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
140 Arena*) const override {
141 this->data(place).read(buf);
142 }
143
144 bool allocates_memory_in_arena() const override { return AllocatesMemoryInArena; }
145
146 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
147 this->data(place).insert_result_into(to);
148 }
149};
