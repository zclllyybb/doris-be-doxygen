56class ColumnChunkReader {
57public:
58 struct Statistics {
59 int64_t decompress_time = 0;
60 int64_t decompress_cnt = 0;
61 int64_t decode_header_time = 0;
62 int64_t decode_value_time = 0;
63 int64_t decode_dict_time = 0;
64 int64_t decode_level_time = 0;
65 };
66
67 ColumnChunkReader(BufferedStreamReader* reader, tparquet::ColumnChunk* column_chunk,
68 FieldSchema* field_schema, cctz::time_zone* ctz);
69 ~ColumnChunkReader() = default;
70
71 // Initialize chunk reader, will generate the decoder and codec.
72 Status init();
73
74 // Whether the chunk reader has a more page to read.
75 bool has_next_page() { return _page_reader->has_next_page(); }
76
77 // Seek to the specific page, page_header_offset must be the start offset of the page header.
78 void seek_to_page(int64_t page_header_offset) {
79 _remaining_num_values = 0;
80 _page_reader->seek_to_page(page_header_offset);
81 _state = INITIALIZED;
82 }
83
84 // Seek to next page. Only read and parse the page header.
85 Status next_page();
86
87 // Skip current page(will not read and parse) if the page is filtered by predicates.
88 Status skip_page() {
89 _remaining_num_values = 0;
90 if (_state == HEADER_PARSED) {
91 return _page_reader->skip_page();
92 }
93 if (_state != DATA_LOADED) {
94 return Status::Corruption("Should parse page header to skip page");
95 }
96 return Status::OK();
97 }
98 // Skip some values(will not read and parse) in current page if the values are filtered by predicates.
99 // when skip_data = false, the underlying decoder will not skip data,
100 // only used when maintaining the consistency of _remaining_num_values.
101 Status skip_values(size_t num_values, bool skip_data = true);
102
103 // Load page data into the underlying container,
104 // and initialize the repetition and definition level decoder for current page data.
105 Status load_page_data();
106 Status load_page_data_idempotent() {
107 if (_state == DATA_LOADED) {
108 return Status::OK();
109 }
110 return load_page_data();
111 }
112 // The remaining number of values in current page(including null values). Decreased when reading or skipping.
113 uint32_t remaining_num_values() const { return _remaining_num_values; }
114 // null values are generated from definition levels
115 // the caller should maintain the consistency after analyzing null values from definition levels.
116 void insert_null_values(MutableColumnPtr& doris_column, size_t num_values);
117 // Get the raw data of current page.
118 Slice& get_page_data() { return _page_data; }
119
120 // Get the repetition levels
121 size_t get_rep_levels(level_t* levels, size_t n);
122 // Get the definition levels
123 size_t get_def_levels(level_t* levels, size_t n);
124
125 // Decode values in current page into doris column.
126 Status decode_values(MutableColumnPtr& doris_column, DataTypePtr& data_type,
127 ColumnSelectVector& select_vector);
128
129 // Get the repetition level decoder of current page.
130 LevelDecoder& rep_level_decoder() { return _rep_level_decoder; }
131 // Get the definition level decoder of current page.
132 LevelDecoder& def_level_decoder() { return _def_level_decoder; }
133
134 level_t max_rep_level() const { return _max_rep_level; }
135 level_t max_def_level() const { return _max_def_level; }
136
137 // Get page decoder
138 Decoder* get_page_decoder() { return _page_decoder; }
139
140 Statistics& statistics() {
141 _statistics.decode_header_time = _page_reader->statistics().decode_header_time;
142 return _statistics;
143 }
144
145private:
146 enum ColumnChunkReaderState { NOT_INIT, INITIALIZED, HEADER_PARSED, DATA_LOADED };
147
148 Status _decode_dict_page();
149 void _reserve_decompress_buf(size_t size);
150 int32_t _get_type_length();
151
152 ColumnChunkReaderState _state = NOT_INIT;
153 FieldSchema* _field_schema;
154 level_t _max_rep_level;
155 level_t _max_def_level;
156 tparquet::LogicalType _parquet_logical_type;
157
158 BufferedStreamReader* _stream_reader;
159 tparquet::ColumnMetaData _metadata;
160 cctz::time_zone* _ctz;
161
162 std::unique_ptr<PageReader> _page_reader = nullptr;
163 BlockCompressionCodec* _block_compress_codec = nullptr;
164
165 LevelDecoder _rep_level_decoder;
166 LevelDecoder _def_level_decoder;
167 uint32_t _remaining_num_values = 0;
168 Slice _page_data;
169 std::unique_ptr<uint8_t[]> _decompress_buf;
170 size_t _decompress_buf_size = 0;
171 bool _has_dict = false;
172 Decoder* _page_decoder = nullptr;
173 // Map: encoding -> Decoder
174 // Plain or Dictionary encoding. If the dictionary grows too big, the encoding will fall back to the plain encoding
175 std::unordered_map<int, std::unique_ptr<Decoder>> _decoders;
176 Statistics _statistics;
177};
