26// data consumer group saves a group of data consumers.
27// These data consumers share the same stream load pipe.
28// This class is not thread safe.
29class DataConsumerGroup {
30public:
31 typedef std::function<void(const Status&)> ConsumeFinishCallback;
32
33 DataConsumerGroup()
34 : _grp_id(UniqueId::gen_uid()), _thread_pool(3, 10, "data_consumer"), _counter(0) {}
35
36 virtual ~DataConsumerGroup() { _consumers.clear(); }
37
38 const UniqueId& grp_id() { return _grp_id; }
39
40 const std::vector<std::shared_ptr<DataConsumer>>& consumers() { return _consumers; }
41
42 void add_consumer(std::shared_ptr<DataConsumer> consumer) {
43 consumer->set_grp(_grp_id);
44 _consumers.push_back(consumer);
45 ++_counter;
46 }
47
48 // start all consumers
49 virtual Status start_all(StreamLoadContext* ctx) { return Status::OK(); }
50
51protected:
52 UniqueId _grp_id;
53 std::vector<std::shared_ptr<DataConsumer>> _consumers;
54 // thread pool to run each consumer in multi thread
55 PriorityThreadPool _thread_pool;
56 // mutex to protect counter.
57 // the counter is init as the number of consumers.
58 // once a consumer is done, decrease the counter.
59 // when the counter becomes zero, shutdown the queue to finish
60 std::mutex _mutex;
61 int _counter;
62};
