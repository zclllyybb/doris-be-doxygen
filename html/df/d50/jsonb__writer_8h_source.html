11/*
12 * This file defines JsonbWriterT (template) and JsonbWriter.
13 *
14 * JsonbWriterT is a template class which implements an JSONB serializer.
15 * Users call various write functions of JsonbWriterT object to write values
16 * directly to JSONB packed bytes. All write functions of value or key return
17 * the number of bytes written to JSONB, or 0 if there is an error. To write an
18 * object, an array, or a string, you must call writeStart[..] before writing
19 * values or key, and call writeEnd[..] after finishing at the end.
20 *
21 * By default, an JsonbWriterT object creates an output stream buffer.
22 * Alternatively, you can also pass any output stream object to a writer, as
23 * long as the stream object implements some basic functions of std::ostream
24 * (such as JsonbOutStream, see JsonbStream.h).
25 *
26 * JsonbWriter specializes JsonbWriterT with JsonbOutStream type (see
27 * JsonbStream.h). So unless you want to provide own a different output stream
28 * type, use JsonbParser object.
29 *
30 * @author Tian Xia <tianx@fb.com>
31 * this file is copied from
32 * https://github.com/facebook/mysql-5.6/blob/fb-mysql-5.6.35/fbson/FbsonWriter.h
33 * and modified by Doris
34 */
35

78 // write a key string (or key id if an external dict is provided)
79 uint32_t writeKey(const char* key, uint8_t len, hDictInsert handler = nullptr) {
80 if (len && !stack_.empty() && verifyKeyState()) {
81 int key_id = -1;
82 if (handler) {
83 key_id = handler(key, len);
84 }
85
86 uint32_t size = sizeof(uint8_t);
87 if (key_id < 0) {
88 os_->put(len);
89 os_->write(key, len);
90 size += len;
91 } else if (key_id <= JsonbKeyValue::sMaxKeyId) {
92 JsonbKeyValue::keyid_type idx = key_id;
93 os_->put(0);
94 os_->write((char*)&idx, sizeof(JsonbKeyValue::keyid_type));
95 size += sizeof(JsonbKeyValue::keyid_type);
96 } else { // key id overflow
97 assert(0);
98 return 0;
99 }
100
101 kvState_ = WS_Key;
102 return size;
103 }
104
105 return 0;
106 }

272 // must call writeStartString before writing a string val
273 bool writeStartString() {
274 if ((first_ && stack_.empty()) || (!stack_.empty() && verifyValueState())) {
275 if (!writeFirstHeader()) return 0;
276 os_->put((JsonbTypeUnder)JsonbType::T_String);
277 str_pos_ = os_->tellp();
278
279 // fill the size bytes with 0 for now
280 uint32_t size = 0;
281 os_->write((char*)&size, sizeof(uint32_t));
282
283 kvState_ = WS_String;
284 return true;
285 }
286
287 return false;
288 }
289
290 // finish writing a string val
291 bool writeEndString() {
292 if (kvState_ == WS_String) {
293 std::streampos cur_pos = os_->tellp();
294 int32_t size = (int32_t)(cur_pos - str_pos_ - sizeof(uint32_t));
295 assert(size >= 0);
296
297 os_->seekp(str_pos_);
298 os_->write((char*)&size, sizeof(uint32_t));
299 os_->seekp(cur_pos);
300
301 kvState_ = WS_Value;
302 return true;
303 }
304
305 return false;
306 }

314 return 0;

374 // must call writeStartObject before writing an object val
375 bool writeStartObject() {
376 if (stack_.empty() || verifyValueState()) {
377 if (stack_.empty()) {
378 // if this is a new JSONB, write the header
379 if (!hasHdr_) {
380 writeHeader();
381 } else
382 return false;
383 }
384
385 // check if the object exceeds the maximum nesting level
386 if (stack_.size() >= MaxNestingLevel) return false;
387
388 os_->put((JsonbTypeUnder)JsonbType::T_Object);
389 // save the size position
390 stack_.push(WriteInfo({WS_Object, os_->tellp()}));
391
392 // fill the size bytes with 0 for now
393 uint32_t size = 0;
394 os_->write((char*)&size, sizeof(uint32_t));
395
396 kvState_ = WS_Value;
397 return true;
398 }
399
400 return false;
401 }
402
403 // finish writing an object val
404 bool writeEndObject() {
405 if (!stack_.empty() && stack_.top().state == WS_Object && kvState_ == WS_Value) {
406 WriteInfo& ci = stack_.top();
407 std::streampos cur_pos = os_->tellp();
408 int32_t size = (int32_t)(cur_pos - ci.sz_pos - sizeof(uint32_t));
409 assert(size >= 0);
410
411 os_->seekp(ci.sz_pos);
412 os_->write((char*)&size, sizeof(uint32_t));
413 os_->seekp(cur_pos);
414 stack_.pop();
415
416 return true;
417 }
418
419 return false;
420 }

480 bool writeEnd() {
481 while (!stack_.empty()) {
482 bool ok = false;
483 switch (stack_.top().state) {
484 case WS_Array:
485 ok = writeEndArray();
486 break;
487 case WS_Object:
488 ok = writeEndObject();
489 break;
490 case WS_String:
491 ok = writeEndString();
492 break;
493 case WS_Binary:
494 ok = writeEndBinary();
495 break;
496 default:
497 ok = false;
498 break;
499 }
500 if (ok == false) return false;
501 }
502 return true;
503 }

505private:
506 // verify we are in the right state before writing a value
507 bool verifyValueState() {
508 assert(!stack_.empty());
509 // The document can only be an Object or an Array which follows
510 // the standard.
511 return (stack_.top().state == WS_Object && kvState_ == WS_Key) ||
512 (stack_.top().state == WS_Array && kvState_ == WS_Value);
513 }
514
515 // verify we are in the right state before writing a key
516 bool verifyKeyState() {
517 assert(!stack_.empty());
518 return stack_.top().state == WS_Object && kvState_ == WS_Value;
519 }
520
521 void writeHeader() {
522 os_->put(JSONB_VER);
523 hasHdr_ = true;
524 }
