26// PageDecoder is used to decode page.
27class PageDecoder {
28public:
29 PageDecoder() {}
30
31 virtual ~PageDecoder() {}
32
33 // Call this to do some preparation for decoder.
34 // eg: parse data page header
35 virtual Status init() = 0;
36
37 // Seek the decoder to the given positional index of the page.
38 // For example, seek_to_position_in_page(0) seeks to the first
39 // stored entry.
40 //
41 // It is an error to call this with a value larger than Count().
42 // Doing so has undefined results.
43 virtual Status seek_to_position_in_page(size_t pos) = 0;
44
45 // Seek the decoder to the given value in the page, or the
46 // lowest value which is greater than the given value.
47 //
48 // If the decoder was able to locate an exact match, then
49 // sets *exact_match to true. Otherwise sets *exact_match to
50 // false, to indicate that the seeked value is _after_ the
51 // requested value.
52 //
53 // If the given value is less than the lowest value in the page,
54 // seeks to the start of the page. If it is higher than the highest
55 // value in the page, then returns Status::NotFound
56 //
57 // This will only return valid results when the data page
58 // consists of values in sorted order.
59 virtual Status seek_at_or_after_value(const void* value, bool* exact_match) {
60 return Status::NotSupported("seek_at_or_after_value"); // FIXME
61 }
62
63 // Seek the decoder forward by a given number of rows, or to the end
64 // of the page. This is primarily used to skip over data.
65 //
66 // Return the step skipped.
67 virtual size_t seek_forward(size_t n) {
68 size_t step = std::min(n, count() - current_index());
69 DCHECK_GE(step, 0);
70 seek_to_position_in_page(current_index() + step);
71 return step;
72 }
73
74 virtual Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) = 0;
75
76 virtual Status read_by_rowids(const rowid_t* rowids, ordinal_t page_first_ordinal, size_t* n,
77 vectorized::MutableColumnPtr& dst) {
78 return Status::NotSupported("not implement vec op now");
79 }
80
81 // Same as `next_batch` except for not moving forward the cursor.
82 // When read array's ordinals in `ArrayFileColumnIterator`, we want to read one extra ordinal
83 // but do not want to move forward the cursor.
84 virtual Status peek_next_batch(size_t* n, vectorized::MutableColumnPtr& dst) {
85 return Status::NotSupported("not implement vec op now");
86 }
87
88 // Return the number of elements in this page.
89 virtual size_t count() const = 0;
90
91 // Return the position within the page of the currently seeked
92 // entry (ie the entry that will next be returned by next_vector())
93 virtual size_t current_index() const = 0;
94
95 bool has_remaining() const { return current_index() < count(); }
96
97private:
98 DISALLOW_COPY_AND_ASSIGN(PageDecoder);
99};
