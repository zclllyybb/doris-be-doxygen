24// Because __int128 in memory is not aligned, but GCC7 will generate SSE instruction
25// for __int128 load/store. This will cause segment fault.
26struct PackedInt128 {
27 // PackedInt128() : value(0) {}
28 PackedInt128() = default;
29
30 PackedInt128(const __int128& value_) { value = value_; }
31 PackedInt128& operator=(const __int128& value_) {
32 value = value_;
33 return *this;
34 }
35 PackedInt128& operator=(const PackedInt128& rhs) = default;
36
37 __int128 value;
38} __attribute__((packed));

40// unalign address directly casted to int128 will core dump
41inline int128_t get_int128_from_unalign(const void* address) {
42 int128_t value = 0;
43 memcpy(&value, address, sizeof(int128_t));
44 return value;
45}
