52// The counter of add_batch rpc of a single node
53struct AddBatchCounter {
54 // total execution time of a add_batch rpc
55 int64_t add_batch_execution_time_us = 0;
56 // lock waiting time in a add_batch rpc
57 int64_t add_batch_wait_execution_time_us = 0;
58 // number of add_batch call
59 int64_t add_batch_num = 0;
60 // time passed between marked close and finish close
61 int64_t close_wait_time_ms = 0;
62
63 AddBatchCounter& operator+=(const AddBatchCounter& rhs) {
64 add_batch_execution_time_us += rhs.add_batch_execution_time_us;
65 add_batch_wait_execution_time_us += rhs.add_batch_wait_execution_time_us;
66 add_batch_num += rhs.add_batch_num;
67 close_wait_time_ms += rhs.close_wait_time_ms;
68 return *this;
69 }
70 friend AddBatchCounter operator+(const AddBatchCounter& lhs, const AddBatchCounter& rhs) {
71 AddBatchCounter sum = lhs;
72 sum += rhs;
73 return sum;
74 }
75};

77// It's very error-prone to guarantee the handler capture vars' & this closure's destruct sequence.
78// So using create() to get the closure pointer is recommended. We can delete the closure ptr before the capture vars destruction.
79// Delete this point is safe, don't worry about RPC callback will run after ReusableClosure deleted.
80template <typename T>
81class ReusableClosure final : public google::protobuf::Closure {
82public:
83 ReusableClosure() : cid(INVALID_BTHREAD_ID) {}
84 ~ReusableClosure() override {
85 // shouldn't delete when Run() is calling or going to be called, wait for current Run() done.
86 join();
87 SCOPED_SWITCH_THREAD_MEM_TRACKER_LIMITER(ExecEnv::GetInstance()->orphan_mem_tracker());
88 cntl.Reset();
89 }
90
91 static ReusableClosure<T>* create() { return new ReusableClosure<T>(); }
92
93 void addFailedHandler(const std::function<void(bool)>& fn) { failed_handler = fn; }
94 void addSuccessHandler(const std::function<void(const T&, bool)>& fn) { success_handler = fn; }
95
96 void join() {
97 // We rely on in_flight to assure one rpc is running,
98 // while cid is not reliable due to memory order.
99 // in_flight is written before getting callid,
100 // so we can not use memory fence to synchronize.
101 while (_packet_in_flight) {
102 // cid here is complicated
103 if (cid != INVALID_BTHREAD_ID) {
104 // actually cid may be the last rpc call id.
105 brpc::Join(cid);
106 }
107 if (_packet_in_flight) {
108 std::this_thread::sleep_for(std::chrono::milliseconds(10));
109 }
110 }
111 }
112
113 // plz follow this order: reset() -> set_in_flight() -> send brpc batch
114 void reset() {
115 SCOPED_SWITCH_THREAD_MEM_TRACKER_LIMITER(ExecEnv::GetInstance()->orphan_mem_tracker());
116 cntl.Reset();
117 cid = cntl.call_id();
118 }
119
120 bool try_set_in_flight() {
121 bool value = false;
122 return _packet_in_flight.compare_exchange_strong(value, true);
123 }
124
125 void clear_in_flight() { _packet_in_flight = false; }
126
127 bool is_packet_in_flight() { return _packet_in_flight; }
128
129 void end_mark() {
130 DCHECK(_is_last_rpc == false);
131 _is_last_rpc = true;
132 }
133
134 void Run() override {
135 DCHECK(_packet_in_flight);
136 if (cntl.Failed()) {
137 LOG(WARNING) << "failed to send brpc batch, error=" << berror(cntl.ErrorCode())
138 << ", error_text=" << cntl.ErrorText();
139 failed_handler(_is_last_rpc);
140 } else {
141 success_handler(result, _is_last_rpc);
142 }
143 clear_in_flight();
144 }
145
146 brpc::Controller cntl;
147 T result;
148
149private:
150 brpc::CallId cid;
151 std::atomic<bool> _packet_in_flight {false};
152 std::atomic<bool> _is_last_rpc {false};
153 std::function<void(bool)> failed_handler;
154 std::function<void(const T&, bool)> success_handler;
155};
