83template <typename T>
84class AtomicMetric : public Metric {
85public:
86 AtomicMetric() : _value(T()) {}
87 virtual ~AtomicMetric() {}
88
89 std::string to_string() const override { return std::to_string(value()); }
90
91 T value() const { return _value.load(); }
92
93 void increment(const T& delta) { _value.fetch_add(delta); }
94
95 void set_value(const T& value) { _value.store(value); }
96
97 rj::Value to_json_value(rj::Document::AllocatorType& allocator) const override {
98 return rj::Value(value());
99 }
100
101protected:
102 std::atomic<T> _value;
103};

105template <typename T>
106class LockSimpleMetric : public Metric {
107public:
108 LockSimpleMetric() : _value(T()) {}
109 virtual ~LockSimpleMetric() {}
110
111 std::string to_string() const override { return std::to_string(value()); }
112
113 T value() const {
114 std::lock_guard<SpinLock> l(_lock);
115 return _value;
116 }
117
118 void increment(const T& delta) {
119 std::lock_guard<SpinLock> l(this->_lock);
120 _value += delta;
121 }
122
123 void set_value(const T& value) {
124 std::lock_guard<SpinLock> l(this->_lock);
125 _value = value;
126 }
127
128 rj::Value to_json_value(rj::Document::AllocatorType& allocator) const override {
129 return rj::Value(value());
130 }
131
132protected:
133 // We use spinlock instead of std::atomic is because atomic don't support
134 // double's fetch_add
135 // TODO(zc): If this is atomic is bottleneck, we change to thread local.
136 // performance: on Intel(R) Xeon(R) CPU E5-2450 int64_t
137 // original type: 2ns/op
138 // single thread spinlock: 26ns/op
139 // multiple thread(8) spinlock: 2500ns/op
140 mutable SpinLock _lock;
141 T _value;
142};

144template <typename T>
145class CoreLocalCounter : public Metric {
146public:
147 CoreLocalCounter() {}
148 virtual ~CoreLocalCounter() {}
149
150 std::string to_string() const override {
151 std::stringstream ss;
152 ss << value();
153 return ss.str();
154 }
155
156 T value() const {
157 T sum = 0;
158 for (int i = 0; i < _value.size(); ++i) {
159 sum += *_value.access_at_core(i);
160 }
161 return sum;
162 }
163
164 void increment(const T& delta) { __sync_fetch_and_add(_value.access(), delta); }
165
166 void reset() { _value.reset(); }
167
168 rj::Value to_json_value(rj::Document::AllocatorType& allocator) const override {
169 return rj::Value(value());
170 }
171
172protected:
173 CoreLocalValue<T> _value;
174};

352 template <typename T>
353 Metric* register_metric(const MetricPrototype* metric_type) {
354 std::lock_guard<SpinLock> l(_lock);
355 auto inserted_metric = _metrics.insert(std::make_pair(metric_type, nullptr));
356 if (inserted_metric.second) {
357 // If not exist, make a new metric pointer

361 }
