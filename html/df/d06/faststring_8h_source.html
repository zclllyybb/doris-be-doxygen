32// A faststring is similar to a std::string, except that it is faster for many
33// common use cases (in particular, resize() will fill with uninitialized data
34// instead of memsetting to \0)
35class faststring {
36public:
37 enum { kInitialCapacity = 32 };
38
39 faststring() : data_(initial_data_), len_(0), capacity_(kInitialCapacity) {}
40
41 // Construct a string with the given capacity, in bytes.
42 explicit faststring(size_t capacity)
43 : data_(initial_data_), len_(0), capacity_(kInitialCapacity) {
44 if (capacity > capacity_) {
45 data_ = new uint8_t[capacity];
46 capacity_ = capacity;
47 }
48 ASAN_POISON_MEMORY_REGION(data_, capacity_);
49 }
50
51 ~faststring() {
52 ASAN_UNPOISON_MEMORY_REGION(initial_data_, arraysize(initial_data_));
53 if (data_ != initial_data_) {
54 delete[] data_;
55 }
56 }
57
58 // Reset the valid length of the string to 0.
59 //
60 // This does not free up any memory. The capacity of the string remains unchanged.
61 void clear() {
62 resize(0);
63 ASAN_POISON_MEMORY_REGION(data_, capacity_);
64 }
65
66 // Resize the string to the given length.
67 // If the new length is larger than the old length, the capacity is expanded as necessary.
68 //
69 // NOTE: in contrast to std::string's implementation, Any newly "exposed" bytes of data are
70 // not cleared.
71 void resize(size_t newsize) {
72 if (newsize > capacity_) {
73 reserve(newsize);
74 }
75 len_ = newsize;
76 ASAN_POISON_MEMORY_REGION(data_ + len_, capacity_ - len_);
77 ASAN_UNPOISON_MEMORY_REGION(data_, len_);
78 }
79
80 // Return the buffer built so far and reset `this` to the initial status (size() == 0).
81 // NOTE: the returned data pointer is not necessarily the pointer returned by data()
82 OwnedSlice build() {
83 uint8_t* ret = data_;
84 if (ret == initial_data_) {
85 ret = new uint8_t[len_];
86 memcpy(ret, data_, len_);
87 }
88 OwnedSlice result(ret, len_);
89 len_ = 0;
90 capacity_ = kInitialCapacity;
91 data_ = initial_data_;
92 ASAN_POISON_MEMORY_REGION(data_, capacity_);
93 return result;
94 }
95
96 // Reserve space for the given total amount of data. If the current capacity is already
97 // larger than the newly requested capacity, this is a no-op (i.e. it does not ever free memory).
98 //
99 // NOTE: even though the new capacity is reserved, it is illegal to begin writing into that memory
100 // directly using pointers. If ASAN is enabled, this is ensured using manual memory poisoning.
101 void reserve(size_t newcapacity) {
102 if (PREDICT_TRUE(newcapacity <= capacity_)) return;
103 GrowArray(newcapacity);
104 }
105
106 // Append the given data to the string, resizing capacity as necessary.
107 void append(const void* src_v, size_t count) {
108 const uint8_t* src = reinterpret_cast<const uint8_t*>(src_v);
109 EnsureRoomForAppend(count);
110 ASAN_UNPOISON_MEMORY_REGION(data_ + len_, count);
111
112 // appending short values is common enough that this
113 // actually helps, according to benchmarks. In theory
114 // memcpy_inlined should already be just as good, but this
115 // was ~20% faster for reading a large prefix-coded string file
116 // where each string was only a few chars different
117 if (count <= 4) {
118 uint8_t* p = &data_[len_];
119 for (int i = 0; i < count; i++) {
120 *p++ = *src++;
121 }
122 } else {
123 strings::memcpy_inlined(&data_[len_], src, count);
124 }
125 len_ += count;
126 }
127
128 // Append the given string to this string.
129 void append(const std::string& str) { append(str.data(), str.size()); }
130
131 // Append the given character to this string.
132 void push_back(const char byte) {
133 EnsureRoomForAppend(1);
134 ASAN_UNPOISON_MEMORY_REGION(data_ + len_, 1);
135 data_[len_] = byte;
136 len_++;
137 }
138
139 // Return the valid length of this string.
140 size_t length() const { return len_; }
141
142 // Return the valid length of this string (identical to length())
143 size_t size() const { return len_; }
144
145 // Return the allocated capacity of this string.
146 size_t capacity() const { return capacity_; }
147
148 // Return a pointer to the data in this string. Note that this pointer
149 // may be invalidated by any later non-const operation.
150 const uint8_t* data() const { return &data_[0]; }
151
152 // Return a pointer to the data in this string. Note that this pointer
153 // may be invalidated by any later non-const operation.
154 uint8_t* data() { return &data_[0]; }
155
156 // Return the given element of this string. Note that this does not perform
157 // any bounds checking.
158 const uint8_t& at(size_t i) const { return data_[i]; }
159
160 // Return the given element of this string. Note that this does not perform
161 // any bounds checking.
162 const uint8_t& operator[](size_t i) const { return data_[i]; }
163
164 // Return the given element of this string. Note that this does not perform
165 // any bounds checking.
166 uint8_t& operator[](size_t i) { return data_[i]; }
167
168 // Reset the contents of this string by copying 'len' bytes from 'src'.
169 void assign_copy(const uint8_t* src, size_t len) {
170 // Reset length so that the first resize doesn't need to copy the current
171 // contents of the array.
172 len_ = 0;
173 resize(len);
174 memcpy(data(), src, len);
175 }
176
177 // Reset the contents of this string by copying from the given std::string.
178 void assign_copy(const std::string& str) {
179 assign_copy(reinterpret_cast<const uint8_t*>(str.c_str()), str.size());
180 }
181
182 // Reallocates the internal storage to fit only the current data.
183 //
184 // This may revert to using internal storage if the current length is shorter than
185 // kInitialCapacity. In that case, after this call, capacity() will go down to
186 // kInitialCapacity.
187 //
188 // Any pointers within this instance may be invalidated.
189 void shrink_to_fit() {
190 if (data_ == initial_data_ || capacity_ == len_) return;
191 ShrinkToFitInternal();
192 }
193
194 // Return a copy of this string as a std::string.
195 std::string ToString() const {
196 return std::string(reinterpret_cast<const char*>(data()), len_);
197 }
198
199private:
200 DISALLOW_COPY_AND_ASSIGN(faststring);
201
202 // If necessary, expand the buffer to fit at least 'count' more bytes.
203 // If the array has to be grown, it is grown by at least 50%.
204 void EnsureRoomForAppend(size_t count) {
205 if (PREDICT_TRUE(len_ + count <= capacity_)) {
206 return;
207 }
208
209 // Call the non-inline slow path - this reduces the number of instructions
210 // on the hot path.
211 GrowToAtLeast(len_ + count);
212 }
213
214 // The slow path of EnsureRoomForAppend. Grows the buffer by either
215 // 'count' bytes, or 50%, whichever is more.
216 void GrowToAtLeast(size_t newcapacity);
217
218 // Grow the array to the given capacity, which must be more than
219 // the current capacity.
220 void GrowArray(size_t newcapacity);
221
222 void ShrinkToFitInternal();
223
224 uint8_t* data_;
225 uint8_t initial_data_[kInitialCapacity];
226 size_t len_;
227 // NOTE: we will make a initial buffer as part of the object, so the smallest
228 // possible value of capacity_ is kInitialCapacity.
229 size_t capacity_;
230};
