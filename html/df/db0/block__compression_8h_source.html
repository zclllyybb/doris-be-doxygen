30// This class is used to encapsulate Compression/Decompression algorithm.
31// This class only used to compress a block data, which means all data
32// should given when call compress or decompress. This class don't handle
33// stream compression.
34//
35// NOTICE!! BlockCompressionCodec is NOT thread safe, it should NOT be shared by threads
36//
37
38// max compression reuse buffer size
39// if max_compress_len is bigger than this, don't use faststring in context
40const static int MAX_COMPRESSION_BUFFER_SIZE_FOR_REUSE = 1024 * 1024 * 8;
41class BlockCompressionCodec {
42public:
43 virtual ~BlockCompressionCodec() {}
44
45 virtual Status init() { return Status::OK(); }
46
47 // This function will compress input data into output.
48 // output should be preallocated, and its capacity must be large enough
49 // for compressed input, which can be get through max_compressed_len function.
50 // Size of compressed data will be set in output's size.
51 virtual Status compress(const Slice& input, faststring* output) = 0;
52
53 // Default implementation will merge input list into a big buffer and call
54 // compress(Slice) to finish compression. If compression type support digesting
55 // slice one by one, it should reimplement this function.
56 virtual Status compress(const std::vector<Slice>& input, size_t uncompressed_size,
57 faststring* output);
58
59 // Decompress input data into output, output's capacity should be large enough
60 // for decompressed data.
61 // Size of decompressed data will be set in output's size.
62 virtual Status decompress(const Slice& input, Slice* output) = 0;
63
64 // Returns an upper bound on the max compressed length.
65 virtual size_t max_compressed_len(size_t len) = 0;
66
67 virtual bool exceed_max_compress_len(size_t uncompressed_size);
68};
