40template <typename ConditionType>
41class PredicateCreator {
42public:
43 virtual ColumnPredicate* create(const TabletColumn& column, int index,
44 const ConditionType& conditions, bool opposite,
45 MemPool* pool) = 0;
46 virtual ~PredicateCreator() = default;
47};

49template <PrimitiveType Type, PredicateType PT, typename ConditionType>
50class IntegerPredicateCreator : public PredicateCreator<ConditionType> {
51public:
52 using CppType = typename PredicatePrimitiveTypeTraits<Type>::PredicateFieldType;
53 ColumnPredicate* create(const TabletColumn& column, int index, const ConditionType& conditions,
54 bool opposite, MemPool* pool) override {
55 if constexpr (PredicateTypeTraits::is_list(PT)) {
56 return new InListPredicateBase<Type, PT>(index, conditions, convert, opposite);
57 } else {
58 static_assert(PredicateTypeTraits::is_comparison(PT));
59 return new ComparisonPredicateBase<Type, PT>(index, convert(conditions), opposite);
60 }
61 }
62
63private:
64 static CppType convert(const std::string& condition) {
65 CppType value = 0;
66 // because std::from_chars can't compile on macOS
67 if constexpr (std::is_same_v<CppType, double>) {
68 value = std::stod(condition, nullptr);
69 } else if constexpr (std::is_same_v<CppType, float>) {
70 value = std::stof(condition, nullptr);
71 } else {
72 std::from_chars(condition.data(), condition.data() + condition.size(), value);
73 }
74 return value;
75 }
76};

78template <PrimitiveType Type, PredicateType PT, typename ConditionType>
79class DecimalPredicateCreator : public PredicateCreator<ConditionType> {
80public:
81 using CppType = typename PredicatePrimitiveTypeTraits<Type>::PredicateFieldType;
82 ColumnPredicate* create(const TabletColumn& column, int index, const ConditionType& conditions,
83 bool opposite, MemPool* pool) override {
84 if constexpr (PredicateTypeTraits::is_list(PT)) {
85 return new InListPredicateBase<Type, PT>(index, conditions, convert, opposite, &column);
86 } else {
87 static_assert(PredicateTypeTraits::is_comparison(PT));
88 return new ComparisonPredicateBase<Type, PT>(index, convert(column, conditions),
89 opposite);
90 }
91 }
92
93private:
94 static CppType convert(const TabletColumn& column, const std::string& condition) {
95 StringParser::ParseResult result = StringParser::ParseResult::PARSE_SUCCESS;
96 // return CppType value cast from int128_t
97 return StringParser::string_to_decimal<int128_t>(
98 condition.data(), condition.size(), column.precision(), column.frac(), &result);
99 }
100};

102template <PrimitiveType Type, PredicateType PT, typename ConditionType>
103class StringPredicateCreator : public PredicateCreator<ConditionType> {
104public:
105 ColumnPredicate* create(const TabletColumn& column, int index, const ConditionType& conditions,
106 bool opposite, MemPool* pool) override {
107 if constexpr (PredicateTypeTraits::is_list(PT)) {
108 return new InListPredicateBase<Type, PT>(index, conditions, convert, opposite, &column,
109 pool);
110 } else {
111 static_assert(PredicateTypeTraits::is_comparison(PT));
112 return new ComparisonPredicateBase<Type, PT>(index, convert(column, conditions, pool),
113 opposite);
114 }
115 }
116
117private:
118 static StringRef convert(const TabletColumn& column, const std::string& condition,
119 MemPool* pool) {
120 size_t length = condition.length();
121 if constexpr (Type == TYPE_CHAR) {
122 length = std::max(static_cast<size_t>(column.length()), length);
123 }
124
125 char* buffer = reinterpret_cast<char*>(pool->allocate(length));
126 memset(buffer, 0, length);
127 memory_copy(buffer, condition.data(), condition.length());
128
129 return {buffer, length};
130 }
131};

133template <PrimitiveType Type, PredicateType PT, typename ConditionType>
134struct CustomPredicateCreator : public PredicateCreator<ConditionType> {
135public:
136 using CppType = typename PredicatePrimitiveTypeTraits<Type>::PredicateFieldType;
137 CustomPredicateCreator(const std::function<CppType(const std::string& condition)>& convert)
138 : _convert(convert) {}
139
140 ColumnPredicate* create(const TabletColumn& column, int index, const ConditionType& conditions,
141 bool opposite, MemPool* pool) override {
142 if constexpr (PredicateTypeTraits::is_list(PT)) {
143 return new InListPredicateBase<Type, PT>(index, conditions, _convert, opposite);
144 } else {
145 static_assert(PredicateTypeTraits::is_comparison(PT));
146 return new ComparisonPredicateBase<Type, PT>(index, _convert(conditions), opposite);
147 }
148 }
149
150private:
151 std::function<CppType(const std::string& condition)> _convert;
152};

154template <PredicateType PT, typename ConditionType>
155inline std::unique_ptr<PredicateCreator<ConditionType>> get_creator(const FieldType& type) {
156 switch (type) {
157 case OLAP_FIELD_TYPE_TINYINT: {
158 return std::make_unique<IntegerPredicateCreator<TYPE_TINYINT, PT, ConditionType>>();
159 }
160 case OLAP_FIELD_TYPE_SMALLINT: {
161 return std::make_unique<IntegerPredicateCreator<TYPE_SMALLINT, PT, ConditionType>>();
162 }
163 case OLAP_FIELD_TYPE_INT: {
164 return std::make_unique<IntegerPredicateCreator<TYPE_INT, PT, ConditionType>>();
165 }
166 case OLAP_FIELD_TYPE_BIGINT: {
167 return std::make_unique<IntegerPredicateCreator<TYPE_BIGINT, PT, ConditionType>>();
168 }
169 case OLAP_FIELD_TYPE_LARGEINT: {
170 return std::make_unique<IntegerPredicateCreator<TYPE_LARGEINT, PT, ConditionType>>();
171 }
172 case OLAP_FIELD_TYPE_FLOAT: {
173 return std::make_unique<IntegerPredicateCreator<TYPE_FLOAT, PT, ConditionType>>();
174 }
175 case OLAP_FIELD_TYPE_DOUBLE: {
176 return std::make_unique<IntegerPredicateCreator<TYPE_DOUBLE, PT, ConditionType>>();
177 }
178 case OLAP_FIELD_TYPE_DECIMAL: {
179 return std::make_unique<CustomPredicateCreator<TYPE_DECIMALV2, PT, ConditionType>>(
180 [](const std::string& condition) {
181 decimal12_t value = {0, 0};
182 value.from_string(condition);
183 return value;
184 });
185 }
186 case OLAP_FIELD_TYPE_DECIMAL32: {
187 return std::make_unique<DecimalPredicateCreator<TYPE_DECIMAL32, PT, ConditionType>>();
188 }
189 case OLAP_FIELD_TYPE_DECIMAL64: {
190 return std::make_unique<DecimalPredicateCreator<TYPE_DECIMAL64, PT, ConditionType>>();
191 }
192 case OLAP_FIELD_TYPE_DECIMAL128I: {
193 return std::make_unique<DecimalPredicateCreator<TYPE_DECIMAL128I, PT, ConditionType>>();
194 }
195 case OLAP_FIELD_TYPE_CHAR: {
196 return std::make_unique<StringPredicateCreator<TYPE_CHAR, PT, ConditionType>>();
197 }
198 case OLAP_FIELD_TYPE_VARCHAR:
199 case OLAP_FIELD_TYPE_STRING: {
200 return std::make_unique<StringPredicateCreator<TYPE_STRING, PT, ConditionType>>();
201 }
202 case OLAP_FIELD_TYPE_DATE: {
203 return std::make_unique<CustomPredicateCreator<TYPE_DATE, PT, ConditionType>>(
204 timestamp_from_date);
205 }
206 case OLAP_FIELD_TYPE_DATEV2: {
207 return std::make_unique<CustomPredicateCreator<TYPE_DATEV2, PT, ConditionType>>(
208 timestamp_from_date_v2);
209 }
210 case OLAP_FIELD_TYPE_DATETIME: {
211 return std::make_unique<CustomPredicateCreator<TYPE_DATETIME, PT, ConditionType>>(
212 timestamp_from_datetime);
213 }
214 case OLAP_FIELD_TYPE_DATETIMEV2: {
215 return std::make_unique<CustomPredicateCreator<TYPE_DATETIMEV2, PT, ConditionType>>(
216 timestamp_from_datetime_v2);
217 }
218 case OLAP_FIELD_TYPE_BOOL: {
219 return std::make_unique<CustomPredicateCreator<TYPE_BOOLEAN, PT, ConditionType>>(
220 [](const std::string& condition) {
221 int32_t ivalue = 0;
222 auto result = std::from_chars(condition.data(),
223 condition.data() + condition.size(), ivalue);
224 if (result.ec == std::errc()) {
225 return bool(ivalue);
226 }
227
228 StringParser::ParseResult parse_result;
229 bool value = StringParser::string_to_bool(condition.data(), condition.size(),
230 &parse_result);
231 return value;
232 });
233 }
234 default:
235 return nullptr;
236 }
237}

264// This method is called in reader and in deletehandler.
265// When it is called by delete handler, then it should use the delete predicate's tablet schema
266// to parse the conditions.
267inline ColumnPredicate* parse_to_predicate(TabletSchemaSPtr tablet_schema,
268 const TCondition& condition, MemPool* mem_pool,
269 bool opposite = false) {
270 int32_t col_unique_id = condition.column_unique_id;
271 // TODO: not equal and not in predicate is not pushed down
272 const TabletColumn& column = tablet_schema->column_by_uid(col_unique_id);
273 uint32_t index = tablet_schema->field_index(col_unique_id);
274
275 if (to_lower(condition.condition_op) == "is") {
276 return new NullPredicate(index, to_lower(condition.condition_values[0]) == "null",
277 opposite);
278 } else if (is_match_condition(condition.condition_op)) {
279 return new MatchPredicate(index, condition.condition_values[0],
280 to_match_type(condition.condition_op));
281 }
282
283 if ((condition.condition_op == "*=" || condition.condition_op == "!*=") &&
284 condition.condition_values.size() > 1) {
285 decltype(create_list_predicate<PredicateType::UNKNOWN>)* create = nullptr;
286
287 if (condition.condition_op == "*=") {
288 create = create_list_predicate<PredicateType::IN_LIST>;
289 } else {
290 create = create_list_predicate<PredicateType::NOT_IN_LIST>;
291 }
292 return create(column, index, condition.condition_values, opposite, mem_pool);
293 }
294
295 decltype(create_comparison_predicate<PredicateType::UNKNOWN>)* create = nullptr;
296 if (condition.condition_op == "*=" || condition.condition_op == "=") {
297 create = create_comparison_predicate<PredicateType::EQ>;
298 } else if (condition.condition_op == "!*=" || condition.condition_op == "!=") {
299 create = create_comparison_predicate<PredicateType::NE>;
300 } else if (condition.condition_op == "<<") {
301 create = create_comparison_predicate<PredicateType::LT>;
302 } else if (condition.condition_op == "<=") {
303 create = create_comparison_predicate<PredicateType::LE>;
304 } else if (condition.condition_op == ">>") {
305 create = create_comparison_predicate<PredicateType::GT>;
306 } else if (condition.condition_op == ">=") {
307 create = create_comparison_predicate<PredicateType::GE>;
308 }
309 return create(column, index, condition.condition_values[0], opposite, mem_pool);
310}
