38inline const int HLL_COLUMN_PRECISION = 14;
39inline const int HLL_ZERO_COUNT_BITS = (64 - HLL_COLUMN_PRECISION);
40inline const int HLL_EXPLICIT_INT64_NUM = 160;

44inline const int HLL_COLUMN_DEFAULT_LEN = HLL_REGISTERS_COUNT + 1;

50// Hyperloglog distinct estimate algorithm.
51// See these papers for more details.
52// 1) Hyperloglog: The analysis of a near-optimal cardinality estimation
53// algorithm (2007)
54// 2) HyperLogLog in Practice (paper from google with some improvements)
55
56// Each HLL value is a set of values. To save space, Doris store HLL value
57// in different format according to its cardinality.
58//
59// HLL_DATA_EMPTY: when set is empty.
60//
61// HLL_DATA_EXPLICIT: when there is only few values in set, store these values explicit.
62// If the number of hash values is not greater than 160, set is encoded in this format.
63// The max space occupied is (1 + 1 + 160 * 8) = 1282. I don't know why 160 is chosen,
64// maybe can be other number. If you are interested, you can try other number and see
65// if it will be better.
66//
67// HLL_DATA_SPARSE: only store non-zero registers. If the number of non-zero registers
68// is not greater than 4096, set is encoded in this format. The max space occupied is
69// (1 + 4 + 3 * 4096) = 12293.
70//
71// HLL_DATA_FULL: most space-consuming, store all registers
72//
73// A HLL value will change in the sequence empty -> explicit -> sparse -> full, and not
74// allow reverse.
75//
76// NOTE: This values are persisted in storage devices, so don't change exist
77// enum values.
78enum HllDataType {
79 HLL_DATA_EMPTY = 0,
80 HLL_DATA_EXPLICIT = 1,
81 HLL_DATA_SPARSE = 2,
82 HLL_DATA_FULL = 3,
83};

85class HyperLogLog {
86public:
87 HyperLogLog() = default;
88 explicit HyperLogLog(uint64_t hash_value) : _type(HLL_DATA_EXPLICIT) {
89 _hash_set.emplace(hash_value);
90 }

93 HyperLogLog(const HyperLogLog& other) {
94 this->_type = other._type;
95 switch (other._type) {
96 case HLL_DATA_EMPTY:
97 break;
98 case HLL_DATA_EXPLICIT: {
99 this->_hash_set = other._hash_set;
100 break;
101 }
102 case HLL_DATA_SPARSE:
103 case HLL_DATA_FULL: {
104 _registers = new uint8_t[HLL_REGISTERS_COUNT];
105 memcpy(_registers, other._registers, HLL_REGISTERS_COUNT);
106 break;
107 }
108 default:
109 break;
110 }
111 }
112
113 HyperLogLog(HyperLogLog&& other) {
114 this->_type = other._type;
115 switch (other._type) {
116 case HLL_DATA_EMPTY:
117 break;
118 case HLL_DATA_EXPLICIT: {
119 this->_hash_set = std::move(other._hash_set);
120 other._type = HLL_DATA_EMPTY;
121 break;
122 }
123 case HLL_DATA_SPARSE:
124 case HLL_DATA_FULL: {
125 this->_registers = other._registers;
126 other._registers = nullptr;
127 other._type = HLL_DATA_EMPTY;
128 break;
129 }
130 default:
131 break;
132 }
133 }
134
135 HyperLogLog& operator=(HyperLogLog&& other) {
136 if (this != &other) {
137 if (_registers != nullptr) {
138 delete[] _registers;
139 _registers = nullptr;
140 }
141
142 this->_type = other._type;
143 switch (other._type) {
144 case HLL_DATA_EMPTY:
145 break;
146 case HLL_DATA_EXPLICIT: {
147 this->_hash_set = std::move(other._hash_set);
148 other._type = HLL_DATA_EMPTY;
149 break;
150 }
151 case HLL_DATA_SPARSE:
152 case HLL_DATA_FULL: {
153 this->_registers = other._registers;
154 other._registers = nullptr;
155 other._type = HLL_DATA_EMPTY;
156 break;
157 }
158 default:
159 break;
160 }
161 }
162 return *this;
163 }

166 if (this != &other) {

168 delete[] _registers;

185 }

187 break;
188 }
189 }
190 return *this;
191 }

280 // update one hash value into this registers
281 void _update_registers(uint64_t hash_value) {
282 // Use the lower bits to index into the number of streams and then
283 // find the first 1 bit after the index bits.
284 int idx = hash_value % HLL_REGISTERS_COUNT;
285 hash_value >>= HLL_COLUMN_PRECISION;
286 // make sure max first_one_bit is HLL_ZERO_COUNT_BITS + 1
287 hash_value |= ((uint64_t)1 << HLL_ZERO_COUNT_BITS);
288 uint8_t first_one_bit = __builtin_ctzl(hash_value) + 1;
289 _registers[idx] = (_registers[idx] < first_one_bit ? first_one_bit : _registers[idx]);
290 }
291
292 // absorb other registers into this registers
293 void _merge_registers(const uint8_t* other_registers) {
294#ifdef __AVX2__
295 int loop = HLL_REGISTERS_COUNT / 32; // 32 = 256/8
296 uint8_t* dst = _registers;
297 const uint8_t* src = other_registers;
298 for (int i = 0; i < loop; i++) {
299 __m256i xa = _mm256_loadu_si256((const __m256i*)dst);
300 __m256i xb = _mm256_loadu_si256((const __m256i*)src);
301 _mm256_storeu_si256((__m256i*)dst, _mm256_max_epu8(xa, xb));
302 src += 32;
303 dst += 32;
304 }
305#else
306 for (int i = 0; i < HLL_REGISTERS_COUNT; ++i) {
307 _registers[i] =
308 (_registers[i] < other_registers[i] ? other_registers[i] : _registers[i]);
309 }
310#endif
311 }
