52 // If all dependencies are finished, this pipeline task should be scheduled.
53 // e.g. Hash join probe task will be scheduled once Hash join build task is finished.
54 bool finish_one_dependency(int dependency_core_id) {
55 DCHECK(_complete_dependency < _dependencies.size());
56 bool finish = _complete_dependency.fetch_add(1) == _dependencies.size() - 1;
57 if (finish) {
58 _previous_schedule_id = dependency_core_id;
59 }
60 return finish;
61 }
