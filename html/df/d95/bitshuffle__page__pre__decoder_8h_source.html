27template <bool USED_IN_DICT_ENCODING>
28struct BitShufflePagePreDecoder : public DataPagePreDecoder {
40 virtual Status decode(std::unique_ptr<char[]>* page, Slice* page_slice,
41 size_t size_of_tail) override {
42 size_t num_elements, compressed_size, num_element_after_padding;
43 int size_of_element;
44
45 size_t size_of_dict_header = 0;
46 Slice data(page_slice->data, page_slice->size - size_of_tail);
47 if constexpr (USED_IN_DICT_ENCODING) {
48 auto type = decode_fixed32_le((const uint8_t*)&data.data[0]);
49 if (static_cast<EncodingTypePB>(type) != EncodingTypePB::DICT_ENCODING) {
50 return Status::OK();
51 }
52 size_of_dict_header = BINARY_DICT_PAGE_HEADER_SIZE;
53 data.remove_prefix(4);
54 }
55
56 RETURN_IF_ERROR(parse_bit_shuffle_header(data, num_elements, compressed_size,
57 num_element_after_padding, size_of_element));
58
59 if (compressed_size != data.size) {
60 return Status::InternalError(
61 "Size information unmatched, compressed_size:{}, num_elements:{}, data size:{}",
62 compressed_size, num_elements, data.size);
63 }
64
65 Slice decoded_slice;
66 decoded_slice.size = size_of_dict_header + BITSHUFFLE_PAGE_HEADER_SIZE +
67 num_element_after_padding * size_of_element + size_of_tail;
68 std::unique_ptr<char[]> decoded_page(new char[decoded_slice.size]);
69 decoded_slice.data = decoded_page.get();
70
71 if constexpr (USED_IN_DICT_ENCODING) {
72 memcpy(decoded_slice.data, page_slice->data, size_of_dict_header);
73 }
74
75 memcpy(decoded_slice.data + size_of_dict_header, data.data, BITSHUFFLE_PAGE_HEADER_SIZE);
76
77 auto bytes = bitshuffle::decompress_lz4(
78 &data.data[BITSHUFFLE_PAGE_HEADER_SIZE],
79 decoded_slice.data + BITSHUFFLE_PAGE_HEADER_SIZE + size_of_dict_header,
80 num_element_after_padding, size_of_element, 0);
81 if (PREDICT_FALSE(bytes < 0)) {
82 // Ideally, this should not happen.
83 warn_with_bitshuffle_error(bytes);
84 return Status::RuntimeError("Unshuffle Process failed");
85 }
86
87 memcpy(decoded_slice.data + decoded_slice.size - size_of_tail,
88 page_slice->data + page_slice->size - size_of_tail, size_of_tail);
89
90 *page_slice = decoded_slice;
91 *page = std::move(decoded_page);
92 return Status::OK();
93 }
94};
