32inline uint8_t bits_less_than_64(const uint64_t v) {
33 return v == 0 ? 0 : 64 - __builtin_clzll(v);
34}
35
36// See https://stackoverflow.com/questions/28423405/counting-the-number-of-leading-zeros-in-a-128-bit-integer
37inline uint8_t bits_may_more_than_64(const uint128_t v) {
38 uint64_t hi = v >> 64;
39 uint64_t lo = v;
40 int z[3] = {__builtin_clzll(hi), __builtin_clzll(lo) + 64, 128};
41 int idx = !hi + ((!lo) & (!hi));
42 return 128 - z[idx];
43}
44
45template <typename T>
46uint8_t bits(const T v) {
47 if (sizeof(T) <= 8) {
48 return bits_less_than_64(v);
49 } else {
50 return bits_may_more_than_64(v);
51 }
52}

54// The implementation for frame-of-reference coding
55// The detail of frame-of-reference coding, please refer to
56// https://lemire.me/blog/2012/02/08/effective-compression-using-frame-of-reference-and-delta-coding/
57// and https://www.elastic.co/cn/blog/frame-of-reference-and-roaring-bitmaps
58//
59// The encoded data format is as follows:
60//
61// 1. Body:
62// BitPackingFrame * FrameCount
63// 2. Footer:
64//
65// (8 bit StorageFormat + 8 bit BitWidth) * FrameCount
66// 8 bit FrameValueNum
67// 32 bit ValuesNum
68//
69// There are currently three storage formats
70// (1) if the StorageFormat == 0: When input data order is not ascending and the BitPackingFrame format is:
71// MinValue, (Value[i] - MinVale) * FrameValueNum
72//
73// (2) if the StorageFormat == 1: When input data order is ascending and the BitPackingFrame format is:
74// MinValue, (Value[i] - Value[i - 1]) * FrameValueNum
75//
76// (3) if the StorageFormat == 2: When overflow occurs when using (1) or (2) and save original values:
77// MinValue, (Value[i]) * FrameValueNum
78//
79// len(MinValue) can be 32(uint32_t), 64(uint64_t), 128(uint128_t)
80//
81// The OrderFlag is 1 represents ascending order, 0 represents not ascending order
82// The last frame value num maybe less than 128
83template <typename T>
84class ForEncoder {
85public:
86 explicit ForEncoder(faststring* buffer) : _buffer(buffer) {}
87
88 void put(const T value) { return put_batch(&value, 1); }
89
90 void put_batch(const T* value, size_t count);
91
92 // Flushes any pending values to the underlying buffer.
93 // Returns the total number of bytes written
94 uint32_t flush();
95
96 // underlying buffer size + footer meta size.
97 // Note: should call this method before flush.
98 uint32_t len() { return _buffer->size() + _storage_formats.size() + _bit_widths.size() + 5; }
99
100 // Resets all the state in the encoder.
101 void clear() {
102 _values_num = 0;
103 _buffered_values_num = 0;
104 _buffer->clear();
105 }
106
107private:
108 void bit_pack(const T* input, uint8_t in_num, int bit_width, uint8_t* output);
109
110 void bit_packing_one_frame_value(const T* input);
111
112 const T* copy_value(const T* val, size_t count);
113
114 const T numeric_limits_max();
115
116 uint32_t _values_num = 0;
117 uint8_t _buffered_values_num = 0;
118 static const uint8_t FRAME_VALUE_NUM = 128;
119 T _buffered_values[FRAME_VALUE_NUM];
120
121 faststring* _buffer;
122 std::vector<uint8_t> _storage_formats;
123 std::vector<uint8_t> _bit_widths;
124};
