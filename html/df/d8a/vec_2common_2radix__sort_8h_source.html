59template <typename KeyBits>
60struct RadixSortFloatTransform {
62 static constexpr bool transform_is_simple = false;
63
64 static KeyBits forward(KeyBits x) {
65 return x ^
66 ((-(x >> (sizeof(KeyBits) * 8 - 1))) | (KeyBits(1) << (sizeof(KeyBits) * 8 - 1)));
67 }
68
69 static KeyBits backward(KeyBits x) {
70 return x ^
71 (((x >> (sizeof(KeyBits) * 8 - 1)) - 1) | (KeyBits(1) << (sizeof(KeyBits) * 8 - 1)));
72 }
73};

161template <typename Traits>
162struct RadixSort {
163private:
164 using Element = typename Traits::Element;
165 using Key = typename Traits::Key;
166 using CountType = typename Traits::CountType;
167 using KeyBits = typename Traits::KeyBits;
168
169 // Use insertion sort if the size of the array is less than equal to this threshold
170 static constexpr size_t INSERTION_SORT_THRESHOLD = 64;
171
172 static constexpr size_t HISTOGRAM_SIZE = 1 << Traits::PART_SIZE_BITS;
173 static constexpr size_t PART_BITMASK = HISTOGRAM_SIZE - 1;
174 static constexpr size_t KEY_BITS = sizeof(Key) * 8;
175 static constexpr size_t NUM_PASSES =
176 (KEY_BITS + (Traits::PART_SIZE_BITS - 1)) / Traits::PART_SIZE_BITS;
177
178 static ALWAYS_INLINE KeyBits get_part(size_t N, KeyBits x) {
179 if (Traits::Transform::transform_is_simple) x = Traits::Transform::forward(x);
180
181 return (x >> (N * Traits::PART_SIZE_BITS)) & PART_BITMASK;
182 }
183
184 static KeyBits key_to_bits(Key x) { return ext::bit_cast<KeyBits>(x); }
185 static Key bits_to_key(KeyBits x) { return ext::bit_cast<Key>(x); }
186
187 static void insertion_sort_internal(Element* arr, size_t size) {
188 Element* end = arr + size;
189 for (Element* i = arr + 1; i < end; ++i) {
190 if (Traits::less(Traits::extract_key(*i), Traits::extract_key(*(i - 1)))) {
191 Element* j;
192 Element tmp = *i;
193 *i = *(i - 1);
194 for (j = i - 1; j > arr && Traits::less(Traits::extract_key(tmp),
195 Traits::extract_key(*(j - 1)));
196 --j)
197 *j = *(j - 1);
198 *j = tmp;
199 }
200 }
201 }

203 /* Main MSD radix sort subroutine
204 * Puts elements to buckets based on PASS-th digit, then recursively calls insertion sort or itself on the buckets
205 */
206 template <size_t PASS>
207 static inline void radix_sort_msd_internal(Element* arr, size_t size, size_t limit) {
208 Element* last_list[HISTOGRAM_SIZE + 1];
209 Element** last = last_list + 1;
210 size_t count[HISTOGRAM_SIZE] = {0};
211
212 for (Element* i = arr; i < arr + size; ++i) ++count[get_part(PASS, *i)];
213
214 last_list[0] = last_list[1] = arr;
215
216 size_t buckets_for_recursion = HISTOGRAM_SIZE;
217 Element* finish = arr + size;
218 for (size_t i = 1; i < HISTOGRAM_SIZE; ++i) {
219 last[i] = last[i - 1] + count[i - 1];
220 if (last[i] >= arr + limit) {
221 buckets_for_recursion = i;
222 finish = last[i];
223 }
224 }
225
226 /* At this point, we have the following variables:
227 * count[i] is the size of i-th bucket
228 * last[i] is a pointer to the beginning of i-th bucket, last[-1] == last[0]
229 * buckets_for_recursion is the number of buckets that should be sorted, the last of them only partially
230 * finish is a pointer to the end of the first buckets_for_recursion buckets
231 */
232
233 // Scatter array elements to buckets until the first buckets_for_recursion buckets are full
234 for (size_t i = 0; i < buckets_for_recursion; ++i) {
235 Element* end = last[i - 1] + count[i];
236 if (end == finish) {
237 last[i] = end;
238 break;
239 }
240 while (last[i] != end) {
241 Element swapper = *last[i];
242 KeyBits tag = get_part(PASS, swapper);
243 if (tag != i) {
244 do {
245 std::swap(swapper, *last[tag]++);
246 } while ((tag = get_part(PASS, swapper)) != i);
247 *last[i] = swapper;
248 }
249 ++last[i];
250 }
251 }
252
253 if constexpr (PASS > 0) {
254 // Recursively sort buckets, except the last one
255 for (size_t i = 0; i < buckets_for_recursion - 1; ++i) {
256 Element* start = last[i - 1];
257 size_t subsize = last[i] - last[i - 1];
258 radix_sort_msd_internal_helper<PASS - 1>(start, subsize, subsize);
259 }
260
261 // Sort last necessary bucket with limit
262 Element* start = last[buckets_for_recursion - 2];
263 size_t subsize = last[buckets_for_recursion - 1] - last[buckets_for_recursion - 2];
264 size_t sublimit = limit - (last[buckets_for_recursion - 1] - arr);
265 radix_sort_msd_internal_helper<PASS - 1>(start, subsize, sublimit);
266 }
267 }

269 // A helper to choose sorting algorithm based on array length
270 template <size_t PASS>
271 static inline void radix_sort_msd_internal_helper(Element* arr, size_t size, size_t limit) {
272 if (size <= INSERTION_SORT_THRESHOLD)
273 insertion_sort_internal(arr, size);
274 else
275 radix_sort_msd_internal<PASS>(arr, size, limit);
276 }

280 static void execute_lsd(Element* arr, size_t size) {
282
284
286 CountType histograms[HISTOGRAM_SIZE * NUM_PASSES] = {0};
287
288 typename Traits::Allocator allocator;
289
291 Element* swap_buffer =
292 reinterpret_cast<Element*>(allocator.allocate(size * sizeof(Element)));
293
296 for (size_t i = 0; i < size; ++i) {
297 if (!Traits::Transform::transform_is_simple)
298 Traits::extract_key(arr[i]) = bits_to_key(
299 Traits::Transform::forward(key_to_bits(Traits::extract_key(arr[i]))));
300
301 for (size_t pass = 0; pass < NUM_PASSES; ++pass)
302 ++histograms[pass * HISTOGRAM_SIZE +
303 get_part(pass, key_to_bits(Traits::extract_key(arr[i])))];
304 }
305
306 {
308 size_t sums[NUM_PASSES] = {0};
309
310 for (size_t i = 0; i < HISTOGRAM_SIZE; ++i) {
311 for (size_t pass = 0; pass < NUM_PASSES; ++pass) {
312 size_t tmp = histograms[pass * HISTOGRAM_SIZE + i] + sums[pass];
313 histograms[pass * HISTOGRAM_SIZE + i] = sums[pass] - 1;
314 sums[pass] = tmp;
315 }
316 }
317 }
318
320 for (size_t pass = 0; pass < NUM_PASSES; ++pass) {
321 Element* writer = pass % 2 ? arr : swap_buffer;
322 Element* reader = pass % 2 ? swap_buffer : arr;
323
324 for (size_t i = 0; i < size; ++i) {
325 size_t pos = get_part(pass, key_to_bits(Traits::extract_key(reader[i])));
326
328 auto& dest = writer[++histograms[pass * HISTOGRAM_SIZE + pos]];
329 dest = reader[i];
330
332 if (!Traits::Transform::transform_is_simple && pass == NUM_PASSES - 1)
333 Traits::extract_key(dest) = bits_to_key(Traits::Transform::backward(
334 key_to_bits(Traits::extract_key(reader[i]))));
335 }
336 }
337
340 if (NUM_PASSES % 2) memcpy(arr, swap_buffer, size * sizeof(Element));
341
342 allocator.deallocate(swap_buffer, size * sizeof(Element));
343 }
