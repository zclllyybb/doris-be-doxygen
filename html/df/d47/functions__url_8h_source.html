77 ColumnString::Chars& res_data, ColumnString::Offsets& res_offsets) {
78 size_t size = offsets.size();
79 res_offsets.resize(size);
80 res_data.reserve(size * Extractor::get_reserve_length_for_element());
81
82 size_t prev_offset = 0;
83 size_t res_offset = 0;
84
86 Pos start;
87 size_t length;
88
89 for (size_t i = 0; i < size; ++i) {
90 Extractor::execute(reinterpret_cast<const char*>(&data[prev_offset]),

92
93 res_data.resize(res_data.size() + length);
94 memcpy_small_allow_read_write_overflow15(&res_data[res_offset], start, length);
95 res_offset += length;
96
97 res_offsets[i] = res_offset;
98 prev_offset = offsets[i];
99 }
100 }

117template <typename Extractor>
118struct CutSubstringImpl {
119 static void vector(const ColumnString::Chars& data, const ColumnString::Offsets& offsets,
120 ColumnString::Chars& res_data, ColumnString::Offsets& res_offsets) {
121 res_data.reserve(data.size());
122 size_t size = offsets.size();
123 res_offsets.resize(size);
124
125 size_t prev_offset = 0;
126 size_t res_offset = 0;
127
129 Pos start;
130 size_t length;
131
132 for (size_t i = 0; i < size; ++i) {
133 const char* current = reinterpret_cast<const char*>(&data[prev_offset]);
134 Extractor::execute(current, offsets[i] - prev_offset, start, length);
135 size_t start_index = start - reinterpret_cast<const char*>(data.data());
136
137 res_data.resize(res_data.size() + offsets[i] - prev_offset - length);
138 memcpy_small_allow_read_write_overflow15(&res_data[res_offset], current,
139 start - current);
140 memcpy_small_allow_read_write_overflow15(&res_data[res_offset + start - current],
141 start + length,
142 offsets[i] - start_index - length);
143 res_offset += offsets[i] - prev_offset - length;
144
145 res_offsets[i] = res_offset;
146 prev_offset = offsets[i];
147 }
148 }

154 res_data.reserve(data.size() - length);
155 res_data.append(data.data(), start);
156 res_data.append(start + length, data.data() + data.size());
