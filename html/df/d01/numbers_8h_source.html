96// ----------------------------------------------------------------------
97// FastIntToBuffer()
98// FastHexToBuffer()
99// FastHex64ToBuffer()
100// FastHex32ToBuffer()
101// FastTimeToBuffer()
102// These are intended for speed. FastIntToBuffer() assumes the
103// integer is non-negative. FastHexToBuffer() puts output in
104// hex rather than decimal. FastTimeToBuffer() puts the output
105// into RFC822 format.
106//
107// FastHex64ToBuffer() puts a 64-bit unsigned value in hex-format,
108// padded to exactly 16 bytes (plus one byte for '\0')
109//
110// FastHex32ToBuffer() puts a 32-bit unsigned value in hex-format,
111// padded to exactly 8 bytes (plus one byte for '\0')
112//
113// All functions take the output buffer as an arg. FastInt() uses
114// at most 22 bytes, FastTime() uses exactly 30 bytes. They all
115// return a pointer to the beginning of the output, which for
116// FastHex() may not be the beginning of the input buffer. (For
117// all others, we guarantee that it is.)
118//
119// NOTE: In 64-bit land, sizeof(time_t) is 8, so it is possible
120// to pass to FastTimeToBuffer() a time whose year cannot be
121// represented in 4 digits. In this case, the output buffer
122// will contain the string "Invalid:<value>"
123// ----------------------------------------------------------------------
124
125// Previously documented minimums -- the buffers provided must be at least this
126// long, though these numbers are subject to change:
127// Int32, UInt32: 12 bytes
128// Int64, UInt64, Hex: 22 bytes
129// Time: 30 bytes
130// Hex32: 9 bytes
131// Hex64: 17 bytes
132// Use kFastToBufferSize rather than hardcoding constants.
133static const int kFastToBufferSize = 32;

385// SimpleAtoi converts a string to an integer.
386// Uses safe_strto?() for actual parsing, so strict checking is
387// applied, which is to say, the string must be a base-10 integer, optionally
388// followed or preceded by whitespace, and value has to be in the range of
389// the corresponding integer type.
390//
391// Returns true if parsing was successful.
392template <typename int_type>
393bool MUST_USE_RESULT SimpleAtoi(const char* s, int_type* out) {
394 // Must be of integer type (not pointer type), with more than 16-bitwidth.
395 COMPILE_ASSERT(sizeof(*out) == 4 || sizeof(*out) == 8, SimpleAtoiWorksWith32Or64BitInts);
396 if (std::numeric_limits<int_type>::is_signed) { // Signed
397 if (sizeof(*out) == 64 / 8) { // 64-bit
398 return safe_strto64(s, reinterpret_cast<int64*>(out));
399 } else { // 32-bit
400 return safe_strto32(s, reinterpret_cast<int32*>(out));
401 }
402 } else { // Unsigned
403 if (sizeof(*out) == 64 / 8) { // 64-bit
404 return safe_strtou64(s, reinterpret_cast<uint64*>(out));
405 } else { // 32-bit
406 return safe_strtou32(s, reinterpret_cast<uint32*>(out));
407 }
408 }
409}

483// ----------------------------------------------------------------------
484// ParseDoubleRange()
485// Parse an expression in 'text' of the form: <double><sep><double>
486// where <double> may be a double-precision number and <sep> is a
487// single char or "..", and must be one of the chars in parameter
488// 'separators', which may contain '-' or '.' (which means "..") or
489// any chars not allowed in a double. If allow_unbounded_markers,
490// <double> may also be a '?' to indicate unboundedness (if on the
491// left of <sep>, means unbounded below; if on the right, means
492// unbounded above). Depending on num_required_bounds, which may be
493// 0, 1, or 2, <double> may also be the empty string, indicating
494// unboundedness. If require_separator is false, then a single
495// <double> is acceptable and is parsed as a range bounded from
496// below. We also check that the character following the range must
497// be in acceptable_terminators. If null_terminator_ok, then it is
498// also OK if the range ends in \0 or after len chars. If
499// allow_currency is true, the first <double> may be optionally
500// preceded by a '$', in which case *is_currency will be true, and
501// the second <double> may similarly be preceded by a '$'. In these
502// cases, the '$' will be ignored (otherwise it's an error). If
503// allow_comparators is true, the expression in 'text' may also be
504// of the form <comparator><double>, where <comparator> is '<' or
505// '>' or '<=' or '>='. separators and require_separator are
506// ignored in this format, but all other parameters function as for
507// the first format. Return true if the expression parsed
508// successfully; false otherwise. If successful, output params are:
509// 'end', which points to the char just beyond the expression;
510// 'from' and 'to' are set to the values of the <double>s, and are
511// -inf and inf (or unchanged, depending on dont_modify_unbounded)
512// if unbounded. Output params are undefined if false is
513// returned. len is the input length, or -1 if text is
514// '\0'-terminated, which is more efficient.
515// ----------------------------------------------------------------------
516struct DoubleRangeOptions {
517 const char* separators;
518 bool require_separator;
519 const char* acceptable_terminators;
520 bool null_terminator_ok;
521 bool allow_unbounded_markers;
522 uint32 num_required_bounds;
523 bool dont_modify_unbounded;
524 bool allow_currency;
525 bool allow_comparators;
526};
