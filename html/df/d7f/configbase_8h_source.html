32class Register {
33public:
34 struct Field {
35 const char* type = nullptr;
36 const char* name = nullptr;
37 void* storage = nullptr;
38 const char* defval = nullptr;
39 bool valmutable = false;
40 Field(const char* ftype, const char* fname, void* fstorage, const char* fdefval,
41 bool fvalmutable)
42 : type(ftype),
43 name(fname),
44 storage(fstorage),
45 defval(fdefval),
46 valmutable(fvalmutable) {}
47 };
48
49public:
50 static std::map<std::string, Field>* _s_field_map;
51
52public:
53 Register(const char* ftype, const char* fname, void* fstorage, const char* fdefval,
54 bool fvalmutable) {
55 if (_s_field_map == nullptr) {
56 _s_field_map = new std::map<std::string, Field>();
57 }
58 Field field(ftype, fname, fstorage, fdefval, fvalmutable);
59 _s_field_map->insert(std::make_pair(std::string(fname), field));
60 }
61};

63// RegisterConfValidator class is used to store validator function of registered config fields in
64// Register::_s_field_map.
65// If any validator return false when BE bootstart, the bootstart will be terminated.
66// If validator return false when use http API to update some config, the config will not
67// be modified and the API will return failure.
68class RegisterConfValidator {
69public:
70 // Validator for each config name.
71 static std::map<std::string, std::function<bool()>>* _s_field_validator;
72
73public:
74 RegisterConfValidator(const char* fname, const std::function<bool()>& validator) {
75 if (_s_field_validator == nullptr) {
76 _s_field_validator = new std::map<std::string, std::function<bool()>>();
77 }
78 // register validator to _s_field_validator
79 _s_field_validator->insert(std::make_pair(std::string(fname), validator));
80 }
81};

83#define DEFINE_FIELD(FIELD_TYPE, FIELD_NAME, FIELD_DEFAULT, VALMUTABLE) \
84 FIELD_TYPE FIELD_NAME; \
85 static Register reg_##FIELD_NAME(#FIELD_TYPE, #FIELD_NAME, &(FIELD_NAME), FIELD_DEFAULT, \
86 VALMUTABLE);
87
88#define DECLARE_FIELD(FIELD_TYPE, FIELD_NAME) extern FIELD_TYPE FIELD_NAME;

90#define DEFINE_VALIDATOR(FIELD_NAME, VALIDATOR) \
91 static auto validator_##FIELD_NAME = VALIDATOR; \
92 static RegisterConfValidator reg_validator_##FIELD_NAME( \
93 #FIELD_NAME, []() -> bool { return validator_##FIELD_NAME(FIELD_NAME); });

95#define DECLARE_VALIDATOR(FIELD_NAME) ;
