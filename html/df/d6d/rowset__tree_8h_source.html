47// Class which encapsulates the set of rowsets which are active for a given
48// Tablet. This provides efficient lookup by key for Rowsets which may overlap
49// that key range.
50//
51// Additionally, the rowset tree maintains information about the implicit
52// intervals generated by the row sets (for instance, if a tablet has
53// rowsets [0, 2] and [1, 3] it has three implicit contiguous intervals:
54// [0, 1], [1, 2], and [2, 3].
55class RowsetTree {
56public:
57 // An RSEndpoint is a POD which associates a rowset, an EndpointType
58 // (either the START or STOP of an interval), and the key at which the
59 // endpoint is located.
60 enum EndpointType { START, STOP };
61 struct RSEndpoint {
62 RSEndpoint(RowsetSharedPtr rowset, uint32_t segment_id, EndpointType endpoint, Slice slice)
63 : rowset_(rowset), segment_id_(segment_id), endpoint_(endpoint), slice_(slice) {}
64
65 RowsetSharedPtr rowset_;
66 uint32_t segment_id_;
67 enum EndpointType endpoint_;
68 Slice slice_;
69 };
70
71 RowsetTree();
72 Status Init(const RowsetVector& rowsets);
73 ~RowsetTree();
74
75 // Return Rowsets whose id in rowset_ids and range may contain the given encoded key.
76 //
77 // The returned pointers are guaranteed to be valid at least until this
78 // RowsetTree object is Reset().
79 void FindRowsetsWithKeyInRange(const Slice& encoded_key, const RowsetIdUnorderedSet* rowset_ids,
80 vector<std::pair<RowsetSharedPtr, int32_t>>* rowsets) const;
81
82 // Call 'cb(rowset, index)' for each (rowset, index) pair such that
83 // 'encoded_keys[index]' may be within the bounds of 'rowset'.
84 //
85 // See IntervalTree::ForEachIntervalContainingPoints for additional
86 // information on the particular order in which the callback will be called.
87 //
88 // REQUIRES: 'encoded_keys' must be in sorted order.
89 void ForEachRowsetContainingKeys(const std::vector<Slice>& encoded_keys,
90 const std::function<void(RowsetSharedPtr, int)>& cb) const;
91
92 // When 'lower_bound' is boost::none, it means negative infinity.
93 // When 'upper_bound' is boost::none, it means positive infinity.
94 // So the query interval can be one of below:
95 // [-OO, +OO)
96 // [-OO, upper_bound)
97 // [lower_bound, +OO)
98 // [lower_bound, upper_bound)
99 void FindRowsetsIntersectingInterval(
100 const std::optional<Slice>& lower_bound, const std::optional<Slice>& upper_bound,
101 vector<std::pair<RowsetSharedPtr, int32_t>>* rowsets) const;
102
103 const RowsetVector& all_rowsets() const { return all_rowsets_; }
104
105 RowsetSharedPtr rs_by_id(RowsetId rs_id) const { return FindPtrOrNull(rs_by_id_, rs_id); }
106
107 // Iterates over RowsetTree::RSEndpoint, guaranteed to be ordered and for
108 // any rowset to appear exactly twice, once at its start slice and once at
109 // its stop slice, equivalent to its GetBounds() values.
110 const std::vector<RSEndpoint>& key_endpoints() const { return key_endpoints_; }
111
112private:
113 // Interval tree of the rowsets. Used to efficiently find rowsets which might contain
114 // a probe row.
115 std::unique_ptr<IntervalTree<RowsetIntervalTraits>> tree_;
116
117 // Ordered map of all the interval endpoints, holding the implicit contiguous
118 // intervals
119 std::vector<RSEndpoint> key_endpoints_;
120
121 // Container for all of the entries in tree_. IntervalTree does
122 // not itself manage memory, so this provides a simple way to enumerate
123 // all the entry structs and free them in the destructor.
124 std::vector<RowsetWithBounds*> entries_;
125
126 // All of the rowsets which were put in this RowsetTree.
127 RowsetVector all_rowsets_;
128
129 // The Rowsets in this RowsetTree, keyed by their id.
130 std::unordered_map<RowsetId, RowsetSharedPtr, HashOfRowsetId> rs_by_id_;
131
132 bool initted_;
133};
