27class FunctionArrayCompact : public IFunction {
28public:
29 static constexpr auto name = "array_compact";
30 static FunctionPtr create() { return std::make_shared<FunctionArrayCompact>(); }
31 using NullMapType = PaddedPODArray<UInt8>;
32
34 String get_name() const override { return name; }
35
36 bool is_variadic() const override { return false; }
37
38 size_t get_number_of_arguments() const override { return 1; }
39
40 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
41 DCHECK(is_array(arguments[0]))
42 << "first argument for function: " << name << " should be DataTypeArray"
43 << " and arguments[0] is " << arguments[0]->get_name();
44 return arguments[0];
45 }
46
47 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
48 size_t result, size_t input_rows_count) override {
49 ColumnPtr src_column =
50 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
51 const auto& src_column_array = check_and_get_column<ColumnArray>(*src_column);
52 if (!src_column_array) {
53 return Status::RuntimeError(
54 fmt::format("unsupported types for function {}({})", get_name(),
55 block.get_by_position(arguments[0]).type->get_name()));
56 }
57 const auto& src_offsets = src_column_array->get_offsets();
58 const auto* src_nested_column = &src_column_array->get_data();
59 DCHECK(src_nested_column != nullptr);
60
61 DataTypePtr src_column_type = block.get_by_position(arguments[0]).type;
62 auto nested_type = assert_cast<const DataTypeArray&>(*src_column_type).get_nested_type();
63 auto dest_column_ptr = ColumnArray::create(nested_type->create_column(),
64 ColumnArray::ColumnOffsets::create());
65 IColumn* dest_nested_column = &dest_column_ptr->get_data();
66 auto& dest_offsets = dest_column_ptr->get_offsets();
67 DCHECK(dest_nested_column != nullptr);
68
69 auto res_val = _execute(*src_nested_column, src_offsets, *dest_nested_column, dest_offsets);
70 if (!res_val) {
71 return Status::RuntimeError(
72 fmt::format("execute failed or unsupported types for function {}({})",
73 get_name(), block.get_by_position(arguments[0]).type->get_name()));
74 }
75
76 block.replace_by_position(result, std::move(dest_column_ptr));
77 return Status::OK();
78 }
79
80private:
81 bool _execute(const IColumn& src_column, const ColumnArray::Offsets64& src_offsets,
82 IColumn& dest_column, ColumnArray::Offsets64& dest_offsets) {
83 ColumnArray::Offset64 src_offsets_size = src_offsets.size();
84 ColumnArray::Offset64 src_pos = 0;
85 ColumnArray::Offset64 dest_pos = 0;
86
87 for (size_t i = 0; i < src_offsets_size; ++i) {
88 auto src_offset = src_offsets[i];
89 if (src_pos < src_offset) {
90 // Insert first element
91 dest_column.insert_from(src_column, src_pos);
92
93 ++src_pos;
94 ++dest_pos;
95
96 // For the rest of elements, insert if the element is different from the previous.
97 for (; src_pos < src_offset; ++src_pos) {
98 if (0 != (src_column.compare_at(src_pos - 1, src_pos, src_column, 1))) {
99 dest_column.insert_from(src_column, src_pos);
100 ++dest_pos;
101 }
102 }
103 }
104 dest_offsets.push_back(dest_pos);
105 }
106 return true;
107 }
108};
