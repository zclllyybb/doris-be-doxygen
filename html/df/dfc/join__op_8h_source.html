66template <typename RowRefType>
67struct Batch {
68 static constexpr size_t MAX_SIZE = 7;
69
70 RowRef::SizeT size = 0;
71 Batch<RowRefType>* next;
72 RowRefType row_refs[MAX_SIZE];
73
74 Batch(Batch<RowRefType>* parent) : next(parent) {}
75
76 bool full() const { return size == MAX_SIZE; }
77
78 Batch<RowRefType>* insert(RowRefType&& row_ref, Arena& pool) {
79 if (full()) {
80 auto batch = pool.alloc<Batch<RowRefType>>();
81 *batch = Batch<RowRefType>(this);
82 batch->insert(std::move(row_ref), pool);
83 return batch;
84 }
85
86 row_refs[size++] = std::move(row_ref);
87 return this;
88 }
89};

91template <typename RowRefListType>
92class ForwardIterator {
93public:
94 using RowRefType = typename RowRefListType::RowRefType;
95 ForwardIterator() : root(nullptr), first(false), batch(nullptr), position(0) {}
96
97 ForwardIterator(RowRefListType* begin)
98 : root(begin), first(true), batch(root->next), position(0) {}
99
100 RowRefType& operator*() {
101 if (first) return *root;
102 return batch->row_refs[position];
103 }
104 RowRefType* operator->() { return &(**this); }
105
106 bool operator==(const ForwardIterator<RowRefListType>& rhs) const {
107 if (ok() != rhs.ok()) {
108 return false;
109 }
110 if (first && rhs.first) {
111 return true;
112 }
113 return batch == rhs.batch && position == rhs.position;
114 }
115 bool operator!=(const ForwardIterator<RowRefListType>& rhs) const { return !(*this == rhs); }
116
117 void operator++() {
118 if (first) {
119 first = false;
120 return;
121 }
122
123 if (batch) {
124 ++position;
125 if (position >= batch->size) {
126 batch = batch->next;
127 position = 0;
128 }
129 }
130 }
131
132 bool ok() const { return first || batch; }
133
134 static ForwardIterator<RowRefListType> end() { return ForwardIterator(); }
135
136private:
137 RowRefListType* root;
138 bool first;
139 Batch<RowRefType>* batch;
140 size_t position;
141};
