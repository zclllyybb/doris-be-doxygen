34// Container struct for statistics read from the /proc filesystem for a thread.
35struct ThreadStats {
36 int64_t user_ns;
37 int64_t kernel_ns;
38 int64_t iowait_ns;
39
40 // Default constructor zeroes all members in case structure can't be filled by
41 // GetThreadStats.
42 ThreadStats() : user_ns(0), kernel_ns(0), iowait_ns(0) {}
43};
44
45// Populates ThreadStats object using a given buffer. The buffer is expected to
46// conform to /proc/<pid>/task/<tid>/stat layout; an error will be returned otherwise.
47//
48// If 'name' is supplied, the extracted thread name will be written to it.
49Status parse_stat(const std::string& buffer, std::string* name, ThreadStats* stats);
50
51// Populates ThreadStats object for a given thread by reading from
52// /proc/<pid>/task/<tid>/stat. Returns OK unless the file cannot be read or is in an
53// unrecognised format, or if the kernel version is not modern enough.
54Status get_thread_stats(int64_t tid, ThreadStats* stats);
55
56// Disable core dumps for this process.
57//
58// This is useful particularly in tests where we have injected failures and don't
59// want to generate a core dump from an "expected" crash.
60void disable_core_dumps();

62} // namespace doris
