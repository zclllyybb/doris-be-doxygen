46template <FieldType Type>
47class BinaryPlainPageBuilder : public PageBuilder {
48public:
49 BinaryPlainPageBuilder(const PageBuilderOptions& options)
50 : _size_estimate(0), _options(options) {
51 reset();
52 }
53
54 bool is_page_full() override {
55 bool ret = false;
56 if (_options.is_dict_page) {
57 // dict_page_size is 0, do not limit the page size
58 ret = _options.dict_page_size != 0 && _size_estimate > _options.dict_page_size;
59 } else {
60 ret = _options.data_page_size != 0 && _size_estimate > _options.data_page_size;
61 }
62 return ret;
63 }
64
65 Status add(const uint8_t* vals, size_t* count) override {
66 DCHECK(!_finished);
67 DCHECK_GT(*count, 0);
68 size_t i = 0;
69
70 // If the page is full, should stop adding more items.
71 while (!is_page_full() && i < *count) {
72 auto src = reinterpret_cast<const Slice*>(vals);
73 if constexpr (Type == OLAP_FIELD_TYPE_OBJECT) {
74 if (_options.need_check_bitmap) {
75 RETURN_IF_ERROR(BitmapTypeCode::validate(*(src->data)));
76 }
77 }
78 size_t offset = _buffer.size();
79 _offsets.push_back(offset);
80 _buffer.append(src->data, src->size);
81
82 _last_value_size = src->size;
83 _size_estimate += src->size;
84 _size_estimate += sizeof(uint32_t);
85
86 i++;
87 vals += sizeof(Slice);
88 }
89
90 *count = i;
91 return Status::OK();
92 }
93
94 OwnedSlice finish() override {
95 DCHECK(!_finished);
96 _finished = true;
97 // Set up trailer
98 for (uint32_t _offset : _offsets) {
99 put_fixed32_le(&_buffer, _offset);
100 }
101 put_fixed32_le(&_buffer, _offsets.size());
102 if (_offsets.size() > 0) {
103 _copy_value_at(0, &_first_value);
104 _copy_value_at(_offsets.size() - 1, &_last_value);
105 }
106 return _buffer.build();
107 }
108
109 void reset() override {
110 _offsets.clear();
111 _buffer.clear();
112 _buffer.reserve(_options.data_page_size == 0
113 ? 1024
114 : std::min(_options.data_page_size, _options.dict_page_size));
115 _size_estimate = sizeof(uint32_t);
116 _finished = false;
117 _last_value_size = 0;
118 }
119
120 size_t count() const override { return _offsets.size(); }
121
122 uint64_t size() const override { return _size_estimate; }
123
124 Status get_first_value(void* value) const override {
125 DCHECK(_finished);
126 if (_offsets.size() == 0) {
127 return Status::NotFound("page is empty");
128 }
129 *reinterpret_cast<Slice*>(value) = Slice(_first_value);
130 return Status::OK();
131 }
132 Status get_last_value(void* value) const override {
133 DCHECK(_finished);
134 if (_offsets.size() == 0) {
135 return Status::NotFound("page is empty");
136 }
137 *reinterpret_cast<Slice*>(value) = Slice(_last_value);
138 return Status::OK();
139 }
140
141 inline Slice operator[](size_t idx) const {
142 DCHECK(!_finished);
143 DCHECK_LT(idx, _offsets.size());
144 size_t value_size =
145 (idx < _offsets.size() - 1) ? _offsets[idx + 1] - _offsets[idx] : _last_value_size;
146 return Slice(&_buffer[_offsets[idx]], value_size);
147 }
148
149 inline Slice get(std::size_t idx) const { return (*this)[idx]; }
150
151private:
152 void _copy_value_at(size_t idx, faststring* value) const {
153 size_t value_size =
154 (idx < _offsets.size() - 1) ? _offsets[idx + 1] - _offsets[idx] : _last_value_size;
155 value->assign_copy(&_buffer[_offsets[idx]], value_size);
156 }
157
158 faststring _buffer;
159 size_t _size_estimate;
160 // Offsets of each entry, relative to the start of the page
161 std::vector<uint32_t> _offsets;
162 bool _finished;
163 PageBuilderOptions _options;
164 // size of last added value
165 uint32_t _last_value_size = 0;
166 faststring _first_value;
167 faststring _last_value;
168};

170template <FieldType Type>
171class BinaryPlainPageDecoder : public PageDecoder {
172public:
173 BinaryPlainPageDecoder(Slice data) : BinaryPlainPageDecoder(data, PageDecoderOptions()) {}
174
175 BinaryPlainPageDecoder(Slice data, const PageDecoderOptions& options)
176 : _data(data),
177 _options(options),
178 _parsed(false),
179 _num_elems(0),
180 _offsets_pos(0),
181 _cur_idx(0) {}
182
183 Status init() override {
184 CHECK(!_parsed);
185
186 if (_data.size < sizeof(uint32_t)) {
187 return Status::Corruption(
188 "file corruption: not enough bytes for trailer in BinaryPlainPageDecoder ."
189 "invalid data size:{}, trailer size:{}",
190 _data.size, sizeof(uint32_t));
191 }
192
193 // Decode trailer
194 _num_elems = decode_fixed32_le((const uint8_t*)&_data[_data.get_size() - sizeof(uint32_t)]);
195 _offsets_pos = _data.get_size() - (_num_elems + 1) * sizeof(uint32_t);
196
197 _parsed = true;
198
199 return Status::OK();
200 }
201
202 Status seek_to_position_in_page(size_t pos) override {
203 DCHECK_LE(pos, _num_elems);
204 _cur_idx = pos;
205 return Status::OK();
206 }
207
208 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override {
209 DCHECK(_parsed);
210 if (PREDICT_FALSE(*n == 0 || _cur_idx >= _num_elems)) {
211 *n = 0;
212 return Status::OK();
213 }
214 const size_t max_fetch = std::min(*n, static_cast<size_t>(_num_elems - _cur_idx));
215
216 uint32_t last_offset = guarded_offset(_cur_idx);
217 uint32_t offsets[max_fetch + 1];
218 offsets[0] = last_offset;
219 for (int i = 0; i < max_fetch - 1; i++, _cur_idx++) {
220 const uint32_t start_offset = last_offset;
221 last_offset = guarded_offset(_cur_idx + 1);
222 offsets[i + 1] = last_offset;
223 if constexpr (Type == OLAP_FIELD_TYPE_OBJECT) {
224 if (_options.need_check_bitmap) {
225 RETURN_IF_ERROR(BitmapTypeCode::validate(*(_data.data + start_offset)));
226 }
227 }
228 }
229 _cur_idx++;
230 offsets[max_fetch] = offset(_cur_idx);
231 if constexpr (Type == OLAP_FIELD_TYPE_OBJECT) {
232 if (_options.need_check_bitmap) {
233 RETURN_IF_ERROR(BitmapTypeCode::validate(*(_data.data + last_offset)));
234 }
235 }
236 dst->insert_many_continuous_binary_data(_data.data, offsets, max_fetch);
237
238 *n = max_fetch;
239 return Status::OK();
240 }
241
242 Status read_by_rowids(const rowid_t* rowids, ordinal_t page_first_ordinal, size_t* n,
243 vectorized::MutableColumnPtr& dst) override {
244 DCHECK(_parsed);
245 if (PREDICT_FALSE(*n == 0)) {
246 *n = 0;
247 return Status::OK();
248 }
249
250 auto total = *n;
251 size_t read_count = 0;
252 uint32_t len_array[total];
253 uint32_t start_offset_array[total];
254 for (size_t i = 0; i < total; ++i) {
255 ordinal_t ord = rowids[i] - page_first_ordinal;
256 if (UNLIKELY(ord >= _num_elems)) {
257 break;
258 }
259
260 const uint32_t start_offset = offset(ord);
261 start_offset_array[read_count] = start_offset;
262 len_array[read_count] = offset(ord + 1) - start_offset;
263 read_count++;
264 }
265
266 if (LIKELY(read_count > 0))
267 dst->insert_many_binary_data(_data.mutable_data(), len_array, start_offset_array,
268 read_count);
269
270 *n = read_count;
271 return Status::OK();
272 }
273
274 size_t count() const override {
275 DCHECK(_parsed);
276 return _num_elems;
277 }
278
279 size_t current_index() const override {
280 DCHECK(_parsed);
281 return _cur_idx;
282 }
283
284 Slice string_at_index(size_t idx) const {
285 const uint32_t start_offset = offset(idx);
286 uint32_t len = offset(idx + 1) - start_offset;
287 return Slice(&_data[start_offset], len);
288 }
289
290 void get_dict_word_info(StringRef* dict_word_info) {
291 if (UNLIKELY(_num_elems <= 0)) {
292 return;
293 }
294
295 char* data_begin = (char*)&_data[0];
296 char* offset_ptr = (char*)&_data[_offsets_pos];
297
298 for (uint32_t i = 0; i < _num_elems; ++i) {
299 dict_word_info[i].data = data_begin + decode_fixed32_le((uint8_t*)offset_ptr);
300 offset_ptr += sizeof(uint32_t);
301 }
302
303 for (int i = 0; i < (int)_num_elems - 1; ++i) {
304 dict_word_info[i].size =
305 (char*)dict_word_info[i + 1].data - (char*)dict_word_info[i].data;
306 }
307
308 dict_word_info[_num_elems - 1].size =
309 (data_begin + _offsets_pos) - (char*)dict_word_info[_num_elems - 1].data;
310 }
311
312private:
313 static constexpr size_t SIZE_OF_INT32 = sizeof(uint32_t);
314 // Return the offset within '_data' where the string value with index 'idx' can be found.
315 uint32_t offset(size_t idx) const {
316 if (idx >= _num_elems) {
317 return _offsets_pos;
318 }
319 const uint8_t* p =
320 reinterpret_cast<const uint8_t*>(&_data[_offsets_pos + idx * SIZE_OF_INT32]);
321 return decode_fixed32_le(p);
322 }
323
324 uint32_t guarded_offset(size_t idx) const {
325 const uint8_t* p =
326 reinterpret_cast<const uint8_t*>(&_data[_offsets_pos + idx * SIZE_OF_INT32]);
327 return decode_fixed32_le(p);
328 }
329
330 Slice _data;
331 PageDecoderOptions _options;
332 bool _parsed;
333
334 uint32_t _num_elems;
335 uint32_t _offsets_pos;
336
337 // Index of the currently seeked element in the page.
338 uint32_t _cur_idx;
339 friend class BinaryDictPageDecoder;
340 friend class FileColumnIterator;
341};
