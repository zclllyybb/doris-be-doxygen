37template <typename T>
38class DecimalPaddedPODArray : public PaddedPODArray<T> {
39public:
40 using Base = PaddedPODArray<T>;
41 using Base::operator[];
42
43 DecimalPaddedPODArray(size_t size, UInt32 scale_) : Base(size), scale(scale_) {}
44
45 DecimalPaddedPODArray(const DecimalPaddedPODArray& other)
46 : Base(other.begin(), other.end()), scale(other.scale) {}
47
48 DecimalPaddedPODArray(DecimalPaddedPODArray&& other) {
49 this->swap(other);
50 std::swap(scale, other.scale);
51 }
52
53 DecimalPaddedPODArray& operator=(DecimalPaddedPODArray&& other) {
54 this->swap(other);
55 std::swap(scale, other.scale);
56 return *this;
57 }
58
59 UInt32 get_scale() const { return scale; }
60
61private:
62 UInt32 scale;
63};

250 template <typename U>
251 void permutation(bool reverse, size_t limit, PaddedPODArray<U>& res) const {
252 size_t s = data.size();
253 res.resize(s);
254 for (U i = 0; i < s; ++i) res[i] = i;
255
256 auto sort_end = res.end();
257 if (limit && limit < s) sort_end = res.begin() + limit;
258
259 if (reverse)
260 std::partial_sort(res.begin(), sort_end, res.end(),
261 [this](size_t a, size_t b) { return data[a] > data[b]; });
262 else
263 std::partial_sort(res.begin(), sort_end, res.end(),
264 [this](size_t a, size_t b) { return data[a] < data[b]; });
265 }

285template <typename Type>
286ColumnPtr ColumnDecimal<T>::index_impl(const PaddedPODArray<Type>& indexes, size_t limit) const {
287 size_t size = indexes.size();
288
289 if (limit == 0)
290 limit = size;
291 else
292 limit = std::min(size, limit);
293
294 auto res = this->create(limit, scale);
295 typename Self::Container& res_data = res->get_data();
296 for (size_t i = 0; i < limit; ++i) res_data[i] = data[indexes[i]];
297
298 return res;
299}
