5// This file contains routines for hashing and fingerprinting.
6//
7// A hash function takes an arbitrary input bitstring (string, char*,
8// number) and turns it into a hash value (a fixed-size number) such
9// that unequal input values have a high likelihood of generating
10// unequal hash values. A fingerprint is a hash whose design is
11// biased towards avoiding hash collisions, possibly at the expense of
12// other characteristics such as execution speed.
13//
14// In general, if you are only using the hash values inside a single
15// executable -- you're not writing the values to disk, and you don't
16// depend on another instance of your program, running on another
17// machine, generating the same hash values as you -- you want to use
18// a HASH. Otherwise, you want to use a FINGERPRINT.
19//
20// RECOMMENDED HASH FOR STRINGS: GoodFastHash
21//
22// It is a functor, so you can use it like this:
23// hash_map<string, xxx, GoodFastHash<string> >
24// hash_set<char *, GoodFastHash<char*> >
25//
26// RECOMMENDED HASH FOR NUMBERS: hash<>
27//
28// Note that this is likely the identity hash, so if your
29// numbers are "non-random" (especially in the low bits), another
30// choice is better. You can use it like this:
31// hash_map<int, xxx>
32// hash_set<uint64>
33//
34// RECOMMENDED HASH FOR POINTERS: hash<>
35//
36// This is also likely the identity hash.
37//
38// RECOMMENDED HASH FOR STRUCTS: hash<some_fingerprint(struct)>
39//
40// Take a fingerprint of the struct, and use that as the key.
41// For instance: const uint64 hash_data[] = { s.foo, bit_cast<uint64>(s.bar) };
42// uint64 fprint = <fingerprint fn>(reinterpret_cast<const char*>(hash_data),
43// sizeof(hash_data));
44// hash_map[fprint] = whatever;
45//
46// RECOMMENDED FINGERPRINT: Fingerprint2011
47//
48// (In util/hash/fingerprint2011.h)
49// In particular, do *not* use Fingerprint in new code; it has
50// problems with excess collisions.
51//
52// OTHER HASHES AND FINGERPRINTS:
53//
54//
55// The wiki page also has good advice for when to use a fingerprint vs
56// a hash.
57//
58//
59// Note: if your file declares hash_map<string, ...> or
60// hash_set<string>, it will use the default hash function,
61// hash<string>. This is not a great choice. Always provide an
62// explicit functor, such as GoodFastHash, as a template argument.
63// (Either way, you will need to #include this file to get the
64// necessary definition.)
65//
66// Some of the hash functions below are documented to be fixed
67// forever; the rest (whether they're documented as so or not) may
68// change over time. If you require a hash function that does not
69// change over time, you should have unittests enforcing this
70// property. We already have several such functions; see
71// hash_unittest.cc for the details and unittests.

95// ----------------------------------------------------------------------
96// Fingerprint()
97// Not recommended for new code. Instead, use Fingerprint2011(),
98// a higher-quality and faster hash function. See fingerprint2011.h.
99//
100// Fingerprinting a string (or char*) will never return 0 or 1,
101// in case you want a couple of special values. However,
102// fingerprinting a numeric type may produce 0 or 1.
103//
104// The hash mapping of Fingerprint() will never change.
105//
106// Note: AVOID USING FINGERPRINT if at all possible. Use
107// Fingerprint2011 (in fingerprint2011.h) instead.
108// Fingerprint() is susceptible to collisions for even short
109// strings with low edit distance; see
110// Example collisions:
111// "01056/02" vs. "11057/02"
112// "LTA 02" vs. "MTA 12"
113// The same study found only one collision each for CityHash64() and
114// MurmurHash64(), from more than 2^32 inputs, and on medium-length
115// strings with large edit distances.These issues, among others,
116// led to the recommendation that new code should avoid Fingerprint().
117// ----------------------------------------------------------------------
118extern uint64 FingerprintReferenceImplementation(const char* s, uint32 len);
119extern uint64 FingerprintInterleavedImplementation(const char* s, uint32 len);
120inline uint64 Fingerprint(const char* s, uint32 len) {
121 if constexpr (sizeof(s) == 8) { // 64-bit systems have 8-byte pointers.
122 // The better choice when we have a decent number of registers.
123 return FingerprintInterleavedImplementation(s, len);
124 } else {
125 return FingerprintReferenceImplementation(s, len);
126 }
127}

129// Routine that combines the hi/lo part of a fingerprint
130// and changes the result appropriately to avoid returning 0/1.
131inline uint64 CombineFingerprintHalves(uint32 hi, uint32 lo) {
132 uint64 result = (static_cast<uint64>(hi) << 32) | static_cast<uint64>(lo);
133 if ((hi == 0) && (lo < 2)) {
134 result ^= GG_ULONGLONG(0x130f9bef94a0a928);
135 }
136 return result;
137}

170// This concatenates two 64-bit fingerprints. It is a convenience function to
171// get a fingerprint for a combination of already fingerprinted components.
172// It assumes that each input is already a good fingerprint itself.
173// Note that this is legacy code and new code should use its replacement
174// FingerprintCat2011().
175//
176// Note that in general it's impossible to construct Fingerprint(str)
177// from the fingerprints of substrings of str. One shouldn't expect
178// FingerprintCat(Fingerprint(x), Fingerprint(y)) to indicate
179// anything about Fingerprint(StrCat(x, y)).
180inline uint64 FingerprintCat(uint64 fp1, uint64 fp2) {
181 return Hash64NumWithSeed(fp1, fp2);
182}

184// This intended to be a "good" hash function. It may change from time to time.
185template <>
186struct std::hash<uint128> {
187 size_t operator()(const uint128& x) const {
188 if (sizeof(&x) == 8) { // 64-bit systems have 8-byte pointers.
189 return Hash128to64(x);
190 } else {
191 uint32 a = static_cast<uint32>(Uint128Low64(x)) + static_cast<uint32>(0x9e3779b9UL);
192 uint32 b =
193 static_cast<uint32>(Uint128Low64(x) >> 32) + static_cast<uint32>(0x9e3779b9UL);
194 uint32 c = static_cast<uint32>(Uint128High64(x)) + MIX32;
195 mix(a, b, c);
196 a += static_cast<uint32>(Uint128High64(x) >> 32);
197 mix(a, b, c);
198 return c;
199 }
200 }
201 static const size_t min_buckets = 8; // 4 and 8 are defaults.
202};

204// Hasher for STL pairs. Requires hashers for both members to be defined
205template <class First, class Second>
206struct std::hash<std::pair<First, Second> > {
207 size_t operator()(const pair<First, Second>& p) const {
208 size_t h1 = std::hash<First>()(p.first);
209 size_t h2 = std::hash<Second>()(p.second);
210 // The decision below is at compile time
211 return (sizeof(h1) <= sizeof(uint32)) ? Hash32NumWithSeed(h1, h2)
212 : Hash64NumWithSeed(h1, h2);
213 }
214 static const size_t min_buckets = 8; // 4 and 8 are defaults.
215};
