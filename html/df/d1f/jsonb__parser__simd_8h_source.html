11/*
12 * This file defines JsonbParserTSIMD (template) and JsonbParser.
13 *
14 * JsonbParserTSIMD is a template class which implements a JSON parser.
15 * JsonbParserTSIMD parses JSON text, and serialize it to JSONB binary format
16 * by using JsonbWriterT object. By default, JsonbParserTSIMD creates a new
17 * JsonbWriterT object with an output stream object. However, you can also
18 * pass in your JsonbWriterT or any stream object that implements some basic
19 * interface of std::ostream (see JsonbStream.h).
20 *
21 * JsonbParser specializes JsonbParserTSIMD with JsonbOutStream type (see
22 * JsonbStream.h). So unless you want to provide own a different output stream
23 * type, use JsonbParser object.
24 *
25 * ** Parsing JSON **
26 * JsonbParserTSIMD parses JSON string, and directly serializes into JSONB
27 * packed bytes. There are three ways to parse a JSON string: (1) using
28 * c-string, (2) using string with len, (3) using std::istream object. You can
29 * use custom streambuf to redirect output. JsonbOutBuffer is a streambuf used
30 * internally if the input is raw character buffer.
31 *
32 * You can reuse an JsonbParserTSIMD object to parse/serialize multiple JSON
33 * strings, and the previous JSONB will be overwritten.
34 *
35 * If parsing fails (returned false), the error code will be set to one of
36 * JsonbErrType, and can be retrieved by calling getErrorCode().
37 *
38 * ** External dictionary **
39 * During parsing a JSON string, you can pass a call-back function to map a key
40 * string to an id, and store the dictionary id in JSONB to save space. The
41 * purpose of using an external dictionary is more towards a collection of
42 * documents (which has common keys) rather than a single document, so that
43 * space saving will be significant.
44 *
45 * ** Endianness **
46 * Note: JSONB serialization doesn't assume endianness of the server. However
47 * you will need to ensure that the endianness at the reader side is the same
48 * as that at the writer side (if they are on different machines). Otherwise,
49 * proper conversion is needed when a number value is returned to the
50 * caller/writer.
51 *
52 * @author Tian Xia <tianx@fb.com>
53 *
54 * this file is copied from
55 * https://github.com/facebook/mysql-5.6/blob/fb-mysql-5.6.35/fbson/FbsonJsonParser.h
56 * and modified by Doris
57 */

78class JsonbParserTSIMD {

84 // parse a UTF-8 JSON string
85 bool parse(const std::string& str, hDictInsert handler = nullptr) {
86 return parse(str.c_str(), (unsigned int)str.size(), handler);
87 }
88
89 // parse a UTF-8 JSON c-style string (NULL terminated)
90 bool parse(const char* c_str, hDictInsert handler = nullptr) {
91 return parse(c_str, (unsigned int)strlen(c_str), handler);
92 }
93
94 // parse a UTF-8 JSON string with length
95 bool parse(const char* pch, unsigned int len, hDictInsert handler = nullptr) {
96 // reset state before parse
97 reset();
98
99 if (!pch || len == 0) {
100 err_ = JsonbErrType::E_EMPTY_DOCUMENT;
101 LOG(WARNING) << "empty json string";
102 return false;
103 }
104
105 // parse json using simdjson, return false on exception
106 try {
107 simdjson::padded_string json_str {pch, len};
108 simdjson::ondemand::document doc = parser_.iterate(json_str);
109
110 // simdjson process top level primitive types specially
111 // so some repeated code here
112 switch (doc.type()) {
113 case simdjson::ondemand::json_type::object:
114 case simdjson::ondemand::json_type::array: {
115 parse(doc.get_value(), handler);
116 break;
117 }
118 case simdjson::ondemand::json_type::null: {
119 if (writer_.writeNull() == 0) {
120 err_ = JsonbErrType::E_OUTPUT_FAIL;
121 LOG(WARNING) << "writeNull failed";
122 }
123 break;
124 }
125 case simdjson::ondemand::json_type::boolean: {
126 if (writer_.writeBool(doc.get_bool()) == 0) {
127 err_ = JsonbErrType::E_OUTPUT_FAIL;
128 LOG(WARNING) << "writeBool failed";
129 }
130 break;
131 }
132 case simdjson::ondemand::json_type::string: {
133 write_string(doc.get_string());
134 break;
135 }
136 case simdjson::ondemand::json_type::number: {
137 write_number(doc.get_number());
138 break;
139 }
140 }
141
142 return err_ == JsonbErrType::E_NONE;
143 } catch (simdjson::simdjson_error& e) {
144 err_ = JsonbErrType::E_EXCEPTION;
145 LOG(WARNING) << "simdjson parse exception: " << e.what();
146 return false;
147 }
148 }
149
150 // parse json, recursively if necessary, by simdjson
151 // and serialize to binary format by writer
152 void parse(simdjson::ondemand::value value, hDictInsert handler = nullptr) {
153 switch (value.type()) {
154 case simdjson::ondemand::json_type::null: {
155 if (writer_.writeNull() == 0) {
156 err_ = JsonbErrType::E_OUTPUT_FAIL;
157 LOG(WARNING) << "writeNull failed";
158 }
159 break;
160 }
161 case simdjson::ondemand::json_type::boolean: {
162 if (writer_.writeBool(value.get_bool()) == 0) {
163 err_ = JsonbErrType::E_OUTPUT_FAIL;
164 LOG(WARNING) << "writeBool failed";
165 }
166 break;
167 }
168 case simdjson::ondemand::json_type::string: {
169 write_string(value.get_string());
170 break;
171 }
172 case simdjson::ondemand::json_type::number: {
173 write_number(value.get_number());
174 break;
175 }
176 case simdjson::ondemand::json_type::object: {
177 if (!writer_.writeStartObject()) {
178 err_ = JsonbErrType::E_OUTPUT_FAIL;
179 LOG(WARNING) << "writeStartObject failed";
180 break;
181 }
182
183 for (auto kv : value.get_object()) {
184 std::string_view key;
185 simdjson::error_code e = kv.unescaped_key().get(key);
186 if (e != simdjson::SUCCESS) {
187 err_ = JsonbErrType::E_INVALID_KEY_STRING;
188 LOG(WARNING) << "simdjson get key failed: " << e;
189 break;
190 }
191
192 int key_id = -1;
193 if (handler) {
194 key_id = handler(key.data(), key.size());
195 }
196
197 if (key_id < 0) {
198 if (writer_.writeKey(key.data(), key.size()) == 0) {
199 err_ = JsonbErrType::E_OUTPUT_FAIL;
200 LOG(WARNING) << "writeKey failed key: " << key;
201 break;
202 }
203 } else {
204 if (writer_.writeKey(key_id) == 0) {
205 err_ = JsonbErrType::E_OUTPUT_FAIL;
206 LOG(WARNING) << "writeKey failed key_id: " << key_id;
207 break;
208 }
209 }
210
211 // parse object value
212 parse(kv.value(), handler);
213 if (err_ != JsonbErrType::E_NONE) {
214 LOG(WARNING) << "parse object value failed";
215 break;
216 }
217 }
218 if (err_ != JsonbErrType::E_NONE) {
219 break;
220 }
221
222 if (!writer_.writeEndObject()) {
223 err_ = JsonbErrType::E_OUTPUT_FAIL;
224 LOG(WARNING) << "writeEndObject failed";
225 break;
226 }
227
228 break;
229 }
230 case simdjson::ondemand::json_type::array: {
231 if (!writer_.writeStartArray()) {
232 err_ = JsonbErrType::E_OUTPUT_FAIL;
233 LOG(WARNING) << "writeStartArray failed";
234 break;
235 }
236
237 for (auto elem : value.get_array()) {
238 // parse array element
239 parse(elem.value(), handler);
240 if (err_ != JsonbErrType::E_NONE) {
241 LOG(WARNING) << "parse array element failed";
242 break;
243 }
244 }
245 if (err_ != JsonbErrType::E_NONE) {
246 break;
247 }
248
249 if (!writer_.writeEndArray()) {
250 err_ = JsonbErrType::E_OUTPUT_FAIL;
251 LOG(WARNING) << "writeEndArray failed";
252 break;
253 }
254
255 break;
256 }
257 default: {
258 err_ = JsonbErrType::E_INVALID_TYPE;
259 LOG(WARNING) << "unknown value type: "; // << value;
260 break;
261 }
262
263 } // end of switch
264 }

266 void write_string(std::string_view str) {
267 // start writing string
268 if (!writer_.writeStartString()) {
269 err_ = JsonbErrType::E_OUTPUT_FAIL;
270 LOG(WARNING) << "writeStartString failed";
271 return;
272 }
273
274 // write string
275 if (str.size() > 0) {
276 if (writer_.writeString(str.data(), str.size()) == 0) {
277 err_ = JsonbErrType::E_OUTPUT_FAIL;
278 LOG(WARNING) << "writeString failed";
279 return;
280 }
281 }
282
283 // end writing string
284 if (!writer_.writeEndString()) {
285 err_ = JsonbErrType::E_OUTPUT_FAIL;
286 LOG(WARNING) << "writeEndString failed";
287 return;
288 }
289 }

291 void write_number(simdjson::ondemand::number num) {
292 if (num.is_double()) {
293 if (writer_.writeDouble(num.get_double()) == 0) {
294 err_ = JsonbErrType::E_OUTPUT_FAIL;
295 LOG(WARNING) << "writeDouble failed";
296 return;
297 }
298 } else if (num.is_int64() || num.is_uint64()) {
299 if (num.is_uint64() && num.get_uint64() > std::numeric_limits<int64_t>::max()) {
300 err_ = JsonbErrType::E_OCTAL_OVERFLOW;
301 LOG(WARNING) << "overflow number: " << num.get_uint64();
302 return;
303 }
304 int64_t val = num.is_int64() ? num.get_int64() : num.get_uint64();
305 int size = 0;
306 if (val <= std::numeric_limits<int8_t>::max()) {
307 size = writer_.writeInt8((int8_t)val);
308 } else if (val <= std::numeric_limits<int16_t>::max()) {
309 size = writer_.writeInt16((int16_t)val);
310 } else if (val <= std::numeric_limits<int32_t>::max()) {
311 size = writer_.writeInt32((int32_t)val);
312 } else { // val <= INT64_MAX
313 size = writer_.writeInt64(val);
314 }
315
316 if (size == 0) {
317 err_ = JsonbErrType::E_OUTPUT_FAIL;
318 LOG(WARNING) << "writeInt failed";
319 return;
320 }
321 } else {
322 err_ = JsonbErrType::E_INVALID_NUMBER;
323 LOG(WARNING) << "invalid number: " << num.as_double();
324 return;
325 }
326 }

328 JsonbWriterT<OS_TYPE>& getWriter() { return writer_; }
329
330 JsonbErrType getErrorCode() { return err_; }
331
332 // clear error code
333 void clearErr() { err_ = JsonbErrType::E_NONE; }
334
335 void reset() {
336 writer_.reset();
337 clearErr();
338 }
339
340private:
341 simdjson::ondemand::parser parser_;
342 JsonbWriterT<OS_TYPE> writer_;
343 JsonbErrType err_;
344};
