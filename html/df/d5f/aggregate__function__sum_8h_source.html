31template <typename T>
32struct AggregateFunctionSumData {
33 T sum {};
34
35 void add(T value) { sum += value; }
36
37 void merge(const AggregateFunctionSumData& rhs) { sum += rhs.sum; }
38
39 void write(BufferWritable& buf) const { write_binary(sum, buf); }
40
41 void read(BufferReadable& buf) { read_binary(sum, buf); }
42
43 T get() const { return sum; }
44};
45
47template <typename T, typename TResult, typename Data>
48class AggregateFunctionSum final
49 : public IAggregateFunctionDataHelper<Data, AggregateFunctionSum<T, TResult, Data>> {
50public:
51 using ResultDataType = std::conditional_t<IsDecimalNumber<T>, DataTypeDecimal<TResult>,
52 DataTypeNumber<TResult>>;
53 using ColVecType = std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<T>, ColumnVector<T>>;
54 using ColVecResult =
55 std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<TResult>, ColumnVector<TResult>>;
56
57 String get_name() const override { return "sum"; }
58
59 AggregateFunctionSum(const DataTypes& argument_types_)
60 : IAggregateFunctionDataHelper<Data, AggregateFunctionSum<T, TResult, Data>>(
61 argument_types_, {}),
62 scale(0) {}
63
64 AggregateFunctionSum(const IDataType& data_type, const DataTypes& argument_types_)
65 : IAggregateFunctionDataHelper<Data, AggregateFunctionSum<T, TResult, Data>>(
66 argument_types_, {}),
67 scale(get_decimal_scale(data_type)) {}
68
69 DataTypePtr get_return_type() const override {
70 if constexpr (IsDecimalNumber<T>) {
71 return std::make_shared<ResultDataType>(ResultDataType::max_precision(), scale);
72 } else {
73 return std::make_shared<ResultDataType>();
74 }
75 }
76
77 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
78 Arena*) const override {
79 const auto& column = static_cast<const ColVecType&>(*columns[0]);
80 this->data(place).add(column.get_data()[row_num]);
81 }
82
83 void reset(AggregateDataPtr place) const override { this->data(place).sum = {}; }
84
85 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
86 Arena*) const override {
87 this->data(place).merge(this->data(rhs));
88 }
89
90 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
91 this->data(place).write(buf);
92 }
93
94 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
95 Arena*) const override {
96 this->data(place).read(buf);
97 }
98
99 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
100 auto& column = static_cast<ColVecResult&>(to);
101 column.get_data().push_back(this->data(place).get());
102 }
103
104 void deserialize_from_column(AggregateDataPtr places, const IColumn& column, Arena* arena,
105 size_t num_rows) const override {
106 auto data = assert_cast<const ColVecResult&>(column).get_data().data();
107 auto dst_data = reinterpret_cast<Data*>(places);
108 for (size_t i = 0; i != num_rows; ++i) {
109 dst_data[i].sum = data[i];
110 }
111 }
112
113 void serialize_to_column(const std::vector<AggregateDataPtr>& places, size_t offset,
114 MutableColumnPtr& dst, const size_t num_rows) const override {
115 auto& col = assert_cast<ColVecResult&>(*dst);
116 col.resize(num_rows);
117 auto* data = col.get_data().data();
118 for (size_t i = 0; i != num_rows; ++i) {
119 data[i] = this->data(places[i] + offset).sum;
120 }
121 }
122
123 void streaming_agg_serialize_to_column(const IColumn** columns, MutableColumnPtr& dst,
124 const size_t num_rows, Arena* arena) const override {
125 auto& col = assert_cast<ColVecResult&>(*dst);
126 auto& src = assert_cast<const ColVecType&>(*columns[0]);
127 col.resize(num_rows);
128 auto* src_data = src.get_data().data();
129 auto* dst_data = col.get_data().data();
130 for (size_t i = 0; i != num_rows; ++i) {
131 dst_data[i] = src_data[i];
132 }
133 }
134
135 void deserialize_and_merge_from_column(AggregateDataPtr __restrict place, const IColumn& column,
136 Arena* arena) const override {
137 auto data = assert_cast<const ColVecResult&>(column).get_data().data();
138 const size_t num_rows = column.size();
139 for (size_t i = 0; i != num_rows; ++i) {
140 this->data(place).sum += data[i];
141 }
142 }
143
144 void serialize_without_key_to_column(ConstAggregateDataPtr __restrict place,
145 MutableColumnPtr& dst) const override {
146 auto& col = assert_cast<ColVecResult&>(*dst);
147 col.resize(1);
148 reinterpret_cast<Data*>(col.get_data().data())->sum = this->data(place).sum;
149 }
150
151 MutableColumnPtr create_serialize_column() const override {
152 return get_return_type()->create_column();
153 }
154
155 DataTypePtr get_serialized_type() const override { return get_return_type(); }
156
157private:
158 UInt32 scale;
159};
