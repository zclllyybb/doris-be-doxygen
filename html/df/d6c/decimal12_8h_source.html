28// the sign of integer must be same as fraction
29struct decimal12_t {
30 decimal12_t& operator+=(const decimal12_t& value) {
31 fraction += value.fraction;
32 integer += value.integer;
33
34 if (fraction >= FRAC_RATIO) {
35 integer += 1;
36 fraction -= FRAC_RATIO;
37 } else if (fraction <= -FRAC_RATIO) {
38 integer -= 1;
39 fraction += FRAC_RATIO;
40 }
41
42 // if sign of fraction is different from integer
43 if ((fraction != 0) && (integer != 0) && (fraction ^ integer) < 0) {
44 bool sign = integer < 0;
45 integer += (sign ? 1 : -1);
46 fraction += (sign ? -FRAC_RATIO : FRAC_RATIO);
47 }
48
49 //LOG(WARNING) << "agg: int=" << integer << ", frac=" << fraction;
50 //_set_flag();
51 return *this;
52 }

66 int32_t cmp(const decimal12_t& other) const {
67 if (integer > other.integer) {
68 return 1;
69 } else if (integer == other.integer) {
70 if (fraction > other.fraction) {
71 return 1;
72 } else if (fraction == other.fraction) {
73 return 0;
74 }
75 }
76
77 return -1;
78 }

80 std::string to_string() const {
81 char buf[128] = {'\0'};
82
83 if (integer < 0 || fraction < 0) {
84 snprintf(buf, sizeof(buf), "-%" PRIu64 ".%09u", std::abs(integer), std::abs(fraction));
85 } else {
86 snprintf(buf, sizeof(buf), "%" PRIu64 ".%09u", std::abs(integer), std::abs(fraction));
87 }
88
89 return std::string(buf);

92 Status from_string(const std::string& str) {
93 integer = 0;
94 fraction = 0;
95 const char* value_string = str.c_str();
96 const char* sign = strchr(value_string, '-');
97
98 if (sign != nullptr) {
99 if (sign != value_string) {
100 return Status::Error<ErrorCode::INVALID_ARGUMENT>();
101 } else {
102 ++value_string;
103 }
104 }
105
106 const char* sepr = strchr(value_string, '.');
107 if ((sepr != nullptr && sepr - value_string > MAX_INT_DIGITS_NUM) ||
108 (sepr == nullptr && strlen(value_string) > MAX_INT_DIGITS_NUM)) {
109 integer = 999999999999999999;
110 fraction = 999999999;
111 } else {
112 int32_t f = 0;
113 int64_t i = 0;
114 if (sepr == value_string) {
115 int32_t f = 0;
116 sscanf(value_string, ".%9d", &f);
117 } else {
118 sscanf(value_string, "%18" PRId64 ".%9d", &i, &f);
119 }
120 integer = i;
121 fraction = f;
122
123 int32_t frac_len = (nullptr != sepr) ? MAX_FRAC_DIGITS_NUM - strlen(sepr + 1)
124 : MAX_FRAC_DIGITS_NUM;
125 frac_len = frac_len > 0 ? frac_len : 0;
126 fraction *= g_power_table[frac_len];
127 }
128
129 if (sign != nullptr) {
130 fraction = -fraction;
131 integer = -integer;
132 }
133
134 return Status::OK();
135 }

139 static const int32_t MAX_FRAC_DIGITS_NUM = 9;

141 int64_t integer;
142 int32_t fraction;
143} __attribute__((packed));
