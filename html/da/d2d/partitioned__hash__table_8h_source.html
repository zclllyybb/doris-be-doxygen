554private:
555 void convert_to_partitioned() {
556 SCOPED_RAW_TIMER(&_convert_timer_ns);
557
558 auto bucket_count = level0_sub_table.get_buffer_size_in_cells();
559 for (size_t i = 0; i < NUM_LEVEL1_SUB_TABLES; ++i) {
560 level1_sub_tables[i] = std::move(Impl(bucket_count / NUM_LEVEL1_SUB_TABLES));
561 }
562
563 auto it = level0_sub_table.begin();
564
565 if constexpr (HashTableTraits<Impl>::is_phmap) {
566 for (; it != level0_sub_table.end(); ++it) {
567 size_t hash_value = level0_sub_table.hash(it.get_first());
568 size_t sub_table_idx = get_sub_table_from_hash(hash_value);
569 level1_sub_tables[sub_table_idx].insert(it.get_first(), hash_value,
570 it.get_second());
571 }
572 } else {
574 if (it != level0_sub_table.end() && it.get_ptr()->is_zero(level0_sub_table)) {
575 insert(it->get_value());
576 ++it;
577 }
578
579 for (; it != level0_sub_table.end(); ++it) {
580 const auto* cell = it.get_ptr();
581 size_t hash_value = cell->get_hash(level0_sub_table);
582 size_t sub_table_idx = get_sub_table_from_hash(hash_value);
583 level1_sub_tables[sub_table_idx].insert_unique_non_zero(cell, hash_value);
584 }
585 }
586
587 _is_partitioned = true;
588 level0_sub_table.clear_and_shrink();
589 }

592 static size_t get_sub_table_from_hash(size_t hash_value) {
593 return (hash_value >> (32 - BITS_FOR_SUB_TABLE)) & MAX_SUB_TABLE;
594 }
