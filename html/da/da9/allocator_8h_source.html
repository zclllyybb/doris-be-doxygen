133template <bool clear_memory_, bool mmap_populate>
134class Allocator {
135public:
137 void* alloc(size_t size, size_t alignment = 0) {
138 void* buf;
139
140 if (size >= MMAP_THRESHOLD) {
141 if (alignment > MMAP_MIN_ALIGNMENT)
142 throw doris::vectorized::Exception(
143 fmt::format(
144 "Too large alignment {}: more than page size when allocating {}.",
145 alignment, size),
146 doris::TStatusCode::VEC_BAD_ARGUMENTS);
147
148 if (!TRY_CONSUME_THREAD_MEM_TRACKER(size)) {
149 RETURN_BAD_ALLOC_IF_PRE_CATCH(
150 fmt::format("Allocator Pre Catch: Cannot mmap {}.", size));
151 // memory exceeds the limit, consume mem tracker fails, but there is no external catch bad_alloc,
152 // alloc will continue to execute, so the consume memtracker is forced.
153 CONSUME_THREAD_MEM_TRACKER(size);
154 }
155 buf = mmap(get_mmap_hint(), size, PROT_READ | PROT_WRITE, mmap_flags, -1, 0);
156 if (MAP_FAILED == buf) {
157 RELEASE_THREAD_MEM_TRACKER(size);
158 RETURN_BAD_ALLOC(fmt::format("Allocator: Cannot mmap {}.", size));
159 }
160
162 } else if (!doris::config::disable_chunk_allocator_in_vec && size >= CHUNK_THRESHOLD) {
163 doris::Chunk chunk;
164 if (!doris::ChunkAllocator::instance()->allocate_align(size, &chunk)) {
165 RETURN_BAD_ALLOC(fmt::format("Allocator: Cannot allocate chunk {}.", size));
166 }
167 buf = chunk.data;
168 if constexpr (clear_memory) memset(buf, 0, chunk.size);
169 } else {
170 if (alignment <= MALLOC_MIN_ALIGNMENT) {
171 if constexpr (clear_memory)
172 buf = ::calloc(size, 1);
173 else
174 buf = ::malloc(size);
175
176 if (nullptr == buf) {
177 RETURN_BAD_ALLOC(fmt::format("Allocator: Cannot malloc {}.", size));
178 }
179 } else {
180 buf = nullptr;
181 int res = posix_memalign(&buf, alignment, size);
182
183 if (0 != res) {
184 RETURN_BAD_ALLOC(
185 fmt::format("Cannot allocate memory (posix_memalign) {}.", size));
186 }
187
188 if constexpr (clear_memory) memset(buf, 0, size);
189 }
190 }
191 return buf;
192 }
193
195 void free(void* buf, size_t size) {
196 if (size >= MMAP_THRESHOLD) {
197 if (0 != munmap(buf, size)) {
198 auto err = fmt::format("Allocator: Cannot munmap {}.", size);
199 LOG(ERROR) << err;
200 if (!doris::enable_thread_catch_bad_alloc)
201 doris::MemTrackerLimiter::print_log_process_usage(err);
202 throw std::bad_alloc {};
203 } else {
204 RELEASE_THREAD_MEM_TRACKER(size);
205 }
206 } else if (!doris::config::disable_chunk_allocator_in_vec && size >= CHUNK_THRESHOLD &&
207 ((size & (size - 1)) == 0)) {
208 // Only power-of-two length are added to ChunkAllocator
209 doris::ChunkAllocator::instance()->free((uint8_t*)buf, size);
210 } else {
211 ::free(buf);
212 }
213 }
214
219 void* realloc(void* buf, size_t old_size, size_t new_size, size_t alignment = 0) {
220 if (old_size == new_size) {
223 } else if (old_size < CHUNK_THRESHOLD && new_size < CHUNK_THRESHOLD &&
224 alignment <= MALLOC_MIN_ALIGNMENT) {
226 void* new_buf = ::realloc(buf, new_size);
227 if (nullptr == new_buf) {
228 RETURN_BAD_ALLOC(fmt::format("Allocator: Cannot realloc from {} to {}.", old_size,
229 new_size));
230 }
231
232 buf = new_buf;
233 if constexpr (clear_memory)
234 if (new_size > old_size)
235 memset(reinterpret_cast<char*>(buf) + old_size, 0, new_size - old_size);
236 } else if (old_size >= MMAP_THRESHOLD && new_size >= MMAP_THRESHOLD) {
238 if (!TRY_CONSUME_THREAD_MEM_TRACKER(new_size - old_size)) {
239 RETURN_BAD_ALLOC_IF_PRE_CATCH(fmt::format(
240 "Allocator Pre Catch: Cannot mremap memory chunk from {} to {}.", old_size,
241 new_size));
242 CONSUME_THREAD_MEM_TRACKER(new_size - old_size);
243 }
244
245 // On apple and freebsd self-implemented mremap used (common/mremap.h)
246 buf = clickhouse_mremap(buf, old_size, new_size, MREMAP_MAYMOVE, PROT_READ | PROT_WRITE,
247 mmap_flags, -1, 0);
248 if (MAP_FAILED == buf) {
249 RELEASE_THREAD_MEM_TRACKER(new_size - old_size);
250 RETURN_BAD_ALLOC(fmt::format("Allocator: Cannot mremap memory chunk from {} to {}.",
251 old_size, new_size));
252 }
253
255
256 if constexpr (mmap_populate) {
257 // MAP_POPULATE seems have no effect for mremap as for mmap,
258 // Clear enlarged memory range explicitly to pre-fault the pages
259 if (new_size > old_size)
260 memset(reinterpret_cast<char*>(buf) + old_size, 0, new_size - old_size);
261 }
262 } else {
263 // CHUNK_THRESHOLD <= old_size <= MMAP_THRESHOLD use system realloc is slow, use ChunkAllocator.
264 // Big allocs that requires a copy.
265 void* new_buf = alloc(new_size, alignment);
266 memcpy(new_buf, buf, std::min(old_size, new_size));
267 free(buf, old_size);
268 buf = new_buf;
269 }
270
271 return buf;
272 }
273
274protected:
275 static constexpr size_t get_stack_threshold() { return 0; }
276
277 static constexpr bool clear_memory = clear_memory_;
278
279 // Freshly mmapped pages are copy-on-write references to a global zero page.
280 // On the first write, a page fault occurs, and an actual writable page is
281 // allocated. If we are going to use this memory soon, such as when resizing
282 // hash tables, it makes sense to pre-fault the pages by passing
283 // MAP_POPULATE to mmap(). This takes some time, but should be faster
284 // overall than having a hot loop interrupted by page faults.
285 // It is only supported on Linux.
286 static constexpr int mmap_flags = MAP_PRIVATE | MAP_ANONYMOUS
287#if defined(OS_LINUX)
288 | (mmap_populate ? MAP_POPULATE : 0)
289#endif
290 ;
291
292private:
293#ifndef NDEBUG
297 void* get_mmap_hint() {
298 // return reinterpret_cast<void *>(std::uniform_int_distribution<intptr_t>(0x100000000000UL, 0x700000000000UL)(thread_local_rng));
299 return nullptr;
300 }
301#else
302 void* get_mmap_hint() { return nullptr; }
303#endif
304};

308template <typename Base, size_t N, size_t Alignment>
309class AllocatorWithStackMemory : private Base {
310private:
311 alignas(Alignment) char stack_memory[N];
312
313public:
317 AllocatorWithStackMemory(const AllocatorWithStackMemory&) = delete;
318 AllocatorWithStackMemory& operator=(const AllocatorWithStackMemory&) = delete;
319 AllocatorWithStackMemory() = default;
320 ~AllocatorWithStackMemory() = default;
321
322 void* alloc(size_t size) {
323 if (size <= N) {
324 if constexpr (Base::clear_memory) memset(stack_memory, 0, N);
325 return stack_memory;
326 }
327
328 return Base::alloc(size, Alignment);
329 }
330
331 void free(void* buf, size_t size) {
332 if (size > N) Base::free(buf, size);
333 }
334
335 void* realloc(void* buf, size_t old_size, size_t new_size) {
337 if (new_size <= N) return buf;
338
340 if (old_size > N) return Base::realloc(buf, old_size, new_size, Alignment);
341
343 void* new_buf = Base::alloc(new_size, Alignment);
344 memcpy(new_buf, buf, old_size);
345 return new_buf;
346 }
347
348protected:
349 static constexpr size_t get_stack_threshold() { return N; }
350};
