28// Return the number of bytes necessary to store the given number of bits.
29inline size_t BitmapSize(size_t num_bits) {
30 return (num_bits + 7) / 8;
31}

33// Set the given bit.
34inline void BitmapSet(uint8_t* bitmap, size_t idx) {
35 bitmap[idx >> 3] |= 1 << (idx & 7);
36}

38// Switch the given bit to the specified value.
39inline void BitmapChange(uint8_t* bitmap, size_t idx, bool value) {
40 bitmap[idx >> 3] = (bitmap[idx >> 3] & ~(1 << (idx & 7))) | ((!!value) << (idx & 7));
41}

43// Clear the given bit.
44inline void BitmapClear(uint8_t* bitmap, size_t idx) {
45 bitmap[idx >> 3] &= ~(1 << (idx & 7));
46}

48// Test/get the given bit.
49inline bool BitmapTest(const uint8_t* bitmap, size_t idx) {
50 return bitmap[idx >> 3] & (1 << (idx & 7));
51}

53// Merge the two bitmaps using bitwise or. Both bitmaps should have at least
54// n_bits valid bits.
55inline void BitmapMergeOr(uint8_t* dst, const uint8_t* src, size_t n_bits) {
56 size_t n_bytes = BitmapSize(n_bits);
57 for (size_t i = 0; i < n_bytes; i++) {
58 *dst++ |= *src++;
59 }
60}

95// Returns true if the two bitmaps are equal.
96//
97// It is assumed that both bitmaps have 'bitmap_size' number of bits.
98inline bool BitmapEquals(const uint8_t* bm1, const uint8_t* bm2, size_t bitmap_size) {
99 // Use memeq() to check all of the full bytes.
100 size_t num_full_bytes = bitmap_size >> 3;
101 if (!strings::memeq(bm1, bm2, num_full_bytes)) {
102 return false;
103 }
104
105 // Check any remaining bits in one extra operation.
106 size_t num_remaining_bits = bitmap_size - (num_full_bytes << 3);
107 if (num_remaining_bits == 0) {
108 return true;
109 }
110 DCHECK_LT(num_remaining_bits, 8);
111 uint8_t mask = (1 << num_remaining_bits) - 1;
112 return (bm1[num_full_bytes] & mask) == (bm2[num_full_bytes] & mask);
113}

122// Iterator which yields ranges of set and unset bits.
123// Example usage:
124// bool value;
125// size_t size;
126// BitmapIterator iter(bitmap, n_bits);
127// while ((size = iter.Next(&value))) {
128// printf("bitmap block len=%lu value=%d\n", size, value);
129// }
130class BitmapIterator {
131public:
132 BitmapIterator(const uint8_t* map, size_t num_bits)
133 : offset_(0), num_bits_(num_bits), map_(map) {}
134
135 void Reset(const uint8_t* map, size_t num_bits) {
136 offset_ = 0;
137 num_bits_ = num_bits;
138 map_ = map;
139 }
140
141 void Reset() {
142 offset_ = 0;
143 num_bits_ = 0;
144 map_ = nullptr;
145 }
146
147 bool done() const { return (num_bits_ - offset_) == 0; }
148
149 void SeekTo(size_t bit) {
150 DCHECK_LE(bit, num_bits_);
151 offset_ = bit;
152 }
153
154 size_t Next(bool* value, size_t max_run) {
155 return NextWithLimit(value, std::min(num_bits_, max_run + offset_));
156 }
157
158 size_t Next(bool* value) { return NextWithLimit(value, num_bits_); }
159
160private:
161 size_t NextWithLimit(bool* value, size_t limit) {
162 size_t len = limit - offset_;
163 if (PREDICT_FALSE(len == 0)) return (0);
164
165 *value = BitmapTest(map_, offset_);
166
167 size_t index;
168 if (BitmapFindFirst(map_, offset_, limit, !(*value), &index)) {
169 len = index - offset_;
170 } else {
171 index = limit;
172 }
173
174 offset_ = index;
175 return len;
176 }
177
178private:
179 size_t offset_;
180 size_t num_bits_;
181 const uint8_t* map_;
182};

190class Bitmap {
191public:
192 Bitmap(int64_t num_bits) {
193 DCHECK_GE(num_bits, 0);
194 buffer_.resize(BitUtil::round_up_numi_64(num_bits));
195 num_bits_ = num_bits;
196 }
197
199 void Reset(int64_t num_bits) {
200 DCHECK_GE(num_bits, 0);
201 buffer_.resize(BitUtil::round_up_numi_64(num_bits));
202 num_bits_ = num_bits;
203 SetAllBits(false);
204 }
205
207 static int64_t MemUsage(int64_t num_bits) {
208 DCHECK_GE(num_bits, 0);
209 return BitUtil::round_up_numi_64(num_bits) * sizeof(int64_t);
210 }
211
213 int64_t MemUsage() const { return MemUsage(num_bits_); }
214
216 void Set(int64_t bit_index, bool v) {
217 int64_t word_index = bit_index >> NUM_OFFSET_BITS;
218 bit_index &= BIT_INDEX_MASK;
219 DCHECK_LT(word_index, buffer_.size());
220 if (v) {
221 buffer_[word_index] |= (1LL << bit_index);
222 } else {
223 buffer_[word_index] &= ~(1LL << bit_index);
224 }
225 }
226
228 bool Get(int64_t bit_index) const {
229 int64_t word_index = bit_index >> NUM_OFFSET_BITS;
230 bit_index &= BIT_INDEX_MASK;
231 DCHECK_LT(word_index, buffer_.size());
232 return (buffer_[word_index] & (1LL << bit_index)) != 0;
233 }
234
235 void SetAllBits(bool b) { memset(&buffer_[0], 255 * b, buffer_.size() * sizeof(uint64_t)); }
236
237 int64_t num_bits() const { return num_bits_; }
238
240 std::string DebugString(bool print_bits) const;
241
242private:
243 std::vector<uint64_t> buffer_;
244 int64_t num_bits_;
245
247 static const int64_t NUM_OFFSET_BITS = 6;
248 static const int64_t BIT_INDEX_MASK = 63;
249};
