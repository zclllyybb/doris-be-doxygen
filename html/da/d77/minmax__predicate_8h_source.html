23namespace doris {
24// only used in Runtime Filter
25class MinMaxFuncBase {
26public:
27 virtual void insert(const void* data) = 0;
28 virtual void insert_fixed_len(const char* data, const int* offsets, int number) = 0;
29 virtual bool find(void* data) = 0;
30 virtual bool is_empty() = 0;
31 virtual void* get_max() = 0;
32 virtual void* get_min() = 0;
33 // assign minmax data
34 virtual Status assign(void* min_data, void* max_data) = 0;
35 // merge from other minmax_func
36 virtual Status merge(MinMaxFuncBase* minmax_func, ObjectPool* pool) = 0;
37 virtual ~MinMaxFuncBase() = default;
38};
39
40template <class T>
41class MinMaxNumFunc : public MinMaxFuncBase {
42public:
43 MinMaxNumFunc() = default;
44 ~MinMaxNumFunc() = default;
45
46 void insert(const void* data) override {
47 if (data == nullptr) {
48 return;
49 }
50
51 T val_data;
52 if constexpr (std::is_same_v<T, DateTimeValue>) {
53 reinterpret_cast<const vectorized::VecDateTimeValue*>(data)->convert_vec_dt_to_dt(
54 &val_data);
55 } else if constexpr (sizeof(T) >= sizeof(int128_t)) {
56 // use dereference operator on unalign address maybe lead segmentation fault
57 memcpy(&val_data, data, sizeof(T));
58 } else {
59 val_data = *reinterpret_cast<const T*>(data);
60 }
61
62 if (_empty) {
63 _min = val_data;
64 _max = val_data;
65 _empty = false;
66 return;
67 }
68 if (val_data < _min) {
69 _min = val_data;
70 } else if (val_data > _max) {
71 _max = val_data;
72 }
73 }
74
75 void insert_fixed_len(const char* data, const int* offsets, int number) override {
76 if (!number) {
77 return;
78 }
79 if (_empty) {
80 _min = *((T*)data + offsets[0]);
81 _max = *((T*)data + offsets[0]);
82 }
83 for (int i = _empty; i < number; i++) {
84 _min = std::min(_min, *((T*)data + offsets[i]));
85 _max = std::max(_max, *((T*)data + offsets[i]));
86 }
87 _empty = false;
88 }
89
90 bool find(void* data) override {
91 if (data == nullptr) {
92 return false;
93 }
94
95 T val_data = *reinterpret_cast<T*>(data);
96 return val_data >= _min && val_data <= _max;
97 }
98
99 Status merge(MinMaxFuncBase* minmax_func, ObjectPool* pool) override {
100 if constexpr (std::is_same_v<T, StringRef>) {
101 MinMaxNumFunc<T>* other_minmax = static_cast<MinMaxNumFunc<T>*>(minmax_func);
102
103 if (other_minmax->_min < _min) {
104 auto& other_min = other_minmax->_min;
105 auto str = pool->add(new std::string(other_min.data, other_min.size));
106 _min.data = str->data();
107 _min.size = str->length();
108 }
109 if (other_minmax->_max > _max) {
110 auto& other_max = other_minmax->_max;
111 auto str = pool->add(new std::string(other_max.data, other_max.size));
112 _max.data = str->data();
113 _max.size = str->length();
114 }
115 } else {
116 MinMaxNumFunc<T>* other_minmax = static_cast<MinMaxNumFunc<T>*>(minmax_func);
117 if (other_minmax->_min < _min) {
118 _min = other_minmax->_min;
119 }
120 if (other_minmax->_max > _max) {
121 _max = other_minmax->_max;
122 }
123 }
124
125 return Status::OK();
126 }
127
128 bool is_empty() override { return _empty; }
129
130 void* get_max() override { return &_max; }
131
132 void* get_min() override { return &_min; }
133
134 Status assign(void* min_data, void* max_data) override {
135 _min = *(T*)min_data;
136 _max = *(T*)max_data;
137 return Status::OK();
138 }
139
140private:
141 T _max = type_limit<T>::min();
142 T _min = type_limit<T>::max();
143 // we use _empty to avoid compare twice
144 bool _empty = true;
145};
