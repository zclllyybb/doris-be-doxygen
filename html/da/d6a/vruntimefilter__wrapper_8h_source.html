46 // if filter rate less than this, bloom filter will set always true
47 constexpr static double EXPECTED_FILTER_RATE = 0.4;
48
49 static void calculate_filter(int64_t filter_rows, int64_t scan_rows, bool& has_calculate,
50 bool& always_true) {
51 if ((!has_calculate) && (scan_rows > config::bloom_filter_predicate_check_row_num)) {
52 if (filter_rows / (scan_rows * 1.0) <
53 vectorized::VRuntimeFilterWrapper::EXPECTED_FILTER_RATE) {
54 always_true = true;
55 }
56 has_calculate = true;
57 }
58 }
