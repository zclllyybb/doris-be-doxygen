55// Case 1. Is pair of floats or pair of ints or pair of uints
56template <typename A, typename B>
57constexpr bool is_safe_conversion =
58 (std::is_floating_point_v<A> && std::is_floating_point_v<B>) ||
59 (std::is_integral_v<A> && std::is_integral_v<B> &&
60 !(std::is_signed_v<A> ^ std::is_signed_v<B>)) ||
61 (std::is_same_v<A, doris::vectorized::Int128> &&
62 std::is_same_v<B, doris::vectorized::Int128>) ||
63 (std::is_integral_v<A> && std::is_same_v<B, doris::vectorized::Int128>) ||
64 (std::is_same_v<A, doris::vectorized::Int128> && std::is_integral_v<B>);
65template <typename A, typename B>
66using bool_if_safe_conversion = std::enable_if_t<is_safe_conversion<A, B>, bool>;
67template <typename A, typename B>
68using bool_if_not_safe_conversion = std::enable_if_t<!is_safe_conversion<A, B>, bool>;

75// Case 2a. Are params IntXX and UIntYY and sizeof(IntXX) >= sizeof(UIntYY) (in such case will use accurate compare)
76template <typename TInt, typename TUInt>
77constexpr bool is_le_int_vs_uint =
78 is_any_int_vs_uint<TInt, TUInt> && (sizeof(TInt) <= sizeof(TUInt));

80template <typename TInt, typename TUInt>
81using bool_if_le_int_vs_uint_t = std::enable_if_t<is_le_int_vs_uint<TInt, TUInt>, bool>;

83template <typename TInt, typename TUInt>
84inline bool_if_le_int_vs_uint_t<TInt, TUInt> greaterOpTmpl(TInt a, TUInt b) {
85 return static_cast<TUInt>(a) > b && a >= 0 &&
86 b <= static_cast<TUInt>(std::numeric_limits<TInt>::max());
87}

91 return a > static_cast<TUInt>(b) || b < 0 ||
92 a > static_cast<TUInt>(std::numeric_limits<TInt>::max());

174// Case 3b. 64-bit integers vs floats comparison.
175// See hint at https://github.com/JuliaLang/julia/issues/257 (but it doesn't work properly for -2**63)
176
177constexpr doris::vectorized::Int64 MAX_INT64_WITH_EXACT_FLOAT64_REPR = 9007199254740992LL; // 2^53
178
179template <>
180inline bool greaterOp<doris::vectorized::Float64, doris::vectorized::Int64>(
181 doris::vectorized::Float64 f, doris::vectorized::Int64 i) {
182 if (-MAX_INT64_WITH_EXACT_FLOAT64_REPR <= i && i <= MAX_INT64_WITH_EXACT_FLOAT64_REPR) {
183 return f > static_cast<doris::vectorized::Float64>(i);
184 }
185
186 return (f >= static_cast<doris::vectorized::Float64>(
187 std::numeric_limits<
188 doris::vectorized::Int64>::max())) // rhs is 2**63 (not 2^63 - 1)
189 || (f > static_cast<doris::vectorized::Float64>(
190 std::numeric_limits<doris::vectorized::Int64>::min()) &&
191 static_cast<doris::vectorized::Int64>(f) > i);
192}
193
194template <>
195inline bool greaterOp<doris::vectorized::Int64, doris::vectorized::Float64>(
196 doris::vectorized::Int64 i, doris::vectorized::Float64 f) {
197 if (-MAX_INT64_WITH_EXACT_FLOAT64_REPR <= i && i <= MAX_INT64_WITH_EXACT_FLOAT64_REPR) {
198 return f < static_cast<doris::vectorized::Float64>(i);
199 }
200
201 return (f < static_cast<doris::vectorized::Float64>(
202 std::numeric_limits<doris::vectorized::Int64>::min())) ||
203 (f < static_cast<doris::vectorized::Float64>(
204 std::numeric_limits<doris::vectorized::Int64>::max()) &&
205 i > static_cast<doris::vectorized::Int64>(f));
206}

208template <>
209inline bool greaterOp<doris::vectorized::Float64, doris::vectorized::UInt64>(
210 doris::vectorized::Float64 f, doris::vectorized::UInt64 u) {
211 if (u <= static_cast<doris::vectorized::UInt64>(MAX_INT64_WITH_EXACT_FLOAT64_REPR)) {
212 return f > static_cast<doris::vectorized::Float64>(u);
213 }
214
215 return (f >= static_cast<doris::vectorized::Float64>(
216 std::numeric_limits<doris::vectorized::UInt64>::max())) ||
217 (f >= 0 && static_cast<doris::vectorized::UInt64>(f) > u);
218}

220template <>
221inline bool greaterOp<doris::vectorized::UInt64, doris::vectorized::Float64>(
222 doris::vectorized::UInt64 u, doris::vectorized::Float64 f) {
223 if (u <= static_cast<doris::vectorized::UInt64>(MAX_INT64_WITH_EXACT_FLOAT64_REPR)) {
224 return static_cast<doris::vectorized::Float64>(u) > f;
225 }
226
227 return (f < 0) || (f < static_cast<doris::vectorized::Float64>(
228 std::numeric_limits<doris::vectorized::UInt64>::max()) &&
229 u > static_cast<doris::vectorized::UInt64>(f));

372inline bool greaterOp(doris::vectorized::Int128 i, doris::vectorized::Float64 f) {
373 static constexpr __int128 min_int128 = __int128(0x8000000000000000ll) << 64;
374 static constexpr __int128 max_int128 =
375 (__int128(0x7fffffffffffffffll) << 64) + 0xffffffffffffffffll;
376
377 if (-MAX_INT64_WITH_EXACT_FLOAT64_REPR <= i && i <= MAX_INT64_WITH_EXACT_FLOAT64_REPR) {
378 return static_cast<doris::vectorized::Float64>(i) > f;
379 }
380
381 return (f < static_cast<doris::vectorized::Float64>(min_int128)) ||
382 (f < static_cast<doris::vectorized::Float64>(max_int128) &&
383 i > static_cast<doris::vectorized::Int128>(f));
384}

386inline bool greaterOp(doris::vectorized::Float64 f, doris::vectorized::Int128 i) {
387 static constexpr __int128 min_int128 = __int128(0x8000000000000000ll) << 64;
388 static constexpr __int128 max_int128 =
389 (__int128(0x7fffffffffffffffll) << 64) + 0xffffffffffffffffll;
390
391 if (-MAX_INT64_WITH_EXACT_FLOAT64_REPR <= i && i <= MAX_INT64_WITH_EXACT_FLOAT64_REPR) {
392 return f > static_cast<doris::vectorized::Float64>(i);
393 }
394
395 return (f >= static_cast<doris::vectorized::Float64>(max_int128)) ||
396 (f > static_cast<doris::vectorized::Float64>(min_int128) &&
397 static_cast<doris::vectorized::Int128>(f) > i);
398}

470template <typename From, typename To, bool strict = true>
471inline bool convertNumeric(From value, To& result) {
473 if constexpr (std::is_same_v<From, To>) {
474 result = value;
475 return true;
476 }
477 if constexpr (std::is_floating_point_v<From> && std::is_floating_point_v<To>) {
479 if (is_nan(value)) {
480 result = value;
481 return true;
482 }
483 if (value == std::numeric_limits<From>::infinity()) {
484 result = std::numeric_limits<To>::infinity();
485 return true;
486 }
487 if (value == -std::numeric_limits<From>::infinity()) {
488 result = -std::numeric_limits<To>::infinity();
489 return true;
490 }
491 }
492 if (greaterOp(value, std::numeric_limits<To>::max()) ||
493 lessOp(value, std::numeric_limits<To>::lowest())) {
494 return false;
495 }
496 result = static_cast<To>(value);
497 if constexpr (strict) {
498 return equalsOp(value, result);
499 }
500 return true;
501}
