62public:
63 AggregateRpcUdafData() = default;
64 AggregateRpcUdafData(int64_t num_args) { set_last_result(false); }

66 bool has_last_result() { return _saved_last_result == true; }

68 void set_last_result(bool flag) { _saved_last_result = flag; }

76 Status merge(AggregateRpcUdafData& rhs) {
77 send_buffer_to_rpc_server();
78 if (has_last_result()) {
79 PFunctionCallRequest request;
80 PFunctionCallResponse response;
81 brpc::Controller cntl;
82 PFunctionCallResponse current_res = rhs.get_result();
83 request.set_function_name(_merge_fn);
84 //last result
85 PValues* arg = request.add_args();
86 arg->CopyFrom(_res.result(0));
87 arg = request.add_args();
88 //current result
89 arg->CopyFrom(current_res.result(0));
90 //send to rpc server that impl the merge op, the will save the result
91 RETURN_IF_ERROR(send_rpc_request(cntl, request, response));
92 _res = response;
93 } else {
94 _res = rhs.get_result();
95 set_last_result(true);
96 }
97 return Status::OK();
98 }

100 Status init(const TFunction& fn) {
101 _update_fn = fn.aggregate_fn.update_fn_symbol;
102 _merge_fn = fn.aggregate_fn.merge_fn_symbol;
103 _server_addr = fn.hdfs_location;
104 _finalize_fn = fn.aggregate_fn.finalize_fn_symbol;
105 set_error(false);
106 _client = ExecEnv::GetInstance()->brpc_function_client_cache()->get_client(_server_addr);
107 if (_client == nullptr) {
108 std::string err_msg = "init rpc error, addr:" + _server_addr;
109 LOG(ERROR) << err_msg;
110 set_error(true);
111 return Status::InternalError(err_msg);
112 }
113 return Status::OK();
114 }

116 Status send_rpc_request(brpc::Controller& cntl, PFunctionCallRequest& request,
117 PFunctionCallResponse& response) {
118 _client->fn_call(&cntl, &request, &response, nullptr);
119 if (cntl.Failed()) {
120 set_error(true);
121 std::stringstream err_msg;
122 err_msg << " call rpc function failed";
123 err_msg << " _server_addr:" << _server_addr;
124 err_msg << " function_name:" << request.function_name();
125 err_msg << " err:" << cntl.ErrorText();
126 LOG(ERROR) << err_msg.str();
127 return Status::InternalError(err_msg.str());
128 }
129 if (!response.has_status() || response.result_size() == 0) {
130 set_error(true);
131 std::stringstream err_msg;
132 err_msg << " call rpc function failed, status or result is not set";
133 err_msg << " _server_addr:" << _server_addr;
134 err_msg << " function_name:" << request.function_name();
135 LOG(ERROR) << err_msg.str();
136 return Status::InternalError(err_msg.str());
137 }
138 if (response.status().status_code() != 0) {
139 set_error(true);
140 std::stringstream err_msg;
141 err_msg << " call rpc function failed";
142 err_msg << " _server_addr:" << _server_addr;
143 err_msg << " function_name:" << request.function_name();
144 err_msg << " err:" << response.status().DebugString();
145 LOG(ERROR) << err_msg.str();
146 return Status::InternalError(err_msg.str());
147 }
148 return Status::OK();
149 }

151 Status gen_request_data(PFunctionCallRequest& request, const IColumn** columns, int start,
152 int end, const DataTypes& argument_types) {
153 for (int i = 0; i < argument_types.size(); i++) {
154 PValues* arg = request.add_args();
155 if (auto* nullable = vectorized::check_and_get_column<const vectorized::ColumnNullable>(
156 *columns[i])) {
157 auto data_col = nullable->get_nested_column_ptr();
158 auto& null_col = nullable->get_null_map_column();
159 auto data_type = std::reinterpret_pointer_cast<const vectorized::DataTypeNullable>(
160 argument_types[i]);
161 data_col->get_data_at(0);
162 convert_nullable_col_to_pvalue(data_col->convert_to_full_column_if_const(),
163 data_type->get_nested_type(), null_col, arg, start,
164 end);
165
166 } else {
167 convert_col_to_pvalue<false>(columns[i]->convert_to_full_column_if_const(),
168 argument_types[i], arg, start, end);
169 }
170 }
171 return Status::OK();
172 }
173
174#define ADD_VALUE(TYPEVALUE) \
175 if (_buffer_request[j].args(i).TYPEVALUE##_##size() > 0) { \
176 arg->add_##TYPEVALUE(_buffer_request[j].args(i).TYPEVALUE(0)); \
177 }

179 PFunctionCallRequest merge_buffer_request(PFunctionCallRequest& request) {
180 int args_size = _buffer_request[0].args_size();
181 request.set_function_name(_update_fn);
182 for (int i = 0; i < args_size; i++) {
183 PValues* arg = request.add_args();
184 arg->mutable_type()->CopyFrom(_buffer_request[0].args(i).type());
185 for (int j = 0; j < _buffer_request.size(); j++) {
186 ADD_VALUE(double_value);
187 ADD_VALUE(float_value);
188 ADD_VALUE(int32_value);
189 ADD_VALUE(int64_value);
190 ADD_VALUE(uint32_value);
191 ADD_VALUE(uint64_value);
192 ADD_VALUE(bool_value);
193 ADD_VALUE(string_value);
194 ADD_VALUE(bytes_value);
195 }
196 }
197 return request;
198 }

200 // called in group agg op
201 Status buffer_add(const IColumn** columns, int start, int end,
202 const DataTypes& argument_types) {
203 PFunctionCallRequest request;
204 gen_request_data(request, columns, start, end, argument_types);
205 _buffer_request.push_back(request);
206 if (_buffer_request.size() >= max_buffered_rows) {
207 send_buffer_to_rpc_server();
208 }
209 return Status::OK();
210 }
211
212 //clear buffer request
213 Status send_buffer_to_rpc_server() {
214 if (_buffer_request.size() > 0) {
215 PFunctionCallRequest request;
216 PFunctionCallResponse response;
217 brpc::Controller cntl;
218 merge_buffer_request(request);
219 if (has_last_result()) {
220 request.mutable_context()
221 ->mutable_function_context()
222 ->mutable_args_data()
223 ->CopyFrom(_res.result());
224 }
225 RETURN_IF_ERROR(send_rpc_request(cntl, request, response));
226 _res = response;
227 set_last_result(true);
228 _buffer_request.clear();
229 }
230 return Status::OK();
231 }

233 Status add(const IColumn** columns, int start, int end, const DataTypes& argument_types) {
234 PFunctionCallRequest request;
235 PFunctionCallResponse response;
236 brpc::Controller cntl;
237 request.set_function_name(_update_fn);
238 gen_request_data(request, columns, start, end, argument_types);
239 if (has_last_result()) {
240 request.mutable_context()->mutable_function_context()->mutable_args_data()->CopyFrom(
241 _res.result());
242 }
243 RETURN_IF_ERROR(send_rpc_request(cntl, request, response));
244 _res = response;
245 set_last_result(true);
246 return Status::OK();
247 }

249 void serialize(BufferWritable& buf) {
250 send_buffer_to_rpc_server();
251 std::string serialize_data = error_default_str;
252 if (!has_error()) {
253 serialize_data = _res.SerializeAsString();
254 } else {
255 LOG(ERROR) << "serialize empty buf";
256 }
257 write_binary(serialize_data, buf);
258 }
259
260 void deserialize(BufferReadable& buf) {
261 send_buffer_to_rpc_server();
262 std::string serialize_data;
263 read_binary(serialize_data, buf);
264 if (error_default_str != serialize_data) {
265 _res.ParseFromString(serialize_data);
266 set_last_result(true);
267 } else {
268 LOG(ERROR) << "deserialize empty buf";
269 set_error(true);
270 }
271 }
272
273#define GETDATA(LOCATTYPE, TYPEVALUE) \
274 if (response.result_size() > 0 && response.result(0).TYPEVALUE##_##value_size() > 0) { \
275 LOCATTYPE ret = response.result(0).TYPEVALUE##_##value(0); \
276 to.insert_data((char*)&ret, 0); \
277 } else { \
278 LOG(ERROR) << "_server_addr:" << _server_addr << ",_finalize_fn:" << _finalize_fn \
279 << ",msg: failed to get final result cause return type need " << #TYPEVALUE \
280 << "but result is empty"; \
281 to.insert_default(); \
282 }

284 //if any unexpected error happen will return NULL
285 Status get(IColumn& to, const DataTypePtr& return_type) {
286 if (has_error()) {
287 to.insert_default();
288 return Status::OK();
289 }
290 send_buffer_to_rpc_server();
291 PFunctionCallRequest request;
292 PFunctionCallResponse response;
293 brpc::Controller cntl;
294 request.set_function_name(_finalize_fn);
295 request.mutable_context()->mutable_function_context()->mutable_args_data()->CopyFrom(
296 _res.result());
297 send_rpc_request(cntl, request, response);
298 if (has_error()) {
299 to.insert_default();
300 return Status::OK();
301 }
302 DataTypePtr result_type = return_type;
303 if (return_type->is_nullable()) {
304 result_type =
305 reinterpret_cast<const DataTypeNullable*>(return_type.get())->get_nested_type();
306 }
307 WhichDataType which(result_type);
308 if (which.is_float32()) {
309 GETDATA(float, float);
310 } else if (which.is_float64()) {
311 GETDATA(double, double);
312 } else if (which.is_int32()) {
313 GETDATA(int32_t, int32);
314 } else if (which.is_uint32()) {
315 GETDATA(uint32_t, uint32);
316 } else if (which.is_int64()) {
317 GETDATA(int64_t, int64);
318 } else if (which.is_uint64()) {
319 GETDATA(uint64_t, uint64);
320 } else if (which.is_uint8()) {
321 GETDATA(uint8_t, bool);
322 } else if (which.is_string()) {
323 if (response.result_size() > 0 && response.result(0).string_value_size() > 0) {
324 std::string ret = response.result(0).string_value(0);
325 to.insert_data(ret.c_str(), ret.size());
326 } else {
327 LOG(ERROR) << "_server_addr:" << _server_addr << ",_finalize_fn:" << _finalize_fn
328 << ",msg: failed to get final result cause return type need string but "
329 "result is empty";
330 to.insert_default();
331 }
332 } else {
333 LOG(ERROR) << "failed to get result cause unkown return type";
334 to.insert_default();
335 }
336 return Status::OK();
337 }

339 PFunctionCallResponse get_result() {
340 send_buffer_to_rpc_server();
341 return _res;
342 }

394 }

397 this->data(const_cast<AggregateDataPtr>(place)).get(to, _return_type);
