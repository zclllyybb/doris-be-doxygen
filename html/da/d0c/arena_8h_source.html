46class Arena : private boost::noncopyable {
47private:
49 static constexpr size_t pad_right = 15;
50
52 struct alignas(16) Chunk : private Allocator<false>
53 {
54 char* begin;
55 char* pos;
56 char* end;
57
58 Chunk* prev;
59
60 Chunk(size_t size_, Chunk* prev_) {
61 begin = reinterpret_cast<char*>(Allocator<false>::alloc(size_));
62 pos = begin;
63 end = begin + size_ - pad_right;
64 prev = prev_;
65
66 ASAN_POISON_MEMORY_REGION(begin, size_);
67 }
68
69 ~Chunk() {
74 ASAN_UNPOISON_MEMORY_REGION(begin, size());
75
76 Allocator<false>::free(begin, size());
77
78 if (prev) delete prev;
79 }
80
81 size_t size() const { return end + pad_right - begin; }
82 size_t remaining() const { return end - pos; }
83 };
84
85 size_t growth_factor;
86 size_t linear_growth_threshold;
87
89 Chunk* head;
90 size_t size_in_bytes;
91
92 static size_t round_up_to_page_size(size_t s) { return (s + 4096 - 1) / 4096 * 4096; }
93
96 size_t next_size(size_t min_next_size) const {
97 size_t size_after_grow = 0;
98
99 if (head->size() < linear_growth_threshold) {
100 size_after_grow = std::max(min_next_size, head->size() * growth_factor);
101 } else {
102 // alloc_continue() combined with linear growth results in quadratic
103 // behavior: we append the data by small amounts, and when it
104 // doesn't fit, we create a new chunk and copy all the previous data
105 // into it. The number of times we do this is directly proportional
106 // to the total size of data that is going to be serialized. To make
107 // the copying happen less often, round the next size up to the
108 // linear_growth_threshold.
109 size_after_grow =
110 ((min_next_size + linear_growth_threshold - 1) / linear_growth_threshold) *
111 linear_growth_threshold;
112 }
113
114 assert(size_after_grow >= min_next_size);
115 return round_up_to_page_size(size_after_grow);
116 }
117
119 void NO_INLINE add_chunk(size_t min_size) {
120 head = new Chunk(next_size(min_size + pad_right), head);
121 size_in_bytes += head->size();
122 }
123
124 friend class ArenaAllocator;
125 template <size_t>
126 friend class AlignedArenaAllocator;
127
128public:
129 Arena(size_t initial_size_ = 4096, size_t growth_factor_ = 2,
130 size_t linear_growth_threshold_ = 128 * 1024 * 1024)
131 : growth_factor(growth_factor_),
132 linear_growth_threshold(linear_growth_threshold_),
133 head(new Chunk(initial_size_, nullptr)),
134 size_in_bytes(head->size()) {}
135
136 ~Arena() { delete head; }
137
139 char* alloc(size_t size) {
140 if (UNLIKELY(head->pos + size > head->end)) add_chunk(size);
141
142 char* res = head->pos;
143 head->pos += size;
144 ASAN_UNPOISON_MEMORY_REGION(res, size + pad_right);
145 return res;
146 }
147
149 char* aligned_alloc(size_t size, size_t alignment) {
150 do {
151 void* head_pos = head->pos;
152 size_t space = head->end - head->pos;
153
154 auto res = static_cast<char*>(std::align(alignment, size, head_pos, space));
155 if (res) {
156 head->pos = static_cast<char*>(head_pos);
157 head->pos += size;
158 ASAN_UNPOISON_MEMORY_REGION(res, size + pad_right);
159 return res;
160 }
161
162 add_chunk(size + alignment);
163 } while (true);
164 }
165
166 template <typename T>
167 T* alloc() {
168 return reinterpret_cast<T*>(aligned_alloc(sizeof(T), alignof(T)));
169 }
170
176 void* rollback(size_t size) {
177 head->pos -= size;
178 ASAN_POISON_MEMORY_REGION(head->pos, size + pad_right);
179 return head->pos;
180 }
181
194 char* alloc_continue(size_t additional_bytes, char const*& range_start,
195 size_t start_alignment = 0) {
196 if (!range_start) {
197 // Start a new memory range.
198 char* result = start_alignment ? aligned_alloc(additional_bytes, start_alignment)
199 : alloc(additional_bytes);
200
201 range_start = result;
202 return result;
203 }
204
205 // Extend an existing memory range with 'additional_bytes'.
206
207 // This method only works for extending the last allocation. For lack of
208 // original size, check a weaker condition: that 'begin' is at least in
209 // the current Chunk.
210 assert(range_start >= head->begin && range_start < head->end);
211
212 if (head->pos + additional_bytes <= head->end) {
213 // The new size fits into the last chunk, so just alloc the
214 // additional size. We can alloc without alignment here, because it
215 // only applies to the start of the range, and we don't change it.
216 return alloc(additional_bytes);
217 }
218
219 // New range doesn't fit into this chunk, will copy to a new one.
220 //
221 // Note: among other things, this method is used to provide a hack-ish
222 // implementation of realloc over Arenas in ArenaAllocators. It wastes a
223 // lot of memory -- quadratically so when we reach the linear allocation
224 // threshold. This deficiency is intentionally left as is, and should be
225 // solved not by complicating this method, but by rethinking the
226 // approach to memory management for aggregate function states, so that
227 // we can provide a proper realloc().
228 const size_t existing_bytes = head->pos - range_start;
229 const size_t new_bytes = existing_bytes + additional_bytes;
230 const char* old_range = range_start;
231
232 char* new_range =
233 start_alignment ? aligned_alloc(new_bytes, start_alignment) : alloc(new_bytes);
234
235 memcpy(new_range, old_range, existing_bytes);
236
237 range_start = new_range;
238 return new_range + existing_bytes;
239 }
240
242 char* realloc(const char* old_data, size_t old_size, size_t new_size) {
243 char* res = alloc(new_size);
244 if (old_data) {
245 memcpy(res, old_data, old_size);
246 ASAN_POISON_MEMORY_REGION(old_data, old_size);
247 }
248 return res;
249 }
250
251 char* aligned_realloc(const char* old_data, size_t old_size, size_t new_size,
252 size_t alignment) {
253 char* res = aligned_alloc(new_size, alignment);
254 if (old_data) {
255 memcpy(res, old_data, old_size);
256 ASAN_POISON_MEMORY_REGION(old_data, old_size);
257 }
258 return res;
259 }
260
262 const char* insert(const char* data, size_t size) {
263 char* res = alloc(size);
264 memcpy(res, data, size);
265 return res;
266 }
267
268 const char* aligned_insert(const char* data, size_t size, size_t alignment) {
269 char* res = aligned_alloc(size, alignment);
270 memcpy(res, data, size);
271 return res;
272 }
273
275 size_t size() const { return size_in_bytes; }
276
277 size_t remaining_space_in_current_chunk() const { return head->remaining(); }
278};
