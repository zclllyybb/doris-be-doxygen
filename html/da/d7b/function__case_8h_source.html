55struct CaseWhenColumnHolder {
56 using OptionalPtr = std::optional<ColumnPtr>;
57
58 std::vector<OptionalPtr> when_ptrs; // case, when, when...
59 std::vector<OptionalPtr> then_ptrs; // else, then, then...
60 size_t pair_count;
61 size_t rows_count;
62
63 CaseWhenColumnHolder(Block& block, const ColumnNumbers& arguments, size_t input_rows_count,
64 bool has_case, bool has_else, bool when_null, bool then_null)
65 : rows_count(input_rows_count) {
66 when_ptrs.emplace_back(has_case ? OptionalPtr(block.get_by_position(arguments[0]).column)
67 : std::nullopt);
68 then_ptrs.emplace_back(
69 has_else
70 ? OptionalPtr(block.get_by_position(arguments[arguments.size() - 1]).column)
71 : std::nullopt);
72
73 int begin = 0 + has_case;
74 int end = arguments.size() - has_else;
75 pair_count = (end - begin) / 2 + 1; // when/then at [1: pair_count)
76
77 for (int i = begin; i < end; i += 2) {
78 when_ptrs.emplace_back(OptionalPtr(block.get_by_position(arguments[i]).column));
79 then_ptrs.emplace_back(OptionalPtr(block.get_by_position(arguments[i + 1]).column));
80 }
81
82 // if case_column/when_column is nullable. cast all case_column/when_column to nullable.
83 if (when_null) {
84 for (OptionalPtr& column_ptr : when_ptrs) {
85 cast_to_nullable(column_ptr);
86 }
87 }
88
89 // if else_column/then_column is nullable. cast all else_column/then_column to nullable.
90 if (then_null) {
91 for (OptionalPtr& column_ptr : then_ptrs) {
92 cast_to_nullable(column_ptr);
93 }
94 }
95 }
96
97 void cast_to_nullable(OptionalPtr& column_ptr) {
98 if (!column_ptr.has_value() || column_ptr.value()->is_nullable()) {
99 return;
100 }
101 column_ptr.emplace(make_nullable(column_ptr.value()));
102 }
103};

105template <bool has_case, bool has_else>
106class FunctionCase : public IFunction {
107public:
108 static constexpr auto name = FunctionCaseName<has_case, has_else>::name;
109 static FunctionPtr create() { return std::make_shared<FunctionCase>(); }
110 String get_name() const override { return name; }
111 size_t get_number_of_arguments() const override { return 0; }
112 bool is_variadic() const override { return true; }
113
114 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
115 int loop_start = has_case ? 2 : 1;
116 int loop_end = has_else ? arguments.size() - 1 : arguments.size();
117
118 bool is_nullable = false;
119 if (!has_else || arguments[loop_end].get()->is_nullable()) {
120 is_nullable = true;
121 }
122 for (int i = loop_start; !is_nullable && i < loop_end; i += 2) {
123 if (arguments[i].get()->is_nullable()) {
124 is_nullable = true;
125 }
126 }
127
128 if (is_nullable) {
129 return make_nullable(arguments[loop_start]);
130 } else {
131 return arguments[loop_start];
132 }
133 }

135 bool use_default_implementation_for_nulls() const override { return false; }

137 template <typename ColumnType, bool when_null, bool then_null>
138 Status execute_short_circuit(const DataTypePtr& data_type, Block& block, size_t result,
139 CaseWhenColumnHolder column_holder) {
140 auto case_column_ptr = column_holder.when_ptrs[0].value_or(nullptr);
141 int rows_count = column_holder.rows_count;
142
143 // `then` data index corresponding to each row of results, 0 represents `else`.
144 int then_idx[rows_count];
145 int* __restrict then_idx_ptr = then_idx;
146 memset(then_idx_ptr, 0, sizeof(then_idx));
147
148 for (int row_idx = 0; row_idx < column_holder.rows_count; row_idx++) {
149 for (int i = 1; i < column_holder.pair_count; i++) {
150 auto when_column_ptr = column_holder.when_ptrs[i].value();
151 if constexpr (has_case) {
152 if (!case_column_ptr->is_null_at(row_idx) &&
153 case_column_ptr->compare_at(row_idx, row_idx, *when_column_ptr, -1) == 0) {
154 then_idx_ptr[row_idx] = i;
155 break;
156 }
157 } else {
158 if constexpr (when_null) {
159 if (!then_idx_ptr[row_idx] && when_column_ptr->get_bool(row_idx)) {
160 then_idx_ptr[row_idx] = i;
161 break;
162 }
163 } else {
164 if (!then_idx_ptr[row_idx]) {
165 then_idx_ptr[row_idx] = i;
166 break;
167 }
168 }
169 }
170 }
171 }
172
173 auto result_column_ptr = data_type->create_column();
174 update_result_normal(result_column_ptr, then_idx, column_holder);
175 block.replace_by_position(result, std::move(result_column_ptr));
176 return Status::OK();
177 }

179 template <typename ColumnType, bool when_null, bool then_null>
180 Status execute_impl(const DataTypePtr& data_type, Block& block, size_t result,
181 CaseWhenColumnHolder column_holder) {
182 if (column_holder.pair_count > UINT8_MAX) {
183 return execute_short_circuit<ColumnType, when_null, then_null>(data_type, block, result,
184 column_holder);
185 }
186
187 int rows_count = column_holder.rows_count;
188
189 // `then` data index corresponding to each row of results, 0 represents `else`.
190 uint8_t then_idx[rows_count];
191 uint8_t* __restrict then_idx_ptr = then_idx;
192 memset(then_idx_ptr, 0, sizeof(then_idx));
193
194 auto case_column_ptr = column_holder.when_ptrs[0].value_or(nullptr);
195
196 for (uint8_t i = 1; i < column_holder.pair_count; i++) {
197 auto when_column_ptr = column_holder.when_ptrs[i].value();
198 if constexpr (has_case) {
199 // TODO: need simd
200 for (int row_idx = 0; row_idx < rows_count; row_idx++) {
201 if (!then_idx_ptr[row_idx] && !case_column_ptr->is_null_at(row_idx) &&
202 case_column_ptr->compare_at(row_idx, row_idx, *when_column_ptr, -1) == 0) {
203 then_idx_ptr[row_idx] = i;
204 }
205 }
206 } else {
207 if constexpr (when_null) {
208 // TODO: need simd
209 for (int row_idx = 0; row_idx < rows_count; row_idx++) {
210 if (!then_idx_ptr[row_idx] && when_column_ptr->get_bool(row_idx)) {
211 then_idx_ptr[row_idx] = i;
212 }
213 }
214 } else {
215 auto* __restrict cond_raw_data =
216 reinterpret_cast<const ColumnUInt8*>(when_column_ptr.get())
217 ->get_data()
218 .data();
219
220 // simd automatically
221 for (int row_idx = 0; row_idx < rows_count; row_idx++) {
222 then_idx_ptr[row_idx] |=
223 (!then_idx_ptr[row_idx]) * cond_raw_data[row_idx] * i;
224 }
225 }
226 }
227 }
228
229 return execute_update_result<ColumnType, then_null>(data_type, result, block, then_idx,
230 column_holder);
231 }

233 template <typename ColumnType, bool then_null>
234 Status execute_update_result(const DataTypePtr& data_type, size_t result, Block& block,
235 uint8* then_idx, CaseWhenColumnHolder& column_holder) {
236 auto result_column_ptr = data_type->create_column();
237
238 if constexpr (std::is_same_v<ColumnType, ColumnString> ||
239 std::is_same_v<ColumnType, ColumnBitmap> ||
240 std::is_same_v<ColumnType, ColumnArray> ||
241 std::is_same_v<ColumnType, ColumnMap> ||
242 std::is_same_v<ColumnType, ColumnStruct> ||
243 std::is_same_v<ColumnType, ColumnHLL>) {
244 // result_column and all then_column is not nullable.
245 // can't simd when type is string.
246 update_result_normal(result_column_ptr, then_idx, column_holder);
247 } else if constexpr (then_null) {
248 // result_column and all then_column is nullable.
249 // TODO: make here simd automatically.
250 update_result_normal(result_column_ptr, then_idx, column_holder);
251 } else {
252 update_result_auto_simd<ColumnType>(result_column_ptr, then_idx, column_holder);
253 }
254
255 block.replace_by_position(result, std::move(result_column_ptr));
256 return Status::OK();
257 }

259 template <typename IndexType>
260 void update_result_normal(MutableColumnPtr& result_column_ptr, IndexType* then_idx,
261 CaseWhenColumnHolder& column_holder) {
262 for (int row_idx = 0; row_idx < column_holder.rows_count; row_idx++) {
263 if constexpr (!has_else) {
264 if (!then_idx[row_idx]) {
265 result_column_ptr->insert_default();
266 continue;
267 }
268 }
269 result_column_ptr->insert_from(*column_holder.then_ptrs[then_idx[row_idx]].value(),
270 row_idx);
271 }
272 }

274 template <typename ColumnType>
275 void update_result_auto_simd(MutableColumnPtr& result_column_ptr,
276 const uint8* __restrict then_idx,
277 CaseWhenColumnHolder& column_holder) {
278 size_t rows_count = column_holder.rows_count;
279 result_column_ptr->resize(rows_count);
280 auto* __restrict result_raw_data =
281 reinterpret_cast<ColumnType*>(result_column_ptr.get())->get_data().data();
282
283 // set default value
284 for (int i = 0; i < rows_count; i++) {
285 result_raw_data[i] = 0;
286 }
287
288 // some types had simd automatically, but some not.
289 for (uint8_t i = (has_else ? 0 : 1); i < column_holder.pair_count; i++) {
290 auto* __restrict column_raw_data =
291 reinterpret_cast<ColumnType*>(
292 column_holder.then_ptrs[i].value()->assume_mutable().get())
293 ->get_data()
294 .data();
295
296 for (int row_idx = 0; row_idx < rows_count; row_idx++) {
297 result_raw_data[row_idx] += (then_idx[row_idx] == i) * column_raw_data[row_idx];
298 }
299 }

302 template <typename ColumnType, bool when_null>
303 Status execute_get_then_null(const DataTypePtr& data_type, Block& block,
304 const ColumnNumbers& arguments, size_t result,
305 size_t input_rows_count) {
306 bool then_null = false;
307 for (int i = 1 + has_case; i < arguments.size() - has_else; i += 2) {
308 auto then_column_ptr = block.get_by_position(arguments[i]).column;
309 if (then_column_ptr->is_nullable()) {
310 then_null = true;
311 }
312 }
313 if constexpr (has_else) {
314 auto else_column_ptr = block.get_by_position(arguments[arguments.size() - 1]).column;
315 if (else_column_ptr->is_nullable()) {
316 then_null = true;
317 }
318 } else {
319 then_null = true;
320 }
321
322 CaseWhenColumnHolder column_holder = CaseWhenColumnHolder(
323 block, arguments, input_rows_count, has_case, has_else, when_null, then_null);
324
325 if (then_null) {
326 return execute_impl<ColumnType, when_null, true>(data_type, block, result,
327 column_holder);
328 } else {
329 return execute_impl<ColumnType, when_null, false>(data_type, block, result,
330 column_holder);
331 }
332 }

334 template <typename ColumnType>
335 Status execute_get_when_null(const DataTypePtr& data_type, Block& block,
336 const ColumnNumbers& arguments, size_t result,
337 size_t input_rows_count) {
338 bool when_null = false;
339 if constexpr (has_case) {
340 auto case_column_ptr = block.get_by_position(arguments[0]).column;
341 if (case_column_ptr->is_nullable()) {
342 when_null = true;
343 }
344 }
345 for (int i = has_case; i < arguments.size() - has_else; i += 2) {
346 auto when_column_ptr = block.get_by_position(arguments[i]).column;
347 if (when_column_ptr->is_nullable()) {
348 when_null = true;
349 }
350 }
351
352 if (when_null) {
353 return execute_get_then_null<ColumnType, true>(data_type, block, arguments, result,
354 input_rows_count);
355 } else {
356 return execute_get_then_null<ColumnType, false>(data_type, block, arguments, result,
357 input_rows_count);
