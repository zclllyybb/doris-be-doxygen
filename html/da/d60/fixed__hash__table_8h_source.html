110class FixedHashTable : private boost::noncopyable,
111 protected Allocator,
112 protected Cell::State,
113 protected Size {
114 static constexpr size_t NUM_CELLS = 1ULL << (sizeof(Key) * 8);
115
116protected:

121 void alloc() { buf = reinterpret_cast<Cell*>(Allocator::alloc(NUM_CELLS * sizeof(Cell))); }
122
123 void free() {
124 if (buf) {

130 void destroy_elements() {
