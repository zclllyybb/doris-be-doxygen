91template <bool save_indices, bool WithEditDistance>
92Regexps constructRegexps(const std::vector<String>& str_patterns,
93 [[maybe_unused]] std::optional<UInt32> edit_distance) {
95 std::vector<const char*> patterns;
96 std::vector<unsigned int> flags;
97
99 std::vector<hs_expr_ext> ext_exprs;
100 std::vector<const hs_expr_ext*> ext_exprs_ptrs;
101
102 patterns.reserve(str_patterns.size());
103 flags.reserve(str_patterns.size());
104
105 if constexpr (WithEditDistance) {
106 ext_exprs.reserve(str_patterns.size());
107 ext_exprs_ptrs.reserve(str_patterns.size());
108 }
109
110 for (const auto& ref : str_patterns) {
111 patterns.push_back(ref.data());
112 /* Flags below are the pattern matching flags.
113 * HS_FLAG_DOTALL is a compile flag where matching a . will not exclude newlines. This is a good
114 * performance practice according to Hyperscan API. https://intel.github.io/hyperscan/dev-reference/performance.html#dot-all-mode
115 * HS_FLAG_ALLOWEMPTY is a compile flag where empty strings are allowed to match.
116 * HS_FLAG_UTF8 is a flag where UTF8 literals are matched.
117 * HS_FLAG_SINGLEMATCH is a compile flag where each pattern match will be returned only once. it is a good performance practice
118 * as it is said in the Hyperscan documentation. https://intel.github.io/hyperscan/dev-reference/performance.html#single-match-flag
119 */
120 flags.push_back(HS_FLAG_DOTALL | HS_FLAG_SINGLEMATCH | HS_FLAG_ALLOWEMPTY | HS_FLAG_UTF8);
121 if constexpr (WithEditDistance) {
123 flags.back() &= ~HS_FLAG_UTF8;
124 ext_exprs.emplace_back();
126 ext_exprs.back().flags = HS_EXT_FLAG_EDIT_DISTANCE;
127 ext_exprs.back().edit_distance = edit_distance.value();
128 ext_exprs_ptrs.push_back(&ext_exprs.back());
129 }
130 }
131 hs_database_t* db = nullptr;
132 hs_compile_error_t* compile_error;
133
134 std::unique_ptr<unsigned int[]> ids;
135
137 if constexpr (save_indices) {
138 ids.reset(new unsigned int[patterns.size()]);
139 for (size_t i = 0; i < patterns.size(); ++i) ids[i] = static_cast<unsigned>(i + 1);
140 }
141
142 for (auto& pattern : patterns) {
143 LOG(INFO) << "pattern: " << pattern << "\n";
144 }
145
146 hs_error_t err;
147 if constexpr (!WithEditDistance)
148 err = hs_compile_multi(patterns.data(), flags.data(), ids.get(),
149 static_cast<unsigned>(patterns.size()), HS_MODE_BLOCK, nullptr, &db,
150 &compile_error);
151 else
152 err = hs_compile_ext_multi(patterns.data(), flags.data(), ids.get(), ext_exprs_ptrs.data(),
153 static_cast<unsigned>(patterns.size()), HS_MODE_BLOCK, nullptr,
154 &db, &compile_error);
155
156 if (err != HS_SUCCESS) {
158 CompilerError error(compile_error);
159
160 if (error->expression < 0)
161 LOG(FATAL) << "Logical error: " + String(error->message);
162 else
163 LOG(FATAL) << "Bad arguments: Pattern " + str_patterns[error->expression] +
164 "failed with error " + String(error->message);
165 }
166
169 hs_scratch_t* scratch = nullptr;
170 err = hs_alloc_scratch(db, &scratch);
171
173 if (err != HS_SUCCESS) LOG(FATAL) << "Could not allocate scratch space for hyperscan";
174
175 return {db, scratch};
176}

182struct GlobalCacheTable {
183 constexpr static size_t CACHE_SIZE = 500;
184
185 struct Bucket {
186 std::vector<String> patterns;
187 std::optional<UInt32> edit_distance;
190 DeferredConstructedRegexpsPtr regexps;
191 };
192
193 std::mutex mutex;
194 std::array<Bucket, CACHE_SIZE> known_regexps;
195
196 static size_t getBucketIndexFor(const std::vector<String> patterns,
197 std::optional<UInt32> edit_distance) {
198 size_t hash = 0;
199 for (const auto& pattern : patterns) boost::hash_combine(hash, pattern);
200 boost::hash_combine(hash, edit_distance);
201 return hash % CACHE_SIZE;
202 }
203};

207template <bool save_indices, bool WithEditDistance>
208DeferredConstructedRegexpsPtr getOrSet(const std::vector<StringRef>& patterns,
209 std::optional<UInt32> edit_distance) {
210 static GlobalCacheTable
211 pool;
212
213 std::vector<String> str_patterns;
214 str_patterns.reserve(patterns.size());
215 for (const auto& pattern : patterns) str_patterns.emplace_back(pattern.to_string());
216
217 size_t bucket_idx = GlobalCacheTable::getBucketIndexFor(str_patterns, edit_distance);
218
220 std::lock_guard lock(pool.mutex);
221
222 GlobalCacheTable::Bucket& bucket = pool.known_regexps[bucket_idx];
223
230
231 if (bucket.regexps == nullptr) {
233 auto deferred_constructed_regexps =
234 std::make_shared<DeferredConstructedRegexps>([str_patterns, edit_distance]() {
235 return constructRegexps<save_indices, WithEditDistance>(str_patterns,
236 edit_distance);
237 });
238 bucket = {std::move(str_patterns), edit_distance, deferred_constructed_regexps};
239 } else if (bucket.patterns != str_patterns || bucket.edit_distance != edit_distance) {
241 auto deferred_constructed_regexps =
242 std::make_shared<DeferredConstructedRegexps>([str_patterns, edit_distance]() {
243 return constructRegexps<save_indices, WithEditDistance>(str_patterns,
244 edit_distance);
245 });
246 bucket = {std::move(str_patterns), edit_distance, deferred_constructed_regexps};
247 }
248
249 return bucket.regexps;
250}
