268// Scans memory for a character.
269// memrchr is used in a few places, but it's linux-specific.
270inline void* memrchr(const void* bytes, int find_char, size_t len) {
271 const unsigned char* cursor = reinterpret_cast<const unsigned char*>(bytes) + len - 1;
272 unsigned char actual_char = find_char;
273 for (; cursor >= bytes; --cursor) {
274 if (*cursor == actual_char) {
275 return const_cast<void*>(reinterpret_cast<const void*>(cursor));
276 }
277 }
278 return NULL;
279}

458// Defined behavior on some of the uarchs:
459// PREFETCH_HINT_T0:
460// prefetch to all levels of the hierarchy (except on p4: prefetch to L2)
461// PREFETCH_HINT_NTA:
462// p4: fetch to L2, but limit to 1 way (out of the 8 ways)
463// core: skip L2, go directly to L1
464// k8 rev E and later: skip L2, can go to either of the 2-ways in L1
465enum PrefetchHint {
466 PREFETCH_HINT_T0 = 3, // More temporal locality
467 PREFETCH_HINT_T1 = 2,
468 PREFETCH_HINT_T2 = 1, // Less temporal locality
469 PREFETCH_HINT_NTA = 0 // No temporal locality
470};
471#else
472// prefetch is a no-op for this target. Feel free to add more sections above.
473#endif

475extern inline void prefetch(const char* x, int hint) {
476#if defined(__llvm__)
477 // In the gcc version of prefetch(), hint is only a constant _after_ inlining
478 // (assumed to have been successful). llvm views things differently, and
479 // checks constant-ness _before_ inlining. This leads to compilation errors
480 // with using the other version of this code with llvm.
481 //
482 // One way round this is to use a switch statement to explicitly match
483 // prefetch hint enumerations, and invoke __builtin_prefetch for each valid
484 // value. llvm's optimization removes the switch and unused case statements
485 // after inlining, so that this boils down in the end to the same as for gcc;
486 // that is, a single inlined prefetchX instruction.
487 //
488 // Note that this version of prefetch() cannot verify constant-ness of hint.
489 // If client code calls prefetch() with a variable value for hint, it will
490 // receive the full expansion of the switch below, perhaps also not inlined.
491 // This should however not be a problem in the general case of well behaved
492 // caller code that uses the supplied prefetch hint enumerations.
493 switch (hint) {
494 case PREFETCH_HINT_T0:
495 __builtin_prefetch(x, 0, PREFETCH_HINT_T0);
496 break;
497 case PREFETCH_HINT_T1:
498 __builtin_prefetch(x, 0, PREFETCH_HINT_T1);
499 break;
500 case PREFETCH_HINT_T2:
501 __builtin_prefetch(x, 0, PREFETCH_HINT_T2);
502 break;
503 case PREFETCH_HINT_NTA:
504 __builtin_prefetch(x, 0, PREFETCH_HINT_NTA);
505 break;
506 default:
507 __builtin_prefetch(x);
508 break;
509 }
510#elif defined(__GNUC__)
511#if !defined(__i386) || defined(__SSE__)
512 if (__builtin_constant_p(hint)) {
513 __builtin_prefetch(x, 0, hint);
514 } else {
515 // Defaults to PREFETCH_HINT_T0
516 __builtin_prefetch(x);
517 }
518#else
519 // We want a __builtin_prefetch, but we build with the default -march=i386
520 // where __builtin_prefetch quietly turns into nothing.
521 // Once we crank up to -march=pentium3 or higher the __SSE__
522 // clause above will kick in with the builtin.
523 // -- mec 2006-06-06
524 if (hint == PREFETCH_HINT_NTA) __asm__ __volatile__("prefetchnta (%0)" : : "r"(x));
525#endif
526#else
527 // You get no effect. Feel free to add more sections above.
528#endif
529}

576inline void* aligned_malloc(size_t size, int minimum_alignment) {
577#if defined(__APPLE__)
578 // mac lacks memalign(), posix_memalign(), however, according to
579 // http://stackoverflow.com/questions/196329/osx-lacks-memalign
580 // mac allocs are already 16-byte aligned.
581 if (minimum_alignment <= 16) return malloc(size);
582// next, try to return page-aligned memory. perhaps overkill
583#ifndef _POSIX_C_SOURCE
584 int page_size = getpagesize();
585#else
586 int page_size = vm_page_size;
587#endif
588 if (minimum_alignment <= page_size) return valloc(size);
589 // give up
590 return NULL;
591#elif defined(OS_CYGWIN)
592 return memalign(minimum_alignment, size);
593#else // !__APPLE__ && !OS_CYGWIN
594 void* ptr = NULL;
595 if (posix_memalign(&ptr, minimum_alignment, size) != 0)
596 return NULL;
597 else
598 return ptr;
599#endif
600}

826 FP_INFINITE, // is either plus or minus infinity.
827 FP_ZERO,

833inline int fpclassify_double(double x) {
834 const int float_point_class = _fpclass(x);
835 int c99_class;
836 switch (float_point_class) {
837 case _FPCLASS_SNAN: // Signaling NaN
838 case _FPCLASS_QNAN: // Quiet NaN
839 c99_class = FP_NAN;
840 break;
841 case _FPCLASS_NZ: // Negative zero ( -0)
842 case _FPCLASS_PZ: // Positive 0 (+0)
843 c99_class = FP_ZERO;
844 break;
845 case _FPCLASS_NINF: // Negative infinity ( -INF)
846 case _FPCLASS_PINF: // Positive infinity (+INF)
847 c99_class = FP_INFINITE;
848 break;
849 case _FPCLASS_ND: // Negative denormalized
850 case _FPCLASS_PD: // Positive denormalized
851 c99_class = FP_SUBNORMAL;
852 break;
853 case _FPCLASS_NN: // Negative normalized non-zero
854 case _FPCLASS_PN: // Positive normalized non-zero
855 c99_class = FP_NORMAL;
856 break;
857 default:
858 c99_class = FP_NAN; // Should never happen
859 break;
860 }
861 return c99_class;
862}

864// This function handle the special subnormal case for float; it will
865// become a normal number while casting to double.
866// bit_cast is avoided to simplify dependency and to create a code that is
867// easy to deploy in C code
868inline int fpclassify_float(float x) {
869 uint32 bitwise_representation;
870 memcpy(&bitwise_representation, &x, 4);
871 if ((bitwise_representation & 0x7f800000) == 0 && (bitwise_representation & 0x007fffff) != 0)
872 return FP_SUBNORMAL;
873 return fpclassify_double(x);
874}

876// This define takes care of the denormalized float; the casting to
877// double make it a normal number
878#define fpclassify(x) ((sizeof(x) == sizeof(float)) ? fpclassify_float(x) : fpclassify_double(x))
