84template <typename Callback>
85static inline bool putNGramASCIICaseInsensitive(const UInt8* pos, int offset,
86 Callback&& putNGramBase) {
87 struct Chars {
88 UInt8 c0;
89 UInt8 c1;
90 };
91
92 union {
93 Ngram n;
94 Chars chars;
95 };
96
97 n = toNGram(pos);
98
99 const auto c0_al = isAlphaASCII(chars.c0);
100 const auto c1_al = isAlphaASCII(chars.c1);
101
102 if (c0_al && c1_al) {
104 putNGramBase(n, offset);
105 chars.c0 = alternateCaseIfAlphaASCII(chars.c0);
106 putNGramBase(n, offset);
107 chars.c1 = alternateCaseIfAlphaASCII(chars.c1);
108 putNGramBase(n, offset);
109 chars.c0 = alternateCaseIfAlphaASCII(chars.c0);
110 putNGramBase(n, offset);
111 } else if (c0_al) {
113 putNGramBase(n, offset);
114 chars.c0 = alternateCaseIfAlphaASCII(chars.c0);
115 putNGramBase(n, offset);
116 } else if (c1_al) {
118 putNGramBase(n, offset);
119 chars.c1 = alternateCaseIfAlphaASCII(chars.c1);
120 putNGramBase(n, offset);
121 } else
123 putNGramBase(n, offset);
124
125 return true;
126}

144template <bool CaseSensitive, bool ASCII, typename FallbackSearcher>
145class VolnitskyBase {
146protected:
147 const UInt8* needle;
148 size_t needle_size;
149 const UInt8* needle_end = needle + needle_size;
151 size_t step = needle_size - sizeof(VolnitskyTraits::Ngram) + 1;
152
155 std::unique_ptr<VolnitskyTraits::Offset[]> hash;
156
157 bool fallback;
158
159 FallbackSearcher fallback_searcher;
160
161public:
162 using Searcher = FallbackSearcher;
163
168 VolnitskyBase(const char* const needle_, const size_t needle_size_,
169 size_t haystack_size_hint = 0)
170 : needle {reinterpret_cast<const UInt8*>(needle_)},
171 needle_size {needle_size_},
172 fallback {VolnitskyTraits::isFallbackNeedle(needle_size, haystack_size_hint)},
173 fallback_searcher {needle_, needle_size} {
174 if (fallback || fallback_searcher.force_fallback) return;
175
176 hash = std::unique_ptr<VolnitskyTraits::Offset[]>(
177 new VolnitskyTraits::Offset[VolnitskyTraits::hash_size] {});
178
179 auto callback = [this](const VolnitskyTraits::Ngram ngram, const int offset) {
180 return this->putNGramBase(ngram, offset);
181 };
184 for (auto i = static_cast<ssize_t>(needle_size - sizeof(VolnitskyTraits::Ngram)); i >= 0;
185 --i) {
186 bool ok = VolnitskyTraits::putNGram<CaseSensitive, ASCII>(needle + i, i + 1, needle,
187 needle_size, callback);
188
193 if (!ok) {
194 fallback_searcher.force_fallback = true;
195 hash = nullptr;
196 return;
197 }
198 }
199 }
200
202 const UInt8* search(const UInt8* const haystack, const size_t haystack_size) const {
203 if (needle_size == 0) return haystack;
204
205 const auto* haystack_end = haystack + haystack_size;
206
207 if (fallback || haystack_size <= needle_size || fallback_searcher.force_fallback)
208 return fallback_searcher.search(haystack, haystack_end);
209
211 const auto* pos = haystack + needle_size - sizeof(VolnitskyTraits::Ngram);
212 for (; pos <= haystack_end - needle_size; pos += step) {
214 for (size_t cell_num = VolnitskyTraits::toNGram(pos) % VolnitskyTraits::hash_size;
215 hash[cell_num]; cell_num = (cell_num + 1) % VolnitskyTraits::hash_size) {
217 const auto* res = pos - (hash[cell_num] - 1);
218
220 if (fallback_searcher.compare(haystack, haystack_end, res)) return res;
221 }
222 }
223
224 return fallback_searcher.search(pos - step + 1, haystack_end);
225 }
226
227 const char* search(const char* haystack, size_t haystack_size) const {
228 return reinterpret_cast<const char*>(
229 search(reinterpret_cast<const UInt8*>(haystack), haystack_size));
230 }
231
232protected:
233 void putNGramBase(const VolnitskyTraits::Ngram ngram, const int offset) {
235 size_t cell_num = ngram % VolnitskyTraits::hash_size;
236
237 while (hash[cell_num])
238 cell_num =
239 (cell_num + 1) % VolnitskyTraits::hash_size;
240
241 hash[cell_num] = offset;
242 }
243};

245template <bool CaseSensitive, bool ASCII, typename FallbackSearcher>
246class MultiVolnitskyBase {
247private:
249 const std::vector<StringRef>& needles;
250
252 std::vector<size_t> fallback_needles;
253 std::vector<FallbackSearcher> fallback_searchers;

256 struct OffsetId {
257 VolnitskyTraits::Id id;
258 VolnitskyTraits::Offset off;
259 };
260
261 std::unique_ptr<OffsetId[]> hash;
262
264 size_t step;
265
267 size_t last;
268
270 static constexpr size_t small_limit = VolnitskyTraits::hash_size / 8;
271
272public:
273 explicit MultiVolnitskyBase(const std::vector<StringRef>& needles_)
274 : needles {needles_}, step {0}, last {0} {
275 fallback_searchers.reserve(needles.size());
276 hash = std::unique_ptr<OffsetId[]>(
277 new OffsetId[VolnitskyTraits::
278 hash_size]);
279 }
280
294 bool hasMoreToSearch() {
295 if (last == needles.size()) return false;
296
297 memset(hash.get(), 0, VolnitskyTraits::hash_size * sizeof(OffsetId));
298 fallback_needles.clear();
299 step = std::numeric_limits<size_t>::max();
300
301 size_t buf = 0;
302 size_t size = needles.size();
303
304 for (; last < size; ++last) {
305 const char* cur_needle_data = needles[last].data;
306 const size_t cur_needle_size = needles[last].size;
307
309 if (VolnitskyTraits::isFallbackNeedle(cur_needle_size)) {
310 fallback_needles.push_back(last);
311 } else {
313 auto callback = [this](const VolnitskyTraits::Ngram ngram, const int offset) {
314 return this->putNGramBase(ngram, offset, this->last);
315 };
316
317 buf += cur_needle_size - sizeof(VolnitskyTraits::Ngram) + 1;
318
320 if (buf > small_limit) break;
321
322 step = std::min(step, cur_needle_size - sizeof(VolnitskyTraits::Ngram) + 1);
323 for (auto i = static_cast<int>(cur_needle_size - sizeof(VolnitskyTraits::Ngram));
324 i >= 0; --i) {
325 VolnitskyTraits::putNGram<CaseSensitive, ASCII>(
326 reinterpret_cast<const UInt8*>(cur_needle_data) + i, i + 1,
327 reinterpret_cast<const UInt8*>(cur_needle_data), cur_needle_size,
328 callback);
329 }
330 }
331 fallback_searchers.emplace_back(cur_needle_data, cur_needle_size);
332 }
333 return true;
334 }

336 inline bool searchOne(const UInt8* haystack, const UInt8* haystack_end) const {
337 const size_t fallback_size = fallback_needles.size();
338 for (size_t i = 0; i < fallback_size; ++i)
339 if (fallback_searchers[fallback_needles[i]].search(haystack, haystack_end) !=
340 haystack_end)
341 return true;
342
344 if (step != std::numeric_limits<size_t>::max()) {
345 const auto* pos = haystack + step - sizeof(VolnitskyTraits::Ngram);
346 for (; pos <= haystack_end - sizeof(VolnitskyTraits::Ngram); pos += step) {
347 for (size_t cell_num = VolnitskyTraits::toNGram(pos) % VolnitskyTraits::hash_size;
348 hash[cell_num].off; cell_num = (cell_num + 1) % VolnitskyTraits::hash_size) {
349 if (pos >= haystack + hash[cell_num].off - 1) {
350 const auto res = pos - (hash[cell_num].off - 1);
351 const size_t ind = hash[cell_num].id;
352 if (res + needles[ind].size <= haystack_end &&
353 fallback_searchers[ind].compare(haystack, haystack_end, res))
354 return true;
355 }
356 }
357 }
358 }
359 return false;
360 }

362 inline size_t searchOneFirstIndex(const UInt8* haystack, const UInt8* haystack_end) const {
363 const size_t fallback_size = fallback_needles.size();
364
365 size_t answer = std::numeric_limits<size_t>::max();
366
367 for (size_t i = 0; i < fallback_size; ++i)
368 if (fallback_searchers[fallback_needles[i]].search(haystack, haystack_end) !=
369 haystack_end)
370 answer = std::min(answer, fallback_needles[i]);
371
373 if (step != std::numeric_limits<size_t>::max()) {
374 const auto* pos = haystack + step - sizeof(VolnitskyTraits::Ngram);
375 for (; pos <= haystack_end - sizeof(VolnitskyTraits::Ngram); pos += step) {
376 for (size_t cell_num = VolnitskyTraits::toNGram(pos) % VolnitskyTraits::hash_size;
377 hash[cell_num].off; cell_num = (cell_num + 1) % VolnitskyTraits::hash_size) {
378 if (pos >= haystack + hash[cell_num].off - 1) {
379 const auto res = pos - (hash[cell_num].off - 1);
380 const size_t ind = hash[cell_num].id;
381 if (res + needles[ind].size <= haystack_end &&
382 fallback_searchers[ind].compare(haystack, haystack_end, res))
383 answer = std::min(answer, ind);
384 }
385 }
386 }
387 }
388
389 /*
390 * if nothing was found, answer + 1 will be equal to zero and we can
391 * assign it into the result because we need to return the position starting with one
392 */
393 return answer + 1;
394 }

396 template <typename CountCharsCallback>
397 inline UInt64 searchOneFirstPosition(const UInt8* haystack, const UInt8* haystack_end,
398 const CountCharsCallback& count_chars) const {
399 const size_t fallback_size = fallback_needles.size();
400
401 UInt64 answer = std::numeric_limits<UInt64>::max();
402
403 for (size_t i = 0; i < fallback_size; ++i)
404 if (auto pos = fallback_searchers[fallback_needles[i]].search(haystack, haystack_end);
405 pos != haystack_end)
406 answer = std::min<UInt64>(answer, pos - haystack);
407
409 if (step != std::numeric_limits<size_t>::max()) {
410 const auto* pos = haystack + step - sizeof(VolnitskyTraits::Ngram);
411 for (; pos <= haystack_end - sizeof(VolnitskyTraits::Ngram); pos += step) {
412 for (size_t cell_num = VolnitskyTraits::toNGram(pos) % VolnitskyTraits::hash_size;
413 hash[cell_num].off; cell_num = (cell_num + 1) % VolnitskyTraits::hash_size) {
414 if (pos >= haystack + hash[cell_num].off - 1) {
415 const auto res = pos - (hash[cell_num].off - 1);
416 const size_t ind = hash[cell_num].id;
417 if (res + needles[ind].size <= haystack_end &&
418 fallback_searchers[ind].compare(haystack, haystack_end, res))
419 answer = std::min<UInt64>(answer, res - haystack);
420 }
421 }
422 }
423 }
424 if (answer == std::numeric_limits<UInt64>::max()) return 0;
425 return count_chars(haystack, haystack + answer);
426 }

428 template <typename CountCharsCallback, typename AnsType>
429 inline void searchOneAll(const UInt8* haystack, const UInt8* haystack_end, AnsType* answer,
430 const CountCharsCallback& count_chars) const {
431 const size_t fallback_size = fallback_needles.size();
432 for (size_t i = 0; i < fallback_size; ++i) {
433 const UInt8* ptr =
434 fallback_searchers[fallback_needles[i]].search(haystack, haystack_end);
435 if (ptr != haystack_end) answer[fallback_needles[i]] = count_chars(haystack, ptr);
436 }
437
439 if (step != std::numeric_limits<size_t>::max()) {
440 const auto* pos = haystack + step - sizeof(VolnitskyTraits::Ngram);
441 for (; pos <= haystack_end - sizeof(VolnitskyTraits::Ngram); pos += step) {
442 for (size_t cell_num = VolnitskyTraits::toNGram(pos) % VolnitskyTraits::hash_size;
443 hash[cell_num].off; cell_num = (cell_num + 1) % VolnitskyTraits::hash_size) {
444 if (pos >= haystack + hash[cell_num].off - 1) {
445 const auto* res = pos - (hash[cell_num].off - 1);
446 const size_t ind = hash[cell_num].id;
447 if (answer[ind] == 0 && res + needles[ind].size <= haystack_end &&
448 fallback_searchers[ind].compare(haystack, haystack_end, res))
449 answer[ind] = count_chars(haystack, res);
450 }
451 }
452 }
453 }
454 }

456 void putNGramBase(const VolnitskyTraits::Ngram ngram, const int offset, const size_t num) {
457 size_t cell_num = ngram % VolnitskyTraits::hash_size;
458
459 while (hash[cell_num].off) cell_num = (cell_num + 1) % VolnitskyTraits::hash_size;
460
461 hash[cell_num] = {static_cast<VolnitskyTraits::Id>(num),
462 static_cast<VolnitskyTraits::Offset>(offset)};
463 }
