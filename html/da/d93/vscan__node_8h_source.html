97 enum class PushDownType {
98 // The predicate can not be pushed down to data source
99 UNACCEPTABLE,
100 // The predicate can be pushed down to data source
101 // and the data source can fully evaludate it
102 ACCEPTABLE,
103 // The predicate can be pushed down to data source
104 // but the data source can not fully evaluate it.
105 PARTIAL_ACCEPTABLE
106 };

112 // Process predicates, extract the predicates in the conjuncts that can be pushed down
113 // to the data source, and convert them into common expressions structure ColumnPredicate.
114 // There are currently 3 types of predicates that can be pushed down to data sources:
115 //
116 // 1. Simple predicate, with column on left and constant on right, such as "a=1", "b in (1,2,3)" etc.
117 // 2. Bloom Filter, predicate condition generated by runtime filter
118 // 3. Function Filter, some data sources can accept function conditions, such as "a like 'abc%'"
119 //
120 // Predicates that can be fully processed by the data source will be removed from conjuncts
121 virtual Status _process_conjuncts() {
122 RETURN_IF_ERROR(_normalize_conjuncts());
123 return Status::OK();
124 }

126 // Create a list of scanners.
127 // The number of scanners is related to the implementation of the data source,
128 // predicate conditions, and scheduling strategy.
129 // So this method needs to be implemented separately by the subclass of ScanNode.
130 // Finally, a set of scanners that have been prepared are returned.
131 virtual Status _init_scanners(std::list<VScanner*>* scanners) { return Status::OK(); }
