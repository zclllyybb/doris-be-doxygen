30class StreamLoadContext;
31// used to register all streams in process so that other module can get this stream
32class NewLoadStreamMgr {
33public:
34 NewLoadStreamMgr();
35 ~NewLoadStreamMgr();
36
37 Status put(const UniqueId& id, std::shared_ptr<StreamLoadContext> stream) {
38 std::lock_guard<std::mutex> l(_lock);
39 auto it = _stream_map.find(id);
40 if (it != std::end(_stream_map)) {
41 return Status::InternalError("id already exist");
42 }
43 _stream_map.emplace(id, stream);
44 VLOG_NOTICE << "put stream load pipe: " << id;
45 return Status::OK();
46 }
47
48 std::shared_ptr<StreamLoadContext> get(const UniqueId& id) {
49 std::lock_guard<std::mutex> l(_lock);
50 auto it = _stream_map.find(id);
51 if (it == std::end(_stream_map)) {
52 return std::shared_ptr<StreamLoadContext>(nullptr);
53 }
54 return it->second;
55 }
56
57 void remove(const UniqueId& id) {
58 std::lock_guard<std::mutex> l(_lock);
59 auto it = _stream_map.find(id);
60 if (it != std::end(_stream_map)) {
61 _stream_map.erase(it);
62 VLOG_NOTICE << "remove stream load pipe: " << id;
63 }
64 }
65
66private:
67 std::mutex _lock;
68 std::unordered_map<UniqueId, std::shared_ptr<StreamLoadContext>> _stream_map;
69};
70} // namespace doris
