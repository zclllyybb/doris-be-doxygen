48 bool evaluate_and(const std::pair<WrapperField*, WrapperField*>& statistic) const override {
49 if (_is_null) {
50 return statistic.first->is_null();
51 } else {
52 return !statistic.second->is_null();
53 }
54 }
55
56 bool evaluate_del(const std::pair<WrapperField*, WrapperField*>& statistic) const override {
57 if (_is_null) {
58 return statistic.first->is_null() && statistic.second->is_null();
59 } else {
60 return !statistic.first->is_null() && !statistic.second->is_null();
61 }
62 }
63
64 bool evaluate_and(const segment_v2::BloomFilter* bf) const override {
65 if (_is_null) {
66 return bf->test_bytes(nullptr, 0);
67 } else {
68 LOG(FATAL) << "Bloom filter is not supported by predicate type: is_null=" << _is_null;
69 return true;
70 }
71 }
72
73 bool can_do_bloom_filter() const override { return _is_null; }
74
75 void evaluate_vec(const vectorized::IColumn& column, uint16_t size, bool* flags) const override;
76
77private:
78 std::string _debug_string() const override {
79 std::string info = "NullPredicate(" + std::string(_is_null ? "is_null" : "not_null") + ")";
80 return info;
81 }
82
83 bool _is_null; //true for null, false for not null
84};
