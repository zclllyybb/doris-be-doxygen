28// This class contains constants useful for text processing with SSE4.2
29// intrinsics.
30namespace sse_util {
31// Number of characters that fit in 64/128 bit register.
32// SSE provides instructions for loading 64 or 128 bits into a register
33// at a time.
34static const int CHARS_PER_64_BIT_REGISTER = 8;
35static const int CHARS_PER_128_BIT_REGISTER = 16;
36
37// SSE4.2 adds instructions for textprocessing. The instructions accept
38// a flag to control what text operation to do.
39// - SIDD_CMP_EQUAL_ANY ~ strchr
40// - SIDD_CMP_EQUAL_EACH ~ strcmp
41// - SIDD_UBYTE_OPS - 8 bit chars (as opposed to 16 bit)
42// - SIDD_NEGATIVE_POLARITY - toggles whether to set result to 1 or 0 when a
43// match is found.
44
45// In this mode, sse text processing functions will return a mask of all the characters that
46// matched
47static const int STRCHR_MODE = _SIDD_CMP_EQUAL_ANY | _SIDD_UBYTE_OPS;
48
49// In this mode, sse text processing functions will return the number of bytes that match
50// consecutively from the beginning.
51static const int STRCMP_MODE = _SIDD_CMP_EQUAL_EACH | _SIDD_UBYTE_OPS | _SIDD_NEGATIVE_POLARITY;
52
53// Precomputed mask values up to 16 bits.
54static const int SSE_BITMASK[CHARS_PER_128_BIT_REGISTER] = {
55 1 << 0, 1 << 1, 1 << 2, 1 << 3, 1 << 4, 1 << 5, 1 << 6, 1 << 7,
56 1 << 8, 1 << 9, 1 << 10, 1 << 11, 1 << 12, 1 << 13, 1 << 14, 1 << 15,
57};
58
59} // namespace sse_util
60} // namespace doris
