56 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
57 size_t result, size_t input_rows_count) override {
58 const ColumnPtr source_col = block.get_by_position(arguments[0]).column;
59 const auto is_nullable = block.get_by_position(result).type->is_nullable();
60 const auto* sources = check_and_get_column<ColumnVector<typename Transform::OpArgType>>(
61 remove_nullable(source_col).get());
62 auto col_res = ColumnString::create();
63
64 // Support all input of datetime is valind to make sure not null return
65 if (sources) {
66 if (is_nullable) {
67 auto null_map = ColumnVector<UInt8>::create(input_rows_count);
68 TransformerToStringOneArgument<Transform>::vector(
69 context, sources->get_data(), col_res->get_chars(), col_res->get_offsets(),
70 null_map->get_data());
71 if (const auto* nullable_col =
72 check_and_get_column<ColumnNullable>(source_col.get())) {
73 NullMap& result_null_map = assert_cast<ColumnUInt8&>(*null_map).get_data();
74 const NullMap& src_null_map =
75 assert_cast<const ColumnUInt8&>(nullable_col->get_null_map_column())
76 .get_data();
77 VectorizedUtils::update_null_map(result_null_map, src_null_map);
78 }
79 block.replace_by_position(
80 result, ColumnNullable::create(std::move(col_res), std::move(null_map)));
81 } else {
82 TransformerToStringOneArgument<Transform>::vector(
83 context, sources->get_data(), col_res->get_chars(), col_res->get_offsets());
84 block.replace_by_position(result, std::move(col_res));
85 }
86 } else {
87 return Status::InternalError("Illegal column {} of first argument of function {}",
88 block.get_by_position(arguments[0]).column->get_name(),
89 name);
90 }
91 return Status::OK();
92 }
93};
