52enum class RuntimeFilterType {

54 IN_FILTER = 0,
55 MINMAX_FILTER = 1,

57 IN_OR_BLOOM_FILTER = 3,
58 BITMAP_FILTER = 4

61inline std::string to_string(RuntimeFilterType type) {
62 switch (type) {
63 case RuntimeFilterType::IN_FILTER: {
64 return std::string("in");
65 }
66 case RuntimeFilterType::BLOOM_FILTER: {
67 return std::string("bloomfilter");
68 }
69 case RuntimeFilterType::MINMAX_FILTER: {

73 return std::string("in_or_bloomfilter");
74 }
75 case RuntimeFilterType::BITMAP_FILTER: {
76 return std::string("bitmapfilter");
77 }
78 default:
79 return std::string("UNKNOWN");
80 }

368template <PrimitiveType T>
369Status create_texpr_literal_node(const void* data, TExprNode* node, int precision = 0,
370 int scale = 0) {
371 if constexpr (T == TYPE_BOOLEAN) {
372 auto origin_value = reinterpret_cast<const bool*>(data);
373 TBoolLiteral boolLiteral;
374 (*node).__set_node_type(TExprNodeType::BOOL_LITERAL);
375 boolLiteral.__set_value(*origin_value);
376 (*node).__set_bool_literal(boolLiteral);
377 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_BOOLEAN));
378 } else if constexpr (T == TYPE_TINYINT) {
379 auto origin_value = reinterpret_cast<const int8_t*>(data);
380 (*node).__set_node_type(TExprNodeType::INT_LITERAL);
381 TIntLiteral intLiteral;
382 intLiteral.__set_value(*origin_value);
383 (*node).__set_int_literal(intLiteral);
384 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_TINYINT));
385 } else if constexpr (T == TYPE_SMALLINT) {
386 auto origin_value = reinterpret_cast<const int16_t*>(data);
387 (*node).__set_node_type(TExprNodeType::INT_LITERAL);
388 TIntLiteral intLiteral;
389 intLiteral.__set_value(*origin_value);
390 (*node).__set_int_literal(intLiteral);
391 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_SMALLINT));
392 } else if constexpr (T == TYPE_INT) {
393 auto origin_value = reinterpret_cast<const int32_t*>(data);
394 (*node).__set_node_type(TExprNodeType::INT_LITERAL);
395 TIntLiteral intLiteral;
396 intLiteral.__set_value(*origin_value);
397 (*node).__set_int_literal(intLiteral);
398 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_INT));
399 } else if constexpr (T == TYPE_BIGINT) {
400 auto origin_value = reinterpret_cast<const int64_t*>(data);
401 (*node).__set_node_type(TExprNodeType::INT_LITERAL);
402 TIntLiteral intLiteral;
403 intLiteral.__set_value(*origin_value);
404 (*node).__set_int_literal(intLiteral);
405 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_BIGINT));
406 } else if constexpr (T == TYPE_LARGEINT) {
407 auto origin_value = reinterpret_cast<const int128_t*>(data);
408 (*node).__set_node_type(TExprNodeType::LARGE_INT_LITERAL);
409 TLargeIntLiteral large_int_literal;
410 large_int_literal.__set_value(LargeIntValue::to_string(*origin_value));
411 (*node).__set_large_int_literal(large_int_literal);
412 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_LARGEINT));
413 } else if constexpr ((T == TYPE_DATE) || (T == TYPE_DATETIME) || (T == TYPE_TIME)) {
414 auto origin_value = reinterpret_cast<const doris::vectorized::VecDateTimeValue*>(data);
415 TDateLiteral date_literal;
416 char convert_buffer[30];
417 origin_value->to_string(convert_buffer);
418 date_literal.__set_value(convert_buffer);
419 (*node).__set_date_literal(date_literal);
420 (*node).__set_node_type(TExprNodeType::DATE_LITERAL);
421 if (origin_value->type() == TimeType::TIME_DATE) {
422 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DATE));
423 } else if (origin_value->type() == TimeType::TIME_DATETIME) {
424 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DATETIME));
425 } else if (origin_value->type() == TimeType::TIME_TIME) {
426 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_TIME));
427 }
428 } else if constexpr (T == TYPE_DATEV2) {
429 auto origin_value = reinterpret_cast<
430 const doris::vectorized::DateV2Value<doris::vectorized::DateV2ValueType>*>(data);
431 TDateLiteral date_literal;
432 char convert_buffer[30];
433 origin_value->to_string(convert_buffer);
434 date_literal.__set_value(convert_buffer);
435 (*node).__set_date_literal(date_literal);
436 (*node).__set_node_type(TExprNodeType::DATE_LITERAL);
437 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DATEV2));
438 } else if constexpr (T == TYPE_DATETIMEV2) {
439 auto origin_value = reinterpret_cast<
440 const doris::vectorized::DateV2Value<doris::vectorized::DateTimeV2ValueType>*>(
441 data);
442 TDateLiteral date_literal;
443 char convert_buffer[30];
444 origin_value->to_string(convert_buffer);
445 date_literal.__set_value(convert_buffer);
446 (*node).__set_date_literal(date_literal);
447 (*node).__set_node_type(TExprNodeType::DATE_LITERAL);
448 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DATETIMEV2));
449 } else if constexpr (T == TYPE_DECIMALV2) {
450 auto origin_value = reinterpret_cast<const DecimalV2Value*>(data);
451 (*node).__set_node_type(TExprNodeType::DECIMAL_LITERAL);
452 TDecimalLiteral decimal_literal;
453 decimal_literal.__set_value(origin_value->to_string());
454 (*node).__set_decimal_literal(decimal_literal);
455 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DECIMALV2, precision, scale));
456 } else if constexpr (T == TYPE_DECIMAL32) {
457 auto origin_value = reinterpret_cast<const vectorized::Decimal<int32_t>*>(data);
458 (*node).__set_node_type(TExprNodeType::DECIMAL_LITERAL);
459 TDecimalLiteral decimal_literal;
460 decimal_literal.__set_value(origin_value->to_string(scale));
461 (*node).__set_decimal_literal(decimal_literal);
462 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DECIMAL32, precision, scale));
463 } else if constexpr (T == TYPE_DECIMAL64) {
464 auto origin_value = reinterpret_cast<const vectorized::Decimal<int64_t>*>(data);
465 (*node).__set_node_type(TExprNodeType::DECIMAL_LITERAL);
466 TDecimalLiteral decimal_literal;
467 decimal_literal.__set_value(origin_value->to_string(scale));
468 (*node).__set_decimal_literal(decimal_literal);
469 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DECIMAL64, precision, scale));
470 } else if constexpr (T == TYPE_DECIMAL128I) {
471 auto origin_value = reinterpret_cast<const vectorized::Decimal<int128_t>*>(data);
472 (*node).__set_node_type(TExprNodeType::DECIMAL_LITERAL);
473 TDecimalLiteral decimal_literal;
474 decimal_literal.__set_value(origin_value->to_string(scale));
475 (*node).__set_decimal_literal(decimal_literal);
476 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DECIMAL128I, precision, scale));
477 } else if constexpr (T == TYPE_FLOAT) {
478 auto origin_value = reinterpret_cast<const float*>(data);
479 (*node).__set_node_type(TExprNodeType::FLOAT_LITERAL);
480 TFloatLiteral float_literal;
481 float_literal.__set_value(*origin_value);
482 (*node).__set_float_literal(float_literal);
483 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_FLOAT));
484 } else if constexpr (T == TYPE_DOUBLE) {
485 auto origin_value = reinterpret_cast<const double*>(data);
486 (*node).__set_node_type(TExprNodeType::FLOAT_LITERAL);
487 TFloatLiteral float_literal;
488 float_literal.__set_value(*origin_value);
489 (*node).__set_float_literal(float_literal);
490 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_DOUBLE));
491 } else if constexpr ((T == TYPE_STRING) || (T == TYPE_CHAR) || (T == TYPE_VARCHAR)) {
492 auto origin_value = reinterpret_cast<const StringRef*>(data);
493 (*node).__set_node_type(TExprNodeType::STRING_LITERAL);
494 TStringLiteral string_literal;
495 string_literal.__set_value(origin_value->to_string());
496 (*node).__set_string_literal(string_literal);
497 (*node).__set_type(create_type_desc(PrimitiveType::TYPE_STRING));
498 } else {
499 return Status::InvalidArgument("Invalid argument type!");
500 }
501 return Status::OK();
502}
