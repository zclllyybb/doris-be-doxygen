33// Wrapper around Cache, and used for cache page of column data
34// in Segment.
35// TODO(zc): We should add some metric to see cache hit/miss rate.
36class StoragePageCache {
37public:
38 // The unique key identifying entries in the page cache.
39 // Each cached page corresponds to a specific offset within
40 // a file.
41 //
42 // TODO(zc): Now we use file name(std::string) as a part of
43 // key, which is not efficient. We should make it better later
44 struct CacheKey {
45 CacheKey(std::string fname_, int64_t offset_) : fname(std::move(fname_)), offset(offset_) {}
46 std::string fname;
47 int64_t offset;
48
49 // Encode to a flat binary which can be used as LRUCache's key
50 std::string encode() const {
51 std::string key_buf(fname);
52 key_buf.append((char*)&offset, sizeof(offset));
53 return key_buf;
54 }
55 };
56
57 static constexpr uint32_t kDefaultNumShards = 16;
58
59 // Create global instance of this class
60 static void create_global_cache(size_t capacity, int32_t index_cache_percentage,
61 uint32_t num_shards = kDefaultNumShards);
62
63 // Return global instance.
64 // Client should call create_global_cache before.
65 static StoragePageCache* instance() { return _s_instance; }
66
67 StoragePageCache(size_t capacity, int32_t index_cache_percentage, uint32_t num_shards);
68
69 // Lookup the given page in the cache.
70 //
71 // If the page is found, the cache entry will be written into handle.
72 // PageCacheHandle will release cache entry to cache when it
73 // destructs.
74 //
75 // Cache type selection is determined by page_type argument
76 //
77 // Return true if entry is found, otherwise return false.
78 bool lookup(const CacheKey& key, PageCacheHandle* handle, segment_v2::PageTypePB page_type);
79
80 // Insert a page with key into this cache.
81 // Given handle will be set to valid reference.
82 // This function is thread-safe, and when two clients insert two same key
83 // concurrently, this function can assure that only one page is cached.
84 // The in_memory page will have higher priority.
85 void insert(const CacheKey& key, const Slice& data, PageCacheHandle* handle,
86 segment_v2::PageTypePB page_type, bool in_memory = false);
87
88 // Page cache available check.
89 // When percentage is set to 0 or 100, the index or data cache will not be allocated.
90 bool is_cache_available(segment_v2::PageTypePB page_type) {
91 return _get_page_cache(page_type) != nullptr;
92 }
93
94 void prune(segment_v2::PageTypePB page_type);
95
96 int64_t get_page_cache_mem_consumption(segment_v2::PageTypePB page_type) {
97 return _get_page_cache(page_type)->mem_consumption();
98 }
99
100private:
101 StoragePageCache();
102 static StoragePageCache* _s_instance;
103
104 int32_t _index_cache_percentage = 0;
105 std::unique_ptr<Cache> _data_page_cache = nullptr;
106 std::unique_ptr<Cache> _index_page_cache = nullptr;
107
108 Cache* _get_page_cache(segment_v2::PageTypePB page_type) {
109 switch (page_type) {
110 case segment_v2::DATA_PAGE: {
111 return _data_page_cache.get();
112 }
113 case segment_v2::INDEX_PAGE:
114 return _index_page_cache.get();
115 default:
116 return nullptr;
117 }
118 }
119};

121// A handle for StoragePageCache entry. This class make it easy to handle
122// Cache entry. Users don't need to release the obtained cache entry. This
123// class will release the cache entry when it is destroyed.
124class PageCacheHandle {
125public:
126 PageCacheHandle() {}
127 PageCacheHandle(Cache* cache, Cache::Handle* handle) : _cache(cache), _handle(handle) {}
128 ~PageCacheHandle() {
129 if (_handle != nullptr) {
130 _cache->release(_handle);
131 }
132 }
133
134 PageCacheHandle(PageCacheHandle&& other) noexcept {
135 // we can use std::exchange if we switch c++14 on
136 std::swap(_cache, other._cache);
137 std::swap(_handle, other._handle);
138 }
139
140 PageCacheHandle& operator=(PageCacheHandle&& other) noexcept {
141 std::swap(_cache, other._cache);
142 std::swap(_handle, other._handle);
143 return *this;
144 }
145
146 Cache* cache() const { return _cache; }
147 Slice data() const { return _cache->value_slice(_handle); }
148
149private:
150 Cache* _cache = nullptr;
151 Cache::Handle* _handle = nullptr;
152
153 // Don't allow copy and assign
154 DISALLOW_COPY_AND_ASSIGN(PageCacheHandle);
155};
