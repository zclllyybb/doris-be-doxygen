198template <doris::vectorized::UInt64 salt>
199inline doris::vectorized::UInt32 int_hash32(doris::vectorized::UInt64 key) {
200 key ^= salt;
201
202 key = (~key) + (key << 18);
203 key = key ^ ((key >> 31) | (key << 33));
204 key = key * 21;
205 key = key ^ ((key >> 11) | (key << 53));
206 key = key + (key << 6);
207 key = key ^ ((key >> 22) | (key << 42));
208
209 return key;
210}
