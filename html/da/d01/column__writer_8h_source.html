85 explicit ColumnWriter(std::unique_ptr<Field> field, bool is_nullable)

87
88 virtual ~ColumnWriter() = default;
89
90 virtual Status init() = 0;
91
92 template <typename CellType>
93 Status append(const CellType& cell) {
94 if (_is_nullable) {
95 uint8_t nullmap = 0;
96 BitmapChange(&nullmap, 0, cell.is_null());
97 return append_nullable(&nullmap, cell.cell_ptr(), 1);
98 } else {
99 auto* cel_ptr = cell.cell_ptr();
100 return append_data((const uint8_t**)&cel_ptr, 1);
101 }
102 }
103
104 // Now we only support append one by one, we should support append
105 // multi rows in one call
106 Status append(bool is_null, void* data) {
107 uint8_t nullmap = 0;
108 BitmapChange(&nullmap, 0, is_null);
109 return append_nullable(&nullmap, data, 1);
110 }

213 // All Pages will be organized into a linked list
214 struct Page {
215 // the data vector may contain:
216 // 1. one OwnedSlice if the page body is compressed
217 // 2. one OwnedSlice if the page body is not compressed and doesn't have nullmap
218 // 3. two OwnedSlice if the page body is not compressed and has nullmap
219 // use vector for easier management for lifetime of OwnedSlice
220 std::vector<OwnedSlice> data;
221 PageFooterPB footer;
222 Page* next = nullptr;
223 };
224
225 struct PageHead {
226 Page* head = nullptr;
227 Page* tail = nullptr;
228 };
229
230 void _push_back_page(Page* page) {
231 // add page to pages' tail
232 if (_pages.tail != nullptr) {
233 _pages.tail->next = page;
234 }
235 _pages.tail = page;
236 if (_pages.head == nullptr) {
237 _pages.head = page;
238 }
239 for (auto& data_slice : page->data) {
240 _data_size += data_slice.slice().size;
241 }
242 // estimate (page footer + footer size + checksum) took 20 bytes
243 _data_size += 20;
244 }
245
246 Status _write_data_page(Page* page);
247
248private:
249 io::FileWriter* _file_writer = nullptr;
250 // total size of data page list
251 uint64_t _data_size;
252
253 // cached generated pages,
254 PageHead _pages;
255 ordinal_t _first_rowid = 0;
256
257 BlockCompressionCodec* _compress_codec;

259 std::unique_ptr<OrdinalIndexWriter> _ordinal_index_builder;
