30struct ArrayReverseImpl {
31 static Status _execute(Block& block, const ColumnNumbers& arguments, size_t result,
32 size_t input_rows_count) {
33 ColumnPtr src_column =
34 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
35 ColumnArrayExecutionData src;
36 if (!extract_column_array_info(*src_column, src)) {
37 return Status::RuntimeError(
38 fmt::format("execute failed, unsupported types for function {}({})", "reverse",
39 block.get_by_position(arguments[0]).type->get_name()));
40 }
41
42 bool is_nullable = src.nested_nullmap_data ? true : false;
43 ColumnArrayMutableData dst = create_mutable_data(src.nested_col, is_nullable);
44 dst.offsets_ptr->reserve(input_rows_count);
45
46 auto res_val = _execute_internal(*src.nested_col, *src.offsets_ptr, *dst.nested_col,
47 *dst.offsets_ptr, src.nested_nullmap_data,
48 dst.nested_nullmap_data);
49 if (!res_val) {
50 return Status::RuntimeError(
51 fmt::format("execute failed or unsupported types for function {}({})",
52 "reverse", block.get_by_position(arguments[0]).type->get_name()));
53 }
54
55 ColumnPtr res_column = assemble_column_array(dst);
56 block.replace_by_position(result, std::move(res_column));
57 return Status::OK();
58 }
59
60 static bool _execute_internal(const IColumn& src_column,
61 const ColumnArray::Offsets64& src_offsets, IColumn& dest_column,
62 ColumnArray::Offsets64& dest_offsets, const UInt8* src_null_map,
63 ColumnUInt8::Container* dest_null_map) {
64 size_t prev_src_offset = 0;
65
66 for (auto curr_src_offset : src_offsets) {
67 size_t array_size = curr_src_offset - prev_src_offset;
68 for (size_t j = 0; j < array_size; ++j) {
69 size_t j_reverse = curr_src_offset - j - 1;
70 if (src_null_map && src_null_map[j_reverse]) {
71 DCHECK(dest_null_map != nullptr);
72 // Note: here we need to insert default value
73 dest_column.insert_default();
74 (*dest_null_map).push_back(true);
75 continue;
76 }
77
78 dest_column.insert_from(src_column, j_reverse);
79 if (dest_null_map) {
80 (*dest_null_map).push_back(false);
81 }
82 }
83
84 dest_offsets.push_back(curr_src_offset);
85 prev_src_offset = curr_src_offset;
86 }
87
88 return true;
89 }
90};
