69struct ParquetInt96 {
70 uint64_t lo; // time of nanoseconds in a day
71 uint32_t hi; // days from julian epoch
72
73 inline uint64_t to_timestamp_micros() const {
74 return (hi - JULIAN_EPOCH_OFFSET_DAYS) * MICROS_IN_DAY + lo / NANOS_PER_MICROSECOND;
75 }
76
77 static const uint32_t JULIAN_EPOCH_OFFSET_DAYS;
78 static const uint64_t MICROS_IN_DAY;
79 static const uint64_t NANOS_PER_MICROSECOND;
80};
