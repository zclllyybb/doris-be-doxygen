200template <typename DecimalPrimitiveType>
201void Decoder::init_decimal_converter(DataTypePtr& data_type) {
202 if (_decode_params == nullptr || _field_schema == nullptr ||
203 _decode_params->decimal_scale.scale_type != DecimalScaleParams::NOT_INIT) {
204 return;
205 }
206 auto scale = _field_schema->parquet_schema.scale;
207 auto* decimal_type = reinterpret_cast<DataTypeDecimal<Decimal<DecimalPrimitiveType>>*>(
208 const_cast<IDataType*>(remove_nullable(data_type).get()));
209 auto dest_scale = decimal_type->get_scale();
210 if (dest_scale > scale) {
211 _decode_params->decimal_scale.scale_type = DecimalScaleParams::SCALE_UP;
212 _decode_params->decimal_scale.scale_factor =
213 DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(dest_scale - scale);
214 } else if (dest_scale < scale) {
215 _decode_params->decimal_scale.scale_type = DecimalScaleParams::SCALE_DOWN;
216 _decode_params->decimal_scale.scale_factor =
217 DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(scale - dest_scale);
218 } else {
219 _decode_params->decimal_scale.scale_type = DecimalScaleParams::NO_SCALE;
220 _decode_params->decimal_scale.scale_factor = 1;
221 }
222}

273template <typename Numeric>
274Status FixLengthDecoder::_decode_numeric(MutableColumnPtr& doris_column,
275 ColumnSelectVector& select_vector) {
276 auto& column_data = static_cast<ColumnVector<Numeric>&>(*doris_column).get_data();
277 size_t data_index = column_data.size();
278 column_data.resize(data_index + select_vector.num_values() - select_vector.num_filtered());
279 size_t dict_index = 0;
280 ColumnSelectVector::DataReadType read_type;
281 while (size_t run_length = select_vector.get_next_run(&read_type)) {
282 switch (read_type) {
283 case ColumnSelectVector::CONTENT: {
284 for (size_t i = 0; i < run_length; ++i) {
285 char* buf_start = _FIXED_GET_DATA_OFFSET(dict_index++);
286 column_data[data_index++] = *(Numeric*)buf_start;
287 _FIXED_SHIFT_DATA_OFFSET();
288 }
289 break;
290 }
291 case ColumnSelectVector::NULL_DATA: {
292 data_index += run_length;
293 break;
294 }
295 case ColumnSelectVector::FILTERED_CONTENT: {
296 if (_has_dict) {
297 dict_index += run_length;
298 } else {
299 _offset += _type_length * run_length;
300 }
301 break;
302 }
303 case ColumnSelectVector::FILTERED_NULL: {
304 // do nothing
305 break;
306 }
307 }
308 }
309 return Status::OK();
310}

451template <typename DecimalPrimitiveType>
452Status FixLengthDecoder::_decode_binary_decimal(MutableColumnPtr& doris_column,
453 DataTypePtr& data_type,
454 ColumnSelectVector& select_vector) {
455 init_decimal_converter<DecimalPrimitiveType>(data_type);
456 auto& column_data =
457 static_cast<ColumnDecimal<Decimal<DecimalPrimitiveType>>&>(*doris_column).get_data();
458 size_t data_index = column_data.size();
459 column_data.resize(data_index + select_vector.num_values() - select_vector.num_filtered());
460 size_t dict_index = 0;
461 DecimalScaleParams& scale_params = _decode_params->decimal_scale;
462
463 ColumnSelectVector::DataReadType read_type;
464 while (size_t run_length = select_vector.get_next_run(&read_type)) {
465 switch (read_type) {
466 case ColumnSelectVector::CONTENT: {
467 for (size_t i = 0; i < run_length; ++i) {
468 char* buf_start = _FIXED_GET_DATA_OFFSET(dict_index++);
469 // When Decimal in parquet is stored in byte arrays, binary and fixed,
470 // the unscaled number must be encoded as two's complement using big-endian byte order.
471 Int128 value = buf_start[0] & 0x80 ? -1 : 0;
472 memcpy(reinterpret_cast<char*>(&value) + sizeof(Int128) - _type_length, buf_start,
473 _type_length);
474 value = BigEndian::ToHost128(value);
475 if (scale_params.scale_type == DecimalScaleParams::SCALE_UP) {
476 value *= scale_params.scale_factor;
477 } else if (scale_params.scale_type == DecimalScaleParams::SCALE_DOWN) {
478 value /= scale_params.scale_factor;
479 }
480 auto& v = reinterpret_cast<DecimalPrimitiveType&>(column_data[data_index++]);
481 v = (DecimalPrimitiveType)value;
482 _FIXED_SHIFT_DATA_OFFSET();
483 }
484 break;
485 }
486 case ColumnSelectVector::NULL_DATA: {
487 data_index += run_length;
488 break;
489 }
490 case ColumnSelectVector::FILTERED_CONTENT: {
491 if (_has_dict) {
492 dict_index += run_length;
493 } else {
494 _offset += _type_length * run_length;
495 }
496 break;
497 }
498 case ColumnSelectVector::FILTERED_NULL: {
499 // do nothing
500 break;
501 }
502 }
503 }
504 return Status::OK();
505}

582template <typename DecimalPrimitiveType>
583Status ByteArrayDecoder::_decode_binary_decimal(MutableColumnPtr& doris_column,
584 DataTypePtr& data_type,
585 ColumnSelectVector& select_vector) {
586 init_decimal_converter<DecimalPrimitiveType>(data_type);
587 auto& column_data =
588 static_cast<ColumnDecimal<Decimal<DecimalPrimitiveType>>&>(*doris_column).get_data();
589 size_t data_index = column_data.size();
590 column_data.resize(data_index + select_vector.num_values() - select_vector.num_filtered());
591 size_t dict_index = 0;
592 DecimalScaleParams& scale_params = _decode_params->decimal_scale;
593 ColumnSelectVector::DataReadType read_type;
594 while (size_t run_length = select_vector.get_next_run(&read_type)) {
595 switch (read_type) {
596 case ColumnSelectVector::CONTENT: {
597 for (size_t i = 0; i < run_length; ++i) {
598 char* buf_start;
599 uint32_t length;
600 if (_has_dict) {
601 StringRef& slice = _dict_items[_indexes[dict_index++]];
602 buf_start = const_cast<char*>(slice.data);
603 length = (uint32_t)slice.size;
604 } else {
605 if (UNLIKELY(_offset + 4 > _data->size)) {
606 return Status::IOError("Can't read byte array length from plain decoder");
607 }
608 length = decode_fixed32_le(reinterpret_cast<const uint8_t*>(_data->data) +
609 _offset);
610 _offset += 4;
611 buf_start = _data->data + _offset;
612 _offset += length;
613 }
614 // When Decimal in parquet is stored in byte arrays, binary and fixed,
615 // the unscaled number must be encoded as two's complement using big-endian byte order.
616 Int128 value = buf_start[0] & 0x80 ? -1 : 0;
617 memcpy(reinterpret_cast<char*>(&value) + sizeof(Int128) - length, buf_start,
618 length);
619 value = BigEndian::ToHost128(value);
620 if (scale_params.scale_type == DecimalScaleParams::SCALE_UP) {
621 value *= scale_params.scale_factor;
622 } else if (scale_params.scale_type == DecimalScaleParams::SCALE_DOWN) {
623 value /= scale_params.scale_factor;
624 }
625 auto& v = reinterpret_cast<DecimalPrimitiveType&>(column_data[data_index++]);
626 v = (DecimalPrimitiveType)value;
627 }
628 break;
629 }
630 case ColumnSelectVector::NULL_DATA: {
631 data_index += run_length;
632 break;
633 }
634 case ColumnSelectVector::FILTERED_CONTENT: {
635 if (_has_dict) {
636 dict_index += run_length;
637 } else {
638 _offset += _type_length * run_length;
639 }
