43template <bool result_is_nullable, typename Derived>
44class AggregateFunctionNullBase : public IAggregateFunctionHelper<Derived> {
45protected:
46 AggregateFunctionPtr nested_function;
47 size_t prefix_size;
48
56 AggregateDataPtr nested_place(AggregateDataPtr __restrict place) const noexcept {
57 return place + prefix_size;
58 }
59
60 ConstAggregateDataPtr nested_place(ConstAggregateDataPtr __restrict place) const noexcept {
61 return place + prefix_size;
62 }
63
64 static void init_flag(AggregateDataPtr __restrict place) noexcept {
65 if constexpr (result_is_nullable) {
66 place[0] = false;
67 }
68 }
69
70 static void set_flag(AggregateDataPtr __restrict place) noexcept {
71 if constexpr (result_is_nullable) {
72 place[0] = true;
73 }
74 }
75
76 static bool get_flag(ConstAggregateDataPtr __restrict place) noexcept {
77 return result_is_nullable ? place[0] : true;
78 }
79
80public:
81 AggregateFunctionNullBase(AggregateFunctionPtr nested_function_, const DataTypes& arguments,
82 const Array& params)
83 : IAggregateFunctionHelper<Derived>(arguments, params),
84 nested_function {nested_function_} {
85 if (result_is_nullable) {
86 prefix_size = nested_function->align_of_data();
87 } else {
88 prefix_size = 0;
89 }
90 }
91
92 String get_name() const override {
94 return nested_function->get_name();
95 }
96
97 DataTypePtr get_return_type() const override {
98 return result_is_nullable ? make_nullable(nested_function->get_return_type())
99 : nested_function->get_return_type();
100 }
101
102 void create(AggregateDataPtr __restrict place) const override {
103 init_flag(place);
104 nested_function->create(nested_place(place));
105 }
106
107 void destroy(AggregateDataPtr __restrict place) const noexcept override {
108 nested_function->destroy(nested_place(place));
109 }
110 void reset(AggregateDataPtr place) const override {
111 init_flag(place);
112 nested_function->reset(nested_place(place));
113 }
114
115 bool has_trivial_destructor() const override {
116 return nested_function->has_trivial_destructor();
117 }
118
119 size_t size_of_data() const override { return prefix_size + nested_function->size_of_data(); }
120
121 size_t align_of_data() const override { return nested_function->align_of_data(); }
122
123 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
124 Arena* arena) const override {
125 if (result_is_nullable && get_flag(rhs)) {
126 set_flag(place);
127 }
128
129 nested_function->merge(nested_place(place), nested_place(rhs), arena);
130 }
131
132 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
133 bool flag = get_flag(place);
134 if (result_is_nullable) {
135 write_binary(flag, buf);
136 }
137 if (flag) {
138 nested_function->serialize(nested_place(place), buf);
139 }
140 }
141
142 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
143 Arena* arena) const override {
144 bool flag = true;
145 if (result_is_nullable) {
146 read_binary(flag, buf);
147 }
148 if (flag) {
149 set_flag(place);
150 nested_function->deserialize(nested_place(place), buf, arena);
151 }
152 }
153
154 void deserialize_and_merge(AggregateDataPtr __restrict place, BufferReadable& buf,
155 Arena* arena) const override {
156 bool flag = true;
157 if (result_is_nullable) {
158 read_binary(flag, buf);
159 }
160 if (flag) {
161 set_flag(place);
162 nested_function->deserialize_and_merge(nested_place(place), buf, arena);
163 }
164 }
165
166 void deserialize_and_merge_from_column(AggregateDataPtr __restrict place, const IColumn& column,
167 Arena* arena) const override {
168 size_t num_rows = column.size();
169 for (size_t i = 0; i != num_rows; ++i) {
170 VectorBufferReader buffer_reader(
171 (assert_cast<const ColumnString&>(column)).get_data_at(i));
172 deserialize_and_merge(place, buffer_reader, arena);
173 }
174 }
175
176 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
177 if constexpr (result_is_nullable) {
178 ColumnNullable& to_concrete = assert_cast<ColumnNullable&>(to);
179 if (get_flag(place)) {
180 nested_function->insert_result_into(nested_place(place),
181 to_concrete.get_nested_column());
182 to_concrete.get_null_map_data().push_back(0);
183 } else {
184 to_concrete.insert_default();
185 }
186 } else {
187 nested_function->insert_result_into(nested_place(place), to);
188 }
189 }
190
191 bool allocates_memory_in_arena() const override {
192 return nested_function->allocates_memory_in_arena();
193 }
194
195 bool is_state() const override { return nested_function->is_state(); }
196};

201template <bool result_is_nullable>
202class AggregateFunctionNullUnary final
203 : public AggregateFunctionNullBase<result_is_nullable,
204 AggregateFunctionNullUnary<result_is_nullable>> {
205public:
206 AggregateFunctionNullUnary(AggregateFunctionPtr nested_function_, const DataTypes& arguments,
207 const Array& params)
208 : AggregateFunctionNullBase<result_is_nullable,
209 AggregateFunctionNullUnary<result_is_nullable>>(
210 std::move(nested_function_), arguments, params) {}
211
212 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
213 Arena* arena) const override {
214 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
215 if (!column->is_null_at(row_num)) {
216 this->set_flag(place);
217 const IColumn* nested_column = &column->get_nested_column();
218 this->nested_function->add(this->nested_place(place), &nested_column, row_num, arena);
219 }
220 }
221
222 void add_not_nullable(AggregateDataPtr __restrict place, const IColumn** columns,
223 size_t row_num, Arena* arena) const {
224 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
225 this->set_flag(place);
226 const IColumn* nested_column = &column->get_nested_column();
227 this->nested_function->add(this->nested_place(place), &nested_column, row_num, arena);
228 }
229
230 void add_batch(size_t batch_size, AggregateDataPtr* places, size_t place_offset,
231 const IColumn** columns, Arena* arena, bool agg_many) const override {
232 int processed_records_num = 0;
233
234 // we can use column->has_null() to judge whether whole batch of data is null and skip batch,
235 // but it's maybe too coarse-grained.
236#ifdef __AVX2__
237 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
238 // The overhead introduced is negligible here, just an extra memory read from NullMap
239 const NullMap& null_map_data = column->get_null_map_data();
240
241 // NullMap use uint8_t type to indicate values is null or not, 1 indicates null, 0 versus.
242 // It's important to keep consistent with element type size in NullMap
243 constexpr int simd_batch_size = 256 / (8 * sizeof(uint8_t));
244 __m256i all0 = _mm256_setzero_si256();
245 auto null_map_start_position = reinterpret_cast<const int8_t*>(null_map_data.data());
246
247 while (processed_records_num + simd_batch_size <= batch_size) {
248 // load unaligned data from null_map, 1 means value is null, 0 versus
249 __m256i f = _mm256_loadu_si256(reinterpret_cast<const __m256i*>(
250 null_map_start_position + processed_records_num));
251 int mask = _mm256_movemask_epi8(_mm256_cmpgt_epi8(f, all0));
252 // all data is null
253 if (mask == 0xffffffff) {
254 } else if (mask == 0) { // all data is not null
255 for (size_t i = processed_records_num; i < processed_records_num + simd_batch_size;
256 i++) {
257 AggregateFunctionNullUnary::add_not_nullable(places[i] + place_offset, columns,
258 i, arena);
259 }
260 } else {
261 // data is partly null
262 for (size_t i = processed_records_num; i < processed_records_num + simd_batch_size;
263 i++) {
264 add(places[i] + place_offset, columns, i, arena);
265 }
266 }
267 processed_records_num += simd_batch_size;
268 }
269
270#elif __SSE2__
271 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
272 // The overhead introduced is negligible here, just an extra memory read from NullMap
273 const NullMap& null_map_data = column->get_null_map_data();
274 // NullMap use uint8_t type to indicate values is null or not, 1 indicates null, 0 versus.
275 // It's important to keep consistent with element type size in NullMap
276 constexpr int simd_batch_size = 128 / (8 * sizeof(uint8_t));
277 __m128i all0 = _mm_setzero_si128();
278 auto null_map_start_position = reinterpret_cast<const int8_t*>(null_map_data.data());
279 while (processed_records_num + simd_batch_size <= batch_size) {
280 // load unaligned data from null_map, 1 means value is null, 0 versus
281 __m128i f = _mm_loadu_si128(reinterpret_cast<const __m128i*>(null_map_start_position +
282 processed_records_num));
283 int mask = _mm_movemask_epi8(_mm_cmpgt_epi8(f, all0));
284 // all data is null
285 if (mask == 0xffff) {
286 } else if (mask == 0) { // all data is not null
287 for (size_t i = processed_records_num; i < processed_records_num + simd_batch_size;
288 i++) {
289 add_not_nullable(places[i] + place_offset, columns, i, arena);
290 }
291 } else {
292 // data is partly null
293 for (size_t i = processed_records_num; i < processed_records_num + simd_batch_size;
294 i++) {
295 add(places[i] + place_offset, columns, i, arena);
296 }
297 }
298 processed_records_num += simd_batch_size;
299 }
300#endif
301
302 for (; processed_records_num < batch_size; ++processed_records_num) {
303 add(places[processed_records_num] + place_offset, columns, processed_records_num,
304 arena);
305 }
306 }
307
308 void add_batch_single_place(size_t batch_size, AggregateDataPtr place, const IColumn** columns,
309 Arena* arena) const override {
310 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
311 bool has_null = column->has_null();
312
313 if (has_null) {
314 for (size_t i = 0; i < batch_size; ++i) {
315 if (!column->is_null_at(i)) {
316 this->set_flag(place);
317 this->add(place, columns, i, arena);
318 }
319 }
320 } else {
321 this->set_flag(place);
322 const IColumn* nested_column = &column->get_nested_column();
323 this->nested_function->add_batch_single_place(batch_size, this->nested_place(place),
324 &nested_column, arena);
325 }
326 }
327
328 void add_batch_range(size_t batch_begin, size_t batch_end, AggregateDataPtr place,
329 const IColumn** columns, Arena* arena, bool has_null) override {
330 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
331
332 if (has_null) {
333 for (size_t i = batch_begin; i <= batch_end; ++i) {
334 if (!column->is_null_at(i)) {
335 this->set_flag(place);
336 this->add(place, columns, i, arena);
337 }
338 }
339 } else {
340 this->set_flag(place);
341 const IColumn* nested_column = &column->get_nested_column();
342 this->nested_function->add_batch_range(
343 batch_begin, batch_end, this->nested_place(place), &nested_column, arena);
344 }
345 }

348template <bool result_is_nullable>
349class AggregateFunctionNullVariadic final
350 : public AggregateFunctionNullBase<result_is_nullable,
351 AggregateFunctionNullVariadic<result_is_nullable>> {
352public:
353 AggregateFunctionNullVariadic(AggregateFunctionPtr nested_function_, const DataTypes& arguments,
354 const Array& params)
355 : AggregateFunctionNullBase<result_is_nullable,
356 AggregateFunctionNullVariadic<result_is_nullable>>(
357 std::move(nested_function_), arguments, params),
358 number_of_arguments(arguments.size()) {
359 if (number_of_arguments == 1) {
360 LOG(FATAL)
361 << "Logical error: single argument is passed to AggregateFunctionNullVariadic";
362 }
363
364 if (number_of_arguments > MAX_ARGS) {
365 LOG(FATAL) << fmt::format(
366 "Maximum number of arguments for aggregate function with Nullable types is {}",
367 size_t(MAX_ARGS));
368 }
369
370 for (size_t i = 0; i < number_of_arguments; ++i) {
371 is_nullable[i] = arguments[i]->is_nullable();
372 }
373 }
374
375 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
376 Arena* arena) const override {
378 const IColumn* nested_columns[number_of_arguments];
379
380 for (size_t i = 0; i < number_of_arguments; ++i) {
381 if (is_nullable[i]) {
382 const ColumnNullable& nullable_col =
383 assert_cast<const ColumnNullable&>(*columns[i]);
384 if (nullable_col.is_null_at(row_num)) {
387 return;
388 }
389 nested_columns[i] = &nullable_col.get_nested_column();
390 } else {
391 nested_columns[i] = columns[i];
392 }
393 }
394
395 this->set_flag(place);
396 this->nested_function->add(this->nested_place(place), nested_columns, row_num, arena);
397 }
398
399 bool allocates_memory_in_arena() const override {
400 return this->nested_function->allocates_memory_in_arena();
401 }
402
403private:
404 // The array length is fixed in the implementation of some aggregate functions.
405 // Therefore we choose 256 as the appropriate maximum length limit.
406 static const size_t MAX_ARGS = 256;
407 size_t number_of_arguments = 0;
408 std::array<char, MAX_ARGS>
409 is_nullable;
410};
