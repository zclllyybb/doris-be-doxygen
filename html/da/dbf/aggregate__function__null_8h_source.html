35template <typename NestFunction, bool result_is_nullable, typename Derived>
36class AggregateFunctionNullBaseInline : public IAggregateFunctionHelper<Derived> {
37protected:
38 std::unique_ptr<NestFunction> nested_function;
39 size_t prefix_size;
40
48 AggregateDataPtr nested_place(AggregateDataPtr __restrict place) const noexcept {
49 return place + prefix_size;
50 }
51
52 ConstAggregateDataPtr nested_place(ConstAggregateDataPtr __restrict place) const noexcept {
53 return place + prefix_size;
54 }
55
56 static void init_flag(AggregateDataPtr __restrict place) noexcept {
57 if constexpr (result_is_nullable) {
58 place[0] = false;
59 }
60 }
61
62 static void set_flag(AggregateDataPtr __restrict place) noexcept {
63 if constexpr (result_is_nullable) {
64 place[0] = true;
65 }
66 }
67
68 static bool get_flag(ConstAggregateDataPtr __restrict place) noexcept {
69 return result_is_nullable ? place[0] : true;
70 }
71
72public:
73 AggregateFunctionNullBaseInline(IAggregateFunction* nested_function_,
74 const DataTypes& arguments)
75 : IAggregateFunctionHelper<Derived>(arguments),
76 nested_function {assert_cast<NestFunction*>(nested_function_)} {
77 if (result_is_nullable) {
78 prefix_size = nested_function->align_of_data();
79 } else {
80 prefix_size = 0;
81 }
82 }
83
84 String get_name() const override {
86 return nested_function->get_name();
87 }
88
89 DataTypePtr get_return_type() const override {
90 return result_is_nullable ? make_nullable(nested_function->get_return_type())
91 : nested_function->get_return_type();
92 }
93
94 void create(AggregateDataPtr __restrict place) const override {
95 init_flag(place);
96 nested_function->create(nested_place(place));
97 }
98
99 void destroy(AggregateDataPtr __restrict place) const noexcept override {
100 nested_function->destroy(nested_place(place));
101 }
102 void reset(AggregateDataPtr place) const override {
103 init_flag(place);
104 nested_function->reset(nested_place(place));
105 }
106
107 bool has_trivial_destructor() const override {
108 return nested_function->has_trivial_destructor();
109 }
110
111 size_t size_of_data() const override { return prefix_size + nested_function->size_of_data(); }
112
113 size_t align_of_data() const override { return nested_function->align_of_data(); }
114
115 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
116 Arena* arena) const override {
117 if (result_is_nullable && get_flag(rhs)) {
118 set_flag(place);
119 }
120
121 nested_function->merge(nested_place(place), nested_place(rhs), arena);
122 }
123
124 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
125 bool flag = get_flag(place);
126 if (result_is_nullable) {
127 write_binary(flag, buf);
128 }
129 if (flag) {
130 nested_function->serialize(nested_place(place), buf);
131 }
132 }
133
134 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
135 Arena* arena) const override {
136 bool flag = true;
137 if (result_is_nullable) {
138 read_binary(flag, buf);
139 }
140 if (flag) {
141 set_flag(place);
142 nested_function->deserialize(nested_place(place), buf, arena);
143 }
144 }
145
146 void deserialize_and_merge(AggregateDataPtr __restrict place, BufferReadable& buf,
147 Arena* arena) const override {
148 bool flag = true;
149 if (result_is_nullable) {
150 read_binary(flag, buf);
151 }
152 if (flag) {
153 set_flag(place);
154 nested_function->deserialize_and_merge(nested_place(place), buf, arena);
155 }
156 }
157
158 void deserialize_and_merge_from_column(AggregateDataPtr __restrict place, const IColumn& column,
159 Arena* arena) const override {
160 size_t num_rows = column.size();
161 for (size_t i = 0; i != num_rows; ++i) {
162 VectorBufferReader buffer_reader(
163 (assert_cast<const ColumnString&>(column)).get_data_at(i));
164 deserialize_and_merge(place, buffer_reader, arena);
165 }
166 }
167
168 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
169 if constexpr (result_is_nullable) {
170 ColumnNullable& to_concrete = assert_cast<ColumnNullable&>(to);
171 if (get_flag(place)) {
172 nested_function->insert_result_into(nested_place(place),
173 to_concrete.get_nested_column());
174 to_concrete.get_null_map_data().push_back(0);
175 } else {
176 to_concrete.insert_default();
177 }
178 } else {
179 nested_function->insert_result_into(nested_place(place), to);
180 }
181 }
182
183 bool allocates_memory_in_arena() const override {
184 return nested_function->allocates_memory_in_arena();
185 }
186
187 bool is_state() const override { return nested_function->is_state(); }
188};

193template <typename NestFuction, bool result_is_nullable>
194class AggregateFunctionNullUnaryInline final
195 : public AggregateFunctionNullBaseInline<
196 NestFuction, result_is_nullable,
197 AggregateFunctionNullUnaryInline<NestFuction, result_is_nullable>> {
198public:
199 AggregateFunctionNullUnaryInline(IAggregateFunction* nested_function_,
200 const DataTypes& arguments)
201 : AggregateFunctionNullBaseInline<
202 NestFuction, result_is_nullable,
203 AggregateFunctionNullUnaryInline<NestFuction, result_is_nullable>>(
204 nested_function_, arguments) {}
205
206 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
207 Arena* arena) const override {
208 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
209 if (!column->is_null_at(row_num)) {
210 this->set_flag(place);
211 const IColumn* nested_column = &column->get_nested_column();
212 this->nested_function->add(this->nested_place(place), &nested_column, row_num, arena);
213 }
214 }
215
216 void add_batch(size_t batch_size, AggregateDataPtr* places, size_t place_offset,
217 const IColumn** columns, Arena* arena, bool agg_many) const override {
218 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
219 // The overhead introduced is negligible here, just an extra memory read from NullMap
220 const auto* __restrict null_map_data = column->get_null_map_data().data();
221 const IColumn* nested_column = &column->get_nested_column();
222 for (int i = 0; i < batch_size; ++i) {
223 if (!null_map_data[i]) {
224 AggregateDataPtr __restrict place = places[i] + place_offset;
225 this->set_flag(place);
226 this->nested_function->add(this->nested_place(place), &nested_column, i, arena);
227 }
228 }
229 }
230
231 void add_batch_single_place(size_t batch_size, AggregateDataPtr place, const IColumn** columns,
232 Arena* arena) const override {
233 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
234 bool has_null = column->has_null();
235
236 if (has_null) {
237 for (size_t i = 0; i < batch_size; ++i) {
238 this->add(place, columns, i, arena);
239 }
240 } else {
241 this->set_flag(place);
242 const IColumn* nested_column = &column->get_nested_column();
243 this->nested_function->add_batch_single_place(batch_size, this->nested_place(place),
244 &nested_column, arena);
245 }
246 }
247
248 void add_batch_range(size_t batch_begin, size_t batch_end, AggregateDataPtr place,
249 const IColumn** columns, Arena* arena, bool has_null) override {
250 const ColumnNullable* column = assert_cast<const ColumnNullable*>(columns[0]);
251
252 if (has_null) {
253 for (size_t i = batch_begin; i <= batch_end; ++i) {
254 this->add(place, columns, i, arena);
255 }
256 } else {
257 this->set_flag(place);
258 const IColumn* nested_column = &column->get_nested_column();
259 this->nested_function->add_batch_range(batch_begin, batch_end,
260 this->nested_place(place), &nested_column, arena,
261 false);
262 }
263 }
264};

266template <typename NestFuction, bool result_is_nullable>
267class AggregateFunctionNullVariadicInline final
268 : public AggregateFunctionNullBaseInline<
269 NestFuction, result_is_nullable,
270 AggregateFunctionNullVariadicInline<NestFuction, result_is_nullable>> {
271public:
272 AggregateFunctionNullVariadicInline(IAggregateFunction* nested_function_,
273 const DataTypes& arguments)
274 : AggregateFunctionNullBaseInline<
275 NestFuction, result_is_nullable,
276 AggregateFunctionNullVariadicInline<NestFuction, result_is_nullable>>(
277 nested_function_, arguments),
278 number_of_arguments(arguments.size()) {
279 if (number_of_arguments == 1) {
280 LOG(FATAL)
281 << "Logical error: single argument is passed to AggregateFunctionNullVariadic";
282 }
283
284 if (number_of_arguments > MAX_ARGS) {
285 LOG(FATAL) << fmt::format(
286 "Maximum number of arguments for aggregate function with Nullable types is {}",
287 size_t(MAX_ARGS));
288 }
289
290 for (size_t i = 0; i < number_of_arguments; ++i) {
291 is_nullable[i] = arguments[i]->is_nullable();
292 }
293 }
294
295 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
296 Arena* arena) const override {
298 const IColumn* nested_columns[number_of_arguments];
299
300 for (size_t i = 0; i < number_of_arguments; ++i) {
301 if (is_nullable[i]) {
302 const ColumnNullable& nullable_col =
303 assert_cast<const ColumnNullable&>(*columns[i]);
304 if (nullable_col.is_null_at(row_num)) {
307 return;
308 }
309 nested_columns[i] = &nullable_col.get_nested_column();
310 } else {
311 nested_columns[i] = columns[i];
312 }
313 }
314
315 this->set_flag(place);
316 this->nested_function->add(this->nested_place(place), nested_columns, row_num, arena);
317 }
318
319 bool allocates_memory_in_arena() const override {
320 return this->nested_function->allocates_memory_in_arena();
321 }
322
323private:
324 // The array length is fixed in the implementation of some aggregate functions.
325 // Therefore we choose 256 as the appropriate maximum length limit.
326 static const size_t MAX_ARGS = 256;
327 size_t number_of_arguments = 0;
328 std::array<char, MAX_ARGS>
329 is_nullable;
330};
