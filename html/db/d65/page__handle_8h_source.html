27// When a column page is read into memory, we use this to store it.
28// A page's data may be in cache, or may not in cache. We use this
29// class to unify these two cases.
30// If client use this struct to wrap data not in cache, this class
31// will free data's memory when it is destroyed.
32class PageHandle {
33public:
34 PageHandle() : _is_data_owner(false) {}
35
36 // This class will take the ownership of input data's memory. It will
37 // free it when deconstructs.
38 PageHandle(const Slice& data) : _is_data_owner(true), _data(data) {}
39
40 // This class will take the content of cache data, and will make input
41 // cache_data to a invalid cache handle.
42 PageHandle(PageCacheHandle cache_data)
43 : _is_data_owner(false), _cache_data(std::move(cache_data)) {}
44
45 // Move constructor
46 PageHandle(PageHandle&& other) noexcept
47 : _is_data_owner(false),
48 _data(std::move(other._data)),
49 _cache_data(std::move(other._cache_data)) {
50 // we can use std::exchange if we switch c++14 on
51 std::swap(_is_data_owner, other._is_data_owner);
52 }
53
54 PageHandle& operator=(PageHandle&& other) noexcept {
55 std::swap(_is_data_owner, other._is_data_owner);
56 _data = std::move(other._data);
57 _cache_data = std::move(other._cache_data);
58 return *this;
59 }
60
61 ~PageHandle() {
62 if (_is_data_owner && _data.size > 0) {
63 delete[] _data.data;
64 }
65 }
66
67 // the return slice contains uncompressed page body, page footer, and footer size
68 Slice data() const {
69 if (_is_data_owner) {
70 return _data;
71 } else {
72 return _cache_data.data();
73 }
74 }
75
76private:
77 // when this is true, it means this struct own data and _data is valid.
78 // otherwise _cache_data is valid, and data is belong to cache.
79 bool _is_data_owner = false;
80 Slice _data;
81 PageCacheHandle _cache_data;
82
83 // Don't allow copy and assign
84 DISALLOW_COPY_AND_ASSIGN(PageHandle);
85};
