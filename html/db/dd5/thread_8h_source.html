123 // Returns a unique, stable identifier for this thread. Note that this is a static
124 // method and thus can be used on any thread, including the main thread of the
125 // process.
126 //
127 // In general, this should be used when a value is required that is unique to
128 // a thread and must work on any thread including the main process thread.
129 //
130 // NOTE: this is _not_ the TID, but rather a unique value assigned by the
131 // thread implementation. So, this value should not be presented to the user
132 // in log messages, etc.
133 static int64_t unique_thread_id();
134
135 // Returns the system thread ID (tid on Linux) for the current thread. Note
136 // that this is a static method and thus can be used from any thread,
137 // including the main thread of the process. This is in contrast to
138 // Thread::tid(), which only works on doris::Threads.
139 //
140 // Thread::tid() will return the same value, but the value is cached in the
141 // Thread object, so will be faster to call.
142 //
143 // Thread::unique_thread_id() (or Thread::tid()) should be preferred for
144 // performance sensitive code, however it is only guaranteed to return a
145 // unique and stable thread ID, not necessarily the system thread ID.
146 static int64_t current_thread_id();

170 // OS-specific thread ID. Once the constructor finishes start_thread(),
171 // guaranteed to be set either to a non-negative integer, or to INVALID_TID.
172 //
173 // The tid_ member goes through the following states:
174 // 1. INVALID_TID: the thread has not been started, or has already exited.
175 // 2. PARENT_WAITING_TID: the parent has started the thread, but the
176 // thread has not yet begun running. Therefore the TID is not yet known
177 // but it will be set once the thread starts.
178 // 3. <positive value>: the thread is running.

202 // Starts the thread running supervise_thread(), and returns once that thread has
203 // initialised and its TID has been read. Waits for notification from the started
204 // thread that initialisation is complete before returning. On success, stores a
205 // reference to the thread in holder.
206 static Status start_thread(const std::string& category, const std::string& name,
207 const ThreadFunctor& functor, uint64_t flags,
208 scoped_refptr<Thread>* holder);
209
210 // Wrapper for the user-supplied function. Invoked from the new thread,
211 // with the Thread as its only argument. Executes _functor, but before
212 // doing so registers with the global ThreadMgr and reads the thread's
213 // system ID. After _functor terminates, unregisters with the ThreadMgr.
214 // Always returns nullptr.
215 //
216 // supervise_thread() notifies start_thread() when thread initialisation is
217 // completed via the _tid, which is set to the new thread's system ID.
218 // By that point in time supervise_thread() has also taken a reference to
219 // the Thread object, allowing it to safely refer to it even after the
220 // caller drops its reference.
221 //
222 // Additionally, start_thread() notifies supervise_thread() when the actual
223 // Thread object has been assigned (supervise_thread() is spinning during
224 // this time). Without this, the new thread may reference the actual
225 // Thread object before it has been assigned by start_thread(). See
226 // KUDU-11 for more details.
227 static void* supervise_thread(void* arg);
228
229 // Invoked when the user-supplied function finishes or in the case of an
230 // abrupt exit (i.e. pthread_exit()). Cleans up after supervise_thread().
231 static void finish_thread(void* arg);

236// Utility to join on a thread, printing warning messages if it
237// takes too long. For example:
238//
239// ThreadJoiner(&my_thread, "processing thread")
240// .warn_after_ms(1000)
241// .warn_every_ms(5000)
242// .Join();
243//
244// TODO: would be nice to offer a way to use ptrace() or signals to
245// dump the stack trace of the thread we're trying to join on if it
246// gets stuck. But, after looking for 20 minutes or so, it seems
247// pretty complicated to get right.
248class ThreadJoiner {
249public:
250 explicit ThreadJoiner(Thread* thread);
251
252 // Start emitting warnings after this many milliseconds.
253 //
254 // Default: 1000 ms.
255 ThreadJoiner& warn_after_ms(int ms);
256
257 // After the warnings after started, emit another warning at the
258 // given interval.
259 //
260 // Default: 1000 ms.
261 ThreadJoiner& warn_every_ms(int ms);
262
263 // If the thread has not stopped after this number of milliseconds, give up
264 // joining on it and return Status::Aborted.
265 //
266 // -1 (the default) means to wait forever trying to join.
267 ThreadJoiner& give_up_after_ms(int ms);
268
269 // Join the thread, subject to the above parameters. If the thread joining
270 // fails for any reason, returns RuntimeError. If it times out, returns
271 // Aborted.
272 Status join();
273
274private:
275 enum {
276 kDefaultWarnAfterMs = 1000,
277 kDefaultWarnEveryMs = 1000,
278 kDefaultGiveUpAfterMs = -1 // forever
279 };
280
281 Thread* _thread;
282
283 int _warn_after_ms;
284 int _warn_every_ms;
285 int _give_up_after_ms;
286
287 DISALLOW_COPY_AND_ASSIGN(ThreadJoiner);
288};
