134//
135// Macros for quickly declaring and implementing exception classes.
136// Unfortunately, we cannot use a template here because character
137// pointers (which we need for specifying the exception name)
138// are not allowed as template arguments.
139//
140#define DORIS_DECLARE_EXCEPTION_CODE(API, CLS, BASE, CODE) \
141 class CLS : public BASE { \
142 public: \
143 CLS(int code = CODE); \
144 CLS(const std::string& msg, int code = CODE); \
145 CLS(const std::string& msg, const std::string& arg, int code = CODE); \
146 CLS(const std::string& msg, const AbstractException& exc, int code = CODE); \
147 CLS(const CLS& exc); \
148 ~CLS() throw(); \
149 CLS& operator=(const CLS& exc); \
150 const char* name() const throw(); \
151 const char* className() const throw(); \
152 AbstractException* clone() const; \
153 void rethrow() const; \
154 };
155
156#define DORIS_DECLARE_EXCEPTION(API, CLS, BASE) POCO_DECLARE_EXCEPTION_CODE(API, CLS, BASE, 0)
157
158#define DORIS_IMPLEMENT_EXCEPTION(CLS, BASE, NAME) \
159 CLS::CLS(int code) : BASE(code) {} \
160 CLS::CLS(const std::string& msg, int code) : BASE(msg, code) {} \
161 CLS::CLS(const std::string& msg, const std::string& arg, int code) : BASE(msg, arg, code) {} \
162 CLS::CLS(const std::string& msg, const AbstractException& exc, int code) \
163 : BASE(msg, exc, code) {} \
164 CLS::CLS(const CLS& exc) : BASE(exc) {} \
165 CLS::~CLS() throw() {} \
166 CLS& CLS::operator=(const CLS& exc) { \
167 BASE::operator=(exc); \
168 return *this; \
169 } \
170 const char* CLS::name() const throw() { return NAME; } \
171 const char* CLS::className() const throw() { return typeid(*this).name(); } \
172 AbstractException* CLS::clone() const { return new CLS(*this); } \
173 void CLS::rethrow() const { throw *this; }
