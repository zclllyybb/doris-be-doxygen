168class VMergeIterator : public RowwiseIterator {
169public:
170 // VMergeIterator takes the ownership of input iterators
171 VMergeIterator(std::vector<RowwiseIteratorUPtr>&& iters, int sequence_id_idx, bool is_unique,
172 bool is_reverse, uint64_t* merged_rows)
173 : _origin_iters(std::move(iters)),
174 _sequence_id_idx(sequence_id_idx),
175 _is_unique(is_unique),
176 _is_reverse(is_reverse),
177 _merged_rows(merged_rows) {}
178
179 ~VMergeIterator() override {
180 while (!_merge_heap.empty()) {
181 auto ctx = _merge_heap.top();
182 _merge_heap.pop();
183 delete ctx;
184 }
185 }
186
187 Status init(const StorageReadOptions& opts) override;
188
189 Status next_batch(Block* block) override { return _next_batch(block); }
190 Status next_block_view(BlockView* block_view) override { return _next_batch(block_view); }
191
192 bool support_return_data_by_ref() override { return true; }
193
194 const Schema& schema() const override { return *_schema; }
195
196 Status current_block_row_locations(std::vector<RowLocation>* block_row_locations) override {
197 DCHECK(_record_rowids);
198 *block_row_locations = _block_row_locations;
199 return Status::OK();
200 }
201
202 bool update_profile(RuntimeProfile* profile) override {
203 if (!_origin_iters.empty()) {
204 return _origin_iters[0]->update_profile(profile);
205 }
206 return false;
207 }
208
209private:
210 int _get_size(Block* block) { return block->rows(); }
211 int _get_size(BlockView* block_view) { return block_view->size(); }
212
213 template <typename T>
214 Status _next_batch(T* block) {
215 if (UNLIKELY(_record_rowids)) {
216 _block_row_locations.resize(_block_row_max);
217 }
218 size_t row_idx = 0;
219 VMergeIteratorContext* pre_ctx = nullptr;
220 while (_get_size(block) < _block_row_max) {
221 if (_merge_heap.empty()) {
222 break;
223 }
224
225 auto ctx = _merge_heap.top();
226 _merge_heap.pop();
227
228 if (!ctx->need_skip()) {
229 ctx->add_cur_batch();
230 if (pre_ctx != ctx) {
231 if (pre_ctx) {
232 pre_ctx->copy_rows(block);
233 }
234 pre_ctx = ctx;
235 }
236 pre_ctx->set_pre_ctx_same(ctx);
237 if (UNLIKELY(_record_rowids)) {
238 _block_row_locations[row_idx] = ctx->current_row_location();
239 }
240 row_idx++;
241 if (ctx->is_cur_block_finished() || row_idx >= _block_row_max) {
242 // current block finished, ctx not advance
243 // so copy start_idx = (_index_in_block - _cur_batch_num + 1)
244 ctx->copy_rows(block, false);
245 pre_ctx = nullptr;
246 }
247 } else if (_merged_rows != nullptr) {
248 (*_merged_rows)++;
249 // need skip cur row, so flush rows in pre_ctx
250 if (pre_ctx) {
251 pre_ctx->copy_rows(block);
252 pre_ctx = nullptr;
253 }
254 }
255
256 RETURN_IF_ERROR(ctx->advance());
257 if (ctx->valid()) {
258 _merge_heap.push(ctx);
259 } else {
260 // Release ctx earlier to reduce resource consumed
261 delete ctx;
262 }
263 }
264 if (!_merge_heap.empty()) {
265 return Status::OK();
266 }
267 // Still last batch needs to be processed
268
269 if (UNLIKELY(_record_rowids)) {
270 _block_row_locations.resize(row_idx);
271 }
272
273 return Status::EndOfFile("no more data in segment");
274 }
