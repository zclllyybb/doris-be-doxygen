215 uint8_t* allocate_from_current_chunk(int64_t size, int alignment) {
216 // Manually ASAN poisoning is complicated and it is hard to make
217 // it work right. There are illustrated examples in
218 // http://blog.hostilefork.com/poison-memory-without-asan/.
219 //
220 // Stacks of use after poison do not provide enough information
221 // to resolve bug, while stacks of use afer free provide.
222 // https://github.com/google/sanitizers/issues/191
223 //
224 // We'd better implement a mempool using malloc/free directly,
225 // thus asan works natively. However we cannot do it in a short
226 // time, so we make manual poisoning work as much as possible.
227 // I refers to https://github.com/mcgov/asan_alignment_example.
228
229 ChunkInfo& info = chunks_[current_chunk_idx_];
230 int64_t aligned_allocated_bytes =
231 BitUtil::RoundUpToPowerOf2(info.allocated_bytes + DEFAULT_PADDING_SIZE, alignment);
232 if (aligned_allocated_bytes + size <= info.chunk.size) {
233 // Ensure the requested alignment is respected.
234 int64_t padding = aligned_allocated_bytes - info.allocated_bytes;
235 uint8_t* result = info.chunk.data + aligned_allocated_bytes;
236 ASAN_UNPOISON_MEMORY_REGION(result, size);
237 DCHECK_LE(info.allocated_bytes + size, info.chunk.size);
238 info.allocated_bytes += padding + size;
239 total_allocated_bytes_ += padding + size;
240 DCHECK_LE(current_chunk_idx_, chunks_.size() - 1);
241 return result;
242 }
243 return nullptr;
244 }

246 template <bool CHECK_LIMIT_FIRST>
247 uint8_t* ALWAYS_INLINE allocate(int64_t size, int alignment, bool free_old_chunks = false) {
248 DCHECK_GE(size, 0);
249 if (UNLIKELY(size == 0)) return reinterpret_cast<uint8_t*>(&k_zero_length_region_);
250
251 if (current_chunk_idx_ != -1) {
252 uint8_t* result = allocate_from_current_chunk(size, alignment);
253 if (result != nullptr) {
254 return result;
255 }
256 }
257
258 // If we couldn't allocate a new chunk, return nullptr. malloc() guarantees alignment
259 // of alignof(std::max_align_t), so we do not need to do anything additional to
260 // guarantee alignment.
261 //static_assert(
262 //INITIAL_CHUNK_SIZE >= config::FLAGS_MEMORY_MAX_ALIGNMENT, "Min chunk size too low");
263 if (UNLIKELY(!find_chunk(size + DEFAULT_PADDING_SIZE, CHECK_LIMIT_FIRST, free_old_chunks)))
264 return nullptr;
265
266 uint8_t* result = allocate_from_current_chunk(size, alignment);
267 return result;
268 }
269
270 template <bool CHECK_LIMIT_FIRST>
271 Status ALWAYS_INLINE allocate_safely(int64_t size, int alignment, uint8_t*& ret) {
272 uint8_t* result = allocate<CHECK_LIMIT_FIRST>(size, alignment);
273 if (result == nullptr) {
274 return Status::Error<ErrorCode::MEM_ALLOC_FAILED>();
275 }
276 ret = result;
277 return Status::OK();
278 }
