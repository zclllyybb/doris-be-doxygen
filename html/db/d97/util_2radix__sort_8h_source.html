90template <typename KeyBits>
91struct RadixSortFloatTransform {
93 static constexpr bool transform_is_simple = false;
94
95 static KeyBits forward(KeyBits x) {
96 return x ^
97 ((-(x >> (sizeof(KeyBits) * 8 - 1))) | (KeyBits(1) << (sizeof(KeyBits) * 8 - 1)));
98 }
99
100 static KeyBits backward(KeyBits x) {
101 return x ^
102 (((x >> (sizeof(KeyBits) * 8 - 1)) - 1) | (KeyBits(1) << (sizeof(KeyBits) * 8 - 1)));
103 }
104};

193/*
194 * To use RadixSort, you should define `Traits` to give out the information for sorting.
195 * `RadixSortFloatTraits` is a good example to refer to.
196 * Then you can run it as follows:
197 * RadixSort<YourTraits>::execute_lsd(arr, size);
198 *
199 * In particular, if you want to sort an array of numeric, you can use it easily as follows:
200 * radixSortLSD(array_of_numeric, array_size);
201 *
202 * See more use cases: be/test/util/radix_sort_test.cpp
203 *
204 */
205template <typename Traits>
206struct RadixSort {
207private:
208 using Element = typename Traits::Element;
209 using Key = typename Traits::Key;
210 using CountType = typename Traits::CountType;
211 using KeyBits = typename Traits::KeyBits;
212
213 // Use insertion sort if the size of the array is less than equal to this threshold
214 static constexpr size_t INSERTION_SORT_THRESHOLD = 64;
215
216 static constexpr size_t HISTOGRAM_SIZE = 1 << Traits::PART_SIZE_BITS;
217 static constexpr size_t PART_BITMASK = HISTOGRAM_SIZE - 1;
218 static constexpr size_t KEY_BITS = sizeof(Key) * 8;
219 static constexpr size_t NUM_PASSES =
220 (KEY_BITS + (Traits::PART_SIZE_BITS - 1)) / Traits::PART_SIZE_BITS;
221
222 static ALWAYS_INLINE KeyBits getPart(size_t N, KeyBits x) {
223 if (Traits::Transform::transform_is_simple) x = Traits::Transform::forward(x);
224
225 return (x >> (N * Traits::PART_SIZE_BITS)) & PART_BITMASK;
226 }
227
228 static KeyBits keyToBits(Key x) { return bit_cast<KeyBits>(x); }
229 static Key bitsToKey(KeyBits x) { return bit_cast<Key>(x); }
230
231public:
233 static void executeLSD(Element* arr, size_t size) {
235
237
239 CountType histograms[HISTOGRAM_SIZE * NUM_PASSES] = {0};
240
241 typename Traits::Allocator allocator;
242
244 Element* swap_buffer =
245 reinterpret_cast<Element*>(allocator.allocate(size * sizeof(Element)));
246
249 for (size_t i = 0; i < size; ++i) {
250 if (!Traits::Transform::transform_is_simple)
251 Traits::extractKey(arr[i]) = bitsToKey(
252 Traits::Transform::forward(keyToBits(Traits::extractKey(arr[i]))));
253
254 for (size_t pass = 0; pass < NUM_PASSES; ++pass)
255 ++histograms[pass * HISTOGRAM_SIZE +
256 getPart(pass, keyToBits(Traits::extractKey(arr[i])))];
257 }
258
259 {
261 size_t sums[NUM_PASSES] = {0};
262
263 for (size_t i = 0; i < HISTOGRAM_SIZE; ++i) {
264 for (size_t pass = 0; pass < NUM_PASSES; ++pass) {
265 size_t tmp = histograms[pass * HISTOGRAM_SIZE + i] + sums[pass];
266 histograms[pass * HISTOGRAM_SIZE + i] = sums[pass] - 1;
267 sums[pass] = tmp;
268 }
269 }
270 }
271
273 for (size_t pass = 0; pass < NUM_PASSES; ++pass) {
274 Element* writer = pass % 2 ? arr : swap_buffer;
275 Element* reader = pass % 2 ? swap_buffer : arr;
276
277 for (size_t i = 0; i < size; ++i) {
278 size_t pos = getPart(pass, keyToBits(Traits::extractKey(reader[i])));
279
281 auto& dest = writer[++histograms[pass * HISTOGRAM_SIZE + pos]];
282 dest = reader[i];
283
285 if (!Traits::Transform::transform_is_simple && pass == NUM_PASSES - 1)
286 Traits::extractKey(dest) = bitsToKey(
287 Traits::Transform::backward(keyToBits(Traits::extractKey(reader[i]))));
288 }
289 }
290
293 if (NUM_PASSES % 2) memcpy(arr, swap_buffer, size * sizeof(Element));
294
295 allocator.deallocate(swap_buffer, size * sizeof(Element));
296 }
297};
