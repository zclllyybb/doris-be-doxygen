28// Encode page use frame-of-reference coding
29template <FieldType Type>
30class FrameOfReferencePageBuilder : public PageBuilder {
31public:
32 explicit FrameOfReferencePageBuilder(const PageBuilderOptions& options)
33 : _options(options), _count(0), _finished(false) {
34 _encoder.reset(new ForEncoder<CppType>(&_buf));
35 }
36
37 bool is_page_full() override { return _encoder->len() >= _options.data_page_size; }
38
39 Status add(const uint8_t* vals, size_t* count) override {
40 DCHECK(!_finished);
41 if (*count == 0) {
42 return Status::OK();
43 }
44 auto new_vals = reinterpret_cast<const CppType*>(vals);
45 if (_count == 0) {
46 _first_val = *new_vals;
47 }
48 _encoder->put_batch(new_vals, *count);
49 _count += *count;
50 _last_val = new_vals[*count - 1];
51 return Status::OK();
52 }
53
54 OwnedSlice finish() override {
55 DCHECK(!_finished);
56 _finished = true;
57 _encoder->flush();
58 return _buf.build();
59 }
60
61 void reset() override {
62 _count = 0;
63 _finished = false;
64 _encoder->clear();
65 }
66
67 size_t count() const override { return _count; }
68
69 uint64_t size() const override { return _buf.size(); }
70
71 Status get_first_value(void* value) const override {
72 if (_count == 0) {
73 return Status::NotFound("page is empty");
74 }
75 memcpy(value, &_first_val, sizeof(CppType));
76 return Status::OK();
77 }
78
79 Status get_last_value(void* value) const override {
80 if (_count == 0) {
81 return Status::NotFound("page is empty");
82 }
83 memcpy(value, &_last_val, sizeof(CppType));
84 return Status::OK();
85 }
86
87private:
88 typedef typename TypeTraits<Type>::CppType CppType;
89 PageBuilderOptions _options;
90 size_t _count;
91 bool _finished;
92 std::unique_ptr<ForEncoder<CppType>> _encoder;
93 faststring _buf;
94 CppType _first_val;
95 CppType _last_val;
96};

98template <FieldType Type>
99class FrameOfReferencePageDecoder : public PageDecoder {
100public:
101 FrameOfReferencePageDecoder(Slice slice, const PageDecoderOptions& options)
102 : _parsed(false), _data(slice), _num_elements(0), _cur_index(0) {
103 _decoder.reset(new ForDecoder<CppType>((uint8_t*)_data.data, _data.size));
104 }
105
106 Status init() override {
107 CHECK(!_parsed);
108 bool result = _decoder->init();
109 if (result) {
110 _num_elements = _decoder->count();
111 _parsed = true;
112 return Status::OK();
113 } else {
114 return Status::Corruption("The frame of reference page metadata maybe broken");
115 }
116 }
117
118 Status seek_to_position_in_page(size_t pos) override {
119 DCHECK(_parsed) << "Must call init() firstly";
120 DCHECK_LE(pos, _num_elements)
121 << "Tried to seek to " << pos << " which is > number of elements (" << _num_elements
122 << ") in the block!";
123 // If the block is empty (e.g. the column is filled with nulls), there is no data to seek.
124 if (PREDICT_FALSE(_num_elements == 0)) {
125 return Status::OK();
126 }
127
128 int32_t skip_num = pos - _cur_index;
129 _decoder->skip(skip_num);
130 _cur_index = pos;
131 return Status::OK();
132 }
133
134 Status seek_at_or_after_value(const void* value, bool* exact_match) override {
135 DCHECK(_parsed) << "Must call init() firstly";
136 bool found = _decoder->seek_at_or_after_value(value, exact_match);
137 if (!found) {
138 return Status::NotFound("not found");
139 }
140 _cur_index = _decoder->current_index();
141 return Status::OK();
142 }
143
144 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override {
145 return Status::NotSupported("frame page not implement vec op now");
146 }
147
148 Status peek_next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override {
149 return Status::NotSupported("frame page not implement vec op now");
150 }
151
152 size_t count() const override { return _num_elements; }
153
154 size_t current_index() const override { return _cur_index; }
155
156private:
157 typedef typename TypeTraits<Type>::CppType CppType;
158
159 bool _parsed;
160 Slice _data;
161 uint32_t _num_elements;
162 size_t _cur_index;
163 std::unique_ptr<ForDecoder<CppType>> _decoder;
164};
