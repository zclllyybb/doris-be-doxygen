29// This is a utility class that aggregates counters from the kernel. These counters
30// come from different sources.
31// - perf counter syscall (/usr/include/linux/perf_event.h")
32// - /proc/self/io: io stats
33// - /proc/self/status: memory stats
34// The complexity here is that all these sources have data in a different and not
35// easy to get at format.
36//
37// A typical usage pattern would be:
38// PerfCounters counters;
39// counters.add_default_counters();
40// counters.snapshot("After Init");
41// <do your work>
42// counters.snapshot("After Work");
43// counters.PrettyPrint(cout);
44//
45// TODO: Expect PerfCounters to be refactored to ProcessState.
46

75 // Add the 'default' counters as ones to collect. Returns false if any of those
76 // counters are not available.
77 // Counters cannot be added after a snapshot has been taken.
78 bool add_default_counters();
79
80 // Add a specific counter to watch. Return false if that counter is not available.
81 // Counters cannot be added after a snapshot has been taken.
82 bool add_counter(Counter);
83
84 // Take a snapshot of all the counters and store it. The caller can specify a name
85 // for the snapshot.
86 void snapshot(const std::string& name);
87
88 // Returns the results of that snapshot
89 const std::vector<int64_t>* counters(int snapshot) const;
90
91 // Returns readable names for the added counters
92 const std::vector<std::string>* counter_names() const { return &_counter_names; }
93
94 // Prints out the names and results for all snapshots to 'out'
95 void pretty_print(std::ostream* out) const;

140 struct CounterData {
141 Counter counter;
142 DataSource source;
143 TUnit::type type;
144
145 // DataSource specific data. This is used to pull the counter values.
146 union {
147 // For SYS_PERF_COUNTER. File descriptor where the counter value is stored.
148 int fd;
149 // For PROC_SELF_IO. Line number from /proc/self/io file with this counter's value
150 int proc_io_line_number;
151 };
152 // For PROC_SELF_STATUS. Field name for counter
153 std::string proc_status_field;
154 };

161 // at the same time. This is useful to better correlate counter values.
162 int _group_fd;

165 static std::string _vm_rss_str;
