29class HdfsFsHandle {
30private:
31 // the number of referenced client
32 std::atomic<int> _ref_cnt;
33 // HdfsFsCache try to remove the oldest handler when the cache is full
34 std::atomic<uint64_t> _last_access_time;
35 // Client will set invalid if error thrown, and HdfsFsCache will not reuse this handler
36 std::atomic<bool> _invalid;
37
38 uint64_t _now() {
39 return std::chrono::duration_cast<std::chrono::milliseconds>(
40 std::chrono::system_clock::now().time_since_epoch())
41 .count();
42 }
43
44public:
45 hdfsFS hdfs_fs;
46 // When cache is full, and all handlers are in use, HdfsFsCache will return an uncached handler.
47 // Client should delete the handler in such case.
48 const bool from_cache;
49
50 HdfsFsHandle(hdfsFS fs, bool cached)
51 : _ref_cnt(0), _last_access_time(0), _invalid(false), hdfs_fs(fs), from_cache(cached) {}
52
53 ~HdfsFsHandle() {
54 DCHECK(_ref_cnt == 0);
55 if (hdfs_fs != nullptr) {
56 // Even if there is an error, the resources associated with the hdfsFS will be freed.
57 hdfsDisconnect(hdfs_fs);
58 }
59 hdfs_fs = nullptr;
60 }
61
62 int64_t last_access_time() { return _last_access_time; }
63
64 void inc_ref() {
65 _ref_cnt++;
66 _last_access_time = _now();
67 }
68
69 void dec_ref() {
70 _ref_cnt--;
71 _last_access_time = _now();
72 }
73
74 int ref_cnt() { return _ref_cnt; }
75
76 bool invalid() { return _invalid; }
77
78 void set_invalid() { _invalid = true; }
79};
