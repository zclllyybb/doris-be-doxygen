137 template <typename Func>
138 void ALWAYS_INLINE merge_to_via_emplace(Self& that, Func&& func) {
139 if (this->m0.hasZero() && that.m0.hasZero())
140 func(that.m0.zero_value()->get_mapped(), this->m0.zero_value()->get_mapped(), false);
141 else if (this->m0.hasZero()) {
142 that.m0.setHasZero();
143 func(that.m0.zero_value()->get_mapped(), this->m0.zero_value()->get_mapped(), true);
144 }
145 this->m1.merge_to_via_emplace(that.m1, func);
146 this->m2.merge_to_via_emplace(that.m2, func);
147 this->m3.merge_to_via_emplace(that.m3, func);
148 this->ms.merge_to_via_emplace(that.ms, func);
149 }
150
156 template <typename Func>
157 void ALWAYS_INLINE merge_to_via_find(Self& that, Func&& func) {
158 if (this->m0.size() && that.m0.size())
159 func(that.m0.zero_value()->get_mapped(), this->m0.zero_value()->get_mapped(), true);
160 else if (this->m0.size())
161 func(this->m0.zero_value()->get_mapped(), this->m0.zero_value()->get_mapped(), false);
162 this->m1.merge_to_via_find(that.m1, func);
163 this->m2.merge_to_via_find(that.m2, func);
164 this->m3.merge_to_via_find(that.m3, func);
165 this->ms.merge_to_via_find(that.ms, func);
166 }

168 TMapped& ALWAYS_INLINE operator[](const Key& x) {
169 LookupResult it;
170 bool inserted;
171 this->emplace(x, it, inserted);
172 if (inserted) new (&it->get_mapped()) TMapped();
173
174 return it->get_mapped();
175 }

198 }

200 template <typename Func>
201 void ALWAYS_INLINE for_each_mapped(Func&& func) {
202 if (this->m0.size()) func(this->m0.zero_value()->get_second());
203 for (auto& v : this->m1) func(v.get_second());
204 for (auto& v : this->m2) func(v.get_second());
205 for (auto& v : this->m3) func(v.get_second());
206 for (auto& v : this->ms) func(v.get_second());
207 }
