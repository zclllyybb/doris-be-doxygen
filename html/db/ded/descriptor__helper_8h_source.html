103class TTupleDescriptorBuilder {
104public:
105 TTupleDescriptorBuilder& add_slot(const TSlotDescriptor& slot_desc) {
106 _slot_descs.push_back(slot_desc);
107 return *this;
108 }
109 void build(TDescriptorTableBuilder* tb) {
110 // build slot desc
111 _tuple_id = tb->next_tuple_id();
112 int num_nullables = 0;
113 for (int i = 0; i < _slot_descs.size(); ++i) {
114 auto& slot_desc = _slot_descs[i];
115 if (slot_desc.nullIndicatorByte >= 0) {
116 num_nullables++;
117 }
118 }
119 int null_bytes = (num_nullables + 7) / 8;
120 int null_offset = 0;
121 for (int i = 0; i < _slot_descs.size(); ++i) {
122 auto& slot_desc = _slot_descs[i];
123 slot_desc.id = tb->next_slot_id();
124 slot_desc.parent = _tuple_id;
125 if (slot_desc.nullIndicatorByte >= 0) {
126 slot_desc.nullIndicatorBit = null_offset % 8;
127 slot_desc.nullIndicatorByte = null_offset / 8;
128 null_offset++;
129 } else {
130 slot_desc.nullIndicatorByte = 0;
131 slot_desc.nullIndicatorBit = -1;
132 }
133 slot_desc.slotIdx = i;
134 }
135
136 _tuple_desc.id = _tuple_id;
137 // Useless not set it.
138 _tuple_desc.byteSize = 0;
139 _tuple_desc.numNullBytes = null_bytes;
140 _tuple_desc.numNullSlots = _slot_descs.size();
141
142 tb->add_slots(_slot_descs);
143 tb->add_tuple(_tuple_desc);
144 }
145
146private:
147 TTupleId _tuple_id;
148 std::vector<TSlotDescriptor> _slot_descs;
149 TTupleDescriptor _tuple_desc;
150};
