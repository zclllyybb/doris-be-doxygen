29// this class used in a hash join node
30// Provide a unified interface for other classes
31template <typename ExprCtxType>
32class RuntimeFilterSlotsBase {
33public:
34 RuntimeFilterSlotsBase(const std::vector<ExprCtxType*>& prob_expr_ctxs,
35 const std::vector<ExprCtxType*>& build_expr_ctxs,
36 const std::vector<TRuntimeFilterDesc>& runtime_filter_descs)
37 : _probe_expr_context(prob_expr_ctxs),
38 _build_expr_context(build_expr_ctxs),
39 _runtime_filter_descs(runtime_filter_descs) {}
40
41 Status init(RuntimeState* state, int64_t hash_table_size) {
42 DCHECK(_probe_expr_context.size() == _build_expr_context.size());
43
44 // runtime filter effect strategy
45 // 1. we will ignore IN filter when hash_table_size is too big
46 // 2. we will ignore BLOOM filter and MinMax filter when hash_table_size
47 // is too small and IN filter has effect
48
49 std::map<int, bool> has_in_filter;
50
51 auto ignore_local_filter = [state](int filter_id) {
52 IRuntimeFilter* consumer_filter = nullptr;
53 state->runtime_filter_mgr()->get_consume_filter(filter_id, &consumer_filter);
54 DCHECK(consumer_filter != nullptr);
55 consumer_filter->set_ignored();
56 consumer_filter->signal();
57 };
58
59 auto ignore_remote_filter = [](IRuntimeFilter* runtime_filter, std::string& msg) {
60 runtime_filter->set_ignored();
61 runtime_filter->set_ignored_msg(msg);
62 runtime_filter->publish();
63 runtime_filter->publish_finally();
64 };
65
66 // ordered vector: IN, IN_OR_BLOOM, others.
67 // so we can ignore other filter if IN Predicate exists.
68 std::vector<TRuntimeFilterDesc> sorted_runtime_filter_descs(_runtime_filter_descs);
69 auto compare_desc = [](TRuntimeFilterDesc& d1, TRuntimeFilterDesc& d2) {
70 if (d1.type == d2.type) {
71 return false;
72 } else if (d1.type == TRuntimeFilterType::IN) {
73 return true;
74 } else if (d2.type == TRuntimeFilterType::IN) {
75 return false;
76 } else if (d1.type == TRuntimeFilterType::IN_OR_BLOOM) {
77 return true;
78 } else if (d2.type == TRuntimeFilterType::IN_OR_BLOOM) {
79 return false;
80 } else {
81 return d1.type < d2.type;
82 }
83 };
84 std::sort(sorted_runtime_filter_descs.begin(), sorted_runtime_filter_descs.end(),
85 compare_desc);
86
87 for (auto& filter_desc : sorted_runtime_filter_descs) {
88 IRuntimeFilter* runtime_filter = nullptr;
89 RETURN_IF_ERROR(state->runtime_filter_mgr()->get_producer_filter(filter_desc.filter_id,
90 &runtime_filter));
91 DCHECK(runtime_filter != nullptr);
92 DCHECK(runtime_filter->expr_order() >= 0);
93 DCHECK(runtime_filter->expr_order() < _probe_expr_context.size());
94
95 // do not create 'in filter' when hash_table size over limit
96 auto max_in_num = state->runtime_filter_max_in_num();
97 bool over_max_in_num = (hash_table_size >= max_in_num);
98
99 bool is_in_filter = (runtime_filter->type() == RuntimeFilterType::IN_FILTER);
100
101 if (over_max_in_num &&
102 runtime_filter->type() == RuntimeFilterType::IN_OR_BLOOM_FILTER) {
103 runtime_filter->change_to_bloom_filter();
104 }
105
106 // Note:
107 // In the case that exist *remote target* and in filter and other filter,
108 // we must merge other filter whatever in filter is over the max num in current node,
109 // because:
110 // case 1: (in filter >= max num) in current node, so in filter will be ignored,
111 // and then other filter can be used
112 // case 2: (in filter < max num) in current node, we don't know whether the in filter
113 // will be ignored in merge node, so we must transfer other filter to merge node
114 if (!runtime_filter->has_remote_target()) {
115 bool exists_in_filter = has_in_filter[runtime_filter->expr_order()];
116 if (is_in_filter && over_max_in_num) {
117 VLOG_DEBUG << "fragment instance " << print_id(state->fragment_instance_id())
118 << " ignore runtime filter(in filter id " << filter_desc.filter_id
119 << ") because: in_num(" << hash_table_size << ") >= max_in_num("
120 << max_in_num << ")";
121 ignore_local_filter(filter_desc.filter_id);
122 continue;
123 } else if (!is_in_filter && exists_in_filter) {
124 // do not create 'bloom filter' and 'minmax filter' when 'in filter' has created
125 // because in filter is exactly filter, so it is enough to filter data
126 VLOG_DEBUG << "fragment instance " << print_id(state->fragment_instance_id())
127 << " ignore runtime filter(" << to_string(runtime_filter->type())
128 << " id " << filter_desc.filter_id
129 << ") because: already exists in filter";
130 ignore_local_filter(filter_desc.filter_id);
131 continue;
132 }
133 } else if (is_in_filter && over_max_in_num) {
134#ifdef VLOG_DEBUG_IS_ON
135 std::string msg = fmt::format(
136 "fragment instance {} ignore runtime filter(in filter id {}) because: "
137 "in_num({}) >= max_in_num({})",
138 print_id(state->fragment_instance_id()), filter_desc.filter_id,
139 hash_table_size, max_in_num);
140 ignore_remote_filter(runtime_filter, msg);
141#else
142 ignore_remote_filter(runtime_filter, "ignored");
143#endif
144 continue;
145 }
146
147 if ((runtime_filter->type() == RuntimeFilterType::IN_FILTER) ||
148 (runtime_filter->type() == RuntimeFilterType::IN_OR_BLOOM_FILTER &&
149 !over_max_in_num)) {
150 has_in_filter[runtime_filter->expr_order()] = true;
151 }
152 _runtime_filters[runtime_filter->expr_order()].push_back(runtime_filter);
153 }
154
155 return Status::OK();
156 }
157
158 void insert(std::unordered_map<const vectorized::Block*, std::vector<int>>& datas) {
159 for (int i = 0; i < _build_expr_context.size(); ++i) {
160 auto iter = _runtime_filters.find(i);
161 if (iter == _runtime_filters.end()) {
162 continue;
163 }
164
165 int result_column_id = _build_expr_context[i]->get_last_result_column_id();
166 for (auto it : datas) {
167 auto& column = it.first->get_by_position(result_column_id).column;
168
169 if (auto* nullable =
170 vectorized::check_and_get_column<vectorized::ColumnNullable>(*column)) {
171 auto& column_nested = nullable->get_nested_column_ptr();
172 auto& column_nullmap = nullable->get_null_map_column_ptr();
173 std::vector<int> indexs;
174 for (int row_num : it.second) {
175 if (assert_cast<const vectorized::ColumnUInt8*>(column_nullmap.get())
176 ->get_bool(row_num)) {
177 continue;
178 }
179 indexs.push_back(row_num);
180 }
181 for (auto filter : iter->second) {
182 filter->insert_batch(column_nested, indexs);
183 }
184
185 } else {
186 for (auto filter : iter->second) {
187 filter->insert_batch(column, it.second);
188 }
189 }
190 }
191 }
192 }
193
194 // should call this method after insert
195 void ready_for_publish() {
196 for (auto& pair : _runtime_filters) {
197 for (auto filter : pair.second) {
198 filter->ready_for_publish();
199 }
200 }
201 }
202 // publish runtime filter
203 void publish() {
204 for (int i = 0; i < _probe_expr_context.size(); ++i) {
205 auto iter = _runtime_filters.find(i);
206 if (iter != _runtime_filters.end()) {
207 for (auto filter : iter->second) {
208 filter->publish();
209 }
210 }
211 }
212 for (auto& pair : _runtime_filters) {
213 for (auto filter : pair.second) {
214 filter->publish_finally();
215 }
216 }
217 }
218
219 void copy_to_shared_context(vectorized::SharedHashTableContextPtr& context) {
220 for (auto& it : _runtime_filters) {
221 for (auto& filter : it.second) {
222 auto& target = context->runtime_filters[filter->filter_id()];
223 filter->copy_to_shared_context(target);
224 }
225 }
226 }
227
228 Status copy_from_shared_context(vectorized::SharedHashTableContextPtr& context) {
229 for (auto& it : _runtime_filters) {
230 for (auto& filter : it.second) {
231 auto filter_id = filter->filter_id();
232 auto ret = context->runtime_filters.find(filter_id);
233 if (ret == context->runtime_filters.end()) {
234 return Status::Aborted("invalid runtime filter id: {}", filter_id);
235 }
236 filter->copy_from_shared_context(ret->second);
237 }
238 }
239 return Status::OK();
240 }
241
242 bool empty() { return !_runtime_filters.size(); }
243
244private:
245 const std::vector<ExprCtxType*>& _probe_expr_context;
246 const std::vector<ExprCtxType*>& _build_expr_context;
247 const std::vector<TRuntimeFilterDesc>& _runtime_filter_descs;
248 // prob_contition index -> [IRuntimeFilter]
249 std::map<int, std::list<IRuntimeFilter*>> _runtime_filters;
250};
