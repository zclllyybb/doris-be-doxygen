78template <PrimitiveType Type, PredicateType PT>
79class InListPredicateBase : public ColumnPredicate {
80public:
81 using T = typename PredicatePrimitiveTypeTraits<Type>::PredicateFieldType;
82 template <typename ConditionType, typename ConvertFunc>
83 InListPredicateBase(uint32_t column_id, const ConditionType& conditions,
84 const ConvertFunc& convert, bool is_opposite = false,
85 const TabletColumn* col = nullptr, MemPool* pool = nullptr)
86 : ColumnPredicate(column_id, is_opposite),
87 _values(new phmap::flat_hash_set<T>()),
88 _min_value(type_limit<T>::max()),
89 _max_value(type_limit<T>::min()) {
90 for (const auto& condition : conditions) {
91 T tmp;
92 if constexpr (Type == TYPE_STRING || Type == TYPE_CHAR) {
93 tmp = convert(*col, condition, pool);
94 } else if constexpr (Type == TYPE_DECIMAL32 || Type == TYPE_DECIMAL64 ||
95 Type == TYPE_DECIMAL128I) {
96 tmp = convert(*col, condition);
97 } else {
98 tmp = convert(condition);
99 }
100 _values->insert(tmp);
101 _update_min_max(tmp);
102 }
103 }

105 InListPredicateBase(uint32_t column_id, const std::shared_ptr<HybridSetBase>& hybrid_set,
106 size_t char_length = 0)
107 : ColumnPredicate(column_id, false),
108 _min_value(type_limit<T>::max()),
109 _max_value(type_limit<T>::min()) {
110 using HybridSetType = std::conditional_t<is_string_type(Type), StringSet, HybridSet<Type>>;
111
112 CHECK(hybrid_set != nullptr);
113
114 if constexpr (is_string_type(Type) || Type == TYPE_DECIMALV2 || is_date_type(Type)) {
115 _values = new phmap::flat_hash_set<T>();
116 auto values = ((HybridSetType*)hybrid_set.get())->get_inner_set();
117
118 if constexpr (is_string_type(Type)) {
119 // values' type is "phmap::flat_hash_set<std::string>"
120 for (const std::string& value : *values) {
121 StringRef sv = value;
122 if constexpr (Type == TYPE_CHAR) {
123 _temp_datas.push_back("");
124 _temp_datas.back().resize(std::max(char_length, value.size()));
125 memcpy(_temp_datas.back().data(), value.data(), value.size());
126 sv = _temp_datas.back();
127 }
128 _values->insert(sv);
129 }
130 } else if constexpr (Type == TYPE_DECIMALV2) {
131 for (auto& value : *values) {
132 _values->insert({value.int_value(), value.frac_value()});
133 }
134 } else if constexpr (Type == TYPE_DATE) {
135 for (auto& value : *values) {
136 _values->insert(value.to_olap_date());
137 }
138 } else if constexpr (Type == TYPE_DATETIME) {
139 for (auto& value : *values) {
140 _values->insert(value.to_olap_datetime());
141 }
142 } else {
143 CHECK(Type == TYPE_DATETIMEV2 || Type == TYPE_DATEV2);
144 for (auto& value : *values) {
145 _values->insert(T(value));
146 }
147 }
148 } else {
149 should_delete = false;
150 _values = ((HybridSetType*)hybrid_set.get())->get_inner_set();
151 }
152
153 for (auto& value : *_values) {
154 _update_min_max(value);
155 }
156 }

158 ~InListPredicateBase() override {
159 if (should_delete) {
160 delete _values;
161 }

177 Status evaluate(BitmapIndexIterator* iterator, uint32_t num_rows,
178 roaring::Roaring* result) const override {
179 if (iterator == nullptr) {
180 return Status::OK();
181 }
182 if (iterator->has_null_bitmap()) {
183 roaring::Roaring null_bitmap;
184 RETURN_IF_ERROR(iterator->read_null_bitmap(&null_bitmap));
185 *result -= null_bitmap;
186 }
187 roaring::Roaring indices;
188 for (auto value : *_values) {
189 bool exact_match;
190 Status s = iterator->seek_dictionary(&value, &exact_match);
191 rowid_t seeked_ordinal = iterator->current_ordinal();
192 if (!s.is<ErrorCode::NOT_FOUND>()) {
193 if (!s.ok()) {
194 return s;
195 }
196 if (exact_match) {
197 roaring::Roaring index;
198 RETURN_IF_ERROR(iterator->read_bitmap(seeked_ordinal, &index));
199 indices |= index;
200 }
201 }
202 }
203
204 if constexpr (PT == PredicateType::IN_LIST) {
205 *result &= indices;
206 } else {
207 *result -= indices;
208 }
209
210 return Status::OK();
211 }

213 Status evaluate(const Schema& schema, InvertedIndexIterator* iterator, uint32_t num_rows,
214 roaring::Roaring* result) const override {
215 if (iterator == nullptr) {
216 return Status::OK();
217 }
218 auto column_desc = schema.column(_column_id);
219 std::string column_name = column_desc->name();
220 roaring::Roaring indices;
221 for (auto value : *_values) {
222 InvertedIndexQueryType query_type = InvertedIndexQueryType::EQUAL_QUERY;
223 roaring::Roaring index;
224 RETURN_IF_ERROR(iterator->read_from_inverted_index(column_name, &value, query_type,
225 num_rows, &index));
226 indices |= index;
227 }
228 if constexpr (PT == PredicateType::IN_LIST) {
229 *result &= indices;
230 } else {
231 *result -= indices;
232 }
233 return Status::OK();
234 }

236 uint16_t evaluate(const vectorized::IColumn& column, uint16_t* sel,
237 uint16_t size) const override {
238 if (column.is_nullable()) {
239 auto* nullable_col =
240 vectorized::check_and_get_column<vectorized::ColumnNullable>(column);
241 auto& null_bitmap = reinterpret_cast<const vectorized::ColumnUInt8&>(
242 nullable_col->get_null_map_column())
243 .get_data();
244 auto& nested_col = nullable_col->get_nested_column();
245
246 if (_opposite) {
247 return _base_evaluate<true, true>(&nested_col, &null_bitmap, sel, size);
248 } else {
249 return _base_evaluate<true, false>(&nested_col, &null_bitmap, sel, size);
250 }
251 } else {
252 if (_opposite) {
253 return _base_evaluate<false, true>(&column, nullptr, sel, size);
254 } else {
255 return _base_evaluate<false, false>(&column, nullptr, sel, size);
256 }
257 }

261 void _evaluate_bit(const vectorized::IColumn& column, const uint16_t* sel, uint16_t size,
262 bool* flags) const {

271 if (_opposite) {
272 return _base_evaluate_bit<true, true, is_and>(&nested_col, &null_bitmap, sel, size,

280 return _base_evaluate_bit<false, true, is_and>(&column, nullptr, sel, size, flags);

282 return _base_evaluate_bit<false, false, is_and>(&column, nullptr, sel, size, flags);

297 bool evaluate_and(const std::pair<WrapperField*, WrapperField*>& statistic) const override {
298 if (statistic.first->is_null()) {
299 return true;
300 }
301 if constexpr (PT == PredicateType::IN_LIST) {
302 if constexpr (Type == TYPE_DATE) {
303 T tmp_min_uint32_value = 0;
304 memcpy((char*)(&tmp_min_uint32_value), statistic.first->cell_ptr(),
305 sizeof(uint24_t));
306 T tmp_max_uint32_value = 0;
307 memcpy((char*)(&tmp_max_uint32_value), statistic.second->cell_ptr(),
308 sizeof(uint24_t));
309 return tmp_min_uint32_value <= _max_value && tmp_max_uint32_value >= _min_value;
310 } else {
311 return _get_zone_map_value<T>(statistic.first->cell_ptr()) <= _max_value &&
312 _get_zone_map_value<T>(statistic.second->cell_ptr()) >= _min_value;
313 }
314 } else {
315 return true;
316 }
317 }
318
319 bool evaluate_del(const std::pair<WrapperField*, WrapperField*>& statistic) const override {
320 if (statistic.first->is_null() || statistic.second->is_null()) {
321 return false;
322 }
323 if constexpr (PT == PredicateType::NOT_IN_LIST) {
324 if constexpr (Type == TYPE_DATE) {
325 T tmp_min_uint32_value = 0;
326 memcpy((char*)(&tmp_min_uint32_value), statistic.first->cell_ptr(),
327 sizeof(uint24_t));
328 T tmp_max_uint32_value = 0;
329 memcpy((char*)(&tmp_max_uint32_value), statistic.second->cell_ptr(),
330 sizeof(uint24_t));
331 return tmp_min_uint32_value > _max_value || tmp_max_uint32_value < _min_value;
332 } else {
333 return _get_zone_map_value<T>(statistic.first->cell_ptr()) > _max_value ||
334 _get_zone_map_value<T>(statistic.second->cell_ptr()) < _min_value;
335 }
336 } else {
337 return false;
338 }

341 bool evaluate_and(const segment_v2::BloomFilter* bf) const override {
342 if constexpr (PT == PredicateType::IN_LIST) {
343 for (auto value : *_values) {
344 if constexpr (std::is_same_v<T, StringRef>) {
345 if (bf->test_bytes(value.data, value.size)) {
346 return true;
347 }
348 } else if constexpr (Type == TYPE_DATE) {
349 if (bf->test_bytes(reinterpret_cast<char*>(&value), sizeof(uint24_t))) {
350 return true;
351 }
352 } else {
353 if (bf->test_bytes(reinterpret_cast<char*>(&value), sizeof(value))) {
354 return true;
355 }
356 }
357 }
358 return false;
359 } else {
360 LOG(FATAL) << "Bloom filter is not supported by predicate type.";
361 return true;
362 }
363 }

368 template <typename LeftT, typename RightT>
369 bool _operator(const LeftT& lhs, const RightT& rhs) const {
370 if constexpr (Type == TYPE_BOOLEAN) {
371 DCHECK(_values->size() == 2);
372 return PT == PredicateType::IN_LIST;
373 } else if constexpr (PT == PredicateType::IN_LIST) {
374 return lhs != rhs;
375 }
376 return lhs == rhs;

379 template <bool is_nullable, bool is_opposite>
380 uint16_t _base_evaluate(const vectorized::IColumn* column,
381 const vectorized::PaddedPODArray<vectorized::UInt8>* null_map,
382 uint16_t* sel, uint16_t size) const {
383 uint16_t new_size = 0;
384
385 if (column->is_column_dictionary()) {
386 if constexpr (std::is_same_v<T, StringRef>) {
387 auto* nested_col_ptr = vectorized::check_and_get_column<
388 vectorized::ColumnDictionary<vectorized::Int32>>(column);
389 auto& data_array = nested_col_ptr->get_data();
390 auto segid = column->get_rowset_segment_id();
391 DCHECK((segid.first.hi | segid.first.mi | segid.first.lo) != 0);
392 auto& value_in_dict_flags = _segment_id_to_value_in_dict_flags[segid];
393 if (value_in_dict_flags.empty()) {
394 nested_col_ptr->find_codes(*_values, value_in_dict_flags);
395 }
396
397 CHECK(value_in_dict_flags.size() == nested_col_ptr->dict_size())
398 << "value_in_dict_flags.size()!=nested_col_ptr->dict_size(), "
399 << value_in_dict_flags.size() << " vs " << nested_col_ptr->dict_size()
400 << " rowsetid=" << segid.first << " segmentid=" << segid.second
401 << "dict_info" << nested_col_ptr->dict_debug_string();
402
403 for (uint16_t i = 0; i < size; i++) {
404 uint16_t idx = sel[i];
405 if constexpr (is_nullable) {
406 if ((*null_map)[idx]) {
407 if constexpr (is_opposite) {
408 sel[new_size++] = idx;
409 }
410 continue;
411 }
412 }
413
414 if constexpr (is_opposite != (PT == PredicateType::IN_LIST)) {
415 if (value_in_dict_flags[data_array[idx]]) {
416 sel[new_size++] = idx;
417 }
418 } else {
419 if (!value_in_dict_flags[data_array[idx]]) {
420 sel[new_size++] = idx;
421 }
422 }
423 }
424 } else {
425 LOG(FATAL) << "column_dictionary must use StringRef predicate.";
426 }
427 } else {
428 auto* nested_col_ptr = vectorized::check_and_get_column<
429 vectorized::PredicateColumnType<PredicateEvaluateType<Type>>>(column);
430 auto& data_array = nested_col_ptr->get_data();
431
432 for (uint16_t i = 0; i < size; i++) {
433 uint16_t idx = sel[i];
434 if constexpr (is_nullable) {
435 if ((*null_map)[idx]) {
436 if constexpr (is_opposite) {
437 sel[new_size++] = idx;
438 }
439 continue;
440 }
441 }
442
443 if constexpr (!is_opposite) {
444 if (_operator(_values->find(reinterpret_cast<const T&>(data_array[idx])),
445 _values->end())) {
446 sel[new_size++] = idx;
447 }
448 } else {
449 if (!_operator(_values->find(reinterpret_cast<const T&>(data_array[idx])),
450 _values->end())) {
451 sel[new_size++] = idx;
452 }
453 }
454 }
455 }
456
457 return new_size;
458 }

460 template <bool is_nullable, bool is_opposite, bool is_and>
461 void _base_evaluate_bit(const vectorized::IColumn* column,
462 const vectorized::PaddedPODArray<vectorized::UInt8>* null_map,
463 const uint16_t* sel, uint16_t size, bool* flags) const {
464 if (column->is_column_dictionary()) {
465 if constexpr (std::is_same_v<T, StringRef>) {
466 auto* nested_col_ptr = vectorized::check_and_get_column<
467 vectorized::ColumnDictionary<vectorized::Int32>>(column);
468 auto& data_array = nested_col_ptr->get_data();
469 auto& value_in_dict_flags =
470 _segment_id_to_value_in_dict_flags[column->get_rowset_segment_id()];
471 if (value_in_dict_flags.empty()) {
472 nested_col_ptr->find_codes(*_values, value_in_dict_flags);
473 }
474
475 for (uint16_t i = 0; i < size; i++) {
476 if (is_and ^ flags[i]) {
477 continue;
478 }
479
480 uint16_t idx = sel[i];
481 if constexpr (is_nullable) {
482 if ((*null_map)[idx]) {
483 if (is_and ^ is_opposite) {
484 flags[i] = !is_and;
485 }
486 continue;
487 }
488 }
489
490 if constexpr (is_opposite != (PT == PredicateType::IN_LIST)) {
491 if (is_and ^ value_in_dict_flags[data_array[idx]]) {
492 flags[i] = !is_and;
493 }
494 } else {
495 if (is_and ^ !value_in_dict_flags[data_array[idx]]) {
496 flags[i] = !is_and;
497 }
498 }
499 }
500 } else {
501 LOG(FATAL) << "column_dictionary must use StringRef predicate.";
502 }
503 } else {
504 auto* nested_col_ptr = vectorized::check_and_get_column<
505 vectorized::PredicateColumnType<PredicateEvaluateType<Type>>>(column);
506 auto& data_array = nested_col_ptr->get_data();
507
508 for (uint16_t i = 0; i < size; i++) {
509 if (is_and ^ flags[i]) {
510 continue;
511 }
512 uint16_t idx = sel[i];
513 if constexpr (is_nullable) {
514 if ((*null_map)[idx]) {
515 if (is_and ^ is_opposite) {
516 flags[i] = !is_and;
517 }
518 continue;
519 }
520 }
521
522 if constexpr (!is_opposite) {
523 if (is_and ^
524 _operator(_values->find(reinterpret_cast<const T&>(data_array[idx])),
525 _values->end())) {
526 flags[i] = !is_and;
527 }
528 } else {
529 if (is_and ^
530 !_operator(_values->find(reinterpret_cast<const T&>(data_array[idx])),
531 _values->end())) {
532 flags[i] = !is_and;
533 }
534 }
535 }
536 }
