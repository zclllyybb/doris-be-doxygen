15// This struct is not part of the public API of this module; clients may not
16// use it. (However, it's exported via BASE_EXPORT because clients implicitly
17// do use it at link time by inlining these functions.)
18// Features of this x86. Values may not be correct before main() is run,
19// but are set conservatively.
20struct AtomicOps_x86CPUFeatureStruct {
21 bool has_amd_lock_mb_bug; // Processor has AMD memory-barrier bug; do lfence
22 // after acquire compare-and-swap.
23 bool has_sse2; // Processor has SSE2.
24};
25BASE_EXPORT extern struct AtomicOps_x86CPUFeatureStruct AtomicOps_Internalx86CPUFeatures;

40inline Atomic32 NoBarrier_CompareAndSwap(volatile Atomic32* ptr, Atomic32 old_value,
41 Atomic32 new_value) {
42 Atomic32 cmp = old_value;
43 __tsan_atomic32_compare_exchange_strong(ptr, &cmp, new_value, __tsan_memory_order_relaxed,
44 __tsan_memory_order_relaxed);
45 return cmp;
46}
