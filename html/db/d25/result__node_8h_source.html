52class PartitionRowBatch {
53public:
54 PartitionRowBatch(int64 partition_key)
55 : _partition_key(partition_key), _cache_value(nullptr), _data_size(0) {}
56
57 ~PartitionRowBatch() {}
58
59 void set_row_batch(const PCacheValue& value);
60 bool is_hit_cache(const PCacheParam& param);
61 void clear();
62
63 int64 get_partition_key() const { return _partition_key; }
64
65 PCacheValue* get_value() { return _cache_value; }
66
67 size_t get_data_size() { return _data_size; }
68
69 const CacheStat* get_stat() const { return &_cache_stat; }
70
71private:
72 bool check_match(const PCacheParam& req_param) {
73 if (req_param.partition_key() != _partition_key) {
74 return false;
75 }
76 if (req_param.last_version() > _cache_value->param().last_version()) {
77 return false;
78 }
79 if (req_param.last_version_time() > _cache_value->param().last_version_time()) {
80 return false;
81 }
82 return true;
83 }
84
85 bool check_newer(const PCacheParam& up_param) {
86 //for init data of sql cache
87 if (up_param.last_version() == 0 || up_param.last_version_time() == 0) {
88 return true;
89 }
90 if (up_param.last_version_time() > _cache_value->param().last_version_time()) {
91 return true;
92 }
93 if (up_param.last_version() > _cache_value->param().last_version()) {
94 return true;
95 }
96 return false;
97 }
98
99private:
100 int64 _partition_key;
101 PCacheValue* _cache_value;
102 size_t _data_size;
103 CacheStat _cache_stat;
104};
