27// Utility class for common local file system operations such as file creation and
28// deletion. This class should NOT be used to read or write data
29// Errors are indicated by the status code RUNTIME_ERROR, and are not
30// handled via exceptions.
31class FileSystemUtil {
32public:
33 // Create the specified directory and any ancestor directories that do not exist yet.
34 // The directory and its contents are destroyed if it already exists.
35 // Returns Status::OK() if successful, or a runtime error with a message otherwise.
36 static Status create_directory(const std::string& directory);
37
38 // Create a file at the specified path.
39 static Status create_file(const std::string& file_path);
40
41 // Resize a file to a specified length - uses unistd truncate().
42 static Status resize_file(const std::string& file_path, int64_t trunc_len);
43
44 // Remove the specified paths and their enclosing files/directories.
45 static Status remove_paths(const std::vector<std::string>& directories);
46
47 // Verify that the specified path is an existing directory.
48 // Returns Status::OK() if it is, or a runtime error with a message otherwise.
49 static Status verify_is_directory(const std::string& directory_path);
50
51 // Returns the space available on the file system containing 'directory_path'
52 // in 'available_bytes'
53 static Status get_space_available(const std::string& directory_path, uint64_t* available_bytes);
54
55 // Returns the currently allowed maximum of possible file descriptors. In case of an
56 // error returns 0.
57 static uint64_t max_num_file_handles();
58
59 static bool contain_path(const std::string& parent_path, const std::string& sub_path);
60};
61
62} // namespace doris
