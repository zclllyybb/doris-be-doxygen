65 template <bool is_and>
66 void _evaluate_vec(const vectorized::IColumn& column, uint16_t size, bool* flags) const {
67 if (column.is_nullable()) {
68 auto* nullable_col =
69 vectorized::check_and_get_column<vectorized::ColumnNullable>(column);
70 auto& null_map_data = nullable_col->get_null_map_column().get_data();
71 auto& nested_col = nullable_col->get_nested_column();
72 if (nested_col.is_column_dictionary()) {
73 auto* nested_col_ptr = vectorized::check_and_get_column<
74 vectorized::ColumnDictionary<vectorized::Int32>>(nested_col);
75 auto& data_array = nested_col_ptr->get_data();
76 for (uint16_t i = 0; i < size; i++) {
77 if (null_map_data[i]) {
78 if constexpr (is_and) {
79 flags[i] &= _opposite;
80 } else {
81 flags[i] = _opposite;
82 }
83 continue;
84 }
85
86 StringRef cell_value = nested_col_ptr->get_shrink_value(data_array[i]);
87 if constexpr (is_and) {
88 unsigned char flag = 0;
89 (_state->scalar_function)(
90 const_cast<vectorized::LikeSearchState*>(&_like_state),
91 StringRef(cell_value.data, cell_value.size), pattern, &flag);
92 flags[i] &= _opposite ^ flag;
93 } else {
94 unsigned char flag = 0;
95 (_state->scalar_function)(
96 const_cast<vectorized::LikeSearchState*>(&_like_state),
97 StringRef(cell_value.data, cell_value.size), pattern, &flag);
98 flags[i] = _opposite ^ flag;
99 }
100 }
101 } else {
102 LOG(FATAL) << "vectorized (not) like predicates should be dict column";
103 }
104 } else {
105 if (column.is_column_dictionary()) {
106 auto* nested_col_ptr = vectorized::check_and_get_column<
107 vectorized::ColumnDictionary<vectorized::Int32>>(column);
108 auto& data_array = nested_col_ptr->get_data();
109 for (uint16_t i = 0; i < size; i++) {
110 StringRef cell_value = nested_col_ptr->get_shrink_value(data_array[i]);
111 if constexpr (is_and) {
112 unsigned char flag = 0;
113 (_state->scalar_function)(
114 const_cast<vectorized::LikeSearchState*>(&_like_state),
115 StringRef(cell_value.data, cell_value.size), pattern, &flag);
116 flags[i] &= _opposite ^ flag;
117 } else {
118 unsigned char flag = 0;
119 (_state->scalar_function)(
120 const_cast<vectorized::LikeSearchState*>(&_like_state),
121 StringRef(cell_value.data, cell_value.size), pattern, &flag);
122 flags[i] = _opposite ^ flag;
123 }
124 }
125 } else {
126 LOG(FATAL) << "vectorized (not) like predicates should be dict column";
127 }
128 }
129 }
