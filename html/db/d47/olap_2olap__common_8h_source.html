374// 8 bit rowset id version
375// 56 bit, inc number from 1
376// 128 bit backend uid, it is a uuid bit, id version
377struct RowsetId {
378 int8_t version = 0;
379 int64_t hi = 0;
380 int64_t mi = 0;
381 int64_t lo = 0;
382
383 void init(const std::string& rowset_id_str) {
384 // for new rowsetid its a 48 hex string
385 // if the len < 48, then it is an old format rowset id
386 if (rowset_id_str.length() < 48) {
387 int64_t high = std::stol(rowset_id_str, nullptr, 10);
388 init(1, high, 0, 0);
389 } else {
390 int64_t high = 0;
391 int64_t middle = 0;
392 int64_t low = 0;
393 from_hex(&high, rowset_id_str.substr(0, 16));
394 from_hex(&middle, rowset_id_str.substr(16, 16));
395 from_hex(&low, rowset_id_str.substr(32, 16));
396 init(high >> 56, high & LOW_56_BITS, middle, low);
397 }
398 }
399
400 // to compatible with old version
401 void init(int64_t rowset_id) { init(1, rowset_id, 0, 0); }
402
403 void init(int64_t id_version, int64_t high, int64_t middle, int64_t low) {
404 version = id_version;
405 if (UNLIKELY(high >= MAX_ROWSET_ID)) {
406 LOG(FATAL) << "inc rowsetid is too large:" << high;
407 }
408 hi = (id_version << 56) + (high & LOW_56_BITS);
409 mi = middle;
410 lo = low;
411 }
412
413 std::string to_string() const {
414 if (version < 2) {
415 return std::to_string(hi & LOW_56_BITS);
416 } else {
417 char buf[48];
418 to_hex(hi, buf);
419 to_hex(mi, buf + 16);
420 to_hex(lo, buf + 32);
421 return {buf, 48};
422 }
423 }
424
425 // std::unordered_map need this api
426 bool operator==(const RowsetId& rhs) const {
427 return hi == rhs.hi && mi == rhs.mi && lo == rhs.lo;
428 }
429
430 bool operator!=(const RowsetId& rhs) const {
431 return hi != rhs.hi || mi != rhs.mi || lo != rhs.lo;
432 }
433
434 bool operator<(const RowsetId& rhs) const {
435 if (hi != rhs.hi) {
436 return hi < rhs.hi;
437 } else if (mi != rhs.mi) {
438 return mi < rhs.mi;
439 } else {
440 return lo < rhs.lo;
441 }
442 }
443
444 friend std::ostream& operator<<(std::ostream& out, const RowsetId& rowset_id) {
445 out << rowset_id.to_string();
446 return out;
447 }
448};
