167 int compare_at(size_t, size_t, const IColumn& rhs, int nan_direction_hint) const override {
168 auto rhs_const_column = assert_cast<const ColumnConst&>(rhs);
169
170 auto* this_nullable = check_and_get_column<ColumnNullable>(data.get());
171 auto* rhs_nullable = check_and_get_column<ColumnNullable>(rhs_const_column.data.get());
172 if (this_nullable && rhs_nullable) {
173 return data->compare_at(0, 0, *rhs_const_column.data, nan_direction_hint);
174 } else if (this_nullable) {
175 auto rhs_nullable_column = make_nullable(rhs_const_column.data, false);
176 return this_nullable->compare_at(0, 0, *rhs_nullable_column, nan_direction_hint);
177 } else if (rhs_nullable) {
178 auto this_nullable_column = make_nullable(data, false);
179 return this_nullable_column->compare_at(0, 0, *rhs_const_column.data,
180 nan_direction_hint);
181 } else {
182 return data->compare_at(0, 0, *rhs_const_column.data, nan_direction_hint);
183 }
184 }
