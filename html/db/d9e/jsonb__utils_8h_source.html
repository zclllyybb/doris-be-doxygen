37class JsonbToJson {
38public:

41 // get json string
42 const std::string to_json_string(const char* data, size_t size) {
43 doris::JsonbValue* pval = doris::JsonbDocument::createDocument(data, size)->getValue();

47 const std::string to_json_string(const JsonbValue* val) {
48 os_.clear();
49 os_.seekp(0);
50
51 if (val) {
52 intern_json(val);
53 }
54
55 os_.put(0);
56
57 std::string json_string {os_.getBuffer()};
58 return json_string;
59 }

67 // recursively convert JsonbValue
68 void intern_json(const JsonbValue* val) {
69 switch (val->type()) {
70 case JsonbType::T_Null: {
71 os_.write("null", 4);
72 break;
73 }
74 case JsonbType::T_True: {
75 os_.write("true", 4);
76 break;
77 }
78 case JsonbType::T_False: {
79 os_.write("false", 5);
80 break;
81 }
82 case JsonbType::T_Int8: {
83 os_.write(((JsonbInt8Val*)val)->val());
84 break;
85 }
86 case JsonbType::T_Int16: {
87 os_.write(((JsonbInt16Val*)val)->val());
88 break;
89 }
90 case JsonbType::T_Int32: {
91 os_.write(((JsonbInt32Val*)val)->val());
92 break;
93 }
94 case JsonbType::T_Int64: {
95 os_.write(((JsonbInt64Val*)val)->val());
96 break;
97 }
98 case JsonbType::T_Double: {
99 os_.write(((JsonbDoubleVal*)val)->val());
100 break;
101 }
102 case JsonbType::T_Float: {
103 os_.write(((JsonbFloatVal*)val)->val());
104 break;
105 }
106 case JsonbType::T_Int128: {
107 os_.write(((JsonbInt128Val*)val)->val());
108 break;
109 }
110 case JsonbType::T_String: {
111 string_to_json(((JsonbStringVal*)val)->getBlob(), ((JsonbStringVal*)val)->length());
112 break;
113 }
114 case JsonbType::T_Binary: {
115 os_.write("\"<BINARY>", 9);
116 os_.write(((JsonbBinaryVal*)val)->getBlob(), ((JsonbBinaryVal*)val)->getBlobLen());
117 os_.write("<BINARY>\"", 9);
118 break;
119 }
120 case JsonbType::T_Object: {
121 object_to_json((ObjectVal*)val);
122 break;
123 }
124 case JsonbType::T_Array: {
125 array_to_json((ArrayVal*)val);
126 break;
127 }
128 default:
129 break;
130 }

133 void string_to_json(const char* str, size_t len) {
134 os_.put('"');
135 if (nullptr == str) {
136 os_.put('"');
137 return;
138 }
139 char char_buffer[16];
140 for (const char* ptr = str; ptr != str + len && *ptr; ++ptr) {
141 if ((unsigned char)*ptr > 31 && *ptr != '\"' && *ptr != '\\')
142 os_.put(*ptr);
143 else {
144 os_.put('\\');
145 unsigned char token;
146 switch (token = *ptr) {
147 case '\\':
148 os_.put('\\');
149 break;
150 case '\"':
151 os_.put('\"');
152 break;
153 case '\b':
154 os_.put('b');
155 break;
156 case '\f':
157 os_.put('f');
158 break;
159 case '\n':
160 os_.put('n');
161 break;
162 case '\r':
163 os_.put('r');
164 break;
165 case '\t':
166 os_.put('t');
167 break;
168 default: {
169 int char_num = snprintf(char_buffer, sizeof(char_buffer), "u%04x", token);
170 os_.write(char_buffer, char_num);
171 break;
172 }
173 }
174 }
175 }
176 os_.put('"');
177 }

179 // convert object
180 void object_to_json(const ObjectVal* val) {
181 os_.put('{');
182
183 auto iter = val->begin();
184 auto iter_fence = val->end();
185
186 while (iter < iter_fence) {
187 // write key
188 if (iter->klen()) {
189 string_to_json(iter->getKeyStr(), iter->klen());
190 } else {
191 os_.write(iter->getKeyId());
192 }
193 os_.put(':');
194
195 // convert value
196 intern_json(iter->value());
197
198 ++iter;
199 if (iter != iter_fence) {
200 os_.put(',');
201 }
202 }
203
204 assert(iter == iter_fence);
205
206 os_.put('}');
207 }
208
209 // convert array to json
210 void array_to_json(const ArrayVal* val) {
211 os_.put('[');
212
213 auto iter = val->begin();
214 auto iter_fence = val->end();
215
216 while (iter != iter_fence) {
217 // convert value
218 intern_json((const JsonbValue*)iter);
219 ++iter;
220 if (iter != iter_fence) {
221 os_.put(',');
222 }
223 }
224
225 assert(iter == iter_fence);
226
227 os_.put(']');
228 }

230private:
231 JsonbOutStream os_;
232 char buffer_[OUT_BUF_SIZE];

235// This class is a utility to create a JsonbValue.
236template <class OS_TYPE>
237class JsonbValueCreaterT {
238public:
239 JsonbValue* operator()(int32_t val) { return operator()((int64_t)val); }
240
241 JsonbValue* operator()(int64_t val) {
242 writer_.reset();
243 writer_.writeStartArray();
244 writer_.writeInt(val);
245 writer_.writeEndArray();
246 return extractValue();
247 }
248 JsonbValue* operator()(double val) {
249 writer_.reset();
250 writer_.writeStartArray();
251 writer_.writeDouble(val);
252 writer_.writeEndArray();
253 return extractValue();
254 }
255 JsonbValue* operator()(const char* str) { return operator()(str, (unsigned int)strlen(str)); }
256 JsonbValue* operator()(const char* str, unsigned int str_len) {
257 writer_.reset();
258 writer_.writeStartArray();
259 writer_.writeStartString();
260 writer_.writeString(str, str_len);
261 writer_.writeEndString();
262 writer_.writeEndArray();
263 return extractValue();
264 }
265 JsonbValue* operator()(bool val) {
266 writer_.reset();
267 writer_.writeStartArray();
268 writer_.writeBool(val);
269 writer_.writeEndArray();
270 return extractValue();
271 }
272 JsonbValue* operator()() {
273 writer_.reset();
274 writer_.writeStartArray();
275 writer_.writeNull();
276 writer_.writeEndArray();
277 return extractValue();
278 }
279
280private:
281 JsonbValue* extractValue() {
282 return static_cast<ArrayVal*>(
283 JsonbDocument::createValue(writer_.getOutput()->getBuffer(),
284 (int)writer_.getOutput()->getSize()))
285 ->get(0);
286 }
287 JsonbWriterT<OS_TYPE> writer_;
288};
