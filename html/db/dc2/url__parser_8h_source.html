30// Example for explaining URL parts:
31// http://user:pass@example.com:80/docs/books/tutorial/index.html?name=networking#DOWNLOADING
32// PROTOCOL = http
33// AUTHORITY = example.com:80
34// HOST = example.com
35// PATH = /docs/books/tutorial/index.html
36// QUERY = name=networking
37// FILENAME = /docs/books/tutorial/index.html?name=networking
38// REF = DOWNLOADING
39// USERINFO = user:pass
40// More details on what exactly the URL parts mean can be found here:
41// http://docs.oracle.com/javase/tutorial/networking/urls/urlInfo.html
42class UrlParser {
43public:
44 // Parts of a URL that can be requested.
45 enum UrlPart { INVALID, AUTHORITY, FILE, HOST, PATH, PROTOCOL, QUERY, REF, USERINFO, PORT };
46
47 // Tries to parse the part from url. Places the result in result.
48 // Returns false if the URL is malformed or if part is invalid. True otherwise.
49 // If false is returned the contents of results are undefined.
50 static bool parse_url(const StringRef& url, UrlPart part, StringRef* result);
51
52 // Tries to parse key from url. Places the result in result.
53 // Returns false if the URL is malformed or if part is invalid. True otherwise.
54 // If false is returned the contents of results are undefined.
55 static bool parse_url_key(const StringRef& url, UrlPart part, const StringRef& key,
56 StringRef* result);
57
58 // Compares part against url_authority, url_file, url_host, etc.,
59 // and returns the corresponding enum.
60 // If part did not match any of the url part constants, returns INVALID.
61 static UrlPart get_url_part(const StringRef& part);
62
63 // Extract parameter value from url
64 // Example for url:
65 // http://doris.apache.org?k1=aa&k2=bb&k3=cc&test=dd#999
66 static StringRef extract_url(StringRef url, StringRef name);
67
68private:
69 // Constants representing parts of a URL.
70 static const StringRef _s_url_authority;
71 static const StringRef _s_url_file;
72 static const StringRef _s_url_host;
73 static const StringRef _s_url_path;
74 static const StringRef _s_url_protocol;
75 static const StringRef _s_url_query;
76 static const StringRef _s_url_ref;
77 static const StringRef _s_url_userinfo;
78 static const StringRef _s_url_port;
79 // Constants used in searching for URL parts.
80 static const StringRef _s_protocol;
81 static const StringRef _s_at;
82 static const StringRef _s_slash;
83 static const StringRef _s_colon;
84 static const StringRef _s_question;
85 static const StringRef _s_hash;
86 static const StringSearch _s_protocol_search;
87 static const StringSearch _s_at_search;
88 static const StringSearch _s_slash_search;
89 static const StringSearch _s_colon_search;
90 static const StringSearch _s_question_search;
91 static const StringSearch _s_hash_search;
92};
