44template <typename T>
45struct AggregateFunctionUniqExactData {
46 static constexpr bool is_string_key = std::is_same_v<T, String>;
47 using Key = std::conditional_t<is_string_key, UInt128, T>;
48 using Hash = std::conditional_t<is_string_key, UInt128TrivialHash, HashCRC32<Key>>;
49
50 using Set = phmap::flat_hash_set<Key, Hash>;
51
52 static UInt128 ALWAYS_INLINE get_key(const StringRef& value) {
53 UInt128 key;
54 SipHash hash;
55 hash.update(value.data, value.size);
56 hash.get128(key.low, key.high);
57 return key;
58 }
59
60 Set set;
61
62 static String get_name() { return "uniqExact"; }
63};

70template <typename T, typename Data>
71struct OneAdder {
72 static void ALWAYS_INLINE add(Data& data, const IColumn& column, size_t row_num) {
73 if constexpr (std::is_same_v<T, String>) {
74 StringRef value = column.get_data_at(row_num);
75 data.set.insert(Data::get_key(value));
76 } else if constexpr (IsDecimalNumber<T>) {
77 data.set.insert(assert_cast<const ColumnDecimal<T>&>(column).get_data()[row_num]);
78 } else {
79 data.set.insert(assert_cast<const ColumnVector<T>&>(column).get_data()[row_num]);
80 }
81 }
82};

87template <typename T, typename Data>
88class AggregateFunctionUniq final
89 : public IAggregateFunctionDataHelper<Data, AggregateFunctionUniq<T, Data>> {
90public:
91 using KeyType = std::conditional_t<std::is_same_v<T, String>, UInt128, T>;
92 AggregateFunctionUniq(const DataTypes& argument_types_)
93 : IAggregateFunctionDataHelper<Data, AggregateFunctionUniq<T, Data>>(argument_types_,
94 {}) {}
95
96 String get_name() const override { return Data::get_name(); }
97
98 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeInt64>(); }
99
100 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
101 Arena*) const override {
102 detail::OneAdder<T, Data>::add(this->data(place), *columns[0], row_num);
103 }
104
105 static ALWAYS_INLINE const KeyType* get_keys(std::vector<KeyType>& keys_container,
106 const IColumn& column, size_t batch_size) {
107 if constexpr (std::is_same_v<T, String>) {
108 keys_container.resize(batch_size);
109 for (size_t i = 0; i != batch_size; ++i) {
110 StringRef value = column.get_data_at(i);
111 keys_container[i] = Data::get_key(value);
112 }
113 return keys_container.data();
114 } else {
115 using ColumnType =
116 std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<T>, ColumnVector<T>>;
117 return assert_cast<const ColumnType&>(column).get_data().data();
118 }
119 }
120
121 void add_batch(size_t batch_size, AggregateDataPtr* places, size_t place_offset,
122 const IColumn** columns, Arena* arena, bool /*agg_many*/) const override {
123 std::vector<KeyType> keys_container;
124 const KeyType* keys = get_keys(keys_container, *columns[0], batch_size);
125
126 std::vector<typename Data::Set*> array_of_data_set(batch_size);
127
128 for (size_t i = 0; i != batch_size; ++i) {
129 array_of_data_set[i] = &(this->data(places[i] + place_offset).set);
130 }
131
132 for (size_t i = 0; i != batch_size; ++i) {
133 if (i + HASH_MAP_PREFETCH_DIST < batch_size) {
134 array_of_data_set[i + HASH_MAP_PREFETCH_DIST]->prefetch(
135 keys[i + HASH_MAP_PREFETCH_DIST]);
136 }
137
138 array_of_data_set[i]->insert(keys[i]);
139 }
140 }
141
142 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
143 Arena*) const override {
144 auto& rhs_set = this->data(rhs).set;
145 if (rhs_set.size() == 0) return;
146
147 auto& set = this->data(place).set;
148 set.rehash(set.size() + rhs_set.size());
149
150 for (auto elem : rhs_set) {
151 set.insert(elem);
152 }
153 }
154
155 void add_batch_single_place(size_t batch_size, AggregateDataPtr place, const IColumn** columns,
156 Arena* arena) const override {
157 std::vector<KeyType> keys_container;
158 const KeyType* keys = get_keys(keys_container, *columns[0], batch_size);
159 auto& set = this->data(place).set;
160
161 for (size_t i = 0; i != batch_size; ++i) {
162 if (i + HASH_MAP_PREFETCH_DIST < batch_size) {
163 set.prefetch(keys[i + HASH_MAP_PREFETCH_DIST]);
164 }
165 set.insert(keys[i]);
166 }
167 }
168
169 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
170 auto& set = this->data(place).set;
171 write_var_uint(set.size(), buf);
172 for (const auto& elem : set) {
173 write_pod_binary(elem, buf);
174 }
175 }
176
177 void deserialize_and_merge(AggregateDataPtr __restrict place, BufferReadable& buf,
178 Arena* arena) const override {
179 auto& set = this->data(place).set;
180 UInt64 size;
181 read_var_uint(size, buf);
182
183 set.rehash(size + set.size());
184
185 for (size_t i = 0; i < size; ++i) {
186 KeyType ref;
187 read_pod_binary(ref, buf);
188 set.insert(ref);
189 }
190 }
191
192 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
193 Arena* arena) const override {
194 deserialize_and_merge(place, buf, arena);
195 }
196
197 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
198 assert_cast<ColumnInt64&>(to).get_data().push_back(this->data(place).set.size());
199 }
200};
