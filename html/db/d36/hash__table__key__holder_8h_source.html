136inline void ALWAYS_INLINE key_holder_discard_key(doris::vectorized::SerializedKeyHolder& holder) {
137 [[maybe_unused]] void* new_head = holder.pool.rollback(holder.key.size);
138 assert(new_head == holder.key.data);
139 holder.key.data = nullptr;
140 holder.key.size = 0;
141}
