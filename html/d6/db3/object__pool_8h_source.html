27// An ObjectPool maintains a list of C++ objects which are deallocated
28// by destroying the pool.
29// Thread-safe.
30class ObjectPool {
31public:
32 ObjectPool() = default;
33
34 ~ObjectPool() { clear(); }
35
36 template <class T>
37 T* add(T* t) {
38 // TODO: Consider using a lock-free structure.
39 std::lock_guard<SpinLock> l(_lock);
40 _objects.emplace_back(Element {t, [](void* obj) { delete reinterpret_cast<T*>(obj); }});
41 return t;
42 }
43
44 template <class T>
45 T* add_array(T* t) {
46 std::lock_guard<SpinLock> l(_lock);
47 _objects.emplace_back(Element {t, [](void* obj) { delete[] reinterpret_cast<T*>(obj); }});
48 return t;
49 }
50
51 void clear() {
52 std::lock_guard<SpinLock> l(_lock);
53 for (Element& elem : _objects) elem.delete_fn(elem.obj);
54 _objects.clear();
55 }
56
57 void acquire_data(ObjectPool* src) {
58 _objects.insert(_objects.end(), src->_objects.begin(), src->_objects.end());
59 src->_objects.clear();
60 }
61
62 uint64_t size() {
63 std::lock_guard<SpinLock> l(_lock);
64 return _objects.size();
65 }
66
67private:
68 ObjectPool(const ObjectPool&) = delete;
69 void operator=(const ObjectPool&) = delete;
70
72 using DeleteFn = void (*)(void*);
73
75 struct Element {
76 void* obj;
77 DeleteFn delete_fn;
78 };
79
80 std::vector<Element> _objects;
81 SpinLock _lock;
82};
83
84} // namespace doris
