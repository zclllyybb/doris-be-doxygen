109 void update(const char* data, doris::vectorized::UInt64 size) {
110 const char* end = data + size;
111
113 if (cnt & 7) {
114 while (cnt & 7 && data < end) {
115 current_bytes[cnt & 7] = *data;
116 ++data;
117 ++cnt;
118 }
119
121 if (cnt & 7) return;
122
123 v3 ^= current_word;
124 SIPROUND;
125 SIPROUND;
126 v0 ^= current_word;
127 }
128
129 cnt += end - data;
130
131 while (data + 8 <= end) {
132 current_word = unaligned_load<doris::vectorized::UInt64>(data);
133
134 v3 ^= current_word;
135 SIPROUND;
136 SIPROUND;
137 v0 ^= current_word;
138
139 data += 8;
140 }
141
143 current_word = 0;
144 switch (end - data) {
145 case 7:
146 current_bytes[6] = data[6];
147 [[fallthrough]];
148 case 6:
149 current_bytes[5] = data[5];
150 [[fallthrough]];
151 case 5:
152 current_bytes[4] = data[4];
153 [[fallthrough]];
154 case 4:
155 current_bytes[3] = data[3];
156 [[fallthrough]];
157 case 3:
158 current_bytes[2] = data[2];
159 [[fallthrough]];
160 case 2:
161 current_bytes[1] = data[1];
162 [[fallthrough]];
163 case 1:
164 current_bytes[0] = data[0];
165 [[fallthrough]];
166 case 0:
167 break;
168 }
169 }
