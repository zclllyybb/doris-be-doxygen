50 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
51 DataTypePtr arg_0 = remove_nullable(arguments[0]);
52 DCHECK(is_array(arg_0) || is_map(arg_0))
53 << "first argument for function: " << name

55 if (is_array(arg_0)) {
56 DCHECK(is_integer(remove_nullable(arguments[1])))

63 check_and_get_data_type<DataTypeMap>(arg_0.get())->get_value_type());
64 } else {
65 LOG(ERROR) << "element_at only support array and map so far.";
66 return nullptr;
67 }

70 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
71 size_t result, size_t input_rows_count) override {
72 auto dst_null_column = ColumnUInt8::create(input_rows_count);
73 UInt8* dst_null_map = dst_null_column->get_data().data();
74 const UInt8* src_null_map = nullptr;
75 ColumnsWithTypeAndName args;
76 auto col_left = block.get_by_position(arguments[0]);
77 if (col_left.column->is_nullable()) {
78 auto null_col = check_and_get_column<ColumnNullable>(*col_left.column);
79 src_null_map = null_col->get_null_map_column().get_data().data();
80 args = {{null_col->get_nested_column_ptr(), remove_nullable(col_left.type),
81 col_left.name},
82 block.get_by_position(arguments[1])};
83 } else {
84 args = {col_left, block.get_by_position(arguments[1])};
85 }
86 ColumnPtr res_column = nullptr;
87 if (args[0].column->is_column_map()) {
88 res_column = _execute_map(args, input_rows_count, src_null_map, dst_null_map);
89 } else {
90 res_column = _execute_nullable(args, input_rows_count, src_null_map, dst_null_map);
91 }
92 if (!res_column) {
93 return Status::RuntimeError("unsupported types for function {}({}, {})", get_name(),
94 block.get_by_position(arguments[0]).type->get_name(),
95 block.get_by_position(arguments[1]).type->get_name());
96 }
97 block.replace_by_position(result,
98 ColumnNullable::create(res_column, std::move(dst_null_column)));
99 return Status::OK();
100 }

104 ColumnPtr _get_mapped_idx(const ColumnArray& column, const ColumnWithTypeAndName& argument) {
105 auto right_column = make_nullable(argument.column->convert_to_full_column_if_const());
106 const ColumnArray::Offsets64& offsets = column.get_offsets();
107 ColumnPtr nested_ptr = make_nullable(column.get_data_ptr());
108 size_t rows = offsets.size();
109 // prepare return data
110 auto matched_indices = ColumnVector<Int8>::create();
111 matched_indices->reserve(rows);
112
113 for (size_t i = 0; i < rows; i++) {
114 bool matched = false;
115 size_t begin = offsets[i - 1];
116 size_t end = offsets[i];
117 for (size_t j = begin; j < end; j++) {
118 if (nested_ptr->compare_at(j, i, *right_column, -1) == 0) {
119 matched_indices->insert_value(j - begin + 1);
120 matched = true;
121 break;
122 }
123 }
124
125 if (!matched) {
126 matched_indices->insert_value(end - begin + 1); // make indices for null
127 }
128 }
129
130 return matched_indices;
131 }

133 template <typename ColumnType>
134 ColumnPtr _execute_number(const ColumnArray::Offsets64& offsets, const IColumn& nested_column,
135 const UInt8* arr_null_map, const IColumn& indices,
136 const UInt8* nested_null_map, UInt8* dst_null_map) {
137 const auto& nested_data = reinterpret_cast<const ColumnType&>(nested_column).get_data();
138
139 auto dst_column = nested_column.clone_empty();
140 auto& dst_data = reinterpret_cast<ColumnType&>(*dst_column).get_data();
141 dst_data.resize(offsets.size());
142
143 // process
144 for (size_t row = 0; row < offsets.size(); ++row) {
145 size_t off = offsets[row - 1];
146 size_t len = offsets[row] - off;
147 auto index = indices.get_int(row);
148 // array is nullable
149 bool null_flag = bool(arr_null_map && arr_null_map[row]);
150 // calc index in nested column
151 if (!null_flag && index > 0 && index <= len) {
152 index += off - 1;
153 } else if (!null_flag && index < 0 && -index <= len) {
154 index += off + len;
155 } else {
156 null_flag = true;
157 }
158 // nested column nullable check
159 if (!null_flag && nested_null_map && nested_null_map[index]) {
160 null_flag = true;
161 }
162 // actual data copy
163 if (null_flag) {
164 dst_null_map[row] = true;
165 dst_data[row] = typename ColumnType::value_type();
166 } else {
167 DCHECK(index >= 0 && index < nested_data.size());
168 dst_null_map[row] = false;
169 dst_data[row] = nested_data[index];
170 }
171 }
172 return dst_column;
173 }

175 ColumnPtr _execute_string(const ColumnArray::Offsets64& offsets, const IColumn& nested_column,
176 const UInt8* arr_null_map, const IColumn& indices,
177 const UInt8* nested_null_map, UInt8* dst_null_map) {
178 const auto& src_str_offs =
179 reinterpret_cast<const ColumnString&>(nested_column).get_offsets();
180 const auto& src_str_chars =
181 reinterpret_cast<const ColumnString&>(nested_column).get_chars();
182
183 // prepare return data
184 auto dst_column = ColumnString::create();
185 auto& dst_str_offs = dst_column->get_offsets();
186 dst_str_offs.resize(offsets.size());
187 auto& dst_str_chars = dst_column->get_chars();
188 dst_str_chars.reserve(src_str_chars.size());
189
190 // process
191 for (size_t row = 0; row < offsets.size(); ++row) {
192 size_t off = offsets[row - 1];
193 size_t len = offsets[row] - off;
194 auto index = indices.get_int(row);
195 // array is nullable
196 bool null_flag = bool(arr_null_map && arr_null_map[row]);
197 // calc index in nested column
198 if (!null_flag && index > 0 && index <= len) {
199 index += off - 1;
200 } else if (!null_flag && index < 0 && -index <= len) {
201 index += off + len;
202 } else {
203 null_flag = true;
204 }
205 // nested column nullable check
206 if (!null_flag && nested_null_map && nested_null_map[index]) {
207 null_flag = true;
208 }
209 // actual string copy
210 if (!null_flag) {
211 DCHECK(index >= 0 && index < src_str_offs.size());
212 dst_null_map[row] = false;
213 auto element_size = src_str_offs[index] - src_str_offs[index - 1];
214 dst_str_offs[row] = dst_str_offs[row - 1] + element_size;
215 auto src_string_pos = src_str_offs[index - 1];
216 auto dst_string_pos = dst_str_offs[row - 1];
217 dst_str_chars.resize(dst_string_pos + element_size);
218 memcpy(&dst_str_chars[dst_string_pos], &src_str_chars[src_string_pos],
219 element_size);
220 } else {
221 dst_null_map[row] = true;
222 dst_str_offs[row] = dst_str_offs[row - 1];
223 }
224 }
225 return dst_column;
226 }

228 ColumnPtr _execute_map(const ColumnsWithTypeAndName& arguments, size_t input_rows_count,
229 const UInt8* src_null_map, UInt8* dst_null_map) {
230 auto left_column = arguments[0].column->convert_to_full_column_if_const();
231 DataTypePtr val_type =
232 reinterpret_cast<const DataTypeMap&>(*arguments[0].type).get_value_type();
233 const auto& map_column = reinterpret_cast<const ColumnMap&>(*left_column);
234
235 // create column array to find keys
236 auto key_arr = ColumnArray::create(map_column.get_keys_ptr(), map_column.get_offsets_ptr());
237 auto val_arr =
238 ColumnArray::create(map_column.get_values_ptr(), map_column.get_offsets_ptr());
239
240 const auto& offsets = map_column.get_offsets();
241 const size_t rows = offsets.size();
242
243 if (rows <= 0) {
244 return nullptr;
245 }
246 if (key_arr->is_nullable()) {
247 }
248 ColumnPtr matched_indices = _get_mapped_idx(*key_arr, arguments[1]);
249 if (!matched_indices) {
250 return nullptr;
251 }
252 DataTypePtr indices_type(std::make_shared<vectorized::DataTypeInt8>());
253 ColumnWithTypeAndName indices(matched_indices, indices_type, "indices");
254 ColumnWithTypeAndName data(val_arr, val_type, "value");
255 ColumnsWithTypeAndName args = {data, indices};
256 return _execute_nullable(args, input_rows_count, src_null_map, dst_null_map);
257 }

259 ColumnPtr _execute_nullable(const ColumnsWithTypeAndName& arguments, size_t input_rows_count,
260 const UInt8* src_null_map, UInt8* dst_null_map) {
261 // check array nested column type and get data
262 auto left_column = arguments[0].column->convert_to_full_column_if_const();
263 const auto& array_column = reinterpret_cast<const ColumnArray&>(*left_column);
264 const auto& offsets = array_column.get_offsets();
265 DCHECK(offsets.size() == input_rows_count);
266 const UInt8* nested_null_map = nullptr;
267 ColumnPtr nested_column = nullptr;
268 if (is_column_nullable(array_column.get_data())) {
269 const auto& nested_null_column =
270 reinterpret_cast<const ColumnNullable&>(array_column.get_data());
271 nested_null_map = nested_null_column.get_null_map_column().get_data().data();
272 nested_column = nested_null_column.get_nested_column_ptr();
273 } else {
274 nested_column = array_column.get_data_ptr();
275 }
276
277 ColumnPtr res = nullptr;
278 // because we impl use_default_implementation_for_nulls
279 // we should handle array index column by-self, and array index should not be nullable.
280 auto idx_col = remove_nullable(arguments[1].column);
281 if (nested_column->is_date_type()) {
282 res = _execute_number<ColumnDate>(offsets, *nested_column, src_null_map, *idx_col,
283 nested_null_map, dst_null_map);
284 } else if (nested_column->is_datetime_type()) {
285 res = _execute_number<ColumnDateTime>(offsets, *nested_column, src_null_map, *idx_col,
286 nested_null_map, dst_null_map);
287 } else if (check_column<ColumnDateV2>(nested_column)) {
288 res = _execute_number<ColumnDateV2>(offsets, *nested_column, src_null_map, *idx_col,
289 nested_null_map, dst_null_map);
290 } else if (check_column<ColumnDateTimeV2>(nested_column)) {
291 res = _execute_number<ColumnDateTime>(offsets, *nested_column, src_null_map, *idx_col,
292 nested_null_map, dst_null_map);
293 } else if (check_column<ColumnUInt8>(*nested_column)) {
294 res = _execute_number<ColumnUInt8>(offsets, *nested_column, src_null_map, *idx_col,
295 nested_null_map, dst_null_map);
296 } else if (check_column<ColumnInt8>(*nested_column)) {
297 res = _execute_number<ColumnInt8>(offsets, *nested_column, src_null_map, *idx_col,
298 nested_null_map, dst_null_map);
299 } else if (check_column<ColumnInt16>(*nested_column)) {
300 res = _execute_number<ColumnInt16>(offsets, *nested_column, src_null_map, *idx_col,
301 nested_null_map, dst_null_map);
302 } else if (check_column<ColumnInt32>(*nested_column)) {
303 res = _execute_number<ColumnInt32>(offsets, *nested_column, src_null_map, *idx_col,
304 nested_null_map, dst_null_map);
305 } else if (check_column<ColumnInt64>(*nested_column)) {
306 res = _execute_number<ColumnInt64>(offsets, *nested_column, src_null_map, *idx_col,
307 nested_null_map, dst_null_map);
308 } else if (check_column<ColumnInt128>(*nested_column)) {
309 res = _execute_number<ColumnInt128>(offsets, *nested_column, src_null_map, *idx_col,
310 nested_null_map, dst_null_map);
311 } else if (check_column<ColumnFloat32>(*nested_column)) {
312 res = _execute_number<ColumnFloat32>(offsets, *nested_column, src_null_map, *idx_col,
313 nested_null_map, dst_null_map);
314 } else if (check_column<ColumnFloat64>(*nested_column)) {
315 res = _execute_number<ColumnFloat64>(offsets, *nested_column, src_null_map, *idx_col,
316 nested_null_map, dst_null_map);
317 } else if (check_column<ColumnDecimal32>(*nested_column)) {
318 res = _execute_number<ColumnDecimal32>(offsets, *nested_column, src_null_map, *idx_col,
319 nested_null_map, dst_null_map);
320 } else if (check_column<ColumnDecimal64>(*nested_column)) {
321 res = _execute_number<ColumnDecimal64>(offsets, *nested_column, src_null_map, *idx_col,
322 nested_null_map, dst_null_map);
323 } else if (check_column<ColumnDecimal128I>(*nested_column)) {
324 res = _execute_number<ColumnDecimal128I>(offsets, *nested_column, src_null_map,
325 *idx_col, nested_null_map, dst_null_map);
326 } else if (check_column<ColumnDecimal128>(*nested_column)) {
327 res = _execute_number<ColumnDecimal128>(offsets, *nested_column, src_null_map, *idx_col,
328 nested_null_map, dst_null_map);
329 } else if (check_column<ColumnString>(*nested_column)) {
330 res = _execute_string(offsets, *nested_column, src_null_map, *idx_col, nested_null_map,
331 dst_null_map);
332 }
333
334 return res;
335 }
