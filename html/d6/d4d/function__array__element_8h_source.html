34class FunctionArrayElement : public IFunction {
35public:
36 static constexpr auto name = "element_at";
37 static FunctionPtr create() { return std::make_shared<FunctionArrayElement>(); }
38
40 String get_name() const override { return name; }
41
42 bool is_variadic() const override { return false; }
43
44 size_t get_number_of_arguments() const override { return 2; }
45
46 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
47 DCHECK(is_array(arguments[0]))
48 << "first argument for function: " << name << " should be DataTypeArray";
49 DCHECK(is_integer(arguments[1]))
50 << "second argument for function: " << name << " should be Integer";
51 return make_nullable(
52 check_and_get_data_type<DataTypeArray>(arguments[0].get())->get_nested_type());
53 }

55 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
56 size_t result, size_t input_rows_count) override {
57 auto dst_null_column = ColumnUInt8::create(input_rows_count);
58 UInt8* dst_null_map = dst_null_column->get_data().data();
59 const UInt8* src_null_map = nullptr;
60 ColumnsWithTypeAndName args;
61 auto col_left = block.get_by_position(arguments[0]);
62 if (col_left.column->is_nullable()) {
63 auto null_col = check_and_get_column<ColumnNullable>(*col_left.column);
64 src_null_map = null_col->get_null_map_column().get_data().data();
65 args = {{null_col->get_nested_column_ptr(), remove_nullable(col_left.type),
66 col_left.name},
67 block.get_by_position(arguments[1])};
68 } else {
69 args = {col_left, block.get_by_position(arguments[1])};
70 }
71
72 auto res_column = _execute_non_nullable(args, input_rows_count, src_null_map, dst_null_map);
73 if (!res_column) {
74 return Status::RuntimeError("unsupported types for function {}({}, {})", get_name(),
75 block.get_by_position(arguments[0]).type->get_name(),
76 block.get_by_position(arguments[1]).type->get_name());
77 }
78 block.replace_by_position(
79 result, ColumnNullable::create(std::move(res_column), std::move(dst_null_column)));
80 return Status::OK();
81 }

83private:
84 template <typename ColumnType>
85 ColumnPtr _execute_number(const ColumnArray::Offsets64& offsets, const IColumn& nested_column,
86 const UInt8* arr_null_map, const IColumn& indices,
87 const UInt8* nested_null_map, UInt8* dst_null_map) {
88 const auto& nested_data = reinterpret_cast<const ColumnType&>(nested_column).get_data();
89
90 auto dst_column = nested_column.clone_empty();
91 auto& dst_data = reinterpret_cast<ColumnType&>(*dst_column).get_data();
92 dst_data.resize(offsets.size());
93
94 // process
95 for (size_t row = 0; row < offsets.size(); ++row) {
96 size_t off = offsets[row - 1];
97 size_t len = offsets[row] - off;
98 auto index = indices.get_int(row);
99 // array is nullable
100 bool null_flag = bool(arr_null_map && arr_null_map[row]);
101 // calc index in nested column
102 if (!null_flag && index > 0 && index <= len) {
103 index += off - 1;
104 } else if (!null_flag && index < 0 && -index <= len) {
105 index += off + len;
106 } else {
107 null_flag = true;
108 }
109 // nested column nullable check
110 if (!null_flag && nested_null_map && nested_null_map[index]) {
111 null_flag = true;
112 }
113 // actual data copy
114 if (null_flag) {
115 dst_null_map[row] = true;
116 dst_data[row] = typename ColumnType::value_type();
117 } else {
118 DCHECK(index >= 0 && index < nested_data.size());
119 dst_null_map[row] = false;
120 dst_data[row] = nested_data[index];
121 }
122 }
123 return dst_column;
124 }

126 ColumnPtr _execute_string(const ColumnArray::Offsets64& offsets, const IColumn& nested_column,
127 const UInt8* arr_null_map, const IColumn& indices,
128 const UInt8* nested_null_map, UInt8* dst_null_map) {
129 const auto& src_str_offs =
130 reinterpret_cast<const ColumnString&>(nested_column).get_offsets();
131 const auto& src_str_chars =
132 reinterpret_cast<const ColumnString&>(nested_column).get_chars();
133
134 // prepare return data
135 auto dst_column = ColumnString::create();
136 auto& dst_str_offs = dst_column->get_offsets();
137 dst_str_offs.resize(offsets.size());
138 auto& dst_str_chars = dst_column->get_chars();
139 dst_str_chars.reserve(src_str_chars.size());
140
141 // process
142 for (size_t row = 0; row < offsets.size(); ++row) {
143 size_t off = offsets[row - 1];
144 size_t len = offsets[row] - off;
145 auto index = indices.get_int(row);
146 // array is nullable
147 bool null_flag = bool(arr_null_map && arr_null_map[row]);
148 // calc index in nested column
149 if (!null_flag && index > 0 && index <= len) {
150 index += off - 1;
151 } else if (!null_flag && index < 0 && -index <= len) {
152 index += off + len;
153 } else {
154 null_flag = true;
155 }
156 // nested column nullable check
157 if (!null_flag && nested_null_map && nested_null_map[index]) {
158 null_flag = true;
159 }
160 // actual string copy
161 if (!null_flag) {
162 DCHECK(index >= 0 && index < src_str_offs.size());
163 dst_null_map[row] = false;
164 auto element_size = src_str_offs[index] - src_str_offs[index - 1];
165 dst_str_offs[row] = dst_str_offs[row - 1] + element_size;
166 auto src_string_pos = src_str_offs[index - 1];
167 auto dst_string_pos = dst_str_offs[row - 1];
168 dst_str_chars.resize(dst_string_pos + element_size);
169 memcpy(&dst_str_chars[dst_string_pos], &src_str_chars[src_string_pos],
170 element_size);
171 } else {
172 dst_null_map[row] = true;
173 dst_str_offs[row] = dst_str_offs[row - 1];
174 }
175 }
176 return dst_column;
177 }

179 ColumnPtr _execute_non_nullable(const ColumnsWithTypeAndName& arguments,
180 size_t input_rows_count, const UInt8* src_null_map,
181 UInt8* dst_null_map) {
182 // check array nested column type and get data
183 auto left_column = arguments[0].column->convert_to_full_column_if_const();
184 const auto& array_column = reinterpret_cast<const ColumnArray&>(*left_column);
185 const auto& offsets = array_column.get_offsets();
186 DCHECK(offsets.size() == input_rows_count);
187 const UInt8* nested_null_map = nullptr;
188 ColumnPtr nested_column = nullptr;
189 if (is_column_nullable(array_column.get_data())) {
190 const auto& nested_null_column =
191 reinterpret_cast<const ColumnNullable&>(array_column.get_data());
192 nested_null_map = nested_null_column.get_null_map_column().get_data().data();
193 nested_column = nested_null_column.get_nested_column_ptr();
194 } else {
195 nested_column = array_column.get_data_ptr();
196 }
197
198 ColumnPtr res = nullptr;
199 if (nested_column->is_date_type()) {
200 res = _execute_number<ColumnDate>(offsets, *nested_column, src_null_map,
201 *arguments[1].column, nested_null_map, dst_null_map);
202 } else if (nested_column->is_datetime_type()) {
203 res = _execute_number<ColumnDateTime>(offsets, *nested_column, src_null_map,
204 *arguments[1].column, nested_null_map,
205 dst_null_map);
206 } else if (check_column<ColumnUInt8>(*nested_column)) {
207 res = _execute_number<ColumnUInt8>(offsets, *nested_column, src_null_map,
208 *arguments[1].column, nested_null_map, dst_null_map);
209 } else if (check_column<ColumnInt8>(*nested_column)) {
210 res = _execute_number<ColumnInt8>(offsets, *nested_column, src_null_map,
211 *arguments[1].column, nested_null_map, dst_null_map);
212 } else if (check_column<ColumnInt16>(*nested_column)) {
213 res = _execute_number<ColumnInt16>(offsets, *nested_column, src_null_map,
214 *arguments[1].column, nested_null_map, dst_null_map);
215 } else if (check_column<ColumnInt32>(*nested_column)) {
216 res = _execute_number<ColumnInt32>(offsets, *nested_column, src_null_map,
217 *arguments[1].column, nested_null_map, dst_null_map);
218 } else if (check_column<ColumnInt64>(*nested_column)) {
219 res = _execute_number<ColumnInt64>(offsets, *nested_column, src_null_map,
220 *arguments[1].column, nested_null_map, dst_null_map);
221 } else if (check_column<ColumnInt128>(*nested_column)) {
222 res = _execute_number<ColumnInt128>(offsets, *nested_column, src_null_map,
223 *arguments[1].column, nested_null_map,
224 dst_null_map);
225 } else if (check_column<ColumnFloat32>(*nested_column)) {
226 res = _execute_number<ColumnFloat32>(offsets, *nested_column, src_null_map,
227 *arguments[1].column, nested_null_map,
228 dst_null_map);
229 } else if (check_column<ColumnFloat64>(*nested_column)) {
230 res = _execute_number<ColumnFloat64>(offsets, *nested_column, src_null_map,
231 *arguments[1].column, nested_null_map,
232 dst_null_map);
233 } else if (check_column<ColumnDecimal128>(*nested_column)) {
234 res = _execute_number<ColumnDecimal128>(offsets, *nested_column, src_null_map,
235 *arguments[1].column, nested_null_map,
236 dst_null_map);
237 } else if (check_column<ColumnString>(*nested_column)) {
238 res = _execute_string(offsets, *nested_column, src_null_map, *arguments[1].column,
239 nested_null_map, dst_null_map);
240 }
241
242 return res;
243 }
