164class WindowFunctionNTile final
165 : public IAggregateFunctionDataHelper<NTileData, WindowFunctionNTile> {
166public:
167 WindowFunctionNTile(const DataTypes& argument_types_, const Array& parameters)
168 : IAggregateFunctionDataHelper(argument_types_, parameters) {}
169
170 String get_name() const override { return "ntile"; }
171
172 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeInt64>(); }
173
174 void add(AggregateDataPtr place, const IColumn**, size_t, Arena*) const override {}
175
176 void add_range_single_place(int64_t partition_start, int64_t partition_end, int64_t frame_start,
177 int64_t frame_end, AggregateDataPtr place, const IColumn** columns,
178 Arena* arena) const override {
179 // some variables are partition related, but there is no chance to init them
180 // when the new partition arrives, so we calculate them evey time now.
181 // Partition = big_bucket_num * big_bucket_size + small_bucket_num * small_bucket_size
182 int64_t row_index = ++WindowFunctionNTile::data(place).rows - 1;
183 int64_t bucket_num = columns[0]->get_int(0);
184 int64_t partition_size = partition_end - partition_start;
185
186 int64 small_bucket_size = partition_size / bucket_num;
187 int64 big_bucket_num = partition_size % bucket_num;
188 int64 first_small_bucket_row_index = big_bucket_num * (small_bucket_size + 1);
189 if (row_index >= first_small_bucket_row_index) {
190 // small_bucket_size can't be zero
191 WindowFunctionNTile::data(place).bucket_index =
192 big_bucket_num + 1 +
193 (row_index - first_small_bucket_row_index) / small_bucket_size;
194 } else {
195 WindowFunctionNTile::data(place).bucket_index = row_index / (small_bucket_size + 1) + 1;
196 }
197 }
198
199 void reset(AggregateDataPtr place) const override { WindowFunctionNTile::data(place).rows = 0; }
200
201 void insert_result_into(ConstAggregateDataPtr place, IColumn& to) const override {
202 assert_cast<ColumnInt64&>(to).get_data().push_back(
203 WindowFunctionNTile::data(place).bucket_index);
204 }
205
206 void merge(AggregateDataPtr place, ConstAggregateDataPtr rhs, Arena*) const override {}
207 void serialize(ConstAggregateDataPtr place, BufferWritable& buf) const override {}
208 void deserialize(AggregateDataPtr place, BufferReadable& buf, Arena*) const override {}
209};

227template <typename ColVecType, bool result_is_nullable, bool arg_is_nullable>
228struct LeadLagData {
229public:
230 void reset() {
231 _data_value.reset();
232 _default_value.reset();
233 _is_inited = false;
234 }
235
236 bool default_is_null() { return _default_value.is_null(); }
237
238 // here _ptr pointer default column from third
239 void set_value_from_default() { this->_data_value = _default_value; }
240
241 void insert_result_into(IColumn& to) const {
242 if constexpr (result_is_nullable) {
243 if (_data_value.is_null()) {
244 auto& col = assert_cast<ColumnNullable&>(to);
245 col.insert_default();
246 } else {
247 auto& col = assert_cast<ColumnNullable&>(to);
248 StringRef value = _data_value.get_value();
249 col.insert_data(value.data, value.size);
250 }
251 } else {
252 StringRef value = _data_value.get_value();
253 to.insert_data(value.data, value.size);
254 }
255 }
256
257 void set_is_null() { this->_data_value.reset(); }
258
259 void set_value(const IColumn** columns, size_t pos) {
260 if constexpr (arg_is_nullable) {
261 if (assert_cast<const ColumnNullable*>(columns[0])->is_null_at(pos)) {
262 // ptr == nullptr means nullable
263 _data_value.reset();
264 return;
265 }
266 }
267 // here ptr is pointer to nullable column or not null column from first
268 _data_value.set_value(columns[0], pos);
269 }
270
271 void check_default(const IColumn* column) {
272 if (!_is_inited) {
273 if (is_column_nullable(*column)) {
274 const auto* nullable_column = assert_cast<const ColumnNullable*>(column);
275 if (nullable_column->is_null_at(0)) {
276 _default_value.reset();
277 }
278 } else {
279 _default_value.set_value(column, 0);
280 }
281 _is_inited = true;
282 }
283 }
284
285private:
286 BaseValue<ColVecType, arg_is_nullable> _data_value;
287 BaseValue<ColVecType, arg_is_nullable> _default_value;
288 bool _is_inited = false;
289};

292struct WindowFunctionLeadImpl : Data {
293 void add_range_single_place(int64_t partition_start, int64_t partition_end, int64_t frame_start,
294 int64_t frame_end, const IColumn** columns) {
295 this->check_default(columns[2]);
296 if (frame_end > partition_end) { //output default value, win end is under partition
297 if (this->default_is_null()) {
298 this->set_is_null();
299 } else {
300 this->set_value_from_default();
301 }
302 return;
303 }
304 this->set_value(columns, frame_end - 1);
305 }
