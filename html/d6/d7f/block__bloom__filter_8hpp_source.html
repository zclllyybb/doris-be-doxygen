39// https://github.com/apache/kudu/blob/master/src/kudu/util/block_bloom_filter.h
40// BlockBloomFilter is modified based on Impala's BlockBloomFilter.
41// For general Bloomfilter implementations, this implementation is
42// more friendly to CPU Cache, and it is easier to use SIMD instructions to
43// speed up the implementation.
44
45// BlockBloomFilter will not store null values, and will always return a false if the input is null.

47// Some constants used in hashing. #defined for efficiency reasons.
48#define BLOOM_HASH_CONSTANTS \
49 0x47b6137bU, 0x44974d91U, 0x8824ad5bU, 0xa2b7289dU, 0x705495c7U, 0x2df1424bU, 0x9efc4947U, \
50 0x5c6bfb31U

68 // Adds an element to the BloomFilter. The function used to generate 'hash' need not
69 // have good uniformity, but it should have low collision probability. For instance, if
70 // the set of values is 32-bit ints, the identity function is a valid hash function for
71 // this Bloom filter, since the collision probability (the probability that two
72 // non-equal values will have the same hash value) is 0.
73 void insert(uint32_t hash) noexcept;
74 // Same as above with convenience of hashing the key.
75 void insert(const Slice& key) noexcept {
76 if (key.data) {
77 insert(HashUtil::murmur_hash3_32(key.data, key.size, _hash_seed));
78 }
79 }

81#ifdef __AVX2__

83 static inline ATTRIBUTE_ALWAYS_INLINE __attribute__((__target__("avx2"))) __m256i make_mark(
84 const uint32_t hash) {
85 const __m256i ones = _mm256_set1_epi32(1);
86 const __m256i rehash = _mm256_setr_epi32(BLOOM_HASH_CONSTANTS);
87 // Load hash into a YMM register, repeated eight times
88 __m256i hash_data = _mm256_set1_epi32(hash);
89 // Multiply-shift hashing ala Dietzfelbinger et al.: multiply 'hash' by eight different
90 // odd constants, then keep the 5 most significant bits from each product.
91 hash_data = _mm256_mullo_epi32(rehash, hash_data);
92 hash_data = _mm256_srli_epi32(hash_data, 27);
93 // Use these 5 bits to shift a single bit to a location in each 32-bit lane
94 return _mm256_sllv_epi32(ones, hash_data);
95 }
96#endif
97 // Finds an element in the BloomFilter, returning true if it is found and false (with
98 // high probability) if it is not.
99 ALWAYS_INLINE bool find(uint32_t hash) const noexcept {
100 if (_always_false) {
101 return false;
102 }
103 const uint32_t bucket_idx = rehash32to32(hash) & _directory_mask;
104#ifdef __AVX2__
105 const __m256i mask = make_mark(hash);
106 const __m256i bucket = reinterpret_cast<__m256i*>(_directory)[bucket_idx];
107 // We should return true if 'bucket' has a one wherever 'mask' does. _mm256_testc_si256
108 // takes the negation of its first argument and ands that with its second argument. In
109 // our case, the result is zero everywhere iff there is a one in 'bucket' wherever
110 // 'mask' is one. testc returns 1 if the result is 0 everywhere and returns 0 otherwise.
111 const bool result = _mm256_testc_si256(bucket, mask);
112 _mm256_zeroupper();
113 return result;
114#else
115 return bucket_find(bucket_idx, hash);
116#endif
117 }
118 // Same as above with convenience of hashing the key.
119 bool find(const Slice& key) const noexcept {
120 if (key.data) {
121 return find(HashUtil::murmur_hash3_32(key.data, key.size, _hash_seed));
122 } else {
123 return false;
124 }
125 }

158 // The BloomFilter is divided up into Buckets and each Bucket comprises of 8 BucketWords of
159 // 4 bytes each.
160 static constexpr uint64_t kBucketWords = 8;
161 typedef uint32_t BucketWord;
162
163 // log2(number of bits in a BucketWord)
164 static constexpr int kLogBucketWordBits = 5;
165 static constexpr BucketWord kBucketWordMask = (1 << kLogBucketWordBits) - 1;
166
167 // log2(number of bytes in a bucket)
168 static constexpr int kLogBucketByteSize = 5;
169 // Bucket size in bytes.
170 static constexpr size_t kBucketByteSize = 1UL << kLogBucketByteSize;
171
172 static_assert(
173 (1 << kLogBucketWordBits) == std::numeric_limits<BucketWord>::digits,
174 "BucketWord must have a bit-width that is be a power of 2, like 64 for uint64_t.");

228 // kRehash is used as 8 odd 32-bit unsigned ints. See Dietzfelbinger et al.'s "A
229 // reliable randomized algorithm for the closest-pair problem".
230 static constexpr uint32_t kRehash[8] __attribute__((aligned(32))) = {BLOOM_HASH_CONSTANTS};

232 // Get 32 more bits of randomness from a 32-bit hash:
233 static ALWAYS_INLINE uint32_t rehash32to32(const uint32_t hash) {
234 // Constants generated by uuidgen(1) with the -r flag
235 static constexpr uint64_t m = 0x7850f11ec6d14889ULL;
236 static constexpr uint64_t a = 0x6773610597ca4c63ULL;
237 // This is strongly universal hashing following Dietzfelbinger's "Universal hashing
238 // and k-wise independent random variables via integer arithmetic without primes". As
239 // such, for any two distinct uint32_t's hash1 and hash2, the probability (over the
240 // randomness of the constants) that any subset of bit positions of
241 // Rehash32to32(hash1) is equal to the same subset of bit positions
242 // Rehash32to32(hash2) is minimal.
243 return (static_cast<uint64_t>(hash) * m + a) >> 32U;
244 }
