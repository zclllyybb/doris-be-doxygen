59// Superclass of all executor nodes.
60// All subclasses need to make sure to check RuntimeState::is_cancelled()
61// periodically in order to ensure timely termination after the cancellation
62// flag gets set.
63class ExecNode {
64public:
65 // Init conjuncts.
66 ExecNode(ObjectPool* pool, const TPlanNode& tnode, const DescriptorTbl& descs);
67
68 virtual ~ExecNode();
69
73 virtual Status init(const TPlanNode& tnode, RuntimeState* state);
74
75 // Sets up internal structures, etc., without doing any actual work.
76 // Must be called prior to open(). Will only be called once in this
77 // node's lifetime.
78 // All code generation (adding functions to the LlvmCodeGen object) must happen
79 // in prepare(). Retrieving the jit compiled function pointer must happen in
80 // open().
81 // If overridden in subclass, must first call superclass's prepare().
82 virtual Status prepare(RuntimeState* state);
83
84 // Performs any preparatory work prior to calling get_next().
85 // Can be called repeatedly (after calls to close()).
86 // Caller must not be holding any io buffers. This will cause deadlock.
87 virtual Status open(RuntimeState* state);
88
89 // Alloc and open resource for the node
90 // Only pipeline operator use exec node need to impl the virtual function
91 // so only vectorized exec node need to impl
92 virtual Status alloc_resource(RuntimeState* state);
93
94 // Retrieves rows and returns them via row_batch. Sets eos to true
95 // if subsequent calls will not retrieve any more rows.
96 // Data referenced by any tuples returned in row_batch must not be overwritten
97 // by the callee until close() is called. The memory holding that data
98 // can be returned via row_batch's tuple_data_pool (in which case it may be deleted
99 // by the caller) or held on to by the callee. The row_batch, including its
100 // tuple_data_pool, will be destroyed by the caller at some point prior to the final
101 // close() call.
102 // In other words, if the memory holding the tuple data will be referenced
103 // by the callee in subsequent get_next() calls, it must *not* be attached to the
104 // row_batch's tuple_data_pool.
105 // Caller must not be holding any io buffers. This will cause deadlock.
106 // TODO: AggregationNode and HashJoinNode cannot be "re-opened" yet.
107 virtual Status get_next(RuntimeState* state, vectorized::Block* block, bool* eos);
108 // new interface to compatible new optimizers in FE
109 Status get_next_after_projects(
110 RuntimeState* state, vectorized::Block* block, bool* eos,
111 const std::function<Status(RuntimeState*, vectorized::Block*, bool*)>& fn);
112
113 // Emit data, both need impl with method: sink
114 // Eg: Aggregation, Sort, Scan
115 virtual Status pull(RuntimeState* state, vectorized::Block* output_block, bool* eos) {
116 return get_next(state, output_block, eos);
117 }
118
119 virtual Status push(RuntimeState* state, vectorized::Block* input_block, bool eos) {
120 return Status::OK();
121 }
122
123 bool can_read() const { return _can_read; }
124
125 // Sink Data to ExecNode to do some stock work, both need impl with method: get_result
126 // `eos` means source is exhausted, exec node should do some finalize work
127 // Eg: Aggregation, Sort
128 virtual Status sink(RuntimeState* state, vectorized::Block* input_block, bool eos);
129
130 // Resets the stream of row batches to be retrieved by subsequent GetNext() calls.
131 // Clears all internal state, returning this node to the state it was in after calling
132 // Prepare() and before calling Open(). This function must not clear memory
133 // still owned by this node that is backing rows returned in GetNext().
134 // Prepare() and Open() must have already been called before calling Reset().
135 // GetNext() may have optionally been called (not necessarily until eos).
136 // Close() must not have been called.
137 // Reset() is not idempotent. Calling it multiple times in a row without a preceding
138 // call to Open() is invalid.
139 // If overridden in a subclass, must call superclass's Reset() at the end. The default
140 // implementation calls Reset() on children.
141 // Note that this function may be called many times (proportional to the input data),
142 // so should be fast.
143 virtual Status reset(RuntimeState* state);
144
145 // This should be called before close() and after get_next(), it is responsible for
146 // collecting statistics sent with row batch, it can't be called when prepare() returns
147 // error.
148 virtual Status collect_query_statistics(QueryStatistics* statistics);
149
150 // close() will get called for every exec node, regardless of what else is called and
151 // the status of these calls (i.e. prepare() may never have been called, or
152 // prepare()/open()/get_next() returned with an error).
153 // close() releases all resources that were allocated in open()/get_next(), even if the
154 // latter ended with an error. close() can be called if the node has been prepared or
155 // the node is closed.
156 // After calling close(), the caller calls open() again prior to subsequent calls to
157 // get_next(). The default implementation updates runtime profile counters and calls
158 // close() on the children. To ensure that close() is called on the entire plan tree,
159 // each implementation should start out by calling the default implementation.
160 virtual Status close(RuntimeState* state);
161
162 void increase_ref() { ++_ref; }
163 int decrease_ref() { return --_ref; }
164
165 // Release and close resource for the node
166 // Only pipeline operator use exec node need to impl the virtual function
167 // so only vectorized exec node need to impl
168 virtual void release_resource(RuntimeState* state);
169
170 // Creates exec node tree from list of nodes contained in plan via depth-first
171 // traversal. All nodes are placed in pool.
172 // Returns error if 'plan' is corrupted, otherwise success.
173 static Status create_tree(RuntimeState* state, ObjectPool* pool, const TPlan& plan,
174 const DescriptorTbl& descs, ExecNode** root);
175
176 // Collect all nodes of given 'node_type' that are part of this subtree, and return in
177 // 'nodes'.
178 void collect_nodes(TPlanNodeType::type node_type, std::vector<ExecNode*>* nodes);
179
180 // Collect all scan node types.
181 void collect_scan_nodes(std::vector<ExecNode*>* nodes);
182
183 virtual void prepare_for_next() {}
184
185 // When the agg node is the scan node direct parent,
186 // we directly return agg object from scan node to agg node,
187 // and don't serialize the agg object.
188 // This improve is cautious, we ensure the correctness firstly.
189 void try_do_aggregate_serde_improve();

191 // Returns a string representation in DFS order of the plan rooted at this.
192 std::string debug_string() const;
193
194 // recursive helper method for generating a string for Debug_string().
195 // implementations should call debug_string(int, std::stringstream) on their children.
196 // Input parameters:
197 // indentation_level: Current level in plan tree.
198 // Output parameters:
199 // out: Stream to accumulate debug string.
200 virtual void debug_string(int indentation_level, std::stringstream* out) const;

202 int id() const { return _id; }
203 TPlanNodeType::type type() const { return _type; }
204 virtual const RowDescriptor& row_desc() const {
205 return _output_row_descriptor ? *_output_row_descriptor : _row_descriptor;
206 }

212 // and add block rows for profile
213 void reached_limit(vectorized::Block* block, bool* eos);
