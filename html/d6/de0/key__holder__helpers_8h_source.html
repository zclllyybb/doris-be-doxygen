28template <bool is_plain_column = false>
29static auto get_key_holder(const IColumn& column, size_t row_num, Arena& arena) {
30 if constexpr (is_plain_column) {
31 return ArenaKeyHolder {column.get_data_at(row_num), arena};
32 } else {
33 const char* begin = nullptr;
34 StringRef serialized = column.serialize_value_into_arena(row_num, arena, begin);
35 assert(serialized.data != nullptr);
36 return SerializedKeyHolder {serialized, arena};
37 }
38}

40template <bool is_plain_column>
41static void deserialize_and_insert(StringRef str, IColumn& data_to) {
42 if constexpr (is_plain_column) {
43 data_to.insert_data(str.data, str.size);
44 } else {
45 data_to.deserialize_and_insert_from_arena(str.data);
46 }
47}
