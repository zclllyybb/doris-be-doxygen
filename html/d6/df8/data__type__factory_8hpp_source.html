47class DataTypeFactory {
48 using DataTypeMap = std::unordered_map<std::string, DataTypePtr>;
49 using InvertedDataTypeMap = std::vector<std::pair<DataTypePtr, std::string>>;

51public:
52 static DataTypeFactory& instance() {
53 static std::once_flag oc;
54 static DataTypeFactory instance;
55 std::call_once(oc, []() {
56 std::unordered_map<std::string, DataTypePtr> base_type_map {
57 {"UInt8", std::make_shared<DataTypeUInt8>()},
58 {"UInt16", std::make_shared<DataTypeUInt16>()},
59 {"UInt32", std::make_shared<DataTypeUInt32>()},
60 {"UInt64", std::make_shared<DataTypeUInt64>()},
61 {"Int8", std::make_shared<DataTypeInt8>()},
62 {"Int16", std::make_shared<DataTypeInt16>()},
63 {"Int32", std::make_shared<DataTypeInt32>()},
64 {"Int64", std::make_shared<DataTypeInt64>()},
65 {"Int128", std::make_shared<DataTypeInt128>()},
66 {"Float32", std::make_shared<DataTypeFloat32>()},
67 {"Float64", std::make_shared<DataTypeFloat64>()},
68 {"Date", std::make_shared<DataTypeDate>()},
69 {"DateV2", std::make_shared<DataTypeDateV2>()},
70 {"DateTime", std::make_shared<DataTypeDateTime>()},
71 {"DateTimeV2", std::make_shared<DataTypeDateTimeV2>()},
72 {"String", std::make_shared<DataTypeString>()},
73 {"Decimal", std::make_shared<DataTypeDecimal<Decimal128>>(27, 9)},
74 {"Decimal32", std::make_shared<DataTypeDecimal<Decimal32>>(
75 BeConsts::MAX_DECIMAL32_PRECISION, 0)},
76 {"Decimal64", std::make_shared<DataTypeDecimal<Decimal64>>(
77 BeConsts::MAX_DECIMAL64_PRECISION, 0)},
78 {"Decimal128", std::make_shared<DataTypeDecimal<Decimal128>>(
79 BeConsts::MAX_DECIMAL128_PRECISION, 0)},
80 {"Decimal128I", std::make_shared<DataTypeDecimal<Decimal128I>>(
81 BeConsts::MAX_DECIMAL128_PRECISION, 0)},
82 {"Jsonb", std::make_shared<DataTypeJsonb>()},
83 {"BitMap", std::make_shared<DataTypeBitMap>()},
84 {"Hll", std::make_shared<DataTypeHLL>()},
85 };
86 for (auto const& [key, val] : base_type_map) {
87 instance.register_data_type(key, val);
88 instance.register_data_type("Array(" + key + ")",
89 std::make_shared<vectorized::DataTypeArray>(val));
90 instance.register_data_type(
91 "Array(Nullable(" + key + "))",
92 std::make_shared<vectorized::DataTypeArray>(
93 std::make_shared<vectorized::DataTypeNullable>(val)));
94 }
95 });
96 return instance;
97 }
98 DataTypePtr get(const std::string& name) { return _data_type_map[name]; }
99 const std::string& get(const DataTypePtr& data_type) const {
100 auto type_ptr = data_type->is_nullable()
101 ? ((DataTypeNullable*)(data_type.get()))->get_nested_type()
102 : data_type;
103 for (const auto& entity : _invert_data_type_map) {
104 if (entity.first->equals(*type_ptr)) {
105 return entity.second;
106 }
107 if (is_decimal(type_ptr) && type_ptr->get_type_id() == entity.first->get_type_id()) {
108 return entity.second;
109 }
110 }
111 return _empty_string;
112 }
