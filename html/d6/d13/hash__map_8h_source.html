162 template <typename Func>
163 void ALWAYS_INLINE merge_to_via_emplace(Self& that, Func&& func) {
164 for (auto it = this->begin(), end = this->end(); it != end; ++it) {
165 typename Self::LookupResult res_it;
166 bool inserted;
167 that.emplace(it->get_first(), res_it, inserted, it.get_hash());
168 func(*lookup_result_get_mapped(res_it), it->get_second(), inserted);
169 }

177 template <typename Func>
178 void ALWAYS_INLINE merge_to_via_find(Self& that, Func&& func) {
179 for (auto it = this->begin(), end = this->end(); it != end; ++it) {
180 auto res_it = that.find(it->get_first(), it.get_hash());
181 if (!res_it)
182 func(it->get_second(), it->get_second(), false);
183 else
184 func(*lookup_result_get_mapped(res_it), it->get_second(), true);
185 }
186 }
