70template <typename T>
71void write_text(Decimal<T> value, UInt32 scale, std::ostream& ostr) {
72 if (value < Decimal<T>(0)) {
73 value *= Decimal<T>(-1);
74 if (value > Decimal<T>(0)) {
75 ostr << '-';
76 }
77 }
78
79 using Type = std::conditional_t<std::is_same_v<T, Int128I>, int128_t, T>;
80 Type whole_part = value;
81
82 if (scale) {
83 whole_part = value / decimal_scale_multiplier<Type>(scale);
84 }
85 if constexpr (std::is_same_v<T, __int128_t> || std::is_same_v<T, Int128I>) {
86 ostr << int128_to_string(whole_part);
87 } else {
88 ostr << whole_part;
89 }
90 if (scale) {
91 ostr << '.';
92 String str_fractional(scale, '0');
93 Int32 pos = scale - 1;
94 if (value < Decimal<T>(0) && pos >= 0) {
95 // Reach here iff this value is a min value of a signed numeric type. It means min<int>()
96 // which is -2147483648 multiply -1 is still -2147483648.
97 str_fractional[pos] += (value / 10 * 10) - value;
98 pos--;
99 value /= 10;
100 value *= Decimal<T>(-1);
101 }
102 for (; pos >= 0; --pos, value /= 10) {
103 str_fractional[pos] += value % 10;
104 }
105 ostr.write(str_fractional.data(), scale);
106 }
107}

249template <typename T>
250bool read_float_text_fast_impl(T& x, ReadBuffer& in) {
251 static_assert(std::is_same_v<T, double> || std::is_same_v<T, float>,
252 "Argument for readFloatTextImpl must be float or double");
253 static_assert('a' > '.' && 'A' > '.' && '\n' < '.' && '\t' < '.' && '\'' < '.' && '"' < '.',
254 "Layout of char is not like ASCII"); //-V590
255
256 StringParser::ParseResult result;
257 x = StringParser::string_to_float<T>(in.position(), in.count(), &result);
258
259 if (UNLIKELY(result != StringParser::PARSE_SUCCESS || std::isnan(x) || std::isinf(x))) {
260 return false;
261 }
262
263 // only to match the is_all_read() check to prevent return null
264 in.position() = in.end();
265 return true;
266}

332template <typename T>
333bool read_decimal_text_impl(T& x, ReadBuffer& buf, UInt32 precision, UInt32 scale) {
334 static_assert(IsDecimalNumber<T>);
335 if constexpr (!std::is_same_v<Decimal128, T>) {
336 StringParser::ParseResult result = StringParser::PARSE_SUCCESS;
337
338 x.value = StringParser::string_to_decimal<typename T::NativeType>(
339 (const char*)buf.position(), buf.count(), precision, scale, &result);
340 // only to match the is_all_read() check to prevent return null
341 buf.position() = buf.end();
342 return result != StringParser::PARSE_FAILURE;
343 } else {
344 auto dv = binary_cast<Int128, DecimalV2Value>(x.value);
345 auto ans = dv.parse_from_str((const char*)buf.position(), buf.count()) == 0;
346
347 // only to match the is_all_read() check to prevent return null
348 buf.position() = buf.end();
349
350 x.value = dv.value();
351 return ans;
352 }

377template <typename T>
378static inline const char* try_read_first_int_text(T& x, const char* pos, const char* end) {
379 const int len = end - pos;
380 int i = 0;
381 while (i < len) {
382 if (pos[i] >= '0' && pos[i] <= '9') {
383 i++;
384 } else {
385 break;
386 }
387 }
388 const char* int_end = pos + i;
389 ReadBuffer in((char*)pos, int_end - pos);
390 const size_t count = in.count();
391 try_read_int_text(x, in);
392 return pos + count;
393}
