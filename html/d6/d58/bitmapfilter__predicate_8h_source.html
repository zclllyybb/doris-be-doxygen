30class BitmapFilterFuncBase {
31public:
32 virtual void insert(const void* data) = 0;
33 virtual void insert_many(const std::vector<const BitmapValue*> bitmaps) = 0;
34 virtual bool empty() = 0;
35 virtual Status assign(BitmapValue* bitmap_value) = 0;
36 virtual void light_copy(BitmapFilterFuncBase* other) { _not_in = other->_not_in; }
37 virtual uint16_t find_fixed_len_olap_engine(const char* data, const uint8* nullmap,
38 uint16_t* offsets, int number) = 0;
39 virtual void find_batch(const char* data, const uint8* nullmap, int number,
40 uint8* results) const = 0;
41 virtual size_t size() const = 0;
42 bool is_not_in() const { return _not_in; }
43 void set_not_in(bool not_in) { _not_in = not_in; }
44 virtual ~BitmapFilterFuncBase() = default;
45
46protected:
47 // true -> not in bitmap, false -> in bitmap
48 bool _not_in {false};
49};

51template <PrimitiveType type>
52class BitmapFilterFunc : public BitmapFilterFuncBase {
53public:
54 using CppType = typename PrimitiveTypeTraits<type>::CppType;

56 BitmapFilterFunc() : _bitmap_value(std::make_shared<BitmapValue>()) {}

65 int number) override;
66

68 uint8* results) const override;
69
70 bool empty() override { return _bitmap_value->empty(); }

73 *_bitmap_value = *bitmap_value;
74 return Status::OK();

85 bool contains_any(CppType left, CppType right) {
86 if (right < 0) {
87 return false;

100template <PrimitiveType type>
101void BitmapFilterFunc<type>::insert(const void* data) {
102 if (data == nullptr) {
103 return;
104 }
105
106 *_bitmap_value |= *reinterpret_cast<const BitmapValue*>(data);
107}

109template <PrimitiveType type>
110void BitmapFilterFunc<type>::insert_many(const std::vector<const BitmapValue*> bitmaps) {
111 if (bitmaps.empty()) {
112 return;
113 }
114 _bitmap_value->fastunion(bitmaps);
115}

117template <PrimitiveType type>
118uint16_t BitmapFilterFunc<type>::find_fixed_len_olap_engine(const char* data, const uint8* nullmap,
119 uint16_t* offsets, int number) {
120 uint16_t new_size = 0;
121 for (int i = 0; i < number; i++) {
122 uint16_t idx = offsets[i];
123 if (nullmap != nullptr && nullmap[idx]) {
124 continue;
125 }
126 if (!find(*((CppType*)data + idx))) {
127 continue;
128 }
129 offsets[new_size++] = idx;
130 }
131 return new_size;
132}
133
134template <PrimitiveType type>
135void BitmapFilterFunc<type>::find_batch(const char* data, const uint8* nullmap, int number,
136 uint8* results) const {
137 for (int i = 0; i < number; i++) {
138 results[i] = false;
139 if (nullmap != nullptr && nullmap[i]) {
140 continue;
141 }
142 if (!find(*((CppType*)data + i))) {
143 continue;
144 }
145 results[i] = true;
146 }
147}

149template <PrimitiveType type>
150void BitmapFilterFunc<type>::light_copy(BitmapFilterFuncBase* bitmapfilter_func) {
151 BitmapFilterFuncBase::light_copy(bitmapfilter_func);
152 auto other_func = reinterpret_cast<BitmapFilterFunc*>(bitmapfilter_func);
153 _bitmap_value = other_func->_bitmap_value;
154}
