72
73 size_t _next_child_index() {
74 if (_next_idx == 0) {
75 _next_idx = 1;
76 if (_queue.size() > 2 && _comp(_queue[1], _queue[2])) ++_next_idx;
77 }
78
79 return _next_idx;
80 }
81
82 void update_top() {
83 size_t size = _queue.size();
84 if (size < 2) return;
85
86 auto begin = _queue.begin();
87
88 size_t child_idx = _next_child_index();
89 auto child_it = begin + child_idx;
90
92 if (_comp(*child_it, *begin)) return;
93
94 _next_idx = 0;
95
96 auto curr_it = begin;
97 auto top = *begin;
98 do {
100 *curr_it = *child_it;
101 curr_it = child_it;
102
103 // recompute the child based off of the updated parent
104 child_idx = 2 * child_idx + 1;
105
106 if (child_idx >= size) break;
107
108 child_it = begin + child_idx;
109
110 if ((child_idx + 1) < size && _comp(*child_it, *(child_it + 1))) {
112 ++child_it;
113 ++child_idx;
114 }
115
117 } while (!(_comp(*child_it, top)));
118 *curr_it = top;
119 }
120};
