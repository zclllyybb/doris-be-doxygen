67class Centroid {
68public:
69 Centroid() : Centroid(0.0, 0.0) {}
70
71 Centroid(Value mean, Weight weight) : _mean(mean), _weight(weight) {}
72
73 Value mean() const noexcept { return _mean; }
74
75 Weight weight() const noexcept { return _weight; }
76
77 Value& mean() noexcept { return _mean; }
78
79 Weight& weight() noexcept { return _weight; }
80
81 void add(const Centroid& c) {
82 DCHECK_GT(c._weight, 0);
83 if (_weight != 0.0) {
84 _weight += c._weight;
85 _mean += c._weight * (c._mean - _mean) / _weight;
86 } else {
87 _weight = c._weight;
88 _mean = c._mean;
89 }
90 }
91
92private:
93 Value _mean = 0;
94 Weight _weight = 0;
95};

97struct CentroidList {
98 CentroidList(const std::vector<Centroid>& s) : iter(s.cbegin()), end(s.cend()) {}
99 std::vector<Centroid>::const_iterator iter;
100 std::vector<Centroid>::const_iterator end;

102 bool advance() { return ++iter != end; }

105class CentroidListComparator {
106public:
107 CentroidListComparator() {}
108
109 bool operator()(const CentroidList& left, const CentroidList& right) const {
110 return left.iter->mean() > right.iter->mean();

227 // merge in a vector of tdigests in the most efficient manner possible
228 // in constant space
229 // works for any value of kHighWater
230 void add(std::vector<const TDigest*>::const_iterator iter,
231 std::vector<const TDigest*>::const_iterator end) {
232 if (iter != end) {
233 auto size = std::distance(iter, end);
234 TDigestQueue pq(TDigestComparator {});
235 for (; iter != end; iter++) {
236 pq.push((*iter));
237 }
238 std::vector<const TDigest*> batch;
239 batch.reserve(size);
240
241 size_t totalSize = 0;
242 while (!pq.empty()) {
243 auto td = pq.top();
244 batch.push_back(td);
245 pq.pop();
246 totalSize += td->totalSize();
247 if (totalSize >= kHighWater || pq.empty()) {
248 mergeProcessed(batch);
249 mergeUnprocessed(batch);
250 processIfNecessary();
251 batch.clear();
252 totalSize = 0;
253 }
254 }
255 updateCumulative();
256 }

279 // return the cdf on the processed values
280 Value cdfProcessed(Value x) const {
281 VLOG_CRITICAL << "cdf value " << x;
282 VLOG_CRITICAL << "processed size " << _processed.size();
283 if (_processed.size() == 0) {
284 // no data to examine
285 VLOG_CRITICAL << "no processed values";
286
287 return 0.0;
288 } else if (_processed.size() == 1) {
289 VLOG_CRITICAL << "one processed value "
290 << " _min " << _min << " _max " << _max;
291 // exactly one centroid, should have _max==_min
292 auto width = _max - _min;
293 if (x < _min) {
294 return 0.0;
295 } else if (x > _max) {
296 return 1.0;
297 } else if (x - _min <= width) {
298 // _min and _max are too close together to do any viable interpolation
299 return 0.5;
300 } else {
301 // interpolate if somehow we have weight > 0 and _max != _min
302 return (x - _min) / (_max - _min);
303 }
304 } else {
305 auto n = _processed.size();
306 if (x <= _min) {
307 VLOG_CRITICAL << "below _min "
308 << " _min " << _min << " x " << x;
309 return 0;
310 }
311
312 if (x >= _max) {
313 VLOG_CRITICAL << "above _max "
314 << " _max " << _max << " x " << x;
315 return 1;
316 }
317
318 // check for the left tail
319 if (x <= mean(0)) {
320 VLOG_CRITICAL << "left tail "
321 << " _min " << _min << " mean(0) " << mean(0) << " x " << x;
322
323 // note that this is different than mean(0) > _min ... this guarantees interpolation works
324 if (mean(0) - _min > 0) {
325 return (x - _min) / (mean(0) - _min) * weight(0) / _processed_weight / 2.0;
326 } else {
327 return 0;
328 }
329 }
330
331 // and the right tail
332 if (x >= mean(n - 1)) {
333 VLOG_CRITICAL << "right tail"
334 << " _max " << _max << " mean(n - 1) " << mean(n - 1) << " x " << x;
335
336 if (_max - mean(n - 1) > 0) {
337 return 1.0 - (_max - x) / (_max - mean(n - 1)) * weight(n - 1) /
338 _processed_weight / 2.0;
339 } else {
340 return 1;
341 }
342 }
343
344 CentroidComparator cc;
345 auto iter =
346 std::upper_bound(_processed.cbegin(), _processed.cend(), Centroid(x, 0), cc);
347
348 auto i = std::distance(_processed.cbegin(), iter);
349 auto z1 = x - (iter - 1)->mean();
350 auto z2 = (iter)->mean() - x;
351 DCHECK_LE(0.0, z1);
352 DCHECK_LE(0.0, z2);
353 VLOG_CRITICAL << "middle "
354 << " z1 " << z1 << " z2 " << z2 << " x " << x;
355
356 return weightedAverage(_cumulative[i - 1], z2, _cumulative[i], z1) / _processed_weight;
357 }
358 }

366 // this returns a quantile on the currently processed values without changing the t-digest
367 // the value will not represent the unprocessed values
368 Value quantileProcessed(Value q) const {
369 if (q < 0 || q > 1) {
370 VLOG_CRITICAL << "q should be in [0,1], got " << q;
371 return NAN;
372 }
373
374 if (_processed.size() == 0) {
375 // no sorted means no data, no way to get a quantile
376 return NAN;
377 } else if (_processed.size() == 1) {
378 // with one data point, all quantiles lead to Rome
379
380 return mean(0);
381 }
382
383 // we know that there are at least two sorted now
384 auto n = _processed.size();
385
386 // if values were stored in a sorted array, index would be the offset we are Weighterested in
387 const auto index = q * _processed_weight;
388
389 // at the boundaries, we return _min or _max
390 if (index <= weight(0) / 2.0) {
391 DCHECK_GT(weight(0), 0);
392 return _min + 2.0 * index / weight(0) * (mean(0) - _min);
393 }
394
395 auto iter = std::lower_bound(_cumulative.cbegin(), _cumulative.cend(), index);
396
397 if (iter + 1 != _cumulative.cend()) {
398 auto i = std::distance(_cumulative.cbegin(), iter);
399 auto z1 = index - *(iter - 1);
400 auto z2 = *(iter)-index;
401 // VLOG_CRITICAL << "z2 " << z2 << " index " << index << " z1 " << z1;
402 return weightedAverage(mean(i - 1), z2, mean(i), z1);
403 }
404
405 DCHECK_LE(index, _processed_weight);
406 DCHECK_GE(index, _processed_weight - weight(n - 1) / 2.0);
407
408 auto z1 = index - _processed_weight - weight(n - 1) / 2.0;
409 auto z2 = weight(n - 1) / 2 - z1;
410 return weightedAverage(mean(n - 1), z1, _max, z2);
411 }

431 void add(std::vector<Centroid>::const_iterator iter,
432 std::vector<Centroid>::const_iterator end) {
433 while (iter != end) {
434 const size_t diff = std::distance(iter, end);
435 const size_t room = _max_unprocessed - _unprocessed.size();
436 auto mid = iter + std::min(diff, room);
437 while (iter != mid) _unprocessed.push_back(*(iter++));
438 if (_unprocessed.size() >= _max_unprocessed) {
439 process();
440 }
441 }

585 // merge all processed centroids together into a single sorted vector
586 void mergeProcessed(const std::vector<const TDigest*>& tdigests) {
587 if (tdigests.size() == 0) return;
588
589 size_t total = 0;
590 CentroidListQueue pq(CentroidListComparator {});
591 for (auto& td : tdigests) {
592 auto& sorted = td->_processed;
593 auto size = sorted.size();
594 if (size > 0) {
595 pq.push(CentroidList(sorted));
596 total += size;
597 _processed_weight += td->_processed_weight;
598 }
599 }
600 if (total == 0) return;
601
602 if (_processed.size() > 0) {
603 pq.push(CentroidList(_processed));
604 total += _processed.size();
605 }
606
607 std::vector<Centroid> sorted;
608 VLOG_CRITICAL << "total " << total;
609 sorted.reserve(total);
610
611 while (!pq.empty()) {
612 auto best = pq.top();
613 pq.pop();
614 sorted.push_back(*(best.iter));
615 if (best.advance()) pq.push(best);
616 }
617 _processed = std::move(sorted);
618 if (_processed.size() > 0) {
619 _min = std::min(_min, _processed[0].mean());
620 _max = std::max(_max, (_processed.cend() - 1)->mean());
621 }
622 }

630 void updateCumulative() {
631 const auto n = _processed.size();
632 _cumulative.clear();
633 _cumulative.reserve(n + 1);
634 auto previous = 0.0;
635 for (Index i = 0; i < n; i++) {
636 auto current = weight(i);
637 auto halfCurrent = current / 2.0;
638 _cumulative.push_back(previous + halfCurrent);
639 previous = previous + current;
640 }
641 _cumulative.push_back(previous);
642 }

644 // merges _unprocessed centroids and _processed centroids together and processes them
645 // when complete, _unprocessed will be empty and _processed will have at most _max_processed centroids
646 void process() {
647 CentroidComparator cc;
648 RadixSort<TDigestRadixSortTraits>::executeLSD(_unprocessed.data(), _unprocessed.size());
649 auto count = _unprocessed.size();
650 _unprocessed.insert(_unprocessed.end(), _processed.cbegin(), _processed.cend());
651 std::inplace_merge(_unprocessed.begin(), _unprocessed.begin() + count, _unprocessed.end(),
652 cc);
653
654 _processed_weight += _unprocessed_weight;
655 _unprocessed_weight = 0;
656 _processed.clear();
657
658 _processed.push_back(_unprocessed[0]);
659 Weight wSoFar = _unprocessed[0].weight();
660 Weight wLimit = _processed_weight * integratedQ(1.0);
661
662 auto end = _unprocessed.end();
663 for (auto iter = _unprocessed.cbegin() + 1; iter < end; iter++) {
664 auto& centroid = *iter;
665 Weight projectedW = wSoFar + centroid.weight();
666 if (projectedW <= wLimit) {
667 wSoFar = projectedW;
668 (_processed.end() - 1)->add(centroid);
669 } else {
670 auto k1 = integratedLocation(wSoFar / _processed_weight);
671 wLimit = _processed_weight * integratedQ(k1 + 1.0);
672 wSoFar += centroid.weight();
673 _processed.emplace_back(centroid);
674 }
675 }
676 _unprocessed.clear();
677 _min = std::min(_min, _processed[0].mean());
678 VLOG_CRITICAL << "new _min " << _min;
679 _max = std::max(_max, (_processed.cend() - 1)->mean());
680 VLOG_CRITICAL << "new _max " << _max;
681 updateCumulative();
682 }

686 size_t checkWeights(const std::vector<Centroid>& sorted, Value total) {
687 size_t badWeight = 0;
688 auto k1 = 0.0;
689 auto q = 0.0;
690 for (auto iter = sorted.cbegin(); iter != sorted.cend(); iter++) {
691 auto w = iter->weight();
692 auto dq = w / total;
693 auto k2 = integratedLocation(q + dq);
694 if (k2 - k1 > 1 && w != 1) {
695 VLOG_CRITICAL << "Oversize centroid at " << std::distance(sorted.cbegin(), iter)

705 }
706 q += dq;
707 k1 = k2;
708 }
709
710 return badWeight;
711 }

731 return _compression * (std::asin(2.0 * q - 1.0) + M_PI / 2) / M_PI;
732 }

736 }

743 static Value weightedAverage(Value x1, Value w1, Value x2, Value w2) {
744 return (x1 <= x2) ? weightedAverageSorted(x1, w1, x2, w2)
745 : weightedAverageSorted(x2, w2, x1, w1);

755 static Value weightedAverageSorted(Value x1, Value w1, Value x2, Value w2) {
756 DCHECK_LE(x1, x2);
757 const Value x = (x1 * w1 + x2 * w2) / (w1 + w2);
758 return std::max(x1, std::min(x, x2));
759 }

775 static Value quantile(Value index, Value previousIndex, Value nextIndex, Value previousMean,
776 Value nextMean) {
777 const auto delta = nextIndex - previousIndex;
778 const auto previousWeight = (nextIndex - index) / delta;
779 const auto nextWeight = (index - previousIndex) / delta;
