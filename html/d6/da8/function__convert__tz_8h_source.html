33template <typename DateValueType, typename ArgType>
34struct ConvertTZImpl {
35 using ColumnType = std::conditional_t<
36 std::is_same_v<DateV2Value<DateV2ValueType>, DateValueType>, ColumnDateV2,
37 std::conditional_t<std::is_same_v<DateV2Value<DateTimeV2ValueType>, DateValueType>,
38 ColumnDateTimeV2, ColumnDateTime>>;
39 using NativeType = std::conditional_t<
40 std::is_same_v<DateV2Value<DateV2ValueType>, DateValueType>, UInt32,
41 std::conditional_t<std::is_same_v<DateV2Value<DateTimeV2ValueType>, DateValueType>,
42 UInt64, Int64>>;
43 using ReturnDateType = std::conditional_t<std::is_same_v<VecDateTimeValue, DateValueType>,
44 VecDateTimeValue, DateV2Value<DateTimeV2ValueType>>;
45 using ReturnNativeType =
46 std::conditional_t<std::is_same_v<VecDateTimeValue, DateValueType>, Int64, UInt64>;
47 using ReturnColumnType = std::conditional_t<std::is_same_v<VecDateTimeValue, DateValueType>,
48 ColumnDateTime, ColumnDateTimeV2>;
49
50 static void execute(FunctionContext* context, const ColumnType* date_column,
51 const ColumnString* from_tz_column, const ColumnString* to_tz_column,
52 ReturnColumnType* result_column, NullMap& result_null_map,
53 size_t input_rows_count) {
54 auto convert_ctx = reinterpret_cast<ConvertTzCtx*>(
55 context->get_function_state(FunctionContext::FunctionStateScope::THREAD_LOCAL));
56 std::map<std::string, cctz::time_zone> time_zone_cache_;
57 auto& time_zone_cache = convert_ctx ? convert_ctx->time_zone_cache : time_zone_cache_;
58 for (size_t i = 0; i < input_rows_count; i++) {
59 if (result_null_map[i]) {
60 result_column->insert_default();
61 continue;
62 }
63
64 auto from_tz = from_tz_column->get_data_at(i).to_string();
65 auto to_tz = to_tz_column->get_data_at(i).to_string();
66
67 DateValueType ts_value =
68 binary_cast<NativeType, DateValueType>(date_column->get_element(i));
69 int64_t timestamp;
70
71 if (time_zone_cache.find(from_tz) == time_zone_cache.cend()) {
72 if (!TimezoneUtils::find_cctz_time_zone(from_tz, time_zone_cache[from_tz])) {
73 result_null_map[i] = true;
74 result_column->insert_default();
75 continue;
76 }
77 }
78
79 if (time_zone_cache.find(to_tz) == time_zone_cache.cend()) {
80 if (!TimezoneUtils::find_cctz_time_zone(to_tz, time_zone_cache[to_tz])) {
81 result_null_map[i] = true;
82 result_column->insert_default();
83 continue;
84 }
85 }
86
87 if (!ts_value.unix_timestamp(&timestamp, time_zone_cache[from_tz])) {
88 result_null_map[i] = true;
89 result_column->insert_default();
90 continue;
91 }
92
93 ReturnDateType ts_value2;
94 if (!ts_value2.from_unixtime(timestamp, time_zone_cache[to_tz])) {
95 result_null_map[i] = true;
96 result_column->insert_default();
97 continue;
98 }
99
100 result_column->insert(binary_cast<ReturnDateType, ReturnNativeType>(ts_value2));
101 }
102 }
103
104 static DataTypes get_variadic_argument_types() {
105 return {std::make_shared<ArgType>(), std::make_shared<DataTypeString>(),
106 std::make_shared<DataTypeString>()};
107 }
108};

110template <typename Transform, typename DateType>
111class FunctionConvertTZ : public IFunction {
112public:
113 static constexpr auto name = "convert_tz";
114
115 static FunctionPtr create() { return std::make_shared<FunctionConvertTZ>(); }
116
117 String get_name() const override { return name; }
118
119 size_t get_number_of_arguments() const override { return 3; }
120
121 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
122 if constexpr (std::is_same_v<DateType, DataTypeDateTime> ||
123 std::is_same_v<DateType, DataTypeDate>) {
124 return make_nullable(std::make_shared<DataTypeDateTime>());
125 } else {
126 return make_nullable(std::make_shared<DataTypeDateTimeV2>());
127 }
128 }
129
130 bool is_variadic() const override { return true; }
131
132 DataTypes get_variadic_argument_types_impl() const override {
133 return Transform::get_variadic_argument_types();
134 }
135
136 bool use_default_implementation_for_constants() const override { return true; }
137 bool use_default_implementation_for_nulls() const override { return false; }
138
139 Status open(FunctionContext* context, FunctionContext::FunctionStateScope scope) override {
140 if (scope != FunctionContext::THREAD_LOCAL) {
141 return Status::OK();
142 }
143 context->set_function_state(scope, std::make_shared<ConvertTzCtx>());
144 return Status::OK();
145 }
146
147 Status close(FunctionContext* context, FunctionContext::FunctionStateScope scope) override {
148 return Status::OK();
149 }
150
151 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
152 size_t result, size_t input_rows_count) override {
153 auto result_null_map_column = ColumnUInt8::create(input_rows_count, 0);
154
155 ColumnPtr argument_columns[3];
156
157 for (int i = 0; i < 3; ++i) {
158 argument_columns[i] =
159 block.get_by_position(arguments[i]).column->convert_to_full_column_if_const();
160 if (auto* nullable = check_and_get_column<ColumnNullable>(*argument_columns[i])) {
161 // Danger: Here must dispose the null map data first! Because
162 // argument_columns[i]=nullable->get_nested_column_ptr(); will release the mem
163 // of column nullable mem of null map
164 VectorizedUtils::update_null_map(result_null_map_column->get_data(),
165 nullable->get_null_map_data());
166 argument_columns[i] = nullable->get_nested_column_ptr();
167 }
168 }
169
170 if constexpr (std::is_same_v<DateType, DataTypeDateTime> ||
171 std::is_same_v<DateType, DataTypeDate>) {
172 auto result_column = ColumnDateTime::create();
173 Transform::execute(context,
174 assert_cast<const ColumnDateTime*>(argument_columns[0].get()),
175 assert_cast<const ColumnString*>(argument_columns[1].get()),
176 assert_cast<const ColumnString*>(argument_columns[2].get()),
177 assert_cast<ColumnDateTime*>(result_column.get()),
178 assert_cast<ColumnUInt8*>(result_null_map_column.get())->get_data(),
179 input_rows_count);
180 block.get_by_position(result).column = ColumnNullable::create(
181 std::move(result_column), std::move(result_null_map_column));
182 } else if constexpr (std::is_same_v<DateType, DataTypeDateV2>) {
183 auto result_column = ColumnDateTimeV2::create();
184 Transform::execute(context, assert_cast<const ColumnDateV2*>(argument_columns[0].get()),
185 assert_cast<const ColumnString*>(argument_columns[1].get()),
186 assert_cast<const ColumnString*>(argument_columns[2].get()),
187 assert_cast<ColumnDateTimeV2*>(result_column.get()),
188 assert_cast<ColumnUInt8*>(result_null_map_column.get())->get_data(),
189 input_rows_count);
190 block.get_by_position(result).column = ColumnNullable::create(
191 std::move(result_column), std::move(result_null_map_column));
192 } else {
193 auto result_column = ColumnDateTimeV2::create();
194 Transform::execute(context,
195 assert_cast<const ColumnDateTimeV2*>(argument_columns[0].get()),
196 assert_cast<const ColumnString*>(argument_columns[1].get()),
197 assert_cast<const ColumnString*>(argument_columns[2].get()),
198 assert_cast<ColumnDateTimeV2*>(result_column.get()),
199 assert_cast<ColumnUInt8*>(result_null_map_column.get())->get_data(),
200 input_rows_count);
201 block.get_by_position(result).column = ColumnNullable::create(
202 std::move(result_column), std::move(result_null_map_column));
203 }
204
205 return Status::OK();
206 }
207};
