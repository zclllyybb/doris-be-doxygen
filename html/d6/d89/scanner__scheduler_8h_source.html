27// Responsible for the scheduling and execution of all Scanners of a BE node.
28// ScannerScheduler has two types of thread pools:
29// 1. Scheduling thread pool
30// Responsible for Scanner scheduling.
31// A set of Scanners for a query will be encapsulated into a ScannerContext
32// and submitted to the ScannerScheduler's scheduling queue.
33// There are multiple scheduling queues in ScannerScheduler, and each scheduling queue
34// is handled by a scheduling thread.
35// The scheduling thread is scheduled in granularity of ScannerContext,
36// that is, a group of Scanners in a ScannerContext are scheduled at a time.
37//
38//2. Execution thread pool
39// The scheduling thread will submit the Scanners selected from the ScannerContext
40// to the execution thread pool to do the actual scan task.
41// Each Scanner will act as a producer, read a group of blocks and put them into
42// the corresponding block queue.
43// The corresponding ScanNode will act as a consumer to consume blocks from the block queue.
44class ScannerScheduler {
45public:
46 ScannerScheduler();
47 ~ScannerScheduler();
48
49 Status init(ExecEnv* env);
50
51 Status submit(ScannerContext* ctx);
52
53 std::unique_ptr<ThreadPoolToken> new_limited_scan_pool_token(ThreadPool::ExecutionMode mode,
54 int max_concurrency);
55
56private:
57 // scheduling thread function
58 void _schedule_thread(int queue_id);
59 // schedule scanners in a certain ScannerContext
60 void _schedule_scanners(ScannerContext* ctx);
61 // execution thread function
62 void _scanner_scan(ScannerScheduler* scheduler, ScannerContext* ctx, VScanner* scanner);
63
64private:
65 // Scheduling queue number.
66 // TODO: make it configurable.
67 static const int QUEUE_NUM = 4;
68 // The ScannerContext will be submitted to the pending queue roundrobin.
69 // _queue_idx pointer to the current queue.
70 // The scheduler thread will take ctx from pending queue, schedule it,
71 // and put it to the _scheduling_map.
72 // If any scanner finish, it will take ctx from and put it to pending queue again.
73 std::atomic_int _queue_idx = {0};
74 BlockingQueue<ScannerContext*>** _pending_queues;
75
76 // scheduling thread pool
77 std::unique_ptr<ThreadPool> _scheduler_pool;
78 // execution thread pool
79 // _local_scan_thread_pool is for local scan task(typically, olap scanner)
80 // _remote_scan_thread_pool is for remote scan task(cold data on s3, hdfs, etc.)
81 // _limited_scan_thread_pool is a special pool for queries with resource limit
82 std::unique_ptr<PriorityThreadPool> _local_scan_thread_pool;
83 std::unique_ptr<ThreadPool> _remote_scan_thread_pool;
84 std::unique_ptr<ThreadPool> _limited_scan_thread_pool;
85
86 // true is the scheduler is closed.
87 std::atomic_bool _is_closed = {false};
88 bool _is_init = false;
89};
