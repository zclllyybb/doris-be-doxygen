29class FunctionArrayRemove : public IFunction {
30public:
31 static constexpr auto name = "array_remove";
32 static FunctionPtr create() { return std::make_shared<FunctionArrayRemove>(); }
33
35 String get_name() const override { return name; }
36
37 bool is_variadic() const override { return false; }
38
39 size_t get_number_of_arguments() const override { return 2; }

41 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
42 DCHECK(is_array(arguments[0]))
43 << "First argument for function: " << name
44 << " should be DataTypeArray but it has type " << arguments[0]->get_name() << ".";
45 return arguments[0];

48 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
49 size_t result, size_t input_rows_count) override {
50 // For default implementation of nulls args
51 ColumnsWithTypeAndName args = {block.get_by_position(arguments[0]),
52 block.get_by_position(arguments[1])};
53
54 auto res_column = _execute_non_nullable(args, input_rows_count);
55 if (!res_column) {
56 return Status::RuntimeError(
57 fmt::format("unsupported types for function {}({}, {})", get_name(),
58 block.get_by_position(arguments[0]).type->get_name(),
59 block.get_by_position(arguments[1]).type->get_name()));
60 }
61 DCHECK_EQ(args[0].column->size(), res_column->size());
62 block.replace_by_position(result, std::move(res_column));
63 return Status::OK();
64 }

66private:
67 template <typename NestedColumnType, typename RightColumnType>
68 ColumnPtr _execute_number(const ColumnArray::Offsets64& offsets, const IColumn& nested_column,
69 const IColumn& right_column, const UInt8* nested_null_map) {
70 // check array nested column type and get data
71 const auto& src_data = reinterpret_cast<const NestedColumnType&>(nested_column).get_data();
72
73 // check target column type and get data
74 const auto& target_data = reinterpret_cast<const RightColumnType&>(right_column).get_data();
75
76 PaddedPODArray<UInt8>* dst_null_map = nullptr;
77 MutableColumnPtr array_nested_column = nullptr;
78 IColumn* dst_column;
79 if (nested_null_map) {
80 auto dst_nested_column =
81 ColumnNullable::create(nested_column.clone_empty(), ColumnUInt8::create());
82 array_nested_column = dst_nested_column->get_ptr();
83 dst_column = dst_nested_column->get_nested_column_ptr();
84 dst_null_map = &dst_nested_column->get_null_map_data();
85 dst_null_map->reserve(offsets.back());
86 } else {
87 auto dst_nested_column = nested_column.clone_empty();
88 array_nested_column = dst_nested_column->get_ptr();
89 dst_column = dst_nested_column;
90 }
91
92 auto& dst_data = reinterpret_cast<NestedColumnType&>(*dst_column).get_data();
93 dst_data.reserve(offsets.back());
94
95 auto dst_offsets_column = ColumnArray::ColumnOffsets::create();
96 auto& dst_offsets = dst_offsets_column->get_data();
97 dst_offsets.reserve(offsets.size());
98
99 size_t cur = 0;
100 for (size_t row = 0; row < offsets.size(); ++row) {
101 size_t off = offsets[row - 1];
102 size_t len = offsets[row] - off;
103
104 if (len == 0) {
105 // case: array:[], target:1 ==> []
106 dst_offsets.push_back(cur);
107 continue;
108 }
109
110 size_t count = 0;
111 for (size_t pos = 0; pos < len; ++pos) {
112 if (nested_null_map && nested_null_map[off + pos]) {
113 // case: array:[Null], target:1 ==> [Null]
114 dst_data.push_back(typename NestedColumnType::value_type());
115 dst_null_map->push_back(1);
116 continue;
117 }
118
119 if (src_data[off + pos] == target_data[row]) {
120 ++count;
121 } else {
122 dst_data.push_back(src_data[off + pos]);
123 if (nested_null_map) {
124 dst_null_map->push_back(0);
125 }
126 }
127 }
128
129 cur += len - count;
130 dst_offsets.push_back(cur);
131 }
132
133 auto dst =
134 ColumnArray::create(std::move(array_nested_column), std::move(dst_offsets_column));
135 return dst;
136 }

138 ColumnPtr _execute_string(const ColumnArray::Offsets64& offsets, const IColumn& nested_column,
139 const IColumn& right_column, const UInt8* nested_null_map) {
140 // check array nested column type and get data
141 const auto& src_offs = reinterpret_cast<const ColumnString&>(nested_column).get_offsets();
142 const auto& src_chars = reinterpret_cast<const ColumnString&>(nested_column).get_chars();
143
144 // check right column type and get data
145 const auto& target_offs = reinterpret_cast<const ColumnString&>(right_column).get_offsets();
146 const auto& target_chars = reinterpret_cast<const ColumnString&>(right_column).get_chars();
147
148 PaddedPODArray<UInt8>* dst_null_map = nullptr;
149 MutableColumnPtr array_nested_column = nullptr;
150 IColumn* dst_column;
151 if (nested_null_map) {
152 auto dst_nested_column =
153 ColumnNullable::create(nested_column.clone_empty(), ColumnUInt8::create());
154 array_nested_column = dst_nested_column->get_ptr();
155 dst_column = dst_nested_column->get_nested_column_ptr();
156 dst_null_map = &dst_nested_column->get_null_map_data();
157 dst_null_map->reserve(offsets.back());
158 } else {
159 auto dst_nested_column = nested_column.clone_empty();
160 array_nested_column = dst_nested_column->get_ptr();
161 dst_column = dst_nested_column;
162 }
163
164 auto& dst_offs = reinterpret_cast<ColumnString&>(*dst_column).get_offsets();
165 auto& dst_chars = reinterpret_cast<ColumnString&>(*dst_column).get_chars();
166 dst_offs.reserve(src_offs.size());
167 dst_chars.reserve(src_offs.back());
168
169 auto dst_offsets_column = ColumnArray::ColumnOffsets::create();
170 auto& dst_offsets = dst_offsets_column->get_data();
171 dst_offsets.reserve(offsets.size());
172
173 size_t cur = 0;
174 for (size_t row = 0; row < offsets.size(); ++row) {
175 size_t off = offsets[row - 1];
176 size_t len = offsets[row] - off;
177
178 if (len == 0) {
179 // case: array:[], target:'str' ==> []
180 dst_offsets.push_back(cur);
181 continue;
182 }
183
184 size_t target_off = target_offs[row - 1];
185 size_t target_len = target_offs[row] - target_off;
186
187 size_t count = 0;
188 for (size_t pos = 0; pos < len; ++pos) {
189 if (nested_null_map && nested_null_map[off + pos]) {
190 // case: array:[Null], target:'str' ==> [Null]
191 // dst_chars.push_back(0);
192 dst_offs.push_back(dst_offs.back());
193 dst_null_map->push_back(1);
194 continue;
195 }
196
197 size_t src_pos = src_offs[pos + off - 1];
198 size_t src_len = src_offs[pos + off] - src_pos;
199 const char* src_raw_v = reinterpret_cast<const char*>(&src_chars[src_pos]);
200 const char* target_raw_v = reinterpret_cast<const char*>(&target_chars[target_off]);
201
202 if (std::string_view(src_raw_v, src_len) ==
203 std::string_view(target_raw_v, target_len)) {
204 ++count;
205 } else {
206 const size_t old_size = dst_chars.size();
207 const size_t new_size = old_size + src_len;
208 dst_chars.resize(new_size);
209 memcpy(&dst_chars[old_size], &src_chars[src_pos], src_len);
210 dst_offs.push_back(new_size);
211 if (nested_null_map) {
212 dst_null_map->push_back(0);
213 }
214 }
215 }
216
217 cur += len - count;
218 dst_offsets.push_back(cur);
219 }
220
221 auto dst =
222 ColumnArray::create(std::move(array_nested_column), std::move(dst_offsets_column));
223 return dst;
224 }

226 template <typename NestedColumnType>
227 ColumnPtr _execute_number_expanded(const ColumnArray::Offsets64& offsets,
228 const IColumn& nested_column, const IColumn& right_column,
229 const UInt8* nested_null_map) {
230 if (check_column<ColumnUInt8>(right_column)) {
231 return _execute_number<NestedColumnType, ColumnUInt8>(offsets, nested_column,
232 right_column, nested_null_map);
233 } else if (check_column<ColumnInt8>(right_column)) {
234 return _execute_number<NestedColumnType, ColumnInt8>(offsets, nested_column,
235 right_column, nested_null_map);
236 } else if (check_column<ColumnInt16>(right_column)) {
237 return _execute_number<NestedColumnType, ColumnInt16>(offsets, nested_column,
238 right_column, nested_null_map);
239 } else if (check_column<ColumnInt32>(right_column)) {
240 return _execute_number<NestedColumnType, ColumnInt32>(offsets, nested_column,
241 right_column, nested_null_map);
242 } else if (right_column.is_date_type()) {
243 return _execute_number<NestedColumnType, ColumnDate>(offsets, nested_column,
244 right_column, nested_null_map);
245 } else if (right_column.is_datetime_type()) {
246 return _execute_number<NestedColumnType, ColumnDateTime>(offsets, nested_column,
247 right_column, nested_null_map);
248 } else if (check_column<ColumnInt64>(right_column)) {
249 return _execute_number<NestedColumnType, ColumnInt64>(offsets, nested_column,
250 right_column, nested_null_map);
251 } else if (check_column<ColumnInt128>(right_column)) {
252 return _execute_number<NestedColumnType, ColumnInt128>(offsets, nested_column,
253 right_column, nested_null_map);
254 } else if (check_column<ColumnFloat32>(right_column)) {
255 return _execute_number<NestedColumnType, ColumnFloat32>(offsets, nested_column,
256 right_column, nested_null_map);
257 } else if (check_column<ColumnFloat64>(right_column)) {
258 return _execute_number<NestedColumnType, ColumnFloat64>(offsets, nested_column,
259 right_column, nested_null_map);
260 } else if (check_column<ColumnDecimal128>(right_column)) {
261 return _execute_number<NestedColumnType, ColumnDecimal128>(
262 offsets, nested_column, right_column, nested_null_map);
263 }
264 return nullptr;
265 }

267 ColumnPtr _execute_non_nullable(const ColumnsWithTypeAndName& arguments,
268 size_t input_rows_count) {
269 // check array nested column type and get data
270 auto left_column = arguments[0].column->convert_to_full_column_if_const();
271 const auto& array_column = reinterpret_cast<const ColumnArray&>(*left_column);
272 const auto& offsets = array_column.get_offsets();
273 DCHECK(offsets.size() == input_rows_count);
274
275 // check array right column type and get data
276 const auto right_column = arguments[1].column->convert_to_full_column_if_const();
277
278 const UInt8* nested_null_map = nullptr;
279 ColumnPtr nested_column = nullptr;
280 if (is_column_nullable(array_column.get_data())) {
281 const auto& nested_null_column =
282 reinterpret_cast<const ColumnNullable&>(array_column.get_data());
283 nested_null_map = nested_null_column.get_null_map_column().get_data().data();
284 nested_column = nested_null_column.get_nested_column_ptr();
285 } else {
286 nested_column = array_column.get_data_ptr();
287 }
288
289 auto left_element_type = remove_nullable(
290 assert_cast<const DataTypeArray&>(*arguments[0].type).get_nested_type());
291 auto right_type = remove_nullable((arguments[1]).type);
292
293 ColumnPtr res = nullptr;
294 if (is_string(right_type) && is_string(left_element_type)) {
295 res = _execute_string(offsets, *nested_column, *right_column, nested_null_map);
296 } else if (is_number(right_type) && is_number(left_element_type)) {
297 if (check_column<ColumnUInt8>(*nested_column)) {
298 res = _execute_number_expanded<ColumnUInt8>(offsets, *nested_column, *right_column,
299 nested_null_map);
300 } else if (check_column<ColumnInt8>(*nested_column)) {
301 res = _execute_number_expanded<ColumnInt8>(offsets, *nested_column, *right_column,
302 nested_null_map);
303 } else if (check_column<ColumnInt16>(*nested_column)) {
304 res = _execute_number_expanded<ColumnInt16>(offsets, *nested_column, *right_column,
305 nested_null_map);
306 } else if (check_column<ColumnInt32>(*nested_column)) {
307 res = _execute_number_expanded<ColumnInt32>(offsets, *nested_column, *right_column,
308 nested_null_map);
309 } else if (check_column<ColumnInt64>(*nested_column)) {
310 res = _execute_number_expanded<ColumnInt64>(offsets, *nested_column, *right_column,
311 nested_null_map);
312 } else if (check_column<ColumnInt128>(*nested_column)) {
313 res = _execute_number_expanded<ColumnInt128>(offsets, *nested_column, *right_column,
314 nested_null_map);
315 } else if (check_column<ColumnFloat32>(*nested_column)) {
316 res = _execute_number_expanded<ColumnFloat32>(offsets, *nested_column,
317 *right_column, nested_null_map);
318 } else if (check_column<ColumnFloat64>(*nested_column)) {
319 res = _execute_number_expanded<ColumnFloat64>(offsets, *nested_column,
320 *right_column, nested_null_map);
321 } else if (check_column<ColumnDecimal128>(*nested_column)) {
322 res = _execute_number_expanded<ColumnDecimal128>(offsets, *nested_column,
323 *right_column, nested_null_map);
324 }
325 } else if (is_date_or_datetime(right_type) && is_date_or_datetime(left_element_type)) {
326 if (nested_column->is_date_type()) {
327 res = _execute_number_expanded<ColumnDate>(offsets, *nested_column, *right_column,
328 nested_null_map);
329 } else if (nested_column->is_datetime_type()) {
330 res = _execute_number_expanded<ColumnDateTime>(offsets, *nested_column,
331 *right_column, nested_null_map);
332 }
333 }
334
335 return res;
336 }
