45// Used to compare row with input scan key. Scan key only contains key columns,
46// row contains all key columns, which is superset of key columns.
47// So we should compare the common prefix columns of lhs and rhs.
48//
49// NOTE: if you are not sure if you can use it, please don't use this function.
50template <typename LhsRowType, typename RhsRowType>
51int compare_row_key(const LhsRowType& lhs, const RhsRowType& rhs) {
52 auto cmp_cids = std::min(lhs.schema()->num_column_ids(), rhs.schema()->num_column_ids());
53 for (uint32_t cid = 0; cid < cmp_cids; ++cid) {
54 auto res = lhs.schema()->column(cid)->compare_cell(lhs.cell(cid), rhs.cell(cid));
55 if (res != 0) {
56 return res;
57 }
58 }
59 return 0;
60}
