33class IColumnDummy : public IColumn {
34public:
35 IColumnDummy() : s(0) {}
36 IColumnDummy(size_t s_) : s(s_) {}
37
38public:
39 virtual MutableColumnPtr clone_dummy(size_t s_) const = 0;
40
41 MutableColumnPtr clone_resized(size_t s) const override { return clone_dummy(s); }
42 size_t size() const override { return s; }
43 void insert_default() override { ++s; }
44 void pop_back(size_t n) override { s -= n; }
45 size_t byte_size() const override { return 0; }
46 size_t allocated_bytes() const override { return 0; }
47 int compare_at(size_t, size_t, const IColumn&, int) const override { return 0; }

49 [[noreturn]] Field operator[](size_t) const override {
50 LOG(FATAL) << "Cannot get value from " << get_name();

54 LOG(FATAL) << "Cannot get value from " << get_name();
55 }
56
57 void insert(const Field&) override {
58 LOG(FATAL) << "Cannot insert element into " << get_name();
59 }
60
61 StringRef get_data_at(size_t) const override { return {}; }

63 void insert_data(const char*, size_t) override { ++s; }
64
65 StringRef serialize_value_into_arena(size_t /*n*/, Arena& arena,

67 return {arena.alloc_continue(0, begin), 0};

69
70 const char* deserialize_and_insert_from_arena(const char* pos) override {
71 ++s;

73 }

75 void insert_from(const IColumn&, size_t) override { ++s; }

77 void insert_range_from(const IColumn& /*src*/, size_t /*start*/, size_t length) override {
78 s += length;

80
81 void insert_indices_from(const IColumn& src, const int* indices_begin,
82 const int* indices_end) override {
83 s += (indices_end - indices_begin);
84 }
85
86 ColumnPtr filter(const Filter& filt, ssize_t /*result_size_hint*/) const override {
87 return clone_dummy(count_bytes_in_filter(filt));
88 }
89
90 ColumnPtr permute(const Permutation& perm, size_t limit) const override {
91 if (s != perm.size()) {
92 LOG(FATAL) << "Size of permutation doesn't match size of column.";
93 }
94
95 return clone_dummy(limit ? std::min(s, limit) : s);
96 }
97
98 void get_permutation(bool /*reverse*/, size_t /*limit*/, int /*nan_direction_hint*/,
99 Permutation& res) const override {
100 res.resize(s);
101 for (size_t i = 0; i < s; ++i) res[i] = i;
102 }
103
104 ColumnPtr replicate(const Offsets& offsets) const override {
105 if (s != offsets.size()) {
106 LOG(FATAL) << "Size of offsets doesn't match size of column.";
107 }
108
109 return clone_dummy(offsets.back());
110 }
111
112 MutableColumns scatter(ColumnIndex num_columns, const Selector& selector) const override {
113 if (s != selector.size()) {
114 LOG(FATAL) << "Size of selector doesn't match size of column.";
115 }
116
117 std::vector<size_t> counts(num_columns);
118 for (auto idx : selector) ++counts[idx];
119
120 MutableColumns res(num_columns);
121 for (size_t i = 0; i < num_columns; ++i) res[i] = clone_resized(counts[i]);
122
123 return res;
124 }
125
126 void append_data_by_selector(MutableColumnPtr& res,
127 const IColumn::Selector& selector) const override {
128 size_t num_rows = size();
129
130 if (num_rows < selector.size()) {
131 LOG(FATAL) << fmt::format("Size of selector: {}, is larger than size of column:{}",
132 selector.size(), num_rows);
133 }
134
135 res->reserve(num_rows);
136
137 for (size_t i = 0; i < selector.size(); ++i) res->insert_from(*this, selector[i]);
138 }
139
140 void get_extremes(Field&, Field&) const override {}
141
142 void addSize(size_t delta) { s += delta; }
143
144 bool is_dummy() const override { return true; }
145
146 void replace_column_data(const IColumn& rhs, size_t row, size_t self_row = 0) override {
147 LOG(FATAL) << "should not call the method in column dummy";
148 }
149
150 void replace_column_data_default(size_t self_row = 0) override {
151 LOG(FATAL) << "should not call the method in column dummy";
152 }
153
154protected:
155 size_t s;
156};
