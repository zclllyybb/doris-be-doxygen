54 std::string debug_string() const {
55 std::stringstream ss;
56 ss << "column_name=" << column_name << ", query_op=" << query_op

59 }
60
61 bool is_empty() const { return column_name.empty() && query_value.empty() && query_op.empty(); }

69 return false;
70 }

72 if (column_pred_info.query_op != query_op) {
73 return false;
74 }
75
76 return true;
77 }

253 void _convert_rowcursor_to_short_key(const RowCursor& key, size_t num_keys) {
254 if (_short_key.capacity() == 0) {
255 _short_key.resize(num_keys);
256 for (auto cid = 0; cid < num_keys; cid++) {
257 auto* field = key.schema()->column(cid);
258 _short_key[cid] = Schema::get_column_by_field(*field);
259
260 if (field->type() == OLAP_FIELD_TYPE_DATE) {
261 _short_key[cid]->set_date_type();
262 } else if (field->type() == OLAP_FIELD_TYPE_DATETIME) {
263 _short_key[cid]->set_datetime_type();
264 }
265 }
266 } else {
267 for (int i = 0; i < num_keys; i++) {
268 _short_key[i]->clear();
269 }
270 }
271
272 for (auto cid = 0; cid < num_keys; cid++) {
273 auto field = key.schema()->column(cid);
274 if (field == nullptr) {
275 break;
276 }
277 auto cell = key.cell(cid);
278 if (cell.is_null()) {
279 _short_key[cid]->insert_default();
280 } else {
281 if (field->type() == OLAP_FIELD_TYPE_VARCHAR ||
282 field->type() == OLAP_FIELD_TYPE_CHAR ||
283 field->type() == OLAP_FIELD_TYPE_STRING) {
284 const Slice* slice = reinterpret_cast<const Slice*>(cell.cell_ptr());
285 _short_key[cid]->insert_data(slice->data, slice->size);
286 } else {
287 _short_key[cid]->insert_many_fix_len_data(
288 reinterpret_cast<const char*>(cell.cell_ptr()), 1);
289 }
290 }
291 }
