32template <typename T>
33struct OverlapSetImpl {
34 using ElementNativeType = typename NativeType<typename T::value_type>::Type;
35 using Set = HashSetWithStackMemory<ElementNativeType, DefaultHash<ElementNativeType>, 4>;
36 Set set;
37 void insert_array(const IColumn* column, size_t start, size_t size) {
38 const auto& vec = assert_cast<const T&>(*column).get_data();
39 for (size_t i = start; i < start + size; ++i) {
40 set.insert(vec[i]);
41 }
42 }
43 bool find_any(const IColumn* column, size_t start, size_t size) {
44 const auto& vec = assert_cast<const T&>(*column).get_data();
45 for (size_t i = start; i < start + size; ++i) {
46 if (set.find(vec[i])) {
47 return true;
48 }
49 }
50 return false;
51 }
52};

73class FunctionArraysOverlap : public IFunction {
74public:
75 static constexpr auto name = "arrays_overlap";
76 static FunctionPtr create() { return std::make_shared<FunctionArraysOverlap>(); }
77
79 String get_name() const override { return name; }
80
81 bool use_default_implementation_for_nulls() const override { return false; }
82
83 bool is_variadic() const override { return false; }
84
85 size_t get_number_of_arguments() const override { return 2; }
86
87 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
88 auto left_data_type = remove_nullable(arguments[0]);
89 auto right_data_type = remove_nullable(arguments[1]);
90 DCHECK(is_array(left_data_type)) << arguments[0]->get_name();
91 DCHECK(is_array(right_data_type)) << arguments[1]->get_name();
92 auto left_nested_type = remove_nullable(
93 assert_cast<const DataTypeArray&>(*left_data_type).get_nested_type());
94 auto right_nested_type = remove_nullable(
95 assert_cast<const DataTypeArray&>(*right_data_type).get_nested_type());
96 DCHECK(left_nested_type->equals(*right_nested_type))
97 << "data type " << arguments[0]->get_name() << " not equal with "
98 << arguments[1]->get_name();
99 return make_nullable(std::make_shared<DataTypeUInt8>());
100 }

102 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
103 size_t result, size_t input_rows_count) override {
104 auto left_column =
105 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
106 auto right_column =
107 block.get_by_position(arguments[1]).column->convert_to_full_column_if_const();
108 ColumnArrayExecutionData left_exec_data;
109 ColumnArrayExecutionData right_exec_data;
110
111 Status ret = Status::RuntimeError(
112 fmt::format("execute failed, unsupported types for function {}({}, {})", get_name(),
113 block.get_by_position(arguments[0]).type->get_name(),
114 block.get_by_position(arguments[1]).type->get_name()));
115
116 // extract array column
117 if (!extract_column_array_info(*left_column, left_exec_data) ||
118 !extract_column_array_info(*right_column, right_exec_data)) {
119 return ret;
120 }
121
122 // prepare return column
123 auto dst_nested_col = ColumnVector<UInt8>::create(input_rows_count, 0);
124 auto dst_null_map = ColumnVector<UInt8>::create(input_rows_count, 0);
125 UInt8* dst_null_map_data = dst_null_map->get_data().data();
126
127 // any array is null or any elements in array is null, return null
128 RETURN_IF_ERROR(_execute_nullable(left_exec_data, dst_null_map_data));
129 RETURN_IF_ERROR(_execute_nullable(right_exec_data, dst_null_map_data));
130
131 // execute overlap check
132 if (left_exec_data.nested_col->is_column_string()) {
133 ret = _execute_internal<ColumnString>(left_exec_data, right_exec_data,
134 dst_null_map_data,
135 dst_nested_col->get_data().data());
136 } else if (left_exec_data.nested_col->is_date_type()) {
137 ret = _execute_internal<ColumnDate>(left_exec_data, right_exec_data, dst_null_map_data,
138 dst_nested_col->get_data().data());
139 } else if (left_exec_data.nested_col->is_datetime_type()) {
140 ret = _execute_internal<ColumnDateTime>(left_exec_data, right_exec_data,
141 dst_null_map_data,
142 dst_nested_col->get_data().data());
143 } else if (left_exec_data.nested_col->is_numeric()) {
144 if (check_column<ColumnUInt8>(*left_exec_data.nested_col)) {
145 ret = _execute_internal<ColumnUInt8>(left_exec_data, right_exec_data,
146 dst_null_map_data,
147 dst_nested_col->get_data().data());
148 } else if (check_column<ColumnInt8>(*left_exec_data.nested_col)) {
149 ret = _execute_internal<ColumnInt8>(left_exec_data, right_exec_data,
150 dst_null_map_data,
151 dst_nested_col->get_data().data());
152 } else if (check_column<ColumnInt16>(*left_exec_data.nested_col)) {
153 ret = _execute_internal<ColumnInt16>(left_exec_data, right_exec_data,
154 dst_null_map_data,
155 dst_nested_col->get_data().data());
156 } else if (check_column<ColumnInt32>(*left_exec_data.nested_col)) {
157 ret = _execute_internal<ColumnInt32>(left_exec_data, right_exec_data,
158 dst_null_map_data,
159 dst_nested_col->get_data().data());
160 } else if (check_column<ColumnInt64>(*left_exec_data.nested_col)) {
161 ret = _execute_internal<ColumnInt64>(left_exec_data, right_exec_data,
162 dst_null_map_data,
163 dst_nested_col->get_data().data());
164 } else if (check_column<ColumnInt128>(*left_exec_data.nested_col)) {
165 ret = _execute_internal<ColumnInt128>(left_exec_data, right_exec_data,
166 dst_null_map_data,
167 dst_nested_col->get_data().data());
168 } else if (check_column<ColumnFloat32>(*left_exec_data.nested_col)) {
169 ret = _execute_internal<ColumnFloat32>(left_exec_data, right_exec_data,
170 dst_null_map_data,
171 dst_nested_col->get_data().data());
172 } else if (check_column<ColumnFloat64>(*left_exec_data.nested_col)) {
173 ret = _execute_internal<ColumnFloat64>(left_exec_data, right_exec_data,
174 dst_null_map_data,
175 dst_nested_col->get_data().data());
176 }
177 } else if (left_exec_data.nested_col->is_column_decimal()) {
178 if (check_column<ColumnDecimal128>(*left_exec_data.nested_col)) {
179 ret = _execute_internal<ColumnDecimal128>(left_exec_data, right_exec_data,
180 dst_null_map_data,
181 dst_nested_col->get_data().data());
182 }
183 }
184
185 if (ret == Status::OK()) {
186 block.replace_by_position(result, ColumnNullable::create(std::move(dst_nested_col),
187 std::move(dst_null_map)));
188 }
189
190 return ret;
191 }

193private:
194 Status _execute_nullable(const ColumnArrayExecutionData& data, UInt8* dst_nullmap_data) {
195 for (ssize_t row = 0; row < data.offsets_ptr->size(); ++row) {
196 if (dst_nullmap_data[row]) {
197 continue;
198 }
199
200 if (data.array_nullmap_data && data.array_nullmap_data[row]) {
201 dst_nullmap_data[row] = 1;
202 continue;
203 }
204
205 // any element inside array is NULL, return NULL
206 if (data.nested_nullmap_data) {
207 ssize_t start = (*data.offsets_ptr)[row - 1];
208 ssize_t size = (*data.offsets_ptr)[row] - start;
209 for (ssize_t i = start; i < start + size; ++i) {
210 if (data.nested_nullmap_data[i]) {
211 dst_nullmap_data[row] = 1;
212 break;
213 }
214 }
215 }
216 }
217 return Status::OK();
218 }

220 template <typename T>
221 Status _execute_internal(const ColumnArrayExecutionData& left_data,
222 const ColumnArrayExecutionData& right_data,
223 const UInt8* dst_nullmap_data, UInt8* dst_data) {
224 using ExecutorImpl = OverlapSetImpl<T>;
225 for (ssize_t row = 0; row < left_data.offsets_ptr->size(); ++row) {
226 if (dst_nullmap_data[row]) {
227 continue;
228 }
229
230 ssize_t left_start = (*left_data.offsets_ptr)[row - 1];
231 ssize_t left_size = (*left_data.offsets_ptr)[row] - left_start;
232 ssize_t right_start = (*right_data.offsets_ptr)[row - 1];
233 ssize_t right_size = (*right_data.offsets_ptr)[row] - right_start;
234 if (left_size == 0 || right_size == 0) {
235 dst_data[row] = 0;
236 continue;
237 }
238
239 ExecutorImpl impl;
240 if (right_size < left_size) {
241 impl.insert_array(right_data.nested_col, right_start, right_size);
242 dst_data[row] = impl.find_any(left_data.nested_col, left_start, left_size);
243 } else {
244 impl.insert_array(left_data.nested_col, left_start, left_size);
245 dst_data[row] = impl.find_any(right_data.nested_col, right_start, right_size);
246 }
247 }
248 return Status::OK();
249 }
250};
