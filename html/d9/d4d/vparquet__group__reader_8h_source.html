66 struct PositionDeleteContext {
67 // the filtered rows in current row group
68 const std::vector<int64_t>& delete_rows;
69 // the first row id of current row group in parquet file
70 const int64_t first_row_id;
71 // the number of rows in current row group
72 const int64_t num_rows;
73 const int64_t last_row_id;
74 // current row id to read in the row group
75 int64_t current_row_id;
76 // start index in delete_rows
77 const int64_t start_index;
78 // end index in delete_rows
79 const int64_t end_index;
80 // current index in delete_rows
81 int64_t index;
82 const bool has_filter;
83
84 PositionDeleteContext(const std::vector<int64_t>& delete_rows, const int64_t num_rows,
85 const int64_t first_row_id, const int64_t start_index,
86 const int64_t end_index)
87 : delete_rows(delete_rows),
88 first_row_id(first_row_id),
89 num_rows(num_rows),
90 last_row_id(first_row_id + num_rows),
91 current_row_id(first_row_id),
92 start_index(start_index),
93 end_index(end_index),
94 index(start_index),
95 has_filter(end_index > start_index) {}
96
97 PositionDeleteContext(const int64_t num_rows, const int64_t first_row)
98 : PositionDeleteContext(NO_DELETE, num_rows, first_row, 0, 0) {}
99
100 PositionDeleteContext(const PositionDeleteContext& filter) = default;
101 };
