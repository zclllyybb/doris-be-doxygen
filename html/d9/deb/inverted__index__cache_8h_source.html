105// A handle for a index_searcher from index_searcher lru cache.
106// The handle can ensure that the index_searcher is valid
107// and will not be closed while the holder of the handle is accessing the index_searcher.
108// The handle will automatically release the cache entry when it is destroyed.
109// So the caller need to make sure the handle is valid in lifecycle.
110class InvertedIndexCacheHandle {
111public:
112 InvertedIndexCacheHandle() {}
113 InvertedIndexCacheHandle(Cache* cache, Cache::Handle* handle)
114 : _cache(cache), _handle(handle) {}
115
116 ~InvertedIndexCacheHandle() {
117 if (_handle != nullptr) {
118 CHECK(_cache != nullptr);
119 CHECK(!owned);
120 // only after get_index_searcher call this destructor will
121 // add `config::index_cache_entry_stay_time_after_lookup_s` on last_visit_time,
122 // this is to extend the retention time of the entries hit by lookup.
123 ((InvertedIndexSearcherCache::CacheValue*)_cache->value(_handle))->last_visit_time =
124 UnixMillis() + config::index_cache_entry_stay_time_after_lookup_s * 1000;
125 _cache->release(_handle);
126 }
127 }
128
129 InvertedIndexCacheHandle(InvertedIndexCacheHandle&& other) noexcept {
130 std::swap(_cache, other._cache);
131 std::swap(_handle, other._handle);
132 this->owned = other.owned;
133 this->index_searcher = std::move(other.index_searcher);
134 }
135
136 InvertedIndexCacheHandle& operator=(InvertedIndexCacheHandle&& other) noexcept {
137 std::swap(_cache, other._cache);
138 std::swap(_handle, other._handle);
139 this->owned = other.owned;
140 this->index_searcher = std::move(other.index_searcher);
141 return *this;
142 }
143
144 IndexSearcherPtr get_index_searcher() {
145 if (owned) {
146 return index_searcher;
147 } else {
148 return ((InvertedIndexSearcherCache::CacheValue*)_cache->value(_handle))
149 ->index_searcher;
150 }
151 }
152
153public:
154 // If set to true, the loaded index_searcher will be saved in index_searcher, not in lru cache;
155 bool owned = false;
156 IndexSearcherPtr index_searcher;
157
158private:
159 Cache* _cache = nullptr;
160 Cache::Handle* _handle = nullptr;
161
162 // Don't allow copy and assign
163 DISALLOW_COPY_AND_ASSIGN(InvertedIndexCacheHandle);
164};
