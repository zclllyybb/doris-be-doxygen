106// A handle for a index_searcher from index_searcher lru cache.
107// The handle can ensure that the index_searcher is valid
108// and will not be closed while the holder of the handle is accessing the index_searcher.
109// The handle will automatically release the cache entry when it is destroyed.
110// So the caller need to make sure the handle is valid in lifecycle.
111class InvertedIndexCacheHandle {
112public:
113 InvertedIndexCacheHandle() {}
114 InvertedIndexCacheHandle(Cache* cache, Cache::Handle* handle)
115 : _cache(cache), _handle(handle) {}
116
117 ~InvertedIndexCacheHandle() {
118 if (_handle != nullptr) {
119 CHECK(_cache != nullptr);
120 CHECK(!owned);
121 // only after get_index_searcher call this destructor will
122 // add `config::index_cache_entry_stay_time_after_lookup_s` on last_visit_time,
123 // this is to extend the retention time of the entries hit by lookup.
124 ((InvertedIndexSearcherCache::CacheValue*)_cache->value(_handle))->last_visit_time =
125 UnixMillis() + config::index_cache_entry_stay_time_after_lookup_s * 1000;
126 _cache->release(_handle);
127 }
128 }
129
130 InvertedIndexCacheHandle(InvertedIndexCacheHandle&& other) noexcept {
131 std::swap(_cache, other._cache);
132 std::swap(_handle, other._handle);
133 this->owned = other.owned;
134 this->index_searcher = std::move(other.index_searcher);
135 }
136
137 InvertedIndexCacheHandle& operator=(InvertedIndexCacheHandle&& other) noexcept {
138 std::swap(_cache, other._cache);
139 std::swap(_handle, other._handle);
140 this->owned = other.owned;
141 this->index_searcher = std::move(other.index_searcher);
142 return *this;
143 }
144
145 IndexSearcherPtr get_index_searcher() {
146 if (owned) {
147 return index_searcher;
148 } else {
149 return ((InvertedIndexSearcherCache::CacheValue*)_cache->value(_handle))
150 ->index_searcher;
151 }
152 }
153
154public:
155 // If set to true, the loaded index_searcher will be saved in index_searcher, not in lru cache;
156 bool owned = false;
157 IndexSearcherPtr index_searcher;
158
159private:
160 Cache* _cache = nullptr;
161 Cache::Handle* _handle = nullptr;
162
163 // Don't allow copy and assign
164 DISALLOW_COPY_AND_ASSIGN(InvertedIndexCacheHandle);
165};
