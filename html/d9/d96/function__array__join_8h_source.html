57 static Status execute(Block& block, const ColumnNumbers& arguments, size_t result,
58 const DataTypeArray* data_type_array, const ColumnArray& array) {
59 ColumnPtr src_column =
60 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
61 ColumnArrayExecutionData src;
62 if (!extract_column_array_info(*src_column, src)) {
63 return Status::RuntimeError(fmt::format(
64 "execute failed, unsupported types for function {}({})", "array_join",
65 block.get_by_position(arguments[0]).type->get_name()));
66 }
67
68 ColumnPtr sep_column =
69 block.get_by_position(arguments[1]).column->convert_to_full_column_if_const();
70 ColumnPtr null_replace_column =
71 (arguments.size() > 2 ? block.get_by_position(arguments[2])
72 .column->convert_to_full_column_if_const()
73 : nullptr);
74
75 std::string sep_str = _get_string_from_column(sep_column);
76 std::string null_replace_str = _get_string_from_column(null_replace_column);
77
78 auto nested_type = data_type_array->get_nested_type();
79 auto dest_column_ptr = ColumnString::create();
80 DCHECK(dest_column_ptr != nullptr);
81
82 auto res_val = _execute_by_type(*src.nested_col, *src.offsets_ptr, src.nested_nullmap_data,
83 sep_str, null_replace_str, nested_type, dest_column_ptr);
84 if (!res_val) {
85 return Status::RuntimeError(fmt::format(
86 "execute failed or unsupported types for function {}({},{},{})", "array_join",
87 block.get_by_position(arguments[0]).type->get_name(),
88 block.get_by_position(arguments[1]).type->get_name(),
89 (arguments.size() > 2 ? block.get_by_position(arguments[2]).type->get_name()
90 : "")));
91 }
92
93 block.replace_by_position(result, std::move(dest_column_ptr));
94 return Status::OK();
95 }

98 static std::string _get_string_from_column(const ColumnPtr& column_ptr) {
99 if (!column_ptr) {

108 static void _fill_result_string(const std::string& input_str, const std::string& sep_str,
109 std::string& result_str, bool& is_first_elem) {
110 if (is_first_elem) {
111 result_str.append(input_str);
112 is_first_elem = false;
113 } else {
114 result_str.append(sep_str);
115 result_str.append(input_str);
116 }
117 return;
118 }

120 template <typename ColumnType>
121 static bool _execute_number(const IColumn& src_column,
122 const ColumnArray::Offsets64& src_offsets,
123 const UInt8* src_null_map, const std::string& sep_str,
124 const std::string& null_replace_str, DataTypePtr& nested_type,
125 ColumnString* dest_column_ptr) {
126 using NestType = typename ColumnType::value_type;
127 bool is_decimal = IsDecimalNumber<NestType>;
128
129 const ColumnType* src_data_concrete = reinterpret_cast<const ColumnType*>(&src_column);
130 if (!src_data_concrete) {
131 return false;
132 }
133
134 size_t prev_src_offset = 0;
135 for (auto curr_src_offset : src_offsets) {
136 std::string result_str;
137 bool is_first_elem = true;
138 for (size_t j = prev_src_offset; j < curr_src_offset; ++j) {
139 if (src_null_map && src_null_map[j]) {
140 if (null_replace_str.size() == 0) {
141 continue;
142 } else {
143 _fill_result_string(null_replace_str, sep_str, result_str, is_first_elem);
144 continue;
145 }
146 }
147
148 if (is_decimal) {
149 DecimalV2Value decimal_value =
150 (DecimalV2Value)(src_data_concrete->get_data()[j]);
151 std::string decimal_str = decimal_value.to_string();
152 _fill_result_string(decimal_str, sep_str, result_str, is_first_elem);
153 } else {
154 std::string elem_str = remove_nullable(nested_type)->to_string(src_column, j);
155 _fill_result_string(elem_str, sep_str, result_str, is_first_elem);
156 }
157 }
158
159 dest_column_ptr->insert_data(result_str.c_str(), result_str.size());
160 prev_src_offset = curr_src_offset;
161 }
162
163 return true;
164 }

166 static bool _execute_string(const IColumn& src_column,
167 const ColumnArray::Offsets64& src_offsets,
168 const UInt8* src_null_map, const std::string& sep_str,
169 const std::string& null_replace_str,
170 ColumnString* dest_column_ptr) {
171 const ColumnString* src_data_concrete = reinterpret_cast<const ColumnString*>(&src_column);
172 if (!src_data_concrete) {
173 return false;
174 }
175
176 size_t prev_src_offset = 0;
177 for (auto curr_src_offset : src_offsets) {
178 std::string result_str;

201 static bool _execute_by_type(const IColumn& src_column,
202 const ColumnArray::Offsets64& src_offsets,
203 const UInt8* src_null_map, const std::string& sep_str,
204 const std::string& null_replace_str, DataTypePtr& nested_type,
205 ColumnString* dest_column_ptr) {
206 bool res = false;
207 WhichDataType which(remove_nullable(nested_type));
208 if (which.is_uint8()) {
209 res = _execute_number<ColumnUInt8>(src_column, src_offsets, src_null_map, sep_str,
210 null_replace_str, nested_type, dest_column_ptr);
211 } else if (which.is_int8()) {
212 res = _execute_number<ColumnInt8>(src_column, src_offsets, src_null_map, sep_str,
213 null_replace_str, nested_type, dest_column_ptr);
214 } else if (which.is_int16()) {
215 res = _execute_number<ColumnInt16>(src_column, src_offsets, src_null_map, sep_str,
216 null_replace_str, nested_type, dest_column_ptr);
217 } else if (which.is_int32()) {
218 res = _execute_number<ColumnInt32>(src_column, src_offsets, src_null_map, sep_str,
219 null_replace_str, nested_type, dest_column_ptr);
220 } else if (which.is_int64()) {
221 res = _execute_number<ColumnInt64>(src_column, src_offsets, src_null_map, sep_str,
222 null_replace_str, nested_type, dest_column_ptr);
223 } else if (which.is_int128()) {
224 res = _execute_number<ColumnInt128>(src_column, src_offsets, src_null_map, sep_str,
225 null_replace_str, nested_type, dest_column_ptr);
226 } else if (which.is_float32()) {
227 res = _execute_number<ColumnFloat32>(src_column, src_offsets, src_null_map, sep_str,
228 null_replace_str, nested_type, dest_column_ptr);
229 } else if (which.is_float64()) {
230 res = _execute_number<ColumnFloat64>(src_column, src_offsets, src_null_map, sep_str,
231 null_replace_str, nested_type, dest_column_ptr);
232 } else if (which.is_date()) {
233 res = _execute_number<ColumnDate>(src_column, src_offsets, src_null_map, sep_str,
234 null_replace_str, nested_type, dest_column_ptr);
235 } else if (which.is_date_time()) {
236 res = _execute_number<ColumnDateTime>(src_column, src_offsets, src_null_map, sep_str,
237 null_replace_str, nested_type, dest_column_ptr);
238 } else if (which.is_date_v2()) {
239 res = _execute_number<ColumnDateV2>(src_column, src_offsets, src_null_map, sep_str,
240 null_replace_str, nested_type, dest_column_ptr);
241 } else if (which.is_date_time_v2()) {
242 res = _execute_number<ColumnDateTimeV2>(src_column, src_offsets, src_null_map, sep_str,
243 null_replace_str, nested_type, dest_column_ptr);
244 } else if (which.is_decimal32()) {
245 res = _execute_number<ColumnDecimal32>(src_column, src_offsets, src_null_map, sep_str,
246 null_replace_str, nested_type, dest_column_ptr);
247 } else if (which.is_decimal64()) {
248 res = _execute_number<ColumnDecimal64>(src_column, src_offsets, src_null_map, sep_str,
249 null_replace_str, nested_type, dest_column_ptr);
250 } else if (which.is_decimal128i()) {
251 res = _execute_number<ColumnDecimal128I>(src_column, src_offsets, src_null_map, sep_str,
252 null_replace_str, nested_type,
253 dest_column_ptr);
254 } else if (which.is_decimal128()) {
255 res = _execute_number<ColumnDecimal128>(src_column, src_offsets, src_null_map, sep_str,
256 null_replace_str, nested_type, dest_column_ptr);
257 } else if (which.is_string()) {
258 res = _execute_string(src_column, src_offsets, src_null_map, sep_str, null_replace_str,
259 dest_column_ptr);
260 }
261 return res;
262 }
