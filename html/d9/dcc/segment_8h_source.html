53// A Segment is used to represent a segment in memory format. When segment is
54// generated, it won't be modified, so this struct aimed to help read operation.
55// It will prepare all ColumnReader to create ColumnIterator as needed.
56// And user can create a RowwiseIterator through new_iterator function.
57//
58// NOTE: This segment is used to a specified TabletSchema, when TabletSchema
59// is changed, this segment can not be used any more. For example, after a schema
60// change finished, client should disable all cached Segment for old TabletSchema.
61class Segment : public std::enable_shared_from_this<Segment> {
62public:
63 static Status open(io::FileSystemSPtr fs, const std::string& path, uint32_t segment_id,
64 RowsetId rowset_id, TabletSchemaSPtr tablet_schema,
65 const io::FileReaderOptions& reader_options,
66 std::shared_ptr<Segment>* output);
67
68 ~Segment();
69
70 Status new_iterator(const Schema& schema, const StorageReadOptions& read_options,
71 std::unique_ptr<RowwiseIterator>* iter);
72
73 uint32_t id() const { return _segment_id; }
74
75 RowsetId rowset_id() const { return _rowset_id; }
76
77 uint32_t num_rows() const { return _footer.num_rows(); }
78
79 Status new_column_iterator(const TabletColumn& tablet_column, ColumnIterator** iter);
80
81 Status new_bitmap_index_iterator(const TabletColumn& tablet_column, BitmapIndexIterator** iter);
82
83 Status new_inverted_index_iterator(const TabletColumn& tablet_column,
84 const TabletIndex* index_meta, OlapReaderStatistics* stats,
85 InvertedIndexIterator** iter);
86
87 const ShortKeyIndexDecoder* get_short_key_index() const {
88 DCHECK(_load_index_once.has_called() && _load_index_once.stored_result().ok());
89 return _sk_index_decoder.get();
90 }
91
92 const PrimaryKeyIndexReader* get_primary_key_index() const {
93 DCHECK(_load_index_once.has_called() && _load_index_once.stored_result().ok());
94 return _pk_index_reader.get();
95 }
96
97 Status lookup_row_key(const Slice& key, RowLocation* row_location);
98
99 Status read_key_by_rowid(uint32_t row_id, std::string* key);
100
101 // only used by UT
102 const SegmentFooterPB& footer() const { return _footer; }
103
104 Status load_index();
105
106 Status load_pk_index_and_bf();
107
108 std::string min_key() {
109 DCHECK(_tablet_schema->keys_type() == UNIQUE_KEYS && _footer.has_primary_key_index_meta());
110 return _footer.primary_key_index_meta().min_key();
111 }
112 std::string max_key() {
113 DCHECK(_tablet_schema->keys_type() == UNIQUE_KEYS && _footer.has_primary_key_index_meta());
114 return _footer.primary_key_index_meta().max_key();
115 }
116
117 io::FileReaderSPtr file_reader() { return _file_reader; }

121private:

124 // open segment file and read the minimum amount of necessary information (footer)
125 Status _open();
126 Status _parse_footer();
127 Status _create_column_readers();
128 Status _load_pk_bloom_filter();
129
130private:
131 friend class SegmentIterator;
132 io::FileReaderSPtr _file_reader;
133
134 uint32_t _segment_id;
135 RowsetId _rowset_id;
136 TabletSchemaSPtr _tablet_schema;
137
138 int64_t _meta_mem_usage;
139 SegmentFooterPB _footer;
140
141 // Map from column unique id to column ordinal in footer's ColumnMetaPB
142 // If we can't find unique id from it, it means this segment is created
143 // with an old schema.
144 std::unordered_map<uint32_t, uint32_t> _column_id_to_footer_ordinal;
145
146 // map column unique id ---> column reader
147 // ColumnReader for each column in TabletSchema. If ColumnReader is nullptr,
148 // This means that this segment has no data for that column, which may be added
149 // after this segment is generated.
150 std::map<int32_t, std::unique_ptr<ColumnReader>> _column_readers;
151
152 // used to guarantee that short key index will be loaded at most once in a thread-safe way
153 DorisCallOnce<Status> _load_index_once;
154 // used to guarantee that primary key bloom filter will be loaded at most once in a thread-safe way
155 DorisCallOnce<Status> _load_pk_bf_once;
156 // used to hold short key index page in memory
157 PageHandle _sk_index_handle;
158 // short key index decoder
159 std::unique_ptr<ShortKeyIndexDecoder> _sk_index_decoder;
160 // primary key index reader
161 std::unique_ptr<PrimaryKeyIndexReader> _pk_index_reader;
162 // Segment may be destructed after StorageEngine, in order to exit gracefully.
163 std::shared_ptr<MemTracker> _segment_meta_mem_tracker;
164};
