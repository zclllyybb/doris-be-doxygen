31// Hash function for TNetworkAddress. This function must be called hash_value to be picked
32// up properly by boost.
33inline std::size_t hash_value(const TNetworkAddress& host_port) {
34 uint32_t hash = HashUtil::hash(host_port.hostname.c_str(), host_port.hostname.length(), 0);
35 return HashUtil::hash(&host_port.port, sizeof(host_port.port), hash);
36}

48// find_or_insert(): if the key is present, return the value; if the key is not present,
49// create a new entry (key, default_val) and return default_val.
50
51template <typename K, typename V>
52V* find_or_insert(std::map<K, V>* m, const K& key, const V& default_val) {
53 typename std::map<K, V>::iterator it = m->find(key);
54
55 if (it == m->end()) {
56 it = m->insert(make_pair(key, default_val)).first;
57 }
58
59 return &it->second;
60}

73// find_with_default: if the key is present, return the corresponding value; if the key
74// is not present, return the supplied default value

76template <typename K, typename V>
77const V& find_with_default(const std::map<K, V>& m, const K& key, const V& default_val) {
78 typename std::map<K, V>::const_iterator it = m.find(key);
79
80 if (it == m.end()) {
