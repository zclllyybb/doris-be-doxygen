147class ArrayTypeInfo : public TypeInfo {
148public:
149 explicit ArrayTypeInfo(TypeInfoPtr item_type_info)
150 : _item_type_info(std::move(item_type_info)), _item_size(_item_type_info->size()) {}
151 ~ArrayTypeInfo() override = default;

153 int cmp(const void* left, const void* right) const override {
154 auto l_value = reinterpret_cast<const CollectionValue*>(left);
155 auto r_value = reinterpret_cast<const CollectionValue*>(right);
156 size_t l_length = l_value->length();
157 size_t r_length = r_value->length();
158 size_t cur = 0;
159
160 if (!l_value->has_null() && !r_value->has_null()) {
161 while (cur < l_length && cur < r_length) {
162 int result = _item_type_info->cmp((uint8_t*)(l_value->data()) + cur * _item_size,
163 (uint8_t*)(r_value->data()) + cur * _item_size);
164 if (result != 0) {
165 return result;
166 }
167 ++cur;
168 }
169 } else {
170 while (cur < l_length && cur < r_length) {
171 if (l_value->is_null_at(cur)) {
172 if (!r_value->is_null_at(cur)) { // left is null & right is not null
173 return -1;
174 }
175 } else if (r_value->is_null_at(cur)) { // left is not null & right is null
176 return 1;
177 } else { // both are not null
178 int result =
179 _item_type_info->cmp((uint8_t*)(l_value->data()) + cur * _item_size,
180 (uint8_t*)(r_value->data()) + cur * _item_size);
181 if (result != 0) {
182 return result;
183 }
184 }
185 ++cur;
186 }
187 }
188
189 if (l_length < r_length) {
190 return -1;
191 } else if (l_length > r_length) {
192 return 1;
193 } else {
194 return 0;
195 }
196 }

198 void deep_copy(void* dest, const void* src, MemPool* mem_pool) const override {
199 auto dest_value = reinterpret_cast<CollectionValue*>(dest);
200 auto src_value = reinterpret_cast<const CollectionValue*>(src);
201
202 if (src_value->length() == 0) {
203 new (dest_value) CollectionValue(src_value->length());
204 return;
205 }
206
207 dest_value->set_length(src_value->length());
208
209 size_t item_size = src_value->length() * _item_size;
210 size_t nulls_size = src_value->has_null() ? src_value->length() : 0;
211 dest_value->set_data(mem_pool->allocate(item_size + nulls_size));
212 dest_value->set_has_null(src_value->has_null());
213 dest_value->set_null_signs(src_value->has_null()
214 ? reinterpret_cast<bool*>(dest_value->mutable_data()) +
215 item_size
216 : nullptr);
217
218 // copy null_signs
219 if (src_value->has_null()) {
220 dest_value->copy_null_signs(src_value);
221 }
222
223 // copy item
224 for (uint32_t i = 0; i < src_value->length(); ++i) {
225 if (dest_value->is_null_at(i)) continue;
226 _item_type_info->deep_copy((uint8_t*)(dest_value->mutable_data()) + i * _item_size,
227 (uint8_t*)(src_value->data()) + i * _item_size, mem_pool);
228 }
229 }

231 void direct_copy(void* dest, const void* src) const override {
232 auto dest_value = static_cast<CollectionValue*>(dest);
233 // NOTICE: The address pointed by null_signs of the dest_value can NOT be modified here.
234 auto base = reinterpret_cast<uint8_t*>(dest_value->mutable_null_signs());
235 direct_copy(&base, dest, src);
236 }

238 void direct_copy(uint8_t** base, void* dest, const void* src) const {
239 auto dest_value = static_cast<CollectionValue*>(dest);
240 auto src_value = static_cast<const CollectionValue*>(src);
241
242 auto nulls_size = src_value->has_null() ? src_value->length() : 0;
243 dest_value->set_data(src_value->length() ? (*base + nulls_size) : nullptr);
244 dest_value->set_length(src_value->length());
245 dest_value->set_has_null(src_value->has_null());
246 if (src_value->has_null()) {
247 // direct copy null_signs
248 dest_value->set_null_signs(reinterpret_cast<bool*>(*base));
249 memory_copy(dest_value->mutable_null_signs(), src_value->null_signs(),
250 src_value->length());
251 }
252 *base += nulls_size + src_value->length() * _item_type_info->size();
253
254 // Direct copy item.
255 if (_item_type_info->type() == OLAP_FIELD_TYPE_ARRAY) {
256 for (uint32_t i = 0; i < src_value->length(); ++i) {
257 if (dest_value->is_null_at(i)) {
258 continue;
259 }
260 dynamic_cast<const ArrayTypeInfo*>(_item_type_info.get())
261 ->direct_copy(base, (uint8_t*)(dest_value->mutable_data()) + i * _item_size,
262 (uint8_t*)(src_value->data()) + i * _item_size);
263 }
264 } else {
265 for (uint32_t i = 0; i < src_value->length(); ++i) {
266 if (dest_value->is_null_at(i)) {
267 continue;
268 }
269 auto dest_address = (uint8_t*)(dest_value->mutable_data()) + i * _item_size;
270 auto src_address = (uint8_t*)(src_value->data()) + i * _item_size;
271 if (is_olap_string_type(_item_type_info->type())) {
272 auto dest_slice = reinterpret_cast<Slice*>(dest_address);
273 auto src_slice = reinterpret_cast<const Slice*>(src_address);
274 dest_slice->data = reinterpret_cast<char*>(*base);
275 dest_slice->size = src_slice->size;
276 *base += src_slice->size;
277 }
278 _item_type_info->direct_copy(dest_address, src_address);
279 }
280 }

565 static Status from_string(void* buf, const std::string& scan_key, const int precision,
566 const int scale) {
567 int128_t value = 0;
568
569 const char* value_string = scan_key.c_str();
570 char* end = nullptr;
571 value = strtol(value_string, &end, 10);
572 if (*end != 0) {
573 value = 0;
574 } else if (value > LONG_MIN && value < LONG_MAX) {
575 // use strtol result directly
576 } else {
577 bool is_negative = false;
578 if (*value_string == '-' || *value_string == '+') {
579 if (*(value_string++) == '-') {
580 is_negative = true;
581 }
582 }
583
584 uint128_t current = 0;
585 uint128_t max_int128 = ~((int128_t)(1) << 127);
586 while (*value_string != 0) {
587 if (current > max_int128 / 10) {
588 break;
589 }
590
591 current = current * 10 + (*(value_string++) - '0');
592 }
593 if (*value_string != 0 || (!is_negative && current > max_int128) ||
594 (is_negative && current > max_int128 + 1)) {
595 current = 0;
596 }
597
598 value = is_negative ? -current : current;
599 }
600
601 *reinterpret_cast<PackedInt128*>(buf) = value;
602
603 return Status::OK();
604 }
605 static std::string to_string(const void* src) {
606 char buf[1024];
607 int128_t value = reinterpret_cast<const PackedInt128*>(src)->value;
608 if (value >= std::numeric_limits<int64_t>::min() &&
609 value <= std::numeric_limits<int64_t>::max()) {
610 snprintf(buf, sizeof(buf), "%" PRId64, (int64_t)value);
611 } else {
612 char* current = buf;
613 uint128_t abs_value = value;
614 if (value < 0) {
615 *(current++) = '-';
616 abs_value = -value;
617 }
618
619 // the max value of uint64_t is 18446744073709551615UL,
620 // so use Z19_UINT64 to divide uint128_t
621 const static uint64_t Z19_UINT64 = 10000000000000000000ULL;
622 uint64_t suffix = abs_value % Z19_UINT64;
623 uint64_t middle = abs_value / Z19_UINT64 % Z19_UINT64;
624 uint64_t prefix = abs_value / Z19_UINT64 / Z19_UINT64;
625
626 char* end = buf + sizeof(buf);
627 if (prefix > 0) {
628 current += snprintf(current, end - current, "%" PRIu64, prefix);
629 current += snprintf(current, end - current, "%.19" PRIu64, middle);
630 current += snprintf(current, end - current, "%.19" PRIu64, suffix);
631 } else if (OLAP_LIKELY(middle > 0)) {
632 current += snprintf(current, end - current, "%" PRIu64, middle);
633 current += snprintf(current, end - current, "%.19" PRIu64, suffix);
634 } else {
635 current += snprintf(current, end - current, "%" PRIu64, suffix);
636 }
637 }
638
639 return std::string(buf);
640 }
