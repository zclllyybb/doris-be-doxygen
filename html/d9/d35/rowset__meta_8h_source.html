55 virtual bool init_from_pb(const RowsetMetaPB& rowset_meta_pb) {

60 // Release ownership of TabletSchemaPB from `rowset_meta_pb` and then set it back to `rowset_meta_pb`,

300 // return true if segments in this rowset has overlapping data.
301 // this is not same as `segments_overlap()` method.
302 // `segments_overlap()` only return the value of "segments_overlap" field in rowset meta,
303 // but "segments_overlap" may be UNKNOWN.
304 //
305 // Returns true iff all of the following conditions are met
306 // 1. the rowset contains more than one segment
307 // 2. the rowset's start version == end version (non-singleton rowset was generated by compaction process
308 // which always produces non-overlapped segments)
309 // 3. segments_overlap() flag is not NONOVERLAPPING (OVERLAP_UNKNOWN and OVERLAPPING are OK)
310 bool is_segments_overlapping() const {
311 return num_segments() > 1 && is_singleton_delta() && segments_overlap() != NONOVERLAPPING;
312 }
313
314 // get the compaction score of this rowset.
315 // if segments are overlapping, the score equals to the number of segments,
316 // otherwise, score is 1.
317 uint32_t get_compaction_score() const {
318 uint32_t score = 0;
319 if (!is_segments_overlapping()) {
320 score = 1;
321 } else {
322 score = num_segments();
323 CHECK(score > 0);
324 }
325 return score;
326 }

377 bool _deserialize_from_pb(const std::string& value) {
378 RowsetMetaPB rowset_meta_pb;

380 return false;
