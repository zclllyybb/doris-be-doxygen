286 void replace_column_data(const IColumn& rhs, size_t row, size_t self_row = 0) override {
287 DCHECK(size() > self_row);

289 null_map->replace_column_data(*nullable_rhs.null_map, row, self_row);
290
291 if (!nullable_rhs.is_null_at(row)) {
292 nested_column->replace_column_data(*nullable_rhs.nested_column, row, self_row);
293 } else {
294 nested_column->replace_column_data_default(self_row);
