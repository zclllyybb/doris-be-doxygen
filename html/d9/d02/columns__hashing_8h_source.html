79template <typename Value, typename Mapped, bool place_string_to_arena = true, bool use_cache = true>
80struct HashMethodString : public columns_hashing_impl::HashMethodBase<
81 HashMethodString<Value, Mapped, place_string_to_arena, use_cache>,
82 Value, Mapped, use_cache> {
83 using Self = HashMethodString<Value, Mapped, place_string_to_arena, use_cache>;
84 using Base = columns_hashing_impl::HashMethodBase<Self, Value, Mapped, use_cache>;
85
86 const IColumn::Offset* offsets;
87 const UInt8* chars;
88
89 HashMethodString(const ColumnRawPtrs& key_columns, const Sizes& /*key_sizes*/,
90 const HashMethodContextPtr&) {
91 const IColumn& column = *key_columns[0];
92 const ColumnString& column_string = assert_cast<const ColumnString&>(column);
93 offsets = column_string.get_offsets().data();
94 chars = column_string.get_chars().data();
95 }
96
97 auto get_key_holder(ssize_t row, [[maybe_unused]] Arena& pool) const {
98 StringRef key(chars + offsets[row - 1], offsets[row] - offsets[row - 1]);
99
100 if constexpr (place_string_to_arena) {
101 return ArenaKeyHolder {key, pool};
102 } else {
103 return key;
104 }
105 }
106
107protected:
108 friend class columns_hashing_impl::HashMethodBase<Self, Value, Mapped, use_cache>;
109};

116template <typename Value, typename Mapped, bool keys_pre_serialized = false>
117struct HashMethodSerialized
118 : public columns_hashing_impl::HashMethodBase<
119 HashMethodSerialized<Value, Mapped, keys_pre_serialized>, Value, Mapped, false> {
120 using Self = HashMethodSerialized<Value, Mapped, keys_pre_serialized>;
121 using Base = columns_hashing_impl::HashMethodBase<Self, Value, Mapped, false>;
122 using KeyHolderType =
123 std::conditional_t<keys_pre_serialized, ArenaKeyHolder, SerializedKeyHolder>;

125 ColumnRawPtrs key_columns;

127 const StringRef* keys;
128
129 HashMethodSerialized(const ColumnRawPtrs& key_columns_, const Sizes& /*key_sizes*/,
130 const HashMethodContextPtr&)
131 : key_columns(key_columns_), keys_size(key_columns_.size()) {}
132
133 void set_serialized_keys(const StringRef* keys_) { keys = keys_; }
134
135 ALWAYS_INLINE KeyHolderType get_key_holder(size_t row, Arena& pool) const {
136 if constexpr (keys_pre_serialized) {
137 return KeyHolderType {keys[row], pool};
138 } else {
139 return KeyHolderType {
140 serialize_keys_to_pool_contiguous(row, keys_size, key_columns, pool), pool};
141 }
142 }
143
144protected:
145 friend class columns_hashing_impl::HashMethodBase<Self, Value, Mapped, false>;

232 template <typename Data>
233 ALWAYS_INLINE EmplaceResult emplace_key(Data& data, size_t row, Arena& pool) {
234 if (key_column->is_null_at(row)) {
235 bool has_null_key = data.has_null_key_data();
236 data.has_null_key_data() = true;
237
238 if constexpr (has_mapped)
239 return EmplaceResult(data.get_null_key_data(), data.get_null_key_data(),
240 !has_null_key);
241 else
242 return EmplaceResult(!has_null_key);
243 }
244
245 auto key_holder = Base::get_key_holder(row, pool);
246
247 bool inserted = false;
248 typename Data::LookupResult it;
249 data.emplace(key_holder, it, inserted);
250
251 if constexpr (has_mapped) {
252 auto& mapped = *lookup_result_get_mapped(it);
253 if (inserted) {
254 new (&mapped) Mapped();
255 }
256 return EmplaceResult(mapped, mapped, inserted);
257 } else
258 return EmplaceResult(inserted);
259 }

261 template <typename Data, typename Func, typename CreatorForNull>
262 ALWAYS_INLINE typename std::enable_if_t<has_mapped, Mapped>& lazy_emplace_key(
263 Data& data, size_t row, Arena& pool, Func&& f, CreatorForNull&& null_creator) {
264 if (key_column->is_null_at(row)) {
265 bool has_null_key = data.has_null_key_data();
266 data.has_null_key_data() = true;
267 if (!has_null_key) std::forward<CreatorForNull>(null_creator)(data.get_null_key_data());
268 return data.get_null_key_data();
269 }
270 auto key_holder = Base::get_key_holder(row, pool);
271 typename Data::LookupResult it;
272 data.lazy_emplace(key_holder, it, std::forward<Func>(f));
273 return *lookup_result_get_mapped(it);
274 }

277 ALWAYS_INLINE typename std::enable_if_t<has_mapped, Mapped>& lazy_emplace_key(
278 Data& data, size_t row, Arena& pool, size_t hash_value, Func&& f,
279 CreatorForNull&& null_creator) {
280 if (key_column->is_null_at(row)) {
281 bool has_null_key = data.has_null_key_data();
282 data.has_null_key_data() = true;
283 if (!has_null_key) std::forward<CreatorForNull>(null_creator)(data.get_null_key_data());
284 return data.get_null_key_data();
285 }
286 auto key_holder = Base::get_key_holder(row, pool);
287 typename Data::LookupResult it;
288 data.lazy_emplace(key_holder, it, hash_value, std::forward<Func>(f));
289 return *lookup_result_get_mapped(it);
290 }

292 template <typename Data>
293 ALWAYS_INLINE FindResult find_key(Data& data, size_t row, Arena& pool) {
294 if (key_column->is_null_at(row)) {
295 bool has_null_key = data.has_null_key_data();
296 if constexpr (has_mapped)
297 return FindResult(&data.get_null_key_data(), has_null_key);
298 else
299 return FindResult(has_null_key);
300 }
301 auto key_holder = Base::get_key_holder(row, pool);
302 auto key = key_holder_get_key(key_holder);
303 auto it = data.find(key);
304 if constexpr (has_mapped)
305 return FindResult(it ? lookup_result_get_mapped(it) : nullptr, it != nullptr);
306 else
307 return FindResult(it != nullptr);
308 }
309};
