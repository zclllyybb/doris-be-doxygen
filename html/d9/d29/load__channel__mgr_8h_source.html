110template <typename Request>
111Status LoadChannelMgr::_get_load_channel(std::shared_ptr<LoadChannel>& channel, bool& is_eof,
112 const UniqueId& load_id, const Request& request) {
113 is_eof = false;
114 std::lock_guard<std::mutex> l(_lock);
115 auto it = _load_channels.find(load_id);
116 if (it == _load_channels.end()) {
117 auto handle = _last_success_channel->lookup(load_id.to_string());
118 // success only when eos be true
119 if (handle != nullptr) {
120 _last_success_channel->release(handle);
121 if (request.has_eos() && request.eos()) {
122 is_eof = true;
123 return Status::OK();
124 }
125 }
126 return Status::InternalError("fail to add batch in load channel. unknown load_id={}",
127 load_id.to_string());
128 }
129 channel = it->second;
130 return Status::OK();
131}

133template <typename TabletWriterAddRequest, typename TabletWriterAddResult>
134Status LoadChannelMgr::add_batch(const TabletWriterAddRequest& request,
135 TabletWriterAddResult* response) {
136 UniqueId load_id(request.id());
137 // 1. get load channel
138 std::shared_ptr<LoadChannel> channel;
139 bool is_eof;
140 auto status = _get_load_channel(channel, is_eof, load_id, request);
141 if (!status.ok() || is_eof) {
142 return status;
143 }
144
145 if (!channel->is_high_priority()) {
146 // 2. check if mem consumption exceed limit
147 // If this is a high priority load task, do not handle this.
148 // because this may block for a while, which may lead to rpc timeout.
149 _handle_mem_exceed_limit();
150 }
151
152 // 3. add batch to load channel
153 // batch may not exist in request(eg: eos request without batch),
154 // this case will be handled in load channel's add batch method.
155 Status st = channel->add_batch(request, response);
156 if (UNLIKELY(!st.ok())) {
157 channel->cancel();
158 return st;
159 }
160
161 // 4. handle finish
162 if (channel->is_finished()) {
163 _finish_load_channel(load_id);
164 }
165 return Status::OK();
166}
