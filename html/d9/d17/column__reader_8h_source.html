88// we should do our best to reduce resource usage through share
89// same information, such as OrdinalPageIndex and Page data.

91class ColumnReader {
92public:
93 // Create an initialized ColumnReader in *reader.
94 // This should be a lightweight operation without I/O.
95 static Status create(const ColumnReaderOptions& opts, const ColumnMetaPB& meta,
96 uint64_t num_rows, const io::FileReaderSPtr& file_reader,
97 std::unique_ptr<ColumnReader>* reader);
98
99 enum DictEncodingType { UNKNOWN_DICT_ENCODING, PARTIAL_DICT_ENCODING, ALL_DICT_ENCODING };
100
101 ~ColumnReader();
102
103 // create a new column iterator. Client should delete returned iterator
104 Status new_iterator(ColumnIterator** iterator);
105 // Client should delete returned iterator
106 Status new_bitmap_index_iterator(BitmapIndexIterator** iterator);
107
108 Status new_inverted_index_iterator(const TabletIndex* index_meta, OlapReaderStatistics* stats,
109 InvertedIndexIterator** iterator);
110
111 // Seek to the first entry in the column.
112 Status seek_to_first(OrdinalPageIndexIterator* iter);
113 Status seek_at_or_before(ordinal_t ordinal, OrdinalPageIndexIterator* iter);
114
115 // read a page from file into a page handle
116 Status read_page(const ColumnIteratorOptions& iter_opts, const PagePointer& pp,
117 PageHandle* handle, Slice* page_body, PageFooterPB* footer,
118 BlockCompressionCodec* codec) const;
119
120 bool is_nullable() const { return _meta.is_nullable(); }
121
122 const EncodingInfo* encoding_info() const { return _encoding_info; }
123
124 bool has_zone_map() const { return _zone_map_index_meta != nullptr; }
125 bool has_bitmap_index() const { return _bitmap_index_meta != nullptr; }
126 bool has_bloom_filter_index() const { return _bf_index_meta != nullptr; }
127
128 // Check if this column could match `cond' using segment zone map.
129 // Since segment zone map is stored in metadata, this function is fast without I/O.
130 // Return true if segment zone map is absent or `cond' could be satisfied, false otherwise.
131 bool match_condition(const AndBlockColumnPredicate* col_predicates) const;
132
133 Status next_batch_of_zone_map(size_t* n, vectorized::MutableColumnPtr& dst) const;
134
135 // get row ranges with zone map
136 // - cond_column is user's query predicate
137 // - delete_condition is a delete predicate of one version
138 Status get_row_ranges_by_zone_map(const AndBlockColumnPredicate* col_predicates,
139 std::vector<const ColumnPredicate*>* delete_predicates,
140 RowRanges* row_ranges);
141
142 // get row ranges with bloom filter index
143 Status get_row_ranges_by_bloom_filter(const AndBlockColumnPredicate* col_predicates,
144 RowRanges* row_ranges);
145
146 PagePointer get_dict_page_pointer() const { return _meta.dict_page(); }
147
148 bool is_empty() const { return _num_rows == 0; }
149
150 CompressionTypePB get_compression() const { return _meta.compression(); }
151
152 uint64_t num_rows() const { return _num_rows; }
153
154 void set_dict_encoding_type(DictEncodingType type) {
155 _set_dict_encoding_type_once.call([&] {
156 _dict_encoding_type = type;
157 return Status::OK();
158 });
159 }
160
161 DictEncodingType get_dict_encoding_type() { return _dict_encoding_type; }
162
163 void disable_index_meta_cache() { _index_meta_use_page_cache = false; }
164
165private:
166 ColumnReader(const ColumnReaderOptions& opts, const ColumnMetaPB& meta, uint64_t num_rows,
167 io::FileReaderSPtr file_reader);
168 Status init();
169
170 // Read and load necessary column indexes into memory if it hasn't been loaded.
171 // May be called multiple times, subsequent calls will no op.
172 Status _ensure_index_loaded() {
173 return _load_index_once.call([this] {
174 bool use_page_cache = !config::disable_storage_page_cache && _index_meta_use_page_cache;
175 RETURN_IF_ERROR(_load_zone_map_index(use_page_cache, _opts.kept_in_memory));
176 RETURN_IF_ERROR(_load_ordinal_index(use_page_cache, _opts.kept_in_memory));
177 RETURN_IF_ERROR(_load_bitmap_index(use_page_cache, _opts.kept_in_memory));
178 RETURN_IF_ERROR(_load_bloom_filter_index(use_page_cache, _opts.kept_in_memory));
179 return Status::OK();
180 });
181 }
182
183 // Read column inverted indexes into memory
184 // May be called multiple times, subsequent calls will no op.
185 Status _ensure_inverted_index_loaded(const TabletIndex* index_meta) {
186 // load inverted index only if not loaded or index_id is changed
187 RETURN_IF_ERROR(_load_inverted_index_index(index_meta));
188 return Status::OK();
189 }
190
191 Status _load_zone_map_index(bool use_page_cache, bool kept_in_memory);
192 Status _load_ordinal_index(bool use_page_cache, bool kept_in_memory);
193 Status _load_bitmap_index(bool use_page_cache, bool kept_in_memory);
194 Status _load_inverted_index_index(const TabletIndex* index_meta);
195 Status _load_bloom_filter_index(bool use_page_cache, bool kept_in_memory);
196
197 bool _zone_map_match_condition(const ZoneMapPB& zone_map, WrapperField* min_value_container,
198 WrapperField* max_value_container,
199 const AndBlockColumnPredicate* col_predicates) const;
200
201 void _parse_zone_map(const ZoneMapPB& zone_map, WrapperField* min_value_container,
202 WrapperField* max_value_container) const;
203
204 Status _get_filtered_pages(const AndBlockColumnPredicate* col_predicates,
205 std::vector<const ColumnPredicate*>* delete_predicates,
206 std::vector<uint32_t>* page_indexes);
207
208 Status _calculate_row_ranges(const std::vector<uint32_t>& page_indexes, RowRanges* row_ranges);
209
210private:
211 ColumnMetaPB _meta;
212 ColumnReaderOptions _opts;
213 uint64_t _num_rows;
214
215 io::FileReaderSPtr _file_reader;
216
217 DictEncodingType _dict_encoding_type;
218
219 TypeInfoPtr _type_info =
220 TypeInfoPtr(nullptr, nullptr); // initialized in init(), may changed by subclasses.
221 const EncodingInfo* _encoding_info =
222 nullptr; // initialized in init(), used for create PageDecoder
223
224 // meta for various column indexes (null if the index is absent)
225 bool _index_meta_use_page_cache = true;
226 const ZoneMapIndexPB* _zone_map_index_meta = nullptr;
227 const OrdinalIndexPB* _ordinal_index_meta = nullptr;
228 const BitmapIndexPB* _bitmap_index_meta = nullptr;
229 const BloomFilterIndexPB* _bf_index_meta = nullptr;
230
231 DorisCallOnce<Status> _load_index_once;
232 mutable std::mutex _load_index_lock;
233 std::unique_ptr<ZoneMapIndexReader> _zone_map_index;
234 std::unique_ptr<OrdinalIndexReader> _ordinal_index;
235 std::unique_ptr<BitmapIndexReader> _bitmap_index;
236 std::unique_ptr<InvertedIndexReader> _inverted_index;
237 std::unique_ptr<BloomFilterIndexReader> _bloom_filter_index;
238
239 std::vector<std::unique_ptr<ColumnReader>> _sub_readers;
240
241 std::once_flag _set_dict_encoding_type_flag;
242 DorisCallOnce<Status> _set_dict_encoding_type_once;

247public:

255

257 virtual Status seek_to_first() = 0;
258
259 // Seek to the given ordinal entry in the column.
260 // Entry 0 is the first entry written to the column.
261 // If provided seek point is past the end of the file,
262 // then returns false.
263 virtual Status seek_to_ordinal(ordinal_t ord) = 0;
264

275 return Status::NotSupported("next_batch_of_zone_map not implement");
276 }
277
278 virtual Status read_by_rowids(const rowid_t* rowids, const size_t count,
279 vectorized::MutableColumnPtr& dst) {
280 return Status::NotSupported("read_by_rowids not implement");
281 }
282
283 virtual ordinal_t get_current_ordinal() const = 0;
284
285 virtual Status get_row_ranges_by_zone_map(
286 const AndBlockColumnPredicate* col_predicates,

288 return Status::OK();
289 }
290

292 RowRanges* row_ranges) {
293 return Status::OK();
294 }
295
296 virtual bool is_all_dict_encoding() const { return false; }
297
298protected:
299 ColumnIteratorOptions _opts;
300};
