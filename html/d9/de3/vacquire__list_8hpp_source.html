25template <typename Element, int batch_size = 8>
26struct AcquireList {
27 using Batch = Element[batch_size];
28
29 Element& acquire(Element&& element) {
30 if (_current_batch == nullptr) {
31 _current_batch.reset(new Element[batch_size]);
32 }
33 if (current_full()) {
34 _lst.emplace_back(std::move(_current_batch));
35 _current_batch.reset(new Element[batch_size]);
36 _current_offset = 0;
37 }
38
39 auto base_addr = _current_batch.get();
40 base_addr[_current_offset] = std::move(element);
41 auto& ref = base_addr[_current_offset];
42 _current_offset++;
43 return ref;
44 }
45
46 void remove_last_element() { _current_offset--; }
47
48private:
49 bool current_full() { return _current_offset == batch_size; }
50 std::vector<std::unique_ptr<Element[]>> _lst;
51 std::unique_ptr<Element[]> _current_batch;
52 int _current_offset = 0;
53};
54} // namespace doris::vectorized
