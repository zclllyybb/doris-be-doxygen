421 // The bits in week_format has the following meaning:
422 // WEEK_MONDAY_FIRST (0)
423 // If not set:
424 // Sunday is first day of week
425 // If set:
426 // Monday is first day of week
427 //
428 // WEEK_YEAR (1)
429 // If not set:
430 // Week is in range 0-53
431 // Week 0 is returned for the last week of the previous year (for
432 // a date at start of january) In this case one can get 53 for the
433 // first week of next year. This flag ensures that the week is
434 // relevant for the given year. Note that this flag is only
435 // relevant if WEEK_JANUARY is not set.
436 // If set:
437 // Week is in range 1-53.
438 // In this case one may get week 53 for a date in January (when
439 // the week is that last week of previous year) and week 1 for a
440 // date in December.
441 //
442 // WEEK_FIRST_WEEKDAY (2)
443 // If not set
444 // Weeks are numbered according to ISO 8601:1988
445 // If set
446 // The week that contains the first 'first-day-of-week' is week 1.
447 //
448 // ISO 8601:1988 means that
449 // if the week containing January 1 has
450 // four or more days in the new year, then it is week 1;
451 // Otherwise it is the last week of the previous year, and the
452 // next week is week 1.

640 void from_packed_time(int64_t packed_time) {
641 int64_t ymdhms = packed_time >> 24;
642 int64_t ymd = ymdhms >> 17;
643 int64_t hms = ymdhms % (1 << 17);
644
645 _day = ymd % (1 << 5);
646 int64_t ym = ymd >> 5;
647 _month = ym % 13;
648 _year = ym / 13;
649 _year %= 10000;
650 _second = hms % (1 << 6);
651 _minute = (hms >> 6) % (1 << 6);
652 _hour = (hms >> 12);
653 _neg = 0;
654 _type = TIME_DATETIME;
655 }

1192int64_t datetime_diff(const VecDateTimeValue& ts_value1, const VecDateTimeValue& ts_value2) {
1193 switch (unit) {
1194 case YEAR: {
1195 int year = (ts_value2.year() - ts_value1.year());
1196 if (year > 0) {
1197 year -= (ts_value2.to_datetime_int64() % 10000000000 -
1198 ts_value1.to_datetime_int64() % 10000000000) < 0;
1199 } else if (year < 0) {
1200 year += (ts_value2.to_datetime_int64() % 10000000000 -
1201 ts_value1.to_datetime_int64() % 10000000000) > 0;
1202 }
1203 return year;
1204 }
1205 case MONTH: {
1206 int month = (ts_value2.year() - ts_value1.year()) * 12 +
1207 (ts_value2.month() - ts_value1.month());
1208 if (month > 0) {
1209 month -= (ts_value2.to_datetime_int64() % 100000000 -
1210 ts_value1.to_datetime_int64() % 100000000) < 0;
1211 } else if (month < 0) {
1212 month += (ts_value2.to_datetime_int64() % 100000000 -
1213 ts_value1.to_datetime_int64() % 100000000) > 0;
1214 }
1215 return month;
1216 }
1217 case WEEK: {
1218 int day = ts_value2.daynr() - ts_value1.daynr();
1219 if (day > 0) {
1220 day -= ts_value2.time_part_diff(ts_value1) < 0;
1221 } else if (day < 0) {
1222 day += ts_value2.time_part_diff(ts_value1) > 0;
1223 }
1224 return day / 7;
1225 }
1226 case DAY: {
1227 int day = ts_value2.daynr() - ts_value1.daynr();
1228 if (day > 0) {
1229 day -= ts_value2.time_part_diff(ts_value1) < 0;
1230 } else if (day < 0) {
1231 day += ts_value2.time_part_diff(ts_value1) > 0;
1232 }
1233 return day;
1234 }
1235 case HOUR: {
1236 int64_t second = ts_value2.second_diff(ts_value1);
1237 int64_t hour = second / 60 / 60;
1238 return hour;
1239 }
1240 case MINUTE: {

1254template <TimeUnit unit, typename T0, typename T1>
1255int64_t datetime_diff(const DateV2Value<T0>& ts_value1, const DateV2Value<T1>& ts_value2) {
1256 constexpr uint64_t uint64_minus_one = -1;
1257 switch (unit) {
1258 case YEAR: {
1259 int year = (ts_value2.year() - ts_value1.year());
1260 if constexpr (std::is_same_v<T0, T1>) {
1261 int year_width =
1262 DateV2Value<T0>::is_datetime ? DATETIMEV2_YEAR_WIDTH : DATEV2_YEAR_WIDTH;
1263 decltype(ts_value2.to_date_int_val()) minus_one = -1;
1264 if (year > 0) {
1265 year -= ((ts_value2.to_date_int_val() & (minus_one >> year_width)) <
1266 (ts_value1.to_date_int_val() & (minus_one >> year_width)));
1267 } else if (year < 0) {
1268 year += ((ts_value2.to_date_int_val() & (minus_one >> year_width)) >
1269 (ts_value1.to_date_int_val() & (minus_one >> year_width)));
1270 }
1271 } else if constexpr (std::is_same_v<T0, DateV2ValueType>) {
1272 auto ts1_int_value = ((uint64_t)ts_value1.to_date_int_val()) << TIME_PART_LENGTH;
1273 if (year > 0) {
1274 year -= ((ts_value2.to_date_int_val() &
1275 (uint64_minus_one >> DATETIMEV2_YEAR_WIDTH)) <
1276 (ts1_int_value & (uint64_minus_one >> DATETIMEV2_YEAR_WIDTH)));
1277 } else if (year < 0) {
1278 year += ((ts_value2.to_date_int_val() &
1279 (uint64_minus_one >> DATETIMEV2_YEAR_WIDTH)) >
1280 (ts1_int_value & (uint64_minus_one >> DATETIMEV2_YEAR_WIDTH)));
1281 }
1282 } else {
