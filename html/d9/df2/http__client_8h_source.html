61 _header_list = curl_slist_append(_header_list, scratch_str.c_str());
62 curl_easy_setopt(_curl, CURLOPT_HTTPHEADER, _header_list);

68 }

82 char* ct = nullptr;

100 // used to get content length
101 // return -1 as error
102 Status get_content_length(uint64_t* length) const {
103 curl_off_t cl;
104 auto code = curl_easy_getinfo(_curl, CURLINFO_CONTENT_LENGTH_DOWNLOAD_T, &cl);
105 if (!code) {
106 *length = cl;
107 return Status::OK();
108 }
109 return Status::InternalError("failed to get content length. err code: {}", code);
110 }
