94template <typename ArgType>
95struct ToDateImpl {
96 using OpArgType = ArgType;
97 using T = typename DateTraits<ArgType>::T;
98 static constexpr auto name = "to_date";
99
100 static inline auto execute(const ArgType& t) {
101 auto dt = binary_cast<ArgType, T>(t);
102 if constexpr (std::is_same_v<T, DateV2Value<DateV2ValueType>>) {
103 return binary_cast<T, ArgType>(dt);
104 } else if constexpr (std::is_same_v<T, VecDateTimeValue>) {
105 dt.cast_to_date();
106 return binary_cast<T, ArgType>(dt);
107 } else {
108 return (UInt32)(binary_cast<T, ArgType>(dt) >> TIME_PART_LENGTH);
109 }
110 }
111
112 static DataTypes get_variadic_argument_types() {
113 return {std::make_shared<typename DateTraits<ArgType>::DateType>()};
114 }
115};

180template <typename DateType, typename ArgType>
181struct DateFormatImpl {
182 using FromType = ArgType;
183
184 static constexpr auto name = "date_format";
185
186 static inline auto execute(const FromType& t, StringRef format, ColumnString::Chars& res_data,
187 size_t& offset) {
188 const auto& dt = (DateType&)t;
189 if (format.size > 128) {
190 return std::pair {offset, true};
191 }
192 char buf[128];
193 if (!dt.to_format_string(format.data, format.size, buf)) {
194 return std::pair {offset, true};
195 }
196
197 auto len = strlen(buf);
198 res_data.insert(buf, buf + len);
199 offset += len;
200 return std::pair {offset, false};
201 }
202
203 static DataTypes get_variadic_argument_types() {
204 return std::vector<DataTypePtr> {
205 std::dynamic_pointer_cast<const IDataType>(
206 std::make_shared<typename DateTraits<ArgType>::DateType>()),
207 std::dynamic_pointer_cast<const IDataType>(
208 std::make_shared<vectorized::DataTypeString>())};
209 }

238template <typename Transform>
239struct TransformerToStringOneArgument {
240 static void vector(FunctionContext* context,
241 const PaddedPODArray<typename Transform::OpArgType>& ts,
242 ColumnString::Chars& res_data, ColumnString::Offsets& res_offsets,
243 NullMap& null_map) {
244 const auto len = ts.size();
245 res_data.resize(len * Transform::max_size);
246 res_offsets.resize(len);
247 null_map.resize(len);
248
249 size_t offset = 0;
250 for (int i = 0; i < len; ++i) {
251 const auto& t = ts[i];
252 const auto& date_time_value =
253 reinterpret_cast<const typename DateTraits<typename Transform::OpArgType>::T&>(
254 t);
255 res_offsets[i] = Transform::execute(date_time_value, res_data, offset);
256 null_map[i] = !date_time_value.is_valid_date();
257 }
258 }

279template <typename Transform>
280struct TransformerToStringTwoArgument {
281 static void vector_constant(FunctionContext* context,
282 const PaddedPODArray<typename Transform::FromType>& ts,
283 const StringRef& format, ColumnString::Chars& res_data,
284 ColumnString::Offsets& res_offsets,
285 PaddedPODArray<UInt8>& null_map) {
286 auto len = ts.size();
287 res_offsets.resize(len);
288 res_data.reserve(len * format.size + len);
289 null_map.resize_fill(len, false);
290
291 size_t offset = 0;
292 for (int i = 0; i < len; ++i) {
293 const auto& t = ts[i];
294 size_t new_offset;
295 bool is_null;
296 if constexpr (is_specialization_of_v<Transform, FromUnixTimeImpl>) {
297 std::tie(new_offset, is_null) = Transform::execute(
298 t, format, res_data, offset, context->state()->timezone_obj());
299 } else {
300 std::tie(new_offset, is_null) = Transform::execute(t, format, res_data, offset);
301 }
302 res_offsets[i] = new_offset;
303 null_map[i] = is_null;
304 }
305 }

369template <typename FromType, typename ToType, typename Transform>
370struct DateTimeTransformImpl {
371 static Status execute(Block& block, const ColumnNumbers& arguments, size_t result,
372 size_t input_rows_count) {
373 using Op = Transformer<FromType, ToType, Transform>;
374
375 const auto is_nullable = block.get_by_position(result).type->is_nullable();
376
377 const ColumnPtr source_col = remove_nullable(block.get_by_position(arguments[0]).column);
378 if (const auto* sources = check_and_get_column<ColumnVector<FromType>>(source_col.get())) {
379 auto col_to = ColumnVector<ToType>::create();
380 if (is_nullable) {
381 auto null_map = ColumnVector<UInt8>::create(input_rows_count);
382 Op::vector(sources->get_data(), col_to->get_data(), null_map->get_data());
383 if (const auto* nullable_col = check_and_get_column<ColumnNullable>(
384 block.get_by_position(arguments[0]).column.get())) {
385 NullMap& result_null_map = assert_cast<ColumnUInt8&>(*null_map).get_data();
386 const NullMap& src_null_map =
387 assert_cast<const ColumnUInt8&>(nullable_col->get_null_map_column())
388 .get_data();
389
390 VectorizedUtils::update_null_map(result_null_map, src_null_map);
391 }
392 block.replace_by_position(
393 result, ColumnNullable::create(std::move(col_to), std::move(null_map)));
394 } else {
395 Op::vector(sources->get_data(), col_to->get_data());
396 block.replace_by_position(result, std::move(col_to));
397 }
398 } else {
399 return Status::RuntimeError("Illegal column {} of first argument of function {}",
400 block.get_by_position(arguments[0]).column->get_name(),
401 Transform::name);
402 }
403 return Status::OK();
404 }
405};
