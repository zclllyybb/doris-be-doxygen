37template <typename Impl>
38class FunctionBitmapSingle : public IFunction {
39public:
40 static constexpr auto name = Impl::name;
41
42 static FunctionPtr create() { return std::make_shared<FunctionBitmapSingle>(); }
43
44 String get_name() const override { return name; }
45
46 size_t get_number_of_arguments() const override { return 1; }
47
48 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
49 return make_nullable(std::make_shared<DataTypeInt64>());
50 }
51
52 bool use_default_implementation_for_constants() const override { return true; }
53
54 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
55 size_t result, size_t input_rows_count) override {
56 auto result_column = ColumnInt64::create();
57 auto result_null_map_column = ColumnUInt8::create(input_rows_count, 0);
58
59 ColumnPtr argument_column =
60 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
61 if (auto* nullable = check_and_get_column<ColumnNullable>(*argument_column)) {
62 // Danger: Here must dispose the null map data first! Because
63 // argument_columns[i]=nullable->get_nested_column_ptr(); will release the mem
64 // of column nullable mem of null map
65 VectorizedUtils::update_null_map(result_null_map_column->get_data(),
66 nullable->get_null_map_data());
67 argument_column = nullable->get_nested_column_ptr();
68 }
69
70 execute_straight(assert_cast<const ColumnBitmap*>(argument_column.get()),
71 assert_cast<ColumnInt64*>(result_column.get()),
72 assert_cast<ColumnUInt8*>(result_null_map_column.get())->get_data(),
73 input_rows_count);
74
75 block.get_by_position(result).column =
76 ColumnNullable::create(std::move(result_column), std::move(result_null_map_column));
77 return Status::OK();
78 }
79
80private:
81 void execute_straight(const ColumnBitmap* date_column, ColumnInt64* result_column,
82 NullMap& result_null_map, size_t input_rows_count) {
83 for (size_t i = 0; i < input_rows_count; i++) {
84 if (result_null_map[i]) {
85 result_column->insert_default();
86 continue;
87 }
88
89 BitmapValue value = date_column->get_element(i);
90 if (!value.cardinality()) {
91 result_null_map[i] = true;
92 result_column->insert_default();
93 continue;
94 }
95
96 result_column->insert(Impl::calculate(value));
97 }
98 }
99};
