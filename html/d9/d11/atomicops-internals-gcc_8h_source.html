56inline Atomic32 NoBarrier_CompareAndSwap(volatile Atomic32* ptr, Atomic32 old_value,
57 Atomic32 new_value) {
58 Atomic32 prev_value = old_value;
59 __atomic_compare_exchange_n(ptr, &prev_value, new_value, 0, __ATOMIC_RELAXED, __ATOMIC_RELAXED);
60 return prev_value;
61}
