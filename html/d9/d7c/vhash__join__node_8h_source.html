41template <typename RowRefListType>
42struct SerializedHashTableContext {
43 using Mapped = RowRefListType;
44 using HashTable = PartitionedHashMap<StringRef, Mapped>;
45 using State =
46 ColumnsHashing::HashMethodSerialized<typename HashTable::value_type, Mapped, true>;
47 using Iter = typename HashTable::iterator;
48
49 HashTable hash_table;
50 Iter iter;
51 bool inited = false;
52 std::vector<StringRef> keys;
53 size_t keys_memory_usage = 0;
54
55 void serialize_keys(const ColumnRawPtrs& key_columns, size_t num_rows) {
56 if (keys.size() < num_rows) {
57 keys.resize(num_rows);
58 }
59
60 _arena.reset(new Arena());
61 keys_memory_usage = 0;
62 size_t keys_size = key_columns.size();
63 for (size_t i = 0; i < num_rows; ++i) {
64 keys[i] = serialize_keys_to_pool_contiguous(i, keys_size, key_columns, *_arena);
65 }
66 keys_memory_usage = _arena->size();
67 }
68
69 void init_once() {
70 if (!inited) {
71 inited = true;
72 iter = hash_table.begin();
73 }
74 }
75
76private:
77 std::unique_ptr<Arena> _arena;
78};
