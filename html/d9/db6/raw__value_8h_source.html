81// Use boost::hash_combine for corner cases. boost::hash_combine is reimplemented
82// here to use int32t's (instead of size_t)
83// boost::hash_combine does:
84// seed ^= v + 0x9e3779b9 + (seed << 6) + (seed >> 2);
85inline uint32_t RawValue::get_hash_value(const void* v, const PrimitiveType& type, uint32_t seed) {
86 // Hash_combine with v = 0
87 if (v == nullptr) {
88 uint32_t value = 0x9e3779b9;
89 return seed ^ (value + (seed << 6) + (seed >> 2));
90 }
91
92 switch (type) {
93 case TYPE_VARCHAR:
94 case TYPE_CHAR:
95 case TYPE_HLL:
96 case TYPE_STRING: {
97 const StringRef* string_value = reinterpret_cast<const StringRef*>(v);
98 return HashUtil::hash(string_value->data, string_value->size, seed);
99 }
100
101 case TYPE_BOOLEAN: {
102 uint32_t value = *reinterpret_cast<const bool*>(v) + 0x9e3779b9;
103 return seed ^ (value + (seed << 6) + (seed >> 2));
104 }
105
106 case TYPE_TINYINT:
107 return HashUtil::hash(v, 1, seed);
108
109 case TYPE_SMALLINT:
110 return HashUtil::hash(v, 2, seed);
111
112 case TYPE_INT:
113 return HashUtil::hash(v, 4, seed);
114
115 case TYPE_BIGINT:
116 return HashUtil::hash(v, 8, seed);
117
118 case TYPE_FLOAT:
119 return HashUtil::hash(v, 4, seed);
120
121 case TYPE_DOUBLE:
122 return HashUtil::hash(v, 8, seed);
123
124 case TYPE_DATE:
125 case TYPE_DATETIME:
126 return HashUtil::hash(v, 16, seed);
127
128 case TYPE_DATEV2:
129 return HashUtil::hash(v, 4, seed);
130
131 case TYPE_DATETIMEV2:
132 return HashUtil::hash(v, 8, seed);
133
134 case TYPE_DECIMALV2:
135 return HashUtil::hash(v, 16, seed);
136 case TYPE_DECIMAL32:
137 return HashUtil::hash(v, 4, seed);
138 case TYPE_DECIMAL64:
139 return HashUtil::hash(v, 8, seed);
140 case TYPE_DECIMAL128I:
141 return HashUtil::hash(v, 16, seed);
142
143 case TYPE_LARGEINT:
144 return HashUtil::hash(v, 16, seed);
145
146 default:
147 DCHECK(false) << "invalid type: " << type;
148 return 0;
149 }
150}

152// NOTE: this is just for split data, decimal use old doris hash function
153// Because crc32 hardware is not equal with zlib crc32
154inline uint32_t RawValue::zlib_crc32(const void* v, const TypeDescriptor& type, uint32_t seed) {
155 // Hash_combine with v = 0
156 if (v == nullptr) {
157 uint32_t value = 0x9e3779b9;
158 return seed ^ (value + (seed << 6) + (seed >> 2));
159 }
160
161 switch (type.type) {
162 case TYPE_VARCHAR:
163 case TYPE_HLL:
164 case TYPE_STRING: {
165 const StringRef* string_value = reinterpret_cast<const StringRef*>(v);
166 return HashUtil::zlib_crc_hash(string_value->data, string_value->size, seed);
167 }
168 case TYPE_CHAR: {
169 // TODO(zc): ugly, use actual value to compute hash value
170 const StringRef* string_value = reinterpret_cast<const StringRef*>(v);
171 int len = 0;
172 while (len < string_value->size) {
173 if (string_value->data[len] == '\0') {
174 break;
175 }
176 len++;
177 }
178 return HashUtil::zlib_crc_hash(string_value->data, len, seed);
179 }
180 case TYPE_BOOLEAN:
181 case TYPE_TINYINT:
182 return HashUtil::zlib_crc_hash(v, 1, seed);
183 case TYPE_SMALLINT:
184 return HashUtil::zlib_crc_hash(v, 2, seed);
185 case TYPE_INT:
186 case TYPE_DATEV2:
187 case TYPE_DECIMAL32:
188 return HashUtil::zlib_crc_hash(v, 4, seed);
189 case TYPE_BIGINT:
190 case TYPE_DATETIMEV2:
191 case TYPE_DECIMAL64:
192 return HashUtil::zlib_crc_hash(v, 8, seed);
193 case TYPE_LARGEINT:
194 case TYPE_DECIMAL128I:
195 return HashUtil::zlib_crc_hash(v, 16, seed);
196 case TYPE_FLOAT:
197 return HashUtil::zlib_crc_hash(v, 4, seed);
198 case TYPE_DOUBLE:
199 return HashUtil::zlib_crc_hash(v, 8, seed);
200 case TYPE_DATE:
201 case TYPE_DATETIME: {
202 const DateTimeValue* date_val = (const DateTimeValue*)v;
203 char buf[64];
204 int len = date_val->to_buffer(buf);
205 return HashUtil::zlib_crc_hash(buf, len, seed);
206 }
207
208 case TYPE_DECIMALV2: {
209 const DecimalV2Value* dec_val = (const DecimalV2Value*)v;
210 int64_t int_val = dec_val->int_value();
211 int32_t frac_val = dec_val->frac_value();
212 seed = HashUtil::zlib_crc_hash(&int_val, sizeof(int_val), seed);
213 return HashUtil::zlib_crc_hash(&frac_val, sizeof(frac_val), seed);
214 }
215 default:
216 DCHECK(false) << "invalid type: " << type;
217 return 0;
218 }
219}
