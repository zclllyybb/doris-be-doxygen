44// NOTE: this is just for split data, decimal use old doris hash function
45// Because crc32 hardware is not equal with zlib crc32
46inline uint32_t RawValue::zlib_crc32(const void* v, size_t len, const PrimitiveType& type,
47 uint32_t seed) {
48 // Hash_combine with v = 0
49 if (v == nullptr) {
50 uint32_t value = 0x9e3779b9;
51 return seed ^ (value + (seed << 6) + (seed >> 2));
52 }
53
54 switch (type) {
55 case TYPE_VARCHAR:
56 case TYPE_HLL:
57 case TYPE_STRING:
58 case TYPE_CHAR: {
59 return HashUtil::zlib_crc_hash(v, len, seed);
60 }
61
62 case TYPE_BOOLEAN:
63 case TYPE_TINYINT:
64 return HashUtil::zlib_crc_hash(v, 1, seed);
65 case TYPE_SMALLINT:
66 return HashUtil::zlib_crc_hash(v, 2, seed);
67 case TYPE_INT:
68 return HashUtil::zlib_crc_hash(v, 4, seed);
69 case TYPE_BIGINT:
70 return HashUtil::zlib_crc_hash(v, 8, seed);
71 case TYPE_LARGEINT:
72 return HashUtil::zlib_crc_hash(v, 16, seed);
73 case TYPE_FLOAT:
74 return HashUtil::zlib_crc_hash(v, 4, seed);
75 case TYPE_DOUBLE:
76 return HashUtil::zlib_crc_hash(v, 8, seed);
77 case TYPE_DATE:
78 case TYPE_DATETIME: {
79 auto* date_val = (const vectorized::VecDateTimeValue*)v;
80 char buf[64];
81 int len = date_val->to_buffer(buf);
82 return HashUtil::zlib_crc_hash(buf, len, seed);
83 }
84
85 case TYPE_DATEV2: {
86 return HashUtil::zlib_crc_hash(v, 4, seed);
87 }
88
89 case TYPE_DATETIMEV2: {
90 return HashUtil::zlib_crc_hash(v, 8, seed);
91 }
92
93 case TYPE_DECIMALV2: {
94 const DecimalV2Value* dec_val = (const DecimalV2Value*)v;
95 int64_t int_val = dec_val->int_value();
96 int32_t frac_val = dec_val->frac_value();
97 seed = HashUtil::zlib_crc_hash(&int_val, sizeof(int_val), seed);
98 return HashUtil::zlib_crc_hash(&frac_val, sizeof(frac_val), seed);
99 }
100 case TYPE_DECIMAL32:
101 return HashUtil::zlib_crc_hash(v, 4, seed);
102 case TYPE_DECIMAL64:
103 return HashUtil::zlib_crc_hash(v, 8, seed);
104 case TYPE_DECIMAL128I:
105 return HashUtil::zlib_crc_hash(v, 16, seed);
106 default:
107 DCHECK(false) << "invalid type: " << type;
108 return 0;
109 }
110}
