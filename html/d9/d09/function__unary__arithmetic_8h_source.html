33template <typename A, typename Op>
34struct UnaryOperationImpl {
35 using ResultType = typename Op::ResultType;
36 using ColVecA = std::conditional_t<IsDecimalNumber<A>, ColumnDecimal<A>, ColumnVector<A>>;
37 using ColVecC = std::conditional_t<IsDecimalNumber<ResultType>, ColumnDecimal<ResultType>,
38 ColumnVector<ResultType>>;
39 using ArrayA = typename ColVecA::Container;
40 using ArrayC = typename ColVecC::Container;
41
42 static void NO_INLINE vector(const ArrayA& a, ArrayC& c) {
43 size_t size = a.size();
44 for (size_t i = 0; i < size; ++i) c[i] = Op::apply(a[i]);
45 }
46
47 static void constant(A a, ResultType& c) { c = Op::apply(a); }
48};

63template <template <typename> class Op, typename Name, bool is_injective>
64class FunctionUnaryArithmetic : public IFunction {
65 static constexpr bool allow_decimal = std::is_same_v<Op<Int8>, NegativeImpl<Int8>> ||
66 std::is_same_v<Op<Int8>, AbsImpl<Int8>> ||
67 std::is_same_v<Op<Int8>, PositiveImpl<Int8>>;
68
69 template <typename F>
70 static bool cast_type(const IDataType* type, F&& f) {
71 return cast_type_to_either<DataTypeUInt8, DataTypeUInt16, DataTypeUInt32, DataTypeUInt64,
72 DataTypeInt8, DataTypeInt16, DataTypeInt32, DataTypeInt64,
73 DataTypeInt128, DataTypeFloat32, DataTypeFloat64,
74 DataTypeDecimal<Decimal32>, DataTypeDecimal<Decimal64>,
75 DataTypeDecimal<Decimal128>, DataTypeDecimal<Decimal128I>>(
76 type, std::forward<F>(f));
77 }
78
79public:
80 static constexpr auto name = Name::name;
81 static FunctionPtr create() { return std::make_shared<FunctionUnaryArithmetic>(); }
82
83 String get_name() const override { return name; }
84
85 size_t get_number_of_arguments() const override { return 1; }
86 bool get_is_injective(const Block&) override { return is_injective; }
87
88 bool use_default_implementation_for_constants() const override { return true; }
89
90 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
91 DataTypePtr result;
92 bool valid = cast_type(arguments[0].get(), [&](const auto& type) {
93 using DataType = std::decay_t<decltype(type)>;
94 using T0 = typename DataType::FieldType;
95
96 if constexpr (IsDataTypeDecimal<DataType>) {
97 if constexpr (!allow_decimal) return false;
98 result = std::make_shared<DataType>(type.get_precision(), type.get_scale());
99 } else {
100 result = std::make_shared<DataTypeNumber<typename Op<T0>::ResultType>>();
101 }
102 return true;
103 });
104 if (!valid) {
105 LOG(FATAL) << fmt::format("Illegal type {} of argument of function {}",
106 arguments[0]->get_name(), get_name());
107 }
108 return result;
109 }
110
111 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
112 size_t result, size_t input_rows_count) override {
113 bool valid =
114 cast_type(block.get_by_position(arguments[0]).type.get(), [&](const auto& type) {
115 using DataType = std::decay_t<decltype(type)>;
116 using T0 = typename DataType::FieldType;
117
118 if constexpr (IsDataTypeDecimal<DataType>) {
119 if constexpr (allow_decimal) {
120 if (auto col = check_and_get_column<ColumnDecimal<T0>>(
121 block.get_by_position(arguments[0]).column.get())) {
122 auto col_res = ColumnDecimal<typename Op<T0>::ResultType>::create(
123 0, type.get_scale());
124 auto& vec_res = col_res->get_data();
125 vec_res.resize(col->get_data().size());
126 UnaryOperationImpl<T0, Op<T0>>::vector(col->get_data(), vec_res);
127 block.replace_by_position(result, std::move(col_res));
128 return true;
129 }
130 }
131 } else {
132 if (auto col = check_and_get_column<ColumnVector<T0>>(
133 block.get_by_position(arguments[0]).column.get())) {
134 auto col_res = ColumnVector<typename Op<T0>::ResultType>::create();
135 auto& vec_res = col_res->get_data();
136 vec_res.resize(col->get_data().size());
137 UnaryOperationImpl<T0, Op<T0>>::vector(col->get_data(), vec_res);
138 block.replace_by_position(result, std::move(col_res));
139 return true;
140 }
141 }
142
143 return false;
144 });
145 if (!valid) {
146 return Status::RuntimeError("{}'s argument does not match the expected data type",
147 get_name());
148 }
149 return Status::OK();
150 }
151
152 bool has_information_about_monotonicity() const override { return false; }
153};
