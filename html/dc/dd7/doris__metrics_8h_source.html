30#define REGISTER_ENTITY_HOOK_METRIC(entity, owner, metric, func) \
31 owner->metric = (UIntGauge*)(entity->register_metric<UIntGauge>(&METRIC_##metric)); \
32 entity->register_hook(#metric, [&]() { owner->metric->set_value(func()); });
33
34#define REGISTER_HOOK_METRIC(metric, func) \
35 REGISTER_ENTITY_HOOK_METRIC(DorisMetrics::instance()->server_entity(), \
36 DorisMetrics::instance(), metric, func)
37
38#define DEREGISTER_ENTITY_HOOK_METRIC(entity, name) \
39 entity->deregister_metric(&METRIC_##name); \
40 entity->deregister_hook(#name);
41
42#define DEREGISTER_HOOK_METRIC(name) \
43 DEREGISTER_ENTITY_HOOK_METRIC(DorisMetrics::instance()->server_entity(), name)
