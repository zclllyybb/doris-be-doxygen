34#define SIP_HASHES_FUNCTION_COLUMN_IMPL() \
35 auto s = hashes.size(); \
36 DCHECK(s == size()); \
37 if (null_data == nullptr) { \
38 for (size_t i = 0; i < s; i++) { \
39 update_hash_with_value(i, hashes[i]); \
40 } \
41 } else { \
42 for (size_t i = 0; i < s; i++) { \
43 if (null_data[i] == 0) update_hash_with_value(i, hashes[i]); \
44 } \
45 }
46
47#define DO_CRC_HASHES_FUNCTION_COLUMN_IMPL() \
48 if (null_data == nullptr) { \
49 for (size_t i = 0; i < s; i++) { \
50 hashes[i] = HashUtil::zlib_crc_hash(&data[i], sizeof(T), hashes[i]); \
51 } \
52 } else { \
53 for (size_t i = 0; i < s; i++) { \
54 if (null_data[i] == 0) \
55 hashes[i] = HashUtil::zlib_crc_hash(&data[i], sizeof(T), hashes[i]); \
56 } \
57 }
