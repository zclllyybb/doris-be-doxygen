29// A "smart" retrieved LRU cache handle.
30//
31// The cache handle is released when this object goes out of scope, possibly
32// closing the opened file if it is no longer in the cache.
33template <class FileType>
34class OpenedFileHandle {
35public:
36 OpenedFileHandle() : _cache(nullptr), _handle(nullptr) {}
37
38 // A opened file handle
39 explicit OpenedFileHandle(Cache* cache, Cache::Handle* handle)
40 : _cache(cache), _handle(handle) {}
41
42 // release cache handle
43 ~OpenedFileHandle() {
44 if (_handle != nullptr) {
45 _cache->release(_handle);
46 }
47 }
48
49 OpenedFileHandle(OpenedFileHandle&& other) noexcept {
50 std::swap(_cache, other._cache);
51 std::swap(_handle, other._handle);
52 }
53
54 OpenedFileHandle& operator=(OpenedFileHandle&& other) noexcept {
55 std::swap(_cache, other._cache);
56 std::swap(_handle, other._handle);
57 return *this;
58 }
59
60 FileType* file() const {
61 DCHECK(_handle != nullptr);
62 return reinterpret_cast<FileType*>(_cache->value(_handle));
63 }
64
65private:
66 Cache* _cache;
67 Cache::Handle* _handle;
68};

70// Cache of open files.
71//
72// The purpose of this cache is to enforce an upper bound on the maximum number
73// of files open at a time. Files opened through the cache may be closed at any
74// time, only to be reopened upon next use.
75//
76// The file cache can be viewed as having two logical parts: the client-facing
77// File handle and the LRU cache.
78//
79// Client-facing API
80// -----------------
81// The core of the client-facing API is the cache descriptor. A descriptor
82
83// LRU cache
84// ---------
85// The lower half of the file cache is a standard LRU cache whose keys are file
86// names and whose values are pointers to opened file objects allocated on the
87// heap. Unlike the descriptor map, this cache has an upper bound on capacity,
88// and handles are evicted (and closed) according to an LRU algorithm.
89//
90// Whenever a descriptor is used by a client in file I/O, its file name is used
91// in an LRU cache lookup. If found, the underlying file is still open and the
92// file access is performed. Otherwise, the file must have been evicted and
93// closed, so it is reopened and reinserted (possibly evicting a different open
94// file) before the file access is performed.
95//
96// Every public method in the file cache is thread safe.
97template <class FileType>
98class FileCache {
99public:
100 // Creates a new file cache.
101 //
102 // The 'cache_name' is used to disambiguate amongst other file cache
103 // instances. The cache will use 'max_open_files' as a soft upper bound on
104 // the number of files open at any given time.
105 // for this constructor, _is_cache_own is set to true, indicating that _cache
106 // is only owned by this.
107 FileCache(const std::string& cache_name, int max_open_files);
108
109 // Creates a new file cache with given cache.
110 //
111 // The 'cache_name' is used to disambiguate amongst other file cache
112 // instances. Please use this constructor only you want to share _cache
113 // with other.
114 // for this constructor, _is_cache_own is set to false, indicating that _cache
115 // is sharing with other (In most case, sharing _cache with storage engine).
116 FileCache(const std::string& cache_name, std::shared_ptr<Cache> cache);
117
118 // Destroys the file cache.
119 ~FileCache() {
120 // If _cache is only owned by this, reset the shared_ptr of _cache.
121 if (_is_cache_own) {
122 _cache.reset();
123 }
124 }
125
126 // find whether the file has been cached
127 // if cached, return true and set the file_handle
128 // else return false
129 bool lookup(const std::string& file_name, OpenedFileHandle<FileType>* file_handle);
130
131 // insert new FileType* into lru cache
132 // and return file_handle
133 void insert(const std::string& file_name, FileType* file,
134 OpenedFileHandle<FileType>* file_handle,
135 void (*deleter)(const CacheKey&, void*) = nullptr);
136
137private:
138 // Name of the cache.
139 std::string _cache_name;
140
141 // Underlying cache instance. Caches opened files.
142 std::shared_ptr<Cache> _cache;
143
144 // Indicates weather _cache is only owned by this,
145 // generally, _cache can be shared by other, in
146 // this case, _is_cache_own is set to false.
147 bool _is_cache_own = false;
148
149 DISALLOW_COPY_AND_ASSIGN(FileCache);
150};
