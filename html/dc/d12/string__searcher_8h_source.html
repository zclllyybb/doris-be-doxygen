70template <bool ASCII>
71class StringSearcher<true, ASCII> : public StringSearcherBase {
72private:
74 const uint8_t* const needle;
75 const uint8_t* const needle_end;
77 uint8_t first {};
78
79#ifdef __SSE4_1__
80 uint8_t second {};
82 __m128i first_pattern;
83 __m128i second_pattern;
85 __m128i cache = _mm_setzero_si128();
86 int cachemask {};
87#endif
88
89public:
90 template <typename CharT>
91 // requires (sizeof(CharT) == 1)
92 StringSearcher(const CharT* needle_, const size_t needle_size)
93 : needle {reinterpret_cast<const uint8_t*>(needle_)},
94 needle_end {needle + needle_size} {
95 if (0 == needle_size) return;
96
97 first = *needle;
98
99#ifdef __SSE4_1__
100 first_pattern = _mm_set1_epi8(first);
101 if (needle + 1 < needle_end) {
102 second = *(needle + 1);
103 second_pattern = _mm_set1_epi8(second);
104 }
105 const auto* needle_pos = needle;
106
107 //for (const auto i : collections::range(0, n))
108 for (size_t i = 0; i < n; i++) {
109 cache = _mm_srli_si128(cache, 1);
110
111 if (needle_pos != needle_end) {
112 cache = _mm_insert_epi8(cache, *needle_pos, n - 1);
113 cachemask |= 1 << i;
114 ++needle_pos;
115 }
116 }
117#endif
118 }

120 template <typename CharT>
121 // requires (sizeof(CharT) == 1)
122 ALWAYS_INLINE bool compare(const CharT* /*haystack*/, const CharT* /*haystack_end*/,
123 const CharT* pos) const {
124#ifdef __SSE4_1__
125 if (needle_end - needle > n && page_safe(pos)) {
126 const auto v_haystack = _mm_loadu_si128(reinterpret_cast<const __m128i*>(pos));
127 const auto v_against_cache = _mm_cmpeq_epi8(v_haystack, cache);
128 const auto mask = _mm_movemask_epi8(v_against_cache);
129
130 if (0xffff == cachemask) {
131 if (mask == cachemask) {
132 pos += n;
133 const auto* needle_pos = needle + n;
134
135 while (needle_pos < needle_end && *pos == *needle_pos) ++pos, ++needle_pos;
136
137 if (needle_pos == needle_end) return true;
138 }
139 } else if ((mask & cachemask) == cachemask)
140 return true;
141
142 return false;
143 }
144#endif
145
146 if (*pos == first) {
147 ++pos;
148 const auto* needle_pos = needle + 1;
149
150 while (needle_pos < needle_end && *pos == *needle_pos) ++pos, ++needle_pos;
151
152 if (needle_pos == needle_end) return true;
153 }
154
155 return false;
156 }

158 template <typename CharT>
159 // requires (sizeof(CharT) == 1)
160 const CharT* search(const CharT* haystack, const CharT* const haystack_end) const {
161 if (needle == needle_end) return haystack;
162
163 const auto needle_size = needle_end - needle;
164#ifdef __SSE4_1__
166 if (needle_size == 1) {
167 while (haystack < haystack_end) {
168 if (haystack + n <= haystack_end && page_safe(haystack)) {
169 const auto v_haystack =
170 _mm_loadu_si128(reinterpret_cast<const __m128i*>(haystack));
171 const auto v_against_pattern = _mm_cmpeq_epi8(v_haystack, first_pattern);
172 const auto mask = _mm_movemask_epi8(v_against_pattern);
173 if (mask == 0) {
174 haystack += n;
175 continue;
176 }
177
178 const auto offset = __builtin_ctz(mask);
179 haystack += offset;
180
181 return haystack;
182 }
183
184 if (haystack == haystack_end) {
185 return haystack_end;
186 }
187
188 if (*haystack == first) {
189 return haystack;
190 }
191 ++haystack;
192 }
193 return haystack_end;
194 }
195#endif
196
197 while (haystack < haystack_end && haystack_end - haystack >= needle_size) {
198#ifdef __SSE4_1__
199 if ((haystack + 1 + n) <= haystack_end && page_safe(haystack)) {
201 const auto v_haystack_block_first =
202 _mm_loadu_si128(reinterpret_cast<const __m128i*>(haystack));
203 const auto v_haystack_block_second =
204 _mm_loadu_si128(reinterpret_cast<const __m128i*>(haystack + 1));
205
206 const auto v_against_pattern_first =
207 _mm_cmpeq_epi8(v_haystack_block_first, first_pattern);
208 const auto v_against_pattern_second =
209 _mm_cmpeq_epi8(v_haystack_block_second, second_pattern);
210
211 const auto mask = _mm_movemask_epi8(
212 _mm_and_si128(v_against_pattern_first, v_against_pattern_second));
214 if (mask == 0) {
215 haystack += n;
216 continue;
217 }
218
219 const auto offset = __builtin_ctz(mask);
220 haystack += offset;
221
222 if (haystack + n <= haystack_end && page_safe(haystack)) {
224 const auto v_haystack_offset =
225 _mm_loadu_si128(reinterpret_cast<const __m128i*>(haystack));
226 const auto v_against_cache = _mm_cmpeq_epi8(v_haystack_offset, cache);
227 const auto mask_offset = _mm_movemask_epi8(v_against_cache);
228
229 if (0xffff == cachemask) {
230 if (mask_offset == cachemask) {
231 const auto* haystack_pos = haystack + n;
232 const auto* needle_pos = needle + n;
233
234 while (haystack_pos < haystack_end && needle_pos < needle_end &&
235 *haystack_pos == *needle_pos)
236 ++haystack_pos, ++needle_pos;
237
238 if (needle_pos == needle_end) return haystack;
239 }
240 } else if ((mask_offset & cachemask) == cachemask)
241 return haystack;
242
243 ++haystack;
244 continue;
245 }
246 }
247#endif
248
249 if (haystack == haystack_end) return haystack_end;
250
251 if (*haystack == first) {
252 const auto* haystack_pos = haystack + 1;
253 const auto* needle_pos = needle + 1;
254
255 while (haystack_pos < haystack_end && needle_pos < needle_end &&
256 *haystack_pos == *needle_pos)
257 ++haystack_pos, ++needle_pos;
258
259 if (needle_pos == needle_end) return haystack;
260 }
261
262 ++haystack;
263 }
264
265 return haystack_end;
266 }

275// Searches for needle surrounded by token-separators.
276// Separators are anything inside ASCII (0-128) and not alphanum.
277// Any value outside of basic ASCII (>=128) is considered a non-separator symbol, hence UTF-8 strings
278// should work just fine. But any Unicode whitespace is not considered a token separtor.
279template <typename StringSearcher>
280class TokenSearcher : public StringSearcherBase {
281 StringSearcher searcher;
282 size_t needle_size;
283
284public:
285 template <typename CharT>
286 // requires (sizeof(CharT) == 1)
287 TokenSearcher(const CharT* needle_, const size_t needle_size_)
288 : searcher {needle_, needle_size_}, needle_size(needle_size_) {
289 if (std::any_of(needle_, needle_ + needle_size_, isTokenSeparator)) {
290 //throw Exception{"Needle must not contain whitespace or separator characters", ErrorCodes::BAD_ARGUMENTS};
291 }
292 }
293
294 template <typename CharT>
295 // requires (sizeof(CharT) == 1)
296 ALWAYS_INLINE bool compare(const CharT* haystack, const CharT* haystack_end,
297 const CharT* pos) const {
298 // use searcher only if pos is in the beginning of token and pos + searcher.needle_size is end of token.
299 if (isToken(haystack, haystack_end, pos))
300 return searcher.compare(haystack, haystack_end, pos);
301
302 return false;
303 }
304
305 template <typename CharT>
306 // requires (sizeof(CharT) == 1)
307 const CharT* search(const CharT* haystack, const CharT* const haystack_end) const {
308 // use searcher.search(), then verify that returned value is a token
309 // if it is not, skip it and re-run
310
311 const auto* pos = haystack;
312 while (pos < haystack_end) {
313 pos = searcher.search(pos, haystack_end);
314 if (pos == haystack_end || isToken(haystack, haystack_end, pos)) return pos;
315
316 // assuming that heendle does not contain any token separators.
317 pos += needle_size;
318 }
319 return haystack_end;
320 }

323 // requires (sizeof(CharT) == 1)
324 const CharT* search(const CharT* haystack, const size_t haystack_size) const {
325 return search(haystack, haystack + haystack_size);
326 }

328 template <typename CharT>
329 // requires (sizeof(CharT) == 1)
330 ALWAYS_INLINE bool isToken(const CharT* haystack, const CharT* const haystack_end,
331 const CharT* p) const {
332 return (p == haystack || isTokenSeparator(*(p - 1))) &&
333 (p + needle_size >= haystack_end || isTokenSeparator(*(p + needle_size)));
334 }
335
336 ALWAYS_INLINE static bool isTokenSeparator(const uint8_t c) {
337 return !(is_alpha_numeric_ascii(c) || !is_ascii(c));
338 }
339};
