63inline size_t base64_encode(const char* data, size_t length, char* encoded_data) {
64 size_t output_length = (size_t)(4.0 * ceil((double)length / 3.0));
65
66 if (encoded_data == nullptr) {
67 return 0;
68 }
69
70 for (uint32_t i = 0, j = 0; i < length;) {
71 uint32_t octet_a = i < length ? (unsigned char)data[i++] : 0;
72 uint32_t octet_b = i < length ? (unsigned char)data[i++] : 0;
73 uint32_t octet_c = i < length ? (unsigned char)data[i++] : 0;
74
75 uint32_t triple = (octet_a << 0x10) + (octet_b << 0x08) + octet_c;
76
77 encoded_data[j++] = encoding_table[(triple >> 3 * 6) & 0x3F];
78 encoded_data[j++] = encoding_table[(triple >> 2 * 6) & 0x3F];
79 encoded_data[j++] = encoding_table[(triple >> 1 * 6) & 0x3F];
80 encoded_data[j++] = encoding_table[(triple >> 0 * 6) & 0x3F];
81 }
82
83 for (int i = 0; i < mod_table[length % 3]; i++) {
84 encoded_data[output_length - 1 - i] = base64_pad;
85 }
86
87 return output_length;
88}

102inline SQLFilterOp to_olap_filter_type(TExprOpcode::type type, bool opposite) {
103 switch (type) {
104 case TExprOpcode::LT:
105 return opposite ? FILTER_LARGER : FILTER_LESS;
106
107 case TExprOpcode::LE:
108 return opposite ? FILTER_LARGER_OR_EQUAL : FILTER_LESS_OR_EQUAL;
109
110 case TExprOpcode::GT:
111 return opposite ? FILTER_LESS : FILTER_LARGER;
112
113 case TExprOpcode::GE:
114 return opposite ? FILTER_LESS_OR_EQUAL : FILTER_LARGER_OR_EQUAL;
115
116 case TExprOpcode::EQ:
117 return opposite ? FILTER_NOT_IN : FILTER_IN;
118
119 case TExprOpcode::NE:
120 return opposite ? FILTER_IN : FILTER_NOT_IN;
121
122 case TExprOpcode::EQ_FOR_NULL:
123 return FILTER_IN;
124
125 default:
126 VLOG_CRITICAL << "TExprOpcode: " << type;
127 DCHECK(false);
128 }
129
130 return FILTER_IN;
131}

133inline SQLFilterOp to_olap_filter_type(const std::string& function_name, bool opposite) {
134 if (function_name == "lt") {
135 return opposite ? FILTER_LARGER : FILTER_LESS;
136 } else if (function_name == "gt") {
137 return opposite ? FILTER_LESS : FILTER_LARGER;
138 } else if (function_name == "le") {
139 return opposite ? FILTER_LARGER_OR_EQUAL : FILTER_LESS_OR_EQUAL;
140 } else if (function_name == "ge") {
141 return opposite ? FILTER_LESS_OR_EQUAL : FILTER_LARGER_OR_EQUAL;
142 } else if (function_name == "eq") {
143 return opposite ? FILTER_NOT_IN : FILTER_IN;
144 } else if (function_name == "ne") {
145 return opposite ? FILTER_IN : FILTER_NOT_IN;
146 } else {
147 DCHECK(false) << "Function Name: " << function_name;
148 return FILTER_IN;
149 }
150}

218inline bool is_match_condition(const std::string& op) {
219 if (0 == strcasecmp(op.c_str(), "match_any") || 0 == strcasecmp(op.c_str(), "match_all") ||
220 0 == strcasecmp(op.c_str(), "match_phrase") ||
221 0 == strcasecmp(op.c_str(), "match_element_eq") ||
222 0 == strcasecmp(op.c_str(), "match_element_lt") ||
223 0 == strcasecmp(op.c_str(), "match_element_gt") ||
224 0 == strcasecmp(op.c_str(), "match_element_le") ||
225 0 == strcasecmp(op.c_str(), "match_element_ge")) {
226 return true;
227 }
228 return false;
