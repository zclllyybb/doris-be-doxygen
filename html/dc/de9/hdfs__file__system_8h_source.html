30class HdfsFileSystemHandle {
31public:
32 HdfsFileSystemHandle(hdfsFS fs, bool cached)
33 : hdfs_fs(fs), from_cache(cached), _ref_cnt(0), _last_access_time(0), _invalid(false) {}
34
35 ~HdfsFileSystemHandle() {
36 DCHECK(_ref_cnt == 0);
37 if (hdfs_fs != nullptr) {
38 // Even if there is an error, the resources associated with the hdfsFS will be freed.
39 hdfsDisconnect(hdfs_fs);
40 }
41 hdfs_fs = nullptr;
42 }
43
44 int64_t last_access_time() { return _last_access_time; }
45
46 void inc_ref() {
47 _ref_cnt++;
48 _last_access_time = _now();
49 }
50
51 void dec_ref() {
52 _ref_cnt--;
53 _last_access_time = _now();
54 }
55
56 int ref_cnt() { return _ref_cnt; }
57
58 bool invalid() { return _invalid; }
59
60 void set_invalid() { _invalid = true; }
61
62 hdfsFS hdfs_fs;
63 // When cache is full, and all handlers are in use, HdfsFileSystemCache will return an uncached handler.
64 // Client should delete the handler in such case.
65 const bool from_cache;
66
67private:
68 // the number of referenced client
69 std::atomic<int> _ref_cnt;
70 // HdfsFileSystemCache try to remove the oldest handler when the cache is full
71 std::atomic<uint64_t> _last_access_time;
72 // Client will set invalid if error thrown, and HdfsFileSystemCache will not reuse this handler
73 std::atomic<bool> _invalid;
74
75 uint64_t _now() {
76 return std::chrono::duration_cast<std::chrono::milliseconds>(
77 std::chrono::system_clock::now().time_since_epoch())
78 .count();
79 }
80};
