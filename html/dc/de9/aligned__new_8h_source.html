30// Objects that should be allocated, for performance or correctness reasons, at alignment
31// greater than that promised by the global new (16) can inherit publicly from AlignedNew.
32template <size_t ALIGNMENT>
33struct alignas(ALIGNMENT) AlignedNew {
34 static_assert(ALIGNMENT > 0, "ALIGNMENT must be positive");
35 static_assert((ALIGNMENT & (ALIGNMENT - 1)) == 0, "ALIGNMENT must be a power of 2");
36 static_assert((ALIGNMENT % sizeof(void*)) == 0,
37 "ALIGNMENT must be a multiple of sizeof(void *)");
38 static void* operator new(std::size_t count) { return Allocate(count); }
39 static void* operator new[](std::size_t count) { return Allocate(count); }
40 static void operator delete(void* ptr) { free(ptr); }
41 static void operator delete[](void* ptr) { free(ptr); }
42
43private:
44 static void* Allocate(std::size_t count) {
45 void* result = nullptr;
46 const auto alloc_failed = posix_memalign(&result, ALIGNMENT, count);
47 if (alloc_failed) {
48 LOG(ERROR) << "Failed to allocate aligned memory; return code " << alloc_failed;
49 throw std::bad_alloc();
50 }
51 DCHECK(result != nullptr);
52 return result;
53 }
54};
55
56using CacheLineAligned = AlignedNew<CACHE_LINE_SIZE>;
57} // namespace doris
