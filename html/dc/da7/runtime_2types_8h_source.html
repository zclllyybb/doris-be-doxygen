36// Describes a type. Includes the enum, children types, and any type-specific metadata
37// (e.g. precision and scale for decimals).
38// TODO for 2.3: rename to TypeDescriptor
39struct TypeDescriptor {
40 PrimitiveType type;
42 int len;
43 static constexpr int MAX_VARCHAR_LENGTH = OLAP_VARCHAR_MAX_LENGTH;
44 static constexpr int MAX_CHAR_LENGTH = 255;
45 static constexpr int MAX_CHAR_INLINE_LENGTH = 128;
46
48 int precision;
49 int scale;
50
52 static constexpr int MAX_PRECISION = 38;
53 static constexpr int MAX_SCALE = MAX_PRECISION;
54
56 static constexpr int MAX_DECIMAL4_PRECISION = 9;
58 static constexpr int MAX_DECIMAL8_PRECISION = 18;
59
60 // Empty for scalar types
61 std::vector<TypeDescriptor> children;
62
63 // Only set if type == TYPE_STRUCT. The field name of each child.
64 std::vector<std::string> field_names;
65
66 // Used for complex types only.
67 bool contains_null = true;
68
69 TypeDescriptor() : type(INVALID_TYPE), len(-1), precision(-1), scale(-1) {}
70
71 // explicit TypeDescriptor(PrimitiveType type) :
72 TypeDescriptor(PrimitiveType type) : type(type), len(-1), precision(-1), scale(-1) {
73 if (type == TYPE_DECIMALV2) {
74 precision = 27;
75 scale = 9;
76 } else if (type == TYPE_DATETIMEV2) {
77 precision = 18;
78 scale = 6;
79 }
80 }
81
82 static TypeDescriptor create_char_type(int len) {
83 DCHECK_GE(len, 1);
84 DCHECK_LE(len, MAX_CHAR_LENGTH);
85 TypeDescriptor ret;
86 ret.type = TYPE_CHAR;
87 ret.len = len;
88 return ret;
89 }
90
91 static TypeDescriptor create_varchar_type(int len) {
92 DCHECK_GE(len, 1);
93 DCHECK_LE(len, MAX_VARCHAR_LENGTH);
94 TypeDescriptor ret;
95 ret.type = TYPE_VARCHAR;
96 ret.len = len;
97 return ret;
98 }
99
100 static TypeDescriptor create_string_type() {
101 TypeDescriptor ret;
102 ret.type = TYPE_STRING;
103 ret.len = config::string_type_length_soft_limit_bytes;
104 return ret;
105 }
106
107 static TypeDescriptor create_hll_type() {
108 TypeDescriptor ret;
109 ret.type = TYPE_HLL;
110 ret.len = HLL_COLUMN_DEFAULT_LEN;
111 return ret;
112 }
113
114 static TypeDescriptor create_decimalv2_type(int precision, int scale) {
115 DCHECK_LE(precision, MAX_PRECISION);
116 DCHECK_LE(scale, MAX_SCALE);
117 DCHECK_GE(precision, 0);
118 DCHECK_LE(scale, precision);
119 TypeDescriptor ret;
120 ret.type = TYPE_DECIMALV2;
121 ret.precision = precision;
122 ret.scale = scale;
123 return ret;
124 }
125
126 static TypeDescriptor from_thrift(const TTypeDesc& t) {
127 int idx = 0;
128 TypeDescriptor result(t.types, &idx);
129 DCHECK_EQ(idx, t.types.size() - 1);
130 return result;
131 }
132
133 static TypeDescriptor from_protobuf(const PTypeDesc& ptype) {
134 int idx = 0;
135 TypeDescriptor result(ptype.types(), &idx);
136 DCHECK_EQ(idx, ptype.types_size() - 1);
137 return result;
138 }
139
140 bool operator==(const TypeDescriptor& o) const {
141 if (type != o.type) {
142 return false;
143 }
144 if (children != o.children) {
145 return false;
146 }
147 if (type == TYPE_CHAR) {
148 return len == o.len;
149 }
150 if (type == TYPE_DECIMALV2) {
151 return precision == o.precision && scale == o.scale;
152 }
153 return true;
154 }
155
156 bool operator!=(const TypeDescriptor& other) const { return !(*this == other); }
157
158 TTypeDesc to_thrift() const {
159 TTypeDesc thrift_type;
160 to_thrift(&thrift_type);
161 return thrift_type;
162 }
163
164 void to_protobuf(PTypeDesc* ptype) const;
165
166 bool is_integer_type() const {
167 return type == TYPE_TINYINT || type == TYPE_SMALLINT || type == TYPE_INT ||
168 type == TYPE_BIGINT;
169 }
170
171 bool is_string_type() const {
172 return type == TYPE_VARCHAR || type == TYPE_CHAR || type == TYPE_HLL ||
173 type == TYPE_OBJECT || type == TYPE_QUANTILE_STATE || type == TYPE_STRING;
174 }
175
176 bool is_date_type() const { return type == TYPE_DATE || type == TYPE_DATETIME; }
177
178 bool is_date_v2_type() const { return type == TYPE_DATEV2; }
179 bool is_datetime_v2_type() const { return type == TYPE_DATETIMEV2; }
180
181 bool is_decimal_v2_type() const { return type == TYPE_DECIMALV2; }
182
183 bool is_decimal_v3_type() const {
184 return (type == TYPE_DECIMAL32) || (type == TYPE_DECIMAL64) || (type == TYPE_DECIMAL128I);
185 }
186
187 bool is_datetime_type() const { return type == TYPE_DATETIME; }
188
189 bool is_var_len_string_type() const {
190 return type == TYPE_VARCHAR || type == TYPE_HLL || type == TYPE_CHAR ||
191 type == TYPE_OBJECT || type == TYPE_QUANTILE_STATE || type == TYPE_STRING;
192 }
193
194 bool is_complex_type() const {
195 return type == TYPE_STRUCT || type == TYPE_ARRAY || type == TYPE_MAP;
196 }
197
198 bool is_collection_type() const { return type == TYPE_ARRAY || type == TYPE_MAP; }
199
200 bool is_array_type() const { return type == TYPE_ARRAY; }
201
202 bool is_bitmap_type() const { return type == TYPE_OBJECT; }
203
204 int get_slot_size() const { return ::doris::get_slot_size(type); }
205
206 static inline int get_decimal_byte_size(int precision) {
207 DCHECK_GT(precision, 0);
208 if (precision <= MAX_DECIMAL4_PRECISION) {
209 return 4;
210 }
211 if (precision <= MAX_DECIMAL8_PRECISION) {
212 return 8;
213 }
214 return 16;
215 }
216
217 std::string debug_string() const;
218
219private:
225 TypeDescriptor(const std::vector<TTypeNode>& types, int* idx);
226 TypeDescriptor(const google::protobuf::RepeatedPtrField<PTypeNode>& types, int* idx);
227
230 void to_thrift(TTypeDesc* thrift_type) const;
231};
