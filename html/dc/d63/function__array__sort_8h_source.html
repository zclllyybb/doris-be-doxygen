29template <typename Name, bool Positive>
30class FunctionArraySort : public IFunction {
31public:
32 static constexpr auto name = Name::name;
33 static FunctionPtr create() { return std::make_shared<FunctionArraySort<Name, Positive>>(); }
34
36 String get_name() const override { return name; }
37
38 bool is_variadic() const override { return false; }
39
40 size_t get_number_of_arguments() const override { return 1; }
41
42 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
43 DCHECK(is_array(arguments[0]))
44 << "first argument for function: " << name << " should be DataTypeArray"
45 << " and arguments[0] is " << arguments[0]->get_name();
46 return arguments[0];
47 }
48
49 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
50 size_t result, size_t input_rows_count) override {
51 ColumnPtr src_column =
52 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
53 const auto& src_column_array = check_and_get_column<ColumnArray>(*src_column);
54 if (!src_column_array) {
55 return Status::RuntimeError(
56 fmt::format("unsupported types for function {}({})", get_name(),
57 block.get_by_position(arguments[0]).type->get_name()));
58 }
59
60 auto dest_column_ptr = _execute(*src_column_array);
61
62 block.replace_by_position(result, std::move(dest_column_ptr));
63 return Status::OK();
64 }
65
66private:
67 ColumnPtr _execute(const ColumnArray& src_column_array) {
68 const auto& src_offsets = src_column_array.get_offsets();
69 const auto& src_nested_column = src_column_array.get_data();
70
71 size_t size = src_offsets.size();
72 size_t nested_size = src_nested_column.size();
73 IColumn::Permutation permutation(nested_size);
74 for (size_t i = 0; i < nested_size; ++i) {
75 permutation[i] = i;
76 }
77
78 for (size_t i = 0; i < size; ++i) {
79 auto current_offset = src_offsets[i - 1];
80 auto next_offset = src_offsets[i];
81
82 std::sort(&permutation[current_offset], &permutation[next_offset],
83 Less(src_nested_column));
84 }
85
86 return ColumnArray::create(src_column_array.get_data().permute(permutation, 0),
87 src_column_array.get_offsets_ptr());
88 }
89
90 struct Less {
91 const IColumn& column;
92
93 explicit Less(const IColumn& column_) : column(column_) {}
94
95 bool operator()(size_t lhs, size_t rhs) const {
96 if (Positive) {
97 return column.compare_at(lhs, rhs, column, -1) < 0;
98 } else {
99 return column.compare_at(lhs, rhs, column, -1) > 0;
100 }
101 }
102 };
103};
