39// space-saving algorithm
40template <typename T>
41struct AggregateFunctionTopNData {
42 using ColVecType =
43 std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<Decimal128>, ColumnVector<T>>;
44 void set_paramenters(int input_top_num, int space_expand_rate = 50) {
45 top_num = input_top_num;
46 capacity = (uint64_t)top_num * space_expand_rate;
47 }
48
49 void add(const StringRef& value, const UInt64& increment = 1) {
50 std::string data = value.to_string();
51 auto it = counter_map.find(data);
52 if (it != counter_map.end()) {
53 it->second = it->second + increment;
54 } else {
55 counter_map.insert({data, increment});
56 }
57 }
58
59 void add(const T& value, const UInt64& increment = 1) {
60 auto it = counter_map.find(value);
61 if (it != counter_map.end()) {
62 it->second = it->second + increment;
63 } else {
64 counter_map.insert({value, increment});
65 }
66 }
67
68 void merge(const AggregateFunctionTopNData& rhs) {
69 if (!rhs.top_num) {
70 return;
71 }
72
73 top_num = rhs.top_num;
74 capacity = rhs.capacity;
75
76 bool lhs_full = (counter_map.size() >= capacity);
77 bool rhs_full = (rhs.counter_map.size() >= capacity);
78
79 uint64_t lhs_min = 0;
80 uint64_t rhs_min = 0;
81
82 if (lhs_full) {
83 lhs_min = UINT64_MAX;
84 for (auto it : counter_map) {
85 lhs_min = std::min(lhs_min, it.second);
86 }
87 }
88
89 if (rhs_full) {
90 rhs_min = UINT64_MAX;
91 for (auto it : rhs.counter_map) {
92 rhs_min = std::min(rhs_min, it.second);
93 }
94
95 for (auto& it : counter_map) {
96 it.second += rhs_min;
97 }
98 }
99
100 for (auto rhs_it : rhs.counter_map) {
101 auto lhs_it = counter_map.find(rhs_it.first);
102 if (lhs_it != counter_map.end()) {
103 lhs_it->second += rhs_it.second - rhs_min;
104 } else {
105 counter_map.insert({rhs_it.first, rhs_it.second + lhs_min});
106 }
107 }
108 }
109
110 std::vector<std::pair<uint64_t, T>> get_remain_vector() const {
111 std::vector<std::pair<uint64_t, T>> counter_vector;
112 for (auto it : counter_map) {
113 counter_vector.emplace_back(it.second, it.first);
114 }
115 std::sort(counter_vector.begin(), counter_vector.end(),
116 std::greater<std::pair<uint64_t, T>>());
117 return counter_vector;
118 }
119
120 void write(BufferWritable& buf) const {
121 write_binary(top_num, buf);
122 write_binary(capacity, buf);
123
124 uint64_t element_number = std::min(capacity, (uint64_t)counter_map.size());
125 write_binary(element_number, buf);
126
127 auto counter_vector = get_remain_vector();
128
129 for (auto i = 0; i < element_number; i++) {
130 auto element = counter_vector[i];
131 write_binary(element.second, buf);
132 write_binary(element.first, buf);
133 }
134 }
135
136 void read(BufferReadable& buf) {
137 read_binary(top_num, buf);
138 read_binary(capacity, buf);
139
140 uint64_t element_number = 0;
141 read_binary(element_number, buf);
142
143 counter_map.clear();
144 std::pair<T, uint64_t> element;
145 for (auto i = 0; i < element_number; i++) {
146 read_binary(element.first, buf);
147 read_binary(element.second, buf);
148 counter_map.insert(element);
149 }
150 }
151
152 std::string get() const {
153 auto counter_vector = get_remain_vector();
154
155 rapidjson::StringBuffer buffer;
156 rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
157
158 writer.StartObject();
159 for (int i = 0; i < std::min((int)counter_vector.size(), top_num); i++) {
160 const auto& element = counter_vector[i];
161 writer.Key(element.second.c_str());
162 writer.Uint64(element.first);
163 }
164 writer.EndObject();
165
166 return buffer.GetString();
167 }
168
169 void insert_result_into(IColumn& to) const {
170 auto counter_vector = get_remain_vector();
171 for (int i = 0; i < std::min((int)counter_vector.size(), top_num); i++) {
172 const auto& element = counter_vector[i];
173 if constexpr (std::is_same_v<T, std::string>) {
174 static_cast<ColumnString&>(to).insert_data(element.second.c_str(),
175 element.second.length());
176 } else {
177 static_cast<ColVecType&>(to).get_data().push_back(element.second);
178 }
179 }
180 }
181
182 void reset() { counter_map.clear(); }
183
184 int top_num = 0;
185 uint64_t capacity = 0;
186 phmap::flat_hash_map<T, uint64_t> counter_map;
187};

308template <typename Impl, typename T, bool is_weighted>
309class AggregateFunctionTopNArray final : public AggregateFunctionTopNBase<Impl, T> {
310public:
311 AggregateFunctionTopNArray(const DataTypes& argument_types_)
312 : AggregateFunctionTopNBase<Impl, T>(argument_types_),
313 _argument_type(argument_types_[0]) {}
314
315 String get_name() const override {
316 if constexpr (is_weighted) {
317 return "topn_weighted";
318 } else {
319 return "topn_array";
320 }
321 }
322
323 DataTypePtr get_return_type() const override {
324 return std::make_shared<DataTypeArray>(make_nullable(_argument_type));
325 }
326
327 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
328 auto& to_arr = assert_cast<ColumnArray&>(to);
329 auto& to_nested_col = to_arr.get_data();
330 if (to_nested_col.is_nullable()) {
331 auto col_null = reinterpret_cast<ColumnNullable*>(&to_nested_col);
332 this->data(place).insert_result_into(col_null->get_nested_column());
333 col_null->get_null_map_data().resize_fill(col_null->get_nested_column().size(), 0);
334 } else {
335 this->data(place).insert_result_into(to_nested_col);
336 }
337 to_arr.get_offsets().push_back(to_nested_col.size());
338 }
339
340private:
341 DataTypePtr _argument_type;
342};
