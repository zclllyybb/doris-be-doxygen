32class CgroupsMgr {
33public:
34 // Input parameters:
35 // exec_env: global variable to get global objects
36 // cgroups_root_path: root cgroup allocated to doris by admin

38 ~CgroupsMgr();
39
40 // Compare the old user resource and new user resource to find deleted user
41 // then delete nonexisting cgroups, create new user cgroups, update all user cgroups
42 Status update_local_cgroups(const TFetchResourceResult& new_fetched_resource);
43
44 // Delete all existing cgroups under root path
45 Status init_cgroups();
46
47 // Modify cgroup resource shares under cgroups_root_path.
48 // Create related cgroups if it not exist.
49 //
50 // Input parameters:
51 // user_name: unique name for the user. it is a dir under cgroups_root_path
52 //
53 // user_share: a mapping for shares for different resource like (cpu.share, 100)
54 // mapping key is resource file name in cgroup; value is share weight
55 //
56 // level_share: a mapping for shares for different levels under the user.
57 // mapping key is level name; value is level's share. Currently, different resource using the same share.
58 Status modify_user_cgroups(const std::string& user_name,
59 const std::map<std::string, int32_t>& user_share,
60 const std::map<std::string, int32_t>& level_share);
61
62 static void apply_cgroup(const std::string& user_name, const std::string& level);
63
64 static void apply_system_cgroup() { apply_cgroup(_s_system_user, _s_system_group); }
65
66 // Assign the thread calling this funciton to the cgroup identified by user name and level
67 //
68 // Input parameters:
69 // user_name&level: the user name and level used to find the cgroup
70 Status assign_to_cgroups(const std::string& user_name, const std::string& level);
71
72 // Assign the thread identified by thread id to the cgroup identified by user name and level
73 //
74 // Input parameters:
75 // thread_id: the unique id for the thread
76 // user_name&level: the user name and level used to find the cgroup
77 Status assign_thread_to_cgroups(int64_t thread_id, const std::string& user_name,
78 const std::string& level);
79
80 // Delete the user's cgroups and its sub level cgroups using DropCgroups
81 // Input parameters:
82 // user name: user name to be deleted
83 Status delete_user_cgroups(const std::string& user_name);
84 // Delete a cgroup
85 // If there are active tasks in this cgroups, they will be relocated
86 // to root cgroups.
87 // If there are sub cgroups, it will return error.
88 // Input parameters:
89 // deleted_cgroups_path: the absolute cgroups path to be deleted
90 Status drop_cgroups(const std::string& deleted_cgroups_path);
91
92 // Relocate all threads or processes in src cgroups to dest cgroups
93 // Ignore errors when echo to dest cgroups
94 // Input parameters:
95 // src_cgroups: absolute path for src cgroups folder
96 // dest_cgroups: absolute path for dest cgroups folder
97 Status relocate_tasks(const std::string& src_cgroups, const std::string& dest_cgroups);
98
99 int64_t get_cgroups_version() { return _cur_version; }
100
101 // set the disk throttle for the user by getting resource value from the map and echo it to the cgroups.
102 // currently, both the user and groups under the user are set to the same value
103 // because throttle does not support hierarchy.
104 // Input parameters:
105 // user_name: name for the user
106 // resource_share: resource value get from fe
107 void _config_user_disk_throttle(std::string user_name,
108 const std::map<TResourceType::type, int32_t>& resource_share);
109
110 // get user resource share value from the map
111 int64_t _get_resource_value(const TResourceType::type resource_type,
112 const std::map<TResourceType::type, int32_t>& resource_share);
113
114 // set disk throttle according to the parameters. currently, we set different
115 // values for hdd and ssd.
116 // Input parameters:
117 // hdd_read_iops: read iops number for hdd disk.
118 // hdd_write_iops: write iops number for hdd disk.
119 // hdd_read_mbps: read bps number for hdd disk, using mb not byte or kb.
120 // hdd_write_mbps: write bps number for hdd disk, using mb not byte or kb.
121 // ssd_read_iops: read iops number for ssd disk.
122 // ssd_write_iops: write iops number for ssd disk.
123 // ssd_read_mbps: read bps number for ssd disk, using mb not byte or kb.
124 // ssd_write_mbps: write bps number for ssd disk, using mb not byte or kb.
125 Status _config_disk_throttle(std::string user_name, std::string level, int64_t hdd_read_iops,
126 int64_t hdd_write_iops, int64_t hdd_read_mbps,
127 int64_t hdd_write_mbps, int64_t ssd_read_iops,
128 int64_t ssd_write_iops, int64_t ssd_read_mbps,
129 int64_t ssd_write_mbps);
130
131 // echo command in string stream to the cgroup file
132 // Input parameters:
133 // ctrl_cmd: stringstream that contains the string to echo
134 // cgroups_path: target cgroup file path
135 void _echo_cmd_to_cgroup(std::stringstream& ctrl_cmd, std::string& cgroups_path);
136
137 // check if the path exists and it is a directory
138 // Input parameters:
139 // file_path: path to the file
140 bool is_directory(const char* file_path);
141
142 // check if the path exists
143 // Input parameters:
144 // file_path: path to the file
145 bool is_file_exist(const char* file_path);
146
147 // check if the path exists
148 // Input parameters:
149 // file_path: string value of the path
150 bool is_file_exist(const std::string& file_path);
151
152public:
153 const static std::string _s_system_user;
154 const static std::string _s_system_group;
155
156private:
157 std::string _root_cgroups_path;
158 int32_t _drop_retry_times = 10;
159 bool _is_cgroups_init_success;
160 std::string _default_user_name = "default";
161 std::string _default_level = "normal";
162 int64_t _cur_version;
163 std::set<std::string> _local_users;
164 std::mutex _update_cgroups_mtx;
165
166 // A static mapping from fe's resource type to cgroups file
167 static std::map<TResourceType::type, std::string> _s_resource_cgroups;
168};
