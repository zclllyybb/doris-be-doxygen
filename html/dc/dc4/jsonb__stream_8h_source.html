11/*
12 * This header file defines JsonbInBuffer and JsonbOutStream classes.
13 *
14 * ** Input Buffer **
15 * JsonbInBuffer is a customer input buffer to wrap raw character buffer. Its
16 * object instances are used to create std::istream objects interally.
17 *
18 * ** Output Stream **
19 * JsonbOutStream is a custom output stream classes, to contain the JSONB
20 * serialized binary. The class is conveniently used to specialize templates of
21 * JsonbParser and JsonbWriter.

23 * @author Tian Xia <tianx@fb.com>

45#define MAX_INT_DIGITS 11

49/*
50 * JSONB's implementation of input buffer
51 */
52class JsonbInBuffer : public std::streambuf {
53public:
54 JsonbInBuffer(const char* str, uint32_t len) {
55 // this is read buffer and the str will not be changed
56 // so we use const_cast (ugly!) to remove constness
57 char* pch(const_cast<char*>(str));
58 setg(pch, pch, pch + len);
59 }
60};

62/*
63 * JSONB's implementation of output stream.
64 *
65 * This is a wrapper of a char buffer. By default, the buffer capacity is 1024
66 * bytes. We will double the buffer if realloc is needed for writes.
67 */
68class JsonbOutStream : public std::ostream {
69public:
70 explicit JsonbOutStream(uint32_t capacity = 1024)
71 : std::ostream(nullptr), head_(nullptr), size_(0), capacity_(capacity), alloc_(true) {
72 if (capacity_ == 0) {
73 capacity_ = 1024;
74 }
75
76 head_ = (char*)malloc(capacity_);
77 }
78
79 JsonbOutStream(char* buffer, uint32_t capacity)
80 : std::ostream(nullptr), head_(buffer), size_(0), capacity_(capacity), alloc_(false) {
81 assert(buffer && capacity_ > 0);
82 }
83
84 ~JsonbOutStream() {
85 if (alloc_) {
86 free(head_);
87 }
88 }
89
90 void put(char c) { write(&c, 1); }
91
92 void write(const char* c_str) { write(c_str, (uint32_t)strlen(c_str)); }
93
94 void write(const char* bytes, uint32_t len) {
95 if (len == 0) return;
96
97 if (size_ + len > capacity_) {
98 realloc(len);
99 }
100
101 memcpy(head_ + size_, bytes, len);
102 size_ += len;
103 }
104
105 // write the integer to string
106 void write(int i) {
107 // snprintf automatically adds a NULL, so we need one more char
108 if (size_ + MAX_INT_DIGITS + 1 > capacity_) {
109 realloc(MAX_INT_DIGITS + 1);
110 }
111
112 int len = snprintf(head_ + size_, MAX_INT_DIGITS + 1, "%d", i);
113 assert(len > 0);
114 size_ += len;
115 }
116
117 // write the 64bit integer to string
118 void write(int64_t l) {
119 // snprintf automatically adds a NULL, so we need one more char
120 if (size_ + MAX_INT64_DIGITS + 1 > capacity_) {
121 realloc(MAX_INT64_DIGITS + 1);
122 }
123
124 int len = snprintf(head_ + size_, MAX_INT64_DIGITS + 1, "%" PRIi64, l);
125 assert(len > 0);
126 size_ += len;
127 }
128
129 void write(__int128 l) {
130 // TODO
131 assert(false);
132 }
133
134 // write the double to string
135 void write(double d) {
136 // snprintf automatically adds a NULL, so we need one more char
137 if (size_ + MAX_DOUBLE_DIGITS + 1 > capacity_) {
138 realloc(MAX_DOUBLE_DIGITS + 1);
139 }
140
141 int len = snprintf(head_ + size_, MAX_DOUBLE_DIGITS + 1, "%.15g", d);
142 assert(len > 0);
143 size_ += len;
144 }
145
146 pos_type tellp() const { return size_; }
147
148 void seekp(pos_type pos) { size_ = (uint32_t)pos; }
149
150 const char* getBuffer() const { return head_; }
151
152 pos_type getSize() const { return tellp(); }
153
154private:
155 void realloc(uint32_t len) {
156 assert(capacity_ > 0);
157
158 capacity_ *= 2;
159 while (capacity_ < size_ + len) {
160 capacity_ *= 2;
161 }
162
163 if (alloc_) {
164 char* new_buf = (char*)::realloc(head_, capacity_);
165 assert(new_buf);
166 head_ = new_buf;
167 } else {
168 char* new_buf = (char*)::malloc(capacity_);
169 assert(new_buf);
170 memcpy(new_buf, head_, size_);
171 head_ = new_buf;
172 alloc_ = true;
173 }
174 }
175
176private:
177 char* head_;
178 uint32_t size_;
179 uint32_t capacity_;
180 bool alloc_;
181};
