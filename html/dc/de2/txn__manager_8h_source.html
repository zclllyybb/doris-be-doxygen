173 using TxnKey = std::pair<int64_t, int64_t>; // partition_id, transaction_id;

175 // Implement TxnKey hash function to support TxnKey as a key for `unordered_map`.
176 struct TxnKeyHash {
177 template <typename T, typename U>
178 size_t operator()(const std::pair<T, U>& e) const {
179 return std::hash<T>()(e.first) ^ std::hash<U>()(e.second);
180 }
181 };
182
183 // Implement TxnKey equal function to support TxnKey as a key for `unordered_map`.
184 struct TxnKeyEqual {
185 template <class T, typename U>
186 bool operator()(const std::pair<T, U>& l, const std::pair<T, U>& r) const {
187 return l.first == r.first && l.second == r.second;
188 }
189 };
