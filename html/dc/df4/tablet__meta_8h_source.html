40// Lifecycle states that a Tablet can be in. Legal state transitions for a
41// Tablet object:
42//
43// NOTREADY -> RUNNING -> TOMBSTONED -> STOPPED -> SHUTDOWN
44// | | | ^^^
45// | | +----------++|
46// | +------------------------+|
47// +-------------------------------------+
48
49enum TabletState {
50 // Tablet is under alter table, rollup, clone
51 TABLET_NOTREADY,
52
53 TABLET_RUNNING,
54
55 // Tablet integrity has been violated, such as missing versions.
56 // In this state, tablet will not accept any incoming request.
57 // Report this state to FE, scheduling BE to drop tablet.
58 TABLET_TOMBSTONED,
59
60 // Tablet is shutting down, files in disk still remained.
61 TABLET_STOPPED,
62
63 // Files have been removed, tablet has been shutdown completely.
64 TABLET_SHUTDOWN
65};
