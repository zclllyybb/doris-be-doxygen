32// Utility class to write bit/byte streams. This class can write data to either be
33// bit packed or byte aligned (and a single stream that has a mix of both).
34class BitWriter {
35public:
36 // buffer: buffer to write bits to.
37 explicit BitWriter(faststring* buffer) : buffer_(buffer) { Clear(); }
38
39 void Clear() {
40 buffered_values_ = 0;
41 byte_offset_ = 0;
42 bit_offset_ = 0;
43 buffer_->clear();
44 }
45
46 // Returns a pointer to the underlying buffer
47 faststring* buffer() const { return buffer_; }
48
49 // The number of current bytes written, including the current byte (i.e. may include a
50 // fraction of a byte). Includes buffered values.
51 int bytes_written() const { return byte_offset_ + BitUtil::Ceil(bit_offset_, 8); }
52
53 // Writes a value to buffered_values_, flushing to buffer_ if necessary. This is bit
54 // packed.
55 void PutValue(uint64_t v, int num_bits);
56
57 // Writes v to the next aligned byte using num_bits. If T is larger than num_bits, the
58 // extra high-order bits will be ignored.
59 template <typename T>
60 void PutAligned(T v, int num_bits);
61
62 // Write a Vlq encoded int to the buffer. The value is written byte aligned.
63 // For more details on vlq: en.wikipedia.org/wiki/Variable-length_quantity
64 void PutVlqInt(int32_t v);
65
66 // Get the index to the next aligned byte and advance the underlying buffer by num_bytes.
67 size_t GetByteIndexAndAdvance(int num_bytes) {
68 uint8_t* ptr = GetNextBytePtr(num_bytes);
69 return ptr - buffer_->data();
70 }
71
72 // Get a pointer to the next aligned byte and advance the underlying buffer by num_bytes.
73 uint8_t* GetNextBytePtr(int num_bytes);
74
75 // Flushes all buffered values to the buffer. Call this when done writing to the buffer.
76 // If 'align' is true, buffered_values_ is reset and any future writes will be written
77 // to the next byte boundary.
78 void Flush(bool align = false);
79
80private:
81 // Bit-packed values are initially written to this variable before being memcpy'd to
82 // buffer_. This is faster than writing values byte by byte directly to buffer_.
83 uint64_t buffered_values_;
84
85 faststring* buffer_;
86 int byte_offset_; // Offset in buffer_
87 int bit_offset_; // Offset in buffered_values_
88};

90// Utility class to read bit/byte stream. This class can read bits or bytes
91// that are either byte aligned or not. It also has utilities to read multiple
92// bytes in one read (e.g. encoded int).
93class BitReader {
94public:
95 // 'buffer' is the buffer to read from. The buffer's length is 'buffer_len'.
96 BitReader(const uint8_t* buffer, int buffer_len);
97
98 BitReader() : buffer_(nullptr), max_bytes_(0) {}
99
100 // Gets the next value from the buffer. Returns true if 'v' could be read or false if
101 // there are not enough bytes left. num_bits must be <= 32.
102 template <typename T>
103 bool GetValue(int num_bits, T* v);
104
105 // Reads a 'num_bytes'-sized value from the buffer and stores it in 'v'. T needs to be a
106 // little-endian native type and big enough to store 'num_bytes'. The value is assumed
107 // to be byte-aligned so the stream will be advanced to the start of the next byte
108 // before 'v' is read. Returns false if there are not enough bytes left.
109 template <typename T>
110 bool GetAligned(int num_bytes, T* v);
111
112 // Reads a vlq encoded int from the stream. The encoded int must start at the
113 // beginning of a byte. Return false if there were not enough bytes in the buffer.
114 bool GetVlqInt(uint32_t* v);
115 // Reads a zigzag encoded int `into` v.
116 bool GetZigZagVlqInt(int32_t* v);
117
118 // Reads a vlq encoded int from the stream. The encoded int must start at the
119 // beginning of a byte. Return false if there were not enough bytes in the buffer.
120 bool GetVlqInt(uint64_t* v);
121 // Reads a zigzag encoded int `into` v.
122 bool GetZigZagVlqInt(int64_t* v);
123
124 // Returns the number of bytes left in the stream, not including the current byte (i.e.,
125 // there may be an additional fraction of a byte).
126 int bytes_left() { return max_bytes_ - (byte_offset_ + BitUtil::Ceil(bit_offset_, 8)); }
127
128 // Current position in the stream, by bit.
129 int position() const { return byte_offset_ * 8 + bit_offset_; }
130
131 // Rewind the stream by 'num_bits' bits
132 void Rewind(int num_bits);
133
134 // Advance the stream by 'num_bits' bits
135 bool Advance(int64_t num_bits);
136
137 // Seek to a specific bit in the buffer
138 void SeekToBit(unsigned int stream_position);
139
140 // Maximum byte length of a vlq encoded int
141 static const int MAX_VLQ_BYTE_LEN = 5;
142
143 // Maximum byte length of a vlq encoded int64
144 static const int MAX_VLQ_BYTE_LEN_FOR_INT64 = 10;
145
146 bool is_initialized() const { return buffer_ != nullptr; }
147
148private:
149 // Used by SeekToBit() and GetValue() to fetch the
150 // the next word into buffer_.
151 void BufferValues();
152
153 const uint8_t* buffer_;
154 int max_bytes_;
155
156 // Bytes are memcpy'd from buffer_ and values are read from this variable. This is
157 // faster than reading values byte by byte directly from buffer_.
158 uint64_t buffered_values_;
159
160 int byte_offset_; // Offset in buffer_
161 int bit_offset_; // Offset in buffered_values_
162};

168class BatchedBitReader {
169public:
172 BatchedBitReader(const uint8_t* buffer, int64_t buffer_len) { Reset(buffer, buffer_len); }
173
174 BatchedBitReader() {}
175
176 // The implicit copy constructor is left defined. If a BatchedBitReader is copied, the
177 // two copies do not share any state. Invoking functions on either copy continues
178 // reading from the current read position without modifying the state of the other
179 // copy.
180
183 void Reset(const uint8_t* buffer, int64_t buffer_len) {
184 DCHECK(buffer != nullptr);
185 DCHECK_GE(buffer_len, 0);
186 buffer_pos_ = buffer;
187 buffer_end_ = buffer + buffer_len;
188 }
