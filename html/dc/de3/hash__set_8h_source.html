32template <typename Key, typename TCell, typename Hash = DefaultHash<Key>,
33 typename Grower = HashTableGrower<>, typename Allocator = HashTableAllocator>
34class HashSetTable : public HashTable<Key, TCell, Hash, Grower, Allocator> {
35public:
36 using Self = HashSetTable;
37 using Cell = TCell;
38
39 using Base = HashTable<Key, TCell, Hash, Grower, Allocator>;
40 using typename Base::LookupResult;
41
42 void merge(const Self& rhs) {
43 if (!this->get_has_zero() && rhs.get_has_zero()) {
44 this->set_get_has_zero();
45 ++this->m_size;
46 }
47
48 for (size_t i = 0; i < rhs.grower.buf_size(); ++i)
49 if (!rhs.buf[i].is_zero(*this)) this->insert(Cell::get_key(rhs.buf[i].get_value()));
50 }
51};

54struct HashSetCellWithSavedHash : public HashTableCell<Key, Hash, TState> {

63 bool key_equals(const Key& key_) const { return this->key == key_; }
