35template <typename T, bool _int, bool _float, bool _decimal, bool _datetime, typename F>
36bool call_on_basic_type(TypeIndex number, F&& f) {
37 if constexpr (_int) {
38 switch (number) {
39 case TypeIndex::UInt8:
40 return f(TypePair<T, UInt8>());
41 case TypeIndex::UInt16:
42 return f(TypePair<T, UInt16>());
43 case TypeIndex::UInt32:
44 return f(TypePair<T, UInt32>());
45 case TypeIndex::UInt64:
46 return f(TypePair<T, UInt64>());
47
48 case TypeIndex::Int8:
49 return f(TypePair<T, Int8>());
50 case TypeIndex::Int16:
51 return f(TypePair<T, Int16>());
52 case TypeIndex::Int32:
53 return f(TypePair<T, Int32>());
54 case TypeIndex::Int64:
55 return f(TypePair<T, Int64>());
56 case TypeIndex::Int128:
57 return f(TypePair<T, Int128>());
58
59 default:
60 break;
61 }
62 }
63
64 if constexpr (_decimal) {
65 switch (number) {
66 case TypeIndex::Decimal32:
67 return f(TypePair<T, Decimal32>());
68 case TypeIndex::Decimal64:
69 return f(TypePair<T, Decimal64>());
70 case TypeIndex::Decimal128:
71 return f(TypePair<T, Decimal128>());
72 case TypeIndex::Decimal128I:

74 default:
75 break;

77 }

81 case TypeIndex::Float32:

83 case TypeIndex::Float64:

87 }
88 }
89
90 return false;
91}
