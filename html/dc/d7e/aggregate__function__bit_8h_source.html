89template <typename T, typename Data>
90class AggregateFunctionBitwise final
91 : public IAggregateFunctionDataHelper<Data, AggregateFunctionBitwise<T, Data>> {
92public:
93 AggregateFunctionBitwise(const DataTypes& argument_types_)
94 : IAggregateFunctionDataHelper<Data, AggregateFunctionBitwise<T, Data>>(
95 argument_types_) {}
96
97 String get_name() const override { return Data::name; }
98
99 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeNumber<T>>(); }
100
101 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
102 Arena*) const override {
103 const auto& column = static_cast<const ColumnVector<T>&>(*columns[0]);
104 this->data(place).add(column.get_data()[row_num]);
105 }
106
107 void reset(AggregateDataPtr place) const override { this->data(place).reset(); }
108
109 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
110 Arena*) const override {
111 this->data(place).merge(this->data(rhs));
112 }
113
114 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
115 this->data(place).write(buf);
116 }
117
118 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
119 Arena*) const override {
120 this->data(place).read(buf);
121 }
122
123 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
124 auto& column = static_cast<ColumnVector<T>&>(to);
125 column.get_data().push_back(this->data(place).get());
126 }
127};
