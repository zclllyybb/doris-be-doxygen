43template <typename Impl, typename Name>
44class FunctionArrayMapped : public IFunction {
45public:
46 static constexpr auto name = Name::name;
47 static FunctionPtr create() { return std::make_shared<FunctionArrayMapped>(); }
48
49 String get_name() const override { return name; }
50 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
51 size_t result, size_t input_rows_count) override {
52 const auto& typed_column = block.get_by_position(arguments[0]);
53 auto ptr = typed_column.column->convert_to_full_column_if_const();
54 const typename Impl::column_type* column_array;
55 if (ptr->is_nullable()) {
56 column_array = check_and_get_column<const typename Impl::column_type>(
57 static_cast<const ColumnNullable*>(ptr.get())->get_nested_column_ptr().get());
58 } else {
59 column_array = check_and_get_column<const typename Impl::column_type>(ptr.get());
60 }
61 const auto* data_type_array =
62 static_cast<const DataTypeArray*>(remove_nullable(typed_column.type).get());
63 return Impl::execute(block, arguments, result, data_type_array, *column_array);
64 }
65
66 bool is_variadic() const override { return Impl::_is_variadic(); }
67
68 size_t get_number_of_arguments() const override { return Impl::_get_number_of_arguments(); }
69
70 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
71 return Impl::get_return_type(arguments);
72 }
73};
