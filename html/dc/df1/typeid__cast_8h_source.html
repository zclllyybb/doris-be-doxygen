36template <typename To, typename From>
37std::enable_if_t<std::is_reference_v<To>, To> typeid_cast(From& from) {
38 try {
39 if (typeid(from) == typeid(To)) {
40 return static_cast<To>(from);
41 }
42 } catch (const std::exception& e) {
43 throw doris::vectorized::Exception(e.what(), doris::TStatusCode::VEC_BAD_CAST);
44 }
45
46 throw doris::vectorized::Exception("Bad cast from type " + demangle(typeid(from).name()) +
47 " to " + demangle(typeid(To).name()),
48 doris::TStatusCode::VEC_BAD_CAST);
49}

51template <typename To, typename From>
52To typeid_cast(From* from) {
53 try {
54 if (typeid(*from) == typeid(std::remove_pointer_t<To>)) {
55 return static_cast<To>(from);
56 } else {
57 return nullptr;
58 }
59 } catch (const std::exception& e) {
60 throw doris::vectorized::Exception(e.what(), doris::TStatusCode::VEC_BAD_CAST);
61 }
