30// Block Column Predicate support do column predicate and support OR and AND predicate
31// Block Column Predicate will replace column predicate as a unified external vectorized interface
32// in the future
33// TODO: support do predicate on Bitmap and ZoneMap, So we can use index of column to do predicate on
34// page and segment
35class BlockColumnPredicate {
36public:
37 BlockColumnPredicate() = default;
38 virtual ~BlockColumnPredicate() = default;
39
40 virtual void get_all_column_ids(std::set<ColumnId>& column_id_set) const = 0;
41
42 virtual void get_all_column_predicate(
43 std::set<const ColumnPredicate*>& predicate_set) const = 0;
44
45 virtual uint16_t evaluate(vectorized::MutableColumns& block, uint16_t* sel,
46 uint16_t selected_size) const {
47 return selected_size;
48 }
49 virtual void evaluate_and(vectorized::MutableColumns& block, uint16_t* sel,
50 uint16_t selected_size, bool* flags) const {}
51 virtual void evaluate_or(vectorized::MutableColumns& block, uint16_t* sel,
52 uint16_t selected_size, bool* flags) const {}
53
54 virtual void evaluate_vec(vectorized::MutableColumns& block, uint16_t size, bool* flags) const {
55 }
56
57 virtual bool evaluate_and(const std::pair<WrapperField*, WrapperField*>& statistic) const {
58 LOG(FATAL) << "should not reach here";
59 return true;
60 }
61
62 virtual bool evaluate_and(const segment_v2::BloomFilter* bf) const {
63 LOG(FATAL) << "should not reach here";
64 return true;
65 }
66 virtual bool can_do_bloom_filter() const { return false; }
67
68 //evaluate predicate on inverted
69 virtual Status evaluate(const std::string& column_name, InvertedIndexIterator* iterator,
70 uint32_t num_rows, roaring::Roaring* bitmap) const {
71 return Status::NotSupported(
72 "Not Implemented evaluate with inverted index, please check the predicate");
73 }
74};

105class MutilColumnBlockPredicate : public BlockColumnPredicate {
106public:
107 MutilColumnBlockPredicate() = default;
108
109 ~MutilColumnBlockPredicate() override {
110 for (auto ptr : _block_column_predicate_vec) {
111 delete ptr;
112 }
113 }
114
115 void add_column_predicate(const BlockColumnPredicate* column_predicate) {
116 _block_column_predicate_vec.push_back(column_predicate);

118
119 size_t num_of_column_predicate() const { return _block_column_predicate_vec.size(); }
120
121 void get_all_column_ids(std::set<ColumnId>& column_id_set) const override {
122 for (auto child_block_predicate : _block_column_predicate_vec) {
123 child_block_predicate->get_all_column_ids(column_id_set);
124 }
125 }
126
127 void get_all_column_predicate(std::set<const ColumnPredicate*>& predicate_set) const override {
128 for (auto child_block_predicate : _block_column_predicate_vec) {
129 child_block_predicate->get_all_column_predicate(predicate_set);
130 }
131 }
132
133protected:
134 std::vector<const BlockColumnPredicate*> _block_column_predicate_vec;
135};
