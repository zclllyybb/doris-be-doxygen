190 Status filter_by_selector(const uint16_t* sel, size_t sel_size, IColumn* col_ptr) override {
191 auto* res_col = reinterpret_cast<vectorized::ColumnString*>(col_ptr);
192 StringRef strings[sel_size];
193 size_t length = 0;
194 for (size_t i = 0; i != sel_size; ++i) {
195 auto& value = _dict.get_value(_codes[sel[i]]);
196 strings[i].data = value.data;
197 strings[i].size = value.size;
198 length += value.size;
199 }
200 res_col->get_offsets().reserve(sel_size + res_col->get_offsets().size());
201 res_col->get_chars().reserve(length + res_col->get_chars().size());
202 res_col->insert_many_strings_without_reserve(strings, sel_size);
203 return Status::OK();
204 }

347 // The function is only used in the runtime filter feature
348 inline void generate_hash_values_for_runtime_filter(FieldType type) {
349 if (_hash_values.empty()) {
350 _hash_values.resize(_dict_data->size());
351 for (size_t i = 0; i < _dict_data->size(); i++) {
352 auto& sv = (*_dict_data)[i];
353 // The char data is stored in the disk with the schema length,
354 // and zeros are filled if the length is insufficient
355
356 // When reading data, use shrink_char_type_column_suffix_zero(_char_type_idx)
357 // Remove the suffix 0
358 // When writing data, use the CharField::consume function to fill in the trailing 0.
359
360 // For dictionary data of char type, sv.size is the schema length,
361 // so use strnlen to remove the 0 at the end to get the actual length.
362 int32_t len = sv.size;
363 if (type == OLAP_FIELD_TYPE_CHAR) {
364 len = strnlen(sv.data, sv.size);
365 }
366 uint32_t hash_val = HashUtil::murmur_hash3_32(sv.data, len, 0);
367 _hash_values[i] = hash_val;
368 }
369 }
370 }

372 inline uint32_t get_hash_value(T code) const { return _hash_values[code]; }
373
374 // For > , code takes upper_bound - 1; For >= , code takes upper_bound
375 // For < , code takes upper_bound; For <=, code takes upper_bound - 1
376 // For example a sorted dict: <'b',0> <'c',1> <'d',2>
377 // Now the predicate value is 'ccc', 'ccc' is not in the dict, 'ccc' is between 'c' and 'd'.
378 // std::upper_bound(..., 'ccc') - begin, will return the encoding of 'd', which is 2
379 // If the predicate is col > 'ccc' and the value of upper_bound-1 is 1,
380 // then evaluate code > 1 and the result is 'd'.
381 // If the predicate is col < 'ccc' and the value of upper_bound is 2,
382 // evaluate code < 2, and the return result is 'b'.
383 // If the predicate is col >= 'ccc' and the value of upper_bound is 2,
384 // evaluate code >= 2, and the return result is 'd'.
385 // If the predicate is col <= 'ccc' and the value of upper_bound-1 is 1,
386 // evaluate code <= 1, and the returned result is 'b'.
387 // If the predicate is col < 'a', 'a' is also not in the dict, and 'a' is less than 'b',
388 // so upper_bound is the code 0 of b, then evaluate code < 0 and returns empty
389 // If the predicate is col <= 'a' and upper_bound-1 is -1,
390 // then evaluate code <= -1 and returns empty
391 int32_t find_code_by_bound(const StringRef& value, bool greater, bool eq) const {
392 auto code = find_code(value);
393 if (code >= 0) {
394 return code;
395 }
396 auto bound = std::upper_bound(_dict_data->begin(), _dict_data->end(), value) -
397 _dict_data->begin();
398 return greater ? bound - greater + eq : bound - eq;
399 }

401 void find_codes(const phmap::flat_hash_set<StringRef>& values,
402 std::vector<vectorized::UInt8>& selected) const {
403 size_t dict_word_num = _dict_data->size();
404 selected.resize(dict_word_num);
405 selected.assign(dict_word_num, false);
406 for (size_t i = 0; i < _dict_data->size(); i++) {
407 if (values.find((*_dict_data)[i]) != values.end()) {
408 selected[i] = true;
409 }
410 }
411 }

421 void sort() {
422 size_t dict_size = _dict_data->size();
423
424 _code_convert_table.reserve(dict_size);
425 _perm.resize(dict_size);
426 for (size_t i = 0; i < dict_size; ++i) {
427 _perm[i] = i;
428 }
429
430 std::sort(_perm.begin(), _perm.end(),
431 [&dict_data = *_dict_data, &comparator = _comparator](const size_t a,
432 const size_t b) {
433 return comparator(dict_data[a], dict_data[b]);
434 });
435
436 auto new_dict_data = new DictContainer(dict_size);
437 for (size_t i = 0; i < dict_size; ++i) {
438 _code_convert_table[_perm[i]] = (T)i;
439 (*new_dict_data)[i] = (*_dict_data)[_perm[i]];
440 }
441 _dict_data.reset(new_dict_data);
442 }
443
444 T convert_code(const T& code) const {
445 if (get_null_code() == code) {
446 return code;
447 }
448 return _code_convert_table[code];

462 }

464 std::string debug_string() const {
465 std::string str = "[";
466 if (_dict_data) {
467 for (size_t i = 0; i < _dict_data->size(); i++) {
468 if (i) {
469 str += ',';
470 }
471 str += (*_dict_data)[i].to_string();
472 }
473 }
474 str += ']';
475 return str;
476 }

478 private:
479 StringRef _null_value = StringRef();
480 StringRef::Comparator _comparator;
481 // dict code -> dict value
482 std::unique_ptr<DictContainer> _dict_data;
483 std::vector<T> _code_convert_table;
484 // hash value of origin string , used for bloom filter
485 // It's a trade-off of space for performance
486 // But in TPC-DS 1GB q60,we see no significant improvement.
487 // This may because the magnitude of the data is not large enough(in q60, only about 80k rows data is filtered for largest table)
488 // So we may need more test here.
489 HashValueContainer _hash_values;
490 IColumn::Permutation _perm;
491 size_t _total_str_len;
