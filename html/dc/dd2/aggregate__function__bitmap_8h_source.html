132template <typename Op>
133class AggregateFunctionBitmapOp final
134 : public IAggregateFunctionDataHelper<AggregateFunctionBitmapData<Op>,
135 AggregateFunctionBitmapOp<Op>> {
136public:
137 using ResultDataType = BitmapValue;
138 using ColVecType = ColumnBitmap;
139 using ColVecResult = ColumnBitmap;
140
141 String get_name() const override { return Op::name; }
142
143 AggregateFunctionBitmapOp(const DataTypes& argument_types_)
144 : IAggregateFunctionDataHelper<AggregateFunctionBitmapData<Op>,
145 AggregateFunctionBitmapOp<Op>>(argument_types_) {}
146
147 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeBitMap>(); }
148
149 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
150 Arena*) const override {
151 const auto& column = static_cast<const ColVecType&>(*columns[0]);
152 this->data(place).add(column.get_data()[row_num]);
153 }
154
155 void add_many(AggregateDataPtr __restrict place, const IColumn** columns,
156 std::vector<int>& rows, Arena*) const override {
157 if constexpr (std::is_same_v<Op, AggregateFunctionBitmapUnionOp>) {
158 const auto& column = static_cast<const ColVecType&>(*columns[0]);
159 std::vector<const BitmapValue*> values;
160 for (int i = 0; i < rows.size(); ++i) {
161 values.push_back(&(column.get_data()[rows[i]]));
162 }
163 this->data(place).add_batch(values);
164 }
165 }
166
167 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
168 Arena*) const override {
169 this->data(place).merge(
170 const_cast<AggregateFunctionBitmapData<Op>&>(this->data(rhs)).get());
171 }
172
173 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
174 this->data(place).write(buf);
175 }
176
177 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
178 Arena*) const override {
179 this->data(place).read(buf);
180 }
181
182 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
183 auto& column = static_cast<ColVecResult&>(to);
184 column.get_data().push_back(
185 const_cast<AggregateFunctionBitmapData<Op>&>(this->data(place)).get());
186 }
187
188 void reset(AggregateDataPtr __restrict place) const override { this->data(place).reset(); }
189};

191template <bool arg_is_nullable, typename ColVecType>
192class AggregateFunctionBitmapCount final
193 : public IAggregateFunctionDataHelper<
194 AggregateFunctionBitmapData<AggregateFunctionBitmapUnionOp>,
195 AggregateFunctionBitmapCount<arg_is_nullable, ColVecType>> {
196public:
197 // using ColVecType = ColumnBitmap;
198 using ColVecResult = ColumnVector<Int64>;
199 using AggFunctionData = AggregateFunctionBitmapData<AggregateFunctionBitmapUnionOp>;
200
201 AggregateFunctionBitmapCount(const DataTypes& argument_types_)
202 : IAggregateFunctionDataHelper<
203 AggregateFunctionBitmapData<AggregateFunctionBitmapUnionOp>,
204 AggregateFunctionBitmapCount<arg_is_nullable, ColVecType>>(argument_types_) {}
205
206 String get_name() const override { return "count"; }
207 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeInt64>(); }
208
209 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
210 Arena*) const override {
211 if constexpr (arg_is_nullable) {
212 auto& nullable_column = assert_cast<const ColumnNullable&>(*columns[0]);
213 if (!nullable_column.is_null_at(row_num)) {
214 const auto& column =
215 static_cast<const ColVecType&>(nullable_column.get_nested_column());
216 this->data(place).add(column.get_data()[row_num]);
217 }
218 } else {
219 const auto& column = static_cast<const ColVecType&>(*columns[0]);
220 this->data(place).add(column.get_data()[row_num]);
221 }
222 }
223
224 void add_many(AggregateDataPtr __restrict place, const IColumn** columns,
225 std::vector<int>& rows, Arena*) const override {
226 if constexpr (arg_is_nullable && std::is_same_v<ColVecType, ColumnBitmap>) {
227 auto& nullable_column = assert_cast<const ColumnNullable&>(*columns[0]);
228 const auto& column =
229 static_cast<const ColVecType&>(nullable_column.get_nested_column());
230 std::vector<const BitmapValue*> values;
231 for (int i = 0; i < rows.size(); ++i) {
232 if (!nullable_column.is_null_at(rows[i])) {
233 values.push_back(&(column.get_data()[rows[i]]));
234 }
235 }
236 this->data(place).add_batch(values);
237 } else if constexpr (std::is_same_v<ColVecType, ColumnBitmap>) {
238 const auto& column = static_cast<const ColVecType&>(*columns[0]);
239 std::vector<const BitmapValue*> values;
240 for (int i = 0; i < rows.size(); ++i) {
241 values.push_back(&(column.get_data()[rows[i]]));
242 }
243 this->data(place).add_batch(values);
244 }
245 }
246
247 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
248 Arena*) const override {
249 this->data(place).merge(const_cast<AggFunctionData&>(this->data(rhs)).get());
250 }
251
252 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
253 this->data(place).write(buf);
254 }
255
256 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
257 Arena*) const override {
258 this->data(place).read(buf);
259 }
260
261 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
262 auto& value_data = const_cast<AggFunctionData&>(this->data(place)).get();
263 auto& column = static_cast<ColVecResult&>(to);
264 column.get_data().push_back(value_data.cardinality());
265 }
266
267 void reset(AggregateDataPtr __restrict place) const override { this->data(place).reset(); }
268};
