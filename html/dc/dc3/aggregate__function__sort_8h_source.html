32struct AggregateFunctionSortData {
33 const SortDescription sort_desc;
34 Block block;
35
36 // The construct only support the template compiler, useless
37 AggregateFunctionSortData() : sort_desc() {};
38 AggregateFunctionSortData(SortDescription sort_desc, const Block& block)
39 : sort_desc(std::move(sort_desc)), block(block.clone_empty()) {}
40
41 void merge(const AggregateFunctionSortData& rhs) {
42 if (block.rows() == 0) {
43 block = rhs.block;
44 } else {
45 for (size_t i = 0; i < block.columns(); i++) {
46 auto column = block.get_by_position(i).column->assume_mutable();
47 auto column_rhs = rhs.block.get_by_position(i).column;
48 column->insert_range_from(*column_rhs, 0, rhs.block.rows());
49 }
50 }
51 }
52
53 void serialize(const RuntimeState* state, BufferWritable& buf) const {
54 PBlock pblock;
55 size_t uncompressed_bytes = 0;
56 size_t compressed_bytes = 0;
57 block.serialize(state->be_exec_version(), &pblock, &uncompressed_bytes, &compressed_bytes,
58 segment_v2::CompressionTypePB::SNAPPY);
59
60 write_string_binary(pblock.SerializeAsString(), buf);
61 }
62
63 void deserialize(BufferReadable& buf) {
64 std::string data;
65 read_binary(data, buf);
66
67 PBlock pblock;
68 pblock.ParseFromString(data);
69 new (&block) Block(pblock);
70 }
71
72 void add(const IColumn** columns, size_t columns_num, size_t row_num) {
73 DCHECK(block.columns() == columns_num)
74 << fmt::format("block.columns()!=columns_num, block.columns()={}, columns_num={}",
75 block.columns(), columns_num);
76
77 for (size_t i = 0; i < columns_num; ++i) {
78 auto column = block.get_by_position(i).column->assume_mutable();
79 column->insert_from(*columns[i], row_num);
80 }
81 }
82
83 void sort() { sort_block(block, block, sort_desc, block.rows()); }
84};

86template <typename Data>
87class AggregateFunctionSort
88 : public IAggregateFunctionDataHelper<Data, AggregateFunctionSort<Data>> {
89private:
90 static constexpr auto prefix_size = sizeof(Data);
91 AggregateFunctionPtr _nested_func;
92 DataTypes _arguments;
93 const SortDescription& _sort_desc;
94 Block _block;
95 const RuntimeState* _state;
96
97 AggregateDataPtr get_nested_place(AggregateDataPtr __restrict place) const noexcept {
98 return place + prefix_size;
99 }
100
101 ConstAggregateDataPtr get_nested_place(ConstAggregateDataPtr __restrict place) const noexcept {
102 return place + prefix_size;
103 }
104
105public:
106 AggregateFunctionSort(const AggregateFunctionPtr& nested_func, const DataTypes& arguments,
107 const SortDescription& sort_desc, const RuntimeState* state)
108 : IAggregateFunctionDataHelper<Data, AggregateFunctionSort>(
109 arguments, nested_func->get_parameters()),
110 _nested_func(nested_func),
111 _arguments(arguments),
112 _sort_desc(sort_desc),
113 _state(state) {
114 for (const auto& type : _arguments) {
115 _block.insert({type, ""});
116 }
117 }
118
119 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
120 Arena* arena) const override {
121 this->data(place).add(columns, _arguments.size(), row_num);
122 }
123
124 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
125 Arena* arena) const override {
126 this->data(place).merge(this->data(rhs));
127 }
128
129 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
130 this->data(place).serialize(_state, buf);
131 }
132
133 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
134 Arena* arena) const override {
135 this->data(place).deserialize(buf);
136 }
137
138 void insert_result_into(ConstAggregateDataPtr targetplace, IColumn& to) const override {
139 auto place = const_cast<AggregateDataPtr>(targetplace);
140 if (!this->data(place).block.is_empty_column()) {
141 this->data(place).sort();
142
143 ColumnRawPtrs arguments_nested;
144 for (int i = 0; i < _arguments.size() - _sort_desc.size(); i++) {
145 arguments_nested.emplace_back(
146 this->data(place).block.get_by_position(i).column.get());
147 }
148 _nested_func->add_batch_single_place(arguments_nested[0]->size(),
149 get_nested_place(place), arguments_nested.data(),
150 nullptr);
151 }
152
153 _nested_func->insert_result_into(get_nested_place(place), to);
154 }
155
156 size_t size_of_data() const override { return prefix_size + _nested_func->size_of_data(); }
157
158 size_t align_of_data() const override { return _nested_func->align_of_data(); }
159
160 void create(AggregateDataPtr __restrict place) const override {
161 new (place) Data(_sort_desc, _block);
162 _nested_func->create(get_nested_place(place));
163 }
164
165 void destroy(AggregateDataPtr __restrict place) const noexcept override {
166 this->data(place).~Data();
167 _nested_func->destroy(get_nested_place(place));
168 }
169
170 String get_name() const override { return _nested_func->get_name() + "Sort"; }
171
172 DataTypePtr get_return_type() const override { return _nested_func->get_return_type(); }
173};
