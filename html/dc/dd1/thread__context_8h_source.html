98// Using gcc11 compiles thread_local variable on lower versions of GLIBC will report an error,
99// see https://github.com/apache/doris/pull/7911
100//
101// If we want to avoid this error,
102// 1. For non-trivial variables in thread_local, such as std::string, you need to store them as pointers to
103// ensure that thread_local is trivial, these non-trivial pointers will uniformly call destructors elsewhere.
104// 2. The default destructor of the thread_local variable cannot be overridden.
105//
106// This is difficult to implement. Because the destructor is not overwritten, it means that the outside cannot
107// be notified when the thread terminates, and the non-trivial pointers in thread_local cannot be released in time.
108// The func provided by pthread and std::thread doesn't help either.
109//
110// So, kudu Class-scoped static thread local implementation was introduced. Solve the above problem by
111// Thread-scoped thread local + Class-scoped thread local.
112//
113// This may look very trick, but it's the best way I can find.
114//
115// refer to:
116// https://gcc.gnu.org/onlinedocs/gcc-3.3.1/gcc/Thread-Local.html
117// https://stackoverflow.com/questions/12049684/
118// https://sourceware.org/glibc/wiki/Destructor%20support%20for%20thread_local%20variables
119// https://www.jianshu.com/p/756240e837dd
120// https://man7.org/linux/man-pages/man3/pthread_tryjoin_np.3.html
121class ThreadContextPtr {
122public:
123 ThreadContextPtr();
124 // Cannot add destructor `~ThreadContextPtr`, otherwise it will no longer be of type POD, the reason is as above.
125
126 // TCMalloc hook is triggered during ThreadContext construction, which may lead to deadlock.
127 bool init = false;
128
129 DECLARE_STATIC_THREAD_LOCAL(ThreadContext, _ptr);
130};

212// Cache the pointer of bthread local in pthead local,
213// Avoid calling bthread_getspecific frequently to get bthread local, which has performance problems.
214static void pthread_attach_bthread() {
215 bthread_id = bthread_self();
216 bthread_context = static_cast<ThreadContext*>(bthread_getspecific(btls_key));
217 if (bthread_context == nullptr) {
218 // A new bthread starts, two scenarios:
219 // 1. First call to bthread_getspecific (and before any bthread_setspecific) returns NULL
220 // 2. There are not enough reusable btls in btls pool.
221 // else, two scenarios:
222 // 1. A new bthread starts, but get a reuses btls.
223 // 2. A pthread switch occurs. Because the pthread switch cannot be accurately identified at the moment.
224 // So tracker call reset 0 like reuses btls.
225 bthread_context = new ThreadContext;
226 // The brpc server should respond as quickly as possible.
227 bthread_context->thread_mem_tracker_mgr->disable_wait_gc();
228 // set the data so that next time bthread_getspecific in the thread returns the data.
229 CHECK_EQ(0, bthread_setspecific(btls_key, bthread_context));
230 }
231}
232
233static ThreadContext* thread_context() {
234 if (bthread_self() != 0) {
235 if (bthread_self() != bthread_id) {
236 // A new bthread starts or pthread switch occurs, during this period, stop the use of thread_context.
237 thread_context_ptr.init = false;
238 pthread_attach_bthread();
239 thread_context_ptr.init = true;
240 }
241 return bthread_context;
242 } else {
243 return thread_context_ptr._ptr;
244 }
245}

356// Mem Hook to consume thread mem tracker
357// TODO: In the original design, the MemTracker consume method is called before the memory is allocated.
358// If the consume succeeds, the memory is actually allocated, otherwise an exception is thrown.
359// But the statistics of memory through TCMalloc new/delete Hook are after the memory is actually allocated,
360// which is different from the previous behavior.
361#define CONSUME_MEM_TRACKER(size) \
362 do { \
363 if (doris::thread_context_ptr.init) { \
364 doris::thread_context()->thread_mem_tracker_mgr->consume(size); \
365 } \
366 } while (0)
367// NOTE, The LOG cannot be printed in the mem hook. If the LOG statement triggers the mem hook LOG,
368// the nested LOG may cause an unknown crash.
369#define TRY_CONSUME_MEM_TRACKER(size, fail_ret) \
370 do { \
371 if (doris::thread_context_ptr.init) { \
372 if (doris::enable_thread_catch_bad_alloc) { \
373 if (!doris::thread_context()->thread_mem_tracker_mgr->try_consume(size)) { \
374 return fail_ret; \
375 } \
376 } else { \
377 doris::thread_context()->thread_mem_tracker_mgr->consume(size); \
378 } \
379 } \
380 } while (0)

382 do { \

384 doris::thread_context()->thread_mem_tracker_mgr->consume(-size); \
385 } \
386 } while (0)

389#define TRY_CONSUME_THREAD_MEM_TRACKER(size) true
390#define RELEASE_THREAD_MEM_TRACKER(size) (void)0
391#define THREAD_MEM_TRACKER_TRANSFER_TO(size, tracker) (void)0
