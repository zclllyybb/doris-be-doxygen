26template <typename T>
27struct AggregateFunctionHistogramData {
28 using ColVecType =
29 std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<Decimal128>, ColumnVector<T>>;
30
31 void set_parameters(double input_sample_rate, size_t input_max_bucket_num) {
32 if (input_sample_rate > 0 && input_sample_rate <= 1) {
33 sample_rate = input_sample_rate;
34 }
35 if (input_max_bucket_num > 0) {
36 max_bucket_num = (uint32_t)input_max_bucket_num;
37 }
38 }
39
40 void reset() { ordered_map.clear(); }
41
42 void add(const StringRef& value, const UInt64& number = 1) {
43 std::string data = value.to_string();
44 auto it = ordered_map.find(data);
45 if (it != ordered_map.end()) {
46 it->second = it->second + number;
47 } else {
48 ordered_map.insert({data, number});
49 }
50 }
51
52 void add(const T& value, const UInt64& number = 1) {
53 auto it = ordered_map.find(value);
54 if (it != ordered_map.end()) {
55 it->second = it->second + number;
56 } else {
57 ordered_map.insert({value, number});
58 }
59 }
60
61 void merge(const AggregateFunctionHistogramData& rhs) {
62 if (!rhs.sample_rate) {
63 return;
64 }
65
66 for (auto rhs_it : rhs.ordered_map) {
67 auto lhs_it = ordered_map.find(rhs_it.first);
68 if (lhs_it != ordered_map.end()) {
69 lhs_it->second += rhs_it.second;
70 } else {
71 ordered_map.insert({rhs_it.first, rhs_it.second});
72 }
73 }
74 }
75
76 void write(BufferWritable& buf) const {
77 write_binary(sample_rate, buf);
78 write_binary(max_bucket_num, buf);
79
80 uint64_t element_number = (uint64_t)ordered_map.size();
81 write_binary(element_number, buf);
82
83 auto pair_vector = map_to_vector();
84
85 for (auto i = 0; i < element_number; i++) {
86 auto element = pair_vector[i];
87 write_binary(element.second, buf);
88 write_binary(element.first, buf);
89 }
90 }
91
92 void read(BufferReadable& buf) {
93 read_binary(sample_rate, buf);
94 read_binary(max_bucket_num, buf);
95
96 uint64_t element_number = 0;
97 read_binary(element_number, buf);
98
99 ordered_map.clear();
100 std::pair<T, uint64_t> element;
101 for (auto i = 0; i < element_number; i++) {
102 read_binary(element.first, buf);
103 read_binary(element.second, buf);
104 ordered_map.insert(element);
105 }
106 }
107
108 void insert_result_into(IColumn& to) const {
109 auto pair_vector = map_to_vector();
110 for (auto i = 0; i < pair_vector.size(); i++) {
111 const auto& element = pair_vector[i];
112 if constexpr (std::is_same_v<T, std::string>) {
113 static_cast<ColumnString&>(to).insert_data(element.second.c_str(),
114 element.second.length());
115 } else {
116 static_cast<ColVecType&>(to).get_data().push_back(element.second);
117 }
118 }
119 }
120
121 std::string get(const DataTypePtr& data_type) const {
122 std::vector<Bucket<T>> buckets;
123 rapidjson::StringBuffer buffer;
124 build_bucket_from_data(buckets, ordered_map, sample_rate, max_bucket_num);
125 build_json_from_bucket(buffer, buckets, data_type, sample_rate, max_bucket_num);
126 return std::string(buffer.GetString());
127 }
128
129 std::vector<std::pair<uint64_t, T>> map_to_vector() const {
130 std::vector<std::pair<uint64_t, T>> pair_vector;
131 for (auto it : ordered_map) {
132 pair_vector.emplace_back(it.second, it.first);
133 }
134 return pair_vector;
135 }
136
137private:
138 double sample_rate = 0.2;
139 uint32_t max_bucket_num = 128;
140 std::map<T, uint64_t> ordered_map;
141};

143template <typename Data, typename T, bool has_input_param>
144class AggregateFunctionHistogram final
145 : public IAggregateFunctionDataHelper<
146 Data, AggregateFunctionHistogram<Data, T, has_input_param>> {
147public:
148 using ColVecType = ColumnVectorOrDecimal<T>;
149
150 AggregateFunctionHistogram() = default;
151 AggregateFunctionHistogram(const DataTypes& argument_types_)
152 : IAggregateFunctionDataHelper<Data,
153 AggregateFunctionHistogram<Data, T, has_input_param>>(
154 argument_types_, {}),
155 _argument_type(argument_types_[0]) {}
156
157 std::string get_name() const override { return "histogram"; }
158
159 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeString>(); }
160
161 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
162 Arena* arena) const override {
163 if (columns[0]->is_null_at(row_num)) {
164 return;
165 }
166
167 if (has_input_param) {
168 this->data(place).set_parameters(
169 static_cast<const ColumnFloat64*>(columns[1])->get_element(row_num),
170 static_cast<const ColumnInt32*>(columns[2])->get_element(row_num));
171 }
172
173 if constexpr (std::is_same_v<T, std::string>) {
174 this->data(place).add(
175 static_cast<const ColumnString&>(*columns[0]).get_data_at(row_num));
176 } else {
177 this->data(place).add(static_cast<const ColVecType&>(*columns[0]).get_data()[row_num]);
178 }
179 }
180
181 void reset(AggregateDataPtr place) const override { this->data(place).reset(); }
182
183 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
184 Arena* arena) const override {
185 this->data(place).merge(this->data(rhs));
186 }
187
188 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
189 this->data(place).write(buf);
190 }
191
192 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
193 Arena*) const override {
194 this->data(place).read(buf);
195 }
196
197 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
198 const std::string bucket_json = this->data(place).get(_argument_type);
199 assert_cast<ColumnString&>(to).insert_data(bucket_json.c_str(), bucket_json.length());
200 }
201
202private:
203 DataTypePtr _argument_type;
204};
