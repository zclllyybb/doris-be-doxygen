34// CachePathPolicy it to define which cache path should be used
35// for the local cache of the given file(path).
36// The dervied class should implement get_cache_path() method
37class CachePathPolicy {
38public:
39 // path: the path of file which will be cached
40 // return value: the cache path of the given file.
41 virtual std::string get_cache_path(const std::string& path) const { return ""; }
42};
43
44class NoCachePathPolicy : public CachePathPolicy {
45public:
46 NoCachePathPolicy() = default;
47 std::string get_cache_path(const std::string& path) const override { return ""; }
48};
49
50class SegmentCachePathPolicy : public CachePathPolicy {
51public:
52 SegmentCachePathPolicy() = default;
53
54 void set_cache_path(const std::string& cache_path) { _cache_path = cache_path; }
55
56 std::string get_cache_path(const std::string& path) const override { return _cache_path; }
57
58private:
59 std::string _cache_path;
60};
61
62class FileBlockCachePathPolicy : public CachePathPolicy {
63public:
64 FileBlockCachePathPolicy() = default;
65 std::string get_cache_path(const std::string& path) const override { return path; }
66};
