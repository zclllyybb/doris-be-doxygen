40// This type of page use dictionary encoding for strings.
41// There is only one dictionary page for all the data pages within a column.
42//
43// Layout for dictionary encoded page:
44// Either header + embedded codeword page, which can be encoded with any
45// int PageBuilder, when mode_ = DICT_ENCODING.
46// Or header + embedded BinaryPlainPage, when mode_ = PLAIN_ENCODING.
47// Data pages start with mode_ = DICT_ENCODING, when the size of dictionary
48// page go beyond the option_->dict_page_size, the subsequent data pages will switch
49// to string plain page automatically.
50class BinaryDictPageBuilder : public PageBuilder {
51public:
52 BinaryDictPageBuilder(const PageBuilderOptions& options);

62 size_t count() const override;

64 uint64_t size() const override;
65
66 Status get_dictionary_page(OwnedSlice* dictionary_page) override;

68 Status get_first_value(void* value) const override;
69
70 Status get_last_value(void* value) const override;
71
72private:
73 PageBuilderOptions _options;
74 bool _finished;
75
76 std::unique_ptr<PageBuilder> _data_page_builder;
77
78 std::unique_ptr<BinaryPlainPageBuilder<OLAP_FIELD_TYPE_VARCHAR>> _dict_builder;

80 EncodingTypePB _encoding_type;
81 struct HashOfSlice {
82 size_t operator()(const Slice& slice) const {

85 };

95 bool _has_empty = false;

97};
