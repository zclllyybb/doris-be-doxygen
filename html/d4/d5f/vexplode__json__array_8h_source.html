31struct ParsedData {
32 static std::string true_value;
33 static std::string false_value;
34
35 // The number parsed from json array
36 // the `_backup` saved the real number entity.
37 std::vector<void*> _data;
38 std::vector<StringRef> _data_string;
39 std::vector<int64_t> _backup_int;
40 std::vector<double> _backup_double;
41 std::vector<std::string> _backup_string;
42 std::vector<bool> _string_nulls;
43 char tmp_buf[128] = {0};
44
45 void reset(ExplodeJsonArrayType type) {
46 switch (type) {
47 case ExplodeJsonArrayType::INT:
48 _data.clear();
49 _backup_int.clear();
50 break;
51 case ExplodeJsonArrayType::DOUBLE:
52 _data.clear();
53 _backup_double.clear();
54 break;
55 case ExplodeJsonArrayType::STRING:
56 _data_string.clear();
57 _backup_string.clear();
58 _string_nulls.clear();
59 break;
60 default:
61 CHECK(false) << type;
62 break;
63 }
64 }
65
66 void* get_value(ExplodeJsonArrayType type, int64_t offset, bool real = false) {
67 switch (type) {
68 case ExplodeJsonArrayType::INT:
69 case ExplodeJsonArrayType::DOUBLE:
70 return _data[offset];
71 case ExplodeJsonArrayType::STRING:
72 return _string_nulls[offset] ? nullptr
73 : real ? reinterpret_cast<void*>(_backup_string[offset].data())
74 : &_data_string[offset];
75 default:
76 return nullptr;
77 }
78 }
79
80 int64 get_value_length(ExplodeJsonArrayType type, int64_t offset) {
81 if (type == ExplodeJsonArrayType::STRING && !_string_nulls[offset]) {
82 return _backup_string[offset].size();
83 }
84 return 0;
85 }
86
87 int set_output(ExplodeJsonArrayType type, rapidjson::Document& document);
88};
