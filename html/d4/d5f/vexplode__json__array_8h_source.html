29enum ExplodeJsonArrayType { INT = 0, DOUBLE, STRING };
30
31struct ParsedData {
32 static std::string true_value;
33 static std::string false_value;
34
35 // The number parsed from json array
36 // the `_backup` saved the real number entity.
37 std::vector<void*> _data;
38 std::vector<StringRef> _data_string;
39 std::vector<int64_t> _backup_int;
40 std::vector<double> _backup_double;
41 std::vector<std::string> _backup_string;
42 std::vector<bool> _string_nulls;
43 char tmp_buf[128] = {0};
44
45 void reset(ExplodeJsonArrayType type) {
46 switch (type) {
47 case ExplodeJsonArrayType::INT:
48 _data.clear();
49 _backup_int.clear();
50 break;
51 case ExplodeJsonArrayType::DOUBLE:
52 _data.clear();
53 _backup_double.clear();
54 break;
55 case ExplodeJsonArrayType::STRING:
56 _data_string.clear();
57 _backup_string.clear();
58 _string_nulls.clear();
59 break;
60 default:
61 CHECK(false) << type;
62 break;
63 }
64 }
65
66 void get_value(ExplodeJsonArrayType type, int64_t offset, void** output, bool real = false) {
67 switch (type) {
68 case ExplodeJsonArrayType::INT:
69 case ExplodeJsonArrayType::DOUBLE:
70 *output = _data[offset];
71 break;
72 case ExplodeJsonArrayType::STRING:
73 *output = _string_nulls[offset] ? nullptr
74 : real ? reinterpret_cast<void*>(_backup_string[offset].data())
75 : &_data_string[offset];
76 break;
77 default:
78 CHECK(false) << type;
79 }
80 }
81
82 void get_value_length(ExplodeJsonArrayType type, int64_t offset, int64_t* length) {
83 switch (type) {
84 case ExplodeJsonArrayType::INT:
85 case ExplodeJsonArrayType::DOUBLE:
86 break;
87 case ExplodeJsonArrayType::STRING:
88 *length = _string_nulls[offset] ? -1 : _backup_string[offset].size();
89 break;
90 default:
91 CHECK(false) << type;
92 }
93 }
94
95 int set_output(ExplodeJsonArrayType type, rapidjson::Document& document);
96};
