85 /* Now the output tuples for table function node is base_table_tuple + tf1 + tf2 + ...
86 But not all slots are used, the real used slots are inside table_function_node.outputSlotIds.
87 For case like explode_bitmap:
88 SELECT a2,count(*) as a3 FROM A WHERE a1 IN
89 (SELECT c1 FROM B LATERAL VIEW explode_bitmap(b1) C as c1)
90 GROUP BY a2 ORDER BY a3;
91 Actually we only need to output column c1, no need to output columns in bitmap table B.
92 Copy large bitmap columns are very expensive and slow.
93
94 Here we check if the slot is really used, otherwise we avoid copy it and just insert a default value.
95
96 A better solution is:
97 1. FE: create a new output tuple based on the real output slots;
98 2. BE: refractor (V)TableFunctionNode output rows based no the new tuple;
99 */
100 inline bool slot_need_copy(SlotId slot_id) const {
101 auto id = _output_slots[slot_id]->id();
102 return (id < _output_slot_ids.size()) && (_output_slot_ids[id]);
103 }
104
105 Status get_expanded_block(RuntimeState* state, Block* output_block, bool* eos);
106
