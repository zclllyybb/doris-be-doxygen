62// The new Split API
63// aka Split2
64// aka strings::Split()
65//
66// This string splitting API consists of a Split() function in the ::strings
67// namespace and a handful of delimiter objects in the ::strings::delimiter
68// namespace (more on delimiter objects below). The Split() function always
69// takes two arguments: the text to be split and the delimiter on which to split
70// the text. An optional third argument may also be given, which is a Predicate
71// functor that will be used to filter the results, e.g., to skip empty strings
72// (more on predicates below). The Split() function adapts the returned
73// collection to the type specified by the caller.
74//
75// Example 1:
76// // Splits the given string on commas. Returns the results in a
77// // vector of strings.
78// vector<string> v = strings::Split("a,b,c", ",");
79// assert(v.size() == 3);
80//
81// Example 2:
82// // By default, empty strings are *included* in the output. See the
83// // strings::SkipEmpty predicate below to omit them.
84// vector<string> v = strings::Split("a,b,,c", ",");
85// assert(v.size() == 4); // "a", "b", "", "c"
86// v = strings::Split("", ",");
87// assert(v.size() == 1); // v contains a single ""
88//
89// Example 3:
90// // Splits the string as in the previous example, except that the results
91// // are returned as StringPiece objects. Note that because we are storing
92// // the results within StringPiece objects, we have to ensure that the input
93// // string outlives any results.
94// vector<StringPiece> v = strings::Split("a,b,c", ",");
95// assert(v.size() == 3);
96//
97// Example 4:
98// // Stores results in a set<string>.
99// set<string> a = strings::Split("a,b,c,a,b,c", ",");
100// assert(a.size() == 3);
101//
102// Example 5:
103// // Stores results in a map. The map implementation assumes that the input
104// // is provided as a series of key/value pairs. For example, the 0th element
105// // resulting from the split will be stored as a key to the 1st element. If
106// // an odd number of elements are resolved, the last element is paired with
107// // a default-constructed value (e.g., empty string).
108// map<string, string> m = strings::Split("a,b,c", ",");
109// assert(m.size() == 2);
110// assert(m["a"] == "b");
111// assert(m["c"] == ""); // last component value equals ""
112//
113// Example 6:
114// // Splits on the empty string, which results in each character of the input
115// // string becoming one element in the output collection.
116// vector<string> v = strings::Split("abc", "");
117// assert(v.size() == 3);
118//
119// Example 7:
120// // Stores first two split strings as the members in an std::pair.
121// std::pair<string, string> p = strings::Split("a,b,c", ",");
122// EXPECT_EQ("a", p.first);
123// EXPECT_EQ("b", p.second);
124// // "c" is omitted because std::pair can hold only two elements.
125//
126// As illustrated above, the Split() function adapts the returned collection to
127// the type specified by the caller. The returned collections may contain
128// string, StringPiece, Cord, or any object that has a constructor (explicit or
129// not) that takes a single StringPiece argument. This pattern works for all
130// standard STL containers including vector, list, deque, set, multiset, map,
131// multimap, unordered_set and unordered_map, and even std::pair which is not
132// actually a container.
133//
134// Splitting to std::pair is an interesting case because it can hold only two
135// elements and is not a collection type. When splitting to an std::pair the
136// first two split strings become the std::pair's .first and .second members
137// respectively. The remaining split substrings are discarded. If there are less
138// than two split substrings, the empty string is used for the corresponding
139// std::pair member.
140//
141// The strings::Split() function can be used multiple times to perform more
142// complicated splitting logic, such as intelligently parsing key-value pairs.
143// For example
144//
145// // The input string "a=b=c,d=e,f=,g" becomes
146// // { "a" => "b=c", "d" => "e", "f" => "", "g" => "" }
147// map<string, string> m;
148// for (StringPiece sp : strings::Split("a=b=c,d=e,f=,g", ",")) {
149// m.insert(strings::Split(sp, strings::delimiter::Limit("=", 1)));
150// }
151// EXPECT_EQ("b=c", m.find("a")->second);
152// EXPECT_EQ("e", m.find("d")->second);
153// EXPECT_EQ("", m.find("f")->second);
154// EXPECT_EQ("", m.find("g")->second);
155//
156// The above example stores the results in an std::map. But depending on your
157// data requirements, you can just as easily store the results in an
158// std::multimap or even a vector<std::pair<>>.
159//
160//
161// Delimiters
162//
163// The Split() function also takes a second argument that is a delimiter. This
164// delimiter is actually an object that defines the boundaries between elements
165// in the provided input. If a string (const char*, ::string, or StringPiece) is
166// passed in place of an explicit Delimiter object, the argument is implicitly
167// converted to a ::strings::delimiter::Literal.
168//
169// With this split API comes the formal concept of a Delimiter (big D). A
170// Delimiter is an object with a Find() function that knows how find the first
171// occurrence of itself in a given StringPiece. Models of the Delimiter concept
172// represent specific kinds of delimiters, such as single characters,
173// substrings, or even regular expressions.
174//
175// The following Delimiter objects are provided as part of the Split() API:
176//
177// - Literal (default)
178// - AnyOf
179// - Limit
180//
181// The following are examples of using some provided Delimiter objects:
182//
183// Example 1:
184// // Because a string literal is converted to a strings::delimiter::Literal,
185// // the following two splits are equivalent.
186// vector<string> v1 = strings::Split("a,b,c", ","); // (1)
187// using ::strings::delimiter::Literal;
188// vector<string> v2 = strings::Split("a,b,c", Literal(",")); // (2)
189//
190// Example 2:
191// // Splits on any of the characters specified in the delimiter string.
192// using ::strings::delimiter::AnyOf;
193// vector<string> v = strings::Split("a,b;c-d", AnyOf(",;-"));
194// assert(v.size() == 4);
195//
196// Example 3:
197// // Uses the Limit meta-delimiter to limit the number of matches a delimiter
198// // can have. In this case, the delimiter of a Literal comma is limited to
199// // to matching at most one time. The last element in the returned
200// // collection will contain all unsplit pieces, which may contain instances
201// // of the delimiter.
202// using ::strings::delimiter::Limit;
203// vector<string> v = strings::Split("a,b,c", Limit(",", 1));
204// assert(v.size() == 2); // Limited to 1 delimiter; so two elements found
205// assert(v[0] == "a");
206// assert(v[1] == "b,c");
207//
208//
209// Predicates
210//
211// Predicates can filter the results of a Split() operation by determining
212// whether or not a resultant element is included in the result set. A predicate
213// may be passed as an *optional* third argument to the Split() function.
214//
215// Predicates are unary functions (or functors) that take a single StringPiece
216// argument and return bool indicating whether the argument should be included
217// (true) or excluded (false).
218//
219// One example where this is useful is when filtering out empty substrings. By
220// default, empty substrings may be returned by strings::Split(), which is
221// similar to the way split functions work in other programming languages. For
222// example:
223//
224// // Empty strings *are* included in the returned collection.
225// vector<string> v = strings::Split(",a,,b,", ",");
226// assert(v.size() == 5); // v[0] == "", v[1] == "a", v[2] == "", ...
227//
228// These empty strings can be filtered out of the results by simply passing the
229// provided SkipEmpty predicate as the third argument to the Split() function.
230// SkipEmpty does not consider a string containing all whitespace to be empty.
231// For that behavior use the SkipWhitespace predicate. For example:
232//
233// Example 1:
234// // Uses SkipEmpty to omit empty strings. Strings containing whitespace are
235// // not empty and are therefore not skipped.
236// using strings::SkipEmpty;
237// vector<string> v = strings::Split(",a, ,b,", ",", SkipEmpty());
238// assert(v.size() == 3);
239// assert(v[0] == "a");
240// assert(v[1] == " "); // <-- The whitespace makes the string not empty.
241// assert(v[2] == "b");
242//
243// Example 2:
244// // Uses SkipWhitespace to skip all strings that are either empty or contain
245// // only whitespace.
246// using strings::SkipWhitespace;
247// vector<string> v = strings::Split(",a, ,b,", ",", SkipWhitespace());
248// assert(v.size() == 2);
249// assert(v[0] == "a");
250// assert(v[1] == "b");
251//
252//
253// Differences between Split1 and Split2
254//
255// Split2 is the strings::Split() API described above. Split1 is a name for the
256// collection of legacy Split*() functions declared later in this file. Most of
257// the Split1 functions follow a set of conventions that don't necessarily match
258// the conventions used in Split2. The following are some of the important
259// differences between Split1 and Split2:
260//
261// Split1 -> Split2
262// ----------------
263// Append -> Assign:
264// The Split1 functions all returned their output collections via a pointer to
265// an out parameter as is typical in Google code. In some cases the comments
266// explicitly stated that results would be *appended* to the output
267// collection. In some cases it was ambiguous whether results were appended.
268// This ambiguity is gone in the Split2 API as results are always assigned to
269// the output collection, never appended.
270//
271// AnyOf -> Literal:
272// Most Split1 functions treated their delimiter argument as a string of
273// individual byte delimiters. For example, a delimiter of ",;" would split on
274// "," and ";", not the substring ",;". This behavior is equivalent to the
275// Split2 delimiter strings::delimiter::AnyOf, which is *not* the default. By
276// default, strings::Split() splits using strings::delimiter::Literal() which
277// would treat the whole string ",;" as a single delimiter string.
278//
279// SkipEmpty -> allow empty:
280// Most Split1 functions omitted empty substrings in the results. To keep
281// empty substrings one would have to use an explicitly named
282// Split*AllowEmpty() function. This behavior is reversed in Split2. By
283// default, strings::Split() *allows* empty substrings in the output. To skip
284// them, use the strings::SkipEmpty predicate.
285//
286// string -> user's choice:
287// Most Split1 functions return collections of string objects. Some return
288// char*, but the type returned is dictated by each Split1 function. With
289// Split2 the caller can choose which string-like object to return. (Note:
290// char* C-strings are not supported in Split2--use StringPiece instead).
291//
292
293// Definitions of the main Split() function.
294template <typename Delimiter>
295inline internal::Splitter<Delimiter> Split(StringPiece text, Delimiter d) {
296 return internal::Splitter<Delimiter>(text, d);
297}

299template <typename Delimiter, typename Predicate>
300inline internal::Splitter<Delimiter, Predicate> Split(StringPiece text, Delimiter d, Predicate p) {
301 return internal::Splitter<Delimiter, Predicate>(text, d, p);
302}

304namespace delimiter {
305// A Delimiter object represents a single separator, such as a character,
306// literal string, or regular expression. A Delimiter object must have the
307// following member:
308//
309// StringPiece Find(StringPiece text);
310//
311// This Find() member function should return a StringPiece referring to the next
312// occurrence of the represented delimiter within the given string text. If no
313// delimiter is found in the given text, a zero-length StringPiece referring to
314// text.end() should be returned (e.g., StringPiece(text.end(), 0)). It is
315// important that the returned StringPiece always be within the bounds of the
316// StringPiece given as an argument--it must not refer to a string that is
317// physically located outside of the given string. The following example is a
318// simple Delimiter object that is created with a single char and will look for
319// that char in the text given to the Find() function:
320//
321// struct SimpleDelimiter {
322// const char c_;
323// explicit SimpleDelimiter(char c) : c_(c) {}
324// StringPiece Find(StringPiece text) {
325// int pos = text.find(c_);
326// if (pos == StringPiece::npos) return StringPiece(text.end(), 0);
327// return StringPiece(text, pos, 1);
328// }
329// };
330
331// Represents a literal string delimiter. Examples:
332//
333// using ::strings::delimiter::Literal;
334// vector<string> v = strings::Split("a=>b=>c", Literal("=>"));
335// assert(v.size() == 3);
336// assert(v[0] == "a");
337// assert(v[1] == "b");
338// assert(v[2] == "c");
339//
340// The next example uses the empty string as a delimiter.
341//
342// using ::strings::delimiter::Literal;
343// vector<string> v = strings::Split("abc", Literal(""));
344// assert(v.size() == 3);
345// assert(v[0] == "a");
346// assert(v[1] == "b");
347// assert(v[2] == "c");
348//
349class Literal {
350public:
351 explicit Literal(StringPiece sp);
352 StringPiece Find(StringPiece text) const;
353
354private:
355 const string delimiter_;
356};
357
358// Represents a delimiter that will match any of the given byte-sized
359// characters. AnyOf is similar to Literal, except that AnyOf uses
360// StringPiece::find_first_of() and Literal uses StringPiece::find(). AnyOf
361// examples:
362//
363// using ::strings::delimiter::AnyOf;
364// vector<string> v = strings::Split("a,b=c", AnyOf(",="));
365//
366// assert(v.size() == 3);
367// assert(v[0] == "a");
368// assert(v[1] == "b");
369// assert(v[2] == "c");
370//
371// If AnyOf is given the empty string, it behaves exactly like Literal and
372// matches each individual character in the input string.
373//
374// Note: The string passed to AnyOf is assumed to be a string of single-byte
375// ASCII characters. AnyOf does not work with multi-byte characters.
376class AnyOf {
377public:
378 explicit AnyOf(StringPiece sp);
379 StringPiece Find(StringPiece text) const;
380
381private:
382 const string delimiters_;
383};

385// Wraps another delimiter and sets a max number of matches for that delimiter.
386// Create LimitImpls using the Limit() function. Example:
387//
388// using ::strings::delimiter::Limit;
389// vector<string> v = strings::Split("a,b,c,d", Limit(",", 2));
390//
391// assert(v.size() == 3); // Split on 2 commas, giving a vector with 3 items
392// assert(v[0] == "a");
393// assert(v[1] == "b");
394// assert(v[2] == "c,d");
395//
396template <typename Delimiter>
397class LimitImpl {
398public:
399 LimitImpl(Delimiter delimiter, int limit)
400 : delimiter_(std::move(delimiter)), limit_(limit), count_(0) {}
401 StringPiece Find(StringPiece text) {
402 if (count_++ == limit_) {
403 return StringPiece(text.end(), 0); // No more matches.
404 }
405 return delimiter_.Find(text);
406 }
407
408private:
409 Delimiter delimiter_;
410 const int limit_;
411 int count_;
412};
413
414// Overloaded Limit() function to create LimitImpl<> objects. Uses the Delimiter
415// Literal as the default if string-like objects are passed as the delimiter
416// parameter. This is similar to the overloads for Split() below.
417template <typename Delimiter>
418inline LimitImpl<Delimiter> Limit(Delimiter delim, int limit) {
419 return LimitImpl<Delimiter>(delim, limit);
420}
421
422inline LimitImpl<Literal> Limit(const char* s, int limit) {
423 return LimitImpl<Literal>(Literal(s), limit);
424}
425
426inline LimitImpl<Literal> Limit(const string& s, int limit) {
427 return LimitImpl<Literal>(Literal(s), limit);
428}
429
430inline LimitImpl<Literal> Limit(StringPiece s, int limit) {
431 return LimitImpl<Literal>(Literal(s), limit);
432}

436//
437// Predicates are functors that return bool indicating whether the given
438// StringPiece should be included in the split output. If the predicate returns
439// false then the string will be excluded from the output from strings::Split().
440//
441
442// Always returns true, indicating that all strings--including empty
443// strings--should be included in the split output. This predicate is not
444// strictly needed because this is the default behavior of the strings::Split()
445// function. But it might be useful at some call sites to make the intent
446// explicit.
447//
448// vector<string> v = Split(" a , ,,b,", ",", AllowEmpty());
449// EXPECT_THAT(v, ElementsAre(" a ", " ", "", "b", ""));
450struct AllowEmpty {
451 bool operator()(StringPiece sp) const { return true; }
452};
453
454// Returns false if the given StringPiece is empty, indicating that the
455// strings::Split() API should omit the empty string.
456//
457// vector<string> v = Split(" a , ,,b,", ",", SkipEmpty());
458// EXPECT_THAT(v, ElementsAre(" a ", " ", "b"));
459struct SkipEmpty {
460 bool operator()(StringPiece sp) const { return !sp.empty(); }
461};
462
463// Returns false if the given StringPiece is empty or contains only whitespace,
464// indicating that the strings::Split() API should omit the string.
465//
466// vector<string> v = Split(" a , ,,b,", ",", SkipWhitespace());
467// EXPECT_THAT(v, ElementsAre(" a ", "b"));
468struct SkipWhitespace {
469 bool operator()(StringPiece sp) const {
470 StripWhiteSpace(&sp);
471 return !sp.empty();
472 }
473};

1084 bool (*parse)(const string& str,

1094 for (const auto& piece : pieces) {
1095 typename Container::value_type t;

1097 insert_policy(result, t);
1098 } else {
1099 retval = false;
1100 }
1101 }
1102 return retval;
1103}
