7// Thread safety
8// -------------
9//
10// Writes require external synchronization, most likely a mutex.
11// Reads require a guarantee that the SkipList will not be destroyed
12// while the read is in progress. Apart from that, reads progress
13// without any internal locking or synchronization.
14//
15// Invariants:
16//
17// (1) Allocated nodes are never deleted until the SkipList is
18// destroyed. This is trivially guaranteed by the code since we
19// never delete any skip list nodes.
20//
21// (2) The contents of a Node except for the next/prev pointers are
22// immutable after the Node has been linked into the SkipList.
23// Only Insert() modifies the list, and it is careful to initialize
24// a node and use release-stores to publish the nodes in one or
25// more lists.
26//
27// ... prev vs. next pointer ordering ...
28

38template <typename Key, class Comparator>
39class SkipList {
40private:
41 struct Node;
42 enum { kMaxHeight = 12 };
43
44public:
45 typedef Key key_type;
46 // One Hint object is to show position info of one row.
47 // It is used in the following scenarios:
48 // // 1. check for existence
49 // bool is_exist = skiplist->Find(key, &hint);
50 // // 2. Do something separately based on the value of is_exist
51 // if (is_exist) {
52 // do_something1 ();
53 // } else {
54 // do_something2 ();
55 // skiplist->InsertWithHint(key, is_exist, hint);
56 // }
57 //
58 // Note: The user should guarantee that there must not be any other insertion
59 // between calling Find() and InsertWithHint().
60 struct Hint {
61 Node* curr;
62 Node* prev[kMaxHeight];
63 };
64
65 // Create a new SkipList object that will use "cmp" for comparing keys,
66 // and will allocate memory using "*arena".
67 // NOTE: Objects allocated in the arena must remain allocated for
68 // the lifetime of the skiplist object.
69 explicit SkipList(Comparator* cmp, vectorized::Arena* arena, bool can_dup);
70
71 // Insert key into the list.
72 void Insert(const Key& key, bool* overwritten);
73 // Use hint to insert a key. the hint is from previous Find()
74 void InsertWithHint(const Key& key, bool is_exist, Hint* hint);
75
76 // Returns true if an entry that compares equal to key is in the list.
77 bool Contains(const Key& key) const;
78 // Like Contains(), but it will return the position info as a hint. We can use this
79 // position info to insert directly using InsertWithHint().
80 bool Find(const Key& key, Hint* hint) const;

86 // The returned iterator is not valid.

121private:
122 // Immutable after construction

138

141 bool Equal(const Key& a, const Key& b) const { return ((*compare_)(a, b) == 0); }
142
143 // Return true if key is greater than the data stored in "n"
144 bool KeyIsAfterNode(const Key& key, Node* n) const;
145
146 // Return the earliest node that comes at or after key.
147 // Return nullptr if there is no such node.
148 //
149 // If prev is non-nullptr, fills prev[level] with pointer to previous
150 // node at "level" for every level in [0..max_height_-1].
151 Node* FindGreaterOrEqual(const Key& key, Node** prev) const;
152
153 // Return the latest node with a key < key.
154 // Return head_ if there is no such node.
155 Node* FindLessThan(const Key& key) const;
156
157 // Return the last node in the list.
158 // Return head_ if list is empty.
159 Node* FindLast() const;

166// Implementation details follow
167template <typename Key, class Comparator>
168struct SkipList<Key, Comparator>::Node {
169 explicit Node(const Key& k) : key(k) {}
170
171 Key const key;
172
173 // Accessors/mutators for links. Wrapped in methods so we can
174 // add the appropriate barriers as necessary.
175 Node* Next(int n) {
176 DCHECK(n >= 0);
177 // Use an 'acquire load' so that we observe a fully initialized
178 // version of the returned Node.
179 return (next_[n].load(std::memory_order_acquire));
180 }
181 void SetNext(int n, Node* x) {
182 DCHECK(n >= 0);
183 // Use a 'release store' so that anybody who reads through this
184 // pointer observes a fully initialized version of the inserted node.
185 next_[n].store(x, std::memory_order_release);
186 }
187
188 // No-barrier variants that can be safely used in a few locations.
189 Node* NoBarrier_Next(int n) {
190 DCHECK(n >= 0);
191 return next_[n].load(std::memory_order_relaxed);
192 }
193 void NoBarrier_SetNext(int n, Node* x) {
194 DCHECK(n >= 0);
195 next_[n].store(x, std::memory_order_relaxed);
196 }
197
198private:
199 // Array of length equal to the node height. next_[0] is lowest level link.
200 std::atomic<Node*> next_[1];
201};

262template <typename Key, class Comparator>
263int SkipList<Key, Comparator>::RandomHeight() {
264 // Increase height with probability 1 in kBranching
265 static const unsigned int kBranching = 4;
266 int height = 1;
267 while (height < kMaxHeight && ((rnd_.Next() % kBranching) == 0)) {
268 height++;
269 }
270 DCHECK(height > 0);
271 DCHECK(height <= kMaxHeight);
272 return height;
273}

281template <typename Key, class Comparator>
282typename SkipList<Key, Comparator>::Node* SkipList<Key, Comparator>::FindGreaterOrEqual(
283 const Key& key, Node** prev) const {
284 Node* x = head_;
285 int level = GetMaxHeight() - 1;
286 while (true) {
287 Node* next = x->Next(level);
288 if (KeyIsAfterNode(key, next)) {
289 // Keep searching in this list
290 x = next;
291 } else {
292 if (prev != nullptr) prev[level] = x;
293 if (level == 0) {
294 return next;
295 } else {
296 // Switch to next list
297 level--;
298 }
299 }
300 }
301}

303template <typename Key, class Comparator>
304typename SkipList<Key, Comparator>::Node* SkipList<Key, Comparator>::FindLessThan(
305 const Key& key) const {
306 Node* x = head_;
307 int level = GetMaxHeight() - 1;
308 while (true) {
309 DCHECK(x == head_ || (*compare_)(x->key, key) < 0);
310 Node* next = x->Next(level);
311 if (next == nullptr || (*compare_)(next->key, key) >= 0) {
312 if (level == 0) {
313 return x;
314 } else {
315 // Switch to next list
316 level--;
317 }
318 } else {
319 x = next;
320 }
321 }
322}

324template <typename Key, class Comparator>
325typename SkipList<Key, Comparator>::Node* SkipList<Key, Comparator>::FindLast() const {
326 Node* x = head_;
327 int level = GetMaxHeight() - 1;
328 while (true) {
329 Node* next = x->Next(level);
330 if (next == nullptr) {
331 if (level == 0) {
332 return x;
333 } else {
334 // Switch to next list
335 level--;
336 }
337 } else {
338 x = next;
339 }
340 }

356template <typename Key, class Comparator>
357void SkipList<Key, Comparator>::Insert(const Key& key, bool* overwritten) {
358 // TODO(opt): We can use a barrier-free variant of FindGreaterOrEqual()
359 // here since Insert() is externally synchronized.
360 Node* prev[kMaxHeight];
361 Node* x = FindGreaterOrEqual(key, prev);
362
363#ifndef BE_TEST
364 // The key already exists and duplicate keys are not allowed, so we need to aggregate them
365 if (!_can_dup && x != nullptr && Equal(key, x->key)) {
366 *overwritten = true;
367 return;
368 }
369#endif
370
371 *overwritten = false;
372 // Our data structure does not allow duplicate insertion
373 int height = RandomHeight();
374 if (height > GetMaxHeight()) {
375 for (int i = GetMaxHeight(); i < height; i++) {
376 prev[i] = head_;
377 }
378 //fprintf(stderr, "Change height from %d to %d\n", max_height_, height);
379
380 // It is ok to mutate max_height_ without any synchronization
381 // with concurrent readers. A concurrent reader that observes
382 // the new value of max_height_ will see either the old value of
383 // new level pointers from head_ (nullptr), or a new value set in
384 // the loop below. In the former case the reader will
385 // immediately drop to the next level since nullptr sorts after all
386 // keys. In the latter case the reader will use the new node.
387 max_height_.store(height, std::memory_order_relaxed);
388 }
389
390 x = NewNode(key, height);
391 for (int i = 0; i < height; i++) {
392 // NoBarrier_SetNext() suffices since we will add a barrier when
393 // we publish a pointer to "x" in prev[i].
394 x->NoBarrier_SetNext(i, prev[i]->NoBarrier_Next(i));
395 prev[i]->SetNext(i, x);
396 }
397}

400template <typename Key, class Comparator>
401void SkipList<Key, Comparator>::InsertWithHint(const Key& key, bool is_exist, Hint* hint) {
402 Node* x = hint->curr;
403 DCHECK(!is_exist || x) << "curr pointer must not be null if row exists";
404
405#ifndef BE_TEST
406 // The key already exists and duplicate keys are not allowed, so we need to aggregate them
407 if (!_can_dup && is_exist) {
408 return;
409 }
410#endif
411
412 Node** prev = hint->prev;
413 // Our data structure does not allow duplicate insertion
414 int height = RandomHeight();
415 if (height > GetMaxHeight()) {
416 for (int i = GetMaxHeight(); i < height; i++) {
417 prev[i] = head_;
418 }
419 //fprintf(stderr, "Change height from %d to %d\n", max_height_, height);

426 // immediately drop to the next level since nullptr sorts after all
427 // keys. In the latter case the reader will use the new node.
428 max_height_.store(height, std::memory_order_relaxed);
