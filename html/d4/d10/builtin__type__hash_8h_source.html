29// This function hashes pointer sized items and returns a 32b hash,
30// convenienty hiding the fact that pointers may be 32b or 64b,
31// depending on the architecture.
32inline uint32 Hash32PointerWithSeed(const void* p, uint32 seed) {
33 uintptr_t pvalue = reinterpret_cast<uintptr_t>(p);
34 uint32 h = seed;
35 // Hash the pointer 32b at a time.
36 for (size_t i = 0; i < sizeof(pvalue); i += 4) {
37 h = Hash32NumWithSeed(static_cast<uint32>(pvalue >> (i * 8)), h);
38 }
39 return h;
40}

42// ----------------------------------------------------------------------
43// Hash64FloatWithSeed
44// Hash64DoubleWithSeed
45// Functions for computing a hash value of floating-point numbers.
46// On systems where float and double comply with IEEE 754, these hashes
47// guarantee that if a == b, Hash64FloatWithSeed(a, c) ==
48// Hash64FloatWithSeed(b, c). Note that NaN does not compare equal to
49// itself, so two NaN inputs will not necessarily hash to the same value.
50//
51// It is often a mistake to compare floating-point values for equality,
52// since floating-point computations do not produce exact values, due to
53// rounding. If equality comparison doesn't make sense in your situation,
54// hashing almost certainly doesn't make sense either.
55//
56// Not guaranteed to return the same value in different builds, or to
57// avoid any reserved values.
58// ----------------------------------------------------------------------
59inline uint64 Hash64FloatWithSeed(float num, uint64 seed) {
60 // +0 and -0 are the only floating point numbers which compare equal but
61 // have distinct bitwise representations in IEEE 754. To work around this,
62 // we force 0 to be +0.
63 if (num == 0) {
64 num = 0;
65 }
66 COMPILE_ASSERT(sizeof(float) == sizeof(uint32), float_has_wrong_size);
67
68 const uint64 kMul = 0xc6a4a7935bd1e995ULL;
69
70 uint64 a = (bit_cast<uint32>(num) + seed) * kMul;
71 a ^= (a >> 47);
72 a *= kMul;
73 a ^= (a >> 47);
74 a *= kMul;
75 return a;
76}
