30template <size_t initial_size_degree = 10>
31struct JoinHashTableGrower {
33 doris::vectorized::UInt8 size_degree = initial_size_degree;
34 doris::vectorized::Int64 double_grow_degree = 31; // 2GB
35
36 size_t bucket_size() const { return 1ULL << (size_degree - 1); }
37
39 size_t buf_size() const { return 1ULL << size_degree; }
40
41 size_t max_fill() const { return buf_size(); }
42
43 size_t mask() const { return bucket_size() - 1; }
44
46 size_t place(size_t x) const { return x & mask(); }
47
49 bool overflow(size_t elems) const { return elems >= max_fill(); }
50
52 void increase_size() { size_degree += size_degree >= 23 ? 1 : 2; }
53
55 void set(size_t num_elems) {
56#ifndef STRICT_MEMORY_USE
57 size_t fill_capacity = static_cast<size_t>(log2(num_elems - 1)) + 2;
58#else
59 size_t fill_capacity = static_cast<size_t>(log2(num_elems - 1)) + 1;
60 fill_capacity =
61 fill_capacity < double_grow_degree
62 ? fill_capacity + 1
63 : (num_elems < (1ULL << fill_capacity) - (1ULL << (fill_capacity - 2))
64 ? fill_capacity
65 : fill_capacity + 1);
66#endif
67 size_degree = num_elems <= 1 ? initial_size_degree
68 : (initial_size_degree > fill_capacity ? initial_size_degree
69 : fill_capacity);
70 }
71
72 void set_buf_size(size_t buf_size_) {
73 size_degree = static_cast<size_t>(log2(buf_size_ - 1) + 1);
74 }
75};

181 // "first" is the buckets of the hash map, and it holds the index of the first key value saved in each bucket,
182 // while other keys can be found by following the indices saved in
183 // "next". "next[0]" represents the end of the list of keys in a bucket.
184 // https://dare.uva.nl/search?identifier=5ccbb60a-38b8-4eeb-858a-e7735dd37487

199 size_t ALWAYS_INLINE find_cell(const Key& x, size_t hash_value, size_t place_value) const {
200 while (place_value && !buf[place_value - 1].is_zero(*this) &&
201 !buf[place_value - 1].key_equals(x, hash_value, *this)) {
202 place_value = next[place_value];
203#ifdef DBMS_HASH_MAP_COUNT_COLLISIONS
204 ++collisions;
205#endif
206 }
207
208 return place_value;
209 }
210
211 std::pair<bool, size_t> ALWAYS_INLINE find_cell_opt(const Key& x, size_t hash_value,
212 size_t place_value) const {
213 bool is_zero = false;
214 do {
215 if (!place_value) return {true, place_value};
216 is_zero = buf[place_value - 1].is_zero(*this);
217 if (is_zero || buf[place_value - 1].key_equals(x, hash_value, *this)) break;
218 place_value = next[place_value];
219 } while (true);
220
221 return {is_zero, place_value};
222 }
223
225 size_t ALWAYS_INLINE find_empty_cell(size_t place_value) const {
226 while (place_value && !buf[place_value - 1].is_zero(*this)) {
227 place_value = next[place_value];
228#ifdef DBMS_HASH_MAP_COUNT_COLLISIONS
229 ++collisions;
230#endif
231 }
232
233 return place_value;
234 }

236 void alloc(const Grower& new_grower) {
237 buf = reinterpret_cast<Cell*>(Allocator::alloc(new_grower.buf_size() * sizeof(Cell)));
238 first = reinterpret_cast<size_t*>(
239 Allocator::alloc(new_grower.bucket_size() * sizeof(size_t)));
240 memset(first, 0, new_grower.bucket_size() * sizeof(size_t));
241 next = reinterpret_cast<size_t*>(
242 Allocator::alloc((new_grower.buf_size() + 1) * sizeof(size_t)));
243 memset(next, 0, (new_grower.buf_size() + 1) * sizeof(size_t));
244 grower = new_grower;
245 }
246
247 void free() {
248 if (buf) {
249 Allocator::free(buf, get_buffer_size_in_bytes());
250 buf = nullptr;
251 }
252 if (first) {
253 Allocator::free(first, grower.bucket_size() * sizeof(size_t));
254 first = nullptr;
255 }
256 if (next) {
257 Allocator::free(next, (grower.buf_size() + 1) * sizeof(size_t));
258 next = nullptr;
259 }
260 }

263 void resize(size_t for_num_elems = 0, size_t for_buf_size = 0) {
264 SCOPED_RAW_TIMER(&_resize_timer_ns);
265#ifdef DBMS_HASH_MAP_DEBUG_RESIZES
266 Stopwatch watch;
267#endif
268
269 size_t old_size = grower.buf_size();
270
276 Grower new_grower = grower;
277 if (for_num_elems) {
278 new_grower.set(for_num_elems);
279 if (new_grower.buf_size() <= old_size) return;
280 } else if (for_buf_size) {
281 new_grower.set_buf_size(for_buf_size);
282 if (new_grower.buf_size() <= old_size) return;
283 } else
284 new_grower.increase_size();
285
287 buf = reinterpret_cast<Cell*>(Allocator::realloc(buf, get_buffer_size_in_bytes(),
288 new_grower.buf_size() * sizeof(Cell)));
289 first = reinterpret_cast<size_t*>(Allocator::realloc(
290 first, get_bucket_size_in_bytes(), new_grower.bucket_size() * sizeof(size_t)));
291 memset(first, 0, new_grower.bucket_size() * sizeof(size_t));
292 next = reinterpret_cast<size_t*>(Allocator::realloc(
293 next, get_buffer_size_in_bytes(), (new_grower.buf_size() + 1) * sizeof(size_t)));
294 memset(next, 0, (new_grower.buf_size() + 1) * sizeof(size_t));
295 grower = new_grower;
296
301 size_t i = 0;
302 for (; i < m_no_zero_size; ++i)
303 if (!buf[i].is_zero(*this)) reinsert(i + 1, buf[i], buf[i].get_hash(*this));
304
305#ifdef DBMS_HASH_MAP_DEBUG_RESIZES
306 watch.stop();
307 std::cerr << std::fixed << std::setprecision(3) << "Resize from " << old_size << " to "
308 << grower.buf_size() << " took " << watch.elapsedSeconds() << " sec."
309 << std::endl;
310#endif
311 }

316 void reinsert(size_t place_value, Cell& x, size_t hash_value) {
317 size_t bucket_value = grower.place(hash_value);
318 next[place_value] = first[bucket_value];
319 first[bucket_value] = place_value;

470 bool ALWAYS_INLINE emplace_if_zero(Key x, LookupResult& it, bool& inserted, size_t hash_value) {
472 if (!Cell::need_zero_value_storage) return false;
473
474 if (Cell::is_zero(x, *this)) {
475 it = this->zero_value();
476
477 if (!this->get_has_zero()) {
478 ++m_size;
479 this->set_get_has_zero();
480 this->zero_value()->set_hash(hash_value);
481 inserted = true;
482 } else
483 inserted = false;
484
485 return true;
486 }
487
488 return false;
489 }

492 template <typename KeyHolder>
493 void ALWAYS_INLINE emplace_non_zero(KeyHolder&& key_holder, LookupResult& it, bool& inserted,
494 size_t hash_value) {
495 it = &buf[m_no_zero_size];
496
497 if (!buf[m_no_zero_size].is_zero(*this)) {
498 key_holder_discard_key(key_holder);
499 inserted = false;
500 return;
501 }
502
503 key_holder_persist_key(key_holder);
504 const auto& key = key_holder_get_key(key_holder);
505
506 new (&buf[m_no_zero_size]) Cell(key, *this);
507 buf[m_no_zero_size].set_hash(hash_value);
508 size_t bucket_value = grower.place(hash_value);
509 inserted = true;
510 ++m_size;
511 ++m_no_zero_size;
512 next[m_no_zero_size] = first[bucket_value];
513 first[bucket_value] = m_no_zero_size;
514
515 if (UNLIKELY(grower.overflow(m_size))) {
516 try {
517 resize();
518 } catch (...) {
523 first[bucket_value] = next[m_no_zero_size];
524 next[m_no_zero_size] = 0;
525 --m_size;
526 --m_no_zero_size;
527 buf[m_no_zero_size].set_zero();
528 throw;
529 }
530 }

542 std::pair<LookupResult, bool> ALWAYS_INLINE insert(const value_type& x) {
543 std::pair<LookupResult, bool> res;
544 size_t hash_value = hash(Cell::get_key(x));
