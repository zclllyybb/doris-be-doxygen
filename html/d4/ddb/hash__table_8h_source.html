243template <size_t initial_size_degree = 10>
244struct HashTableGrower {
246 doris::vectorized::UInt8 size_degree = initial_size_degree;
247 doris::vectorized::Int64 double_grow_degree = 31; // 2GB
248
250 size_t buf_size() const { return 1ULL << size_degree; }
251
252#ifndef STRICT_MEMORY_USE
253 size_t max_fill() const { return 1ULL << (size_degree - 1); }
254#else
255 // When capacity is greater than 2G, grow when 75% of the capacity is satisfied.
256 size_t max_fill() const {
257 return size_degree < double_grow_degree
258 ? 1ULL << (size_degree - 1)
259 : (1ULL << size_degree) - (1ULL << (size_degree - 2));
260 }
261#endif
262
263 size_t mask() const { return buf_size() - 1; }
264
266 size_t place(size_t x) const { return x & mask(); }
267
269 size_t next(size_t pos) const {
270 ++pos;
271 return pos & mask();
272 }
273
275 bool overflow(size_t elems) const { return elems > max_fill(); }
276
278 void increase_size() { size_degree += size_degree >= 23 ? 1 : 2; }
279
281 void set(size_t num_elems) {
282#ifndef STRICT_MEMORY_USE
283 size_t fill_capacity = static_cast<size_t>(log2(num_elems - 1)) + 2;
284#else
285 size_t fill_capacity = static_cast<size_t>(log2(num_elems - 1)) + 1;
286 fill_capacity =
287 fill_capacity < double_grow_degree
288 ? fill_capacity + 1
289 : (num_elems < (1ULL << fill_capacity) - (1ULL << (fill_capacity - 2))
290 ? fill_capacity
291 : fill_capacity + 1);
292#endif
293 size_degree = num_elems <= 1 ? initial_size_degree
294 : (initial_size_degree > fill_capacity ? initial_size_degree
295 : fill_capacity);
296 }
297
298 void set_buf_size(size_t buf_size_) {
299 size_degree = static_cast<size_t>(log2(buf_size_ - 1) + 1);
300 }
301};

307template <size_t initial_size_degree = 8>
308class alignas(64) HashTableGrowerWithPrecalculation {
310
311 doris::vectorized::UInt8 size_degree_ = initial_size_degree;
312 size_t precalculated_mask = (1ULL << initial_size_degree) - 1;
313 size_t precalculated_max_fill = 1ULL << (initial_size_degree - 1);
314
315public:
316 doris::vectorized::UInt8 size_degree() const { return size_degree_; }
317
318 void increase_size_degree(doris::vectorized::UInt8 delta) {
319 size_degree_ += delta;
320 DCHECK(size_degree_ <= 64);
321 precalculated_mask = (1ULL << size_degree_) - 1;
322 precalculated_max_fill = 1ULL << (size_degree_ - 1);
323 }
324
325 static constexpr auto initial_count = 1ULL << initial_size_degree;
326
328 static constexpr auto performs_linear_probing_with_single_step = true;
329
331 size_t buf_size() const { return 1ULL << size_degree_; }
332
334 size_t place(size_t x) const { return x & precalculated_mask; }
335
337 size_t next(size_t pos) const { return (pos + 1) & precalculated_mask; }
338
340 bool overflow(size_t elems) const { return elems > precalculated_max_fill; }
341
343 void increase_size() { increase_size_degree(size_degree_ >= 23 ? 1 : 2); }
344
346 void set(size_t num_elems) {
347 size_degree_ =
348 num_elems <= 1
349 ? initial_size_degree
350 : ((initial_size_degree > static_cast<size_t>(log2(num_elems - 1)) + 2)
351 ? initial_size_degree
352 : (static_cast<size_t>(log2(num_elems - 1)) + 2));
353 increase_size_degree(0);
354 }
355
356 void set_buf_size(size_t buf_size_) {
357 size_degree_ = static_cast<size_t>(log2(buf_size_ - 1) + 1);
358 increase_size_degree(0);
359 }

474 size_t ALWAYS_INLINE find_cell(const Key& x, size_t hash_value, size_t place_value) const {
475 while (!buf[place_value].is_zero(*this) &&
476 !buf[place_value].key_equals(x, hash_value, *this)) {
477 place_value = grower.next(place_value);
478#ifdef DBMS_HASH_MAP_COUNT_COLLISIONS
479 ++collisions;
480#endif
481 }
482
483 return place_value;
484 }

486 std::pair<bool, size_t> ALWAYS_INLINE find_cell_opt(const Key& x, size_t hash_value,
487 size_t place_value) const {
488 bool is_zero = false;
489 do {
490 is_zero = buf[place_value].is_zero(*this);
491 if (is_zero || buf[place_value].key_equals(x, hash_value, *this)) break;
492 place_value = grower.next(place_value);
493 } while (true);
494
495 return {is_zero, place_value};

752 template <typename KeyHolder>
753 void ALWAYS_INLINE emplace_non_zero_impl(size_t place_value, KeyHolder&& key_holder,
754 LookupResult& it, bool& inserted, size_t hash_value) {
755 it = &buf[place_value];
756
757 if (!buf[place_value].is_zero(*this)) {
758 key_holder_discard_key(key_holder);
759 inserted = false;
760 return;
761 }
762
763 key_holder_persist_key(key_holder);
764 const auto& key = key_holder_get_key(key_holder);
765
766 new (&buf[place_value]) Cell(key, *this);
767 buf[place_value].set_hash(hash_value);
768 inserted = true;
769 ++m_size;
770
771 if (UNLIKELY(grower.overflow(m_size))) {
772 try {
773 resize();
774 } catch (...) {
779 --m_size;
780 buf[place_value].set_zero();
781 throw;
782 }
783
784 if (LIKELY(!_need_partition)) {
785 // The hash table was rehashed, so we have to re-find the key.
786 size_t new_place = find_cell(key, hash_value, grower.place(hash_value));
787 assert(!buf[new_place].is_zero(*this));
788 it = &buf[new_place];
789 }
790 }
791 }
