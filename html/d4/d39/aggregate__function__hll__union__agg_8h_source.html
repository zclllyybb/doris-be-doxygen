32template <bool is_nullable>
33struct AggregateFunctionHLLData {
34 HyperLogLog dst_hll {};
35
36 void merge(const AggregateFunctionHLLData& rhs) { dst_hll.merge(rhs.dst_hll); }
37
38 void write(BufferWritable& buf) const {
39 std::string result(dst_hll.max_serialized_size(), '0');
40 int size = dst_hll.serialize((uint8_t*)result.c_str());
41 result.resize(size);
42 write_binary(result, buf);
43 }
44
45 void read(BufferReadable& buf) {
46 StringRef ref;
47 read_binary(ref, buf);
48 dst_hll.deserialize(Slice(ref.data, ref.size));
49 }
50
51 Int64 get_cardinality() const { return dst_hll.estimate_cardinality(); }
52
53 HyperLogLog get() const { return dst_hll; }
54
55 void reset() { dst_hll.clear(); }
56
57 void add(const IColumn* column, size_t row_num) {
58 if constexpr (is_nullable) {
59 auto* nullable_column = check_and_get_column<const ColumnNullable>(*column);
60 if (nullable_column->is_null_at(row_num)) {
61 return;
62 }
63 const auto& sources =
64 static_cast<const ColumnHLL&>((nullable_column->get_nested_column()));
65 dst_hll.merge(sources.get_element(row_num));
66
67 } else {
68 const auto& sources = static_cast<const ColumnHLL&>(*column);
69 dst_hll.merge(sources.get_element(row_num));
70 }
71 }
72};
