50template <typename Value, bool consecutive_keys_optimization_>
51struct LastElementCache {
52 static constexpr bool consecutive_keys_optimization = consecutive_keys_optimization_;
53 Value value;

56
57 bool check(const Value& value_) { return !empty && value == value_; }

218 template <typename Data, typename KeyHolder>
219 ALWAYS_INLINE EmplaceResult emplaceImpl(KeyHolder& key_holder, Data& data) {
220 if constexpr (Cache::consecutive_keys_optimization) {
221 if (cache.found && cache.check(key_holder_get_key(key_holder))) {
222 if constexpr (has_mapped)
223 return EmplaceResult(cache.value.second, cache.value.second, false);
224 else
225 return EmplaceResult(false);
226 }
227 }
228
229 typename Data::LookupResult it;
230 bool inserted = false;
231 data.emplace(key_holder, it, inserted);
232
233 [[maybe_unused]] Mapped* cached = nullptr;
234 if constexpr (has_mapped) cached = lookup_result_get_mapped(it);
235
236 if (inserted) {
237 if constexpr (has_mapped) {
238 new (lookup_result_get_mapped(it)) Mapped();
239 }
240 }
241
242 if constexpr (consecutive_keys_optimization) {
243 cache.found = true;
244 cache.empty = false;
245
246 if constexpr (has_mapped) {
247 cache.value.first = *lookup_result_get_key(it);
248 cache.value.second = *lookup_result_get_mapped(it);
249 cached = &cache.value.second;
250 } else {
251 cache.value = *lookup_result_get_key(it);
252 }
253 }
254
255 if constexpr (has_mapped)
256 return EmplaceResult(*lookup_result_get_mapped(it), *cached, inserted);
257 else
258 return EmplaceResult(inserted);
259 }

321 template <typename Data, typename Key>
322 ALWAYS_INLINE FindResult find_key_impl(Key key, Data& data) {
323 if constexpr (Cache::consecutive_keys_optimization) {
324 if (cache.check(key)) {
325 if constexpr (has_mapped)
326 return FindResult(&cache.value.second, cache.found);
327 else
328 return FindResult(cache.found);
329 }
330 }
331
332 auto it = data.find(key);
333
334 if constexpr (consecutive_keys_optimization) {
335 cache.found = it != nullptr;
336 cache.empty = false;
337
338 if constexpr (has_mapped) {
339 cache.value.first = key;
340 if (it) {
341 cache.value.second = *lookup_result_get_mapped(it);
342 }
343 } else {
344 cache.value = key;
345 }
346 }
347
348 if constexpr (has_mapped)
349 return FindResult(it ? lookup_result_get_mapped(it) : nullptr, it != nullptr);
350 else
351 return FindResult(it != nullptr);
352 }

401template <typename Key>
402class BaseStateKeysFixed<Key, true> {
403protected:
404 BaseStateKeysFixed(const ColumnRawPtrs& key_columns) {
405 null_maps.reserve(key_columns.size());
406 actual_columns.reserve(key_columns.size());
407
408 for (const auto& col : key_columns) {
409 if (auto* nullable_col = check_and_get_column<ColumnNullable>(col)) {
410 actual_columns.push_back(&nullable_col->get_nested_column());
411 null_maps.push_back(&nullable_col->get_null_map_column());
412 } else {
413 actual_columns.push_back(col);
414 null_maps.push_back(nullptr);
415 }
416 }
417 }
418
422 const ColumnRawPtrs& get_actual_columns() const { return actual_columns; }
423
426 KeysNullMap<Key> create_bitmap(size_t row) const {
427 KeysNullMap<Key> bitmap {};
428
429 for (size_t k = 0; k < null_maps.size(); ++k) {
430 if (null_maps[k] != nullptr) {
431 const auto& null_map = assert_cast<const ColumnUInt8&>(*null_maps[k]).get_data();
432 if (null_map[row] == 1) {
433 size_t bucket = k / 8;
434 size_t offset = k % 8;
435 bitmap[bucket] |= UInt8(1) << offset;
436 }
437 }
438 }
439
440 return bitmap;
441 }
442
443private:
444 ColumnRawPtrs actual_columns;
445 ColumnRawPtrs null_maps;
446};
