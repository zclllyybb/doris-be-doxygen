99// Load an unaligned little endian word from memory.
100//
101// These routines are named Word32At(), Word64At() and Google1At().
102// Long ago, the 32-bit version of this operation was implemented using
103// signed characters. The hash function that used this variant creates
104// persistent hash values. The hash routine needs to remain backwards
105// compatible, so we renamed the word loading function 'Google1At' to
106// make it clear this implements special functionality.
107//
108// If a machine has alignment constraints or is big endian, we must
109// load the word a byte at a time. Otherwise we can load the whole word
110// from memory.
111//
112// [Plausibly, Word32At() and Word64At() should really be called
113// UNALIGNED_LITTLE_ENDIAN_LOAD32() and UNALIGNED_LITTLE_ENDIAN_LOAD64()
114// but that seems overly verbose.]
115

117static inline uint64 Word64At(const char* ptr) {
118 return UNALIGNED_LOAD64(ptr);
119}

122 return UNALIGNED_LOAD32(ptr);

125// This produces the same results as the byte-by-byte version below.
126// Here, we mask off the sign bits and subtract off two copies. To
127// see why this is the same as adding together the sign extensions,
128// start by considering the low-order byte. If we loaded an unsigned
129// word and wanted to sign extend it, we isolate the sign bit and subtract
130// that from zero which gives us a sequence of bits matching the sign bit
131// at and above the sign bit. If we remove (subtract) the sign bit and
132// add in the low order byte, we now have a sign-extended byte as desired.
133// We can then operate on all four bytes in parallel because addition
134// is associative and commutative.
135//
136// For example, consider sign extending the bytes 0x01 and 0x81. For 0x01,
137// the sign bit is zero, and 0x01 - 0 -0 = 1. For 0x81, the sign bit is 1
138// and we are computing 0x81 - 0x80 + (-0x80) == 0x01 + 0xFFFFFF80.
139//
140// Similarily, if we start with 0x8200 and want to sign extend that,
141// we end up calculating 0x8200 - 0x8000 + (-0x8000) == 0xFFFF8000 + 0x0200
142//
143// Suppose we have two bytes at the same time. Doesn't the adding of all
144// those F's generate something wierd? Ignore the F's and reassociate
145// the addition. For 0x8281, processing the bytes one at a time (like
146// we used to do) calculates
147// [0x8200 - 0x8000 + (-0x8000)] + [0x0081 - 0x80 + (-0x80)]
148// == 0x8281 - 0x8080 - 0x8000 - 0x80
149// == 0x8281 - 0x8080 - 0x8080
150
151static inline uint32 Google1At(const char* ptr) {
152 uint32 t = UNALIGNED_LOAD32(ptr);
153 uint32 masked = t & 0x80808080;
154 return t - masked - masked;
155}
