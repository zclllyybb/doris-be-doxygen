75 template <typename JoinOpType, bool set_build_side_flag, bool set_probe_side_flag>
76 Status _generate_join_block_data(RuntimeState* state, JoinOpType& join_op_variants) {
77 constexpr bool ignore_null = JoinOpType::value == TJoinOp::LEFT_ANTI_JOIN ||
78 JoinOpType::value == TJoinOp::NULL_AWARE_LEFT_ANTI_JOIN ||
79 JoinOpType::value == TJoinOp::RIGHT_ANTI_JOIN;
80
81 MutableBlock mutable_join_block(&_join_block);
82
83 while (_join_block.rows() < state->batch_size() && !_matched_rows_done) {
84 // If this left block is exhausted or empty, we need to pull data from left child.
85 if (_left_block_pos == _left_block.rows()) {
86 if (_left_side_eos) {
87 _matched_rows_done = true;
88 } else {
89 _left_block_pos = 0;
90 _need_more_input_data = true;
91 return Status::OK();
92 }
93 }
94
95 // We should try to join rows if there still are some rows from probe side.
96 if (!_matched_rows_done && _current_build_pos < _build_blocks.size()) {
97 do {
98 const auto& now_process_build_block = _build_blocks[_current_build_pos++];
99 if constexpr (set_build_side_flag) {
100 _offset_stack.push(mutable_join_block.rows());
101 }
102 _process_left_child_block(mutable_join_block, now_process_build_block);
103 } while (_join_block.rows() < state->batch_size() &&
104 _current_build_pos < _build_blocks.size());
105 }
106
107 if constexpr (set_probe_side_flag) {
108 auto status =
109 _do_filtering_and_update_visited_flags<set_build_side_flag,
110 set_probe_side_flag, ignore_null>(
111 &_join_block, !_is_left_semi_anti);
112 _update_additional_flags(&_join_block);
113 if (!status.ok()) {
114 return status;
115 }
116 mutable_join_block = MutableBlock(&_join_block);
117 // If this join operation is left outer join or full outer join, when
118 // `_current_build_pos == _build_blocks.size()`, means all rows from build
119 // side have been joined with the current probe row, we should output current
120 // probe row with null from build side.
121 if (_current_build_pos == _build_blocks.size()) {
122 if (!_matched_rows_done) {
123 _finalize_current_phase<false,
124 JoinOpType::value == TJoinOp::LEFT_SEMI_JOIN>(
125 mutable_join_block, state->batch_size());
126 _reset_with_next_probe_row();
127 }
128 break;
129 }
130 }
131
132 if (!_matched_rows_done && _current_build_pos == _build_blocks.size()) {
133 if (_is_mark_join && _build_blocks.empty()) {
134 DCHECK_EQ(JoinOpType::value, TJoinOp::CROSS_JOIN);
135 _append_left_data_with_null(mutable_join_block);
136 _reset_with_next_probe_row();
137 break;
138 }
139 _reset_with_next_probe_row();
140 }
141 }
142
143 if constexpr (!set_probe_side_flag) {
144 Status status =
145 _do_filtering_and_update_visited_flags<set_build_side_flag, set_probe_side_flag,
146 ignore_null>(&_join_block,
147 !_is_right_semi_anti);
148 _update_additional_flags(&_join_block);
149 mutable_join_block = MutableBlock(&_join_block);
150 if (!status.ok()) {
151 return status;
152 }
153 }
154
155 if constexpr (set_build_side_flag) {
156 if (_matched_rows_done && _output_null_idx_build_side < _build_blocks.size()) {
157 _finalize_current_phase<true, JoinOpType::value == TJoinOp::RIGHT_SEMI_JOIN>(
158 mutable_join_block, state->batch_size());
159 }
160 }
161 return Status::OK();
162 }
