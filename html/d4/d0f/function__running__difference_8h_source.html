42class FunctionRunningDifference : public IFunction {
43public:
44 static constexpr auto name = "running_difference";
45
46 static FunctionPtr create() { return std::make_shared<FunctionRunningDifference>(); }
47
48 String get_name() const override { return name; }
49
50 size_t get_number_of_arguments() const override { return 1; }
51
52 bool use_default_implementation_for_nulls() const override { return false; }
53
54 bool use_default_implementation_for_constants() const override { return true; }

56 template <typename SrcFieldType>
57 using DstFieldType =
58 typename NumberTraits::ResultOfSubtraction<SrcFieldType, SrcFieldType>::Type;
59 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
60 bool is_nullable = arguments[0]->is_nullable();
61 auto nested_type = remove_nullable(arguments[0]);
62 WhichDataType which(nested_type);
63 //return type is promoted to prevent result overflow
64 //like: input is int32 ---> return type will be int64
65 DataTypePtr return_type = nullptr;
66 if (which.is_uint8() || which.is_int8()) {
67 return_type = std::make_shared<DataTypeInt16>();
68 } else if (which.is_uint16() || which.is_int16()) {
69 return_type = std::make_shared<DataTypeInt32>();
70 } else if (which.is_uint32() || which.is_uint64() || which.is_int32() || which.is_int64()) {
71 return_type = std::make_shared<DataTypeInt64>();
72 } else if (which.is_int128()) {
73 return_type = std::make_shared<DataTypeInt128>();
74 } else if (which.is_float32() || which.is_float64()) {
75 return_type = std::make_shared<DataTypeFloat64>();
76 } else if (which.is_decimal()) {
77 return_type = nested_type;
78 } else if (which.is_date_time() || which.is_date_time_v2()) {
79 return_type = std::make_shared<DataTypeTime>();
80 } else if (which.is_date() || which.is_date_v2()) {
81 return_type = std::make_shared<DataTypeInt32>();
82 }
83
84 return_type = is_nullable ? make_nullable(return_type) : return_type;
85 const ColumnsWithTypeAndName subtract_cols {{nullptr, arguments[0], "first_arg"},
86 {nullptr, arguments[0], "second_arg"}};
87 if (which.is_date_time() || which.is_date_time_v2()) {
88 func_subtract = SimpleFunctionFactory::instance().get_function(
89 "timediff", subtract_cols, return_type);
90 } else if (which.is_date() || which.is_date_v2()) {
91 func_subtract = SimpleFunctionFactory::instance().get_function(
92 "datediff", subtract_cols, return_type);
93 } else {
94 func_subtract = SimpleFunctionFactory::instance().get_function(
95 "subtract", subtract_cols, return_type);
96 }
97 func_return_type = return_type;
98 return return_type;
99 }
100
101 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
102 size_t result, size_t input_rows_count) override {
103 if (is_column_const(*block.get_by_position(arguments[0]).column)) {
104 auto res = func_return_type->create_column_const_with_default_value(input_rows_count);
105 block.replace_by_position(result, std::move(res));
106 return Status::OK();
107 }
108
109 auto arg_first = block.get_by_position(arguments[0]).column;
110 auto arg_type = block.get_by_position(arguments[0]).type;
111 auto arg_second = arg_type->create_column();
112
113 if (is_first_block) {
114 arg_second->insert_from(*arg_first, 0);
115 is_first_block = false;
116 } else {
117 arg_second->insert_data(last_value.c_str(), last_value.length());
118 }
119 arg_second->insert_range_from(*arg_first, 0, input_rows_count - 1);
120 last_value = arg_first->get_data_at(input_rows_count - 1).to_string();
121
122 Block temporary_block {
123 ColumnsWithTypeAndName {block.get_by_position(arguments[0]),
124 {std::move(arg_second), arg_type, "second_arg"},
125 block.get_by_position(result)}};
126
127 func_subtract->execute(context, temporary_block, {0, 1}, 2, input_rows_count);
128 block.get_by_position(result).column = temporary_block.get_by_position(2).column;
129 return Status::OK();
130 }
131
132private:
133 mutable FunctionBasePtr func_subtract;
134 mutable DataTypePtr func_return_type;
135 bool is_first_block = true;
136 std::string last_value;
137};
