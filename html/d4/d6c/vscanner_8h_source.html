38class VScanner {

40 VScanner(RuntimeState* state, VScanNode* parent, int64_t limit, RuntimeProfile* profile);

63public:

72 // Call start_wait_worker_timer() when submit the scanner to the thread pool.

79 void start_scan_cpu_timer() {
80 _cpu_watch.reset();

87
88 RuntimeState* runtime_state() { return _state; }

90 bool is_open() { return _is_open; }

94
95 virtual doris::TabletStorageType get_storage_type() {
96 return doris::TabletStorageType::STORAGE_TYPE_LOCAL;
97 }
98
99 bool need_to_close() { return _need_to_close; }
100
101 void mark_to_need_to_close() {
102 _update_counters_before_close();
103 _need_to_close = true;
104 }
105
106 void set_status_on_failure(const Status& st) { _status = st; }
107
108 VExprContext** vconjunct_ctx_ptr() { return &_vconjunct_ctx; }
109
110 // return false if _is_counted_down is already true,
111 // otherwise, set _is_counted_down to true and return true.
112 bool set_counted_down() {
113 if (_is_counted_down) {
114 return false;
115 }
116 _is_counted_down = true;
117 return true;
118 }
119
120protected:
121 void _discard_conjuncts() {
122 if (_vconjunct_ctx) {
123 _vconjunct_ctx->mark_as_stale();
124 _stale_vexpr_ctxs.push_back(_vconjunct_ctx);
125 _vconjunct_ctx = nullptr;
126 }
127 }
128
129protected:
130 RuntimeState* _state;
131 VScanNode* _parent;
132 // Set if scan node has sort limit info
133 int64_t _limit = -1;
134
135 RuntimeProfile* _profile;
136
137 const TupleDescriptor* _input_tuple_desc = nullptr;
138 const TupleDescriptor* _output_tuple_desc = nullptr;
139 const TupleDescriptor* _real_tuple_desc = nullptr;
140
141 // If _input_tuple_desc is set, the scanner will read data into
142 // this _input_block first, then convert to the output block.
143 Block _input_block;
144 // If _input_tuple_desc is set, this will point to _input_block,
145 // otherwise, it will point to the output block.
146 Block* _input_block_ptr;
147
148 bool _is_open = false;
149 bool _is_closed = false;
150 bool _need_to_close = false;
151 Status _status;
152
153 // If _applied_rf_num == _total_rf_num
154 // means all runtime filters are arrived and applied.
155 int _applied_rf_num = 0;
156 int _total_rf_num = 0;
157 // Cloned from _vconjunct_ctx of scan node.
158 // It includes predicate in SQL and runtime filters.
159 VExprContext* _vconjunct_ctx = nullptr;
160 // Late arriving runtime filters will update _vconjunct_ctx.
161 // The old _vconjunct_ctx will be temporarily placed in _stale_vexpr_ctxs
162 // and will be destroyed at the end.
163 std::vector<VExprContext*> _stale_vexpr_ctxs;
164
165 // num of rows read from scanner
166 int64_t _num_rows_read = 0;
167
168 // num of rows return from scanner, after filter block
169 int64_t _num_rows_return = 0;
170
171 // Set true after counter is updated finally
172 bool _has_updated_counter = false;
173
174 // watch to count the time wait for scanner thread
175 MonotonicStopWatch _watch;
176 // Do not use ScopedTimer. There is no guarantee that, the counter
177 ThreadCpuStopWatch _cpu_watch;
178 int64_t _scanner_wait_worker_timer = 0;
179 int64_t _scan_cpu_timer = 0;
180
181 bool _is_load = false;
182 // set to true after decrease the "_num_unfinished_scanners" in scanner context
183 bool _is_counted_down = false;
184
185 ScannerCounter _counter;
186 int64_t _per_scanner_timer = 0;
187};
