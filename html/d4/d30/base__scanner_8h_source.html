38// The counter will be passed to each scanner.
39// Note that this struct is not thread safe.
40// So if we support concurrent scan in the future, we need to modify this struct.
41struct ScannerCounter {
42 ScannerCounter() : num_rows_filtered(0), num_rows_unselected(0) {}
43
44 int64_t num_rows_filtered; // unqualified rows (unmatched the dest schema, or no partition)
45 int64_t num_rows_unselected; // rows filtered by predicates
46};
