51 template <typename T>
52 static typename std::enable_if<std::is_arithmetic<T>::value, std::string>::type print(
53 T value, TUnit::type unit, bool verbose = false) {
54 std::stringstream ss;
55 ss.flags(std::ios::fixed);
56 switch (unit) {
57 case TUnit::NONE: {
58 ss << value;
59 return ss.str();
60 }
61
62 case TUnit::UNIT: {
63 std::string unit;
64 double output = get_unit(value, &unit);
65 if (unit.empty()) {
66 ss << value;
67 } else {
68 ss << std::setprecision(PRECISION) << output << unit;
69 }
70 if (verbose) ss << " (" << value << ")";
71 break;
72 }
73
74 case TUnit::UNIT_PER_SECOND: {
75 std::string unit;
76 double output = get_unit(value, &unit);
77 if (output == 0) {
78 ss << "0";
79 } else {
80 ss << std::setprecision(PRECISION) << output << " " << unit << "/sec";
81 }
82 break;
83 }
84
85 case TUnit::CPU_TICKS: {
86 if (value < CpuInfo::cycles_per_ms()) {
87 ss << std::setprecision(PRECISION) << (value / 1000.) << "K clock cycles";
88 } else {
89 value /= CpuInfo::cycles_per_ms();
90 print_timems(value, &ss);
91 }
92 break;
93 }
94
95 case TUnit::TIME_NS: {
96 ss << std::setprecision(TIME_NS_PRECISION);
97 if (value >= BILLION) {
99 value /= MILLION;
100 print_timems(value, &ss);
101 } else if (value >= MILLION) {
103 ss << DOUBLE_TRUNCATE(static_cast<double>(value) / MILLION, TIME_NS_PRECISION)
104 << "ms";
105 } else if (value > 1000) {
107 ss << DOUBLE_TRUNCATE(static_cast<double>(value) / 1000, TIME_NS_PRECISION) << "us";
108 } else {
109 ss << DOUBLE_TRUNCATE(value, TIME_NS_PRECISION) << "ns";
110 }
111 break;
112 }
113
114 case TUnit::TIME_MS: {
115 print_timems(value, &ss);
116 break;
117 }
118
119 case TUnit::TIME_S: {
120 print_timems(value * 1000, &ss);
121 break;
122 }
123
124 case TUnit::BYTES: {
125 std::string unit;
126 double output = get_byte_unit(value, &unit);
127 if (output == 0) {
128 ss << "0";
129 } else {
130 ss << std::setprecision(PRECISION) << output << " " << unit;
131 if (verbose) ss << " (" << value << ")";
132 }
133 break;
134 }
135
136 case TUnit::BYTES_PER_SECOND: {
137 std::string unit;
138 double output = get_byte_unit(value, &unit);
139 ss << std::setprecision(PRECISION) << output << " " << unit << "/sec";
140 break;
141 }
142
144 case TUnit::DOUBLE_VALUE: {
145 double output = binary_cast<T, double>(value);
146 ss << std::setprecision(PRECISION) << output << " ";
147 break;
148 }
149
150 default:
151 DCHECK(false) << "Unsupported TUnit: " << value;
152 break;
153 }
154 return ss.str();
155 }

170 template <typename I>
171 static void print_stringList(const I& iterable, TUnit::type unit, std::stringstream* out) {
172 std::vector<std::string> strings;
173 for (typename I::const_iterator it = iterable.begin(); it != iterable.end(); ++it) {
174 std::stringstream ss;
175 ss << PrettyPrinter::print(*it, unit);
176 strings.push_back(ss.str());
177 }
178
179 (*out) << "[" << boost::algorithm::join(strings, ", ") << "]";
180 }

202 template <typename T>
203 static double get_byte_unit(T value, std::string* unit) {
204 if (value == 0) {
205 *unit = "";
206 return value;
207 } else if (value >= GIGABYTE) {
208 *unit = "GB";
209 return value / (double)GIGABYTE;
210 } else if (value >= MEGABYTE) {
211 *unit = "MB";
212 return value / (double)MEGABYTE;
213 } else if (value >= KILOBYTE) {
214 *unit = "KB";
215 return value / (double)KILOBYTE;
216 } else {
217 *unit = "B";
218 return value;
219 }
220 }

222 template <typename T>
223 static double get_unit(T value, std::string* unit) {
224 if (value >= BILLION) {
225 *unit = "B";
226 return value / (1. * BILLION);
227 } else if (value >= MILLION) {
228 *unit = "M";
229 return value / (1. * MILLION);
230 } else if (value >= THOUSAND) {
231 *unit = "K";
232 return value / (1. * THOUSAND);
233 } else {
234 *unit = "";
235 return value;
236 }

253 template <typename T>
254 static void print_timems(T value, std::stringstream* ss) {
255 DCHECK_GE(value, static_cast<T>(0));
256 if (value == 0) {
257 *ss << "0";
258 } else {
259 bool hour = false;
260 bool minute = false;
261 bool second = false;
262 if (value >= HOUR) {
263 *ss << static_cast<int64_t>(value / HOUR) << "h";
264 value = mod(value, HOUR);
265 hour = true;
266 }
267 if (value >= MINUTE) {
268 *ss << static_cast<int64_t>(value / MINUTE) << "m";
269 value = mod(value, MINUTE);
270 minute = true;
271 }
272 if (!hour && value >= SECOND) {
273 *ss << static_cast<int64_t>(value / SECOND) << "s";
274 value = mod(value, SECOND);
275 second = true;
276 }
277 if (!hour && !minute) {
278 if (second) *ss << std::setw(3) << std::setfill('0');
279 *ss << static_cast<int64_t>(value) << "ms";
280 }
281 }
282 }
