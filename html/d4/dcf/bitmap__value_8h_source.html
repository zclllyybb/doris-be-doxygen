43// serialized bitmap := TypeCode(1), Payload
44// The format of payload depends on value of TypeCode which is defined below
45struct BitmapTypeCode {
46 enum type {
47 // An empty bitmap. Payload is 0 byte.
48 // added in 0.11
49 EMPTY = 0,
50 // A bitmap containing only one element that is in [0, UINT32_MAX]
51 // Payload := UInt32LittleEndian(4 byte)
52 // added in 0.11
53 SINGLE32 = 1,
54 // A bitmap whose maximum element is in [0, UINT32_MAX]
55 // Payload := the standard RoaringBitmap format described by
56 // https://github.com/RoaringBitmap/RoaringFormatSpec/
57 // added in 0.11
58 BITMAP32 = 2,
59 // A bitmap containing only one element that is in (UINT32_MAX, UINT64_MAX]
60 // Payload := UInt64LittleEndian(8 byte)
61 // added in 0.12
62 SINGLE64 = 3,
63 // A bitmap whose maximum element is in (UINT32_MAX, UINT64_MAX].
64 //
65 // To support 64-bits elements, all elements with the same high 32 bits are stored in a
66 // RoaringBitmap containing only the lower 32 bits. Thus we could use
67 // map<uint32_t, RoaringBitmap> to represent bitmap of 64-bits ints.
68 //
69 // Since there is no standard format for 64-bits RoaringBitmap, we define our own as below
70 // Payload := NumRoaring(vint64), { MapKey, MapValue }^NumRoaring
71 // - MapKey := the shared high 32 bits in UInt32LittleEndian(4 byte)
72 // - MapValue := the standard RoaringBitmap format
73 //
74 // added in 0.12
75 BITMAP64 = 4
76 };
77 Status static inline validate(int bitmap_type) {
78 if (UNLIKELY(bitmap_type < type::EMPTY || bitmap_type > type::BITMAP64)) {
79 std::string err_msg =
80 fmt::format("BitmapTypeCode invalid, should between: {} and {} actrual is {}",
81 BitmapTypeCode::EMPTY, BitmapTypeCode::BITMAP64, bitmap_type);
82 LOG(ERROR) << err_msg;
83 return Status::IOError(err_msg);
84 }
85 return Status::OK();
86 }
87};

411 uint64_t card = 0;

437 bool isFull() const {
438 // only bother to check if map is fully saturated
439 //
440 // we put std::numeric_limits<>::max/min in parenthesis
441 // to avoid a clash with the Windows.h header under Windows
442 return roarings.size() == ((uint64_t)(std::numeric_limits<uint32_t>::max)()) + 1
443 ? std::all_of(roarings.cbegin(), roarings.cend(),
444 [](const std::pair<const uint32_t, roaring::Roaring>&
445 roaring_map_entry) {
446 // roarings within map are saturated if cardinality
447 // is uint32_t max + 1
448 return roaring_map_entry.second.cardinality() ==
449 ((uint64_t)(std::numeric_limits<uint32_t>::max)()) +
450 1;
451 })
452 : false;
453 }

498 bool operator==(const Roaring64Map& r) const {
499 // we cannot use operator == on the map because either side may contain
500 // empty Roaring Bitmaps
501 auto lhs_iter = roarings.cbegin();
502 auto rhs_iter = r.roarings.cbegin();
503 do {
504 // if the left map has reached its end, ensure that the right map
505 // contains only empty Bitmaps
506 if (lhs_iter == roarings.cend()) {
507 while (rhs_iter != r.roarings.cend()) {
508 if (rhs_iter->second.isEmpty()) {
509 ++rhs_iter;
510 continue;
511 }
512 return false;
513 }
514 return true;
515 }
516 // if the left map has an empty bitmap, skip it
517 if (lhs_iter->second.isEmpty()) {
518 ++lhs_iter;
519 continue;
520 }
521
522 do {
523 // if the right map has reached its end, ensure that the right
524 // map contains only empty Bitmaps
525 if (rhs_iter == r.roarings.cend()) {
526 while (lhs_iter != roarings.cend()) {
527 if (lhs_iter->second.isEmpty()) {
528 ++lhs_iter;
529 continue;
530 }
531 return false;
532 }
533 return true;
534 }
535 // if the right map has an empty bitmap, skip it
536 if (rhs_iter->second.isEmpty()) {
537 ++rhs_iter;
538 continue;
539 }
540 } while (false);
541 // if neither map has reached its end ensure elements are equal and
542 // move to the next element in both
543 } while (lhs_iter++->second == rhs_iter++->second);
544 return false;
545 }

551 void flip(uint64_t range_start, uint64_t range_end) {
552 uint32_t start_high = highBytes(range_start);
553 uint32_t start_low = lowBytes(range_start);
554 uint32_t end_high = highBytes(range_end);
555 uint32_t end_low = lowBytes(range_end);
556
557 if (start_high == end_high) {
558 roarings[start_high].flip(start_low, end_low);
559 return;
560 }
561 // we put std::numeric_limits<>::max/min in parenthesis
562 // to avoid a clash with the Windows.h header under Windows
563 roarings[start_high].flip(start_low, (std::numeric_limits<uint32_t>::max)());
564 roarings[start_high++].setCopyOnWrite(copyOnWrite);
565
566 for (; start_high <= highBytes(range_end) - 1; ++start_high) {
567 roarings[start_high].flip((std::numeric_limits<uint32_t>::min)(),
568 (std::numeric_limits<uint32_t>::max)());
569 roarings[start_high].setCopyOnWrite(copyOnWrite);
570 }
571
572 roarings[start_high].flip((std::numeric_limits<uint32_t>::min)(), end_low);
573 roarings[start_high].setCopyOnWrite(copyOnWrite);
574 }

647 bool select(uint64_t rnk, uint64_t* element) const {
648 for (const auto& map_entry : roarings) {
649 uint64_t sub_cardinality = (uint64_t)map_entry.second.cardinality();
650 if (rnk < sub_cardinality) {
651 *element = ((uint64_t)map_entry.first) << 32;
652 // assuming little endian
653 return map_entry.second.select((uint32_t)rnk, ((uint32_t*)element));
654 }

656 }
657 return false;

663 uint64_t rank(uint64_t x) const {
664 uint64_t result = 0;
665 auto roaring_destination = roarings.find(highBytes(x));
666 if (roaring_destination != roarings.cend()) {
667 for (auto roaring_iter = roarings.cbegin(); roaring_iter != roaring_destination;

675 for (auto roaring_iter = roarings.cbegin(); roaring_iter != roaring_destination;
676 ++roaring_iter) {
677 result += roaring_iter->second.cardinality();
678 }
679 return result;

686 size_t write(char* buf) const {
687 if (is32BitsEnough()) {
688 *(buf++) = BitmapTypeCode::type::BITMAP32;
689 auto it = roarings.find(0);
690 if (it == roarings.end()) { // empty bitmap
691 roaring::Roaring r;
692 return r.write(buf) + 1;
693 }
694 return it->second.write(buf) + 1;
695 }
696
697 const char* orig = buf;
698 // put type code
699 *(buf++) = BitmapTypeCode::type::BITMAP64;
700 // push map size
701 buf = (char*)encode_varint64((uint8_t*)buf, roarings.size());
702 std::for_each(roarings.cbegin(), roarings.cend(),
703 [&buf](const std::pair<const uint32_t, roaring::Roaring>& map_entry) {
704 // push map key
705 encode_fixed32_le((uint8_t*)buf, map_entry.first);
706 buf += sizeof(uint32_t);
707 // push map value Roaring
708 buf += map_entry.second.write(buf);
709 });
710 return buf - orig;
711 }

719 static Roaring64Map read(const char* buf) {
720 Roaring64Map result;
721
722 if (*buf == BitmapTypeCode::BITMAP32) {
723 roaring::Roaring read = roaring::Roaring::read(buf + 1);
724 result.emplaceOrInsert(0, std::move(read));
725 return result;
726 }
727
728 DCHECK_EQ(BitmapTypeCode::BITMAP64, *buf);
729 buf++;
730
731 // get map size (varint64 took 1~10 bytes)
732 uint64_t map_size;
733 buf = reinterpret_cast<const char*>(
734 decode_varint64_ptr(reinterpret_cast<const uint8_t*>(buf),
735 reinterpret_cast<const uint8_t*>(buf + 10), &map_size));
736 DCHECK(buf != nullptr);
737 for (uint64_t lcv = 0; lcv < map_size; lcv++) {
738 // get map key
739 uint32_t key = decode_fixed32_le(reinterpret_cast<const uint8_t*>(buf));
740 buf += sizeof(uint32_t);
741 // read map value Roaring
742 roaring::Roaring read_var = roaring::Roaring::read(buf);
743 // forward buffer past the last Roaring Bitmap
744 buf += read_var.getSizeInBytes();
745 result.emplaceOrInsert(key, std::move(read_var));
746 }
747 return result;
748 }

753 size_t getSizeInBytes() const {

755 auto it = roarings.find(0);
