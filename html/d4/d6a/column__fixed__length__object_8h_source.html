54 void resize(size_t n) override {
55 DCHECK(_item_size > 0) << "_item_size should be greater than 0";
56 _data.resize(n * _item_size);
57 _item_count = n;
58 }
59
60 MutableColumnPtr clone_resized(size_t size) const override {
61 auto res = this->create(_item_size);
62
63 if (size > 0) {
64 auto& new_col = assert_cast<Self&>(*res);
65 new_col.resize(size);
66 auto* new_data = new_col._data.data();
67
68 size_t count = std::min(this->size(), size);
69 memcpy(new_data, _data.data(), count * _item_size);
70
71 if (size > count) memset(new_data + count * _item_size, 0, (size - count) * _item_size);
72 }
73
74 return res;
75 }
76
77 void insert_indices_from(const IColumn& src, const int* indices_begin,
78 const int* indices_end) override {
79 const Self& src_vec = assert_cast<const Self&>(src);
80 auto origin_size = size();
81 auto new_size = indices_end - indices_begin;
82 if (_item_size == 0) {
83 _item_size = src_vec._item_size;
84 }
85 DCHECK(_item_size == src_vec._item_size) << "dst and src should have the same _item_size";
86 resize(origin_size + new_size);
87
88 for (int i = 0; i < new_size; ++i) {
89 int offset = indices_begin[i];
90 if (offset > -1) {
91 memcpy(&_data[(origin_size + i) * _item_size], &src_vec._data[offset * _item_size],
92 _item_size);
93 } else {
94 memset(&_data[(origin_size + i) * _item_size], 0, _item_size);
95 }
96 }
97 }
98
99 void clear() override {
100 _data.clear();
101 _item_count = 0;
102 }
