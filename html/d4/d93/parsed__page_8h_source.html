35// This contains information when one page is loaded, and ready for read
36// This struct can be reused, client should call reset first before reusing
37// this object
38struct ParsedPage {
39 static Status create(PageHandle handle, const Slice& body, const DataPageFooterPB& footer,
40 const EncodingInfo* encoding, const PagePointer& page_pointer,
41 uint32_t page_index, ParsedPage* result,
42 PageDecoderOptions opts = PageDecoderOptions()) {
43 result->~ParsedPage();
44 ParsedPage* page = new (result)(ParsedPage);
45 page->page_handle = std::move(handle);
46
47 auto null_size = footer.nullmap_size();
48 page->has_null = null_size > 0;
49 page->null_bitmap = Slice(body.data + body.size - null_size, null_size);
50
51 if (page->has_null) {
52 page->null_decoder =
53 RleDecoder<bool>((const uint8_t*)page->null_bitmap.data, null_size, 1);
54 }
55
56 Slice data_slice(body.data, body.size - null_size);
57 RETURN_IF_ERROR(encoding->create_page_decoder(data_slice, opts, &page->data_decoder));
58 RETURN_IF_ERROR(page->data_decoder->init());
59
60 if (encoding->encoding() == DICT_ENCODING) {
61 auto dict_decoder = static_cast<BinaryDictPageDecoder*>(page->data_decoder);
62 page->is_dict_encoding = dict_decoder->is_dict_encoding();
63 }
64
65 page->first_ordinal = footer.first_ordinal();
66 page->num_rows = footer.num_values();
67
68 page->page_pointer = page_pointer;
69 page->page_index = page_index;
70
71 page->next_array_item_ordinal = footer.next_array_item_ordinal();
72
73 return Status::OK();
74 }
75
76 ~ParsedPage() {
77 delete data_decoder;
78 data_decoder = nullptr;
79 }
80
81 PageHandle page_handle;
82
83 bool has_null;
84 Slice null_bitmap;
85 RleDecoder<bool> null_decoder;
86 PageDecoder* data_decoder = nullptr;
87
88 // ordinal of the first value in this page
89 ordinal_t first_ordinal = 0;
90 // number of rows including nulls and not-nulls
91 ordinal_t num_rows = 0;
92 // record it to get the last array element's size
93 // should be none zero if set in page
94 ordinal_t next_array_item_ordinal = 0;
95
96 PagePointer page_pointer;
97 uint32_t page_index = 0;
98
99 // current offset when read this page
100 // this means next row we will read
101 ordinal_t offset_in_page = 0;
102
103 bool is_dict_encoding = false;
104
105 bool contains(ordinal_t ord) {
106 return ord >= first_ordinal && ord < (first_ordinal + num_rows);
107 }
108
109 operator bool() const { return data_decoder != nullptr; }
110
111 bool has_remaining() const { return offset_in_page < num_rows; }
112
113 size_t remaining() const { return num_rows - offset_in_page; }
114};
