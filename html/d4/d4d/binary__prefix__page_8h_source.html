38// prefix encoding for string dictionary
39//
40// BinaryPrefixPage := Entry^EntryNum, Trailer
41// Entry := SharedPrefixLength(vint), UnsharedLength(vint), Byte^UnsharedLength
42// Trailer := NumEntry(uint32_t), RESTART_POINT_INTERVAL(uint8_t)
43// RestartPointStartOffset(uint32_t)^NumRestartPoints,NumRestartPoints(uint32_t)
44class BinaryPrefixPageBuilder : public PageBuilder {
45public:
46 BinaryPrefixPageBuilder(const PageBuilderOptions& options) : _options(options) { reset(); }
47
48 bool is_page_full() override { return size() >= _options.data_page_size; }
49
50 Status add(const uint8_t* vals, size_t* add_count) override;
51
52 OwnedSlice finish() override;
53
54 void reset() override {
55 _restart_points_offset.clear();
56 _last_entry.clear();
57 _count = 0;
58 _buffer.clear();
59 _finished = false;
60 }
61
62 uint64_t size() const override {
63 if (_finished) {
64 return _buffer.size();
65 } else {
66 return _buffer.size() + (_restart_points_offset.size() + 2) * sizeof(uint32_t);
67 }
68 }
69
70 size_t count() const override { return _count; }
71
72 Status get_first_value(void* value) const override {
73 DCHECK(_finished);
74 if (_count == 0) {
75 return Status::NotFound("page is empty");
76 }
77 *reinterpret_cast<Slice*>(value) = Slice(_first_entry);
78 return Status::OK();
79 }
80
81 Status get_last_value(void* value) const override {
82 DCHECK(_finished);
83 if (_count == 0) {
84 return Status::NotFound("page is empty");
85 }
86 *reinterpret_cast<Slice*>(value) = Slice(_last_entry);
87 return Status::OK();
88 }
89
90private:
91 PageBuilderOptions _options;
92 std::vector<uint32_t> _restart_points_offset;
93 faststring _first_entry;
94 faststring _last_entry;
95 size_t _count = 0;
96 bool _finished = false;
97 faststring _buffer;
98 // This is a empirical value, Kudu and LevelDB use this default value
99 static const uint8_t RESTART_POINT_INTERVAL = 16;
100};

102class BinaryPrefixPageDecoder : public PageDecoder {
103public:
104 BinaryPrefixPageDecoder(Slice data, const PageDecoderOptions& options)
105 : _data(data), _parsed(false) {}
106
107 Status init() override;
108
109 Status seek_to_position_in_page(size_t pos) override;
110
111 Status seek_at_or_after_value(const void* value, bool* exact_match) override;
112
113 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override;
114
115 size_t count() const override {
116 DCHECK(_parsed);
117 return _num_values;
118 }
119
120 size_t current_index() const override {
121 DCHECK(_parsed);
122 return _cur_pos;
123 }
124
125private:
126 // decode shared and non-shared entry length from `ptr`.
127 // return ptr past the parsed value when success.
128 // return nullptr on failure
129 const uint8_t* _decode_value_lengths(const uint8_t* ptr, uint32_t* shared,
130 uint32_t* non_shared);
131
132 // return start pointer of the restart point at index `restart_point_index`
133 const uint8_t* _get_restart_point(size_t restart_point_index) const {
134 return reinterpret_cast<const uint8_t*>(_data.get_data()) +
135 decode_fixed32_le(_restarts_ptr + restart_point_index * sizeof(uint32_t));
136 }
137
138 // read next value at `_cur_pos` and `_next_ptr` into `_current_value`.
139 // return OK and advance `_next_ptr` on success. `_cur_pos` is not modified.
140 // return NotFound when no more entry can be read.
141 // return other error status otherwise.
142 Status _read_next_value();
143
144 // seek to the first value at the given restart point
145 Status _seek_to_restart_point(size_t restart_point_index);
146
147 Slice _data;
148 bool _parsed = false;
149 size_t _num_values = 0;
150 uint8_t _restart_point_internal = 0;
151 uint32_t _num_restarts = 0;
152 // pointer to _footer start
153 const uint8_t* _footer_start = nullptr;
154 // pointer to restart offsets array
155 const uint8_t* _restarts_ptr = nullptr;
156 // ordinal of the first value to return in next_batch()
157 uint32_t _cur_pos = 0;
158 // first value to return in next_batch()
159 faststring _current_value;
160 // pointer to the start of next value to read, advanced by `_read_next_value`
161 const uint8_t* _next_ptr = nullptr;
162};
