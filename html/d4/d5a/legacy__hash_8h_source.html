27// ----------------------------------------------------------------------
28// HashTo32()
29// HashTo16()
30// These functions take various types of input (through operator
31// overloading) and return 32 or 16 bit quantities, respectively.
32// The basic rule of our hashing is: always mix(). Thus, even for
33// char outputs we cast to a uint32 and mix with two arbitrary numbers.
34// HashTo32 never returns kIllegalHash32, and similarity,
35// HashTo16 never returns kIllegalHash16.
36//
37// Note that these methods avoid returning certain reserved values, while
38// the corresponding HashXXStringWithSeed() methods may return any value.
39// ----------------------------------------------------------------------
40
41// This macro defines the HashTo32 and HashTo16 versions all in one go.
42// It takes the argument list and a command that hashes your number.
43// (For 16 we just mod retval before returning it.) Example:
44// HASH_TO((char c), Hash32NumWithSeed(c, MIX32_1))
45// evaluates to
46// uint32 retval;
47// retval = Hash32NumWithSeed(c, MIX32_1);
48// return retval == kIllegalHash32 ? retval-1 : retval;
49//
50
51#define HASH_TO(arglist, command) \
52 inline uint32 HashTo32 arglist { \
53 uint32 retval = command; \
54 return retval == kIllegalHash32 ? retval - 1 : retval; \
55 }
56
57// This defines:
58// HashToXX(char *s, int slen);
59// HashToXX(char c);
60// etc

62HASH_TO((const char* s, uint32 slen), Hash32StringWithSeed(s, slen, MIX32))
63HASH_TO((const wchar_t* s, uint32 slen),

65 static_cast<uint32>(sizeof(wchar_t) * slen), MIX32))

67HASH_TO((schar c), Hash32NumWithSeed(static_cast<uint32>(c), MIX32))
68HASH_TO((uint16 c), Hash32NumWithSeed(static_cast<uint32>(c), MIX32))
69HASH_TO((int16 c), Hash32NumWithSeed(static_cast<uint32>(c), MIX32))
70HASH_TO((uint32 c), Hash32NumWithSeed(static_cast<uint32>(c), MIX32))
71HASH_TO((int32 c), Hash32NumWithSeed(static_cast<uint32>(c), MIX32))
72HASH_TO((uint64 c), static_cast<uint32>(Hash64NumWithSeed(c, MIX64) >> 32))
73HASH_TO((int64 c), static_cast<uint32>(Hash64NumWithSeed(c, MIX64) >> 32))

77inline uint16 HashTo16(const char* s, uint32 slen) {
78 uint16 retval = Hash32StringWithSeed(s, slen, MIX32) >> 16;
79 return retval == kIllegalHash16 ? static_cast<uint16>(retval - 1) : retval;
80}
