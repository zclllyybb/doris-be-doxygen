31/*
32 * legal utf-8 byte sequence
33 * http://www.unicode.org/versions/Unicode6.0.0/ch03.pdf - page 94
34 *
35 * Code Points 1st 2s 3s 4s
36 * U+0000..U+007F 00..7F
37 * U+0080..U+07FF C2..DF 80..BF
38 * U+0800..U+0FFF E0 A0..BF 80..BF
39 * U+1000..U+CFFF E1..EC 80..BF 80..BF
40 * U+D000..U+D7FF ED 80..9F 80..BF
41 * U+E000..U+FFFF EE..EF 80..BF 80..BF
42 * U+10000..U+3FFFF F0 90..BF 80..BF 80..BF
43 * U+40000..U+FFFFF F1..F3 80..BF 80..BF 80..BF
44 * U+100000..U+10FFFF F4 80..8F 80..BF 80..BF
45 *
46 */

48// all byte values must be no larger than 0xF4
49static inline void checkSmallerThan0xF4(__m128i current_bytes, __m128i* has_error) {
50 // unsigned, saturates to 0 below max
51 *has_error = _mm_or_si128(*has_error, _mm_subs_epu8(current_bytes, _mm_set1_epi8(0xF4)));
52}

54static inline __m128i continuationLengths(__m128i high_nibbles) {
55 return _mm_shuffle_epi8(_mm_setr_epi8(1, 1, 1, 1, 1, 1, 1, 1, // 0xxx (ASCII)
56 0, 0, 0, 0, // 10xx (continuation)
57 2, 2, // 110x
58 3, // 1110
59 4), // 1111, next should be 0 (not checked here)
60 high_nibbles);
61}

63static inline __m128i carryContinuations(__m128i initial_lengths, __m128i previous_carries) {
64 __m128i right1 = _mm_subs_epu8(_mm_alignr_epi8(initial_lengths, previous_carries, 16 - 1),
65 _mm_set1_epi8(1));
66 __m128i sum = _mm_add_epi8(initial_lengths, right1);
67
68 __m128i right2 =
69 _mm_subs_epu8(_mm_alignr_epi8(sum, previous_carries, 16 - 2), _mm_set1_epi8(2));
70 return _mm_add_epi8(sum, right2);
71}

73static inline void checkContinuations(__m128i initial_lengths, __m128i carries,
74 __m128i* has_error) {
75 // overlap || underlap
76 // carry > length && length > 0 || !(carry > length) && !(length > 0)
77 // (carries > length) == (lengths > 0)
78 __m128i overunder = _mm_cmpeq_epi8(_mm_cmpgt_epi8(carries, initial_lengths),
79 _mm_cmpgt_epi8(initial_lengths, _mm_setzero_si128()));
80
81 *has_error = _mm_or_si128(*has_error, overunder);
82}

84// when 0xED is found, next byte must be no larger than 0x9F
85// when 0xF4 is found, next byte must be no larger than 0x8F
86// next byte must be continuation, ie sign bit is set, so signed < is ok
87static inline void checkFirstContinuationMax(__m128i current_bytes, __m128i off1_current_bytes,
88 __m128i* has_error) {
89 __m128i maskED = _mm_cmpeq_epi8(off1_current_bytes, _mm_set1_epi8(0xED));
90 __m128i maskF4 = _mm_cmpeq_epi8(off1_current_bytes, _mm_set1_epi8(0xF4));
91
92 __m128i badfollowED = _mm_and_si128(_mm_cmpgt_epi8(current_bytes, _mm_set1_epi8(0x9F)), maskED);
93 __m128i badfollowF4 = _mm_and_si128(_mm_cmpgt_epi8(current_bytes, _mm_set1_epi8(0x8F)), maskF4);
94
95 *has_error = _mm_or_si128(*has_error, _mm_or_si128(badfollowED, badfollowF4));
96}

98// map off1_hibits => error condition
99// hibits off1 cur
100// C => < C2 && true
101// E => < E1 && < A0
102// F => < F1 && < 90
103// else false && false
104static inline void checkOverlong(__m128i current_bytes, __m128i off1_current_bytes, __m128i hibits,
105 __m128i previous_hibits, __m128i* has_error) {
106 __m128i off1_hibits = _mm_alignr_epi8(hibits, previous_hibits, 16 - 1);
107 __m128i initial_mins =
108 _mm_shuffle_epi8(_mm_setr_epi8(-128, -128, -128, -128, -128, -128, -128, -128, -128,
109 -128, -128, -128, // 10xx => false
110 0xC2, -128, // 110x
111 0xE1, // 1110
112 0xF1),
113 off1_hibits);
114
115 __m128i initial_under = _mm_cmpgt_epi8(initial_mins, off1_current_bytes);
116
117 __m128i second_mins =
118 _mm_shuffle_epi8(_mm_setr_epi8(-128, -128, -128, -128, -128, -128, -128, -128, -128,
119 -128, -128, -128, // 10xx => false
120 127, 127, // 110x => true
121 0xA0, // 1110
122 0x90),
123 off1_hibits);
124 __m128i second_under = _mm_cmpgt_epi8(second_mins, current_bytes);
125 *has_error = _mm_or_si128(*has_error, _mm_and_si128(initial_under, second_under));
126}

139// check whether the current bytes are valid UTF-8
140// at the end of the function, previous gets updated
141static struct processed_utf_bytes checkUTF8Bytes(__m128i current_bytes,
142 struct processed_utf_bytes* previous,
143 __m128i* has_error) {
144 struct processed_utf_bytes pb;
145 count_nibbles(current_bytes, &pb);
146
147 checkSmallerThan0xF4(current_bytes, has_error);
148
149 __m128i initial_lengths = continuationLengths(pb.high_nibbles);
150
151 pb.carried_continuations = carryContinuations(initial_lengths, previous->carried_continuations);
152
153 checkContinuations(initial_lengths, pb.carried_continuations, has_error);
154
155 __m128i off1_current_bytes = _mm_alignr_epi8(pb.rawbytes, previous->rawbytes, 16 - 1);
156 checkFirstContinuationMax(current_bytes, off1_current_bytes, has_error);
157
158 checkOverlong(current_bytes, off1_current_bytes, pb.high_nibbles, previous->high_nibbles,
159 has_error);
160 return pb;
161}

163static bool validate_utf8_fast(const char* src, size_t len) {
164 size_t i = 0;
165 __m128i has_error = _mm_setzero_si128();
166 struct processed_utf_bytes previous = {.rawbytes = _mm_setzero_si128(),
167 .high_nibbles = _mm_setzero_si128(),
168 .carried_continuations = _mm_setzero_si128()};
169 if (len >= 16) {
170 for (; i <= len - 16; i += 16) {
171 __m128i current_bytes = _mm_loadu_si128((const __m128i*)(src + i));
172 previous = checkUTF8Bytes(current_bytes, &previous, &has_error);
173 }
174 }
175
176 // last part
177 if (i < len) {
178 char buffer[16];
179 memset(buffer, 0, 16);
180 memcpy(buffer, src + i, len - i);
181 __m128i current_bytes = _mm_loadu_si128((const __m128i*)(buffer));
182 previous = checkUTF8Bytes(current_bytes, &previous, &has_error);
183 } else {
184 has_error = _mm_or_si128(
185 _mm_cmpgt_epi8(previous.carried_continuations,
186 _mm_setr_epi8(9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 1)),
187 has_error);
188 }
189
190 return _mm_testz_si128(has_error, has_error);
191}
