35inline bool allow_decimal_comparison(const DataTypePtr& left_type, const DataTypePtr& right_type) {
36 if (is_decimal(left_type)) {
37 if (is_decimal(right_type) || is_not_decimal_but_comparable_to_decimal(right_type))
38 return true;
39 } else if (is_not_decimal_but_comparable_to_decimal(left_type) && is_decimal(right_type))
40 return true;
41 return false;
42}

66template <typename A, typename B, template <typename, typename> typename Operation,
67 bool _check_overflow = true, bool _actual = IsDecimalNumber<A> || IsDecimalNumber<B>>
68class DecimalComparison {
69public:
70 using CompareInt = typename DecCompareInt<A, B>::Type;
71 using Op = Operation<CompareInt, CompareInt>;
72 using ColVecA = std::conditional_t<IsDecimalNumber<A>, ColumnDecimal<A>, ColumnVector<A>>;
73 using ColVecB = std::conditional_t<IsDecimalNumber<B>, ColumnDecimal<B>, ColumnVector<B>>;
74 using ArrayA = typename ColVecA::Container;
75 using ArrayB = typename ColVecB::Container;

77 DecimalComparison(Block& block, size_t result, const ColumnWithTypeAndName& col_left,
78 const ColumnWithTypeAndName& col_right) {
79 if (!apply(block, result, col_left, col_right)) {
80 LOG(FATAL) << fmt::format("Wrong decimal comparison with {} and {}",
81 col_left.type->get_name(), col_right.type->get_name());
82 }
83 }

85 static bool apply(Block& block, size_t result [[maybe_unused]],
86 const ColumnWithTypeAndName& col_left,
87 const ColumnWithTypeAndName& col_right) {
88 if constexpr (_actual) {
89 ColumnPtr c_res;
90 Shift shift = getScales<A, B>(col_left.type, col_right.type);
91
92 c_res = apply_with_scale(col_left.column, col_right.column, shift);
93 if (c_res) {
94 block.replace_by_position(result, std::move(c_res));
95 }
96 return true;
97 }
98 return false;
99 }

101 static bool compare(A a, B b, UInt32 scale_a, UInt32 scale_b) {
102 static const UInt32 max_scale = max_decimal_precision<Decimal128>();
103 if (scale_a > max_scale || scale_b > max_scale) {
104 LOG(FATAL) << "Bad scale of decimal field";
105 }
106
107 Shift shift;
108 if (scale_a < scale_b)
109 shift.a = DataTypeDecimal<B>(max_decimal_precision<B>(), scale_b)
110 .get_scale_multiplier(scale_b - scale_a);
111 if (scale_a > scale_b)
112 shift.b = DataTypeDecimal<A>(max_decimal_precision<A>(), scale_a)
113 .get_scale_multiplier(scale_a - scale_b);
114
115 return apply_with_scale(a, b, shift);
116 }
117
118private:
119 struct Shift {
120 CompareInt a = 1;
121 CompareInt b = 1;
122
123 bool none() const { return a == 1 && b == 1; }
124 bool left() const { return a != 1; }
125 bool right() const { return b != 1; }
126 };

128 template <typename T, typename U>
129 static auto apply_with_scale(T a, U b, const Shift& shift) {
130 if (shift.left())
131 return apply<true, false>(a, b, shift.a);
132 else if (shift.right())
133 return apply<false, true>(a, b, shift.b);
134 return apply<false, false>(a, b, 1);
135 }

137 template <typename T, typename U>
138 static std::enable_if_t<IsDecimalNumber<T> && IsDecimalNumber<U>, Shift> getScales(
139 const DataTypePtr& left_type, const DataTypePtr& right_type) {
140 const DataTypeDecimal<T>* decimal0 = check_decimal<T>(*left_type);
141 const DataTypeDecimal<U>* decimal1 = check_decimal<U>(*right_type);
142
143 Shift shift;
144 if (decimal0 && decimal1) {
145 using Type = std::conditional_t<sizeof(T) >= sizeof(U), T, U>;
146 auto type_ptr = decimal_result_type(*decimal0, *decimal1, false, false, false);
147 const DataTypeDecimal<Type>* result_type = check_decimal<Type>(*type_ptr);
148 shift.a = result_type->scale_factor_for(*decimal0, false);
149 shift.b = result_type->scale_factor_for(*decimal1, false);
150 } else if (decimal0) {
151 shift.b = decimal0->get_scale_multiplier();
152 } else if (decimal1) {
153 shift.a = decimal1->get_scale_multiplier();
154 }
155
156 return shift;
157 }

177 template <bool scale_left, bool scale_right>
178 static ColumnPtr apply(const ColumnPtr& c0, const ColumnPtr& c1, CompareInt scale) {
179 auto c_res = ColumnUInt8::create();
180
181 if constexpr (_actual) {
182 bool c0_is_const = is_column_const(*c0);
183 bool c1_is_const = is_column_const(*c1);
184
185 if (c0_is_const && c1_is_const) {
186 const ColumnConst* c0_const = check_and_get_column_const<ColVecA>(c0.get());
187 const ColumnConst* c1_const = check_and_get_column_const<ColVecB>(c1.get());
188
189 A a = c0_const->template get_value<A>();
190 B b = c1_const->template get_value<B>();
191 UInt8 res = apply<scale_left, scale_right>(a, b, scale);
192 return DataTypeUInt8().create_column_const(c0->size(), to_field(res));
193 }
194
195 ColumnUInt8::Container& vec_res = c_res->get_data();
196 vec_res.resize(c0->size());
197
198 if (c0_is_const) {
199 const ColumnConst* c0_const = check_and_get_column_const<ColVecA>(c0.get());
200 A a = c0_const->template get_value<A>();
201 if (const ColVecB* c1_vec = check_and_get_column<ColVecB>(c1.get()))
202 constant_vector<scale_left, scale_right>(a, c1_vec->get_data(), vec_res, scale);
203 else {
204 LOG(FATAL) << "Wrong column in Decimal comparison";
205 }
206 } else if (c1_is_const) {
207 const ColumnConst* c1_const = check_and_get_column_const<ColVecB>(c1.get());
208 B b = c1_const->template get_value<B>();
209 if (const ColVecA* c0_vec = check_and_get_column<ColVecA>(c0.get()))
210 vector_constant<scale_left, scale_right>(c0_vec->get_data(), b, vec_res, scale);
211 else {
212 LOG(FATAL) << "Wrong column in Decimal comparison";
213 }
214 } else {
215 if (const ColVecA* c0_vec = check_and_get_column<ColVecA>(c0.get())) {
216 if (const ColVecB* c1_vec = check_and_get_column<ColVecB>(c1.get()))
217 vector_vector<scale_left, scale_right>(c0_vec->get_data(),
218 c1_vec->get_data(), vec_res, scale);
219 else {
220 LOG(FATAL) << "Wrong column in Decimal comparison";
221 }
222 } else {
223 LOG(FATAL) << "Wrong column in Decimal comparison";
224 }
225 }
226 }
227
228 return c_res;
229 }

231 template <bool scale_left, bool scale_right>
232 static NO_INLINE UInt8 apply(A a, B b, CompareInt scale [[maybe_unused]]) {
233 CompareInt x = a;
234 CompareInt y = b;
235
236 if constexpr (_check_overflow) {
237 bool overflow = false;
238
239 if constexpr (sizeof(A) > sizeof(CompareInt)) overflow |= (A(x) != a);
240 if constexpr (sizeof(B) > sizeof(CompareInt)) overflow |= (B(y) != b);
241 if constexpr (std::is_unsigned_v<A>) overflow |= (x < 0);
242 if constexpr (std::is_unsigned_v<B>) overflow |= (y < 0);
243
244 if constexpr (scale_left) overflow |= common::mul_overflow(x, scale, x);
245 if constexpr (scale_right) overflow |= common::mul_overflow(y, scale, y);
246
247 if (overflow) {
248 LOG(FATAL) << "Can't compare";
249 }
250 } else {
251 if constexpr (scale_left) x *= scale;
252 if constexpr (scale_right) y *= scale;
253 }
254
255 return Op::apply(x, y);
256 }

258 template <bool scale_left, bool scale_right>
259 static void NO_INLINE vector_vector(const ArrayA& a, const ArrayB& b, PaddedPODArray<UInt8>& c,
260 CompareInt scale) {
261 size_t size = a.size();
262 const A* a_pos = a.data();
263 const B* b_pos = b.data();
264 UInt8* c_pos = c.data();
265 const A* a_end = a_pos + size;
266
267 while (a_pos < a_end) {
268 *c_pos = apply<scale_left, scale_right>(*a_pos, *b_pos, scale);
269 ++a_pos;
270 ++b_pos;
271 ++c_pos;
272 }
273 }
