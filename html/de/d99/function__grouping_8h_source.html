25class FunctionGroupingBase : public IFunction {
26public:
27 size_t get_number_of_arguments() const override { return 1; }
28
29 bool use_default_implementation_for_constants() const override { return false; }
30
31 bool use_default_implementation_for_nulls() const override { return false; }
32
33 DataTypePtr get_return_type_impl(const ColumnsWithTypeAndName& arguments) const override {
34 return std::make_shared<DataTypeInt64>();
35 }
36
37 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
38 size_t result, size_t input_rows_count) override {
39 const ColumnWithTypeAndName& src_column = block.get_by_position(arguments[0]);
40 DCHECK(src_column.column->size() == input_rows_count);
41 // result of functions grouping and grouping_id is always not nullable,
42 // but outer join will convert the column to nullable when necessary,
43 // so need to remove nullable here when functions grouping and grouping_id are executed
44 block.get_by_position(result).column = remove_nullable(src_column.column);
45 return Status::OK();
46 }
47};
