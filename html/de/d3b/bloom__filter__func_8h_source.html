25class BloomFilterAdaptor {
26public:
27 BloomFilterAdaptor() { _bloom_filter = std::make_shared<doris::BlockBloomFilter>(); }
28 static int64_t optimal_bit_num(int64_t expect_num, double fpp) {
29 return doris::segment_v2::BloomFilter::optimal_bit_num(expect_num, fpp) / 8;
30 }
31
32 static BloomFilterAdaptor* create() { return new BloomFilterAdaptor(); }
33
34 Status merge(BloomFilterAdaptor* other) { return _bloom_filter->merge(*other->_bloom_filter); }
35
36 Status init(int len) {
37 int log_space = log2(len);
38 return _bloom_filter->init(log_space, /*hash_seed*/ 0);
39 }
40
41 Status init(butil::IOBufAsZeroCopyInputStream* data, const size_t data_size) {
42 int log_space = log2(data_size);
43 return _bloom_filter->init_from_directory(log_space, data, data_size, false, 0);
44 }
45
46 char* data() { return (char*)_bloom_filter->directory().data; }
47
48 size_t size() { return _bloom_filter->directory().size; }
49
50 template <typename T>
51 bool test(T data) const {
52 return _bloom_filter->find(data);
53 }
54
55 void add_bytes(const char* data, size_t len) { _bloom_filter->insert(Slice(data, len)); }
56
57 // test_element/find_element only used on vectorized engine
58 template <typename T>
59 bool test_element(T element) const {
60 if constexpr (std::is_same_v<T, Slice>) {
61 return _bloom_filter->find(element);
62 } else {
63 return _bloom_filter->find(HashUtil::fixed_len_to_uint32(element));
64 }
65 }
66
67 template <typename T>
68 void add_element(T element) {
69 if constexpr (std::is_same_v<T, Slice>) {
70 _bloom_filter->insert(element);
71 } else {
72 _bloom_filter->insert(HashUtil::fixed_len_to_uint32(element));
73 }
74 }
75
76private:
77 std::shared_ptr<doris::BlockBloomFilter> _bloom_filter;
78};

81class BloomFilterFuncBase {
82public:
83 BloomFilterFuncBase() : _inited(false) {}
84
85 virtual ~BloomFilterFuncBase() = default;
86
87 Status init(int64_t expect_num, double fpp) {
88 size_t filter_size = BloomFilterAdaptor::optimal_bit_num(expect_num, fpp);
89 return init_with_fixed_length(filter_size);
90 }
91
92 void set_length(int64_t bloom_filter_length) { _bloom_filter_length = bloom_filter_length; }
93
94 Status init_with_fixed_length() { return init_with_fixed_length(_bloom_filter_length); }
95
96 Status init_with_fixed_length(int64_t bloom_filter_length) {
97 if (_inited) {
98 return Status::OK();
99 }
100 std::lock_guard<std::mutex> l(_lock);
101 if (_inited) {
102 return Status::OK();
103 }
104 DCHECK(bloom_filter_length >= 0);
105 DCHECK_EQ((bloom_filter_length & (bloom_filter_length - 1)), 0);
106 _bloom_filter_alloced = bloom_filter_length;
107 _bloom_filter.reset(BloomFilterAdaptor::create());
108 RETURN_IF_ERROR(_bloom_filter->init(bloom_filter_length));
109 _inited = true;
110 return Status::OK();
111 }
112
113 Status merge(BloomFilterFuncBase* bloomfilter_func) {
114 // If `_inited` is false, there is no memory allocated in bloom filter and this is the first
115 // call for `merge` function. So we just reuse this bloom filter, and we don't need to
116 // allocate memory again.
117 if (_inited) {
118 DCHECK(bloomfilter_func != nullptr);
119 auto other_func = static_cast<BloomFilterFuncBase*>(bloomfilter_func);
120 if (_bloom_filter_alloced != other_func->_bloom_filter_alloced) {
121 LOG(WARNING) << "bloom filter size not the same: already allocated bytes = "
122 << _bloom_filter_alloced << ", expected allocated bytes = "
123 << other_func->_bloom_filter_alloced;
124 return Status::InvalidArgument("bloom filter size invalid");
125 }
126 return _bloom_filter->merge(other_func->_bloom_filter.get());
127 }
128 {
129 std::lock_guard<std::mutex> l(_lock);
130 if (!_inited) {
131 auto other_func = static_cast<BloomFilterFuncBase*>(bloomfilter_func);
132 DCHECK(_bloom_filter == nullptr);
133 DCHECK(bloomfilter_func != nullptr);
134 _bloom_filter = bloomfilter_func->_bloom_filter;
135 _bloom_filter_alloced = other_func->_bloom_filter_alloced;
136 _inited = true;
137 return Status::OK();
138 } else {
139 DCHECK(bloomfilter_func != nullptr);
140 auto other_func = static_cast<BloomFilterFuncBase*>(bloomfilter_func);
141 if (_bloom_filter_alloced != other_func->_bloom_filter_alloced) {
142 LOG(WARNING) << "bloom filter size not the same: already allocated bytes = "
143 << _bloom_filter_alloced << ", expected allocated bytes = "
144 << other_func->_bloom_filter_alloced;
145 return Status::InvalidArgument("bloom filter size invalid");
146 }
147 return _bloom_filter->merge(other_func->_bloom_filter.get());
148 }
149 }
150 }
151
152 Status assign(butil::IOBufAsZeroCopyInputStream* data, const size_t data_size) {
153 if (_bloom_filter == nullptr) {
154 _bloom_filter.reset(BloomFilterAdaptor::create());
155 }
156
157 _bloom_filter_alloced = data_size;
158 return _bloom_filter->init(data, data_size);
159 }
160
161 Status get_data(char** data, int* len) {
162 *data = _bloom_filter->data();
163 *len = _bloom_filter->size();
164 return Status::OK();
165 }
166
167 void light_copy(BloomFilterFuncBase* bloomfilter_func) {
168 auto other_func = static_cast<BloomFilterFuncBase*>(bloomfilter_func);
169 _bloom_filter_alloced = other_func->_bloom_filter_alloced;
170 _bloom_filter = other_func->_bloom_filter;
171 _inited = other_func->_inited;
172 }
173
174 virtual void insert(const void* data) = 0;
175
176 virtual bool find(const void* data) const = 0;
177
178 virtual bool find_olap_engine(const void* data) const = 0;
179
180 virtual bool find_uint32_t(uint32_t data) const = 0;
181
182 virtual void insert_fixed_len(const char* data, const int* offsets, int number) = 0;
183
184 virtual void insert_fixed_len(const char* data) = 0;
185
186 virtual uint16_t find_fixed_len_olap_engine(const char* data, const uint8* nullmap,
187 uint16_t* offsets, int number,
188 bool is_parse_column) = 0;
189
190 virtual void find_fixed_len(const char* data, const uint8* nullmap, int number,
191 uint8* results) = 0;
192
193protected:
194 // bloom filter size
195 int32_t _bloom_filter_alloced;
196 std::shared_ptr<BloomFilterAdaptor> _bloom_filter;
197 bool _inited;
198 std::mutex _lock;
199 int64_t _bloom_filter_length;
200};

202template <class T>
203struct CommonFindOp {
204 // test_batch/find_batch/find_batch_olap_engine only used on vectorized engine
205 void insert_batch(BloomFilterAdaptor& bloom_filter, const char* data, const int* offsets,
206 int number) const {
207 for (int i = 0; i < number; i++) {
208 bloom_filter.add_element(*((T*)data + offsets[i]));
209 }
210 }
211
212 void insert_single(BloomFilterAdaptor& bloom_filter, const char* data) const {
213 bloom_filter.add_element(*((T*)data));
214 }
215
216 uint16_t find_batch_olap_engine(const BloomFilterAdaptor& bloom_filter, const char* data,
217 const uint8* nullmap, uint16_t* offsets, int number,
218 const bool is_parse_column) const {
219 uint16_t new_size = 0;
220 if (is_parse_column) {
221 if (nullmap == nullptr) {
222 for (int i = 0; i < number; i++) {
223 uint16_t idx = offsets[i];
224 if (!bloom_filter.test_element(*((T*)data + idx))) {
225 continue;
226 }
227 offsets[new_size++] = idx;
228 }
229 } else {
230 for (int i = 0; i < number; i++) {
231 uint16_t idx = offsets[i];
232 if (nullmap[idx]) {
233 continue;
234 }
235 if (!bloom_filter.test_element(*((T*)data + idx))) {
236 continue;
237 }
238 offsets[new_size++] = idx;
239 }
240 }
241 } else {
242 if (nullmap == nullptr) {
243 for (int i = 0; i < number; i++) {
244 if (!bloom_filter.test_element(*((T*)data + i))) {
245 continue;
246 }
247 offsets[new_size++] = i;
248 }
249 } else {
250 for (int i = 0; i < number; i++) {
251 if (nullmap[i]) {
252 continue;
253 }
254 if (!bloom_filter.test_element(*((T*)data + i))) {
255 continue;
256 }
257 offsets[new_size++] = i;
258 }
259 }
260 }
261 return new_size;
262 }
263
264 void find_batch(const BloomFilterAdaptor& bloom_filter, const char* data, const uint8* nullmap,
265 int number, uint8* results) const {
266 for (int i = 0; i < number; i++) {
267 results[i] = false;
268 if (nullmap != nullptr && nullmap[i]) {
269 continue;
270 }
271 if (!bloom_filter.test_element(*((T*)data + i))) {
272 continue;
273 }
274 results[i] = true;
275 }
276 }
277
278 void insert(BloomFilterAdaptor& bloom_filter, const void* data) const {
279 bloom_filter.add_bytes((char*)data, sizeof(T));
280 }
281 bool find(const BloomFilterAdaptor& bloom_filter, const void* data) const {
282 return bloom_filter.test(Slice((char*)data, sizeof(T)));
283 }
284 bool find_olap_engine(const BloomFilterAdaptor& bloom_filter, const void* data) const {
285 return find(bloom_filter, data);
286 }
287 bool find(const BloomFilterAdaptor& bloom_filter, uint32_t data) const {
288 return bloom_filter.test(data);
289 }
290};

292struct StringFindOp {
293 void insert_batch(BloomFilterAdaptor& bloom_filter, const char* data, const int* offsets,
294 int number) const {
295 LOG(FATAL) << "StringFindOp does not support insert_batch";
296 }
297
298 void insert_single(BloomFilterAdaptor& bloom_filter, const char* data) const {
299 LOG(FATAL) << "StringFindOp does not support insert_single";
300 }
301
302 uint16_t find_batch_olap_engine(const BloomFilterAdaptor& bloom_filter, const char* data,
303 const uint8* nullmap, uint16_t* offsets, int number,
304 const bool is_parse_column) const {
305 LOG(FATAL) << "StringFindOp does not support find_batch_olap_engine";
306 return 0;
307 }
308
309 void find_batch(const BloomFilterAdaptor& bloom_filter, const char* data, const uint8* nullmap,
310 int number, uint8* results) const {
311 LOG(FATAL) << "StringFindOp does not support find_batch";
312 }
313
314 void insert(BloomFilterAdaptor& bloom_filter, const void* data) const {
315 const auto* value = reinterpret_cast<const StringRef*>(data);
316 if (value) {
317 bloom_filter.add_bytes(value->data, value->size);
318 }
319 }
320 bool find(const BloomFilterAdaptor& bloom_filter, const void* data) const {
321 const auto* value = reinterpret_cast<const StringRef*>(data);
322 if (value == nullptr) {
323 return false;
324 }
325 return bloom_filter.test(Slice(value->data, value->size));
326 }
327 bool find_olap_engine(const BloomFilterAdaptor& bloom_filter, const void* data) const {
328 return StringFindOp::find(bloom_filter, data);
329 }
330 bool find(const BloomFilterAdaptor& bloom_filter, uint32_t data) const {
331 return bloom_filter.test(data);
332 }
333};

335// We do not need to judge whether data is empty, because null will not appear
336// when filer used by the storage engine
337struct FixedStringFindOp : public StringFindOp {
338 bool find_olap_engine(const BloomFilterAdaptor& bloom_filter, const void* input_data) const {
339 const auto* value = reinterpret_cast<const StringRef*>(input_data);
340 int64_t size = value->size;
341 const char* data = value->data;
342 while (size > 0 && data[size - 1] == '\0') {
343 size--;
344 }
345 return bloom_filter.test(Slice(value->data, size));
346 }
347};

358// https://gcc.gnu.org/bugzilla/show_bug.cgi?id=101684

360struct DateFindOp : public CommonFindOp<vectorized::VecDateTimeValue> {
361 bool find_olap_engine(const BloomFilterAdaptor& bloom_filter, const void* data) const {
362 uint24_t date = *static_cast<const uint24_t*>(data);
363 uint64_t value = uint32_t(date);
364
365 vectorized::VecDateTimeValue date_value;
366 date_value.from_olap_date(value);
367 // So confusing here. For join node with condition (a.date_col = b.date_col), the actual
368 // expression is CAST(a.date_col AS DATETIME) = CAST(b.date_col AS DATETIME). So we build
369 // this bloom filter by CAST(a.date_col AS DATETIME) and also need to probe this bloom
370 // filter by a datetime value.
371 date_value.set_type(TimeType::TIME_DATETIME);
372
373 return bloom_filter.test(Slice((char*)&date_value, sizeof(vectorized::VecDateTimeValue)));
374 }

377struct DecimalV2FindOp : public CommonFindOp<DecimalV2Value> {
378 bool find_olap_engine(const BloomFilterAdaptor& bloom_filter, const void* data) const {
379 auto packed_decimal = *static_cast<const decimal12_t*>(data);
380 DecimalV2Value value;
381 int64_t int_value = packed_decimal.integer;
382 int32_t frac_value = packed_decimal.fraction;
383 value.from_olap_decimal(int_value, frac_value);
384
385 constexpr int decimal_value_sz = sizeof(DecimalV2Value);
386 char data_bytes[decimal_value_sz];
387 memcpy(&data_bytes, &value, decimal_value_sz);
388 return bloom_filter.test(Slice(data_bytes, decimal_value_sz));
389 }
390};
