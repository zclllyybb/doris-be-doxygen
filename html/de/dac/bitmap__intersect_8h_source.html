154// Calculate the intersection of two or more bitmaps
155// Usage: intersect_count(bitmap_column_to_count, filter_column, filter_values ...)
156// Example: intersect_count(user_id, event, 'A', 'B', 'C'), meaning find the intersect count of user_id in all A/B/C 3 bitmaps
157// Todo(kks) Use Array type instead of variable arguments
158template <typename T>
159struct BitmapIntersect {
160public:
161 BitmapIntersect() = default;
162
163 explicit BitmapIntersect(const char* src) { deserialize(src); }
164
165 void add_key(const T key) {
166 BitmapValue empty_bitmap;
167 _bitmaps[key] = empty_bitmap;
168 }
169
170 void update(const T& key, const BitmapValue& bitmap) {
171 if (_bitmaps.find(key) != _bitmaps.end()) {
172 _bitmaps[key] |= bitmap;
173 }
174 }
175
176 void merge(const BitmapIntersect& other) {
177 for (auto& kv : other._bitmaps) {
178 if (_bitmaps.find(kv.first) != _bitmaps.end()) {
179 _bitmaps[kv.first] |= kv.second;
180 } else {
181 _bitmaps[kv.first] = kv.second;
182 }
183 }
184 }
185
186 // intersection
187 BitmapValue intersect() const {
188 BitmapValue result;
189 auto it = _bitmaps.begin();
190 result |= it->second;
191 it++;
192 for (; it != _bitmaps.end(); it++) {
193 result &= it->second;
194 }
195 return result;
196 }
197
198 // calculate the intersection for _bitmaps's bitmap values
199 int64_t intersect_count() const {
200 if (_bitmaps.empty()) {
201 return 0;
202 }
203 return intersect().cardinality();
204 }
205
206 // the serialize size
207 size_t size() {
208 size_t size = 4;
209 for (auto& kv : _bitmaps) {
210 size += detail::Helper::serialize_size(kv.first);
211 size += kv.second.getSizeInBytes();
212 }
213 return size;
214 }
215
216 //must call size() first
217 void serialize(char* dest) {
218 char* writer = dest;
219 *(int32_t*)writer = _bitmaps.size();
220 writer += 4;
221 for (auto& kv : _bitmaps) {
222 writer = detail::Helper::write_to(kv.first, writer);
223 kv.second.write(writer);
224 writer += kv.second.getSizeInBytes();
225 }
226 }
227
228 void deserialize(const char* src) {
229 const char* reader = src;
230 int32_t bitmaps_size = *(int32_t*)reader;
231 reader += 4;
232 for (int32_t i = 0; i < bitmaps_size; i++) {
233 T key;
234 detail::Helper::read_from(&reader, &key);
235 BitmapValue bitmap(reader);
236 reader += bitmap.getSizeInBytes();
237 _bitmaps[key] = bitmap;
238 }
239 }
240
241private:
242 std::map<T, BitmapValue> _bitmaps;
243};
