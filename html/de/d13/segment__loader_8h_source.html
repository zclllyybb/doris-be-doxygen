35// SegmentLoader is used to load the Segment of BetaRowset.
36// An LRUCache is encapsulated inside it, which is used to cache the opened segments.
37// The caller should use the following method to load and obtain
38// the segments of a specified rowset:
39//
40// SegmentCacheHandle cache_handle;
41// RETURN_NOT_OK(SegmentCache::instance()->load_segments(_rowset, &cache_handle));
42// for (auto& seg_ptr : cache_handle.value()->segments) {
43// ... visit segment ...
44// }
45//
46// Make sure that cache_handle is valid during the segment usage period.
47using BetaRowsetSharedPtr = std::shared_ptr<BetaRowset>;
48class SegmentLoader {
49public:
50 // The cache key or segment lru cache
51 struct CacheKey {
52 CacheKey(RowsetId rowset_id_) : rowset_id(rowset_id_) {}
53 RowsetId rowset_id;
54
55 // Encode to a flat binary which can be used as LRUCache's key
56 std::string encode() const { return rowset_id.to_string(); }
57 };
58
59 // The cache value of segment lru cache.
60 // Holding all opened segments of a rowset.
61 struct CacheValue {
62 // Save the last visit time of this cache entry.
63 // Use atomic because it may be modified by multi threads.
64 std::atomic<int64_t> last_visit_time = 0;
65 std::vector<segment_v2::SegmentSharedPtr> segments;
66 };
67
68 // Create global instance of this class.
69 // "capacity" is the capacity of lru cache.
70 // TODO: Currently we use the number of rowset as the cache capacity.
71 // That is, the limit of cache is the number of rowset.
72 // This is because currently we cannot accurately estimate the memory occupied by a segment.
73 // After the estimation of segment memory usage is provided later, it is recommended
74 // to use Memory as the capacity limit of the cache.
75 static void create_global_instance(size_t capacity);
76
77 // Return global instance.
78 // Client should call create_global_cache before.
79 static SegmentLoader* instance() { return _s_instance; }
80
81 SegmentLoader(size_t capacity);
82
83 // Load segments of "rowset", return the "cache_handle" which contains segments.
84 // If use_cache is true, it will be loaded from _cache.
85 Status load_segments(const BetaRowsetSharedPtr& rowset, SegmentCacheHandle* cache_handle,
86 bool use_cache = false);
87
88 // Try to prune the segment cache if expired.
89 Status prune();
90
91private:
92 SegmentLoader();
93
94 // Lookup the given rowset in the cache.
95 // If the rowset is found, the cache entry will be written into handle.
96 // Return true if entry is found, otherwise return false.
97 bool _lookup(const SegmentLoader::CacheKey& key, SegmentCacheHandle* handle);
98
99 // Insert a cache entry by key.
100 // And the cache entry will be returned in handle.
101 // This function is thread-safe.
102 void _insert(const SegmentLoader::CacheKey& key, CacheValue& value, SegmentCacheHandle* handle);
103
104private:
105 static SegmentLoader* _s_instance;
106 // A LRU cache to cache all opened segments
107 std::unique_ptr<Cache> _cache = nullptr;
108};
109
110// A handle for a single rowset from segment lru cache.
111// The handle can ensure that the segment is valid
112// and will not be closed while the holder of the handle is accessing the segment.
113// The handle will automatically release the cache entry when it is destroyed.
114// So the caller need to make sure the handle is valid in lifecycle.
115class SegmentCacheHandle {
116public:
117 SegmentCacheHandle() {}
118 SegmentCacheHandle(Cache* cache, Cache::Handle* handle) : _cache(cache), _handle(handle) {}
119
120 ~SegmentCacheHandle() {
121 if (_handle != nullptr) {
122 CHECK(_cache != nullptr);
123 CHECK(segments.empty()) << segments.size();
124 CHECK(!owned);
125 // last_visit_time is set when release.
126 // because it only be needed when pruning.
127 ((SegmentLoader::CacheValue*)_cache->value(_handle))->last_visit_time = UnixMillis();
128 _cache->release(_handle);
129 }
130 }
131
132 SegmentCacheHandle(SegmentCacheHandle&& other) noexcept {
133 std::swap(_cache, other._cache);
134 std::swap(_handle, other._handle);
135 this->owned = other.owned;
136 this->segments = std::move(other.segments);
137 }
138
139 SegmentCacheHandle& operator=(SegmentCacheHandle&& other) noexcept {
140 std::swap(_cache, other._cache);
141 std::swap(_handle, other._handle);
142 this->owned = other.owned;
143 this->segments = std::move(other.segments);
144 return *this;
145 }
146
147 std::vector<segment_v2::SegmentSharedPtr>& get_segments() {
148 if (owned) {
149 return segments;
150 } else {
151 return ((SegmentLoader::CacheValue*)_cache->value(_handle))->segments;
152 }
153 }
154
155public:
156 // If set to true, the loaded segments will be saved in segments, not in lru cache;
157 bool owned = false;
158 std::vector<segment_v2::SegmentSharedPtr> segments;
159
160private:
161 Cache* _cache = nullptr;
162 Cache::Handle* _handle = nullptr;
163
164 // Don't allow copy and assign
165 DISALLOW_COPY_AND_ASSIGN(SegmentCacheHandle);
166};
