29template <class T, class Tag>
30struct StrongTypedef : boost::totally_ordered1<StrongTypedef<T, Tag>,
31 boost::totally_ordered2<StrongTypedef<T, Tag>, T>> {
32private:
33 using Self = StrongTypedef;
34 T t;
35
36public:
37 template <class Enable = typename std::is_copy_constructible<T>::type>
38 explicit StrongTypedef(const T& t_) : t(t_) {}
39 template <class Enable = typename std::is_move_constructible<T>::type>
40 explicit StrongTypedef(T&& t_) : t(std::move(t_)) {}
41
42 template <class Enable = typename std::is_default_constructible<T>::type>
43 StrongTypedef() : t() {}
44
45 StrongTypedef(const Self&) = default;
46 StrongTypedef(Self&&) = default;
47
48 Self& operator=(const Self&) = default;
49 Self& operator=(Self&&) = default;
50
51 template <class Enable = typename std::is_copy_assignable<T>::type>
52 Self& operator=(const T& rhs) {
53 t = rhs;
54 return *this;
55 }
56
57 template <class Enable = typename std::is_move_assignable<T>::type>
58 Self& operator=(T&& rhs) {
59 t = std::move(rhs);
60 return *this;
61 }
62
63 operator const T&() const { return t; }
64 operator T&() { return t; }
65
66 bool operator==(const Self& rhs) const { return t == rhs.t; }
67 bool operator<(const Self& rhs) const { return t < rhs.t; }
68
69 T& to_under_type() { return t; }
70 const T& to_under_type() const { return t; }
71};
