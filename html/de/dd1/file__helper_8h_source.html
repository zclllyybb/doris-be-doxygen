186template <typename MessageType, typename ExtraType = uint32_t,
187 typename FileHandlerType = FileHandler>
188class FileHeader {
189public:
190 FileHeader() {
191 memset(&_fixed_file_header, 0, sizeof(_fixed_file_header));
192 memset(&_extra_fixed_header, 0, sizeof(_extra_fixed_header));
193 _fixed_file_header_size = sizeof(_fixed_file_header);
194 }
195 ~FileHeader() {}
196
197 // To calculate the length of the proto part, it needs to be called after the proto is operated, and prepare must be called before calling serialize
198 Status prepare(FileHandlerType* file_handler);
199
200 // call prepare() first, serialize() will write fixed header and protobuffer.
201 // Write the header to the starting position of the incoming file handle
202 Status serialize(FileHandlerType* file_handler);
203
204 // read from file, validate file length, signature and alder32 of protobuffer.
205 // Read the header from the beginning of the incoming file handle
206 Status unserialize(FileHandlerType* file_handler);
207
208 // Check the validity of Header
209 // it is actually call unserialize().
210 Status validate(const std::string& filename);
211
212 uint64_t file_length() const { return _fixed_file_header.file_length; }
213 uint32_t checksum() const { return _fixed_file_header.checksum; }
214 const ExtraType& extra() const { return _extra_fixed_header; }
215 ExtraType* mutable_extra() { return &_extra_fixed_header; }
216 const MessageType& message() const { return _proto; }
217 MessageType* mutable_message() { return &_proto; }
218 uint64_t size() const {
219 return _fixed_file_header_size + sizeof(_extra_fixed_header) +
220 _fixed_file_header.protobuf_length;
221 }
222
223 void set_file_length(uint64_t file_length) { _fixed_file_header.file_length = file_length; }
224 void set_checksum(uint32_t checksum) { _fixed_file_header.checksum = checksum; }
225
226private:
227 FixedFileHeaderV2 _fixed_file_header;
228 uint32_t _fixed_file_header_size;
229
230 std::string _proto_string;
231 ExtraType _extra_fixed_header;
232 MessageType _proto;
233};

236template <typename MessageType, typename ExtraType, typename FileHandlerType>
237Status FileHeader<MessageType, ExtraType, FileHandlerType>::prepare(FileHandlerType* file_handler) {
238 if (nullptr == file_handler) {
239 return Status::Error<ErrorCode::INVALID_ARGUMENT>();
240 }
241
242 // Use the file name as Signature to prevent problems caused by some misoperations
243 // _proto.set_signature(basename(file_handler->file_name().c_str()));
244
245 try {
246 if (!_proto.SerializeToString(&_proto_string)) {
247 LOG(WARNING) << "serialize file header to string error. [path='"
248 << file_handler->file_name() << "']";
249 return Status::Error<ErrorCode::SERIALIZE_PROTOBUF_ERROR>();
250 }
251 } catch (...) {
252 LOG(WARNING) << "serialize file header to string error. [path='"
253 << file_handler->file_name() << "']";
254 return Status::Error<ErrorCode::SERIALIZE_PROTOBUF_ERROR>();
255 }
256
257 _fixed_file_header.protobuf_checksum =
258 olap_adler32(ADLER32_INIT, _proto_string.c_str(), _proto_string.size());
259
260 _fixed_file_header.checksum = 0;
261 _fixed_file_header.protobuf_length = _proto_string.size();
262 _fixed_file_header.file_length = size();
263 _fixed_file_header.version = OLAP_DATA_VERSION_APPLIED;
264 _fixed_file_header.magic_number = OLAP_FIX_HEADER_MAGIC_NUMBER;
265
266 return Status::OK();
267}

269template <typename MessageType, typename ExtraType, typename FileHandlerType>
270Status FileHeader<MessageType, ExtraType, FileHandlerType>::serialize(
271 FileHandlerType* file_handler) {
272 if (nullptr == file_handler) {
273 return Status::Error<ErrorCode::INVALID_ARGUMENT>();
274 }
275
276 // write to file
277 if (!file_handler->pwrite(&_fixed_file_header, _fixed_file_header_size, 0)) {
278 char errmsg[64];
279 LOG(WARNING) << "fail to write fixed header to file. [file='" << file_handler->file_name()
280 << "' err=" << strerror_r(errno, errmsg, 64) << "]";
281 return Status::Error<ErrorCode::IO_ERROR>();
282 }
283
284 if (!file_handler->pwrite(&_extra_fixed_header, sizeof(_extra_fixed_header),
285 _fixed_file_header_size)) {
286 char errmsg[64];
287 LOG(WARNING) << "fail to write extra fixed header to file. [file='"
288 << file_handler->file_name() << "' err=" << strerror_r(errno, errmsg, 64)
289 << "]";
290 return Status::Error<ErrorCode::IO_ERROR>();
291 }
292
293 if (!file_handler->pwrite(_proto_string.c_str(), _proto_string.size(),
294 _fixed_file_header_size + sizeof(_extra_fixed_header))) {
295 char errmsg[64];
296 LOG(WARNING) << "fail to write proto header to file. [file='" << file_handler->file_name()
297 << "' err='" << strerror_r(errno, errmsg, 64) << "']";
298 return Status::Error<ErrorCode::IO_ERROR>();
299 }
300
301 return Status::OK();
302}

304template <typename MessageType, typename ExtraType, typename FileHandlerType>
305Status FileHeader<MessageType, ExtraType, FileHandlerType>::unserialize(
306 FileHandlerType* file_handler) {
307 if (nullptr == file_handler) {
308 return Status::Error<ErrorCode::INVALID_ARGUMENT>();
309 }
310
311 off_t real_file_length = 0;
312 uint32_t real_protobuf_checksum = 0;
313
314 if (!file_handler->pread(&_fixed_file_header, _fixed_file_header_size, 0)) {
315 char errmsg[64];
316 LOG(WARNING) << "fail to load header structure from file. file="
317 << file_handler->file_name() << ", error=" << strerror_r(errno, errmsg, 64);
318 return Status::Error<ErrorCode::IO_ERROR>();
319 }
320
321 if (_fixed_file_header.magic_number != OLAP_FIX_HEADER_MAGIC_NUMBER) {
322 VLOG_TRACE << "old fix header found, magic num=" << _fixed_file_header.magic_number;
323 FixedFileHeader tmp_header;
324
325 if (!file_handler->pread(&tmp_header, sizeof(tmp_header), 0)) {
326 char errmsg[64];
327 LOG(WARNING) << "fail to load header structure from file. file="
328 << file_handler->file_name()
329 << ", error=" << strerror_r(errno, errmsg, 64);
330 return Status::Error<ErrorCode::IO_ERROR>();
331 }
332
333 _fixed_file_header.file_length = tmp_header.file_length;
334 _fixed_file_header.checksum = tmp_header.checksum;
335 _fixed_file_header.protobuf_length = tmp_header.protobuf_length;
336 _fixed_file_header.protobuf_checksum = tmp_header.protobuf_checksum;
337 _fixed_file_header.magic_number = OLAP_FIX_HEADER_MAGIC_NUMBER;
338 _fixed_file_header.version = OLAP_DATA_VERSION_APPLIED;
339
340 _fixed_file_header_size = sizeof(tmp_header);
341 }
342
343 VLOG_NOTICE << "fix head loaded. file_length=" << _fixed_file_header.file_length
344 << ", checksum=" << _fixed_file_header.checksum
345 << ", protobuf_length=" << _fixed_file_header.protobuf_length
346 << ", magic_number=" << _fixed_file_header.magic_number
347 << ", version=" << _fixed_file_header.version;
348
349 if (!file_handler->pread(&_extra_fixed_header, sizeof(_extra_fixed_header),
350 _fixed_file_header_size)) {
351 char errmsg[64];
352 LOG(WARNING) << "fail to load extra fixed header from file. file="
353 << file_handler->file_name() << ", error=" << strerror_r(errno, errmsg, 64);
354 return Status::Error<ErrorCode::IO_ERROR>();
355 }
356
357 std::unique_ptr<char[]> buf(new (std::nothrow) char[_fixed_file_header.protobuf_length]);
358
359 if (nullptr == buf.get()) {
360 char errmsg[64];
361 LOG(WARNING) << "malloc protobuf buf error. file=" << file_handler->file_name()
362 << ", error=" << strerror_r(errno, errmsg, 64);
363 return Status::Error<ErrorCode::MEM_ALLOC_FAILED>();
364 }
365
366 if (!file_handler->pread(buf.get(), _fixed_file_header.protobuf_length,
367 _fixed_file_header_size + sizeof(_extra_fixed_header))) {
368 char errmsg[64];
369 LOG(WARNING) << "fail to load protobuf from file. file=" << file_handler->file_name()
370 << ", error=" << strerror_r(errno, errmsg, 64);
371 return Status::Error<ErrorCode::IO_ERROR>();
372 }
373
374 real_file_length = file_handler->length();
375
376 if (file_length() != static_cast<uint64_t>(real_file_length)) {
377 LOG(WARNING) << "file length is not match. file=" << file_handler->file_name()
378 << ", file_length=" << file_length()
379 << ", real_file_length=" << real_file_length;
380 return Status::Error<ErrorCode::FILE_DATA_ERROR>();
381 }
382
383 // check proto checksum
384 real_protobuf_checksum =
385 olap_adler32(ADLER32_INIT, buf.get(), _fixed_file_header.protobuf_length);
386
387 if (real_protobuf_checksum != _fixed_file_header.protobuf_checksum) {
388 LOG(WARNING) << "checksum is not match. file=" << file_handler->file_name()
389 << ", expect=" << _fixed_file_header.protobuf_checksum
390 << ", actual=" << real_protobuf_checksum;
391 return Status::Error<ErrorCode::CHECKSUM_ERROR>();
392 }
393
394 try {
395 std::string protobuf_str(buf.get(), _fixed_file_header.protobuf_length);
396
397 if (!_proto.ParseFromString(protobuf_str)) {
398 LOG(WARNING) << "fail to parse file content to protobuf object. file="
399 << file_handler->file_name();
400 return Status::Error<ErrorCode::PARSE_PROTOBUF_ERROR>();
401 }
402 } catch (...) {
403 LOG(WARNING) << "fail to load protobuf. file='" << file_handler->file_name();
404 return Status::Error<ErrorCode::PARSE_PROTOBUF_ERROR>();
405 }
406
407 return Status::OK();
408}

410template <typename MessageType, typename ExtraType, typename FileHandlerType>
411Status FileHeader<MessageType, ExtraType, FileHandlerType>::validate(const std::string& filename) {
412 FileHandler file_handler;
413 Status res = Status::OK();
414
415 if (!file_handler.open(filename.c_str(), O_RDONLY)) {
416 char errmsg[64];
417 LOG(WARNING) << "fail to open file. [file='" << filename
418 << "' err=" << strerror_r(errno, errmsg, 64) << "]";
419 return Status::Error<ErrorCode::IO_ERROR>();
420 }
421
422 if (!(res = unserialize(&file_handler))) {
423 LOG(WARNING) << "unserialize file header error. [file='" << filename << "']";
424 return res;
425 }
426
427 return Status::OK();
428}
