30// VSortedRunMerger is used to merge multiple sorted runs of blocks. A run is a sorted
31// sequence of blocks, which are fetched from a BlockSupplier function object.
32// Merging is implemented using a binary min-heap that maintains the run with the next
33// rows in sorted order at the top of the heap.
34//
35// Merged block of rows are retrieved from VSortedRunMerger via calls to get_next().
36class VSortedRunMerger {
37public:
38 // Function that returns the next block of rows from an input sorted run. The batch
39 // is owned by the supplier (i.e. not VSortedRunMerger). eos is indicated by an NULL
40 // batch being returned.
41 VSortedRunMerger(const std::vector<VExprContext*>& ordering_expr,
42 const std::vector<bool>& _is_asc_order, const std::vector<bool>& _nulls_first,
43 const size_t batch_size, int64_t limit, size_t offset,
44 RuntimeProfile* profile);
45
46 VSortedRunMerger(const SortDescription& desc, const size_t batch_size, int64_t limit,
47 size_t offset, RuntimeProfile* profile);
48
49 virtual ~VSortedRunMerger() = default;
50
51 // Prepare this merger to merge and return rows from the sorted runs in 'input_runs'.
52 // Retrieves the first batch from each run and sets up the binary heap implementing
53 // the priority queue.
54 Status prepare(const std::vector<BlockSupplier>& input_runs);
55
56 // Return the next block of sorted rows from this merger.
57 Status get_next(Block* output_block, bool* eos);
58
59protected:
60 const std::vector<VExprContext*> _ordering_expr;
61 SortDescription _desc;
62 const std::vector<bool> _is_asc_order;
63 const std::vector<bool> _nulls_first;
64 const size_t _batch_size;
65
66 bool _use_sort_desc = false;
67 size_t _num_rows_returned = 0;
68 int64_t _limit = -1;
69 size_t _offset = 0;
70
71 std::vector<BlockSupplierSortCursorImpl> _cursors;
72 std::priority_queue<MergeSortCursor> _priority_queue;
73
74 Block _empty_block;
75
76 // Times calls to get_next().
77 RuntimeProfile::Counter* _get_next_timer;
78
79 // Times calls to get the next batch of rows from the input run.
80 RuntimeProfile::Counter* _get_next_block_timer;
81
82private:
83 void init_timers(RuntimeProfile* profile);
84 void next_heap(MergeSortCursor& current);
85 bool has_next_block(MergeSortCursor& current);
86};
