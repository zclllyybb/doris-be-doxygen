42// Ordinal index is implemented by one IndexPage that stores the first value ordinal
43// and file pointer for each data page.
44// But if there is only one data page, there is no need for index page. So we store
45// the file pointer to that data page directly in index meta (OrdinalIndexPB).
46class OrdinalIndexWriter {
47public:
48 OrdinalIndexWriter() : _page_builder(new IndexPageBuilder(0, true)) {}
49
50 void append_entry(ordinal_t ordinal, const PagePointer& data_pp);
51
52 uint64_t size() { return _page_builder->size(); }
53
54 Status finish(io::FileWriter* file_writer, ColumnIndexMetaPB* meta);
55
56private:
57 DISALLOW_COPY_AND_ASSIGN(OrdinalIndexWriter);
58 std::unique_ptr<IndexPageBuilder> _page_builder;
59 PagePointer _last_pp;
60};

64class OrdinalIndexReader {
65public:
66 explicit OrdinalIndexReader(io::FileReaderSPtr file_reader, const OrdinalIndexPB* index_meta,
67 ordinal_t num_values)
68 : _file_reader(std::move(file_reader)),
69 _index_meta(index_meta),
70 _num_values(num_values) {}
71
72 // load and parse the index page into memory
73 Status load(bool use_page_cache, bool kept_in_memory);
74
75 // the returned iter points to the largest element which is less than `ordinal`,
76 // or points to the first element if all elements are greater than `ordinal`,
77 // or points to "end" if all elements are smaller than `ordinal`.
78 OrdinalPageIndexIterator seek_at_or_before(ordinal_t ordinal);
79 inline OrdinalPageIndexIterator begin();
80 inline OrdinalPageIndexIterator end();
81 ordinal_t get_first_ordinal(int page_index) const { return _ordinals[page_index]; }
82
83 ordinal_t get_last_ordinal(int page_index) const {
84 return get_first_ordinal(page_index + 1) - 1;
85 }
86
87 // for test
88 int32_t num_data_pages() const { return _num_pages; }
89
90private:
91 friend OrdinalPageIndexIterator;
92
93 io::FileReaderSPtr _file_reader;
94 const OrdinalIndexPB* _index_meta;
95 // total number of values (including NULLs) in the indexed column,
96 // equals to 1 + 'last ordinal of last data pages'
97 ordinal_t _num_values;
98
99 // valid after load
100 int _num_pages = 0;
101 // _ordinals[i] = first ordinal of the i-th data page,
102 std::vector<ordinal_t> _ordinals;
103 // _pages[i] = page pointer to the i-th data page
104 std::vector<PagePointer> _pages;
105};

107class OrdinalPageIndexIterator {
108public:
109 OrdinalPageIndexIterator() : _index(nullptr), _cur_idx(-1) {}
110 OrdinalPageIndexIterator(OrdinalIndexReader* index) : _index(index), _cur_idx(0) {}
111 OrdinalPageIndexIterator(OrdinalIndexReader* index, int cur_idx)
112 : _index(index), _cur_idx(cur_idx) {}
113 bool valid() const { return _cur_idx < _index->_num_pages; }
114 void next() {
115 DCHECK_LT(_cur_idx, _index->_num_pages);
116 _cur_idx++;
117 }
118 int32_t page_index() const { return _cur_idx; }
119 const PagePointer& page() const { return _index->_pages[_cur_idx]; }
120 ordinal_t first_ordinal() const { return _index->get_first_ordinal(_cur_idx); }
121 ordinal_t last_ordinal() const { return _index->get_last_ordinal(_cur_idx); }
122
123private:
124 OrdinalIndexReader* _index;
125 int32_t _cur_idx;
126};
