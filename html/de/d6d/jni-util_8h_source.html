136template <class T>
137Status SerializeThriftMsg(JNIEnv* env, T* msg, jbyteArray* serialized_msg) {
138 int buffer_size = 100 * 1024; // start out with 100KB
139 ThriftSerializer serializer(false, buffer_size);
140
141 uint8_t* buffer = NULL;
142 uint32_t size = 0;
143 RETURN_IF_ERROR(serializer.serialize(msg, &size, &buffer));
144
145 // Make sure that 'size' is within the limit of INT_MAX as the use of
146 // 'size' below takes int.
147 if (size > INT_MAX) {
148 return Status::InternalError(
149 "The length of the serialization buffer ({} bytes) exceeds the limit of {} bytes",
150 size, INT_MAX);
151 }
152
154 *serialized_msg = env->NewByteArray(size);
155 RETURN_ERROR_IF_EXC(env);
156 if (*serialized_msg == NULL) return Status::InternalError("couldn't construct jbyteArray");
157 env->SetByteArrayRegion(*serialized_msg, 0, size, reinterpret_cast<jbyte*>(buffer));
158 RETURN_ERROR_IF_EXC(env);
159 return Status::OK();
160}
