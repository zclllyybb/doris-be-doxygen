35template <typename T, int MAX_SAMPLES>
36class StreamingSampler {
37public:
38 StreamingSampler(int initial_period = 500)
39 : samples_collected_(0),
40 period_(initial_period),
41 current_sample_sum_(0),
42 current_sample_count_(0),
43 current_sample_total_time_(0) {}
44
46 StreamingSampler(int period, const std::vector<T>& initial_samples)
47 : samples_collected_(initial_samples.size()),
48 period_(period),
49 current_sample_sum_(0),
50 current_sample_count_(0),
51 current_sample_total_time_(0) {
52 DCHECK_LE(samples_collected_, MAX_SAMPLES);
53 memcpy(samples_, &initial_samples[0], sizeof(T) * samples_collected_);
54 }
55
64 void AddSample(T sample, int ms) {
65 std::lock_guard<SpinLock> l(lock_);
66 ++current_sample_count_;
67 current_sample_sum_ += sample;
68 current_sample_total_time_ += ms;
69
70 if (current_sample_total_time_ >= period_) {
71 samples_[samples_collected_++] = current_sample_sum_ / current_sample_count_;
72 current_sample_count_ = 0;
73 current_sample_sum_ = 0;
74 current_sample_total_time_ = 0;
75
76 if (samples_collected_ == MAX_SAMPLES) {
78 period_ *= 2;
79 for (int i = 0; i < MAX_SAMPLES / 2; ++i) {
80 samples_[i] = (samples_[i * 2] + samples_[i * 2 + 1]) / 2;
81 }
82 samples_collected_ /= 2;
83 }
84 }
85 }

91 const T* GetSamples(int* num_samples, int* period, SpinLock** lock = nullptr) const {
92 if (lock != nullptr) {
93 lock_.lock();
94 *lock = &lock_;
95 }
96 *num_samples = samples_collected_;
97 *period = period_;
98 return samples_;
99 }
100
102 void SetSamples(int period, const std::vector<T>& samples) {
103 DCHECK_LE(samples.size(), MAX_SAMPLES);
104
105 std::lock_guard<SpinLock> l(lock_);
106 period_ = period;
107 samples_collected_ = samples.size();
108 memcpy(samples_, &samples[0], sizeof(T) * samples_collected_);
109 current_sample_sum_ = 0;
110 current_sample_count_ = 0;
111 current_sample_total_time_ = 0;
112 }
113
114 std::string DebugString(const std::string& prefix = "") const {
115 std::lock_guard<SpinLock> l(lock_);
116 std::stringstream ss;
117 ss << prefix << "Period = " << period_ << std::endl
118 << prefix << "Num = " << samples_collected_ << std::endl
119 << prefix << "Samples = {";
120 for (int i = 0; i < samples_collected_; ++i) {
121 ss << samples_[i] << ", ";
122 }
123 ss << prefix << "}" << std::endl;
124 return ss.str();
125 }
