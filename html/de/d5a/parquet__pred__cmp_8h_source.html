27#define _PLAIN_DECODE(T, value, min_bytes, max_bytes, out_value, out_min, out_max) \
28 const T out_min = reinterpret_cast<const T*>(min_bytes)[0]; \
29 const T out_max = reinterpret_cast<const T*>(max_bytes)[0]; \
30 T out_value = *((T*)value);
31
32#define _PLAIN_DECODE_SINGLE(T, value, bytes, conjunct_value, out) \
33 const T out = reinterpret_cast<const T*>(bytes)[0]; \
34 T conjunct_value = *((T*)value);
35
36#define _FILTER_GROUP_BY_EQ_PRED(conjunct_value, min, max) \
37 if (conjunct_value < min || conjunct_value > max) { \
38 return true; \
39 }
40
41#define _FILTER_GROUP_BY_IN(T, in_pred_values, min_bytes, max_bytes) \
42 std::vector<T> in_values; \
43 for (auto val : in_pred_values) { \
44 T value = reinterpret_cast<T*>(val)[0]; \
45 in_values.emplace_back(value); \
46 } \
47 if (in_values.empty()) { \
48 return false; \
49 } \
50 auto result = std::minmax_element(in_values.begin(), in_values.end()); \
51 T in_min = *result.first; \
52 T in_max = *result.second; \
53 const T group_min = reinterpret_cast<const T*>(min_bytes)[0]; \
54 const T group_max = reinterpret_cast<const T*>(max_bytes)[0]; \
55 if (in_max < group_min || in_min > group_max) { \
56 return true; \
57 }

73template <typename T>
74static void _align_decimal_v2_scale(T* conjunct_value, int32_t value_scale, T* parquet_value,
75 int32_t parquet_scale) {
76 if (value_scale > parquet_scale) {
77 *parquet_value = *parquet_value * common::exp10_i32(value_scale - parquet_scale);
78 } else if (value_scale < parquet_scale) {
79 *conjunct_value = *conjunct_value * common::exp10_i32(parquet_scale - value_scale);
80 }
81}

103static bool _eval_in_val(const ColumnMinMaxParams& params) {
104 switch (params.conjunct_type) {
105 case TYPE_TINYINT: {
106 _FILTER_GROUP_BY_IN(int8_t, params.in_pred_values, params.min_bytes, params.max_bytes)
107 break;
108 }
109 case TYPE_SMALLINT: {
110 _FILTER_GROUP_BY_IN(int16_t, params.in_pred_values, params.min_bytes, params.max_bytes)
111 break;
112 }
113 case TYPE_DECIMAL32:
114 case TYPE_INT: {
115 _FILTER_GROUP_BY_IN(int32_t, params.in_pred_values, params.min_bytes, params.max_bytes)
116 break;
117 }
118 case TYPE_DECIMAL64:
119 case TYPE_BIGINT: {
120 _FILTER_GROUP_BY_IN(int64_t, params.in_pred_values, params.min_bytes, params.max_bytes)
121 break;
122 }
123 case TYPE_DECIMALV2: {
124 break;
125 }
126 case TYPE_STRING:
127 case TYPE_VARCHAR:
128 case TYPE_CHAR: {
129 std::vector<std::string> in_values;
130 for (auto val : params.in_pred_values) {
131 in_values.emplace_back(((StringRef*)val)->to_string());
132 }
133 if (in_values.empty()) {
134 return false;
135 }
136 auto result = std::minmax_element(in_values.begin(), in_values.end());
137 std::string& in_min = *result.first;
138 std::string& in_max = *result.second;
139 if (strcmp(in_max.data(), params.min_bytes) < 0 ||
140 strcmp(in_min.data(), params.max_bytes) > 0) {
141 return true;
142 }
143 break;
144 }
145 default:
146 return false;
147 }
148 return false;
149}

151static bool _eval_eq(const ColumnMinMaxParams& params) {
152 switch (params.conjunct_type) {
153 case TYPE_TINYINT: {
154 _PLAIN_DECODE(int16_t, params.value, params.min_bytes, params.max_bytes, conjunct_value,
155 min, max)
156 _FILTER_GROUP_BY_EQ_PRED(conjunct_value, min, max)
157 break;
158 }
159 case TYPE_SMALLINT: {
160 _PLAIN_DECODE(int16_t, params.value, params.min_bytes, params.max_bytes, conjunct_value,
161 min, max)
162 _FILTER_GROUP_BY_EQ_PRED(conjunct_value, min, max)
163 break;
164 }
165 case TYPE_DECIMAL32:
166 case TYPE_INT: {
167 _PLAIN_DECODE(int32_t, params.value, params.min_bytes, params.max_bytes, conjunct_value,
168 min, max)
169 _FILTER_GROUP_BY_EQ_PRED(conjunct_value, min, max)
170 break;
171 }
172 case TYPE_DECIMAL64:
173 case TYPE_BIGINT: {
174 _PLAIN_DECODE(int64_t, params.value, params.min_bytes, params.max_bytes, conjunct_value,
175 min, max)
176 _FILTER_GROUP_BY_EQ_PRED(conjunct_value, min, max)
177 break;
178 }
179 case TYPE_DECIMALV2: {
180 if (params.parquet_type == tparquet::Type::INT32) {
181 int32_t min_value = reinterpret_cast<const int32_t*>(params.min_bytes)[0];
182 int32_t max_value = reinterpret_cast<const int32_t*>(params.max_bytes)[0];
183 DecimalV2Value conjunct_value = *((DecimalV2Value*)params.value);

195 _align_decimal_v2_scale(&conjunct_int_value, conjunct_value.scale(), &min_value,
196 params.parquet_scale);
197 _align_decimal_v2_scale(&conjunct_int_value, conjunct_value.scale(), &max_value,
198 params.parquet_scale);
199 _FILTER_GROUP_BY_EQ_PRED(conjunct_int_value, min_value, max_value)
200 }
201 break;

215 }
216 case TYPE_STRING:
217 case TYPE_VARCHAR:
218 case TYPE_CHAR: {
219 std::string conjunct_value = ((StringRef*)params.value)->to_string();

252 case TYPE_SMALLINT: {

254 return _filter_group_by_gt_or_ge(conjunct_value, max, is_eq);

257 case TYPE_INT: {

413template <PrimitiveType primitive_type>
414static void to_filter(const ColumnValueRange<primitive_type>& col_val_range,
415 std::vector<ScanPredicate>& filters) {
416 using CppType = typename PrimitiveTypeTraits<primitive_type>::CppType;
417 const auto& high_value = col_val_range.get_range_max_value();
418 const auto& low_value = col_val_range.get_range_min_value();
419 const auto& high_op = col_val_range.get_range_high_op();
420 const auto& low_op = col_val_range.get_range_low_op();
421
422 // todo: process equals
423 if (col_val_range.is_fixed_value_range()) {
424 // 1. convert to in filter condition
425 ScanPredicate condition;
426 condition._col_name = col_val_range.column_name();
427 condition._op = TExprOpcode::FILTER_NEW_IN;
428 condition._scale = col_val_range.scale();
429 if (col_val_range.get_fixed_value_set().empty()) {
430 return;
431 }
432 for (const auto& value : col_val_range.get_fixed_value_set()) {
433 condition._values.push_back(const_cast<CppType*>(&value));
434 }
435 filters.push_back(condition);
436 } else if (low_value < high_value) {
437 // 2. convert to min max filter condition
438 ScanPredicate null_pred;
439 if (col_val_range.is_high_value_maximum() && high_op == SQLFilterOp::FILTER_LESS_OR_EQUAL &&
440 col_val_range.is_low_value_mininum() && low_op == SQLFilterOp::FILTER_LARGER_OR_EQUAL &&
441 !col_val_range.contain_null()) {
442 null_pred._col_name = col_val_range.column_name();
443 null_pred._null_op = true;
444 null_pred._is_null = false;
445 filters.push_back(null_pred);
446 return;
447 }
448 ScanPredicate low;
449 if (!col_val_range.is_low_value_mininum() ||
450 SQLFilterOp::FILTER_LARGER_OR_EQUAL != low_op) {
451 low._col_name = col_val_range.column_name();
452 low._op = (low_op == SQLFilterOp::FILTER_LARGER_OR_EQUAL ? TExprOpcode::GE
453 : TExprOpcode::GT);
454 // NOTICE: use get_range_min_value_ptr, not "low_value"'s addr,
455 // to avoid stack-use-after-return bug
456 low._values.push_back(const_cast<CppType*>(col_val_range.get_range_min_value_ptr()));
457 low._scale = col_val_range.scale();
458 filters.push_back(low);
459 }
460
461 ScanPredicate high;
462 if (!col_val_range.is_high_value_maximum() ||
463 SQLFilterOp::FILTER_LESS_OR_EQUAL != high_op) {
464 high._col_name = col_val_range.column_name();
465 high._op = (high_op == SQLFilterOp::FILTER_LESS_OR_EQUAL ? TExprOpcode::LE
466 : TExprOpcode::LT);
467 // NOTICE: use get_range_max_value_ptr, not "high_value"'s addr,
468 // to avoid stack-use-after-return bug
469 high._values.push_back(const_cast<CppType*>(col_val_range.get_range_max_value_ptr()));
470 high._scale = col_val_range.scale();
471 filters.push_back(high);
472 }
473 } else {
474 // 3. convert to is null and is not null filter condition
475 ScanPredicate null_pred;
476 if (col_val_range.is_low_value_maximum() && col_val_range.is_high_value_mininum() &&
477 col_val_range.contain_null()) {
478 null_pred._col_name = col_val_range.column_name();
479 null_pred._null_op = true;
480 null_pred._is_null = true;
481 filters.push_back(null_pred);
482 }
483 }
484}

486static void _eval_predicate(const ScanPredicate& filter, ColumnMinMaxParams* params,
487 bool* need_filter) {
488 if (filter._values.empty()) {
489 return;
490 }
491 if (filter._op == TExprOpcode::FILTER_NEW_IN) {
492 if (filter._values.size() == 1) {
493 params->value = filter._values[0];
494 *need_filter = _eval_eq(*params);
495 return;
496 }
497 params->in_pred_values = filter._values;
498 *need_filter = _eval_in_val(*params);
499 return;
500 }
501 // preserve TExprOpcode::FILTER_NEW_NOT_IN
502 params->value = filter._values[0];
503 switch (filter._op) {
504 case TExprOpcode::EQ:
505 *need_filter = _eval_eq(*params);
506 break;
507 case TExprOpcode::NE:
508 break;
509 case TExprOpcode::GT:
510 *need_filter = _eval_gt(*params, false);
511 break;
512 case TExprOpcode::GE:
513 *need_filter = _eval_gt(*params, true);
514 break;
515 case TExprOpcode::LT:
516 *need_filter = _eval_lt(*params, false);
517 break;
518 case TExprOpcode::LE:
519 *need_filter = _eval_lt(*params, true);
520 break;
521 default:
522 break;
523 }
524}

526static bool determine_filter_min_max(const ColumnValueRangeType& col_val_range,
527 const FieldSchema* col_schema, const std::string& encoded_min,
528 const std::string& encoded_max) {
529 const char* min_bytes = encoded_min.data();
530 const char* max_bytes = encoded_max.data();
531 bool need_filter = false;
532 std::vector<ScanPredicate> filters;
533 PrimitiveType col_type;
534 std::visit(
535 [&](auto&& range) {
536 col_type = range.type();
537 to_filter(range, filters);
538 },
539 col_val_range);
540 if (filters.empty()) {
541 return false;
542 }
543
544 ColumnMinMaxParams params;
545 params.conjunct_type = col_type;
546 params.parquet_type = col_schema->physical_type;
547 params.parquet_precision = col_schema->parquet_schema.precision;
548 params.parquet_scale = col_schema->parquet_schema.scale;
549 params.parquet_type_length = col_schema->parquet_schema.type_length;
550 params.min_bytes = min_bytes;
551 params.max_bytes = max_bytes;
552 for (int i = 0; i < filters.size(); i++) {
553 _eval_predicate(filters[i], &params, &need_filter);
554 if (need_filter) {
555 break;
556 }
557 }
558 return need_filter;
559}
