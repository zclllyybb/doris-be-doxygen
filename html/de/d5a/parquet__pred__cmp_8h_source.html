58 template <typename DecimalPrimitiveType, typename DecimalPhysicalType>
59 static DecimalPrimitiveType _decode_primitive_decimal(const FieldSchema* col_schema,
60 const std::string& encoded_data,
61 int dest_scale) {
62 int scale = col_schema->parquet_schema.scale;
63 Int128 value = *reinterpret_cast<const DecimalPhysicalType*>(encoded_data.data());
64 if (dest_scale > scale) {
65 value *= DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(dest_scale - scale);
66 } else if (dest_scale < scale) {
67 value /= DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(scale - dest_scale);
68 }
69 return (DecimalPrimitiveType)value;
70 }

72 template <typename DecimalPrimitiveType>
73 static DecimalPrimitiveType _decode_binary_decimal(const FieldSchema* col_schema,
74 const std::string& encoded_data,
75 int dest_scale) {
76 int scale = col_schema->parquet_schema.scale;
77 const char* buf_start = encoded_data.data();
78 Int128 value = buf_start[0] & 0x80 ? -1 : 0;
79 memcpy(reinterpret_cast<char*>(&value) + sizeof(Int128) - encoded_data.size(), buf_start,
80 encoded_data.size());
81 value = BigEndian::ToHost128(value);
82 if (dest_scale > scale) {
83 value *= DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(dest_scale - scale);
84 } else if (dest_scale < scale) {
85 value /= DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(scale - dest_scale);
86 }
87 return (DecimalPrimitiveType)value;
88 }

90 template <typename CppType>
91 static bool _filter_by_min_max(const SQLFilterOp op,
92 const std::vector<CppType>& predicate_values, CppType& min_value,
93 CppType& max_value) {
94 if (predicate_values.empty()) {
95 return false;
96 }
97 switch (op) {
98 case FILTER_IN:
99 for (const CppType& in_value : predicate_values) {
100 if (in_value >= min_value && in_value <= max_value) {
101 return false;
102 }
103 }
104 return true;
105 case FILTER_LESS:
106 return min_value >= predicate_values[0];
107 case FILTER_LESS_OR_EQUAL:
108 return min_value > predicate_values[0];
109 case FILTER_LARGER:
110 return max_value <= predicate_values[0];
111 case FILTER_LARGER_OR_EQUAL:
112 return max_value < predicate_values[0];
113 default:
114 return false;
115 }
116 }

118 template <PrimitiveType primitive_type>
119 static bool _filter_by_min_max(const ColumnValueRange<primitive_type>& col_val_range,
120 const ScanPredicate& predicate, const FieldSchema* col_schema,
121 const std::string& encoded_min, const std::string& encoded_max,
122 const cctz::time_zone& ctz) {
123 using CppType = typename PrimitiveTypeTraits<primitive_type>::CppType;
124 std::vector<CppType> predicate_values;
125 for (const void* v : predicate.values) {
126 predicate_values.emplace_back(*reinterpret_cast<const CppType*>(v));
127 }
128
129 CppType min_value;
130 CppType max_value;
131 tparquet::Type::type physical_type = col_schema->physical_type;
132 switch (col_val_range.type()) {
133#define DISPATCH(REINTERPRET_TYPE, PARQUET_TYPE) \
134 case REINTERPRET_TYPE: \
135 if (col_schema->physical_type != PARQUET_TYPE) return false; \
136 min_value = *reinterpret_cast<const CppType*>(encoded_min.data()); \
137 max_value = *reinterpret_cast<const CppType*>(encoded_max.data()); \
138 break;
139 FOR_REINTERPRET_TYPES(DISPATCH)
140#undef DISPATCH
141 case TYPE_VARCHAR:
142 [[fallthrough]];
143 case TYPE_CHAR:
144 [[fallthrough]];
145 case TYPE_STRING:
146 if constexpr (std::is_same_v<CppType, StringRef>) {
147 min_value = StringRef(encoded_min);
148 max_value = StringRef(encoded_max);
149 } else {
150 return false;
151 };
152 break;
153 case TYPE_DECIMALV2:
154 if constexpr (std::is_same_v<CppType, DecimalV2Value>) {
155 size_t max_precision = max_decimal_precision<Decimal<__int128_t>>();
156 if (col_schema->parquet_schema.precision < 1 ||
157 col_schema->parquet_schema.precision > max_precision ||
158 col_schema->parquet_schema.scale > max_precision) {
159 return false;
160 }
161 int v2_scale = DecimalV2Value::SCALE;
162 if (physical_type == tparquet::Type::FIXED_LEN_BYTE_ARRAY) {
163 min_value = DecimalV2Value(
164 _decode_binary_decimal<Int128>(col_schema, encoded_min, v2_scale));
165 max_value = DecimalV2Value(
166 _decode_binary_decimal<Int128>(col_schema, encoded_max, v2_scale));
167 } else if (physical_type == tparquet::Type::INT32) {
168 min_value = DecimalV2Value(_decode_primitive_decimal<Int128, Int32>(
169 col_schema, encoded_min, v2_scale));
170 max_value = DecimalV2Value(_decode_primitive_decimal<Int128, Int32>(
171 col_schema, encoded_max, v2_scale));
172 } else if (physical_type == tparquet::Type::INT64) {
173 min_value = DecimalV2Value(_decode_primitive_decimal<Int128, Int64>(
174 col_schema, encoded_min, v2_scale));
175 max_value = DecimalV2Value(_decode_primitive_decimal<Int128, Int64>(
176 col_schema, encoded_max, v2_scale));
177 } else {
178 return false;
179 }
180 } else {
181 return false;
182 }
183 break;
184 case TYPE_DECIMAL32:
185 [[fallthrough]];
186 case TYPE_DECIMAL64:
187 [[fallthrough]];
188 case TYPE_DECIMAL128I:
189 if constexpr (std::is_same_v<CppType, int32_t> || std::is_same_v<CppType, int64_t> ||
190 std::is_same_v<CppType, __int128_t>) {
191 size_t max_precision = max_decimal_precision<Decimal<CppType>>();
192 if (col_schema->parquet_schema.precision < 1 ||
193 col_schema->parquet_schema.precision > max_precision ||
194 col_schema->parquet_schema.scale > max_precision) {
195 return false;
196 }
197 if (physical_type == tparquet::Type::FIXED_LEN_BYTE_ARRAY) {
198 min_value = _decode_binary_decimal<CppType>(col_schema, encoded_min,
199 predicate.scale);
200 max_value = _decode_binary_decimal<CppType>(col_schema, encoded_max,
201 predicate.scale);
202 } else if (physical_type == tparquet::Type::INT32) {
203 min_value = _decode_primitive_decimal<CppType, Int32>(col_schema, encoded_min,
204 predicate.scale);
205 max_value = _decode_primitive_decimal<CppType, Int32>(col_schema, encoded_max,
206 predicate.scale);
207 } else if (physical_type == tparquet::Type::INT64) {
208 min_value = _decode_primitive_decimal<CppType, Int64>(col_schema, encoded_min,
209 predicate.scale);
210 max_value = _decode_primitive_decimal<CppType, Int64>(col_schema, encoded_max,
211 predicate.scale);
212 } else {
213 return false;
214 }
215 } else {
216 return false;
217 }
218 break;
219 case TYPE_DATE:
220 [[fallthrough]];
221 case TYPE_DATEV2:
222 if (physical_type == tparquet::Type::INT32) {
223 int64_t min_date_value =
224 static_cast<int64_t>(*reinterpret_cast<const int32_t*>(encoded_min.data()));
225 int64_t max_date_value =
226 static_cast<int64_t>(*reinterpret_cast<const int32_t*>(encoded_max.data()));
227 if constexpr (std::is_same_v<CppType, VecDateTimeValue> ||
228 std::is_same_v<CppType, DateV2Value<DateV2ValueType>>) {
229 min_value.from_unixtime(min_date_value * 24 * 60 * 60, ctz);
230 max_value.from_unixtime(max_date_value * 24 * 60 * 60, ctz);
231 } else {
232 return false;
233 }
234 } else {
235 return false;
236 }
237 break;
238 case TYPE_DATETIME:
239 [[fallthrough]];
240 case TYPE_DATETIMEV2:
241 if (physical_type == tparquet::Type::INT96) {
242 ParquetInt96 datetime96_min =
243 *reinterpret_cast<const ParquetInt96*>(encoded_min.data());
244 int64_t micros_min = datetime96_min.to_timestamp_micros();
245 ParquetInt96 datetime96_max =
246 *reinterpret_cast<const ParquetInt96*>(encoded_max.data());
247 int64_t micros_max = datetime96_max.to_timestamp_micros();
248 if constexpr (std::is_same_v<CppType, VecDateTimeValue> ||
249 std::is_same_v<CppType, DateV2Value<DateTimeV2ValueType>>) {
250 min_value.from_unixtime(micros_min / 1000000, ctz);
251 max_value.from_unixtime(micros_max / 1000000, ctz);
252 if constexpr (std::is_same_v<CppType, DateV2Value<DateTimeV2ValueType>>) {
253 min_value.set_microsecond(micros_min % 1000000);
254 max_value.set_microsecond(micros_max % 1000000);
255 }
256 } else {
257 return false;
258 }
259 } else if (physical_type == tparquet::Type::INT64) {
260 int64_t date_value_min = *reinterpret_cast<const int64_t*>(encoded_min.data());
261 int64_t date_value_max = *reinterpret_cast<const int64_t*>(encoded_max.data());
262
263 int64_t second_mask = 1;
264 int64_t scale_to_nano_factor = 1;
265 cctz::time_zone resolved_ctz = ctz;
266 const auto& schema = col_schema->parquet_schema;
267 if (schema.__isset.logicalType && schema.logicalType.__isset.TIMESTAMP) {
268 const auto& timestamp_info = schema.logicalType.TIMESTAMP;
269 if (!timestamp_info.isAdjustedToUTC) {
270 // should set timezone to utc+0
271 resolved_ctz = cctz::utc_time_zone();
272 }
273 const auto& time_unit = timestamp_info.unit;
274 if (time_unit.__isset.MILLIS) {
275 second_mask = 1000;
276 scale_to_nano_factor = 1000000;
277 } else if (time_unit.__isset.MICROS) {
278 second_mask = 1000000;
279 scale_to_nano_factor = 1000;
280 } else if (time_unit.__isset.NANOS) {
281 second_mask = 1000000000;
282 scale_to_nano_factor = 1;
283 }
284 } else if (schema.__isset.converted_type) {
285 const auto& converted_type = schema.converted_type;
286 if (converted_type == tparquet::ConvertedType::TIMESTAMP_MILLIS) {
287 second_mask = 1000;
288 scale_to_nano_factor = 1000000;
289 } else if (converted_type == tparquet::ConvertedType::TIMESTAMP_MICROS) {
290 second_mask = 1000000;
291 scale_to_nano_factor = 1000;
292 }
293 }
294
295 if constexpr (std::is_same_v<CppType, VecDateTimeValue> ||
296 std::is_same_v<CppType, DateV2Value<DateTimeV2ValueType>>) {
297 min_value.from_unixtime(date_value_min / second_mask, resolved_ctz);
298 max_value.from_unixtime(date_value_max / second_mask, resolved_ctz);
299 if constexpr (std::is_same_v<CppType, DateV2Value<DateTimeV2ValueType>>) {
300 min_value.set_microsecond((date_value_min % second_mask) *
301 scale_to_nano_factor / 1000);
302 max_value.set_microsecond((date_value_max % second_mask) *
303 scale_to_nano_factor / 1000);
304 }
305 } else {
306 return false;
307 }
308 } else {
309 return false;
310 }
311 break;
312 default:
313 return false;
314 }
315 return _filter_by_min_max(predicate.op, predicate_values, min_value, max_value);
316 }

318 template <PrimitiveType primitive_type>
319 static std::vector<ScanPredicate> _value_range_to_predicate(
320 const ColumnValueRange<primitive_type>& col_val_range) {
321 using CppType = typename PrimitiveTypeTraits<primitive_type>::CppType;
322 std::vector<ScanPredicate> predicates;
323
324 if (col_val_range.is_fixed_value_range()) {
325 ScanPredicate in_predicate;
326 in_predicate.op = SQLFilterOp::FILTER_IN;
327 in_predicate.scale = col_val_range.scale();
328 for (const auto& value : col_val_range.get_fixed_value_set()) {
329 in_predicate.values.emplace_back(&value);
330 }
331 if (!in_predicate.values.empty()) {
332 predicates.emplace_back(in_predicate);
333 }
334 return predicates;
335 }
336
337 const CppType high_value = col_val_range.get_range_max_value();
338 const CppType low_value = col_val_range.get_range_min_value();
339 const SQLFilterOp high_op = col_val_range.get_range_high_op();
340 const SQLFilterOp low_op = col_val_range.get_range_low_op();
341
342 // orc can only push down is_null. When col_value_range._contain_null = true, only indicating that
343 // value can be null, not equals null, so ignore _contain_null in col_value_range
344 if (col_val_range.is_high_value_maximum() && high_op == SQLFilterOp::FILTER_LESS_OR_EQUAL &&
345 col_val_range.is_low_value_mininum() && low_op == SQLFilterOp::FILTER_LARGER_OR_EQUAL) {
346 return predicates;
347 }
348
349 if (low_value < high_value) {
350 if (!col_val_range.is_low_value_mininum() ||
351 SQLFilterOp::FILTER_LARGER_OR_EQUAL != low_op) {
352 ScanPredicate low_predicate;
353 low_predicate.scale = col_val_range.scale();
354 low_predicate.op = low_op;
355 low_predicate.values.emplace_back(col_val_range.get_range_min_value_ptr());
356 predicates.emplace_back(low_predicate);
357 }
358 if (!col_val_range.is_high_value_maximum() ||
359 SQLFilterOp::FILTER_LESS_OR_EQUAL != high_op) {
360 ScanPredicate high_predicate;
361 high_predicate.scale = col_val_range.scale();
362 high_predicate.op = high_op;
363 high_predicate.values.emplace_back(col_val_range.get_range_max_value_ptr());
364 predicates.emplace_back(high_predicate);
365 }
366 }
367 return predicates;
368 }

370public:
371 static bool filter_by_min_max(const ColumnValueRangeType& col_val_range,
372 const FieldSchema* col_schema, const std::string& encoded_min,
373 const std::string& encoded_max, const cctz::time_zone& ctz) {
374 bool need_filter = false;
375 std::visit(
376 [&](auto&& range) {
377 std::vector<ScanPredicate> filters = _value_range_to_predicate(range);
378 for (auto& filter : filters) {
379 need_filter |= _filter_by_min_max(range, filter, col_schema, encoded_min,
380 encoded_max, ctz);
381 if (need_filter) {
382 break;
383 }
384 }
385 },
386 col_val_range);
387 return need_filter;
388 }
