110 Status copy_column_data_to_block(doris::vectorized::IColumn* input_col_ptr,
111 uint16_t* sel_rowid_idx, uint16_t select_size, int block_cid,
112 size_t batch_size) {
113 // Only the additional deleted filter condition need to materialize column be at the end of the block
114 // We should not to materialize the column of query engine do not need. So here just return OK.
115 // Eg:
116 // `delete from table where a = 10;`
117 // `select b from table;`
118 // a column only effective in segment iterator, the block from query engine only contain the b column.
119 // so the `block_cid >= data.size()` is true
120 if (block_cid >= data.size()) {
121 return Status::OK();
122 }
123
124 MutableColumnPtr raw_res_ptr = this->get_by_position(block_cid).column->assume_mutable();
125 raw_res_ptr->reserve(batch_size);
126
127 // adapt for outer join change column to nullable
128 if (raw_res_ptr->is_nullable() && !input_col_ptr->is_nullable()) {
129 auto col_ptr_nullable =
130 reinterpret_cast<vectorized::ColumnNullable*>(raw_res_ptr.get());
131 col_ptr_nullable->get_null_map_column().insert_many_defaults(select_size);
132 raw_res_ptr = col_ptr_nullable->get_nested_column_ptr();
133 }
134
135 return input_col_ptr->filter_by_selector(sel_rowid_idx, select_size, raw_res_ptr);
136 }

489 template <typename T>
490 Status merge(T&& block) {
491 // merge is not supported in dynamic block
492 if (_columns.size() == 0 && _data_types.size() == 0) {
493 _data_types = block.get_data_types();
494 _names = block.get_names();
495 _columns.resize(block.columns());
496 for (size_t i = 0; i < block.columns(); ++i) {
497 if (block.get_by_position(i).column) {
498 _columns[i] = (*std::move(block.get_by_position(i)
499 .column->convert_to_full_column_if_const()))
500 .mutate();
501 } else {
502 _columns[i] = _data_types[i]->create_column();
503 }
504 }
505 initialize_index_by_name();
506 } else {
507 if (_columns.size() != block.columns()) {
508 return Status::Error<ErrorCode::INTERNAL_ERROR>(
509 "Merge block not match, self:[{}], input:[{}], ", dump_types(),
510 block.dump_types());
511 }
512 for (int i = 0; i < _columns.size(); ++i) {
513 if (!_data_types[i]->equals(*block.get_by_position(i).type)) {
514 DCHECK(_data_types[i]->is_nullable())
515 << " target type: " << _data_types[i]->get_name()
516 << " src type: " << block.get_by_position(i).type->get_name();
517 DCHECK(((DataTypeNullable*)_data_types[i].get())
518 ->get_nested_type()
519 ->equals(*block.get_by_position(i).type));
520 DCHECK(!block.get_by_position(i).type->is_nullable());
521 _columns[i]->insert_range_from(*make_nullable(block.get_by_position(i).column)
522 ->convert_to_full_column_if_const(),
523 0, block.rows());
524 } else {
525 _columns[i]->insert_range_from(
526 *block.get_by_position(i)
527 .column->convert_to_full_column_if_const()
528 .get(),
529 0, block.rows());
530 }
531 }
532 }
533 return Status::OK();
534 }
