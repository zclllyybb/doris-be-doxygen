37inline void TextConverter::write_string_column(const SlotDescriptor* slot_desc,
38 vectorized::MutableColumnPtr* column_ptr,
39 const char* data, size_t len) {
40 DCHECK(column_ptr->get()->is_nullable());
41 auto* nullable_column = reinterpret_cast<vectorized::ColumnNullable*>(column_ptr->get());
42 if (len == 2 && data[0] == '\\' && data[1] == 'N') {
43 nullable_column->get_null_map_data().push_back(1);
44 reinterpret_cast<vectorized::ColumnString&>(nullable_column->get_nested_column())
45 .insert_default();
46 } else {
47 nullable_column->get_null_map_data().push_back(0);
48 reinterpret_cast<vectorized::ColumnString&>(nullable_column->get_nested_column())
49 .insert_data(data, len);
50 }
51}

60inline bool TextConverter::write_vec_column(const SlotDescriptor* slot_desc,
61 vectorized::IColumn* nullable_col_ptr, const char* data,
62 size_t len, bool copy_string, bool need_escape) {
63 vectorized::IColumn* col_ptr = nullable_col_ptr;
64 // \N means it's NULL
65 if (slot_desc->is_nullable()) {
66 auto* nullable_column = reinterpret_cast<vectorized::ColumnNullable*>(nullable_col_ptr);
67 if ((len == 2 && data[0] == '\\' && data[1] == 'N') || len == SQL_NULL_DATA) {
68 nullable_column->insert_data(nullptr, 0);
69 return true;
70 } else {
71 nullable_column->get_null_map_data().push_back(0);
72 col_ptr = &nullable_column->get_nested_column();
73 }
74 }
75
76 bool insert_after_parse_failure = true;
77 StringParser::ParseResult parse_result = StringParser::PARSE_SUCCESS;
78 // Parse the raw-text data. Translate the text string to internal format.
79 switch (slot_desc->type().type) {
80 case TYPE_HLL: {
81 reinterpret_cast<vectorized::ColumnHLL*>(col_ptr)->get_data().emplace_back(
82 HyperLogLog(Slice(data, len)));
83 break;
84 }
85 case TYPE_STRING:
86 case TYPE_VARCHAR:
87 case TYPE_CHAR: {
88 if (need_escape) {
89 unescape_string_on_spot(data, &len);
90 }
91 reinterpret_cast<vectorized::ColumnString*>(col_ptr)->insert_data(data, len);
92 break;
93 }
94
95 case TYPE_BOOLEAN: {
96 bool num = StringParser::string_to_bool(data, len, &parse_result);
97 reinterpret_cast<vectorized::ColumnVector<vectorized::UInt8>*>(col_ptr)->insert_value(
98 (uint8_t)num);
99 break;
100 }
101 case TYPE_TINYINT: {
102 int8_t num = StringParser::string_to_int<int8_t>(data, len, &parse_result);
103 reinterpret_cast<vectorized::ColumnVector<vectorized::Int8>*>(col_ptr)->insert_value(num);
104 break;
105 }
106 case TYPE_SMALLINT: {
107 int16_t num = StringParser::string_to_int<int16_t>(data, len, &parse_result);
108 reinterpret_cast<vectorized::ColumnVector<vectorized::Int16>*>(col_ptr)->insert_value(num);
109 break;
110 }
111 case TYPE_INT: {
112 int32_t num = StringParser::string_to_int<int32_t>(data, len, &parse_result);
113 reinterpret_cast<vectorized::ColumnVector<vectorized::Int32>*>(col_ptr)->insert_value(num);
114 break;
115 }
116 case TYPE_BIGINT: {
117 int64_t num = StringParser::string_to_int<int64_t>(data, len, &parse_result);
118 reinterpret_cast<vectorized::ColumnVector<vectorized::Int64>*>(col_ptr)->insert_value(num);
119 break;
120 }
121 case TYPE_LARGEINT: {
122 __int128 num = StringParser::string_to_int<__int128>(data, len, &parse_result);
123 reinterpret_cast<vectorized::ColumnVector<vectorized::Int128>*>(col_ptr)->insert_value(num);
124 break;
125 }
126
127 case TYPE_FLOAT: {
128 float num = StringParser::string_to_float<float>(data, len, &parse_result);
129 reinterpret_cast<vectorized::ColumnVector<vectorized::Float32>*>(col_ptr)->insert_value(
130 num);
131 break;
132 }
133 case TYPE_DOUBLE: {
134 double num = StringParser::string_to_float<double>(data, len, &parse_result);
135 reinterpret_cast<vectorized::ColumnVector<vectorized::Float64>*>(col_ptr)->insert_value(
136 num);
137 break;
138 }
139 case TYPE_DATE: {
140 vectorized::VecDateTimeValue ts_slot;
141 if (!ts_slot.from_date_str(data, len)) {
142 parse_result = StringParser::PARSE_FAILURE;
143 insert_after_parse_failure = false;
144 break;
145 }
146 ts_slot.cast_to_date();
147 reinterpret_cast<vectorized::ColumnVector<vectorized::Int64>*>(col_ptr)->insert_data(
148 reinterpret_cast<char*>(&ts_slot), 0);
149 break;
150 }
151 case TYPE_DATEV2: {
152 vectorized::VecDateTimeValue ts_slot;
153 if (!ts_slot.from_date_str(data, len)) {
154 parse_result = StringParser::PARSE_FAILURE;
155 insert_after_parse_failure = false;
156 break;
157 }
158 ts_slot.cast_to_date();
159 uint32_t num = ts_slot.to_date_v2();
160 reinterpret_cast<vectorized::ColumnVector<vectorized::UInt32>*>(col_ptr)->insert_value(num);
161 break;
162 }
163 case TYPE_DATETIME: {
164 vectorized::VecDateTimeValue ts_slot;
165 if (!ts_slot.from_date_str(data, len)) {
166 parse_result = StringParser::PARSE_FAILURE;
167 insert_after_parse_failure = false;
168 break;
169 }
170 ts_slot.to_datetime();
171 reinterpret_cast<vectorized::ColumnVector<vectorized::Int64>*>(col_ptr)->insert_data(
172 reinterpret_cast<char*>(&ts_slot), 0);
173 break;
174 }
175 case TYPE_DATETIMEV2: {
176 vectorized::VecDateTimeValue ts_slot;
177 if (!ts_slot.from_date_str(data, len)) {
178 parse_result = StringParser::PARSE_FAILURE;
179 insert_after_parse_failure = false;
180 break;
181 }
182 ts_slot.to_datetime();
183 uint64_t num = ts_slot.to_datetime_v2();
184 reinterpret_cast<vectorized::ColumnVector<vectorized::UInt64>*>(col_ptr)->insert_value(num);
185 break;
186 }
187
188 case TYPE_DECIMALV2: {
189 DecimalV2Value decimal_slot;
190 if (decimal_slot.parse_from_str(data, len)) {
191 parse_result = StringParser::PARSE_FAILURE;
192 insert_after_parse_failure = false;
193 break;
194 }
195 reinterpret_cast<vectorized::ColumnVector<vectorized::Int128>*>(col_ptr)->insert_value(
196 decimal_slot.value());
197 break;
198 }
199
200 default:
201 DCHECK(false) << "bad slot type: " << slot_desc->type();
202 break;
203 }
204
205 if (UNLIKELY(parse_result == StringParser::PARSE_FAILURE)) {
206 if (slot_desc->is_nullable()) {
207 auto* nullable_column = reinterpret_cast<vectorized::ColumnNullable*>(nullable_col_ptr);
208 size_t size = nullable_column->get_null_map_data().size();
209 doris::vectorized::NullMap& null_map_data = nullable_column->get_null_map_data();
210 null_map_data[size - 1] = 1;
211 if (!insert_after_parse_failure) {
212 nullable_column->get_nested_column().insert_default();
213 }
214 }
215 return false;
216 }
217 return true;
218}
