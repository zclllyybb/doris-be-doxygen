56 template <typename T, typename ReturnType>

58 const auto& src_data = col->get_data();

76 template <typename T, typename ReturnType>
77 static bool execute(Block& block, const ColumnDecimal<T>* col, const size_t result) {
78 const auto& src_data = col->get_data();
79 const size_t size = src_data.size();
80 UInt32 scale = src_data.get_scale();
81
82 auto dst = ColumnVector<ReturnType>::create();
83 auto& dst_data = dst->get_data();
84 dst_data.resize(size);
85
86 auto null_column = ColumnVector<UInt8>::create();
87 auto& null_map = null_column->get_data();
88 null_map.resize(size);
89
90 for (size_t i = 0; i < size; ++i)
91 dst_data[i] = convert_from_decimal<DataTypeDecimal<T>, DataTypeNumber<ReturnType>>(
92 src_data[i], scale);
93
94 execute_in_iterations(dst_data.data(), dst_data.data(), null_map, size);
95
96 block.replace_by_position(result,
97 ColumnNullable::create(std::move(dst), std::move(null_column)));
98 return true;

103 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
104 size_t result, size_t input_rows_count) override {
105 const ColumnWithTypeAndName& col = block.get_by_position(arguments[0]);
106
107 auto call = [&](const auto& types) -> bool {
108 using Types = std::decay_t<decltype(types)>;
109 using Type = typename Types::RightType;
110 using ColVecType = std::conditional_t<IsDecimalNumber<Type>, ColumnDecimal<Type>,
111 ColumnVector<Type>>;
112
113 const auto col_vec = check_and_get_column<ColVecType>(col.column.get());
114 return execute<Type, typename Impl::RetType>(block, col_vec, result);
115 };
116
117 if (!call_on_basic_type<void, true, true, true, false>(col.type->get_type_id(), call)) {
118 return Status::InvalidArgument("Illegal column {} of argument of function {}",
119 col.column->get_name(), get_name());
120 }
121 return Status::OK();
122 }
