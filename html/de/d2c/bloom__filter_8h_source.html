39// Base class for bloom filter
40// To support null value, the size of bloom filter is optimize bytes + 1.
41// The last byte is for null value flag.
42class BloomFilter {
43public:
44 // Default seed for the hash function. It comes from date +%s.
45 static const uint32_t DEFAULT_SEED = 1575457558;
46
47 // Minimum Bloom filter size, set to the size of a tiny Bloom filter block
48 static const uint32_t MINIMUM_BYTES = 32;
49
50 // Maximum Bloom filter size, set it to half of max segment file size
51 static const uint32_t MAXIMUM_BYTES = 128 * 1024 * 1024;
52
53 // Factory function for BloomFilter
54 static Status create(BloomFilterAlgorithmPB algorithm, std::unique_ptr<BloomFilter>* bf,
55 size_t bf_size = 0);
56
57 BloomFilter() : _data(nullptr), _num_bytes(0), _size(0), _has_null(nullptr) {}
58
59 virtual ~BloomFilter() {
60 if (_data) {
61 delete[] _data;
62 }
63 }
64
65 virtual bool is_ngram_bf() const { return false; }
66
67 // for write
68 Status init(uint64_t n, double fpp, HashStrategyPB strategy) {
69 return this->init(optimal_bit_num(n, fpp) / 8, strategy);
70 }
71
72 Status init(uint64_t filter_size) { return init(filter_size, HASH_MURMUR3_X64_64); }
73
74 Status init(uint64_t filter_size, HashStrategyPB strategy) {
75 if (strategy == HASH_MURMUR3_X64_64) {
76 _hash_func = murmur_hash3_x64_64;
77 } else {
78 return Status::InvalidArgument("invalid strategy:{}", strategy);
79 }
80 _num_bytes = filter_size;
81 DCHECK((_num_bytes & (_num_bytes - 1)) == 0);
82 _size = _num_bytes + 1;
83 // reserve last byte for null flag
84 _data = new char[_size];
85 memset(_data, 0, _size);
86 _has_null = (bool*)(_data + _num_bytes);
87 *_has_null = false;
88 return Status::OK();
89 }
90
91 // for read
92 // use deep copy to acquire the data
93 virtual Status init(const char* buf, uint32_t size, HashStrategyPB strategy) {
94 DCHECK(size > 1);
95 if (strategy == HASH_MURMUR3_X64_64) {
96 _hash_func = murmur_hash3_x64_64;
97 } else {
98 return Status::InvalidArgument("invalid strategy:{}", strategy);
99 }
100 if (size == 0) {
101 return Status::InvalidArgument("invalid size:{}", size);
102 }
103 _data = new char[size];
104 memcpy(_data, buf, size);
105 _size = size;
106 _num_bytes = _size - 1;
107 DCHECK((_num_bytes & (_num_bytes - 1)) == 0);
108 _has_null = (bool*)(_data + _num_bytes);
109 return Status::OK();
110 }
111
112 void reset() { memset(_data, 0, _size); }
113
114 uint64_t hash(const char* buf, uint32_t size) const {
115 uint64_t hash_code;
116 _hash_func(buf, size, DEFAULT_SEED, &hash_code);
117 return hash_code;
118 }
119
120 virtual void add_bytes(const char* buf, uint32_t size) {
121 if (buf == nullptr) {
122 *_has_null = true;
123 return;
124 }
125 uint64_t code = hash(buf, size);
126 add_hash(code);
127 }
128
129 bool test_bytes(const char* buf, uint32_t size) const {
130 if (buf == nullptr) {
131 return *_has_null;
132 }
133 uint64_t code = hash(buf, size);
134 return test_hash(code);
135 }
136
139 virtual bool contains(const BloomFilter& bf_) const = 0;
140
141 virtual char* data() const { return _data; }
142
143 uint32_t num_bytes() const { return _num_bytes; }
144
145 virtual uint32_t size() const { return _size; }
146
147 void set_has_null(bool has_null) { *_has_null = has_null; }
148
149 virtual bool has_null() const { return *_has_null; }
150
151 virtual void add_hash(uint64_t hash) = 0;
152 virtual bool test_hash(uint64_t hash) const = 0;
153
154 Status merge(const BloomFilter* other) {
155 DCHECK(other->size() == _size);
156 for (uint32_t i = 0; i < other->size(); i++) {
157 _data[i] |= other->_data[i];
158 }
159 return Status::OK();
160 }
161
162 static uint32_t used_bits(uint64_t value);
163
164 // Compute the optimal bit number according to the following rule:
165 // m = -n * ln(fpp) / (ln(2) ^ 2)
166 // n: expected distinct record number
167 // fpp: false positive probability
168 // the result will be power of 2
169 static uint32_t optimal_bit_num(uint64_t n, double fpp);
170
171protected:
172 // bloom filter data
173 // specially add one byte for null flag
174 char* _data;
175 // optimal bloom filter num bytes
176 // it is calculated by optimal_bit_num() / 8
177 uint32_t _num_bytes;
178 // equal to _num_bytes + 1
179 // last byte is for has_null flag
180 uint32_t _size;
181 // last byte's pointer in data for null flag
182 bool* _has_null;
183
184private:
185 std::function<void(const void*, const int, const uint64_t, void*)> _hash_func;
186};
