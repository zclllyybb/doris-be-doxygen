39// Super class for templatized thrift clients.
40class ThriftClientImpl {
41public:
42 virtual ~ThriftClientImpl() { close(); }
43 const std::string& ipaddress() { return _ipaddress; }
44 int port() { return _port; }
45
46 // Open the connection to the remote server. May be called
47 // repeatedly, is idempotent unless there is a failure to connect.
48 Status open();
49
50 // Retry the Open num_retries time waiting wait_ms milliseconds between retries.
51 Status open_with_retry(int num_retries, int wait_ms);
52
53 // close the connection with the remote server. May be called
54 // repeatedly.
55 void close();
56
57 // Set the connect timeout
58 void set_conn_timeout(int ms) { _socket->setConnTimeout(ms); }
59
60 // Set the receive timeout
61 void set_recv_timeout(int ms) { _socket->setRecvTimeout(ms); }
62
63 // Set the send timeout
64 void set_send_timeout(int ms) { _socket->setSendTimeout(ms); }
65
66protected:
67 ThriftClientImpl(const std::string& ipaddress, int port)
68 : _ipaddress(ipaddress),
69 _port(port),
70 _socket(new apache::thrift::transport::TSocket(ipaddress, port)) {}
71
72protected:
73 std::string _ipaddress;
74 int _port;
75
76 // All shared pointers, because Thrift requires them to be
77 std::shared_ptr<apache::thrift::transport::TSocket> _socket;
78 std::shared_ptr<apache::thrift::transport::TTransport> _transport;
79 std::shared_ptr<apache::thrift::protocol::TBinaryProtocol> _protocol;
80};

82// Utility client to a Thrift server. The parameter type is the
83// Thrift interface type that the server implements.
84template <class InterfaceType>
85class ThriftClient : public ThriftClientImpl {
86public:
87 ThriftClient(const std::string& ipaddress, int port);
88
89 ThriftClient(const std::string& ipaddress, int port, ThriftServer::ServerType server_type);
90
91 // Returns the object used to actually make RPCs against the remote server
92 InterfaceType* iface() { return _iface.get(); }
93
94private:
95 std::shared_ptr<InterfaceType> _iface;
96};

98template <class InterfaceType>
99ThriftClient<InterfaceType>::ThriftClient(const std::string& ipaddress, int port)
100 : ThriftClientImpl(ipaddress, port) {
101 _transport.reset(new apache::thrift::transport::TBufferedTransport(_socket));
102 _protocol.reset(new apache::thrift::protocol::TBinaryProtocol(_transport));
103 _iface.reset(new InterfaceType(_protocol));
104}
105
106template <class InterfaceType>
107ThriftClient<InterfaceType>::ThriftClient(const std::string& ipaddress, int port,
108 ThriftServer::ServerType server_type)
109 : ThriftClientImpl(ipaddress, port), _iface(new InterfaceType(_protocol)) {
110 switch (server_type) {
111 case ThriftServer::NON_BLOCKING:
112 _transport.reset(new apache::thrift::transport::TFramedTransport(_socket));
113 break;
114 case ThriftServer::THREADED:
115 case ThriftServer::THREAD_POOL:
116 _transport.reset(new apache::thrift::transport::TBufferedTransport(_socket));
117 break;
118 default:
119 std::stringstream error_msg;
120 error_msg << "Unsupported server type: " << server_type;
121 LOG(ERROR) << error_msg.str();
122 DCHECK(false);
123 break;
124 }
125
126 _protocol.reset(new apache::thrift::protocol::TBinaryProtocol(_transport));
127 _iface.reset(new InterfaceType(_protocol));
128}
