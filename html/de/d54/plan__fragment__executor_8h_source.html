50// PlanFragmentExecutor handles all aspects of the execution of a single plan fragment,
51// including setup and tear-down, both in the success and error case.
52// Tear-down frees all memory allocated for this plan fragment and closes all data
53// streams; it happens automatically in the d'tor.
54//
55// The executor makes an aggregated profile for the entire fragment available,
56// which includes profile information for the plan itself as well as the output
57// sink, if any.
58// The ReportStatusCallback passed into the c'tor is invoked periodically to report the
59// execution status. The frequency of those reports is controlled by the flag
60// status_report_interval; setting that flag to 0 disables periodic reporting altogether
61// Regardless of the value of that flag, if a report callback is specified, it is
62// invoked at least once at the end of execution with an overall status and profile
63// (and 'done' indicator). The only exception is when execution is cancelled, in which
64// case the callback is *not* invoked (the coordinator already knows that execution
65// stopped, because it initiated the cancellation).
66//
67// Aside from Cancel(), which may be called asynchronously, this class is not
68// thread-safe.
69class PlanFragmentExecutor {
70public:
71 // Callback to report execution status of plan fragment.
72 // 'profile' is the cumulative profile, 'done' indicates whether the execution
73 // is done or still continuing.
74 // Note: this does not take a const RuntimeProfile&, because it might need to call
75 // functions like PrettyPrint() or to_thrift(), neither of which is const
76 // because they take locks.
77 using report_status_callback = std::function<void(const Status&, RuntimeProfile*, bool)>;
78
79 // report_status_cb, if !empty(), is used to report the accumulated profile
80 // information periodically during execution (open() or get_next()).
81 PlanFragmentExecutor(ExecEnv* exec_env, const report_status_callback& report_status_cb);
82
83 // Closes the underlying plan fragment and frees up all resources allocated
84 // in open()/get_next().
85 // It is an error to delete a PlanFragmentExecutor with a report callback
86 // before open()/get_next() (depending on whether the fragment has a sink)
87 // indicated that execution is finished.
88 ~PlanFragmentExecutor();
89
90 // prepare for execution. Call this prior to open().
91 // This call won't block.
92 // runtime_state() and row_desc() will not be valid until prepare() is called.
93 // If request.query_options.mem_limit > 0, it is used as an approximate limit on the
94 // number of bytes this query can consume at runtime.
95 // The query will be aborted (MEM_LIMIT_EXCEEDED) if it goes over that limit.
96 // If fragments_ctx is not null, some components will be got from fragments_ctx.
97 Status prepare(const TExecPlanFragmentParams& request,
98 QueryFragmentsCtx* fragments_ctx = nullptr);
99
100 // Start execution. Call this prior to get_next().
101 // If this fragment has a sink, open() will send all rows produced
102 // by the fragment to that sink. Therefore, open() may block until
103 // all rows are produced (and a subsequent call to get_next() will not return
104 // any rows).
105 // This also starts the status-reporting thread, if the interval flag
106 // is > 0 and a callback was specified in the c'tor.
107 // If this fragment has a sink, report_status_cb will have been called for the final
108 // time when open() returns, and the status-reporting thread will have been stopped.
109 Status open();
110
111 // Closes the underlying plan fragment and frees up all resources allocated
112 // in open()/get_next().
113 void close();
114
115 // Initiate cancellation. Must not be called until after prepare() returned.
116 void cancel(const PPlanFragmentCancelReason& reason = PPlanFragmentCancelReason::INTERNAL_ERROR,
117 const std::string& msg = "");
118
119 // call these only after prepare()
120 RuntimeState* runtime_state() { return _runtime_state.get(); }
121 const RowDescriptor& row_desc();
122
123 // Profile information for plan and output sink.
124 RuntimeProfile* profile();
125
126 const Status& status() const { return _status; }
127
128 DataSink* get_sink() const { return _sink.get(); }
129
130 void set_is_report_on_cancel(bool val) { _is_report_on_cancel = val; }
131
132private:
133 ExecEnv* _exec_env; // not owned
134 ExecNode* _plan; // lives in _runtime_state->obj_pool()
135 TUniqueId _query_id;
136
137 // profile reporting-related
138 report_status_callback _report_status_cb;
139 std::promise<bool> _report_thread_promise;
140 std::future<bool> _report_thread_future;
141 std::mutex _report_thread_lock;
142
143 // Indicates that profile reporting thread should stop.
144 // Tied to _report_thread_lock.
145 std::condition_variable _stop_report_thread_cv;
146
147 // Indicates that profile reporting thread started.
148 // Tied to _report_thread_lock.
149 std::condition_variable _report_thread_started_cv;
150 bool _report_thread_active; // true if we started the thread
151
152 // true if _plan->get_next() indicated that it's done
153 bool _done;
154
155 // true if prepare() returned OK
156 bool _prepared;
157
158 // true if close() has been called
159 bool _closed;
160
161 bool _is_report_success;
162
163 // If this is set to false, and '_is_report_success' is false as well,
164 // This executor will not report status to FE on being cancelled.
165 bool _is_report_on_cancel;
166
167 // Overall execution status. Either ok() or set to the first error status that
168 // was encountered.
169 Status _status;
170
171 // Protects _status
172 // lock ordering:
173 // 1. _report_thread_lock
174 // 2. _status_lock
175 std::mutex _status_lock;
176
177 // note that RuntimeState should be constructed before and destructed after `_sink' and `_row_batch',
178 // therefore we declare it before `_sink' and `_row_batch'
179 std::unique_ptr<RuntimeState> _runtime_state;
180 // Output sink for rows sent to this fragment. May not be set, in which case rows are
181 // returned via get_next's row batch
182 // Created in prepare (if required), owned by this object.
183 std::unique_ptr<DataSink> _sink;
184
185 // Number of rows returned by this fragment
186 RuntimeProfile::Counter* _rows_produced_counter;
187
188 // Number of blocks returned by this fragment
189 RuntimeProfile::Counter* _blocks_produced_counter;
190
191 RuntimeProfile::Counter* _fragment_cpu_timer;
192
193 // It is shared with BufferControlBlock and will be called in two different
194 // threads. But their calls are all at different time, there is no problem of
195 // multithreaded access.
196 std::shared_ptr<QueryStatistics> _query_statistics;
197 bool _collect_query_statistics_with_every_batch;
198
199 // Record the cancel information when calling the cancel() method, return it to FE
200 PPlanFragmentCancelReason _cancel_reason;
201 std::string _cancel_msg;
202
203 ObjectPool* obj_pool() { return _runtime_state->obj_pool(); }
204
205 // typedef for TPlanFragmentExecParams.per_node_scan_ranges
206 using PerNodeScanRanges = std::map<TPlanNodeId, std::vector<TScanRangeParams>>;
207
208 // Main loop of profile reporting thread.
209 // Exits when notified on _done_cv.
210 // On exit, *no report is sent*, ie, this will not send the final report.
211 void report_profile();
212
213 // Invoked the report callback if there is a report callback and the current
214 // status isn't CANCELLED. Sets 'done' to true in the callback invocation if
215 // done == true or we have an error status.
216 void send_report(bool done);
217
218 // Executes open() logic and returns resulting status. Does not set _status.
219 // If this plan fragment has no sink, open_internal() does nothing.
220 // If this plan fragment has a sink and open_internal() returns without an
221 // error condition, all rows will have been sent to the sink, the sink will
222 // have been closed, a final report will have been sent and the report thread will
223 // have been stopped. _sink will be set to nullptr after successful execution.
224 Status open_vectorized_internal();
225
226 // Executes get_next() logic and returns resulting status.
227 Status get_vectorized_internal(::doris::vectorized::Block* block, bool* eos);
228
229 // Stops report thread, if one is running. Blocks until report thread terminates.
230 // Idempotent.
231 void stop_report_thread();
232
233 const DescriptorTbl& desc_tbl() const { return _runtime_state->desc_tbl(); }
234
235 void _collect_query_statistics();
236
237 void _collect_node_statistics();
238};
