33template <typename T>
34struct AggregateFunctionAvgData {
35 T sum = 0;
36 UInt64 count = 0;
37
38 AggregateFunctionAvgData& operator=(const AggregateFunctionAvgData<T>& src) {
39 sum = src.sum;
40 count = src.count;
41 return *this;
42 }
43
44 template <typename ResultT>
45 ResultT result() const {
46 if constexpr (std::is_floating_point_v<ResultT>) {
47 if constexpr (std::numeric_limits<ResultT>::is_iec559) {
48 return static_cast<ResultT>(sum) / count;
49 }
50 }
51
52 if (!count) {
53 // null is handled in AggregationNode::_get_without_key_result
54 return static_cast<ResultT>(sum);
55 }
56 // to keep the same result with row vesion; see AggregateFunctions::decimalv2_avg_get_value
57 if constexpr (IsDecimalV2<T> && IsDecimalV2<ResultT>) {
58 DecimalV2Value decimal_val_count(count, 0);
59 DecimalV2Value decimal_val_sum(static_cast<Int128>(sum));
60 DecimalV2Value cal_ret = decimal_val_sum / decimal_val_count;
61 Decimal128 ret(cal_ret.value());
62 return ret;
63 } else {
64 return static_cast<ResultT>(sum) / count;
65 }
66 }
67
68 void write(BufferWritable& buf) const {
69 write_binary(sum, buf);
70 write_binary(count, buf);
71 }
72
73 void read(BufferReadable& buf) {
74 read_binary(sum, buf);
75 read_binary(count, buf);
76 }
77};

80template <typename T, typename Data>
81class AggregateFunctionAvg final
82 : public IAggregateFunctionDataHelper<Data, AggregateFunctionAvg<T, Data>> {
83public:
84 using ResultType = DisposeDecimal<T, Float64>;
85 using ResultDataType =
86 std::conditional_t<IsDecimalV2<T>, DataTypeDecimal<Decimal128>,
87 std::conditional_t<IsDecimalNumber<T>, DataTypeDecimal<Decimal128I>,
88 DataTypeNumber<Float64>>>;
89 using ColVecType = std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<T>, ColumnVector<T>>;
90 using ColVecResult =
91 std::conditional_t<IsDecimalV2<T>, ColumnDecimal<Decimal128>,
92 std::conditional_t<IsDecimalNumber<T>, ColumnDecimal<Decimal128I>,
93 ColumnVector<Float64>>>;
94
96 AggregateFunctionAvg(const DataTypes& argument_types_)
97 : IAggregateFunctionDataHelper<Data, AggregateFunctionAvg<T, Data>>(argument_types_),
98 scale(get_decimal_scale(*argument_types_[0])) {}
99
100 String get_name() const override { return "avg"; }
101
102 DataTypePtr get_return_type() const override {
103 if constexpr (IsDecimalNumber<T>) {
104 return std::make_shared<ResultDataType>(ResultDataType::max_precision(), scale);
105 } else {
106 return std::make_shared<ResultDataType>();
107 }
108 }
109
110 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
111 Arena*) const override {
112 const auto& column = static_cast<const ColVecType&>(*columns[0]);
113 if constexpr (IsDecimalNumber<T>) {
114 this->data(place).sum += column.get_data()[row_num].value;
115 } else {
116 this->data(place).sum += column.get_data()[row_num];
117 }
118 ++this->data(place).count;
119 }
120
121 void reset(AggregateDataPtr place) const override {
122 this->data(place).sum = 0;
123 this->data(place).count = 0;
124 }
125
126 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
127 Arena*) const override {
128 if constexpr (IsDecimalNumber<T>) {
129 this->data(place).sum += this->data(rhs).sum.value;
130 } else {
131 this->data(place).sum += this->data(rhs).sum;
132 }
133 this->data(place).count += this->data(rhs).count;
134 }
135
136 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
137 this->data(place).write(buf);
138 }
139
140 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
141 Arena*) const override {
142 this->data(place).read(buf);
143 }
144
145 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
146 auto& column = static_cast<ColVecResult&>(to);
147 column.get_data().push_back(this->data(place).template result<ResultType>());
148 }
149
150 void deserialize_from_column(AggregateDataPtr places, const IColumn& column, Arena* arena,
151 size_t num_rows) const override {
152 auto& col = assert_cast<const ColumnFixedLengthObject&>(column);
153 DCHECK(col.size() >= num_rows) << "source column's size should greater than num_rows";
154 auto* data = col.get_data().data();
155 memcpy(places, data, sizeof(Data) * num_rows);
156 }
157
158 void serialize_to_column(const std::vector<AggregateDataPtr>& places, size_t offset,
159 MutableColumnPtr& dst, const size_t num_rows) const override {
160 auto& col = assert_cast<ColumnFixedLengthObject&>(*dst);
161 col.set_item_size(sizeof(Data));
162 col.resize(num_rows);
163 auto* data = col.get_data().data();
164 for (size_t i = 0; i != num_rows; ++i) {
165 *reinterpret_cast<Data*>(&data[sizeof(Data) * i]) =
166 *reinterpret_cast<Data*>(places[i] + offset);
167 }
168 }
169
170 void streaming_agg_serialize_to_column(const IColumn** columns, MutableColumnPtr& dst,
171 const size_t num_rows, Arena* arena) const override {
172 auto* src_data = assert_cast<const ColVecType&>(*columns[0]).get_data().data();
173 auto& dst_col = static_cast<ColumnFixedLengthObject&>(*dst);
174 dst_col.set_item_size(sizeof(Data));
175 dst_col.resize(num_rows);
176 auto* data = dst_col.get_data().data();
177 for (size_t i = 0; i != num_rows; ++i) {
178 auto& state = *reinterpret_cast<Data*>(&data[sizeof(Data) * i]);
179 state.sum = src_data[i];
180 state.count = 1;
181 }
182 }
183
184 void deserialize_and_merge_from_column(AggregateDataPtr __restrict place, const IColumn& column,
185 Arena* arena) const override {
186 auto& col = assert_cast<const ColumnFixedLengthObject&>(column);
187 const size_t num_rows = column.size();
188 DCHECK(col.size() >= num_rows) << "source column's size should greater than num_rows";
189 auto* data = reinterpret_cast<const Data*>(col.get_data().data());
190
191 for (size_t i = 0; i != num_rows; ++i) {
192 this->data(place).sum += data[i].sum;
193 this->data(place).count += data[i].count;
194 }
195 }
196
197 void serialize_without_key_to_column(ConstAggregateDataPtr __restrict place,
198 MutableColumnPtr& dst) const override {
199 auto& col = assert_cast<ColumnFixedLengthObject&>(*dst);
200 col.set_item_size(sizeof(Data));
201 col.resize(1);
202 *reinterpret_cast<Data*>(col.get_data().data()) = this->data(place);
203 }
204
205 MutableColumnPtr create_serialize_column() const override {
206 return ColumnFixedLengthObject::create(sizeof(Data));
207 }
208
209 DataTypePtr get_serialized_type() const override {
210 return std::make_shared<DataTypeFixedLengthObject>();
211 }
212
213private:
214 UInt32 scale;
215};
