123 void close() {
124 if (is_closed) {
125 return;
126 }
127 VLOG_DEBUG << "Free resources for JniContext";
128 JNIEnv* env;
129 Status status = JniUtil::GetJNIEnv(&env);
130 if (!status.ok()) {
131 LOG(WARNING) << "errors while get jni env " << status;
132 return;
133 }
134 env->CallNonvirtualVoidMethodA(executor, parent->executor_cl_,
135 parent->executor_close_id_, NULL);
136 Status s = JniUtil::GetJniExceptionMsg(env);
137 if (!s.ok()) LOG(WARNING) << s;
138 env->DeleteGlobalRef(executor);
139 is_closed = true;
140 }
