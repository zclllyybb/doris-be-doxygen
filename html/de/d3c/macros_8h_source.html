24// The COMPILE_ASSERT macro can be used to verify that a compile time
25// expression is true. For example, you could use it to verify the
26// size of a static array:
27//
28// COMPILE_ASSERT(ARRAYSIZE(content_type_names) == CONTENT_NUM_TYPES,
29// content_type_names_incorrect_size);
30//
31// or to make sure a struct is smaller than a certain size:
32//
33// COMPILE_ASSERT(sizeof(foo) < 128, foo_too_large);
34//
35// The second argument to the macro is the name of the variable. If
36// the expression is false, most compilers will issue a warning/error
37// containing the name of the variable.
38
39template <bool>
40struct CompileAssert {};
41
42#ifndef COMPILE_ASSERT
43
44#define COMPILE_ASSERT(expr, msg) \
45 typedef CompileAssert<(bool(expr))> msg[bool(expr) ? 1 : -1] ATTRIBUTE_UNUSED
46
47// Implementation details of COMPILE_ASSERT:
48//
49// - COMPILE_ASSERT works by defining an array type that has -1
50// elements (and thus is invalid) when the expression is false.
51//
52// - The simpler definition
53//
54// #define COMPILE_ASSERT(expr, msg) typedef char msg[(expr) ? 1 : -1]
55//
56// does not work, as gcc supports variable-length arrays whose sizes
57// are determined at run-time (this is gcc's extension and not part
58// of the C++ standard). As a result, gcc fails to reject the
59// following code with the simple definition:
60//
61// int foo;
62// COMPILE_ASSERT(foo, msg); // not supposed to compile as foo is
63// // not a compile-time constant.
64//
65// - By using the type CompileAssert<(bool(expr))>, we ensures that
66// expr is a compile-time constant. (Template arguments must be
67// determined at compile-time.)
68//
69// - The outer parentheses in CompileAssert<(bool(expr))> are necessary
70// to work around a bug in gcc 3.4.4 and 4.0.1. If we had written
71//
72// CompileAssert<bool(expr)>
73//
74// instead, these compilers will refuse to compile
75//
76// COMPILE_ASSERT(5 > 0, some_message);
77//
78// (They seem to think the ">" in "5 > 0" marks the end of the
79// template argument list.)
80//
81// - The array size is (bool(expr) ? 1 : -1), instead of simply
82//
83// ((expr) ? 1 : -1).
84//
85// This is to avoid running into a bug in MS VC 7.1, which
86// causes ((0.0) ? 1 : -1) to incorrectly evaluate to 1.
87#endif // COMPILE_ASSERT

89// A macro to disallow the copy constructor and operator= functions
90// This should be used in the private: declarations for a class
91//
92// For disallowing only assign or copy, write the code directly, but declare
93// the intend in a comment, for example:
94// void operator=(const TypeName&); // DISALLOW_ASSIGN
95// Note, that most uses of DISALLOW_ASSIGN and DISALLOW_COPY are broken
96// semantically, one should either use disallow both or neither. Try to
97// avoid these in new code.
98#ifndef DISALLOW_COPY_AND_ASSIGN
99#define DISALLOW_COPY_AND_ASSIGN(TypeName) \
100 TypeName(const TypeName&) = delete; \
101 void operator=(const TypeName&) = delete
102#endif
103
104// An older, politically incorrect name for the above.
105// Prefer DISALLOW_COPY_AND_ASSIGN for new code.
106#define DISALLOW_EVIL_CONSTRUCTORS(TypeName) DISALLOW_COPY_AND_ASSIGN(TypeName)

120// The following enum should be used only as a constructor argument to indicate
121// that the variable has static storage class, and that the constructor should
122// do nothing to its state. It indicates to the reader that it is legal to
123// declare a static instance of the class, provided the constructor is given
124// the base::LINKER_INITIALIZED argument. Normally, it is unsafe to declare a
125// static variable that has a constructor or a destructor because invocation
126// order is undefined. However, IF the type can be initialized by filling with
127// zeroes (which the loader does for static variables), AND the type's
128// destructor does nothing to the storage, then a constructor for static
129// initialization can be declared as
130// explicit MyClass(base::LinkerInitialized x) {}
131// and invoked as
132// static MyClass my_variable_name(base::LINKER_INITIALIZED);
133namespace base {
134enum LinkerInitialized { LINKER_INITIALIZED };
135}

137// The FALLTHROUGH_INTENDED macro can be used to annotate implicit fall-through
138// between switch labels:
139// switch (x) {
140// case 40:
141// case 41:
142// if (truth_is_out_there) {
143// ++x;
144// FALLTHROUGH_INTENDED; // Use instead of/along with annotations in
145// // comments.
146// } else {
147// return x;
148// }
149// case 42:
150// ...
151//
152// As shown in the example above, the FALLTHROUGH_INTENDED macro should be
153// followed by a semicolon. It is designed to mimic control-flow statements
154// like 'break;', so it can be placed in most places where 'break;' can, but
155// only if there are no statements on the execution path between it and the
156// next switch label.
157//
158// When compiled with clang in C++11 mode, the FALLTHROUGH_INTENDED macro is
159// expanded to [[clang::fallthrough]] attribute, which is analysed when
160// performing switch labels fall-through diagnostic ('-Wimplicit-fallthrough').
161// See clang documentation on language extensions for details:
162// http://clang.llvm.org/docs/LanguageExtensions.html#clang__fallthrough
163//
164// When used with unsupported compilers, the FALLTHROUGH_INTENDED macro has no
165// effect on diagnostics.
166//
167// In either case this macro has no effect on runtime behavior and performance
168// of code.
169#if defined(__clang__) && defined(LANG_CXX11) && defined(__has_warning)
170#if __has_feature(cxx_attributes) && __has_warning("-Wimplicit-fallthrough")
171#define FALLTHROUGH_INTENDED [[clang::fallthrough]] // NOLINT
172#endif
173#endif
174
175#ifndef FALLTHROUGH_INTENDED
176#define FALLTHROUGH_INTENDED \
177 do { \
178 } while (0)
179#endif

181// Retry on EINTR for functions like read() that return -1 on error.
182#define RETRY_ON_EINTR(err, expr) \
183 do { \
184 static_assert(std::is_signed<decltype(err)>::value, #err " must be a signed integer"); \
185 (err) = (expr); \
186 } while ((err) == -1 && errno == EINTR)
187
188// Same as above but for stream API calls like fread() and fwrite().
189#define STREAM_RETRY_ON_EINTR(nread, stream, expr) \
190 do { \
191 static_assert(std::is_unsigned<decltype(nread)>::value == true, \
192 #nread " must be an unsigned integer"); \
193 (nread) = (expr); \
194 } while ((nread) == 0 && ferror(stream) == EINTR)
195
196// Same as above but for functions that return pointer types (like
197// fopen() and freopen()).
198#define POINTER_RETRY_ON_EINTR(ptr, expr) \
199 do { \
200 static_assert(std::is_pointer<decltype(ptr)>::value == true, #ptr " must be a pointer"); \
201 (ptr) = (expr); \
202 } while ((ptr) == nullptr && errno == EINTR)
