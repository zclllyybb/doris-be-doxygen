37template <typename T, size_t N>
38class FixedContainer {
39public:
40 using Self = FixedContainer;
41 using ElementType = T;
42
43 class Iterator;
44
45 FixedContainer() : _size(0) { static_assert(N >= 1 && N <= 12); }
46
47 ~FixedContainer() = default;
48
49 void insert(const T& value) {
50 DCHECK(_size < N);
51 _data[_size++] = value;
52 }
53
54 void insert(Iterator begin, Iterator end) {
55 for (auto iter = begin; iter != end; ++iter) {
56 DCHECK(_size < N);
57 _data[_size++] = (*iter);
58 }
59 }

61 // Use '|' instead of '||' has better performance by test.
62 bool find(const T& value) const {
63 if constexpr (N == 1) {
64 return (value == _data[0]);
65 }
66 if constexpr (N == 2) {
67 return (uint8_t)(value == _data[0]) | (uint8_t)(value == _data[1]);
68 }
69 if constexpr (N == 3) {
70 return (uint8_t)(value == _data[0]) | (uint8_t)(value == _data[1]) |
71 (uint8_t)(value == _data[2]);
72 }
73 if constexpr (N == 4) {
74 return (uint8_t)(value == _data[0]) | (uint8_t)(value == _data[1]) |
75 (uint8_t)(value == _data[2]) | (uint8_t)(value == _data[3]);
76 }
77 if constexpr (N == 5) {

79 (uint8_t)(value == _data[2]) | (uint8_t)(value == _data[3]) |
80 (uint8_t)(value == _data[4]);
81 }

135 class Iterator {
136 public:
137 explicit Iterator(std::array<T, N>& data, size_t index) : _data(data), _index(index) {}
138 Iterator& operator++() {
139 ++_index;
140 return *this;
141 }
142 Iterator operator++(int) {
143 Iterator ret_val = *this;
144 ++(*this);
145 return ret_val;
146 }
147 bool operator==(Iterator other) const { return _index == other._index; }
148 bool operator!=(Iterator other) const { return !(*this == other); }
149 T& operator*() const { return _data[_index]; }
150
151 T* operator->() const { return &operator*(); }
152
153 // iterator traits
154 using iterator_category = std::forward_iterator_tag;
155 using difference_type = std::ptrdiff_t;
156 using value_type = T;
157 using pointer = T*;
158 using reference = T&;
159
160 private:
161 std::array<T, N>& _data;
162 size_t _index;
163 };
164 Iterator begin() { return Iterator(_data, 0); }
165 Iterator end() { return Iterator(_data, _size); }
166
167private:
168 std::array<T, N> _data;
169 size_t _size;
170};

286template <PrimitiveType T,
287 typename _ContainerType = DynamicContainer<typename VecPrimitiveTypeTraits<T>::CppType>,
288 typename _ColumnType = typename VecPrimitiveTypeTraits<T>::ColumnType>
289class HybridSet : public HybridSetBase {
290public:
291 using ContainerType = _ContainerType;
292 using ElementType = typename ContainerType::ElementType;
293 using ColumnType = _ColumnType;
294
295 HybridSet() = default;
296
297 ~HybridSet() override = default;
298
299 void insert(const void* data) override {
300 if (data == nullptr) {
301 return;
302 }
303
304 if constexpr (sizeof(ElementType) >= 16) {
305 // for large int, it will core dump with no memcpy
306 ElementType value;
307 memcpy(&value, data, sizeof(ElementType));
308 _set.insert(value);
309 } else {
310 _set.insert(*reinterpret_cast<const ElementType*>(data));
311 }
312 }
313 void insert(void* data, size_t) override { insert(data); }
314
315 void insert_fixed_len(const char* data, const int* offsets, int number) override {
316 for (int i = 0; i < number; i++) {
317 insert((void*)((ElementType*)data + offsets[i]));
318 }
319 }
320
321 int size() override { return _set.size(); }
322
323 bool find(const void* data) const override {
324 if (data == nullptr) {
325 return false;
326 }
327
328 return _set.find(*reinterpret_cast<const ElementType*>(data));
329 }
330
331 bool find(const void* data, size_t) const override { return find(data); }
332
333 void find_fixed_len(const char* __restrict data, const uint8* __restrict null_map, int number,
334 uint8* __restrict results) override {
335 ElementType* value = (ElementType*)data;
336 if (null_map == nullptr) {
337 for (int i = 0; i < number; i++) {
338 results[i] = _set.find(value[i]);
339 }
340 } else {
341 for (int i = 0; i < number; i++) {
342 results[i] = _set.find(value[i]) & !null_map[i];
343 }
344 }
345 }

347 void find_batch(const doris::vectorized::IColumn& column, size_t rows,
348 doris::vectorized::ColumnUInt8::Container& results) override {
349 _find_batch<false, false>(column, rows, nullptr, results);

352 void find_batch_negative(const doris::vectorized::IColumn& column, size_t rows,

369 template <bool is_nullable, bool is_negative>
370 void _find_batch(const doris::vectorized::IColumn& column, size_t rows,
371 const doris::vectorized::NullMap* null_map,
372 doris::vectorized::ColumnUInt8::Container& results) {
373 auto& col = assert_cast<const ColumnType&>(column);
374 const auto* __restrict data = (ElementType*)col.get_data().data();
375 const uint8_t* __restrict null_map_data;
376 if constexpr (is_nullable) {
377 null_map_data = null_map->data();
378 }
379 auto* __restrict result_data = results.data();
380 for (size_t i = 0; i < rows; ++i) {
381 if constexpr (!is_nullable && !is_negative) {
382 result_data[i] = _set.find(data[i]);
383 } else if constexpr (!is_nullable && is_negative) {
384 result_data[i] = !_set.find(data[i]);
385 } else if constexpr (is_nullable && !is_negative) {
386 result_data[i] = _set.find(data[i]) & (!null_map_data[i]);
387 } else { // (is_nullable && is_negative)
388 result_data[i] = !(_set.find(data[i]) & (!null_map_data[i]));
389 }
390 }
391 }
