64template <PrimitiveType T>
65class HybridSet : public HybridSetBase {
66public:
67 using CppType = typename VecPrimitiveTypeTraits<T>::CppType;
68
69 HybridSet() = default;
70
71 ~HybridSet() override = default;
72
73 void insert(const void* data) override {
74 if (data == nullptr) {
75 return;
76 }
77
78 if constexpr (sizeof(CppType) >= 16) {
79 // for large int, it will core dump with no memcpy
80 CppType value;
81 memcpy(&value, data, sizeof(CppType));
82 _set.insert(value);
83 } else {
84 _set.insert(*reinterpret_cast<const CppType*>(data));
85 }
86 }
87 void insert(void* data, size_t) override { insert(data); }
88
89 void insert_fixed_len(const char* data, const int* offsets, int number) override {
90 for (int i = 0; i < number; i++) {
91 insert((void*)((CppType*)data + offsets[i]));
92 }
93 }
94
95 void insert(HybridSetBase* set) override {
96 HybridSet<T>* hybrid_set = reinterpret_cast<HybridSet<T>*>(set);
97 _set.insert(hybrid_set->_set.begin(), hybrid_set->_set.end());
98 }
99
100 int size() override { return _set.size(); }
101
102 bool find(const void* data) override {
103 if (data == nullptr) {
104 return false;
105 }
106
107 auto it = _set.find(*reinterpret_cast<const CppType*>(data));
108 return !(it == _set.end());
109 }
110
111 bool find(const void* data, size_t) override { return find(data); }
112
113 void find_fixed_len(const char* data, const uint8* nullmap, int number,
114 uint8* results) override {
115 for (int i = 0; i < number; i++) {
116 if (nullmap != nullptr && nullmap[i]) {
117 results[i] = false;
118 } else {
119 results[i] = _set.count(*((CppType*)data + i));
120 }
121 }
122 }

124 template <class _iT>
125 class Iterator : public IteratorBase {
126 public:
127 Iterator(typename phmap::flat_hash_set<_iT>::iterator begin,
128 typename phmap::flat_hash_set<_iT>::iterator end)
129 : _begin(begin), _end(end) {}
130 ~Iterator() override = default;
131 bool has_next() const override { return !(_begin == _end); }
132 const void* get_value() override { return _begin.operator->(); }
133 void next() override { ++_begin; }
134
135 private:
136 typename phmap::flat_hash_set<_iT>::iterator _begin;
137 typename phmap::flat_hash_set<_iT>::iterator _end;
138 };
139
140 IteratorBase* begin() override {
141 return _pool.add(new (std::nothrow) Iterator<CppType>(_set.begin(), _set.end()));
142 }
143
144 phmap::flat_hash_set<CppType>* get_inner_set() { return &_set; }
145
146private:
147 phmap::flat_hash_set<CppType> _set;
148 ObjectPool _pool;
149};
