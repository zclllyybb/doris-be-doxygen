35// File system for S3 compatible object storage
36// When creating S3FileSystem, all required info should be set in S3Conf,
37// such as ak, sk, region, endpoint, bucket.
38// And the root_path of S3FileSystem is s3_conf.prefix.
39// When using S3FileSystem, it accepts 2 kinds of path:
40// 1. Full path: s3://bucket/path/to/file.txt
41// In this case, the root_path is not used.
42// 2. only key: path/to/file.txt
43// In this case, the final key will be "prefix + path/to/file.txt"
44//
45// This class is thread-safe.(Except `set_xxx` method)
46class S3FileSystem final : public RemoteFileSystem {
47public:
48 static Status create(S3Conf s3_conf, std::string id, std::shared_ptr<S3FileSystem>* fs);
49 ~S3FileSystem() override;
50 // Guarded by external lock.
51 void set_conf(S3Conf s3_conf) { _s3_conf = std::move(s3_conf); }
52
53 std::shared_ptr<Aws::S3::S3Client> get_client() const {
54 std::lock_guard lock(_client_mu);
55 return _client;
56 }
57
58protected:
59 Status connect_impl() override;
60 Status create_file_impl(const Path& file, FileWriterPtr* writer) override;
61 Status open_file_internal(const Path& file, int64_t file_size, FileReaderSPtr* reader) override;
62 Status create_directory_impl(const Path& dir, bool failed_if_exists = false) override;
63 Status delete_file_impl(const Path& file) override;
64 Status delete_directory_impl(const Path& dir) override;
65 Status batch_delete_impl(const std::vector<Path>& files) override;
66 Status exists_impl(const Path& path, bool* res) const override;
67 Status file_size_impl(const Path& file, int64_t* file_size) const override;
68 Status list_impl(const Path& dir, bool only_file, std::vector<FileInfo>* files,
69 bool* exists) override;
70 Status rename_impl(const Path& orig_name, const Path& new_name) override;
71 Status rename_dir_impl(const Path& orig_name, const Path& new_name) override;
72
73 Status upload_impl(const Path& local_file, const Path& remote_file) override;
74 Status batch_upload_impl(const std::vector<Path>& local_files,
75 const std::vector<Path>& remote_files) override;
76 Status direct_upload_impl(const Path& remote_file, const std::string& content) override;
77 Status upload_with_checksum_impl(const Path& local_file, const Path& remote_file,
78 const std::string& checksum) override;
79 Status download_impl(const Path& remote_file, const Path& local_file) override;
80 Status direct_download_impl(const Path& remote_file, std::string* content) override;
81
82 Path absolute_path(const Path& path) const override {
83 if (path.string().find("://") != std::string::npos) {
84 // the path is with schema, which means this is a full path like:
85 // s3://bucket/path/to/file.txt
86 // so no need to concat with prefix
87 return path;
88 } else {
89 // path with no schema
90 return _root_path / path;
91 }
92 }
93
94private:
95 S3FileSystem(S3Conf&& s3_conf, std::string&& id);
96
97 template <typename AwsOutcome>
98 std::string error_msg(const std::string& key, const AwsOutcome& outcome) const;
99 std::string error_msg(const std::string& key, const std::string& err) const;
101 Status copy(const Path& src, const Path& dst);
103 Status copy_dir(const Path& src, const Path& dst);
104 Status get_key(const Path& path, std::string* key) const;
105
106private:
107 S3Conf _s3_conf;
108 // TODO(cyx): We can use std::atomic<std::shared_ptr> since c++20.
109 mutable std::mutex _client_mu;
110 std::shared_ptr<Aws::S3::S3Client> _client;
111 std::shared_ptr<Aws::Utils::Threading::PooledThreadExecutor> _executor;
112};
