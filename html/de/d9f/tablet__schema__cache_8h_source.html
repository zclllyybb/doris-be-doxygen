31class TabletSchemaCache {
32public:
33 static void create_global_schema_cache() {
34 DCHECK(_s_instance == nullptr);
35 static TabletSchemaCache instance;
36 _s_instance = &instance;
37 std::thread t(&TabletSchemaCache::_recycle, _s_instance);
38 t.detach();
39 }
40
41 static TabletSchemaCache* instance() { return _s_instance; }
42
43 TabletSchemaSPtr insert(const std::string& key) {
44 DCHECK(_s_instance != nullptr);
45 std::lock_guard guard(_mtx);
46 auto iter = _cache.find(key);
47 if (iter == _cache.end()) {
48 TabletSchemaSPtr tablet_schema_ptr = std::make_shared<TabletSchema>();
49 TabletSchemaPB pb;
50 pb.ParseFromString(key);
51 tablet_schema_ptr->init_from_pb(pb);
52 _cache[key] = tablet_schema_ptr;
53 DorisMetrics::instance()->tablet_schema_cache_count->increment(1);
54 DorisMetrics::instance()->tablet_schema_cache_memory_bytes->increment(
55 tablet_schema_ptr->mem_size());
56 return tablet_schema_ptr;
57 }
58 return iter->second;
59 }
60
61private:
65 void _recycle() {
66 int64_t tablet_schema_cache_recycle_interval = 86400; // s, one day
67 for (;;) {
68 std::this_thread::sleep_for(std::chrono::seconds(tablet_schema_cache_recycle_interval));
69 std::lock_guard guard(_mtx);
70 LOG(INFO) << "Tablet Schema Cache Capacity " << _cache.size();
71 for (auto iter = _cache.begin(), last = _cache.end(); iter != last;) {
72 if (iter->second.unique()) {
73 DorisMetrics::instance()->tablet_schema_cache_memory_bytes->increment(
74 -iter->second->mem_size());
75 DorisMetrics::instance()->tablet_schema_cache_count->increment(-1);
76 iter = _cache.erase(iter);
77 } else {
78 ++iter;
79 }
80 }
81 }
82 }
83
84private:
85 static inline TabletSchemaCache* _s_instance = nullptr;
86 std::mutex _mtx;
87 std::unordered_map<std::string, TabletSchemaSPtr> _cache;
88};
