32// Template metaprogramming utility functions.
33//
34// This code is compiled directly on many platforms, including client
35// platforms like Windows, Mac, and embedded systems. Before making
36// any changes here, make sure that you're not breaking any platforms.
37//
38//
39// The names chosen here reflect those used in tr1 and the boost::mpl
40// library, there are similar operations used in the Loki library as
41// well. I prefer the boost names for 2 reasons:
42// 1. I think that portions of the Boost libraries are more likely to
43// be included in the c++ standard.
44// 2. It is not impossible that some of the boost libraries will be
45// included in our own build in the future.
46// Both of these outcomes means that we may be able to directly replace
47// some of these with boost equivalents.
48//

53// Types small_ and big_ are guaranteed such that sizeof(small_) <
54// sizeof(big_)
55typedef char small_;
56
57struct big_ {
58 char dummy[2];
59};
60
61// Types YesType and NoType are guaranteed such that sizeof(YesType) <
62// sizeof(NoType).
63typedef small_ YesType;
64typedef big_ NoType;

66// Identity metafunction.
67template <class T>
68struct identity_ {
69 typedef T type;
70};

72// integral_constant, defined in tr1, is a wrapper for an integer
73// value. We don't really need this generality; we could get away
74// with hardcoding the integer type to bool. We use the fully
75// general integer_constant for compatibility with tr1.
76
77template <class T, T v>
78struct integral_constant {
79 static const T value = v;
80 typedef T value_type;
81 typedef integral_constant<T, v> type;
82};
83
84template <class T, T v>
85const T integral_constant<T, v>::value;
86
87// Abbreviations: true_type and false_type are structs that represent boolean
88// true and false values. Also define the boost::mpl versions of those names,
89// true_ and false_.
90typedef integral_constant<bool, true> true_type;
91typedef integral_constant<bool, false> false_type;
92typedef true_type true_;
93typedef false_type false_;

95template <class T>
96struct is_non_const_reference : false_type {};
97template <class T>
98struct is_non_const_reference<T&> : true_type {};
99template <class T>
100struct is_non_const_reference<const T&> : false_type {};

102template <class T>
103struct is_const : false_type {};
104template <class T>
105struct is_const<const T> : true_type {};

107template <class T>
108struct is_void : false_type {};
109template <>
110struct is_void<void> : true_type {};

112// if_ is a templatize conditional statement.
113// if_<cond, A, B> is a compile time evaluation of cond.
114// if_<>::type contains A if cond is true, B otherwise.
115template <bool cond, typename A, typename B>
116struct if_ {
117 typedef A type;
118};
119
120template <typename A, typename B>
121struct if_<false, A, B> {
122 typedef B type;
123};

125// type_equals_ is a template type comparator, similar to Loki IsSameType.
126// type_equals_<A, B>::value is true iff "A" is the same type as "B".
127//
128// New code should prefer base::is_same, defined in base/type_traits.h.
129// It is functionally identical, but is_same is the standard spelling.
130template <typename A, typename B>
131struct type_equals_ : public false_ {};
132
133template <typename A>
134struct type_equals_<A, A> : public true_ {};
135
136// and_ is a template && operator.
137// and_<A, B>::value evaluates "A::value && B::value".
138template <typename A, typename B>
139struct and_ : public integral_constant<bool, (A::value && B::value)> {};
140
141// or_ is a template || operator.
142// or_<A, B>::value evaluates "A::value || B::value".
143template <typename A, typename B>
144struct or_ : public integral_constant<bool, (A::value || B::value)> {};

146// Used to determine if a type is a struct/union/class. Inspired by Boost's
147// is_class type_trait implementation.
148struct IsClassHelper {
149 template <typename C>
150 static YesType Test(void (C::*)(void));
151
152 template <typename C>
153 static NoType Test(...);
154};
155
156template <typename T>
157struct is_class : integral_constant<bool, sizeof(IsClassHelper::Test<T>(0)) == sizeof(YesType)> {};
