33// Encode a segment short key indices to one ShortKeyPage. This version
34// only accepts binary key, client should assure that input key is sorted,
35// otherwise error could happens. This builder would arrange the page body in the
36// following format:
37// ShortKeyPageBody := KeyContent^NumEntry, KeyOffset(vint)^NumEntry
38// NumEntry, KeyBytes, OffsetBytes is stored in ShortKeyFooterPB
39// Usage:
40// ShortKeyIndexBuilder builder(segment_id, num_rows_per_block);
41// builder.add_item(key1);
42// ...
43// builder.add_item(keyN);
44// builder.finalize(segment_size, num_rows, &slices);
45// NOTE: This is used for BetaRowset
46// TODO(zc):
47// 1. If this can leverage binary page to save key and offset data
48// 2. Extending this to save in a BTree like struct, which can index full key
49// more than short key
50class ShortKeyIndexBuilder {
51public:
52 ShortKeyIndexBuilder(uint32_t segment_id, uint32_t num_rows_per_block)
53 : _segment_id(segment_id), _num_rows_per_block(num_rows_per_block), _num_items(0) {}
54
55 Status add_item(const Slice& key);
56
57 uint64_t size() { return _key_buf.size() + _offset_buf.size(); }
58
59 Status finalize(uint32_t num_rows, std::vector<Slice>* body, segment_v2::PageFooterPB* footer);
60
61private:
62 uint32_t _segment_id;
63 uint32_t _num_rows_per_block;
64 uint32_t _num_items;
65
66 faststring _key_buf;
67 faststring _offset_buf;
68};

72// An Iterator to iterate one short key index.
73// Client can use this class to iterator all items in this index.
74class ShortKeyIndexIterator {
75public:
76 using iterator_category = std::random_access_iterator_tag;
77 using value_type = Slice;

79 using reference = Slice&;
80 using difference_type = ssize_t;
81
82 ShortKeyIndexIterator(const ShortKeyIndexDecoder* decoder, uint32_t ordinal = 0)
83 : _decoder(decoder), _ordinal(ordinal) {}
84
85 ShortKeyIndexIterator& operator-=(ssize_t step) {
86 _ordinal -= step;
87 return *this;
88 }
89
90 ShortKeyIndexIterator& operator+=(ssize_t step) {
91 _ordinal += step;
92 return *this;
93 }
94
95 ShortKeyIndexIterator& operator++() {
96 _ordinal++;
97 return *this;
98 }
99
100 ShortKeyIndexIterator& operator--() {
101 --_ordinal;
102 return *this;

104
105 bool operator!=(const ShortKeyIndexIterator& other) {
106 return _ordinal != other._ordinal || _decoder != other._decoder;
107 }
108
109 bool operator==(const ShortKeyIndexIterator& other) {
110 return _ordinal == other._ordinal && _decoder == other._decoder;
111 }
112

114 return _ordinal - other._ordinal;
115 }
116
117 bool valid() const;
118

120
121 ssize_t ordinal() const { return _ordinal; }
122
123private:

128// Used to decode short key to header and encoded index data.
129// Usage:
130// ShortKeyIndexDecoder decoder;
131// decoder.parse(body, footer);
132// auto iter = decoder.lower_bound(key);
133class ShortKeyIndexDecoder {
134public:
135 ShortKeyIndexDecoder() : _parsed(false) {}
136
137 // client should assure that body is available when this class is used
138 Status parse(const Slice& body, const segment_v2::ShortKeyFooterPB& footer);
139
140 ShortKeyIndexIterator begin() const {
141 DCHECK(_parsed);
142 return {this, 0};
143 }
144
145 ShortKeyIndexIterator end() const {
146 DCHECK(_parsed);
147 return {this, num_items()};
148 }
149
150 // Return an iterator which locates at the first item who is
151 // equal with or greater than the given key.
152 // NOTE: If one key is the prefix of other key, this function thinks
153 // that longer key is greater than the shorter key.
154 ShortKeyIndexIterator lower_bound(const Slice& key) const {
155 DCHECK(_parsed);
156 return seek<true>(key);
157 }
158
159 // Return the iterator which locates the first item greater than the
160 // input key.
161 ShortKeyIndexIterator upper_bound(const Slice& key) const {
162 DCHECK(_parsed);
163 return seek<false>(key);
164 }
165
166 uint32_t num_items() const {
167 DCHECK(_parsed);
168 return _footer.num_items();
169 }
170
171 uint32_t num_rows_per_block() const {
172 DCHECK(_parsed);
173 return _footer.num_rows_per_block();
174 }
175
176 Slice key(ssize_t ordinal) const {
177 DCHECK(_parsed);
178 DCHECK(ordinal >= 0 && ordinal < num_items());
179 return {_key_data.data + _offsets[ordinal], _offsets[ordinal + 1] - _offsets[ordinal]};
180 }
181
182private:
183 template <bool lower_bound>
184 ShortKeyIndexIterator seek(const Slice& key) const {
185 auto comparator = [](const Slice& lhs, const Slice& rhs) { return lhs.compare(rhs) < 0; };
186 if (lower_bound) {
187 return std::lower_bound(begin(), end(), key, comparator);
188 } else {
189 return std::upper_bound(begin(), end(), key, comparator);
190 }
191 }
192
193private:
194 bool _parsed;
195
196 // All following fields are only valid after parse has been executed successfully
197 segment_v2::ShortKeyFooterPB _footer;
198 std::vector<uint32_t> _offsets;
199 Slice _key_data;
200};
201
202inline Slice ShortKeyIndexIterator::operator*() const {
203 return _decoder->key(_ordinal);
204}
205
206inline bool ShortKeyIndexIterator::valid() const {
207 return _ordinal >= 0 && _ordinal < _decoder->num_items();
208}
