49template <typename LockType, typename T>
50class InternalQueueBase {
51public:
52 struct Node {
53 public:
54 Node() : parent_queue(nullptr), next_node(nullptr), prev_node(nullptr) {}
55 virtual ~Node() {}
56
58 bool in_queue() const { return parent_queue != nullptr; }
59
61 T* next() const {

63 return reinterpret_cast<T*>(next_node);
64 }
65 T* prev() const {
66 std::lock_guard<LockType> lock(parent_queue->lock_);
67 return reinterpret_cast<T*>(prev_node);
68 }
69
70 private:
71 friend class InternalQueueBase<LockType, T>;
72

77 };

79 InternalQueueBase() : head_(nullptr), tail_(nullptr), size_(0) {}

83 T* head() const {
84 std::lock_guard<LockType> lock(lock_);
85 if (empty()) return nullptr;
86 return reinterpret_cast<T*>(head_);

92 std::lock_guard<LockType> lock(lock_);
93 if (empty()) return nullptr;

98 void enqueue(T* n) {
99 Node* node = (Node*)n;
100 DCHECK(node->next_node == nullptr);
101 DCHECK(node->prev_node == nullptr);
102 DCHECK(node->parent_queue == nullptr);
103 node->parent_queue = this;
104 {
105 std::lock_guard<LockType> lock(lock_);
106 if (tail_ != nullptr) tail_->next_node = node;
107 node->prev_node = tail_;
108 tail_ = node;
109 if (head_ == nullptr) head_ = node;
110 ++size_;
111 }
112 }
113
116 T* dequeue() {
117 Node* result = nullptr;
118 {
119 std::lock_guard<LockType> lock(lock_);
120 if (empty()) return nullptr;
121 --size_;
122 result = head_;
123 head_ = head_->next_node;
124 if (head_ == nullptr) {
125 tail_ = nullptr;
126 } else {
127 head_->prev_node = nullptr;
128 }
129 }
130 DCHECK(result != nullptr);
131 result->next_node = result->prev_node = nullptr;
132 result->parent_queue = nullptr;
133 return reinterpret_cast<T*>(result);
134 }
135
138 T* pop_back() {
139 Node* result = nullptr;
140 {
141 std::lock_guard<LockType> lock(lock_);
142 if (empty()) return nullptr;
143 --size_;
144 result = tail_;
145 tail_ = tail_->prev_node;
146 if (tail_ == nullptr) {
147 head_ = nullptr;
148 } else {
149 tail_->next_node = nullptr;
150 }
151 }
152 DCHECK(result != nullptr);
153 result->next_node = result->prev_node = nullptr;
154 result->parent_queue = nullptr;
155 return reinterpret_cast<T*>(result);
156 }

160 bool remove(T* n) {
161 Node* node = (Node*)n;
162 if (node->parent_queue != this) return false;
163 {
164 std::lock_guard<LockType> lock(lock_);
165 if (node->next_node == nullptr && node->prev_node == nullptr) {
166 // Removing only node
167 DCHECK(node == head_);
168 DCHECK(tail_ == node);
169 head_ = tail_ = nullptr;
170 --size_;
171 node->parent_queue = nullptr;
172 return true;
173 }
174
175 if (head_ == node) {
176 DCHECK(node->prev_node == nullptr);
177 head_ = node->next_node;
178 } else {
179 DCHECK(node->prev_node != nullptr);
180 node->prev_node->next_node = node->next_node;
181 }
182
183 if (node == tail_) {
184 DCHECK(node->next_node == nullptr);
185 tail_ = node->prev_node;
186 } else if (node->next_node != nullptr) {
187 node->next_node->prev_node = node->prev_node;
188 }
189 --size_;
190 }
191 node->next_node = node->prev_node = nullptr;
192 node->parent_queue = nullptr;
193 return true;
194 }
195
197 void clear() {
198 std::lock_guard<LockType> lock(lock_);
199 Node* cur = head_;
200 while (cur != nullptr) {
201 Node* tmp = cur;
202 cur = cur->next_node;
203 tmp->prev_node = tmp->next_node = nullptr;
204 tmp->parent_queue = nullptr;
205 }
206 size_ = 0;
207 head_ = tail_ = nullptr;
208 }
209
210 int size() const { return size_; }
211 bool empty() const { return head_ == nullptr; }
212
214 bool contains(const T* target) const { return target->parent_queue == this; }
215
217 bool validate() {
218 int num_elements_found = 0;
219 std::lock_guard<LockType> lock(lock_);
220 if (head_ == nullptr) {
221 if (tail_ != nullptr) return false;
222 if (size() != 0) return false;
223 return true;
224 }
225
226 if (head_->prev_node != nullptr) return false;
227 Node* current = head_;
228 while (current != nullptr) {
229 if (current->parent_queue != this) return false;
230 ++num_elements_found;
231 Node* next_node = current->next_node;
232 if (next_node == nullptr) {
233 if (current != tail_) return false;
234 } else {
235 if (next_node->prev_node != current) return false;
236 }
237 current = next_node;
238 }
239 if (num_elements_found != size()) return false;
240 return true;
241 }

243 // Iterate over elements of queue, calling 'fn' for each element. If 'fn' returns
244 // false, terminate iteration. It is invalid to call other InternalQueue methods
245 // from 'fn'.
246 void iterate(std::function<bool(T*)> fn) {
247 std::lock_guard<LockType> lock(lock_);
248 for (Node* current = head_; current != nullptr; current = current->next_node) {
249 if (!fn(reinterpret_cast<T*>(current))) return;
250 }
251 }
252
254 std::string DebugString() {
255 std::stringstream ss;
256 ss << "(";
257 {
258 std::lock_guard<LockType> lock(lock_);
259 Node* curr = head_;
260 while (curr != nullptr) {
261 ss << (void*)curr;
262 curr = curr->next_node;
263 }
264 }
265 ss << ")";
266 return ss.str();
267 }
268
269private:
270 friend struct Node;
271 mutable LockType lock_;
272 Node *head_, *tail_;
273 int size_;
274};
