37// chartorune copies (decodes) at most UTFmax bytes starting at s to
38// one rune, pointed to by r, and returns the number of bytes consumed.
39// If the input is not exactly in UTF format, chartorune will set *r
40// to Runeerror and return 1.
41//
42// Note: There is no special case for a "null-terminated" string. A
43// string whose first byte has the value 0 is the UTF8 encoding of the
44// Unicode value 0 (i.e., ASCII NULL). A byte value of 0 is illegal
45// anywhere else in a UTF sequence.
46
47int chartorune(Rune* r, const char* s);
48
49// charntorune is like chartorune, except that it will access at most
50// n bytes of s. If the UTF sequence is incomplete within n bytes,
51// charntorune will set *r to Runeerror and return 0. If it is complete
52// but not in UTF format, it will set *r to Runeerror and return 1.
53//
54// Added 2004-09-24 by Wei-Hwa Huang
55
56int charntorune(Rune* r, const char* s, int n);

155// The following routines test types and modify cases for Unicode
156// characters. Unicode defines some characters as letters and
157// specifies three cases: upper, lower, and title. Mappings among the
158// cases are also defined, although they are not exhaustive: some
159// upper case letters have no lower case mapping, and so on. Unicode
160// also defines several character properties, a subset of which are
161// checked by these routines. These routines are based on Unicode
162// version 3.0.0.
163//
164// NOTE: The routines are implemented in C, so the boolean functions
165// (e.g., isupperrune) return 0 for false and 1 for true.
166//
167//
168// toupperrune, tolowerrune, and totitlerune are the Unicode case
169// mappings. These routines return the character unchanged if it has
170// no defined mapping.
171
172Rune toupperrune(Rune r);
173Rune tolowerrune(Rune r);
174Rune totitlerune(Rune r);
