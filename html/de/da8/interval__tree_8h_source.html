46// Implements an Interval Tree.
47//
48// An Interval Tree is a data structure which stores a set of intervals and supports
49// efficient searches to determine which intervals in that set overlap a query
50// point or interval. These operations are O(lg n + k) where 'n' is the number of
51// intervals in the tree and 'k' is the number of results returned for a given query.
52//
53// This particular implementation is a static tree -- intervals may not be added or
54// removed once the tree is instantiated.
55//
56// This class also assumes that all intervals are "closed" intervals -- the intervals
57// are inclusive of their start and end points.
58//
59// The Traits class should have the following members:
60// Traits::point_type
61// a typedef for what a "point" in the range is
62//
63// Traits::interval_type
64// a typedef for an interval
65//
66// static point_type get_left(const interval_type &)
67// static point_type get_right(const interval_type &)
68// accessors which fetch the left and right bound of the interval, respectively.
69//
70// static int compare(const point_type &a, const point_type &b)
71// return < 0 if a < b, 0 if a == b, > 0 if a > b
72//
73// See interval_tree-test.cc for an example Traits class for 'int' ranges.
74template <class Traits>
75class IntervalTree {
76private:
77 // Import types from the traits class to make code more readable.
78 typedef typename Traits::interval_type interval_type;
79 typedef typename Traits::point_type point_type;
80
81 // And some convenience types.
82 typedef std::vector<interval_type> IntervalVector;
83 typedef interval_tree_internal::ITNode<Traits> node_type;
84
85public:
86 // Construct an Interval Tree containing the given set of intervals.
87 explicit IntervalTree(const IntervalVector& intervals);
88
89 ~IntervalTree();
90
91 // Find all intervals in the tree which contain the query point.
92 // The resulting intervals are added to the 'results' vector.
93 // The vector is not cleared first.
94 //
95 // NOTE: 'QueryPointType' is usually point_type, but can be any other
96 // type for which there exists the appropriate Traits::Compare(...) method.
97 template <class QueryPointType>
98 void FindContainingPoint(const QueryPointType& query, IntervalVector* results) const;
99
100 // For each of the query points in the STL container 'queries', find all
101 // intervals in the tree which may contain those points. Calls 'cb(point, interval)'
102 // for each such interval.
103 //
104 // The points in the query container must be comparable to 'point_type'
105 // using Traits::Compare().
106 //
107 // The implementation sequences the calls to 'cb' with the following guarantees:
108 // 1) all of the results corresponding to a given interval will be yielded in at
109 // most two "groups" of calls (i.e. sub-sequences of calls with the same interval).
110 // 2) within each "group" of calls, the query points will be in ascending order.
111 //
112 // For example, the callback sequence may be:
113 //
114 // cb(q1, interval_1) -
115 // cb(q2, interval_1) | first group of interval_1
116 // cb(q6, interval_1) |
117 // cb(q7, interval_1) -
118 //
119 // cb(q2, interval_2) -
120 // cb(q3, interval_2) | first group of interval_2
121 // cb(q4, interval_2) -
122 //
123 // cb(q3, interval_1) -
124 // cb(q4, interval_1) | second group of interval_1
125 // cb(q5, interval_1) -
126 //
127 // cb(q2, interval_3) -
128 // cb(q3, interval_3) | first group of interval_3
129 // cb(q4, interval_3) -
130 //
131 // cb(q5, interval_2) -
132 // cb(q6, interval_2) | second group of interval_2
133 // cb(q7, interval_2) -
134 //
135 // REQUIRES: The input points must be pre-sorted or else this will return invalid
136 // results.
137 template <class Callback, class QueryContainer>
138 void ForEachIntervalContainingPoints(const QueryContainer& queries, const Callback& cb) const;
139
140 // Find all intervals in the tree which intersect the given interval.
141 // The resulting intervals are added to the 'results' vector.
142 // The vector is not cleared first.
143 template <class QueryPointType>
144 void FindIntersectingInterval(const QueryPointType& lower_bound,
145 const QueryPointType& upper_bound, IntervalVector* results) const;
146
147private:
148 static void Partition(const IntervalVector& in, point_type* split_point, IntervalVector* left,
149 IntervalVector* overlapping, IntervalVector* right);
150
151 // Create a node containing the given intervals, recursively splitting down the tree.
152 static node_type* CreateNode(const IntervalVector& intervals);
153
154 node_type* root_;
155
156 DISALLOW_COPY_AND_ASSIGN(IntervalTree);
157};
