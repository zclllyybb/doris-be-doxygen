25// This Bloom filter is implemented using block-based Bloom filter algorithm
26// from Putze et al.'s "Cache-, Hash- and Space-Efficient Bloom filters". The basic
27// idea is to hash the item to a tiny Bloom filter which size fit a single cache line
28// or smaller. This implementation sets 8 bits in each tiny Bloom filter. Each tiny
29// Bloom filter is 32 bytes to take advantage of 32-byte SIMD instruction.
30class BlockSplitBloomFilter : public BloomFilter {
31public:
32 void add_hash(uint64_t hash) override;
33
34 bool test_hash(uint64_t hash) const override;
35 bool contains(const BloomFilter&) const override { return true; }
36
37private:
38 // Bytes in a tiny Bloom filter block.
39 static constexpr int BYTES_PER_BLOCK = 32;
40 // The number of bits to set in a tiny Bloom filter block
41 static constexpr int BITS_SET_PER_BLOCK = 8;
42
43 static constexpr uint32_t SALT[BITS_SET_PER_BLOCK] = {0x47b6137bU, 0x44974d91U, 0x8824ad5bU,
44 0xa2b7289dU, 0x705495c7U, 0x2df1424bU,
45 0x9efc4947U, 0x5c6bfb31U};
46
47 struct BlockMask {
48 uint32_t item[BITS_SET_PER_BLOCK];
49 };
50
51private:
52 void _set_masks(uint32_t key, BlockMask& block_mask) const {
53 for (int i = 0; i < BITS_SET_PER_BLOCK; ++i) {
54 block_mask.item[i] = key * SALT[i];
55 }
56
57 for (int i = 0; i < BITS_SET_PER_BLOCK; ++i) {
58 block_mask.item[i] = block_mask.item[i] >> 27;
59 }
60
61 for (int i = 0; i < BITS_SET_PER_BLOCK; ++i) {
62 block_mask.item[i] = uint32_t(0x1) << block_mask.item[i];
63 }
64 }
65};
