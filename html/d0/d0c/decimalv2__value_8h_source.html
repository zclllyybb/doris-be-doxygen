94 bool from_olap_decimal(int64_t int_value, int64_t frac_value) {
95 bool success = true;
96 bool is_negative = (int_value < 0 || frac_value < 0);
97 if (is_negative) {
98 int_value = std::abs(int_value);
99 frac_value = std::abs(frac_value);
100 }
101
102 //if (int_value > MAX_INT_VALUE) {
103 // int_value = MAX_INT_VALUE;
104 // success = false;
105 //}
106
107 if (frac_value > MAX_FRAC_VALUE) {
108 frac_value = MAX_FRAC_VALUE;
109 success = false;
110 }
111
112 _value = static_cast<int128_t>(int_value) * ONE_BILLION + frac_value;
113 if (is_negative) _value = -_value;
114
115 return success;
116 }
