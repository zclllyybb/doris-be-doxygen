481 // Dispatch is written in a way that maximizes the performance:
482 // 1. Always memcpy 8 times bytes
483 // 2. Use switch case extension to generate fast dispatching table
484 // 3. Funcs are named callables that can be force_inlined
485 //
486 // NOTE: It relies on Little Endianness
487 //
488 // NOTE: It requires padded to 8 bytes keys (IOW you cannot pass
489 // std::string here, but you can pass i.e. ColumnString::getDataAt()),
490 // since it copies 8 bytes at a time.
491 template <typename Self, typename KeyHolder, typename Func>
492 static auto ALWAYS_INLINE dispatch(Self& self, KeyHolder&& key_holder, Func&& func) {
493 StringHashTableHash hash;
494 const doris::StringRef& x = key_holder_get_key(key_holder);
495 const size_t sz = x.size;
496 if (sz == 0) {
497 key_holder_discard_key(key_holder);
498 return func(self.m0, std::forward<KeyHolder>(key_holder), 0);
499 }
500
501 if (x.data[sz - 1] == 0) {
502 // Strings with trailing zeros are not representable as fixed-size
503 // string keys. Put them to the generic table.
504 return func(self.ms, std::forward<KeyHolder>(key_holder), hash(x));
505 }
506
507 const char* p = x.data;
508 // pending bits that needs to be shifted out
509 const char s = (-sz & 7) * 8;
510 union {
511 StringKey8 k8;
512 StringKey16 k16;
513 StringKey24 k24;
514 doris::vectorized::UInt64 n[3];
515 };
516 switch ((sz - 1) >> 3) {
517 case 0: // 1..8 bytes
518 {
519 // first half page
520 if ((reinterpret_cast<uintptr_t>(p) & 2048) == 0) {
521 memcpy(&n[0], p, 8);
522 n[0] &= -1ULL >> s;
523 } else {
524 const char* lp = x.data + x.size - 8;
525 memcpy(&n[0], lp, 8);
526 n[0] >>= s;
527 }
528 key_holder_discard_key(key_holder);
529 return func(self.m1, k8, hash(k8));
530 }
531 case 1: // 9..16 bytes
532 {
533 memcpy(&n[0], p, 8);
534 const char* lp = x.data + x.size - 8;
535 memcpy(&n[1], lp, 8);
536 n[1] >>= s;
537 key_holder_discard_key(key_holder);
538 return func(self.m2, k16, hash(k16));
539 }
540 case 2: // 17..24 bytes
541 {
542 memcpy(&n[0], p, 16);
543 const char* lp = x.data + x.size - 8;
544 memcpy(&n[2], lp, 8);
545 n[2] >>= s;
546 key_holder_discard_key(key_holder);
547 return func(self.m3, k24, hash(k24));
548 }
549 default: // >= 25 bytes
550 {
551 return func(self.ms, std::forward<KeyHolder>(key_holder), hash(x));
552 }
553 }
554 }
