31// Row source represent row location in multi-segments
32// use a uint16_t to store info
33// the lower 15 bits means segment_id in segment pool, and the higher 1 bits means agg flag.
34// In unique-key, agg flags means this key should be deleted, this comes from two way: old version
35// key or delete_sign.
36class RowSource {
37public:
38 RowSource(uint16_t data) : _data(data) {}
39 RowSource(uint16_t source_num, bool agg_flag);
40
41 uint16_t get_source_num() const;
42 bool agg_flag() const;
43 void set_agg_flag(bool agg_flag);
44 uint16_t data() const;
45
46private:
47 uint16_t _data;
48 static const uint16_t SOURCE_FLAG = 0x7FFF;
49 static const uint16_t AGG_FLAG = 0x8000;
50};

52/* rows source buffer
53this buffer should have a memory limit, once reach memory limit, write
54buffer data to tmp file.
55usage:
56 RowSourcesBuffer buffer(tablet_id, tablet_storage_path, reader_type);
57 buffer.append()
58 buffer.append()
59 buffer.flush()
60 buffer.seek_to_begin()
61 while (buffer.has_remaining().ok()) {
62 auto cur = buffer.current().get_source_num();
63 auto same = buffer.same_source_count(cur, limit);
64 // do copy block data
65 buffer.advance(same);
66 }
67*/
68class RowSourcesBuffer {
69public:
70 RowSourcesBuffer(int64_t tablet_id, const std::string& tablet_path, ReaderType reader_type)
71 : _tablet_id(tablet_id),
72 _tablet_path(tablet_path),
73 _reader_type(reader_type),
74 _buffer(ColumnUInt16::create()) {}
75
76 ~RowSourcesBuffer() {
77 _reset_buffer();
78 if (_fd > 0) {
79 ::close(_fd);
80 }
81 }
82
83 // write batch row source
84 Status append(const std::vector<RowSource>& row_sources);
85 Status flush();
86
87 RowSource current() {
88 DCHECK(_buf_idx < _buffer->size());
89 return RowSource(_buffer->get_element(_buf_idx));
90 }
91 void advance(int32_t step = 1) {
92 DCHECK(_buf_idx + step <= _buffer->size());
93 _buf_idx += step;
94 }
95
96 uint64_t buf_idx() { return _buf_idx; }
97 uint64_t total_size() { return _total_size; }
98 uint64_t buffered_size() { return _buffer->size(); }
99 void set_agg_flag(uint64_t index, bool agg);
100
101 Status has_remaining();
102
103 Status seek_to_begin();
104
105 size_t same_source_count(uint16_t source, size_t limit);
106
107 // return continous agg_flag=true count from index
108 size_t continuous_agg_count(uint64_t index);
109
110private:
111 Status _create_buffer_file();
112 Status _serialize();
113 Status _deserialize();
114 void _reset_buffer() {
115 _buffer->clear();
116 _buf_idx = 0;
117 }
118
119private:
120 int64_t _tablet_id;
121 std::string _tablet_path;
122 ReaderType _reader_type;
123 uint64_t _buf_idx = 0;
124 int _fd = -1;
125 ColumnUInt16::MutablePtr _buffer;
126 uint64_t _total_size = 0;
127};

219 // VerticalMergeIterator takes the ownership of input iterators

232
233 ~VerticalHeapMergeIterator() override {
234 while (!_merge_heap.empty()) {

251private:

254private:

274 VMergeHeap _merge_heap;

276 int _block_row_max = 0;
277 KeysType _keys_type;
