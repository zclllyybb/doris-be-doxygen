44// Superclass of all data sinks.
45class DataSink {
46public:
47 DataSink() : _closed(false) {}
48 virtual ~DataSink() {}
49
50 virtual Status init(const TDataSink& thrift_sink);
51
52 // Setup. Call before send(), Open(), or Close().
53 // Subclasses must call DataSink::Prepare().
54 virtual Status prepare(RuntimeState* state);
55
56 // Setup. Call before send() or close().
57 virtual Status open(RuntimeState* state) = 0;
58
59 // Send a Block into this sink.
60 virtual Status send(RuntimeState* state, vectorized::Block* block, bool eos = false) {
61 return Status::NotSupported("Not support send block");
62 }
63 // Releases all resources that were allocated in prepare()/send().
64 // Further send() calls are illegal after calling close().
65 // It must be okay to call this multiple times. Subsequent calls should
66 // be ignored.
67 virtual Status close(RuntimeState* state, Status exec_status) {
68 profile()->add_to_span();
69 _closed = true;
70 return Status::OK();
71 }
72
73 // Creates a new data sink from thrift_sink. A pointer to the
74 // new sink is written to *sink, and is owned by the caller.
75 static Status create_data_sink(ObjectPool* pool, const TDataSink& thrift_sink,
76 const std::vector<TExpr>& output_exprs,
77 const TPlanFragmentExecParams& params,
78 const RowDescriptor& row_desc, RuntimeState* state,
79 std::unique_ptr<DataSink>* sink, DescriptorTbl& desc_tbl);
80
81 // Returns the runtime profile for the sink.
82 virtual RuntimeProfile* profile() = 0;
83
84 virtual void set_query_statistics(std::shared_ptr<QueryStatistics> statistics) {
85 _query_statistics = statistics;
86 }
87
88 void end_send_span() {
89 if (_send_span) {
90 _send_span->End();
91 }
92 }
93
94protected:
95 // Set to true after close() has been called. subclasses should check and set this in
96 // close().
97 bool _closed;
98 std::string _name;
99
100 // Maybe this will be transferred to BufferControlBlock.
101 std::shared_ptr<QueryStatistics> _query_statistics;
102
103 OpentelemetrySpan _send_span {};
104};
105
106} // namespace doris
