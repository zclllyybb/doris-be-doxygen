36class HttpParser {
37public:
38 enum ParseState {
39 PARSE_OK, // reach trunk data, you can read data
40 PARSE_AGAIN, // continue call this function
41 PARSE_DONE, // trunk is over
42 PARSE_ERROR, // parse failed.
43 };
44
45 // Used to parse http Chunked Transfer Coding(rfc2616 3.6.1)
46 // Params:
47 // buf network buffer's pointer, changed when receive data
48 // when this function return, *buf point to position where
49 // next call will begin processing
50 // buf_len length of receive http data
51 // ctx parse context, used to save parse state and data size
52 //
53 // Returns:
54 // PARSE_OK return this means we reach chunk-data, *buf point to begin of data
55 // size of chunk-data is saved in ctx->size. Caller need subtract size
56 // consumed from ctx->size before next call of this function.
57 // PARSE_AGAIN return this means that caller need to call this function with new data
58 // from network
59 // PARSE_DONE All of chunks readed
60 // PARSE_ERROR Error happened
61 static ParseState http_parse_chunked(const uint8_t** buf, const int64_t buf_len,
62 HttpChunkParseCtx* ctx);
63};
