31struct AggregateFunctionApproxCountDistinctData {
32 HyperLogLog hll_data;
33
34 void add(uint64_t hash_value) {
35 if (hash_value != 0) {
36 hll_data.update(hash_value);
37 }
38 }
39
40 void merge(const AggregateFunctionApproxCountDistinctData& rhs) {
41 hll_data.merge(rhs.hll_data);
42 }
43
44 void write(BufferWritable& buf) const {
45 std::string result;
46 result.resize(hll_data.max_serialized_size());
47 int size = hll_data.serialize((uint8_t*)result.data());
48 result.resize(size);
49 write_binary(result, buf);
50 }
51
52 void read(BufferReadable& buf) {
53 StringRef result;
54 read_binary(result, buf);
55 Slice data = Slice(result.data, result.size);
56 hll_data.deserialize(data);
57 }
58
59 int64_t get() const { return hll_data.estimate_cardinality(); }
60
61 void reset() { hll_data.clear(); }
62};

64template <typename ColumnDataType>
65class AggregateFunctionApproxCountDistinct final
66 : public IAggregateFunctionDataHelper<
67 AggregateFunctionApproxCountDistinctData,
68 AggregateFunctionApproxCountDistinct<ColumnDataType>> {
69public:
70 String get_name() const override { return "approx_count_distinct"; }
71
72 AggregateFunctionApproxCountDistinct(const DataTypes& argument_types_)
73 : IAggregateFunctionDataHelper<AggregateFunctionApproxCountDistinctData,
74 AggregateFunctionApproxCountDistinct<ColumnDataType>>(
75 argument_types_, {}) {}
76
77 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeInt64>(); }
78
79 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
80 Arena*) const override {
81 if constexpr (IsFixLenColumnType<ColumnDataType>::value) {
82 auto column = static_cast<const ColumnDataType*>(columns[0]);
83 auto value = column->get_element(row_num);
84 this->data(place).add(
85 HashUtil::murmur_hash64A((char*)&value, sizeof(value), HashUtil::MURMUR_SEED));
86 } else {
87 auto value = static_cast<const ColumnDataType*>(columns[0])->get_data_at(row_num);
88 StringVal sv = value.to_string_val();
89 uint64_t hash_value = HashUtil::murmur_hash64A(sv.ptr, sv.len, HashUtil::MURMUR_SEED);
90 this->data(place).add(hash_value);
91 }
92 }
93
94 void reset(AggregateDataPtr place) const override { this->data(place).reset(); }
95
96 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
97 Arena*) const override {
98 this->data(place).merge(this->data(rhs));
99 }
100
101 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
102 this->data(place).write(buf);
103 }
104
105 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
106 Arena*) const override {
107 this->data(place).read(buf);
108 }
109
110 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
111 auto& column = static_cast<ColumnInt64&>(to);
112 column.get_data().push_back(this->data(place).get());
113 }
114};
