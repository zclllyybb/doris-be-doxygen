26struct NodeInfo {
27 /*
28 There are two types of value with different structure: one is the
29 basic data type and array, and the other one is the Object. We
30 need to differentiate it here, because the actual value is stored
31 differently. For example, for basic data type and array, the
32 stored value is JsonbValue, but for Object data type, the stored
33 value is JsonbKeyValue.
34 The pointer addr points to the beginning of the data value, which
35 means that for basic data type, it points ot JsonbValue and for
36 the elements in the Object, it points to JsonbKeyValue.
37 The pointer, jsonb_value, points to the actually value part, which
38 means that for basic data type, it's the same with addr and for
39 the JsonbKeyValue, it points to the payload of it.
40 */
41 char* addr; // The offset of the address from the beginning
42 JsonbValue* jsonb_value; // The address of the value of current node
43};
44/*
45 * This class is used to update the given document. The given document
46 * is stored in a buffer. When created, the the updater points to the
47 * root value in the document, and pushPathKey and popPathKey can be
48 * used to go to different key paths in the document.
49 */
50class JsonbUpdater {
51public:
52 // buffer_size is the maximum number of bytes can be packed in document
53 JsonbUpdater(JsonbDocument* doc, uint buffer_size, double shrink_ratio = 0.5)
54 : document_(doc), buffer_size_(buffer_size), str_shrink_ratio_(shrink_ratio) {
55 root_ = document_->getValue();
56 clearPath();
57 }
58 JsonbValue* getRoot() { return root_; }
59 JsonbDocument* getDocument() { return document_; }
60 JsonbValue* getCurrent() {
61 return 0 == path_node_.size() ? nullptr : path_node_.back().jsonb_value;
62 }
63 // Remove the current node. It can't be root
64 JsonbErrType remove() {
65 // Remove root is not allowed
66 if (path_node_.size() == 1) {
67 return JsonbErrType::E_INVALID_OPER;
68 }
69
70 // Get the current node and because we want to delete it, pop it from stack
71 NodeInfo curr_node = path_node_.back();
72 int pack_size = (int)((char*)curr_node.jsonb_value - curr_node.addr +
73 curr_node.jsonb_value->numPackedBytes());
74
75 // Move the data after the current node to new place
76 if (!moveTo(curr_node.addr + pack_size, curr_node.addr)) {
77 return JsonbErrType::E_OUTOFMEMORY;
78 }
79 path_node_.pop_back();
80 return JsonbErrType::E_NONE;
81 }
82
83 JsonbErrType popPathKey() {
84 /* It's root, it means that it's the first elements in the
85 * document, which means that it points to the payload of the
86 * document. It doesn't make sense if we pop the key.
87 */
88 if (path_node_.size() <= 1) return JsonbErrType::E_INVALID_OPER;
89 path_node_.pop_back();
90 return JsonbErrType::E_NONE;
91 }
92
93 // For ArrayVal
94 JsonbErrType pushPathKey(int idx) {
95 // Check whether it's an array
96 if (nullptr == path_node_.back().jsonb_value || !path_node_.back().jsonb_value->isArray()) {
97 return JsonbErrType::E_NOTARRAY;
98 }
99
100 // Get the element in given idx
101 ArrayVal::iterator ite;
102 JsonbErrType ret = getElemInArray(idx, path_node_.back().jsonb_value, ite);
103 if (JsonbErrType::E_NONE != ret) return ret;
104
105 if (ite == ((ArrayVal*)path_node_.back().jsonb_value)->end())
106 return JsonbErrType::E_OUTOFBOUNDARY;
107
108 path_node_.push_back({(char*)(ArrayVal::pointer)ite, (ArrayVal::pointer)ite});
109 return JsonbErrType::E_NONE;
110 }
111
112 // For ObjectVal
113 JsonbErrType pushPathKey(const char* key, // Should be a
114 // null-terminated string
115 hDictFind handler = nullptr) {
116 if (nullptr == key) return JsonbErrType::E_INVALID_KEY_STRING;
117 return pushPathKey(key, (unsigned int)strlen(key), handler);
118 }
119 JsonbErrType pushPathKey(const char* key, unsigned int klen, hDictFind handler = nullptr) {
120 // Check node validity
121 if (nullptr == key) {
122 return JsonbErrType::E_INVALID_KEY_STRING;
123 }
124 // Check whether it's an object
125 if (nullptr == path_node_.back().jsonb_value ||
126 !path_node_.back().jsonb_value->isObject()) {
127 return JsonbErrType::E_NOTOBJ;
128 }
129 // Get current node and search the one we want
130 ObjectVal* current = static_cast<ObjectVal*>(path_node_.back().jsonb_value);
131 ObjectVal::iterator kv = current->search(key, klen, handler);
132 if (kv == current->end()) {
133 return JsonbErrType::E_KEYNOTEXIST;
134 } else {
135 JsonbKeyValue* fb_kv = static_cast<ObjectVal::iterator::pointer>(kv);
136 path_node_.push_back({(char*)fb_kv, fb_kv->value()});
137 }
138 return JsonbErrType::E_NONE;
139 }
140
141 // Update a field
142 JsonbErrType updateValue(const JsonbValue* value) { return updateValueInternal(value); }
143
144 // Append a value to an array
145 JsonbErrType appendValue(const JsonbValue* value) {
146 // Check whether it's an array
147 if (nullptr == path_node_.back().jsonb_value || !path_node_.back().jsonb_value->isArray()) {
148 return JsonbErrType::E_NOTARRAY;
149 }
150 return insertArrayInternal(((ArrayVal*)path_node_.back().jsonb_value)->end(),
151 (const char*)value,
152 (const char*)value + value->numPackedBytes());
153 }
154
155 // Insert a value to an array
156 JsonbErrType insertValue(int idx, const JsonbValue* value) {
157 ArrayVal::iterator ite;
158 JsonbErrType ret = getElemInArray(idx, path_node_.back().jsonb_value, ite);
159 if (JsonbErrType::E_NONE != ret) return ret;
160 return insertArrayInternal(ite, (const char*)value,
161 (const char*)value + value->numPackedBytes());
162 }
163
164 // Insert a range of value to an array
165 JsonbErrType insertValue(int idx, ArrayVal::const_iterator beg, ArrayVal::const_iterator end) {
166 ArrayVal::iterator ite;
167 JsonbErrType ret = getElemInArray(idx, path_node_.back().jsonb_value, ite);
168 if (JsonbErrType::E_NONE != ret) return ret;
169
170 const char* beg_addr = (const char*)ArrayVal::const_iterator::pointer(beg);
171 const char* end_addr = (const char*)ArrayVal::const_iterator::pointer(end);
172 return insertArrayInternal(ite, beg_addr, end_addr);
173 }
174
175 // For adding element to dict
176 JsonbErrType insertValue(JsonbKeyValue* kvalue) {
177 if (nullptr == kvalue) {
178 return JsonbErrType::E_INVALID_JSONB_OBJ;
179 }
180 // Check whether it's an object
181 if (nullptr == path_node_.back().jsonb_value ||
182 !path_node_.back().jsonb_value->isObject()) {
183 return JsonbErrType::E_NOTOBJ;
184 }
185 return insertValue(ObjectVal::const_iterator(kvalue), ++ObjectVal::const_iterator(kvalue));
186 }
187
188 // Insert a range of elements to a dictionary
189 JsonbErrType insertValue(ObjectVal::const_iterator beg, ObjectVal::const_iterator end) {
190 // Check whether it's an object
191 if (nullptr == path_node_.back().jsonb_value ||
192 !path_node_.back().jsonb_value->isObject()) {
193 return JsonbErrType::E_NOTOBJ;
194 }
195
196 ObjectVal::const_iterator obj_end = ((ObjectVal*)path_node_.back().jsonb_value)->end();
197 char* curr_addr = (char*)(ObjectVal::const_iterator::pointer)(obj_end);
198
199 // Calculate the bytes needed
200 int need_bytes = (int)((char*)(ObjectVal::const_iterator::pointer)end -
201 (char*)(ObjectVal::const_iterator::pointer)beg);
202 char* new_addr = curr_addr + need_bytes;
203 if (!moveTo(curr_addr, new_addr)) {
204 return JsonbErrType::E_OUTOFMEMORY;
205 }
206 memcpy(curr_addr, (char*)(ObjectVal::const_iterator::pointer)beg, need_bytes);
207 return JsonbErrType::E_NONE;
208 }
209
210 // Go back to the root node
211 void clearPath() {
212 path_node_.clear();
213 path_node_.push_back({(char*)root_, root_});
214 }

217 JsonbErrType updateValueInternal(const JsonbValue* value) {
218 // If we want to update root, it must be an array or object
219 if (path_node_.size() == 1 && !(value->isObject() || value->isArray())) {
220 return JsonbErrType::E_INVALID_OPER;
221 }
222 // For the updating that we don't need to move data
223 JsonbValue* curr = path_node_.back().jsonb_value;
224
225 if (value->isInt() && curr->isInt()) {
226 // Both are ints and optimization can be done here
227 int64_t val = ((const JsonbIntVal*)value)->val();
228 // setVal may fail because the new value can't fit into the current one.
229 if (((JsonbIntVal*)curr)->setVal(val)) {
230 return JsonbErrType::E_NONE;
231 }
232 }
233
234 // If both are strings and the new string is shorter than the
235 // allocated space, then we can update in place.
236 if (value->isString() && curr->isString()) {
237 // When both are strings, optimization is possible
238 const char* str = value->getValuePtr();
239 int str_len = value->size();
240 if ('\0' == str[str_len - 1]) {
241 // There are tailing NULLs in the string
242 str_len = (int)strlen(str);
243 }
244
245 // If the new length of the string range in
246 // [old_len * (1 - str_shrink_ratio), old_len],
247 // then we update it in place. Which means that
248 // when the length of the new string is less than
249 // old_len * str_shrink_ratio, we will do shrink
250 if (str_len <= (int)curr->size() &&
251 str_len >= ((int)curr->size() * (1.0 - str_shrink_ratio_))) {
252 // We can update string in place
253 if (!((JsonbStringVal*)curr)->setVal(str, str_len)) {
254 return JsonbErrType::E_INVALID_OPER;
255 }
256 return JsonbErrType::E_NONE;
257 }
258 }
259
260 // We can't update in place, extand it.
261 if (curr->numPackedBytes() != value->numPackedBytes()) {
262 char* curr_addr = (char*)curr;
263 char* next_addr = (char*)curr_addr + curr->numPackedBytes();
264 char* new_next_addr = curr_addr + value->numPackedBytes();
265 // Expand it or shrink it.
266 if (!moveTo(next_addr, new_next_addr)) {
267 // If move failed, then we do restore
268 return JsonbErrType::E_OUTOFMEMORY;
269 }
270 }
271 // Copy the new data to replace the old one
272 memcpy(curr, value, value->numPackedBytes());
273 return JsonbErrType::E_NONE;
274 }

276 /* Insert a node. Insert can only be applied for array. The node
277 will be inserted at the given iterator. If the array is not long
278 enough, it will be expanded automatically. Null will be the value
279 for the expaneded pard. */
280 JsonbErrType insertArrayInternal(ArrayVal::iterator ite, const char* val_beg,
281 const char* val_end) {
282 int bytes_needed = (int)(val_end - val_beg);
283 char* idx_addr = (char*)ArrayVal::iterator::pointer(ite);
284 if (!moveTo(idx_addr, idx_addr + bytes_needed)) {
285 return JsonbErrType::E_OUTOFMEMORY;
286 }
287
288 // Copy the inserted value
289 memcpy(idx_addr, val_beg, bytes_needed);
290 return JsonbErrType::E_NONE;
291 }

293private:
294 JsonbErrType getElemInArray(int idx, JsonbValue* array, ArrayVal::iterator& ite) {
295 if (nullptr == array || !array->isArray()) return JsonbErrType::E_NOTARRAY;
296
297 if (idx < 0) {
298 return JsonbErrType::E_OUTOFBOUNDARY;
299 }
300
301 // Now we want to get the node at the given index.
302 ArrayVal* arr_val = (ArrayVal*)path_node_.back().jsonb_value;
303 int count = 0;
304 ite = arr_val->begin();
305 while (ite != arr_val->end() && count != idx) {
306 ++count;
307 ++ite;
308 }
309
310 if (ite == arr_val->end() && count != idx) {
311 return JsonbErrType::E_OUTOFBOUNDARY;
312 }
313
314 return JsonbErrType::E_NONE;
315 }

317 /*
318 Whenever the size of a sub node in the stack is updated, the
319 parents should also be updated. inc_size can be positive
320 (expanding) or negative (shrinking)
321 */
322 void updatePackageSize(int inc_size) {
323 if (inc_size == 0) return;
324 /*
325 There are two cases for the top elements in the stack:
326 when we do inserting, the first one must be a container,
327 and we need to expand the size of it. But when for assignment,
328 the first one can be basic type or container and we don't want
329 to update it. But now, if we see it's the basic type, we will
330 not update it and when it's the container, we still update it.
331 It's fine, because, it will be rewritten outside.
332 */
333 for (auto ite = path_node_.begin(); ite != path_node_.end(); ++ite) {
334 JsonbValue* value = ite->jsonb_value;
335 if (value->type() >= JsonbType::T_Null && value->type() <= JsonbType::T_Double) {
336 // For the last elements in the stack, it can be a scalar and
337 // there is no size field stored in them. Just skip to update
338 // it.
339 assert(path_node_.end() == ite + 1);
340 } else {
341 uint32_t* size_pointer = (uint32_t*)((char*)value + sizeof(JsonbTypeUnder));
342 *size_pointer += inc_size;
343 }
344 }
345 }

347 // Move the data from "from" to the end of the document to the new
348 // address "to".
349 bool moveTo(char* from, char* to) {
350 size_t remaining = root_->numPackedBytes() - ((char*)from - (char*)root_);
351
352 // Check whether it exceed the buffer
353 if (to + remaining > (char*)document_ + buffer_size_) return false;
354 updatePackageSize((int)(to - from));
355 memmove(to, from, remaining);
356 return true;
357 }

367 }
