32template <typename Impl, typename Name, bool is_injective = false>
33class FunctionStringToString : public IFunction {
34public:
35 static constexpr auto name = Name::name;
36 static constexpr bool has_variadic_argument =
37 !std::is_void_v<decltype(has_variadic_argument_types(std::declval<Impl>()))>;
38
39 static FunctionPtr create() { return std::make_shared<FunctionStringToString>(); }
40
41 String get_name() const override { return name; }
42
43 size_t get_number_of_arguments() const override { return 1; }
44
45 bool get_is_injective(const Block&) override { return is_injective; }
46
47 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
48 if (!is_string_or_fixed_string(arguments[0])) {
49 LOG(FATAL) << fmt::format("Illegal type {} of argument of function {}",
50 arguments[0]->get_name(), get_name());
51 }
52
53 return arguments[0];
54 }
55
56 bool use_default_implementation_for_constants() const override { return true; }
57
58 DataTypes get_variadic_argument_types_impl() const override {
59 if constexpr (has_variadic_argument) return Impl::get_variadic_argument_types();
60 return {};
61 }
62
63 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
64 size_t result, size_t input_rows_count) override {
65 const ColumnPtr column = block.get_by_position(arguments[0]).column;
66 if (const ColumnString* col = check_and_get_column<ColumnString>(column.get())) {
67 auto col_res = ColumnString::create();
68 Impl::vector(col->get_chars(), col->get_offsets(), col_res->get_chars(),
69 col_res->get_offsets());
70 block.replace_by_position(result, std::move(col_res));
71 } else {
72 return Status::RuntimeError("Illegal column {} of argument of function {}",
73 block.get_by_position(arguments[0]).column->get_name(),
74 get_name());
75 }
76 return Status::OK();
77 }
78};
