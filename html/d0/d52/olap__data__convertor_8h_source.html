120 static bool should_padding(const ColumnString* column, size_t padding_length) {
121 // Check sum of data length, including terminating zero.
122 return column->size() * padding_length != column->chars.size();
123 }
124
125 static ColumnPtr clone_and_padding(const ColumnString* input, size_t padding_length) {
126 auto column = vectorized::ColumnString::create();
127 auto padded_column =
128 assert_cast<vectorized::ColumnString*>(column->assume_mutable().get());
129
130 column->offsets.resize(input->size());
131 column->chars.resize(input->size() * padding_length);
132 memset(padded_column->chars.data(), 0, input->size() * padding_length);
133
134 for (size_t i = 0; i < input->size(); i++) {
135 column->offsets[i] = (i + 1) * padding_length;
136
137 auto str = input->get_data_at(i);
138
139 DCHECK(str.size <= padding_length)
140 << "char type data length over limit, padding_length=" << padding_length
141 << ", real=" << str.size;
142
143 if (str.size) {
144 memcpy(padded_column->chars.data() + i * padding_length, str.data, str.size);
145 }
146 }
147
148 return column;
149 }
150
151 size_t _length;
152 PaddedPODArray<Slice> _slice;
153 ColumnPtr _column = nullptr;
154 };

213 // class OlapColumnDataConvertorSimple for simple types, which don't need to do any convert, like int, float, double, etc...
214 template <typename T>
215 class OlapColumnDataConvertorSimple : public OlapColumnDataConvertorBase {
216 public:
217 OlapColumnDataConvertorSimple() = default;
218 ~OlapColumnDataConvertorSimple() override = default;
219
220 const void* get_data() const override { return _values; }
221
222 const void* get_data_at(size_t offset) const override {
223 assert(offset < _num_rows);
224 UInt8 null_flag = 0;
225 if (_nullmap) {
226 null_flag = _nullmap[offset];
227 }
228 return null_flag ? nullptr : _values + offset;
229 }
230
231 Status convert_to_olap() override {
232 const vectorized::ColumnVector<T>* column_data = nullptr;
233 if (_nullmap) {
234 auto nullable_column =
235 assert_cast<const vectorized::ColumnNullable*>(_typed_column.column.get());
236 column_data = assert_cast<const vectorized::ColumnVector<T>*>(
237 nullable_column->get_nested_column_ptr().get());
238 } else {
239 column_data =
240 assert_cast<const vectorized::ColumnVector<T>*>(_typed_column.column.get());
241 }
242
243 assert(column_data);
244 _values = (const T*)(column_data->get_data().data()) + _row_pos;
245 return Status::OK();
246 }
247
248 protected:
249 const T* _values = nullptr;
250 };
