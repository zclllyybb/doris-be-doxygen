35class Decompressor {
36public:
37 virtual ~Decompressor() = default;
38
39 // implement in derived class
40 // input(in): buf where decompress begin
41 // input_len(in): max length of input buf
42 // input_bytes_read(out): bytes which is consumed by decompressor
43 // output(out): buf where to save decompressed data
44 // output_max_len(in): max length of output buf
45 // decompressed_len(out): decompressed data size in output buf
46 // stream_end(out): true if reach the and of stream,
47 // or normally finished decompressing entire block
48 // more_input_bytes(out): decompressor need more bytes to consume
49 // more_output_bytes(out): decompressor need more space to save decompressed data
50 //
51 // input and output buf should be allocated and released outside
52 virtual Status decompress(uint8_t* input, size_t input_len, size_t* input_bytes_read,
53 uint8_t* output, size_t output_max_len, size_t* decompressed_len,
54 bool* stream_end, size_t* more_input_bytes,
55 size_t* more_output_bytes) = 0;
56
57public:
58 static Status create_decompressor(CompressType type, Decompressor** decompressor);
59
60 virtual std::string debug_info();
61
62 CompressType get_type() { return _ctype; }
63
64protected:
65 virtual Status init() = 0;
66
67 Decompressor(CompressType ctype) : _ctype(ctype) {}
68
69 CompressType _ctype;
70};

160 uint8_t* get_uint8(uint8_t* ptr, uint8_t* value) {
161 *value = *ptr;
162 return ptr + sizeof(uint8_t);

164
165 uint8_t* get_uint16(uint8_t* ptr, uint16_t* value) {
166 *value = *ptr << 8 | *(ptr + 1);
167 return ptr + sizeof(uint16_t);
168 }
169
170 uint8_t* get_uint32(uint8_t* ptr, uint32_t* value) {
171 *value = (*ptr << 24) | (*(ptr + 1) << 16) | (*(ptr + 2) << 8) | *(ptr + 3);
172 return ptr + sizeof(uint32_t);
173 }
