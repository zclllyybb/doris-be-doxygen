94class IndexedColumnIterator {
95public:
96 explicit IndexedColumnIterator(const IndexedColumnReader* reader)
97 : _reader(reader),
98 _ordinal_iter(&reader->_ordinal_index_reader),
99 _value_iter(&reader->_value_index_reader) {}
100
101 // Seek to the given ordinal entry. Entry 0 is the first entry.
102 // Return NotFound if provided seek point is past the end.
103 // Return NotSupported for column without ordinal index.
104 Status seek_to_ordinal(ordinal_t idx);
105
106 // Seek the index to the given key, or to the index entry immediately
107 // before it. Then seek the data block to the value matching value or to
108 // the value immediately after it.
109 //
110 // Sets *exact_match to indicate whether the seek found the exact
111 // key requested.
112 //
113 // Return NotFound if the given key is greater than all keys in this column.
114 // Return NotSupported for column without value index.
115 Status seek_at_or_after(const void* key, bool* exact_match);
116 Status seek_at_or_after(const std::string* key, bool* exact_match) {
117 Slice slice(key->data(), key->size());
118 return seek_at_or_after(static_cast<const void*>(&slice), exact_match);
119 }
120
121 // Get the ordinal index that the iterator is currently pointed to.
122 ordinal_t get_current_ordinal() const {
123 DCHECK(_seeked);
124 return _current_ordinal;
125 }
126
127 // After one seek, we can only call this function once to read data
128 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst);
129
130private:
131 Status _read_data_page(const PagePointer& pp);
132
133 const IndexedColumnReader* _reader;
134 // iterator for ordinal index page
135 IndexPageIterator _ordinal_iter;
136 // iterator for value index page
137 IndexPageIterator _value_iter;
138
139 bool _seeked = false;
140 // current in-use index iterator, could be `&_ordinal_iter` or `&_value_iter` or null
141 IndexPageIterator* _current_iter = nullptr;
142 // seeked data page, containing value at `_current_ordinal`
143 ParsedPage _data_page;
144 // next_batch() will read from this position
145 ordinal_t _current_ordinal = 0;
146 // iterator owned compress codec, should NOT be shared by threads, initialized before used
147 BlockCompressionCodec* _compress_codec = nullptr;
148};
