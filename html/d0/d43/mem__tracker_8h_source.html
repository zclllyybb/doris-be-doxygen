29// Used to track memory usage.
30//
31// MemTracker can be consumed manually by consume()/release(), or put into SCOPED_CONSUME_MEM_TRACKER,
32// which will automatically track all memory usage of the code segment where it is located.
33//
34// This class is thread-safe.
35class MemTracker {
36public:
37 struct Snapshot {
38 std::string type = "";
39 std::string label;
40 std::string parent_label = "";
41 int64_t limit = 0;
42 int64_t cur_consumption = 0;
43 int64_t peak_consumption = 0;
44 };

46 // A counter that keeps track of the current and peak value seen.
47 // Relaxed ordering, not accurate in real time.
48 class MemCounter {
49 public:
50 MemCounter() : _current_value(0), _peak_value(0) {}
51
52 void add(int64_t delta) {
53 _current_value.fetch_add(delta, std::memory_order_relaxed);
54 update_peak();
55 }
56
57 void add_no_update_peak(int64_t delta) {
58 _current_value.fetch_add(delta, std::memory_order_relaxed);
59 }
60
61 bool try_add(int64_t delta, int64_t max) {
62 if (UNLIKELY(_current_value.load(std::memory_order_relaxed) + delta > max))
63 return false;
64 _current_value.fetch_add(delta, std::memory_order_relaxed);
65 update_peak();
66 return true;
67 }
68
69 void set(int64_t v) {
70 _current_value.store(v, std::memory_order_relaxed);
71 update_peak();
72 }
73
74 void update_peak() {
75 if (_current_value.load(std::memory_order_relaxed) >
76 _peak_value.load(std::memory_order_relaxed)) {
77 _peak_value.store(_current_value.load(std::memory_order_relaxed),
78 std::memory_order_relaxed);
79 }
80 }
81
82 int64_t current_value() const { return _current_value.load(std::memory_order_relaxed); }
83 int64_t peak_value() const { return _peak_value.load(std::memory_order_relaxed); }
84
85 private:
86 std::atomic<int64_t> _current_value;
87 std::atomic<int64_t> _peak_value;
88 };
89
90 // Creates and adds the tracker to the mem_tracker_pool.
91 MemTracker(const std::string& label, RuntimeProfile* profile, MemTrackerLimiter* parent,
92 const std::string& profile_counter_name);
93 MemTracker(const std::string& label, MemTrackerLimiter* parent = nullptr);
94 // For MemTrackerLimiter
95 MemTracker() { _parent_group_num = -1; }
96
97 ~MemTracker();
98
99 static std::string print_bytes(int64_t bytes) {
100 return bytes >= 0 ? PrettyPrinter::print(bytes, TUnit::BYTES)
101 : "-" + PrettyPrinter::print(std::abs(bytes), TUnit::BYTES);
102 }
103
104public:
105 const std::string& label() const { return _label; }
106 const std::string& parent_label() const { return _parent_label; }
107 const std::string& set_parent_label() const { return _parent_label; }
108 // Returns the memory consumed in bytes.
109 int64_t consumption() const { return _consumption->current_value(); }
110 int64_t peak_consumption() const { return _consumption->peak_value(); }
111
112 void consume(int64_t bytes) {
113 if (bytes == 0) return;
114 _consumption->add(bytes);
115 }
116 void consume_no_update_peak(int64_t bytes) { // need extreme fast
117 _consumption->add_no_update_peak(bytes);
118 }
119 void release(int64_t bytes) { consume(-bytes); }
120 void set_consumption(int64_t bytes) { _consumption->set(bytes); }
121
122 void refresh_profile_counter() {
123 if (_profile_counter) {
124 _profile_counter->set(_consumption->current_value());
125 }
126 }
127 static void refresh_all_tracker_profile();
128
129public:
130 Snapshot make_snapshot() const;
131 // Specify group_num from mem_tracker_pool to generate snapshot.
132 static void make_group_snapshot(std::vector<Snapshot>* snapshots, int64_t group_num,
133 std::string parent_label);
134 static std::string log_usage(MemTracker::Snapshot snapshot);
135
136 std::string debug_string() {
137 std::stringstream msg;
138 msg << "label: " << _label << "; "
139 << "consumption: " << consumption() << "; "
140 << "peak_consumption: " << peak_consumption() << "; ";
141 return msg.str();
142 }
143
144protected:
145 void bind_parent(MemTrackerLimiter* parent);
146
147 // label used in the make snapshot, not guaranteed unique.
148 std::string _label;
149
150 std::shared_ptr<MemCounter> _consumption;
151 std::shared_ptr<RuntimeProfile::HighWaterMarkCounter> _profile_counter;
152
153 // Tracker is located in group num in mem_tracker_pool
154 int64_t _parent_group_num = 0;
155 // Use _parent_label to correlate with parent limiter tracker.
156 std::string _parent_label = "-";
157
158 // Iterator into mem_tracker_pool for this object. Stored to have O(1) remove.
159 std::list<MemTracker*>::iterator _tracker_group_it;
160};
