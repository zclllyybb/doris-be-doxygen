29// Used to track memory usage.
30//
31// MemTracker can be consumed manually by consume()/release(), or put into SCOPED_CONSUME_MEM_TRACKER,
32// which will automatically track all memory usage of the code segment where it is located.
33//
34// This class is thread-safe.
35class MemTracker {
36public:
37 struct Snapshot {
38 std::string type = "";
39 std::string label;
40 std::string parent_label = "";
41 int64_t limit = 0;
42 int64_t cur_consumption = 0;
43 int64_t peak_consumption = 0;
44 };
45
46 // Creates and adds the tracker to the mem_tracker_pool.
47 MemTracker(const std::string& label, RuntimeProfile* profile, MemTrackerLimiter* parent,
48 const std::string& profile_counter_name);
49 MemTracker(const std::string& label, MemTrackerLimiter* parent = nullptr);
50 // For MemTrackerLimiter
51 MemTracker() { _parent_group_num = -1; }
52
53 ~MemTracker();
54
55 static std::string print_bytes(int64_t bytes) {
56 return bytes >= 0 ? PrettyPrinter::print(bytes, TUnit::BYTES)
57 : "-" + PrettyPrinter::print(std::abs(bytes), TUnit::BYTES);
58 }
59
60public:
61 const std::string& label() const { return _label; }
62 const std::string& parent_label() const { return _parent_label; }
63 const std::string& set_parent_label() const { return _parent_label; }
64 // Returns the memory consumed in bytes.
65 int64_t consumption() const { return _consumption->current_value(); }
66 int64_t peak_consumption() const { return _consumption->value(); }
67
68 void consume(int64_t bytes) {
69 if (bytes == 0) return;
70 _consumption->add(bytes);
71 }
72 void release(int64_t bytes) { consume(-bytes); }
73 void set_consumption(int64_t bytes) { _consumption->set(bytes); }
74
75public:
76 Snapshot make_snapshot() const;
77 // Specify group_num from mem_tracker_pool to generate snapshot.
78 static void make_group_snapshot(std::vector<Snapshot>* snapshots, int64_t group_num,
79 std::string parent_label);
80 static std::string log_usage(MemTracker::Snapshot snapshot);
81
82 std::string debug_string() {
83 std::stringstream msg;
84 msg << "label: " << _label << "; "
85 << "consumption: " << consumption() << "; "
86 << "peak_consumption: " << peak_consumption() << "; ";
87 return msg.str();
88 }
89
90protected:
91 void bind_parent(MemTrackerLimiter* parent);
92
93 // label used in the make snapshot, not guaranteed unique.
94 std::string _label;
95
96 std::shared_ptr<RuntimeProfile::HighWaterMarkCounter> _consumption; // in bytes
97
98 // Tracker is located in group num in mem_tracker_pool
99 int64_t _parent_group_num = 0;
100 // Use _parent_label to correlate with parent limiter tracker.
101 std::string _parent_label = "-";
102
103 // Iterator into mem_tracker_pool for this object. Stored to have O(1) remove.
104 std::list<MemTracker*>::iterator _tracker_group_it;
105};
106
107} // namespace doris
