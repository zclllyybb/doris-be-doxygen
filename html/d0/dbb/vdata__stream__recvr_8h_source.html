205class VDataStreamRecvr::PipSenderQueue : public SenderQueue {
206public:
207 PipSenderQueue(VDataStreamRecvr* parent_recvr, int num_senders, RuntimeProfile* profile)
208 : SenderQueue(parent_recvr, num_senders, profile) {}
209
210 bool should_wait() override {
211 return !_is_cancelled && _block_queue_empty && _num_remaining_senders > 0;
212 }
213
214 void _update_block_queue_empty() override { _block_queue_empty = _block_queue.empty(); }
215
216 Status get_batch(Block** next_block) override {
217 CHECK(!should_wait()) << " _is_cancelled: " << _is_cancelled
218 << ", _block_queue_empty: " << _block_queue_empty
219 << ", _num_remaining_senders: " << _num_remaining_senders;
220 std::lock_guard<std::mutex> l(_lock); // protect _block_queue
221 return _inner_get_batch(next_block);
222 }
223
224 void add_block(Block* block, bool use_move) override {
225 // Avoid deadlock when calling SenderQueue::cancel() in tcmalloc hook,
226 // limit memory via DataStreamRecvr::exceeds_limit.
227 STOP_CHECK_THREAD_MEM_TRACKER_LIMIT();
228
229 if (_is_cancelled || !block->rows()) {
230 return;
231 }
232 Block* nblock = new Block(block->get_columns_with_type_and_name());
233
234 // local exchange should copy the block contented if use move == false
235 if (use_move) {
236 block->clear();
237 } else {
238 auto rows = block->rows();
239 for (int i = 0; i < nblock->columns(); ++i) {
240 nblock->get_by_position(i).column =
241 nblock->get_by_position(i).column->clone_resized(rows);
242 }
243 }
244 materialize_block_inplace(*nblock);
245
246 size_t block_size = nblock->bytes();
247 auto block_mem_size = nblock->allocated_bytes();
248 {
249 std::unique_lock<std::mutex> l(_lock);
250 _block_queue.emplace_back(block_size, nblock);
251 }
252 COUNTER_UPDATE(_recvr->_local_bytes_received_counter, block_size);
253 _recvr->_blocks_memory_usage->add(block_mem_size);
254 _update_block_queue_empty();
255 _data_arrival_cv.notify_one();
256
257 _recvr->_num_buffered_bytes += block_size;
258 COUNTER_UPDATE(_recvr->_local_bytes_received_counter, block_size);
259 }
