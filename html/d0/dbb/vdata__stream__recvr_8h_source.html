204class VDataStreamRecvr::PipSenderQueue : public SenderQueue {
205public:
206 PipSenderQueue(VDataStreamRecvr* parent_recvr, int num_senders, RuntimeProfile* profile)
207 : SenderQueue(parent_recvr, num_senders, profile) {}
208
209 bool should_wait() override {
210 return !_is_cancelled && _block_queue_empty && _num_remaining_senders > 0;
211 }
212
213 void _update_block_queue_empty() override { _block_queue_empty = _block_queue.empty(); }
214
215 Status get_batch(Block* block, bool* eos) override {
216 CHECK(!should_wait()) << " _is_cancelled: " << _is_cancelled
217 << ", _block_queue_empty: " << _block_queue_empty
218 << ", _num_remaining_senders: " << _num_remaining_senders;
219 std::lock_guard<std::mutex> l(_lock); // protect _block_queue
220 return _inner_get_batch(block, eos);
221 }
222
223 void add_block(Block* block, bool use_move) override {
224 // Avoid deadlock when calling SenderQueue::cancel() in tcmalloc hook,
225 // limit memory via DataStreamRecvr::exceeds_limit.
226 STOP_CHECK_THREAD_MEM_TRACKER_LIMIT();
227
228 if (_is_cancelled || !block->rows()) {
229 return;
230 }
231 BlockUPtr nblock = std::make_unique<Block>(block->get_columns_with_type_and_name());
232
233 // local exchange should copy the block contented if use move == false
234 if (use_move) {
235 block->clear();
236 } else {
237 auto rows = block->rows();
238 for (int i = 0; i < nblock->columns(); ++i) {
239 nblock->get_by_position(i).column =
240 nblock->get_by_position(i).column->clone_resized(rows);
241 }
242 }
243 materialize_block_inplace(*nblock);
244
245 auto block_mem_size = nblock->allocated_bytes();
246 {
247 std::unique_lock<std::mutex> l(_lock);
248 _block_queue.emplace_back(std::move(nblock), block_mem_size);
249 }
250 COUNTER_UPDATE(_recvr->_local_bytes_received_counter, block_mem_size);
251 _recvr->_blocks_memory_usage->add(block_mem_size);
252 _update_block_queue_empty();
253 _data_arrival_cv.notify_one();
254 }
