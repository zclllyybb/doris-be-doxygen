32// When the tuple/block data is greater than 2G, embed the tuple/block data
33// and the request serialization string in the attachment, and use "http" brpc.
34// "http"brpc requires that only one of request and attachment be non-null.
35//
36// 2G: In the default "baidu_std" brpcd, upper limit of the request and attachment length is 2G.
37constexpr size_t MIN_HTTP_BRPC_SIZE = (1ULL << 31);
38
39// Embed column_values and brpc request serialization string in controller attachment.
40template <typename Params, typename Closure>
41Status request_embed_attachment_contain_block(Params* brpc_request, Closure* closure) {
42 auto block = brpc_request->block();
43 Status st = request_embed_attachment(brpc_request, block.column_values(), closure);
44 block.set_column_values("");
45 return st;
46}

48inline bool enable_http_send_block(
49 const PTransmitDataParams& request,
50 bool transfer_large_data_by_brpc = config::transfer_large_data_by_brpc) {
51 if (!config::transfer_large_data_by_brpc) {
52 return false;
53 }
54 if (!request.has_block() || !request.block().has_column_values()) {
55 return false;
56 }
57 if (request.ByteSizeLong() < MIN_HTTP_BRPC_SIZE) {
58 return false;
59 }
60 return true;
61}

70template <typename Closure>
71Status transmit_block_http(RuntimeState* state, Closure* closure, PTransmitDataParams& params,
72 TNetworkAddress brpc_dest_addr) {
73 RETURN_IF_ERROR(request_embed_attachment_contain_block(&params, closure));
74
75 //format an ipv6 address
76 std::string brpc_url = get_brpc_http_url(brpc_dest_addr.hostname, brpc_dest_addr.port);
77
78 std::shared_ptr<PBackendService_Stub> brpc_http_stub =
79 state->exec_env()->brpc_internal_client_cache()->get_new_client_no_cache(brpc_url,
80 "http");
81 closure->cntl.http_request().uri() = brpc_url + "/PInternalServiceImpl/transmit_block_by_http";
82 closure->cntl.http_request().set_method(brpc::HTTP_METHOD_POST);
83 closure->cntl.http_request().set_content_type("application/json");
84 brpc_http_stub->transmit_block_by_http(&closure->cntl, nullptr, &closure->result, closure);
85
86 return Status::OK();
87}

145template <typename Params, typename Closure>
146Status request_embed_attachment(Params* brpc_request, const std::string& data, Closure* closure) {
147 butil::IOBuf attachment;
148
149 // step1: serialize brpc_request to string, and append to attachment.
150 std::string req_str;
151 brpc_request->SerializeToString(&req_str);
152 int64_t req_str_size = req_str.size();
153 attachment.append(&req_str_size, sizeof(req_str_size));
154 attachment.append(req_str);
155
156 // step2: append data to attachment and put it in the closure.
157 int64_t data_size = data.size();
158 attachment.append(&data_size, sizeof(data_size));
159 try {
160 attachment.append(data);
161 } catch (...) {
162 LOG(WARNING) << "Try to alloc " << data_size
163 << " bytes for append data to attachment failed. ";
164 return Status::MemoryAllocFailed("request embed attachment failed to memcpy {} bytes",
165 data_size);
166 }
167 // step3: attachment add to closure.
168 closure->cntl.request_attachment().swap(attachment);
169 return Status::OK();
170}

182template <typename Params>
183Status attachment_extract_request(const Params* brpc_request, brpc::Controller* cntl,
184 std::string* data) {
185 const butil::IOBuf& io_buf = cntl->request_attachment();
186
187 // step1: deserialize request string to brpc_request from attachment.
188 int64_t req_str_size;
189 io_buf.copy_to(&req_str_size, sizeof(req_str_size), 0);
190 std::string req_str;
191 io_buf.copy_to(&req_str, req_str_size, sizeof(req_str_size));
192 Params* req = const_cast<Params*>(brpc_request);
193 req->ParseFromString(req_str);
194
195 // step2: extract data from attachment.
196 int64_t data_size;
197 io_buf.copy_to(&data_size, sizeof(data_size), sizeof(req_str_size) + req_str_size);
198 try {
199 io_buf.copy_to(data, data_size, sizeof(data_size) + sizeof(req_str_size) + req_str_size);
200 } catch (...) {
201 LOG(WARNING) << "Try to alloc " << data_size
202 << " bytes for extract data from attachment failed. ";
203 return Status::MemoryAllocFailed("attachment extract request failed to memcpy {} bytes",
204 data_size);
205 }
206 return Status::OK();
207}
