27// Helper class to Prepare() , Open() and Close() the ordering expressions used to perform
28// comparisons in a sort. Used by TopNNode, SortNode. When two
29// rows are compared, the ordering expressions are evaluated once for each side.
30// TopN and Sort materialize input rows into a single tuple before sorting.
31// If _materialize_tuple is true, SortExecExprs also stores the slot expressions used to
32// materialize the sort tuples.
33namespace vectorized {
34
35class VSortExecExprs {
36public:
37 // Initialize the expressions from a TSortInfo using the specified pool.
38 Status init(const TSortInfo& sort_info, ObjectPool* pool);
39
40 // prepare all expressions used for sorting and tuple materialization.
41 Status prepare(RuntimeState* state, const RowDescriptor& child_row_desc,
42 const RowDescriptor& output_row_desc);
43
44 // open all expressions used for sorting and tuple materialization.
45 Status open(RuntimeState* state);
46
47 // close all expressions used for sorting and tuple materialization.
48 void close(RuntimeState* state);
49
50 const std::vector<VExprContext*>& sort_tuple_slot_expr_ctxs() const {
51 return _sort_tuple_slot_expr_ctxs;
52 }
53
54 // Can only be used after calling prepare()
55 const std::vector<VExprContext*>& lhs_ordering_expr_ctxs() const {
56 return _lhs_ordering_expr_ctxs;
57 }
58
59 // Can only be used after calling open()
60 const std::vector<VExprContext*>& rhs_ordering_expr_ctxs() const {
61 return _rhs_ordering_expr_ctxs;
62 }
63
64 bool need_materialize_tuple() const { return _materialize_tuple; }
65
66 const std::vector<bool>& get_convert_nullable_flags() const {
67 return _need_convert_to_nullable_flags;
68 }
69
70private:
71 // Create two VExprContexts for evaluating over the TupleRows.
72 std::vector<VExprContext*> _lhs_ordering_expr_ctxs;
73 std::vector<VExprContext*> _rhs_ordering_expr_ctxs;
74
75 // If true, the tuples to be sorted are materialized by
76 // _sort_tuple_slot_exprs before the actual sort is performed.
77 bool _materialize_tuple;
78
79 // Expressions used to materialize slots in the tuples to be sorted.
80 // One expr per slot in the materialized tuple. Valid only if
81 // _materialize_tuple is true.
82 std::vector<VExprContext*> _sort_tuple_slot_expr_ctxs;
83
84 // for some reason, _sort_tuple_slot_expr_ctxs is not-null but _lhs_ordering_expr_ctxs is nullable
85 // this flag list would be used to convert column to nullable.
86 std::vector<bool> _need_convert_to_nullable_flags;
87
88 // Initialize directly from already-created VExprContexts. Callers should manually call
89 // Prepare(), Open(), and Close() on input VExprContexts (instead of calling the
90 // analogous functions in this class). Used for testing.
91 Status init(const std::vector<VExprContext*>& lhs_ordering_expr_ctxs,
92 const std::vector<VExprContext*>& rhs_ordering_expr_ctxs);
93
94 // Initialize the ordering and (optionally) materialization expressions from the thrift
95 // TExprs into the specified pool. sort_tuple_slot_exprs is NULL if the tuple is not
96 // materialized.
97 Status init(const std::vector<TExpr>& ordering_exprs,
98 const std::vector<TExpr>* sort_tuple_slot_exprs, ObjectPool* pool);
99};
