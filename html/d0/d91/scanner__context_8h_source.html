41// ScannerContext is responsible for recording the execution status
42// of a group of Scanners corresponding to a ScanNode.
43// Including how many scanners are being scheduled, and maintaining
44// a producer-consumer blocks queue between scanners and scan nodes.
45//
46// ScannerContext is also the scheduling unit of ScannerScheduler.
47// ScannerScheduler schedules a ScannerContext at a time,
48// and submits the Scanners to the scanner thread pool for data scanning.
49class ScannerContext {
50public:
51 ScannerContext(RuntimeState* state_, VScanNode* parent, const TupleDescriptor* input_tuple_desc,
52 const TupleDescriptor* output_tuple_desc, const std::list<VScanner*>& scanners_,
53 int64_t limit_, int64_t max_bytes_in_blocks_queue_);
54
55 virtual ~ScannerContext() = default;
56 Status init();
57
58 vectorized::BlockUPtr get_free_block(bool* has_free_block);
59 void return_free_block(std::unique_ptr<vectorized::Block> block);
60
61 // Append blocks from scanners to the blocks queue.
62 virtual void append_blocks_to_queue(std::vector<vectorized::BlockUPtr>& blocks);
63 // Get next block from blocks queue. Called by ScanNode
64 // Set eos to true if there is no more data to read.
65 // And if eos is true, the block returned must be nullptr.
66 virtual Status get_block_from_queue(RuntimeState* state, vectorized::BlockUPtr* block,
67 bool* eos, int id, bool wait = true);
68
69 // When a scanner complete a scan, this method will be called
70 // to return the scanner to the list for next scheduling.
71 void push_back_scanner_and_reschedule(VScanner* scanner);
72
73 bool set_status_on_error(const Status& status);
74
75 Status status() {
76 std::lock_guard l(_transfer_lock);
77 return _process_status;
78 }
79
80 // Called by ScanNode.
81 // Used to notify the scheduler that this ScannerContext can stop working.
82 void set_should_stop() {
83 std::lock_guard l(_transfer_lock);
84 _should_stop = true;
85 _blocks_queue_added_cv.notify_one();
86 }
87
88 // Return true if this ScannerContext need no more process
89 virtual bool done() {
90 std::unique_lock l(_transfer_lock);
91 return _is_finished || _should_stop || !_process_status.ok();
92 }
93
94 // Update the running num of scanners and contexts
95 void update_num_running(int32_t scanner_inc, int32_t sched_inc) {
96 std::lock_guard l(_transfer_lock);
97 _num_running_scanners += scanner_inc;
98 _num_scheduling_ctx += sched_inc;
99 _blocks_queue_added_cv.notify_one();
100 _ctx_finish_cv.notify_one();
101 }
102
103 int get_num_running_scanners() const { return _num_running_scanners; }
104
105 int get_num_scheduling_ctx() const { return _num_scheduling_ctx; }
106
107 void get_next_batch_of_scanners(std::list<VScanner*>* current_run);
108
109 void clear_and_join(VScanNode* node, RuntimeState* state);
110
111 virtual bool no_schedule();
112
113 std::string debug_string();
114
115 RuntimeState* state() { return _state; }
116
117 void incr_num_ctx_scheduling(int64_t num) { _scanner_ctx_sched_counter->update(num); }
118 void incr_num_scanner_scheduling(int64_t num) { _scanner_sched_counter->update(num); }
119
120 VScanNode* parent() { return _parent; }
121
122 virtual bool empty_in_queue(int id);
123
124 virtual void set_max_queue_size(int max_queue_size) {};
125
126 // todo(wb) rethinking how to calculate ```_max_bytes_in_queue``` when executing shared scan
127 virtual inline bool has_enough_space_in_blocks_queue() const {
128 return _cur_bytes_in_queue < _max_bytes_in_queue / 2;
129 }

131 void reschedule_scanner_ctx();

133 // the unique id of this context
134 std::string ctx_id;
135 int32_t queue_idx = -1;
136 ThreadPoolToken* thread_token;
137 std::vector<bthread_t> _btids;

139private:
140 Status _close_and_clear_scanners(VScanNode* node, RuntimeState* state);
141
142protected:
143 RuntimeState* _state;
144 VScanNode* _parent;
145
146 // the comment of same fields in VScanNode
147 const TupleDescriptor* _input_tuple_desc;
148 const TupleDescriptor* _output_tuple_desc;
149 // If _input_tuple_desc is not null, _real_tuple_desc point to _input_tuple_desc,
150 // otherwise, _real_tuple_desc point to _output_tuple_desc
151 const TupleDescriptor* _real_tuple_desc;
152
153 // _transfer_lock is used to protect the critical section
154 // where the ScanNode and ScannerScheduler interact.
155 // Including access to variables such as blocks_queue, _process_status, _is_finished, etc.
156 doris::Mutex _transfer_lock;
157 // The blocks got from scanners will be added to the "blocks_queue".
158 // And the upper scan node will be as a consumer to fetch blocks from this queue.
159 // Should be protected by "_transfer_lock"
160 std::list<vectorized::BlockUPtr> _blocks_queue;
161 // Wait in get_block_from_queue(), by ScanNode.
162 doris::ConditionVariable _blocks_queue_added_cv;
163 // Wait in clear_and_join(), by ScanNode.
164 doris::ConditionVariable _ctx_finish_cv;
165
166 // The following 3 variables control the process of the scanner scheduling.
167 // Use _transfer_lock to protect them.
168 // 1. _process_status
169 // indicates the global status of this scanner context.
170 // Set to non-ok if encounter errors.
171 // And if it is non-ok, the scanner process should stop.
172 // Set be set by either ScanNode or ScannerScheduler.
173 // 2. _should_stop
174 // Always be set by ScanNode.
175 // True means no more data need to be read(reach limit or closed)
176 // 3. _is_finished
177 // Always be set by ScannerScheduler.
178 // True means all scanners are finished to scan.
179 Status _process_status;
180 std::atomic_bool _status_error = false;
181 std::atomic_bool _should_stop = false;
182 std::atomic_bool _is_finished = false;
183
184 // Pre-allocated blocks for all scanners to share, for memory reuse.
185 doris::Mutex _free_blocks_lock;
186 std::vector<vectorized::BlockUPtr> _free_blocks;
187
188 int _batch_size;
189 // The limit from SQL's limit clause
190 int64_t limit;
191
192 // Current number of running scanners.
193 std::atomic_int32_t _num_running_scanners = 0;
194 // Current number of ctx being scheduled.
195 // After each Scanner finishes a task, it will put the corresponding ctx
196 // back into the scheduling queue.
197 // Therefore, there will be multiple pointer of same ctx in the scheduling queue.
198 // Here we record the number of ctx in the scheduling queue to clean up at the end.
199 std::atomic_int32_t _num_scheduling_ctx = 0;
200 // Num of unfinished scanners. Should be set in init()
201 int32_t _num_unfinished_scanners = 0;
202 // Max number of scan thread for this scanner context.
203 int32_t _max_thread_num = 0;
204 // How many blocks a scanner can use in one task.
205 int32_t _block_per_scanner = 0;
206
207 // The current bytes of blocks in blocks queue
208 int64_t _cur_bytes_in_queue = 0;
209 // The max limit bytes of blocks in blocks queue
210 int64_t _max_bytes_in_queue;
211
212 doris::vectorized::ScannerScheduler* _scanner_scheduler;
213 // List "scanners" saves all "unfinished" scanners.
214 // The scanner scheduler will pop scanners from this list, run scanner,
215 // and then if the scanner is not finished, will be pushed back to this list.
216 // Not need to protect by lock, because only one scheduler thread will access to it.
217 doris::Mutex _scanners_lock;
218 std::list<VScanner*> _scanners;
219 std::vector<int64_t> _finished_scanner_runtime;
220 std::vector<int64_t> _finished_scanner_rows_read;
221
222 std::shared_ptr<RuntimeProfile> _scanner_profile;
223 RuntimeProfile::Counter* _scanner_sched_counter = nullptr;
224 RuntimeProfile::Counter* _scanner_ctx_sched_counter = nullptr;
225 RuntimeProfile::HighWaterMarkCounter* _free_blocks_memory_usage = nullptr;
226 RuntimeProfile::HighWaterMarkCounter* _queued_blocks_memory_usage = nullptr;
227 RuntimeProfile::Counter* _newly_create_free_blocks_num = nullptr;
228 RuntimeProfile::Counter* _scanner_wait_batch_timer = nullptr;
229};
