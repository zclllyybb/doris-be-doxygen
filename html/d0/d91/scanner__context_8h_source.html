41// ScannerContext is responsible for recording the execution status
42// of a group of Scanners corresponding to a ScanNode.
43// Including how many scanners are being scheduled, and maintaining
44// a producer-consumer blocks queue between scanners and scan nodes.
45//
46// ScannerContext is also the scheduling unit of ScannerScheduler.
47// ScannerScheduler schedules a ScannerContext at a time,
48// and submits the Scanners to the scanner thread pool for data scanning.
49class ScannerContext {
50public:
51 ScannerContext(RuntimeState* state_, VScanNode* parent, const TupleDescriptor* input_tuple_desc,
52 const TupleDescriptor* output_tuple_desc, const std::list<VScanner*>& scanners_,
53 int64_t limit_, int64_t max_bytes_in_blocks_queue_)
54 : _state(state_),
55 _parent(parent),
56 _input_tuple_desc(input_tuple_desc),
57 _output_tuple_desc(output_tuple_desc),
58 _process_status(Status::OK()),
59 limit(limit_),
60 _max_bytes_in_queue(max_bytes_in_blocks_queue_),
61 _scanners(scanners_) {
62 ctx_id = UniqueId::gen_uid().to_string();
63 if (_scanners.empty()) {
64 _is_finished = true;
65 }
66 }
67
68 virtual ~ScannerContext() = default;
69
70 Status init();
71
72 vectorized::Block* get_free_block(bool* get_free_block);
73 void return_free_block(vectorized::Block* block);
74
75 // Append blocks from scanners to the blocks queue.
76 void append_blocks_to_queue(const std::vector<vectorized::Block*>& blocks);
77
78 // Get next block from blocks queue. Called by ScanNode
79 // Set eos to true if there is no more data to read.
80 // And if eos is true, the block returned must be nullptr.
81 virtual Status get_block_from_queue(vectorized::Block** block, bool* eos, bool wait = true);
82
83 // When a scanner complete a scan, this method will be called
84 // to return the scanner to the list for next scheduling.
85 void push_back_scanner_and_reschedule(VScanner* scanner);
86
87 bool set_status_on_error(const Status& status);
88
89 Status status() {
90 std::lock_guard<std::mutex> l(_transfer_lock);
91 return _process_status;
92 }
93
94 // Called by ScanNode.
95 // Used to notify the scheduler that this ScannerContext can stop working.
96 void set_should_stop() {
97 std::lock_guard<std::mutex> l(_transfer_lock);
98 _should_stop = true;
99 _blocks_queue_added_cv.notify_one();
100 }
101
102 // Return true if this ScannerContext need no more process
103 virtual bool done() {
104 std::unique_lock<std::mutex> l(_transfer_lock);
105 return _is_finished || _should_stop || !_process_status.ok();
106 }
107
108 // Update the running num of scanners and contexts
109 void update_num_running(int32_t scanner_inc, int32_t sched_inc) {
110 std::lock_guard<std::mutex> l(_transfer_lock);
111 _num_running_scanners += scanner_inc;
112 _num_scheduling_ctx += sched_inc;
113 _blocks_queue_added_cv.notify_one();
114 _ctx_finish_cv.notify_one();
115 }
116
117 int get_num_running_scanners() const { return _num_running_scanners; }
118
119 int get_num_scheduling_ctx() const { return _num_scheduling_ctx; }
120
121 void get_next_batch_of_scanners(std::list<VScanner*>* current_run);
122
123 void clear_and_join();
124
125 virtual bool no_schedule();
126
127 std::string debug_string();
128
129 RuntimeState* state() { return _state; }
130
131 void incr_num_ctx_scheduling(int64_t num) { _num_ctx_scheduling += num; }
132 void incr_num_scanner_scheduling(int64_t num) { _num_scanner_scheduling += num; }
133
134 VScanNode* parent() { return _parent; }
135
136 virtual bool empty_in_queue();
137
138 // the unique id of this context
139 std::string ctx_id;
140 int32_t queue_idx = -1;
141 ThreadPoolToken* thread_token;
142
143private:
144 Status _close_and_clear_scanners();
145
146 inline bool _has_enough_space_in_blocks_queue() const {
147 return _cur_bytes_in_queue < _max_bytes_in_queue / 2;
148 }
149
150 // do nothing here, we only do update on pip_scanner_context
151 virtual void _update_block_queue_empty() {}
152
153protected:
154 RuntimeState* _state;
155 VScanNode* _parent;
156
157 // the comment of same fields in VScanNode
158 const TupleDescriptor* _input_tuple_desc;
159 const TupleDescriptor* _output_tuple_desc;
160 // If _input_tuple_desc is not null, _real_tuple_desc point to _input_tuple_desc,
161 // otherwise, _real_tuple_desc point to _output_tuple_desc
162 const TupleDescriptor* _real_tuple_desc;
163
164 // _transfer_lock is used to protect the critical section
165 // where the ScanNode and ScannerScheduler interact.
166 // Including access to variables such as blocks_queue, _process_status, _is_finished, etc.
167 std::mutex _transfer_lock;
168 // The blocks got from scanners will be added to the "blocks_queue".
169 // And the upper scan node will be as a consumer to fetch blocks from this queue.
170 // Should be protected by "_transfer_lock"
171 std::list<vectorized::Block*> _blocks_queue;
172 // Wait in get_block_from_queue(), by ScanNode.
173 std::condition_variable _blocks_queue_added_cv;
174 // Wait in clear_and_join(), by ScanNode.
175 std::condition_variable _ctx_finish_cv;
176
177 // The following 3 variables control the process of the scanner scheduling.
178 // Use _transfer_lock to protect them.
179 // 1. _process_status
180 // indicates the global status of this scanner context.
181 // Set to non-ok if encounter errors.
182 // And if it is non-ok, the scanner process should stop.
183 // Set be set by either ScanNode or ScannerScheduler.
184 // 2. _should_stop
185 // Always be set by ScanNode.
186 // True means no more data need to be read(reach limit or closed)
187 // 3. _is_finished
188 // Always be set by ScannerScheduler.
189 // True means all scanners are finished to scan.
190 Status _process_status;
191 std::atomic_bool _status_error = false;
192 std::atomic_bool _should_stop = false;
193 std::atomic_bool _is_finished = false;
194
195 // Pre-allocated blocks for all scanners to share, for memory reuse.
196 std::mutex _free_blocks_lock;
197 std::vector<vectorized::Block*> _free_blocks;
198
199 // The limit from SQL's limit clause
200 int64_t limit;
201
202 // Current number of running scanners.
203 std::atomic_int32_t _num_running_scanners = 0;
204 // Current number of ctx being scheduled.
205 // After each Scanner finishes a task, it will put the corresponding ctx
206 // back into the scheduling queue.
207 // Therefore, there will be multiple pointer of same ctx in the scheduling queue.
208 // Here we record the number of ctx in the scheduling queue to clean up at the end.
209 std::atomic_int32_t _num_scheduling_ctx = 0;
210 // Num of unfinished scanners. Should be set in init()
211 int32_t _num_unfinished_scanners = 0;
212 // Max number of scan thread for this scanner context.
213 int32_t _max_thread_num = 0;
214 // How many blocks a scanner can use in one task.
215 int32_t _block_per_scanner = 0;
216
217 // The current bytes of blocks in blocks queue
218 int64_t _cur_bytes_in_queue = 0;
219 // The max limit bytes of blocks in blocks queue
220 int64_t _max_bytes_in_queue;
221
222 // List "scanners" saves all "unfinished" scanners.
223 // The scanner scheduler will pop scanners from this list, run scanner,
224 // and then if the scanner is not finished, will be pushed back to this list.
225 // Not need to protect by lock, because only one scheduler thread will access to it.
226 std::mutex _scanners_lock;
227 std::list<VScanner*> _scanners;
228 std::vector<int64_t> _finished_scanner_runtime;
229 std::vector<int64_t> _finished_scanner_rows_read;
230
231 int64_t _num_ctx_scheduling = 0;
232 int64_t _num_scanner_scheduling = 0;
233};
234} // namespace vectorized
235} // namespace doris
