63template <typename TUInt>
64void write_hex_uint_impl(TUInt uint_, char* out, const char* const table) {
65 union {
66 TUInt value;
67 UInt8 uint8[sizeof(TUInt)];
68 };
69
70 value = uint_;
71
73 for (size_t i = 0; i < sizeof(TUInt); ++i) {
74 memcpy(out + i * 2, &table[static_cast<size_t>(uint8[sizeof(TUInt) - 1 - i]) * 2], 2);
75 }
76}

120template <typename TUInt>
121TUInt unhex_uint(const char* data) {
122 TUInt res = TUInt(0);
123 if constexpr ((sizeof(TUInt) <= 8) || ((sizeof(TUInt) % 8) != 0)) {
124 for (size_t i = 0; i < sizeof(TUInt) * 2; ++i, ++data) {
125 res <<= 4;
126 res += unhex(*data);
127 }
128 } else {
129 for (size_t i = 0; i < sizeof(TUInt) / 8; ++i, data += 16) {
130 res <<= TUInt(64);
131 res += TUInt(unhex_uint<UInt64>(data));
132 }
133 }
134 return res;
135}
