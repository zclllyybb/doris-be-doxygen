37 static ColumnsWithTypeAndName create_columns_with_type_and_name(
38 const RowDescriptor& row_desc, bool ignore_trivial_slot = true) {
39 ColumnsWithTypeAndName columns_with_type_and_name;
40 for (const auto& tuple_desc : row_desc.tuple_descriptors()) {
41 for (const auto& slot_desc : tuple_desc->slots()) {
42 if (ignore_trivial_slot && !slot_desc->need_materialize()) {
43 continue;
44 }
45 columns_with_type_and_name.emplace_back(nullptr, slot_desc->get_data_type_ptr(),
46 slot_desc->col_name());

68 static void update_null_map(NullMap& dst, const NullMap& src) {

87 static VExpr* dfs_peel_conjunct(RuntimeState* state, VExprContext* context, VExpr* expr,
88 int& leaf_index, std::function<bool(int)> checker) {
89 static constexpr auto is_leaf = [](VExpr* expr) { return !expr->is_and_expr(); };
90
91 if (is_leaf(expr)) {
92 if (checker(leaf_index++)) {
93 expr->close(state, context, context->get_function_state_scope());
94 return nullptr;
95 }
96 return expr;
97 } else {
98 VExpr* left_child =
99 dfs_peel_conjunct(state, context, expr->children()[0], leaf_index, checker);
100 VExpr* right_child =
101 dfs_peel_conjunct(state, context, expr->children()[1], leaf_index, checker);
102
103 if (left_child != nullptr && right_child != nullptr) {
104 expr->set_children({left_child, right_child});
105 return expr;
106 } else {
107 // here only close the and expr self, do not close the child
108 expr->set_children({});
109 expr->close(state, context, context->get_function_state_scope());
110 }
111
112 return left_child != nullptr ? left_child : right_child;
113 }
114 }

120template <typename ThriftStruct>
121ThriftStruct from_json_string(const std::string& json_val) {
122 using namespace apache::thrift::transport;
123 using namespace apache::thrift::protocol;
124 ThriftStruct ts;
125 TMemoryBuffer* buffer =
126 new TMemoryBuffer((uint8_t*)json_val.c_str(), (uint32_t)json_val.size());
127 std::shared_ptr<TTransport> trans(buffer);
128 TJSONProtocol protocol(trans);
129 ts.read(&protocol);
130 return ts;
131}
