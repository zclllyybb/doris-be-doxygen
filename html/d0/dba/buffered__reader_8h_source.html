34class PrefetchBufferedReader;
35struct PrefetchBuffer : std::enable_shared_from_this<PrefetchBuffer> {
36 enum class BufferStatus { RESET, PENDING, PREFETCHED, CLOSED };
37 PrefetchBuffer() = default;
38 PrefetchBuffer(size_t start_offset, size_t end_offset, size_t buffer_size,
39 size_t whole_buffer_size, io::FileReader* reader)
40 : _start_offset(start_offset),
41 _end_offset(end_offset),
42 _size(buffer_size),
43 _whole_buffer_size(whole_buffer_size),
44 _reader(reader),
45 _buf(buffer_size, '0') {}
46 PrefetchBuffer(PrefetchBuffer&& other)
47 : _offset(other._offset),
48 _start_offset(other._start_offset),
49 _end_offset(other._end_offset),
50 _size(other._size),
51 _whole_buffer_size(other._whole_buffer_size),
52 _reader(other._reader),
53 _buf(std::move(other._buf)) {}
54 ~PrefetchBuffer() = default;
55 size_t _offset;
56 // [_start_offset, _end_offset) is the range that can be prefetched.
57 // Notice that the reader can read out of [_start_offset, _end_offset), because FE does not align the file
58 // according to the format when splitting it.
59 size_t _start_offset;
60 size_t _end_offset;
61 size_t _size;
62 size_t _len {0};
63 size_t _whole_buffer_size;
64 io::FileReader* _reader;
65 std::string _buf;
66 BufferStatus _buffer_status {BufferStatus::RESET};
67 std::mutex _lock;
68 std::condition_variable _prefetched;
69 Status _prefetch_status {Status::OK()};
70 // @brief: reset the start offset of this buffer to offset
71 // @param: the new start offset for this buffer
72 void reset_offset(size_t offset);
73 // @brief: start to fetch the content between [_offset, _offset + _size)
74 void prefetch_buffer();
75 // @brief: used by BufferedReader to read the prefetched data
76 // @param[off] read start address
77 // @param[buf] buffer to put the actual content
78 // @param[buf_len] maximum len trying to read
79 // @param[bytes_read] actual bytes read
80 Status read_buffer(size_t off, const char* buf, size_t buf_len, size_t* bytes_read);
81 // @brief: shut down the buffer until the prior prefetching task is done
82 void close();
83 // @brief: to detect whether this buffer contains off
84 // @param[off] detect offset
85 bool inline contains(size_t off) const { return _offset <= off && off < _offset + _size; }
86};

95class PrefetchBufferedReader : public io::FileReader {
96public:
97 PrefetchBufferedReader(io::FileReaderSPtr reader, int64_t offset, int64_t length,
98 int64_t buffer_size = -1L);
99 ~PrefetchBufferedReader() override;
100
101 Status close() override;
102
103 const io::Path& path() const override { return _reader->path(); }
104
105 size_t size() const override { return _size; }
106
107 bool closed() const override { return _closed; }
108
109 std::shared_ptr<io::FileSystem> fs() const override { return _reader->fs(); }
110
111protected:
112 Status read_at_impl(size_t offset, Slice result, size_t* bytes_read,
113 const IOContext* io_ctx) override;
114
115private:
116 size_t get_buffer_pos(int64_t position) const {
117 return (position % _whole_pre_buffer_size) / s_max_pre_buffer_size;
118 }
119 size_t get_buffer_offset(int64_t position) const {
120 return (position / s_max_pre_buffer_size) * s_max_pre_buffer_size;
121 }
122 void reset_all_buffer(size_t position) {
123 for (int64_t i = 0; i < _pre_buffers.size(); i++) {
124 int64_t cur_pos = position + i * s_max_pre_buffer_size;
125 int cur_buf_pos = get_buffer_pos(cur_pos);
126 // reset would do all the prefetch work
127 _pre_buffers[cur_buf_pos]->reset_offset(get_buffer_offset(cur_pos));
128 }
129 }
130
131 io::FileReaderSPtr _reader;
132 int64_t _start_offset;
133 int64_t _end_offset;
134 int64_t s_max_pre_buffer_size = 4 * 1024 * 1024; // 4MB
135 std::vector<std::shared_ptr<PrefetchBuffer>> _pre_buffers;
136 int64_t _whole_pre_buffer_size;
137 bool _initialized = false;
138 bool _closed = false;
139 size_t _size;
140};
