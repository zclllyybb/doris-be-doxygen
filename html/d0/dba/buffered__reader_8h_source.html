32// Buffered Reader
33// Add a cache layer between the caller and the file reader to reduce the
34// times of calls to the read function to speed up.
35class BufferedReader : public FileReader {
36public:
37 // If the reader need the file size, set it when construct FileReader.
38 // There is no other way to set the file size.
39 // buffered_reader will acquire reader
40 // -1 means using config buffered_reader_buffer_size_bytes
41 BufferedReader(RuntimeProfile* profile, FileReader* reader, int64_t = -1L);
42 virtual ~BufferedReader();
43
44 virtual Status open() override;
45
46 // Read
47 virtual Status read(uint8_t* buf, int64_t buf_len, int64_t* bytes_read, bool* eof) override;
48 virtual Status readat(int64_t position, int64_t nbytes, int64_t* bytes_read,
49 void* out) override;
50 virtual Status read_one_message(std::unique_ptr<uint8_t[]>* buf, int64_t* length) override;
51 virtual int64_t size() override;
52 virtual Status seek(int64_t position) override;
53 virtual Status tell(int64_t* position) override;
54 virtual void close() override;
55 virtual bool closed() override;
56
57private:
58 Status _fill();
59 Status _read_once(int64_t position, int64_t nbytes, int64_t* bytes_read, void* out);
60
61private:
62 RuntimeProfile* _profile;
63 std::unique_ptr<FileReader> _reader;
64 char* _buffer;
65 int64_t _buffer_size;
66 int64_t _buffer_offset;
67 int64_t _buffer_limit;
68 int64_t _cur_offset;
69
70 int64_t _read_count = 0;
71 int64_t _remote_read_count = 0;
72 int64_t _remote_bytes = 0;
73
74 // total time cost in this reader
75 RuntimeProfile::Counter* _read_timer = nullptr;
76 // time cost of "_reader", "remote" because "_reader" is always a remote reader
77 RuntimeProfile::Counter* _remote_read_timer = nullptr;
78 // counter of calling read()
79 RuntimeProfile::Counter* _read_counter = nullptr;
80 // counter of calling "remote read()"
81 RuntimeProfile::Counter* _remote_read_counter = nullptr;
82 RuntimeProfile::Counter* _remote_read_bytes = nullptr;
83 RuntimeProfile::Counter* _remote_read_rate = nullptr;
84};
