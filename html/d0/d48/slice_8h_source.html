213inline int Slice::compare(const Slice& b) const {
214 const int min_len = (size < b.size) ? size : b.size;
215 int r = mem_compare(data, b.data, min_len);
216 if (r == 0) {
217 if (size < b.size)
218 r = -1;
219 else if (size > b.size)
220 r = +1;
221 }
222 return r;
223}

246// A move-only type which manage the lifecycle of externally allocated data.
247// Unlike std::unique_ptr<uint8_t[]>, OwnedSlice remembers the size of data so that clients can access
248// the underlying buffer as a Slice.
249//
250// Usage example:
251// OwnedSlice read_page(PagePointer pp);
252// {
253// OwnedSlice page_data(new uint8_t[pp.size], pp.size);
254// Status s = _file.read_at(pp.offset, owned.slice());
255// if (!s.ok()) {
256// return s; // `page_data` destructs, deallocate underlying buffer
257// }
258// return page_data; // transfer ownership of buffer into the caller
259// }
260//
261class OwnedSlice {
262public:
263 OwnedSlice() : _slice((uint8_t*)nullptr, 0) {}
264
265 OwnedSlice(uint8_t* _data, size_t size) : _slice(_data, size) {}
266
267 OwnedSlice(OwnedSlice&& src) : _slice(src._slice) {
268 src._slice.data = nullptr;
269 src._slice.size = 0;
270 }
271
272 OwnedSlice& operator=(OwnedSlice&& src) {
273 if (this != &src) {
274 std::swap(_slice, src._slice);
275 }
276 return *this;
277 }
278
279 ~OwnedSlice() { delete[] _slice.data; }
280
281 const Slice& slice() const { return _slice; }
282
283private:
284 // disable copy constructor and copy assignment
285 OwnedSlice(const OwnedSlice&) = delete;
286 void operator=(const OwnedSlice&) = delete;
287
288 Slice _slice;
289};
