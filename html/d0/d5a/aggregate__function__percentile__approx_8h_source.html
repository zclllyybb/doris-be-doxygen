32struct PercentileApproxState {
33 static constexpr double INIT_QUANTILE = -1.0;
34 PercentileApproxState() = default;
35 ~PercentileApproxState() = default;
36
37 void init(double compression = 10000) {
38 if (!init_flag) {
39 //https://doris.apache.org/zh-CN/sql-reference/sql-functions/aggregate-functions/percentile_approx.html#description
40 //The compression parameter setting range is [2048, 10000].
41 //If the value of compression parameter is not specified set, or is outside the range of [2048, 10000],
42 //will use the default value of 10000
43 if (compression < 2048 || compression > 10000) {
44 compression = 10000;
45 }
46 digest.reset(new TDigest(compression));
47 compressions = compression;
48 init_flag = true;
49 }
50 }
51
52 void write(BufferWritable& buf) const {
53 write_binary(init_flag, buf);
54 if (!init_flag) {
55 return;
56 }
57
58 write_binary(target_quantile, buf);
59 write_binary(compressions, buf);
60 uint32_t serialize_size = digest->serialized_size();
61 std::string result(serialize_size, '0');
62 DCHECK(digest.get() != nullptr);
63 digest->serialize((uint8_t*)result.c_str());
64
65 write_binary(result, buf);
66 }
67
68 void read(BufferReadable& buf) {
69 read_binary(init_flag, buf);
70 if (!init_flag) {
71 return;
72 }
73
74 read_binary(target_quantile, buf);
75 read_binary(compressions, buf);
76 std::string str;
77 read_binary(str, buf);
78 digest.reset(new TDigest(compressions));
79 digest->unserialize((uint8_t*)str.c_str());
80 }
81
82 double get() const {
83 if (init_flag) {
84 return digest->quantile(target_quantile);
85 } else {
86 return std::nan("");
87 }
88 }
89
90 void merge(const PercentileApproxState& rhs) {
91 if (!rhs.init_flag) {
92 return;
93 }
94 if (init_flag) {
95 DCHECK(digest.get() != nullptr);
96 digest->merge(rhs.digest.get());
97 } else {
98 digest.reset(new TDigest(compressions));
99 digest->merge(rhs.digest.get());
100 init_flag = true;
101 }
102 if (target_quantile == PercentileApproxState::INIT_QUANTILE) {
103 target_quantile = rhs.target_quantile;
104 }
105 }
106
107 void add(double source, double quantile) {
108 digest->add(source);
109 target_quantile = quantile;
110 }
111
112 void reset() {
113 target_quantile = INIT_QUANTILE;
114 init_flag = false;
115 digest.reset(new TDigest(compressions));
116 }
117
118 bool init_flag = false;
119 std::unique_ptr<TDigest> digest = nullptr;
120 double target_quantile = INIT_QUANTILE;
121 double compressions = 10000;
122};

184template <bool is_nullable>
185class AggregateFunctionPercentileApproxTwoParams : public AggregateFunctionPercentileApprox {
186public:
187 AggregateFunctionPercentileApproxTwoParams(const DataTypes& argument_types_)
188 : AggregateFunctionPercentileApprox(argument_types_) {}
189 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
190 Arena*) const override {
191 if constexpr (is_nullable) {
192 double column_data[2] = {0, 0};
193
194 for (int i = 0; i < 2; ++i) {
195 const auto* nullable_column = check_and_get_column<ColumnNullable>(columns[i]);
196 if (nullable_column == nullptr) { //Not Nullable column
197 const auto& column = static_cast<const ColumnVector<Float64>&>(*columns[i]);
198 column_data[i] = column.get_float64(row_num);
199 } else if (!nullable_column->is_null_at(
200 row_num)) { // Nullable column && Not null data
201 const auto& column = static_cast<const ColumnVector<Float64>&>(
202 nullable_column->get_nested_column());
203 column_data[i] = column.get_float64(row_num);
204 } else { // Nullable column && null data
205 if (i == 0) {
206 return;
207 }
208 }
209 }
210
211 this->data(place).init();
212 this->data(place).add(column_data[0], column_data[1]);
213
214 } else {
215 const auto& sources = static_cast<const ColumnVector<Float64>&>(*columns[0]);
216 const auto& quantile = static_cast<const ColumnVector<Float64>&>(*columns[1]);
217
218 this->data(place).init();
219 this->data(place).add(sources.get_float64(row_num), quantile.get_float64(row_num));
220 }
221 }

265struct PercentileState {
266 std::vector<Counts> vec_counts;
267 std::vector<double> vec_quantile {-1};
268 bool inited_flag = false;
269
270 void write(BufferWritable& buf) const {
271 write_binary(inited_flag, buf);
272 int size_num = vec_quantile.size();
273 write_binary(size_num, buf);
274 for (const auto& quantile : vec_quantile) {
275 write_binary(quantile, buf);
276 }
277 std::string serialize_str;
278 for (const auto& counts : vec_counts) {
279 serialize_str.resize(counts.serialized_size(), '0');
280 counts.serialize((uint8_t*)serialize_str.c_str());
281 write_binary(serialize_str, buf);
282 }
283 }
284
285 void read(BufferReadable& buf) {
286 read_binary(inited_flag, buf);
287 int size_num = 0;
288 read_binary(size_num, buf);
289 double data = 0.0;
290 vec_quantile.clear();
291 for (int i = 0; i < size_num; ++i) {
292 read_binary(data, buf);
293 vec_quantile.emplace_back(data);
294 }
295 StringRef ref;
296 vec_counts.clear();
297 vec_counts.resize(size_num);
298 for (int i = 0; i < size_num; ++i) {
299 read_binary(ref, buf);
300 vec_counts[i].unserialize((uint8_t*)ref.data);
301 }
302 }
303
304 void add(int64_t source, const PaddedPODArray<Float64>& quantiles, int arg_size) {
305 if (!inited_flag) {
306 vec_counts.resize(arg_size);
307 vec_quantile.resize(arg_size, -1);
308 inited_flag = true;
309 for (int i = 0; i < arg_size; ++i) {
310 vec_quantile[i] = quantiles[i];
311 }
312 }
313 for (int i = 0; i < arg_size; ++i) {
314 vec_counts[i].increment(source, 1);
315 }
316 }
317
318 void merge(const PercentileState& rhs) {
319 if (!rhs.inited_flag) {
320 return;
321 }
322 int size_num = rhs.vec_quantile.size();
323 if (!inited_flag) {
324 vec_counts.resize(size_num);
325 vec_quantile.resize(size_num, -1);
326 inited_flag = true;
327 }
328
329 for (int i = 0; i < size_num; ++i) {
330 if (vec_quantile[i] == -1.0) {
331 vec_quantile[i] = rhs.vec_quantile[i];
332 }
333 vec_counts[i].merge(&(rhs.vec_counts[i]));
334 }
335 }
336
337 void reset() {
338 vec_counts.clear();
339 vec_quantile.clear();
340 inited_flag = false;
341 }
342
343 double get() const {
344 auto result = vec_counts[0].terminate(vec_quantile[0]); //DoubleVal
345 return result.val;
346 }
347
348 void insert_result_into(IColumn& to) const {
349 auto& column_data = static_cast<ColumnVector<Float64>&>(to).get_data();
350 for (int i = 0; i < vec_counts.size(); ++i) {
351 column_data.push_back(vec_counts[i].terminate(vec_quantile[i]).val);
352 }
353 }
354};

400 : public IAggregateFunctionDataHelper<PercentileState, AggregateFunctionPercentileArray> {
401public:
402 AggregateFunctionPercentileArray(const DataTypes& argument_types_)
403 : IAggregateFunctionDataHelper<PercentileState, AggregateFunctionPercentileArray>(
404 argument_types_, {}) {}
405
406 String get_name() const override { return "percentile_array"; }
407
408 DataTypePtr get_return_type() const override {

411
412 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
413 Arena*) const override {
414 const auto& sources = static_cast<const ColumnVector<Int64>&>(*columns[0]);
415 const auto& quantile_array = static_cast<const ColumnArray&>(*columns[1]);
416 const auto& offset_column_data = quantile_array.get_offsets();
417 const auto& nested_column =
418 static_cast<const ColumnNullable&>(quantile_array.get_data()).get_nested_column();
419 const auto& nested_column_data = static_cast<const ColumnVector<Float64>&>(nested_column);
420
421 AggregateFunctionPercentileArray::data(place).add(
422 sources.get_int(row_num), nested_column_data.get_data(),
423 offset_column_data.data()[row_num] - offset_column_data.data()[row_num - 1]);
424 }

427 AggregateFunctionPercentileArray::data(place).reset();
428 }

433 AggregateFunctionPercentileArray::data(rhs));

437 AggregateFunctionPercentileArray::data(place).write(buf);

441 Arena*) const override {

444
445 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {

450 AggregateFunctionPercentileArray::data(place).insert_result_into(
451 col_null->get_nested_column());
452 col_null->get_null_map_data().resize_fill(col_null->get_nested_column().size(), 0);
453 } else {
454 AggregateFunctionPercentileArray::data(place).insert_result_into(to_nested_col);
455 }
456 to_arr.get_offsets().push_back(to_nested_col.size());
457 }
