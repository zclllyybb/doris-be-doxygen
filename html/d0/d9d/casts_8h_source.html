22// Use implicit_cast as a safe version of static_cast or const_cast
23// for implicit conversions. For example:
24// - Upcasting in a type hierarchy.
25// - Performing arithmetic conversions (int32 to int64, int to double, etc.).
26// - Adding const or volatile qualifiers.
27//
28// In general, implicit_cast can be used to convert this code
29// To to = from;
30// DoSomething(to);
31// to this
32// DoSomething(implicit_cast<To>(from));
33//
34// base::identity_ is used to make a non-deduced context, which
35// forces all callers to explicitly specify the template argument.
36template <typename To>
37inline To implicit_cast(typename base::identity_<To>::type to) {
38 return to;
39}

42// are specified. It's obsolete and should not be used.

48// When you upcast (that is, cast a pointer from type Foo to type
49// SuperclassOfFoo), it's fine to use implicit_cast<>, since upcasts
50// always succeed. When you downcast (that is, cast a pointer from
51// type Foo to type SubclassOfFoo), static_cast<> isn't safe, because
52// how do you know the pointer is really of type SubclassOfFoo? It
53// could be a bare Foo, or of type DifferentSubclassOfFoo. Thus,
54// when you downcast, you should use this macro. In debug mode, we
55// use dynamic_cast<> to double-check the downcast is legal (we die
56// if it's not). In normal mode, we do the efficient static_cast<>
57// instead. Thus, it's important to test in debug mode to make sure
58// the cast is legal!
59// This is the only place in the code we should use dynamic_cast<>.
60// In particular, you SHOULDN'T be using dynamic_cast<> in order to
61// do RTTI (eg code like this:
62// if (dynamic_cast<Subclass1>(foo)) HandleASubclass1Object(foo);
63// if (dynamic_cast<Subclass2>(foo)) HandleASubclass2Object(foo);
64// You should design the code some other way not to need this.
65
66template <typename To, typename From> // use like this: down_cast<T*>(foo);
67inline To down_cast(From* f) { // so we only accept pointers
68 // Ensures that To is a sub-type of From *. This test is here only
69 // for compile-time type checking, and has no overhead in an
70 // optimized build at run-time, as it will be optimized away
71 // completely.
72
73 // TODO(user): This should use COMPILE_ASSERT.
74 if (false) {
75 ::implicit_cast<From*, To>(NULL);
76 }
77
78 // uses RTTI in dbg and fastbuild. asserts are disabled in opt builds.
79 assert(f == NULL || dynamic_cast<To>(f) != NULL);
80 return static_cast<To>(f);
81}

83// Overload of down_cast for references. Use like this: down_cast<T&>(foo).
84// The code is slightly convoluted because we're still using the pointer
85// form of dynamic cast. (The reference form throws an exception if it
86// fails.)
87//
88// There's no need for a special const overload either for the pointer
89// or the reference form. If you call down_cast with a const T&, the
90// compiler will just bind From to const T.
91template <typename To, typename From>
92inline To down_cast(From& f) {
93 COMPILE_ASSERT(std::is_reference<To>::value, target_type_not_a_reference);
94 using ToAsPointer = typename std::remove_reference<To>::type*;
95 if (false) {
96 // Compile-time check that To inherits from From. See above for details.
97 ::implicit_cast<From*, ToAsPointer>(NULL);
98 }
99
100 assert(dynamic_cast<ToAsPointer>(&f) != NULL); // RTTI: debug mode only
101 return static_cast<To>(f);
102}

104// bit_cast<Dest,Source> is a template function that implements the
105// equivalent of "*reinterpret_cast<Dest*>(&source)". We need this in
106// very low-level functions like the protobuf library and fast math
107// support.
108//
109// float f = 3.14159265358979;
110// int i = bit_cast<int32>(f);
111// // i = 0x40490fdb
112//
113// The classical address-casting method is:
114//
115// // WRONG
116// float f = 3.14159265358979; // WRONG
117// int i = * reinterpret_cast<int*>(&f); // WRONG
118//
119// The address-casting method actually produces undefined behavior
120// according to ISO C++ specification section 3.10 -15 -. Roughly, this
121// section says: if an object in memory has one type, and a program
122// accesses it with a different type, then the result is undefined
123// behavior for most values of "different type".
124//
125// This is true for any cast syntax, either *(int*)&f or
126// *reinterpret_cast<int*>(&f). And it is particularly true for
127// conversions between integral lvalues and floating-point lvalues.
128//
129// The purpose of 3.10 -15- is to allow optimizing compilers to assume
130// that expressions with different types refer to different memory. gcc
131// 4.0.1 has an optimizer that takes advantage of this. So a
132// non-conforming program quietly produces wildly incorrect output.
133//
134// The problem is not the use of reinterpret_cast. The problem is type
135// punning: holding an object in memory of one type and reading its bits
136// back using a different type.
137//
138// The C++ standard is more subtle and complex than this, but that
139// is the basic idea.
140//
141// Anyways ...
142//
143// bit_cast<> calls memcpy() which is blessed by the standard,
144// especially by the example in section 3.9 . Also, of course,
145// bit_cast<> wraps up the nasty logic in one place.
146//
147// Fortunately memcpy() is very fast. In optimized mode, with a
148// constant size, gcc 2.95.3, gcc 4.0.1, and msvc 7.1 produce inline
149// code with the minimal amount of data movement. On a 32-bit system,
150// memcpy(d,s,4) compiles to one load and one store, and memcpy(d,s,8)
151// compiles to two loads and two stores.
152//
153// I tested this code with gcc 2.95.3, gcc 4.0.1, icc 8.1, and msvc 7.1.
154//
155// WARNING: if Dest or Source is a non-POD type, the result of the memcpy
156// is likely to surprise you.
157//
158// Props to Bill Gibbons for the compile time assertion technique and
159// Art Komninos and Igor Tandetnik for the msvc experiments.
160//
161// -- mec 2005-10-17
162
163template <class Dest, class Source>
164inline Dest bit_cast(const Source& source) {
165 // Compile time assertion: sizeof(Dest) == sizeof(Source)
166 // A compile error here means your Dest and Source have different sizes.
167 COMPILE_ASSERT(sizeof(Dest) == sizeof(Source), VerifySizesAreEqual);
168
169 Dest dest;
170 memcpy(&dest, &source, sizeof(dest));
171 return dest;
172}

174// **** Enumeration Casts and Tests
175//
176// C++ requires that the value of an integer that is converted to an
177// enumeration be within the value bounds of the enumeration. Modern
178// compilers can and do take advantage of this requirement to optimize
179// programs. So, using a raw static_cast with enums can be bad. See
180//
181// The following templates and macros enable casting from an int to an enum
182// with checking against the appropriate bounds. First, when defining an
183// enumeration, identify the limits of the values of its enumerators.
184//
185// enum A { A_min = -18, A_max = 33 };
186// MAKE_ENUM_LIMITS(A, A_min, A_max)
187//
188// Convert an enum to an int in one of two ways. The preferred way is a
189// tight conversion, which ensures that A_min <= value <= A_max.
190//
191// A var = tight_enum_cast<A>(3);
192//
193// However, the C++ language defines the set of possible values for an
194// enumeration to be essentially the range of a bitfield that can represent
195// all the enumerators, i.e. those within the nearest containing power
196// of two. In the example above, the nearest positive power of two is 64,
197// and so the upper bound is 63. The nearest negative power of two is
198// -32 and so the lower bound is -32 (two's complement), which is upgraded
199// to match the upper bound, becoming -64. The values within this range
200// of -64 to 63 are valid, according to the C++ standard. You can cast
201// values within this range as follows.
202//
203// A var = loose_enum_cast<A>(45);
204//
205// These casts will log a message if the value does not reside within the
206// specified range, and will be fatal when in debug mode.
207//
208// For those times when an assert too strong, there are test functions.
209//
210// bool var = tight_enum_test<A>(3);
211// bool var = loose_enum_test<A>(45);
212//
213// For code that needs to use the enumeration value if and only if
214// it is good, there is a function that both tests and casts.
215//
216// int i = ....;
217// A var;
218// if (tight_enum_test_cast<A>(i, &var))
219// .... // use valid var with value as indicated by i
220// else
221// .... // handle invalid enum cast
222//
223// The enum test/cast facility is currently limited to enumerations that
224// fit within an int. It is also limited to two's complement ints.
225
226// ** Implementation Description
227//
228// The enum_limits template class captures the minimum and maximum
229// enumerator. All uses of this template are intended to be of
230// specializations, so the generic has a field to identify itself as
231// not specialized. The test/cast templates assert specialization.
232
233template <typename Enum>
234class enum_limits {
235public:
236 static const Enum min_enumerator = 0;
237 static const Enum max_enumerator = 0;
238 static const bool is_specialized = false;
239};
240
241// Now we define the macro to define the specialization for enum_limits.
242// The specialization checks that the enumerators fit within an int.
243// This checking relies on integral promotion.
244
245#define MAKE_ENUM_LIMITS(ENUM_TYPE, ENUM_MIN, ENUM_MAX) \
246 template <> \
247 class enum_limits<ENUM_TYPE> { \
248 public: \
249 static const ENUM_TYPE min_enumerator = ENUM_MIN; \
250 static const ENUM_TYPE max_enumerator = ENUM_MAX; \
251 static const bool is_specialized = true; \
252 COMPILE_ASSERT(ENUM_MIN >= INT_MIN, enumerator_too_negative_for_int); \
253 COMPILE_ASSERT(ENUM_MAX <= INT_MAX, enumerator_too_positive_for_int); \
254 };

256// The loose enum test/cast is actually the more complicated one,
257// because of the problem of finding the bounds.
258//
259// The unary upper bound, ub, on a positive number is its positive
260// saturation, i.e. for a value v within pow(2,k-1) <= v < pow(2,k),
261// the upper bound is pow(2,k)-1.
262//
263// The unary lower bound, lb, on a negative number is its negative
264// saturation, i.e. for a value v within -pow(2,k) <= v < -pow(2,k-1),
265// the lower bound is -pow(2,k).
266//
267// The actual bounds are (1) the binary upper bound over the maximum
268// enumerator and the one's complement of a negative minimum enumerator
269// and (2) the binary lower bound over the minimum enumerator and the
270// one's complement of the positive maximum enumerator, except that if no
271// enumerators are negative, the lower bound is zero.
272//
273// The algorithm relies heavily on the observation that
274//
275// a,b>0 then ub(a,b) == ub(a) | ub(b) == ub(a|b)
276// a,b<0 then lb(a,b) == lb(a) & lb(b) == lb(a&b)
277//
278// Note that the compiler will boil most of this code away
279// because of value propagation on the constant enumerator bounds.
280
281template <typename Enum>
282inline bool loose_enum_test(int e_val) {
283 COMPILE_ASSERT(enum_limits<Enum>::is_specialized, missing_MAKE_ENUM_LIMITS);
284 const Enum e_min = enum_limits<Enum>::min_enumerator;
285 const Enum e_max = enum_limits<Enum>::max_enumerator;
286 COMPILE_ASSERT(sizeof(e_val) == 4 || sizeof(e_val) == 8, unexpected_int_size);
287
288 // Find the unary bounding negative number of e_min and e_max.
289
290 // Find the unary bounding negative number of e_max.
291 // This would be b_min = e_max < 0 ? e_max : ~e_max,
292 // but we want to avoid branches to help the compiler.
293 int e_max_sign = e_max >> (sizeof(e_val) * 8 - 1);
294 int b_min = ~e_max_sign ^ e_max;
295
296 // Find the binary bounding negative of both e_min and e_max.
297 b_min &= e_min;
298
299 // However, if e_min is positive, the result will be positive.
300 // Now clear all bits right of the most significant clear bit,
301 // which is a negative saturation for negative numbers.
302 // In the case of positive numbers, this is flush to zero.
303 b_min &= b_min >> 1;
304 b_min &= b_min >> 2;
305 b_min &= b_min >> 4;
306 b_min &= b_min >> 8;
307 b_min &= b_min >> 16;
308#if INT_MAX > 2147483647
309 b_min &= b_min >> 32;
310#endif
311
312 // Find the unary bounding positive number of e_max.
313 int b_max = e_max_sign ^ e_max;
314
315 // Find the binary bounding positive number of that
316 // and the unary bounding positive number of e_min.
317 int e_min_sign = e_min >> (sizeof(e_val) * 8 - 1);
318 b_max |= e_min_sign ^ e_min;
319
320 // Now set all bits right of the most significant set bit,
321 // which is a positive saturation for positive numbers.
322 b_max |= b_max >> 1;
323 b_max |= b_max >> 2;
324 b_max |= b_max >> 4;
325 b_max |= b_max >> 8;
326 b_max |= b_max >> 16;
327#if INT_MAX > 2147483647
328 b_max |= b_max >> 32;
329#endif
330
331 // Finally test the bounds.
332 return b_min <= e_val && e_val <= b_max;
333}

335template <typename Enum>
336inline bool tight_enum_test(int e_val) {
337 COMPILE_ASSERT(enum_limits<Enum>::is_specialized, missing_MAKE_ENUM_LIMITS);
338 const Enum e_min = enum_limits<Enum>::min_enumerator;
339 const Enum e_max = enum_limits<Enum>::max_enumerator;
340 return e_min <= e_val && e_val <= e_max;
341}
