372 std::string to_string(UInt32 scale) const {
373 if (value == std::numeric_limits<T>::min()) {
374 fmt::memory_buffer buffer;
375 fmt::format_to(buffer, "{}", value);
376 std::string res {buffer.data(), buffer.size()};
377 res.insert(res.size() - scale, ".");
378 return res;
379 }
380
381 static constexpr auto precision =
382 std::is_same_v<T, Int32>
383 ? BeConsts::MAX_DECIMAL32_PRECISION
384 : (std::is_same_v<T, Int64> ? BeConsts::MAX_DECIMAL64_PRECISION
385 : BeConsts::MAX_DECIMAL128_PRECISION);
386 bool is_nagetive = value < 0;
387 int max_result_length = precision + (scale > 0) // Add a space for decimal place
388 + (scale == precision) // Add a space for leading 0
389 + (is_nagetive); // Add a space for negative sign
390 std::string str = std::string(max_result_length, '0');
391
392 T abs_value = value;
393 int pos = 0;
394
395 if (is_nagetive) {
396 abs_value = -value;
397 str[pos++] = '-';
398 }
399
400 T whole_part = abs_value;
401 T frac_part;
402 if (scale) {
403 whole_part = abs_value / decimal_scale_multiplier<T>(scale);
404 frac_part = abs_value % decimal_scale_multiplier<T>(scale);
405 }
406 auto end = fmt::format_to(str.data() + pos, "{}", whole_part);
407 pos = end - str.data();
408
409 if (scale) {
410 str[pos++] = '.';
411 for (auto end_pos = pos + scale - 1; end_pos >= pos && frac_part > 0;
412 --end_pos, frac_part /= 10) {
413 str[end_pos] += frac_part % 10;
414 }
415 }
416
417 str.resize(pos + scale);
418 return str;
419 }
420
421 T value;
422};
