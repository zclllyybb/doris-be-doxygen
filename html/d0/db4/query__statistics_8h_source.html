47// This is responsible for collecting query statistics, usually it consists of
48// two parts, one is current fragment or plan's statistics, the other is sub fragment
49// or plan's statistics and QueryStatisticsRecvr is responsible for collecting it.
50class QueryStatistics {
51public:
52 QueryStatistics()
53 : scan_rows(0), scan_bytes(0), cpu_ms(0), returned_rows(0), max_peak_memory_bytes(0) {}
54 ~QueryStatistics();
55
56 void merge(const QueryStatistics& other);
57
58 void add_scan_rows(int64_t scan_rows) { this->scan_rows += scan_rows; }
59
60 void add_scan_bytes(int64_t scan_bytes) { this->scan_bytes += scan_bytes; }
61
62 void add_cpu_ms(int64_t cpu_ms) { this->cpu_ms += cpu_ms; }
63
64 NodeStatistics* add_nodes_statistics(int64_t node_id) {
65 NodeStatistics* nodeStatistics = nullptr;
66 auto iter = _nodes_statistics_map.find(node_id);
67 if (iter == _nodes_statistics_map.end()) {
68 nodeStatistics = new NodeStatistics;
69 _nodes_statistics_map[node_id] = nodeStatistics;
70 } else {
71 nodeStatistics = iter->second;
72 }
73 return nodeStatistics;
74 }
75
76 void set_returned_rows(int64_t num_rows) { this->returned_rows = num_rows; }
77
78 void set_max_peak_memory_bytes(int64_t max_peak_memory_bytes) {
79 this->max_peak_memory_bytes = max_peak_memory_bytes;
80 }
81
82 void merge(QueryStatisticsRecvr* recvr);
83
84 // Get the maximum value from the peak memory collected by all node statistics
85 int64_t calculate_max_peak_memory_bytes();
86
87 void clearNodeStatistics();
88
89 void clear() {
90 scan_rows = 0;
91 scan_bytes = 0;
92 cpu_ms = 0;
93 returned_rows = 0;
94 max_peak_memory_bytes = 0;
95 clearNodeStatistics();
96 }
97
98 void to_pb(PQueryStatistics* statistics);
99
100 void from_pb(const PQueryStatistics& statistics);
101
102private:
103 int64_t scan_rows;
104 int64_t scan_bytes;
105 int64_t cpu_ms;
106 // number rows returned by query.
107 // only set once by result sink when closing.
108 int64_t returned_rows;
109 // Maximum memory peak for all backends.
110 // only set once by result sink when closing.
111 int64_t max_peak_memory_bytes;
112 // The statistics of the query on each backend.
113 typedef std::unordered_map<int64_t, NodeStatistics*> NodeStatisticsMap;
114 NodeStatisticsMap _nodes_statistics_map;
115};
