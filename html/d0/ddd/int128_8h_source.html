65// This is a POD form of uint128 which can be used for static variables which

67struct uint128_pod {
68 // Note: The ordering of fields is different than 'class uint128' but the
69 // same as its 2-arg constructor. This enables more obvious initialization

145inline uint128 operator-(const uint128& val) {
146 const uint64 hi_flip = ~Uint128High64(val);
147 const uint64 lo_flip = ~Uint128Low64(val);
148 const uint64 lo_add = lo_flip + 1;
149 if (lo_add < lo_flip) {
150 return uint128(hi_flip + 1, lo_add);
151 }
152 return uint128(hi_flip, lo_add);
153}

192inline uint128 operator<<(const uint128& val, int amount) {
193 // uint64 shifts of >= 64 are undefined, so we will need some special-casing.
194 if (amount < 64) {
195 if (amount == 0) {
196 return val;
197 }
198 uint64 new_hi = (Uint128High64(val) << amount) | (Uint128Low64(val) >> (64 - amount));
199 uint64 new_lo = Uint128Low64(val) << amount;
200 return uint128(new_hi, new_lo);
201 } else if (amount < 128) {
202 return uint128(Uint128Low64(val) << (amount - 64), 0);
203 } else {
204 return uint128(0, 0);
205 }
206}
207
208inline uint128 operator>>(const uint128& val, int amount) {
209 // uint64 shifts of >= 64 are undefined, so we will need some special-casing.
210 if (amount < 64) {
211 if (amount == 0) {
212 return val;
213 }
214 uint64 new_hi = Uint128High64(val) >> amount;
215 uint64 new_lo = (Uint128Low64(val) >> amount) | (Uint128High64(val) << (64 - amount));
216 return uint128(new_hi, new_lo);
217 } else if (amount < 128) {
218 return uint128(0, Uint128High64(val) >> (amount - 64));
219 } else {
220 return uint128(0, 0);
221 }
222}

224inline uint128& uint128::operator<<=(int amount) {
225 // uint64 shifts of >= 64 are undefined, so we will need some special-casing.
226 if (amount < 64) {
227 if (amount != 0) {
228 hi_ = (hi_ << amount) | (lo_ >> (64 - amount));
229 lo_ = lo_ << amount;
230 }
231 } else if (amount < 128) {
232 hi_ = lo_ << (amount - 64);
233 lo_ = 0;
234 } else {
235 hi_ = 0;
236 lo_ = 0;
237 }
238 return *this;
239}

285inline uint128& uint128::operator*=(const uint128& b) {
286 uint64 a96 = hi_ >> 32;
287 uint64 a64 = hi_ & 0xffffffffu;
288 uint64 a32 = lo_ >> 32;
289 uint64 a00 = lo_ & 0xffffffffu;
290 uint64 b96 = b.hi_ >> 32;
291 uint64 b64 = b.hi_ & 0xffffffffu;
292 uint64 b32 = b.lo_ >> 32;
293 uint64 b00 = b.lo_ & 0xffffffffu;
294 // multiply [a96 .. a00] x [b96 .. b00]
295 // terms higher than c96 disappear off the high side
296 // terms c96 and c64 are safe to ignore carry bit
297 uint64 c96 = a96 * b00 + a64 * b32 + a32 * b64 + a00 * b96;
298 uint64 c64 = a64 * b00 + a32 * b32 + a00 * b64;
299 this->hi_ = (c96 << 32) + c64;
300 this->lo_ = 0;
301 // add terms after this one at a time to capture carry
302 *this += uint128(a32 * b00) << 32;
303 *this += uint128(a00 * b32) << 32;
304 *this += a00 * b00;
305 return *this;
306}
