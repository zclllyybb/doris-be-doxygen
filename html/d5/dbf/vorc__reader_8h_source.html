124 template <typename CppType, typename OrcColumnType>
125 Status _decode_flat_column(const std::string& col_name, const MutableColumnPtr& data_column,
126 orc::ColumnVectorBatch* cvb, size_t num_values) {
127 SCOPED_RAW_TIMER(&_statistics.decode_value_time);
128 OrcColumnType* data = dynamic_cast<OrcColumnType*>(cvb);
129 if (data == nullptr) {
130 return Status::InternalError("Wrong data type for colum '{}'", col_name);
131 }
132 auto* cvb_data = data->data.data();
133 auto& column_data = static_cast<ColumnVector<CppType>&>(*data_column).get_data();
134 auto origin_size = column_data.size();
135 column_data.resize(origin_size + num_values);
136 for (int i = 0; i < num_values; ++i) {
137 column_data[origin_size + i] = (CppType)cvb_data[i];
138 }
139 return Status::OK();
140 }

142 template <typename DecimalPrimitiveType>
143 void _init_decimal_converter(const DataTypePtr& data_type, DecimalScaleParams& scale_params,
144 const int32_t orc_decimal_scale) {
145 if (scale_params.scale_type != DecimalScaleParams::NOT_INIT) {
146 return;
147 }
148 auto* decimal_type = reinterpret_cast<DataTypeDecimal<Decimal<DecimalPrimitiveType>>*>(
149 const_cast<IDataType*>(remove_nullable(data_type).get()));
150 auto dest_scale = decimal_type->get_scale();
151 if (dest_scale > orc_decimal_scale) {
152 scale_params.scale_type = DecimalScaleParams::SCALE_UP;
153 scale_params.scale_factor = DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(
154 dest_scale - orc_decimal_scale);
155 } else if (dest_scale < orc_decimal_scale) {
156 scale_params.scale_type = DecimalScaleParams::SCALE_DOWN;
157 scale_params.scale_factor = DecimalScaleParams::get_scale_factor<DecimalPrimitiveType>(
158 orc_decimal_scale - dest_scale);
159 } else {
160 scale_params.scale_type = DecimalScaleParams::NO_SCALE;
161 scale_params.scale_factor = 1;
162 }
163 }

165 template <typename DecimalPrimitiveType, typename OrcColumnType>
166 Status _decode_explicit_decimal_column(const std::string& col_name,
167 const MutableColumnPtr& data_column,
168 const DataTypePtr& data_type,
169 DecimalScaleParams& scale_params,
170 orc::ColumnVectorBatch* cvb, size_t num_values) {
171 OrcColumnType* data = dynamic_cast<OrcColumnType*>(cvb);
172 if (data == nullptr) {
173 return Status::InternalError("Wrong data type for colum '{}'", col_name);
174 }
175 _init_decimal_converter<DecimalPrimitiveType>(data_type, scale_params, data->scale);
176
177 auto* cvb_data = data->values.data();
178 auto& column_data =
179 static_cast<ColumnVector<DecimalPrimitiveType>&>(*data_column).get_data();
180 auto origin_size = column_data.size();
181 column_data.resize(origin_size + num_values);
182
183 for (int i = 0; i < num_values; ++i) {
184 int128_t value;
185 if constexpr (std::is_same_v<OrcColumnType, orc::Decimal64VectorBatch>) {
186 value = static_cast<int128_t>(cvb_data[i]);
187 } else {
188 uint64_t hi = data->values[i].getHighBits();
189 uint64_t lo = data->values[i].getLowBits();
190 value = (((int128_t)hi) << 64) | (int128_t)lo;
191 }
192 if (scale_params.scale_type == DecimalScaleParams::SCALE_UP) {
193 value *= scale_params.scale_factor;
194 } else if (scale_params.scale_type == DecimalScaleParams::SCALE_DOWN) {
195 value /= scale_params.scale_factor;
196 }
197 auto& v = reinterpret_cast<DecimalPrimitiveType&>(column_data[origin_size + i]);
198 v = (DecimalPrimitiveType)value;
199 }
200 return Status::OK();
201 }

203 template <typename DecimalPrimitiveType>
204 Status _decode_decimal_column(const std::string& col_name, const MutableColumnPtr& data_column,
205 const DataTypePtr& data_type, DecimalScaleParams& scale_params,
206 orc::ColumnVectorBatch* cvb, size_t num_values) {
207 SCOPED_RAW_TIMER(&_statistics.decode_value_time);
208 if (dynamic_cast<orc::Decimal64VectorBatch*>(cvb) != nullptr) {
209 return _decode_explicit_decimal_column<DecimalPrimitiveType, orc::Decimal64VectorBatch>(
210 col_name, data_column, data_type, scale_params, cvb, num_values);
211 } else {
212 return _decode_explicit_decimal_column<DecimalPrimitiveType,
213 orc::Decimal128VectorBatch>(
214 col_name, data_column, data_type, scale_params, cvb, num_values);
215 }

218 template <typename CppType, typename DorisColumnType, typename OrcColumnType>
219 Status _decode_time_column(const std::string& col_name, const MutableColumnPtr& data_column,
220 orc::ColumnVectorBatch* cvb, size_t num_values) {
221 SCOPED_RAW_TIMER(&_statistics.decode_value_time);
222 auto* data = dynamic_cast<OrcColumnType*>(cvb);
223 if (data == nullptr) {
224 return Status::InternalError("Wrong data type for colum '{}'", col_name);
225 }
226 auto& column_data = static_cast<ColumnVector<DorisColumnType>&>(*data_column).get_data();
227 auto origin_size = column_data.size();
228 column_data.resize(origin_size + num_values);
229 for (int i = 0; i < num_values; ++i) {
230 auto& v = reinterpret_cast<CppType&>(column_data[origin_size + i]);
231 if constexpr (std::is_same_v<OrcColumnType, orc::LongVectorBatch>) { // date
232 int64_t& date_value = data->data[i];
233 v.from_unixtime(date_value * 24 * 60 * 60, _time_zone); // day to seconds
234 if constexpr (std::is_same_v<CppType, VecDateTimeValue>) {
235 // we should cast to date if using date v1.
236 v.cast_to_date();
237 }
238 } else { // timestamp
239 v.from_unixtime(data->data[i], _time_zone);
240 if constexpr (std::is_same_v<CppType, DateV2Value<DateTimeV2ValueType>>) {
241 // nanoseconds will lose precision. only keep microseconds.
242 v.set_microsecond(data->nanoseconds[i] / 1000);
243 }
244 }
245 }
246 return Status::OK();
247 }
