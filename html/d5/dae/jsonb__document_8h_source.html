11/*
12 * This header defines JsonbDocument, JsonbKeyValue, and various value classes
13 * which are derived from JsonbValue, and a forward iterator for container
14 * values - essentially everything that is related to JSONB binary data
15 * structures.
16 *
17 * Implementation notes:
18 *
19 * None of the classes in this header file can be instantiated directly (i.e.
20 * you cannot create a JsonbKeyValue or JsonbValue object - all constructors
21 * are declared non-public). We use the classes as wrappers on the packed JSONB
22 * bytes (serialized), and cast the classes (types) to the underlying packed
23 * byte array.
24 *
25 * For the same reason, we cannot define any JSONB value class to be virtual,
26 * since we never call constructors, and will not instantiate vtbl and vptrs.
27 *
28 * Therefore, the classes are defined as packed structures (i.e. no data
29 * alignment and padding), and the private member variables of the classes are
30 * defined precisely in the same order as the JSONB spec. This ensures we
31 * access the packed JSONB bytes correctly.
32 *
33 * The packed structures are highly optimized for in-place operations with low
34 * overhead. The reads (and in-place writes) are performed directly on packed
35 * bytes. There is no memory allocation at all at runtime.
36 *
37 * For updates/writes of values that will expand the original JSONB size, the
38 * write will fail, and the caller needs to handle buffer increase.
39 *
40 * ** Iterator **
41 * Both ObjectVal class and ArrayVal class have iterator type that you can use
42 * to declare an iterator on a container object to go through the key-value
43 * pairs or value list. The iterator has both non-const and const types.
44 *
45 * Note: iterators are forward direction only.
46 *
47 * ** Query **
48 * Querying into containers is through the member functions find (for key/value
49 * pairs) and get (for array elements), and is in streaming style. We don't
50 * need to read/scan the whole JSONB packed bytes in order to return results.
51 * Once the key/index is found, we will stop search. You can use text to query
52 * both objects and array (for array, text will be converted to integer index),
53 * and use index to retrieve from array. Array index is 0-based.
54 *
55 * ** External dictionary **
56 * During query processing, you can also pass a call-back function, so the
57 * search will first try to check if the key string exists in the dictionary.
58 * If so, search will be based on the id instead of the key string.
59 * @author Tian Xia <tianx@fb.com>
60 *
61 * this file is copied from
62 * https://github.com/facebook/mysql-5.6/blob/fb-mysql-5.6.35/fbson/FbsonDocument.h
63 * and modified by Doris
64 */
65

91/*
92 * JsonbType defines 10 primitive types and 2 container types, as described
93 * below.
94 *
95 * primitive_value ::=
96 * 0x00 //null value (0 byte)
97 * | 0x01 //boolean true (0 byte)
98 * | 0x02 //boolean false (0 byte)
99 * | 0x03 int8 //char/int8 (1 byte)
100 * | 0x04 int16 //int16 (2 bytes)
101 * | 0x05 int32 //int32 (4 bytes)
102 * | 0x06 int64 //int64 (8 bytes)
103 * | 0x07 double //floating point (8 bytes)
104 * | 0x08 string //variable length string
105 * | 0x09 binary //variable length binary
106 *
107 * container ::=
108 * 0x0A int32 key_value_list //object, int32 is the total bytes of the object
109 * | 0x0B int32 value_list //array, int32 is the total bytes of the array
110 */
111enum class JsonbType : char {
112 T_Null = 0x00,
113 T_True = 0x01,
114 T_False = 0x02,
115 T_Int8 = 0x03,
116 T_Int16 = 0x04,
117 T_Int32 = 0x05,
118 T_Int64 = 0x06,
119 T_Double = 0x07,
120 T_String = 0x08,
121 T_Binary = 0x09,
122 T_Object = 0x0A,
123 T_Array = 0x0B,
124 T_Int128 = 0x0C,
125 T_Float = 0x0D,
126 NUM_TYPES,
127};

132 * JSONB value. However, you can use the static method "createValue" to get any

134 *
135 * JsonbDocument object also dereferences to an object container value
136 * (ObjectVal) once JSONB is loaded.
137 *

213/*
214 * JsonbFwdIteratorT implements JSONB's iterator template.
215 *
216 * Note: it is an FORWARD iterator only due to the design of JSONB format.
217 */
218template <class Iter_Type, class Cont_Type>
219class JsonbFwdIteratorT {
220public:
221 typedef Iter_Type iterator;
222 typedef typename std::iterator_traits<Iter_Type>::pointer pointer;
223 typedef typename std::iterator_traits<Iter_Type>::reference reference;
224
225public:
226 explicit JsonbFwdIteratorT() : current_(nullptr) {}
227 explicit JsonbFwdIteratorT(const iterator& i) : current_(i) {}
228
229 // allow non-const to const iterator conversion (same container type)
230 template <class Iter_Ty>
231 JsonbFwdIteratorT(const JsonbFwdIteratorT<Iter_Ty, Cont_Type>& rhs) : current_(rhs.base()) {}
232
233 bool operator==(const JsonbFwdIteratorT& rhs) const { return (current_ == rhs.current_); }
234
235 bool operator!=(const JsonbFwdIteratorT& rhs) const { return !operator==(rhs); }
236
237 bool operator<(const JsonbFwdIteratorT& rhs) const { return (current_ < rhs.current_); }
238
239 bool operator>(const JsonbFwdIteratorT& rhs) const { return !operator<(rhs); }
240
241 JsonbFwdIteratorT& operator++() {
242 current_ = (iterator)(((char*)current_) + current_->numPackedBytes());
243 return *this;
244 }
245
246 JsonbFwdIteratorT operator++(int) {
247 auto tmp = *this;
248 current_ = (iterator)(((char*)current_) + current_->numPackedBytes());
249 return tmp;
250 }
251
252 explicit operator pointer() { return current_; }
253
254 reference operator*() const { return *current_; }
255
256 pointer operator->() const { return current_; }
257
258 iterator base() const { return current_; }
259
260private:
261 iterator current_;
262};

269/*
270 * JsonbKeyValue class defines JSONB key type, as described below.
271 *
272 * key ::=
273 * 0x00 int8 //1-byte dictionary id
274 * | int8 (byte*) //int8 (>0) is the size of the key string
275 *
276 * value ::= primitive_value | container
277 *
278 * JsonbKeyValue can be either an id mapping to the key string in an external
279 * dictionary, or it is the original key string. Whether to read an id or a
280 * string is decided by the first byte (size_).
281 *
282 * Note: a key object must be followed by a value object. Therefore, a key
283 * object implicitly refers to a key-value pair, and you can get the value
284 * object right after the key object. The function numPackedBytes hence
285 * indicates the total size of the key-value pair, so that we will be able go
286 * to next pair from the key.
287 *
288 * ** Dictionary size **
289 * By default, the dictionary size is 255 (1-byte). Users can define
290 * "USE_LARGE_DICT" to increase the dictionary size to 655535 (2-byte).
291 */
292class JsonbKeyValue {
293public:
294 static const int sMaxKeyId = 65535;
295 typedef uint16_t keyid_type;
296
297 static const uint8_t sMaxKeyLen = 64;
298
299 // size of the key. 0 indicates it is stored as id
300 uint8_t klen() const { return size_; }
301
302 // get the key string. Note the string may not be null terminated.
303 const char* getKeyStr() const { return key_.str_; }
304
305 keyid_type getKeyId() const { return key_.id_; }
306
307 unsigned int keyPackedBytes() const {
308 return size_ ? (sizeof(size_) + size_) : (sizeof(size_) + sizeof(keyid_type));
309 }
310
311 JsonbValue* value() const { return (JsonbValue*)(((char*)this) + keyPackedBytes()); }
312
313 // size of the total packed bytes (key+value)
314 unsigned int numPackedBytes() const;
315
316private:
317 uint8_t size_;
318
319 union key_ {
320 keyid_type id_;
321 char str_[1];
322 } key_;
323
324 JsonbKeyValue();
325};

562/*
563 * BlobVal is the base class (derived from JsonbValue) for string and binary
564 * types. The size_ indicates the total bytes of the payload_.
565 */
566class JsonbBlobVal : public JsonbValue {
567public:
568 // size of the blob payload only
569 unsigned int getBlobLen() const { return size_; }
570
571 // return the blob as byte array
572 const char* getBlob() const { return payload_; }
573
574 // size of the total packed bytes
575 unsigned int numPackedBytes() const { return sizeof(JsonbValue) + sizeof(size_) + size_; }
576 friend class JsonbDocument;
577
578protected:
579 uint32_t size_;
580 char payload_[0];
581
582 // set new blob bytes
583 bool internalSetVal(const char* blob, uint32_t blobSize) {
584 // if we cannot fit the new blob, fail the operation
585 if (blobSize > size_) {
586 return false;
587 }
588
589 memcpy(payload_, blob, blobSize);
590
591 // Set the reset of the bytes to 0. Note we cannot change the size_ of the
592 // current payload, as all values are packed.
593 memset(payload_ + blobSize, 0, size_ - blobSize);
594
595 return true;
596 }
597
598 JsonbBlobVal();
599};

618/*
619 * String type
620 * Note: JSONB string may not be a c-string (NULL-terminated)
621 */
622class JsonbStringVal : public JsonbBlobVal {
623public:
624 bool setVal(const char* str, uint32_t blobSize) {
625 if (!isString()) {
626 return false;
627 }
628
629 return internalSetVal(str, blobSize);
630 }
631 /*
632 This function return the actual size of a string. Since for
633 a string, it can be null-terminated with null paddings or it
634 can take all the space in the payload_ without null in the end.
635 So we need to check it to get the true actual length of a string.
636 */
637 size_t length() {
638 // It's an empty string
639 if (0 == size_) return size_;
640 // The string stored takes all the spaces in payload_
641 if (payload_[size_ - 1] != 0) {
642 return size_;
643 }
644 // It's shorter than the size of payload_
645 return strnlen(payload_, size_);
646 }
647 // convert the string (case insensitive) to a boolean value
648 // "false": 0
649 // "true": 1
650 // all other strings: -1
651 int getBoolVal() {
652 if (size_ == 4 && tolower(payload_[0]) == 't' && tolower(payload_[1]) == 'r' &&
653 tolower(payload_[2]) == 'u' && tolower(payload_[3]) == 'e')
654 return 1;
655 else if (size_ == 5 && tolower(payload_[0]) == 'f' && tolower(payload_[1]) == 'a' &&
656 tolower(payload_[2]) == 'l' && tolower(payload_[3]) == 's' &&
657 tolower(payload_[4]) == 'e')
658 return 0;
659 else
660 return -1;
661 }
662
663private:
664 JsonbStringVal();
665};

690/*
691 * Object type
692 */
693class ObjectVal : public ContainerVal {
694public:
695 typedef JsonbKeyValue value_type;
696 typedef value_type* pointer;
697 typedef const value_type* const_pointer;
698 typedef JsonbFwdIteratorT<pointer, ObjectVal> iterator;
699 typedef JsonbFwdIteratorT<const_pointer, ObjectVal> const_iterator;
700
701public:
702 const_iterator search(const char* key, hDictFind handler = nullptr) const {
703 return const_cast<ObjectVal*>(this)->search(key, handler);
704 }
705
706 const_iterator search(const char* key, unsigned int klen, hDictFind handler = nullptr) const {
707 return const_cast<ObjectVal*>(this)->search(key, klen, handler);
708 }
709
710 const_iterator search(int key_id) const { return const_cast<ObjectVal*>(this)->search(key_id); }
711 iterator search(const char* key, hDictFind handler = nullptr) {
712 if (!key) {
713 return end();
714 }
715 return search(key, (unsigned int)strlen(key), handler);
716 }
717
718 iterator search(const char* key, unsigned int klen, hDictFind handler = nullptr) {
719 if (!key || !klen) return end();
720
721 int key_id = -1;
722 if (handler && (key_id = handler(key, klen)) >= 0) {
723 return search(key_id);
724 }
725 return internalSearch(key, klen);
726 }
727
728 iterator search(int key_id) {
729 if (key_id < 0 || key_id > JsonbKeyValue::sMaxKeyId) return end();
730
731 const char* pch = payload_;
732 const char* fence = payload_ + size_;
733
734 while (pch < fence) {
735 JsonbKeyValue* pkey = (JsonbKeyValue*)(pch);
736 if (!pkey->klen() && key_id == pkey->getKeyId()) {
737 return iterator(pkey);
738 }
739 pch += pkey->numPackedBytes();
740 }
741
742 assert(pch == fence);
743 return end();
744 }

747 return const_cast<ObjectVal*>(this)->find(key, handler);
748 }

751 return const_cast<ObjectVal*>(this)->find(key, klen, handler);

753 JsonbValue* find(int key_id) const { return const_cast<ObjectVal*>(this)->find(key_id); }

755 // find the JSONB value by a key string (null terminated)

1071inline JsonbValue* JsonbValue::findPath(const char* key_path, unsigned int kp_len,
1072 const char* delim = ".", hDictFind handler = nullptr) {
1073 if (!key_path) return nullptr;
1074 if (kp_len == 0) return this;
1075
1076 // skip $ and . at beginning
1077 if (kp_len > 0 && *key_path == '$') {
1078 key_path++;
1079 kp_len--;
1080 if (kp_len > 0 && *key_path == '.') {
1081 key_path++;
1082 kp_len--;
1083 }
1084 }
1085
1086 if (kp_len == 0) return this;
1087
1088 if (!delim) delim = "."; // default delimiter
1089
1090 JsonbValue* pval = this;
1091 const char* fence = key_path + kp_len;
1092 char idx_buf[21]; // buffer to parse array index (integer value)
1093
1094 while (pval && key_path < fence) {
1095 const char* key = key_path;
1096 unsigned int klen = 0;
1097 const char* left_bracket = nullptr;
1098 const char* right_bracket = nullptr;
1099 size_t idx_len = 0;
1100 // find the current key and [] bracket position
1101 for (; key_path != fence && *key_path != *delim; ++key_path, ++klen) {
1102 if ('[' == *key_path) {
1103 left_bracket = key_path;
1104 } else if (']' == *key_path) {
1105 right_bracket = key_path;
1106 }
1107 }
1108
1109 // check brackets and array index length
1110 if (left_bracket || right_bracket) {
1111 if (!left_bracket || !right_bracket) {
1112 return nullptr;
1113 }
1114 // check the last char is ]
1115 if (key + klen - 1 != right_bracket) {
1116 return nullptr;
1117 }
1118 // the part before left_bracket is object key
1119 klen = left_bracket - key;
1120 // the part between left_bracket and right_bracket is array index
1121 idx_len = right_bracket - left_bracket - 1;
1122 }
1123
1124 if (!klen && !idx_len) return nullptr;
1125
1126 // get value of key in object
1127 if (klen) {
1128 if (LIKELY(pval->type_ == JsonbType::T_Object)) {
1129 pval = ((ObjectVal*)pval)->find(key, klen, handler);
1130 if (!pval) return nullptr;
1131 } else {
1132 return nullptr;
1133 }
1134 }
1135
1136 // get value at idx in array
1137 if (idx_len) {
1138 if (LIKELY(pval->type_ == JsonbType::T_Array)) {
1139 if (idx_len >= sizeof(idx_buf)) return nullptr;
1140 memcpy(idx_buf, left_bracket + 1, idx_len);
1141 idx_buf[idx_len] = 0;
1142
1143 char* end = nullptr;
1144 int index = (int)strtol(idx_buf, &end, 10);
1145 if (end && !*end)
1146 pval = ((ArrayVal*)pval)->get(index);
1147 else
1148 // incorrect index string
1149 return nullptr;
1150
1151 // doris::StringParser::ParseResult parse_result;
1152 // int index = doris::StringParser::string_to_int<int>(left_bracket + 1, idx_len, &parse_result);
1153 // if (parse_result == doris::StringParser::ParseResult::PARSE_SUCCESS)
1154 } else {
1155 return nullptr;
1156 }
1157 }
1158
1159 // skip the delimiter
1160 if (key_path < fence) {
1161 ++key_path;
1162 if (key_path == fence)
1163 // we have a trailing delimiter at the end
1164 return nullptr;
1165 }
1166 }
1167
1168 return pval;
1169}
