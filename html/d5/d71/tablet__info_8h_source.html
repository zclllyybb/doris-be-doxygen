105class VOlapTablePartKeyComparator {
106public:
107 VOlapTablePartKeyComparator(const std::vector<uint16_t>& slot_locs) : _slot_locs(slot_locs) {}
108
109 // return true if lhs < rhs
110 // 'row' is -1 mean
111 bool operator()(const BlockRow* lhs, const BlockRow* rhs) const {
112 if (lhs->second == -1) {
113 return false;
114 } else if (rhs->second == -1) {
115 return true;
116 }
117
118 for (auto slot_loc : _slot_locs) {
119 auto res = lhs->first->get_by_position(slot_loc).column->compare_at(
120 lhs->second, rhs->second, *rhs->first->get_by_position(slot_loc).column, -1);
121 if (res != 0) {
122 return res < 0;
123 }
124 }
125 // equal, return false
126 return false;
127 }
128
129private:
130 const std::vector<uint16_t>& _slot_locs;
131};
