38// Utility functions for doing atoi/atof on non-null terminated strings. On micro benchmarks,
39// this is significantly faster than libc (atoi/strtol and atof/strtod).
40//
41// Strings with leading and trailing whitespaces are accepted.
42// Branching is heavily optimized for the non-whitespace successful case.
43// All the StringTo* functions first parse the input string assuming it has no leading whitespace.
44// If that first attempt was unsuccessful, these functions retry the parsing after removing
45// whitespace. Therefore, strings with whitespace take a perf hit on branch mis-prediction.
46//
47// For overflows, we are following the mysql behavior, to cap values at the max/min value for that
48// data type. This is different from hive, which returns NULL for overflow slots for int types
49// and inf/-inf for float types.
50//
51// Things we tried that did not work:
52// - lookup table for converting character to digit
53// Improvements (TODO):
54// - Validate input using _sidd_compare_ranges
55// - Since we know the length, we can parallelize this: i.e. result = 100*s[0] + 10*s[1] + s[2]

134 template <typename T>
135 static Status split_string_to_map(const std::string& base, const T element_separator,
136 const T key_value_separator,
137 std::map<std::string, std::string>* result) {
138 int key_pos = 0;
139 int key_end;
140 int val_pos;
141 int val_end;
142
143 while ((key_end = base.find(key_value_separator, key_pos)) != std::string::npos) {
144 if ((val_pos = base.find_first_not_of(key_value_separator, key_end)) ==
145 std::string::npos) {
146 break;
147 }
148 if ((val_end = base.find(element_separator, val_pos)) == std::string::npos) {
149 val_end = base.size();
150 }
151 result->insert(std::make_pair(base.substr(key_pos, key_end - key_pos),
152 base.substr(val_pos, val_end - val_pos)));
153 key_pos = val_end;
154 if (key_pos != std::string::npos) {
155 ++key_pos;
156 }
157 }
158
159 return Status::OK();
160 }

213 static inline int skip_leading_whitespace(const char* s, int len) {
214 int i = 0;

216 ++i;
217 }

223 return LIKELY(c == ' ') ||

229template <typename T>
230T StringParser::string_to_int_internal(const char* s, int len, ParseResult* result) {
231 if (UNLIKELY(len <= 0)) {
232 *result = PARSE_FAILURE;
233 return 0;
234 }
235
236 typedef typename std::make_unsigned<T>::type UnsignedT;
237 UnsignedT val = 0;
238 UnsignedT max_val = StringParser::numeric_limits<T>(false);
239 bool negative = false;
240 int i = 0;
241 switch (*s) {
242 case '-':
243 negative = true;
244 max_val = StringParser::numeric_limits<T>(false) + 1;
245 [[fallthrough]];
246 case '+':
247 ++i;
248 }
249
250 // This is the fast path where the string cannot overflow.
251 if (LIKELY(len - i < StringParseTraits<T>::max_ascii_len())) {
252 val = string_to_int_no_overflow<UnsignedT>(s + i, len - i, result);
253 return static_cast<T>(negative ? -val : val);
254 }
255
256 const T max_div_10 = max_val / 10;
257 const T max_mod_10 = max_val % 10;
258
259 int first = i;
260 for (; i < len; ++i) {
261 if (LIKELY(s[i] >= '0' && s[i] <= '9')) {
262 T digit = s[i] - '0';
263 // This is a tricky check to see if adding this digit will cause an overflow.
264 if (UNLIKELY(val > (max_div_10 - (digit > max_mod_10)))) {
265 *result = PARSE_OVERFLOW;
266 return negative ? -max_val : max_val;
267 }
268 val = val * 10 + digit;
269 } else {
270 if ((UNLIKELY(i == first || !is_all_whitespace(s + i, len - i)))) {
271 // Reject the string because either the first char was not a digit,
272 // or the remaining chars are not all whitespace
273 *result = PARSE_FAILURE;
274 return 0;
275 }
276 // Returning here is slightly faster than breaking the loop.
277 *result = PARSE_SUCCESS;
278 return static_cast<T>(negative ? -val : val);
279 }
280 }
281 *result = PARSE_SUCCESS;
282 return static_cast<T>(negative ? -val : val);
283}

287 if (UNLIKELY(len <= 0)) {
288 *result = PARSE_FAILURE;

294 int i = 0;

302

332template <typename T>
333T StringParser::string_to_int_internal(const char* s, int len, int base, ParseResult* result) {
334 typedef typename std::make_unsigned<T>::type UnsignedT;
335 UnsignedT val = 0;
336 UnsignedT max_val = StringParser::numeric_limits<T>(false);
337 bool negative = false;
338 if (UNLIKELY(len <= 0)) {
339 *result = PARSE_FAILURE;
340 return 0;
341 }
342 int i = 0;
343 switch (*s) {
344 case '-':
345 negative = true;
346 max_val = StringParser::numeric_limits<T>(false) + 1;
347 [[fallthrough]];
348 case '+':
349 i = 1;
350 }
351
352 const T max_div_base = max_val / base;
353 const T max_mod_base = max_val % base;
354
355 int first = i;
356 for (; i < len; ++i) {
357 T digit;
358 if (LIKELY(s[i] >= '0' && s[i] <= '9')) {
359 digit = s[i] - '0';
360 } else if (s[i] >= 'a' && s[i] <= 'z') {
361 digit = (s[i] - 'a' + 10);
362 } else if (s[i] >= 'A' && s[i] <= 'Z') {
363 digit = (s[i] - 'A' + 10);
364 } else {
365 if ((UNLIKELY(i == first || !is_all_whitespace(s + i, len - i)))) {
366 // Reject the string because either the first char was not an alpha/digit,
367 // or the remaining chars are not all whitespace
368 *result = PARSE_FAILURE;
369 return 0;
370 }
371 // skip trailing whitespace.
372 break;
373 }
374
375 // Bail, if we encounter a digit that is not available in base.
376 if (digit >= base) {
377 break;
378 }
379
380 // This is a tricky check to see if adding this digit will cause an overflow.
381 if (UNLIKELY(val > (max_div_base - (digit > max_mod_base)))) {
382 *result = PARSE_OVERFLOW;
383 return static_cast<T>(negative ? -max_val : max_val);
384 }
385 val = val * base + digit;
386 }
387 *result = PARSE_SUCCESS;
388 return static_cast<T>(negative ? -val : val);
389}

391template <typename T>
392T StringParser::string_to_int_no_overflow(const char* s, int len, ParseResult* result) {
393 T val = 0;
394 if (UNLIKELY(len == 0)) {
395 *result = PARSE_SUCCESS;
396 return val;
397 }
398 // Factor out the first char for error handling speeds up the loop.
399 if (LIKELY(s[0] >= '0' && s[0] <= '9')) {
400 val = s[0] - '0';
401 } else {
402 *result = PARSE_FAILURE;
403 return 0;
404 }
405 for (int i = 1; i < len; ++i) {
406 if (LIKELY(s[i] >= '0' && s[i] <= '9')) {
407 T digit = s[i] - '0';
408 val = val * 10 + digit;
409 } else {
410 if ((UNLIKELY(!is_all_whitespace(s + i, len - i)))) {
411 *result = PARSE_FAILURE;
412 return 0;
413 }
414 *result = PARSE_SUCCESS;
415 return val;
416 }
417 }
418 *result = PARSE_SUCCESS;
419 return val;
420}

422template <typename T>
423T StringParser::string_to_float_internal(const char* s, int len, ParseResult* result) {
424 int i = 0;
425 // skip leading spaces
426 for (; i < len; ++i) {
427 if (!is_whitespace(s[i])) {
428 break;
429 }
430 }
431
432 // skip back spaces
433 int j = len - 1;
434 for (; j >= i; j--) {
435 if (!is_whitespace(s[j])) {
436 break;
437 }
438 }
439
440 // skip leading '+', from_chars can handle '-'
441 if (i < len && s[i] == '+') {
442 i++;
443 }
444 if (UNLIKELY(i > j)) {
445 *result = PARSE_FAILURE;
446 return 0;
447 }
448
449 // Use double here to not lose precision while accumulating the result
450 double val = 0;
451 auto res = fast_float::from_chars(s + i, s + j + 1, val);
452
453 if (res.ec == std::errc() && res.ptr == s + j + 1) {
454 if (abs(val) == std::numeric_limits<T>::infinity()) {
455 auto contain_inf = false;
456 for (int k = i; k < j + 1; k++) {
457 if (s[k] == 'i' || s[k] == 'I') {
458 contain_inf = true;
459 break;
460 }
461 }
462
463 *result = contain_inf ? PARSE_SUCCESS : PARSE_OVERFLOW;
464 } else {
465 *result = PARSE_SUCCESS;
466 }
467 return val;
468 } else {
469 *result = PARSE_FAILURE;
470 }
471 return 0;
472}

474inline bool StringParser::string_to_bool_internal(const char* s, int len, ParseResult* result) {
475 *result = PARSE_SUCCESS;
476
477 if (len >= 4 && (s[0] == 't' || s[0] == 'T')) {
478 bool match = (s[1] == 'r' || s[1] == 'R') && (s[2] == 'u' || s[2] == 'U') &&
479 (s[3] == 'e' || s[3] == 'E');
480 if (match && LIKELY(is_all_whitespace(s + 4, len - 4))) {
481 return true;
482 }
483 } else if (len >= 5 && (s[0] == 'f' || s[0] == 'F')) {
484 bool match = (s[1] == 'a' || s[1] == 'A') && (s[2] == 'l' || s[2] == 'L') &&
485 (s[3] == 's' || s[3] == 'S') && (s[4] == 'e' || s[4] == 'E');
486 if (match && LIKELY(is_all_whitespace(s + 5, len - 5))) {
487 return false;
488 }
489 }
490
491 *result = PARSE_FAILURE;
492 return false;
493}

548template <typename T>
549T StringParser::string_to_decimal(const char* s, int len, int type_precision, int type_scale,
550 ParseResult* result) {
551 // Special cases:
552 // 1) '' == Fail, an empty string fails to parse.
553 // 2) ' # ' == #, leading and trailing white space is ignored.
554 // 3) '.' == 0, a single dot parses as zero (for consistency with other types).
555 // 4) '#.' == '#', a trailing dot is ignored.
556
557 // Ignore leading and trailing spaces.
558 while (len > 0 && is_whitespace(*s)) {
559 ++s;
560 --len;
561 }
562 while (len > 0 && is_whitespace(s[len - 1])) {
563 --len;
564 }
565
566 bool is_negative = false;
567 if (len > 0) {
568 switch (*s) {
569 case '-':
570 is_negative = true;
571 [[fallthrough]];
572 case '+':
573 ++s;
574 --len;
575 }
576 }
577
578 // Ignore leading zeros.
579 bool found_value = false;
580 while (len > 0 && UNLIKELY(*s == '0')) {
581 found_value = true;
582 ++s;
583 --len;
584 }
585
586 // Ignore leading zeros even after a dot. This allows for differentiating between
587 // cases like 0.01e2, which would fit in a DECIMAL(1, 0), and 0.10e2, which would
588 // overflow.
589 int scale = 0;
590 int found_dot = 0;
591 if (len > 0 && *s == '.') {
592 found_dot = 1;
593 ++s;
594 --len;
595 while (len > 0 && UNLIKELY(*s == '0')) {
596 found_value = true;
597 ++scale;
598 ++s;
599 --len;
600 }
601 }
602
603 int precision = 0;
604 bool found_exponent = false;
605 int8_t exponent = 0;
606 T value = 0;
607 for (int i = 0; i < len; ++i) {
608 const char& c = s[i];
609 if (LIKELY('0' <= c && c <= '9')) {
610 found_value = true;
611 // Ignore digits once the type's precision limit is reached. This avoids
612 // overflowing the underlying storage while handling a string like
613 // 10000000000e-10 into a DECIMAL(1, 0). Adjustments for ignored digits and
614 // an exponent will be made later.
615 if (LIKELY(type_precision > precision)) {
616 value = (value * 10) + (c - '0'); // Benchmarks are faster with parenthesis...
617 }
618 DCHECK(value >= 0); // For some reason //DCHECK_GE doesn't work with __int128.
619 ++precision;
620 scale += found_dot;
621 } else if (c == '.' && LIKELY(!found_dot)) {
622 found_dot = 1;
623 } else if ((c == 'e' || c == 'E') && LIKELY(!found_exponent)) {
624 found_exponent = true;
625 exponent = string_to_int_internal<int8_t>(s + i + 1, len - i - 1, result);
626 if (UNLIKELY(*result != StringParser::PARSE_SUCCESS)) {
627 if (*result == StringParser::PARSE_OVERFLOW && exponent < 0) {
628 *result = StringParser::PARSE_UNDERFLOW;
629 }
630 return 0;
631 }
632 break;
633 } else {
634 if (value == 0) {
635 *result = StringParser::PARSE_FAILURE;
636 return 0;
637 }
638 *result = StringParser::PARSE_SUCCESS;
639 if constexpr (std::is_same_v<T, vectorized::Int128I>) {
640 value *= get_scale_multiplier<__int128>(type_scale - scale);
641 } else {
642 value *= get_scale_multiplier<T>(type_scale - scale);
643 }
644
645 return is_negative ? T(-value) : T(value);
646 }
647 }
648
649 // Find the number of truncated digits before adjusting the precision for an exponent.
650 if (exponent > scale) {
651 // Ex: 0.1e3 (which at this point would have precision == 1 and scale == 1), the
652 // scale must be set to 0 and the value set to 100 which means a precision of 3.
653 precision += exponent - scale;
654
655 if constexpr (std::is_same_v<T, vectorized::Int128I>) {
656 value *= get_scale_multiplier<__int128>(exponent - scale);
657 } else {
658 value *= get_scale_multiplier<T>(exponent - scale);
659 }
660 scale = 0;
661 } else {
662 // Ex: 100e-4, the scale must be set to 4 but no adjustment to the value is needed,
663 // the precision must also be set to 4 but that will be done below for the
664 // non-exponent case anyways.
665 scale -= exponent;
666 }
667 // Ex: 0.001, at this point would have precision 1 and scale 3 since leading zeros
668 // were ignored during previous parsing.
669 if (scale > precision) {
670 precision = scale;
671 }
672
673 // Microbenchmarks show that beyond this point, returning on parse failure is slower
674 // than just letting the function run out.
675 *result = StringParser::PARSE_SUCCESS;
676 if (UNLIKELY(precision - scale > type_precision - type_scale)) {
677 *result = StringParser::PARSE_OVERFLOW;
678 } else if (UNLIKELY(scale > type_scale)) {
679 *result = StringParser::PARSE_UNDERFLOW;
680 int shift = scale - type_scale;
681 if (shift > 0) {
