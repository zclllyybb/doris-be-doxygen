80class KafkaEventCb : public RdKafka::EventCb {
81public:
82 void event_cb(RdKafka::Event& event) {
83 switch (event.type()) {
84 case RdKafka::Event::EVENT_ERROR:
85 LOG(INFO) << "kafka error: " << RdKafka::err2str(event.err())
86 << ", event: " << event.str();
87 break;
88 case RdKafka::Event::EVENT_STATS:
89 LOG(INFO) << "kafka stats: " << event.str();
90 break;
91
92 case RdKafka::Event::EVENT_LOG:
93 LOG(INFO) << "kafka log-" << event.severity() << "-" << event.fac().c_str()
94 << ", event: " << event.str();
95 break;
96
97 case RdKafka::Event::EVENT_THROTTLE:
98 LOG(INFO) << "kafka throttled: " << event.throttle_time() << "ms by "
99 << event.broker_name() << " id " << (int)event.broker_id();
100 break;
101
102 default:
103 LOG(INFO) << "kafka event: " << event.type()
104 << ", err: " << RdKafka::err2str(event.err()) << ", event: " << event.str();
105 break;
106 }
107 }
108};
