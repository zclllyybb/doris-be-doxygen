190struct FilePathDesc {

194 std::string filepath;

199 std::string debug_string() const {
200 std::stringstream ss;
201 ss << "storage_medium: " << to_string(storage_medium) << ", local_path: " << filepath;
202 if (!remote_path.empty()) {
203 ss << ", storage_name: " << storage_name << ", remote_path: " << remote_path;
204 }
205 return ss.str();
206 }
207 // REMOTE_CACHE is the local cache path for remote path, if a data_dir is REMOTE_CACHE,
208 // it means the tablet in it will be set as a remote path.
209 static bool is_remote(TStorageMedium::type checked_storage_medium) {
210 return checked_storage_medium == TStorageMedium::S3 ||
211 checked_storage_medium == TStorageMedium::REMOTE_CACHE;
212 }
213 bool is_remote() const { return is_remote(storage_medium); }

304 // Read "result.size" bytes from the file starting at "offset".
305 // Copies the resulting data into "result.data".
306 //
307 // If an error was encountered, returns a non-OK status.
308 //
309 // This method will internally retry on EINTR and "short reads" in order to
310 // fully read the requested number of bytes. In the event that it is not
311 // possible to read exactly 'length' bytes, an IOError is returned.
312 //
313 // Safe for concurrent use by multiple threads.
314 virtual Status read_at(uint64_t offset, const Slice* result) const = 0;
315
316 // Reads up to the "results" aggregate size, based on each Slice's "size",
317 // from the file starting at 'offset'. The Slices must point to already-allocated
318 // buffers for the data to be written to.
319 //
320 // If an error was encountered, returns a non-OK status.
321 //
322 // This method will internally retry on EINTR and "short reads" in order to
323 // fully read the requested number of bytes. In the event that it is not
324 // possible to read exactly 'length' bytes, an IOError is returned.
325 //
326 // Safe for concurrent use by multiple threads.
327 virtual Status readv_at(uint64_t offset, const Slice* result, size_t res_cnt) const = 0;
328
329 // read all data from this file
330 virtual Status read_all(std::string* content) const = 0;
331
332 // Return the size of this file
333 virtual Status size(uint64_t* size) const = 0;
334
335 // Return name of this file
336 virtual const std::string& file_name() const = 0;

339// A file abstraction for sequential writing. The implementation
340// must provide buffering since callers may append small fragments
341// at a time to the file.
342// Note: To avoid user misuse, WritableFile's API should support only
343// one of Append or PositionedAppend. We support only Append here.
344class WritableFile {
345public:
346 enum FlushMode { FLUSH_SYNC, FLUSH_ASYNC };
347
348 WritableFile() {}
349 virtual ~WritableFile() {}
350
351 // Append data to the end of the file
352 virtual Status append(const Slice& data) = 0;
353
354 // If possible, uses scatter-gather I/O to efficiently append
355 // multiple buffers to a file. Otherwise, falls back to regular I/O.
356 //
357 // For implementation specific quirks and details, see comments in
358 // implementation source code (e.g., env_posix.cc)
359 virtual Status appendv(const Slice* data, size_t cnt) = 0;
360
361 // Pre-allocates 'size' bytes for the file in the underlying filesystem.
362 // size bytes are added to the current pre-allocated size or to the current
363 // offset, whichever is bigger. In no case is the file truncated by this
364 // operation.
365 //
366 // On some implementations, preallocation is done without initializing the
367 // contents of the data blocks (as opposed to writing zeroes), requiring no
368 // IO to the data blocks.
369 //
370 // In no case is the file truncated by this operation.
371 virtual Status pre_allocate(uint64_t size) = 0;
372
373 virtual Status close() = 0;
374
375 // Flush all dirty data (not metadata) to disk.
376 //
377 // If the flush mode is synchronous, will wait for flush to finish and
378 // return a meaningful status.
379 virtual Status flush(FlushMode mode) = 0;
380
381 virtual Status sync() = 0;
382
383 virtual uint64_t size() const = 0;
384
385 // Returns the filename provided when the WritableFile was constructed.
386 virtual const std::string& filename() const = 0;
387
388private:

390 WritableFile(const WritableFile&);
