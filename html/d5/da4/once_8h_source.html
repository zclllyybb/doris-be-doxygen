29// Utility class for implementing thread-safe call-once semantics.
30//
31// call() will return stored result regardless of whether the first invocation
32// returns a success status or not.
33//
34// Example:
35// class Resource {
36// public:
37// Status init() {
38// _init_once.call([this] { return _do_init(); });
39// }
40//
41// bool is_inited() const {
42// return _init_once.has_called() && _init_once.stored_result().ok();
43// }
44// private:
45// Status _do_init() { /* init logic here */ }
46// DorisCallOnce<Status> _init_once;
47// };
48template <typename ReturnType>
49class DorisCallOnce {
50public:
51 DorisCallOnce() : _has_called(false) {}
52
53 // If the underlying `once_flag` has yet to be invoked, invokes the provided
54 // lambda and stores its return value. Otherwise, returns the stored Status.
55 template <typename Fn>
56 ReturnType call(Fn fn) {
57 std::call_once(_once_flag, [this, fn] {
58 _status = fn();
59 _has_called.store(true, std::memory_order_release);
60 });
61 return _status;
62 }
63
64 // Return whether `call` has been invoked or not.
65 bool has_called() const {
66 // std::memory_order_acquire here and std::memory_order_release in
67 // init(), taken together, mean that threads can safely synchronize on
68 // _has_called.
69 return _has_called.load(std::memory_order_acquire);
70 }
71
72 // Return the stored result. The result is only meaningful when `has_called() == true`.
73 ReturnType stored_result() const { return _status; }
74
75private:
76 std::atomic<bool> _has_called;
77 std::once_flag _once_flag;
78 ReturnType _status;
79};
80
81} // namespace doris
