40 class Allocator {
41 public:
42 using value_type = T;
43
44 T* allocate(size_t n) { return static_cast<T*>(::operator new(sizeof(T) * n)); }
45
46 template <typename... Args>
47 void construct(T* p, Args&&... args) {
48 new (p) T(std::forward<Args>(args)...);
49 }
50
51 void destroy(T* p) { p->~T(); }
52

56 class TrieNode {
57 public:
58 TrieNode(const std::string& key, const std::string& wildcard)
59 : _value(nullptr), _wildcard(wildcard) {
60 if (is_named_wildcard(key)) {
61 _named_wildcard = extract_template(key);
62 }
63 }
64
65 TrieNode(const std::string& key, const T& value, const std::string& wildcard)
66 : _value(nullptr), _wildcard(wildcard) {
67 _value = _allocator.allocate(1);
68 _allocator.construct(_value, value);
69 if (is_named_wildcard(key)) {
70 _named_wildcard = extract_template(key);
71 }
72 }
73
74 ~TrieNode() {
75 for (auto& iter : _children) {
76 delete iter.second;
77 iter.second = nullptr;
78 }
79 if (_value != nullptr) {
80 _allocator.destroy(_value);
81 _allocator.deallocate(_value, 1);
82 }
83 }

85 // Return true if insert success.
86 bool insert(const std::vector<std::string> path, int index, const T& value) {
87 if (index >= path.size()) {
88 return false;
89 }
90 const std::string& token = path[index];
91 std::string key = token;
92
93 if (is_named_wildcard(token)) {
94 key = _wildcard;
95 }
96
97 TrieNode* node = get_child(key);
98
99 if (node == nullptr) {
100 // no exist child for this key
101 if (index == path.size() - 1) {
102 node = new TrieNode(token, value, _wildcard);
103 _children.insert(std::make_pair(key, node));
104 return true;
105 } else {
106 node = new TrieNode(token, _wildcard);
107 _children.insert(std::make_pair(key, node));
108 }
109 } else {
110 // If this is a template, set this to the node
111 if (is_named_wildcard(token)) {
112 std::string temp = extract_template(token);
113 if (node->_named_wildcard.empty() || node->_named_wildcard.compare(temp) == 0) {
114 node->_named_wildcard = temp;
115 } else {
116 // Duplicated
117 return false;
118 }
119 }
120 if (index == path.size() - 1) {
121 if (node->_value == nullptr) {
122 node->_value = _allocator.allocate(1);
123 _allocator.construct(node->_value, value);
124 return true;
125 }
126 // Already register by other path
127 return false;
128 }
129 }
130 return node->insert(path, index + 1, value);
131 }
132
133 bool retrieve(const std::vector<std::string> path, int index, T* value,
134 std::map<std::string, std::string>* params) {
135 // check max index
136 if (index >= path.size()) {
137 return false;
138 }
139 bool use_wildcard = false;
140 const std::string& token = path[index];
141 TrieNode* node = get_child(token);
142 if (node == nullptr) {
143 node = get_child(_wildcard);
144 if (node == nullptr) {
145 return false;
146 }
147 use_wildcard = true;
148 } else {
149 // If we the last one, but we have no value, check wildcard
150 if (index == path.size() - 1 && node->_value == nullptr &&
151 get_child(_wildcard) != nullptr) {
152 node = get_child(_wildcard);
153 use_wildcard = true;
154 } else {
155 use_wildcard = (token.compare(_wildcard) == 0);
156 }
157 }
158
159 put(params, node, token);
160
161 if (index == path.size() - 1) {
162 if (node->_value == nullptr) {
163 return false;
164 }
165 _allocator.construct(value, *node->_value);
166 return true;
167 }
168
169 // find exact
170 if (node->retrieve(path, index + 1, value, params)) {
171 return true;
172 }
173
174 // backtrace to test if wildcard can match
175 if (!use_wildcard) {
176 node = get_child(_wildcard);
177 if (node != nullptr) {
178 put(params, node, token);
179 return node->retrieve(path, index + 1, value, params);
180 }
181 }
182 return false;
183 }
184
185 private:
186 bool is_named_wildcard(const std::string& key) {
187 if (key.find('{') != std::string::npos && key.find('}') != std::string::npos) {
188 return true;
189 }
190 return false;
191 }
192
193 std::string extract_template(const std::string& key) {
194 std::size_t left = key.find_first_of('{') + 1;
195 std::size_t right = key.find_last_of('}');
196 return key.substr(left, right - left);
197 }
198
199 TrieNode* get_child(const std::string& key) {
200 auto pair = _children.find(key);
201 if (pair == _children.end()) {
202 return nullptr;
203 }
204 return pair->second;
205 }
206
207 void put(std::map<std::string, std::string>* params, TrieNode* node,
208 const std::string& token) {
209 if (params != nullptr && !node->_named_wildcard.empty()) {
210 params->insert(std::make_pair(node->_named_wildcard, token));
211 }
212 }
213
214 T* _value;
215 std::string _wildcard;
216 std::string _named_wildcard;
217 std::map<std::string, TrieNode*> _children;
218 Allocator _allocator;
219 };

221 bool insert(const std::string& path, const T& value) {
222 std::vector<std::string> path_array;
223 split(path, &path_array);
224 if (path_array.empty()) {
225 if (_root_value == nullptr) {
226 _root_value = _allocator.allocate(1);
227 _allocator.construct(_root_value, value);
228 return true;
229 } else {
230 return false;
231 }
232 }
233 int index = 0;
234 if (path_array[0].empty()) {
235 index = 1;
236 }
237 return _root.insert(path_array, index, value);
238 }
239
240 bool retrieve(const std::string& path, T* value) { return retrieve(path, value, nullptr); }
241
242 bool retrieve(const std::string& path, T* value, std::map<std::string, std::string>* params) {
243 if (path.empty()) {
244 if (_root_value == nullptr) {
245 return false;
246 } else {
247 _allocator.construct(value, *_root_value);
248 return true;
249 }
250 }
251 std::vector<std::string> path_array;
252 split(path, &path_array);
253 if (path_array.empty()) {
254 if (_root_value == nullptr) {
255 return false;
256 } else {
257 _allocator.construct(value, *_root_value);
258 return true;
259 }
260 }
261 int index = 0;
262 if (path_array[0].empty()) {
263 index = 1;
264 }
265 return _root.retrieve(path_array, index, value, params);
266 }
267
268private:
269 void split(const std::string& path, std::vector<std::string>* array) {
270 const char* path_str = path.c_str();
271 std::size_t start = 0;
272 std::size_t pos = 0;
273 for (; pos < path.length(); ++pos) {
274 if (path_str[pos] == _separator) {
275 if (pos - start > 0) {
276 array->push_back(path.substr(start, pos - start));
277 }
278 start = pos + 1;
279 }
280 }
281 if (pos - start > 0) {
282 array->push_back(path.substr(start, pos - start));
283 }
284 }

287 T* _root_value;
288 char _separator;
289 Allocator _allocator;
290};
