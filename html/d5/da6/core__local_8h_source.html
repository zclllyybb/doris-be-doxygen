53template <typename T>
54class CoreLocalValueController {
55public:
56 CoreLocalValueController() {
57 int num_cpus = static_cast<int>(std::thread::hardware_concurrency());
58 _size = 8;
59 while (_size < num_cpus) {
60 _size <<= 1;
61 }
62 _allocators.resize(_size, nullptr);
63 for (int i = 0; i < _size; ++i) {
64 _allocators[i] = CoreDataAllocatorFactory::instance()->get_allocator(i, sizeof(T));
65 }
66 }
67
68 ~CoreLocalValueController() {}
69
70 int get_id() {
71 std::lock_guard<std::mutex> l(_lock);
72 int id = 0;
73 if (_free_ids.empty()) {
74 id = _next_id++;
75 } else {
76 id = _free_ids.back();
77 _free_ids.pop_back();
78 }
79 return id;
80 }
81 void reclaim_id(int id) {
82 std::lock_guard<std::mutex> l(_lock);
83 _free_ids.push_back(id);
84 }
85 size_t size() const { return _size; }
86 CoreDataAllocator* allocator(int i) const { return _allocators[i]; }
87
88 static CoreLocalValueController<T>* instance() {
89 static CoreLocalValueController<T> _s_instance;
90 return &_s_instance;
91 }
92
93private:
94 DISALLOW_COPY_AND_ASSIGN(CoreLocalValueController);
95
96private:
97 std::mutex _lock;
98 int _next_id = 0;
99 std::deque<int> _free_ids;
100 std::vector<CoreDataAllocator*> _allocators;
101 size_t _size;
102};

104template <typename T>
105class CoreLocalValue {
106public:
107 CoreLocalValue(const T init_value = T()) {
108 CoreLocalValueController<T>* controller = CoreLocalValueController<T>::instance();
109 _id = controller->get_id();
110 _size = controller->size();
111 _values.resize(_size, nullptr);
112 for (int i = 0; i < _size; ++i) {
113 void* ptr = controller->allocator(i)->get_or_create(_id);
114 _values[i] = new (ptr) T(init_value);
115 }
116 }
117
118 ~CoreLocalValue() {
119 for (int i = 0; i < _size; ++i) {
120 _values[i]->~T();
121 }
122 CoreLocalValueController<T>::instance()->reclaim_id(_id);
123 }
124
125 size_t size() const { return _size; }
126 T* access() const {
127#ifdef __APPLE__
128 size_t cpu_id = 0;
129#else
130 size_t cpu_id = sched_getcpu();
131#endif
132 if (cpu_id >= _size) {
133 cpu_id &= _size - 1;
134 }
135 return access_at_core(cpu_id);
136 }
137 T* access_at_core(size_t core_idx) const { return _values[core_idx]; }
138
139 inline void reset() {
140 for (int i = 0; i < _size; ++i) {
141 _values[i]->~T();
142 }
143 _values.clear();
144 _values.resize(_size, nullptr);
145 CoreLocalValueController<T>* controller = CoreLocalValueController<T>::instance();
146 for (int i = 0; i < _size; ++i) {
147 void* ptr = controller->allocator(i)->get_or_create(_id);
148 _values[i] = new (ptr) T();
149 }
150 }
151
152private:
153 int _id = -1;
154 size_t _size = 0;
155 std::vector<T*> _values;
156};
