117template <typename Key, typename Mapped, typename Cell = FixedHashMapCell<Key, Mapped>,
118 typename Size = FixedHashTableStoredSize<Cell>, typename Allocator = HashTableAllocator>
119class FixedHashMap : public FixedHashTable<Key, Cell, Size, Allocator> {
120public:
121 using Base = FixedHashTable<Key, Cell, Size, Allocator>;
122 using Self = FixedHashMap;
123 using LookupResult = typename Base::LookupResult;
124
125 using Base::Base;
126
127 template <typename Func>
128 void ALWAYS_INLINE merge_to_via_emplace(Self& that, Func&& func) {
129 for (auto it = this->begin(), end = this->end(); it != end; ++it) {
130 typename Self::LookupResult res_it;
131 bool inserted;
132 that.emplace(it->get_key(), res_it, inserted, it.get_hash());
133 func(res_it->get_mapped(), it->get_mapped(), inserted);
134 }
135 }
136
137 template <typename Func>
138 void ALWAYS_INLINE merge_to_via_find(Self& that, Func&& func) {
139 for (auto it = this->begin(), end = this->end(); it != end; ++it) {
140 auto res_it = that.find(it->get_key(), it.get_hash());
141 if (!res_it)
142 func(it->get_mapped(), it->get_mapped(), false);
143 else
144 func(res_it->get_mapped(), it->get_mapped(), true);
145 }
146 }
147
148 template <typename Func>
149 void for_each_value(Func&& func) {
150 for (auto& v : *this) func(v.get_key(), v.get_mapped());
151 }
152
153 template <typename Func>
154 void for_each_mapped(Func&& func) {
155 for (auto& v : *this) func(v.get_second());
156 }
157
158 Mapped& ALWAYS_INLINE operator[](const Key& x) {
159 LookupResult it;
160 bool inserted;
161 this->emplace(x, it, inserted);
162 if (inserted) new (&it->get_mapped()) Mapped();
163
164 return it->get_mapped();
165 }
166
167 // fixed hash map never overflow
168 bool add_elem_size_overflow(size_t add_size) const { return false; }
169
170 char* get_null_key_data() { return nullptr; }
171 bool has_null_key_data() const { return false; }
172};
