48 uint64_t get_date_at(uint16_t idx) {
49 const T val = data[idx];
50 const char* val_ptr = reinterpret_cast<const char*>(&val);
51 uint64_t value = 0;
52 value = *(unsigned char*)(val_ptr + 2);
53 value <<= 8;
54 value |= *(unsigned char*)(val_ptr + 1);
55 value <<= 8;
56 value |= *(unsigned char*)(val_ptr);
57 return value;
58 }

92 void insert_string_to_res_column(const uint16_t* sel, size_t sel_size,
93 vectorized::ColumnString* res_ptr) {
94 StringRef refs[sel_size];
95 size_t length = 0;
96 for (size_t i = 0; i < sel_size; i++) {
97 uint16_t n = sel[i];
98 auto& sv = reinterpret_cast<StringRef&>(data[n]);
99 refs[i].data = sv.data;
100 refs[i].size = sv.size;
101 length += sv.size;
102 }
103 res_ptr->get_offsets().reserve(sel_size + res_ptr->get_offsets().size());
104 res_ptr->get_chars().reserve(length + res_ptr->get_chars().size());
105 res_ptr->insert_many_strings_without_reserve(refs, sel_size);
106 }

225 void insert_many_date(const char* data_ptr, size_t num) {
226 size_t intput_type_size = sizeof(uint24_t);
227 size_t res_type_size = sizeof(uint32_t);
228 char* input_data_ptr = const_cast<char*>(data_ptr);
229
230 char* res_ptr = (char*)data.get_end_ptr();
231 memset(res_ptr, 0, res_type_size * num);
232 for (int i = 0; i < num; i++) {
233 memcpy(res_ptr, input_data_ptr, intput_type_size);
234 res_ptr += res_type_size;
235 input_data_ptr += intput_type_size;
236 }
237 data.set_end_ptr(res_ptr);
238 }

264 void insert_many_continuous_binary_data(const char* data_, const uint32_t* offsets,
265 const size_t num) override {
266 if (UNLIKELY(num == 0)) {
267 return;
268 }
269 if constexpr (std::is_same_v<T, StringRef>) {
270 if (_pool == nullptr) {
271 _pool.reset(new MemPool());
272 }
273 const auto total_mem_size = offsets[num] - offsets[0];
274 char* destination = (char*)_pool->allocate(total_mem_size);
275 memcpy(destination, data_ + offsets[0], total_mem_size);
276 size_t org_elem_num = data.size();
277 data.resize(org_elem_num + num);
278
279 auto* data_ptr = &data[org_elem_num];
280 for (size_t i = 0; i != num; ++i) {
281 data_ptr[i].data = destination + offsets[i] - offsets[0];
282 data_ptr[i].size = offsets[i + 1] - offsets[i];
283 }
284 DCHECK(data_ptr[num - 1].data + data_ptr[num - 1].size == destination + total_mem_size);
285 }
286 }

288 void insert_many_binary_data(char* data_array, uint32_t* len_array,
289 uint32_t* start_offset_array, size_t num) override {
290 if (num == 0) {
291 return;
292 }
293 if constexpr (std::is_same_v<T, StringRef>) {
294 if (_pool == nullptr) {
295 _pool.reset(new MemPool());
296 }
297
298 size_t total_mem_size = 0;
299 for (size_t i = 0; i < num; i++) {
300 total_mem_size += len_array[i];
301 }
302
303 char* destination = (char*)_pool->allocate(total_mem_size);
304 char* org_dst = destination;
305 size_t org_elem_num = data.size();
306 data.resize(org_elem_num + num);
307 uint32_t fragment_start_offset = start_offset_array[0];
308 size_t fragment_len = 0;
309 for (size_t i = 0; i < num; i++) {
310 data[org_elem_num + i].data = destination + fragment_len;
311 data[org_elem_num + i].size = len_array[i];
312 fragment_len += len_array[i];
313 // Compute the largest continuous memcpy block and copy them.
314 // If this is the last element in data array, then should copy the current memory block.
315 if (i == num - 1 ||
316 start_offset_array[i + 1] != start_offset_array[i] + len_array[i]) {
317 memcpy(destination, data_array + fragment_start_offset, fragment_len);
318 destination += fragment_len;
319 fragment_start_offset = (i == num - 1 ? 0 : start_offset_array[i + 1]);
320 fragment_len = 0;
321 }
322 }
323 CHECK(destination - org_dst == total_mem_size)
324 << "Copied size not equal to expected size";
325 } else {
326 LOG(FATAL) << "Method insert_many_binary_data is not supported";
327 }
328 }

449 Status filter_by_selector(const uint16_t* sel, size_t sel_size, IColumn* col_ptr) override {
450 if constexpr (std::is_same_v<T, StringRef>) {
451 insert_string_to_res_column(sel, sel_size,
452 reinterpret_cast<vectorized::ColumnString*>(col_ptr));
453 } else if constexpr (std::is_same_v<T, decimal12_t>) {
454 insert_decimal_to_res_column(
455 sel, sel_size,
456 reinterpret_cast<vectorized::ColumnDecimal<Decimal128>*>(col_ptr));
457 } else if constexpr (std::is_same_v<T, doris::vectorized::Int8>) {
458 insert_default_value_res_column(
459 sel, sel_size,
460 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::Int8>*>(col_ptr));
461 } else if constexpr (std::is_same_v<T, doris::vectorized::Int16>) {
462 insert_default_value_res_column(
463 sel, sel_size,
464 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::Int16>*>(col_ptr));
465 } else if constexpr (std::is_same_v<T, doris::vectorized::Int32>) {
466 insert_default_value_res_column(
467 sel, sel_size,
468 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::Int32>*>(col_ptr));
469 } else if constexpr (std::is_same_v<T, doris::vectorized::Int64>) {
470 insert_default_value_res_column(
471 sel, sel_size,
472 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::Int64>*>(col_ptr));
473 } else if constexpr (std::is_same_v<T, doris::vectorized::Float32>) {
474 insert_default_value_res_column(
475 sel, sel_size,
476 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::Float32>*>(
477 col_ptr));
478 } else if constexpr (std::is_same_v<T, doris::vectorized::Float64>) {
479 insert_default_value_res_column(
480 sel, sel_size,
481 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::Float64>*>(
482 col_ptr));
483 } else if constexpr (std::is_same_v<T, uint64_t>) {
484 if (const vectorized::ColumnVector<UInt64>* date_col =
485 check_and_get_column<vectorized::ColumnVector<UInt64>>(
486 const_cast<const IColumn*>(col_ptr))) {
487 insert_default_value_res_column(
488 sel, sel_size, const_cast<vectorized::ColumnVector<UInt64>*>(date_col));
489 } else {
490 insert_datetime_to_res_column(
491 sel, sel_size, reinterpret_cast<vectorized::ColumnVector<Int64>*>(col_ptr));
492 }
493 } else if constexpr (std::is_same_v<T, uint24_t>) {
494 insert_date_to_res_column(sel, sel_size,
495 reinterpret_cast<vectorized::ColumnVector<Int64>*>(col_ptr));
496 } else if constexpr (std::is_same_v<T, uint32_t>) {
497 if (const vectorized::ColumnVector<Int64>* date_col =
498 check_and_get_column<vectorized::ColumnVector<Int64>>(
499 const_cast<const IColumn*>(col_ptr))) {
500 // a trick type judge, need refactor it.
501 insert_date32_to_res_column(sel, sel_size,
502 const_cast<vectorized::ColumnVector<Int64>*>(date_col));
503 } else {
504 insert_default_value_res_column(
505 sel, sel_size,
506 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::UInt32>*>(
507 col_ptr));
508 }
509 } else if constexpr (std::is_same_v<T, doris::vectorized::Int128>) {
510 insert_default_value_res_column(
511 sel, sel_size,
512 reinterpret_cast<vectorized::ColumnVector<doris::vectorized::Int128>*>(
513 col_ptr));
514 } else if (std::is_same_v<T, bool>) {
515 insert_byte_to_res_column(sel, sel_size, col_ptr);
516 } else {
517 return Status::NotSupported("not supported output type in predicate_column");
518 }
519 return Status::OK();
520 }
