80class TaggableLogger {
81public:
82 template <typename... Args>
83 TaggableLogger(std::ostream& stream, std::string_view fmt, Args&&... args) : _stream(stream) {
84 if constexpr (sizeof...(args) == 0) {
85 _stream << fmt;
86 } else {
87 _stream << fmt::format(fmt, std::forward<Args>(args)...);
88 }
89 }
90
91 template <typename V>
92 TaggableLogger& tag(std::string_view key, const V& value) {
93 _stream << '|' << key << '=';
94 if constexpr (std::is_same_v<V, TUniqueId> || std::is_same_v<V, PUniqueId>) {
95 _stream << print_id(value);
96 } else {
97 _stream << value;
98 }
99 return *this;
100 }
101
102 template <typename E>
103 TaggableLogger& error(const E& error) {
104 _stream << "|error=" << error;
105 return *this;
106 }
107
108private:
109 std::ostream& _stream;
110};
