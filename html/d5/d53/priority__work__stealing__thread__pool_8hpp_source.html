28// Work-Stealing threadpool which processes items (of type T) in parallel which were placed on multi
29// blocking queues by Offer(). Each item is processed by a single user-supplied method.
30class PriorityWorkStealingThreadPool : public PriorityThreadPool {
31public:
32 // Creates a new thread pool and start num_threads threads.
33 // -- num_threads: how many threads are part of this pool
34 // -- num_queues: how many queues are part of this pool
35 // -- queue_size: the maximum size of the queue on which work items are offered. If the
36 // queue exceeds this size, subsequent calls to Offer will block until there is
37 // capacity available.
38 PriorityWorkStealingThreadPool(uint32_t num_threads, uint32_t num_queues, uint32_t queue_size,
39 const std::string& name)
40 : PriorityThreadPool(0, 0, name) {
41 DCHECK_GT(num_queues, 0);
42 DCHECK_GE(num_threads, num_queues);
43 // init _work_queues first because the work thread needs it
44 for (int i = 0; i < num_queues; ++i) {
45 _work_queues.emplace_back(std::make_shared<BlockingPriorityQueue<Task>>(queue_size));
46 }
47 for (int i = 0; i < num_threads; ++i) {
48 _threads.create_thread(std::bind<void>(
49 std::mem_fn(&PriorityWorkStealingThreadPool::work_thread), this, i));
50 }
51 }
52
53 virtual ~PriorityWorkStealingThreadPool() {
54 shutdown();
55 join();
56 }
57
58 // Blocking operation that puts a work item on the queue. If the queue is full, blocks
59 // until there is capacity available.
60 //
61 // 'work' is copied into the work queue, but may be referenced at any time in the
62 // future. Therefore the caller needs to ensure that any data referenced by work (if T
63 // is, e.g., a pointer type) remains valid until work has been processed, and it's up to
64 // the caller to provide their own signalling mechanism to detect this (or to wait until
65 // after DrainAndshutdown returns).
66 //
67 // Returns true if the work item was successfully added to the queue, false otherwise
68 // (which typically means that the thread pool has already been shut down).
69 bool offer(Task task) override { return _work_queues[task.queue_id]->blocking_put(task); }
70
71 bool offer(WorkFunction func) override {
72 PriorityThreadPool::Task task = {0, func, 0};
73 return _work_queues[task.queue_id]->blocking_put(task);
74 }
75
76 // Shuts the thread pool down, causing the work queue to cease accepting offered work
77 // and the worker threads to terminate once they have processed their current work item.
78 // Returns once the shutdown flag has been set, does not wait for the threads to
79 // terminate.
80 void shutdown() override {
81 PriorityThreadPool::shutdown();
82 for (auto work_queue : _work_queues) {
83 work_queue->shutdown();
84 }
85 }
86
87 uint32_t get_queue_size() const override {
88 uint32_t size = 0;
89 for (auto work_queue : _work_queues) {
90 size += work_queue->get_size();
91 }
92 return size;
93 }
94
95 // Blocks until the work queue is empty, and then calls shutdown to stop the worker
96 // threads and Join to wait until they are finished.
97 // Any work Offer()'ed during DrainAndshutdown may or may not be processed.
98 void drain_and_shutdown() override {
99 {
100 std::unique_lock<std::mutex> l(_lock);
101 while (get_queue_size() != 0) {
102 _empty_cv.wait(l);
103 }
104 }
105 shutdown();
106 join();
107 }
108
109private:
110 // Driver method for each thread in the pool. Continues to read work from the queue
111 // until the pool is shutdown.
112 void work_thread(int thread_id) {
113 auto queue_id = thread_id % _work_queues.size();
114 auto steal_queue_id = (queue_id + 1) % _work_queues.size();
115 while (!is_shutdown()) {
116 Task task;
117 // avoid blocking get
118 bool is_other_queues_empty = true;
119 // steal work in round-robin if nothing to do
120 while (_work_queues[queue_id]->get_size() == 0 && queue_id != steal_queue_id &&
121 !is_shutdown()) {
122 if (_work_queues[steal_queue_id]->non_blocking_get(&task)) {
123 is_other_queues_empty = false;
124 task.work_function();
125 }
126 steal_queue_id = (steal_queue_id + 1) % _work_queues.size();
127 }
128 if (queue_id == steal_queue_id) {
129 steal_queue_id = (steal_queue_id + 1) % _work_queues.size();
130 }
131 if (is_other_queues_empty &&
132 _work_queues[queue_id]->blocking_get(
133 &task, config::doris_blocking_priority_queue_wait_timeout_ms)) {
134 task.work_function();
135 }
136 if (_work_queues[queue_id]->get_size() == 0) {
137 _empty_cv.notify_all();
138 }
139 }
140 }
141
142 // Queue on which work items are held until a thread is available to process them in
143 // FIFO order.
144 std::vector<std::shared_ptr<BlockingPriorityQueue<Task>>> _work_queues;
145};
