32struct RetentionState {
33 static constexpr size_t MAX_EVENTS = 32;
34 uint8_t events[MAX_EVENTS] = {0};
35
36 RetentionState() {}
37
38 void reset() {
39 for (int64_t i = 0; i < MAX_EVENTS; i++) {
40 events[i] = 0;
41 }
42 }
43
44 void set(int event) { events[event] = 1; }
45
46 void merge(const RetentionState& other) {
47 for (int64_t i = 0; i < MAX_EVENTS; i++) {
48 events[i] |= other.events[i];
49 }
50 }
51
52 void write(BufferWritable& out) const {
53 int64_t serialized_events = 0;
54 for (int64_t i = 0; i < MAX_EVENTS; i++) {
55 serialized_events |= events[i];
56 serialized_events <<= 1;
57 }
58 write_var_int(serialized_events, out);
59 }
60
61 void read(BufferReadable& in) {
62 int64_t serialized_events = 0;
63 uint64_t u_serialized_events = 0;
64 read_var_int(serialized_events, in);
65 u_serialized_events = serialized_events;
66
67 u_serialized_events >>= 1;
68 for (int64_t i = MAX_EVENTS - 1; i >= 0; i--) {
69 events[i] = (uint8)(1 & u_serialized_events);
70 u_serialized_events >>= 1;
71 }
72 }
73
74 void insert_result_into(IColumn& to, size_t events_size, const uint8_t* events) const {
75 auto& data_to = static_cast<ColumnUInt8&>(to).get_data();
76
77 ColumnArray::Offset64 current_offset = data_to.size();
78 data_to.resize(current_offset + events_size);
79
80 bool first_flag = events[0];
81 data_to[current_offset] = first_flag;
82 ++current_offset;
83
84 for (size_t i = 1; i < events_size; ++i) {
85 data_to[current_offset] = (first_flag && events[i]);
86 ++current_offset;
87 }
88 }
89};

91class AggregateFunctionRetention
92 : public IAggregateFunctionDataHelper<RetentionState, AggregateFunctionRetention> {
93public:
94 AggregateFunctionRetention(const DataTypes& argument_types_)
95 : IAggregateFunctionDataHelper<RetentionState, AggregateFunctionRetention>(
96 argument_types_) {}
97
98 String get_name() const override { return "retention"; }
99
100 DataTypePtr get_return_type() const override {
101 return std::make_shared<DataTypeArray>(make_nullable(std::make_shared<DataTypeUInt8>()));
102 }
103
104 void reset(AggregateDataPtr __restrict place) const override { this->data(place).reset(); }
105 void add(AggregateDataPtr __restrict place, const IColumn** columns, const size_t row_num,
106 Arena*) const override {
107 for (int i = 0; i < get_argument_types().size(); i++) {
108 auto event = assert_cast<const ColumnVector<UInt8>*>(columns[i])->get_data()[row_num];
109 if (event) {
110 this->data(place).set(i);
111 }
112 }
113 }
114
115 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
116 Arena*) const override {
117 this->data(place).merge(this->data(rhs));
118 }
119
120 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
121 this->data(place).write(buf);
122 }
123
124 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
125 Arena*) const override {
126 this->data(place).read(buf);
127 }
128
129 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
130 auto& to_arr = assert_cast<ColumnArray&>(to);
131 auto& to_nested_col = to_arr.get_data();
132 if (to_nested_col.is_nullable()) {
133 auto col_null = reinterpret_cast<ColumnNullable*>(&to_nested_col);
134 this->data(place).insert_result_into(col_null->get_nested_column(),
135 get_argument_types().size(),
136 this->data(place).events);
137 col_null->get_null_map_data().resize_fill(col_null->get_nested_column().size(), 0);
138 } else {
139 this->data(place).insert_result_into(to_nested_col, get_argument_types().size(),
140 this->data(place).events);
141 }
142 to_arr.get_offsets().push_back(to_nested_col.size());
143 }
144};
