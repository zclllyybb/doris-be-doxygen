34public:

36 static constexpr bool has_variadic_argument =

44 size_t get_number_of_arguments() const override { return 0; }
45 DataTypes get_variadic_argument_types_impl() const override {
46 if constexpr (has_variadic_argument) return Transform::get_variadic_argument_types();
47 return {};
48 }

51 return make_nullable(std::make_shared<DataTypeString>());
52 }
53
54 bool use_default_implementation_for_nulls() const override { return false; }

58 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
59 size_t result, size_t input_rows_count) override {
60 const ColumnPtr source_col = block.get_by_position(arguments[0]).column;
61
62 const auto* nullable_column = check_and_get_column<ColumnNullable>(source_col.get());
63 const auto* sources = check_and_get_column<ColumnVector<typename Transform::FromType>>(
64 nullable_column ? nullable_column->get_nested_column_ptr().get()
65 : source_col.get());
66
67 if (sources) {
68 auto col_res = ColumnString::create();
69 ColumnUInt8::MutablePtr col_null_map_to;
70 col_null_map_to = ColumnUInt8::create();
71 auto& vec_null_map_to = col_null_map_to->get_data();
72
73 if (arguments.size() == 2) {
74 const IColumn& source_col1 = *block.get_by_position(arguments[1]).column;
75 if (const auto* delta_const_column =
76 typeid_cast<const ColumnConst*>(&source_col1)) {
77 TransformerToStringTwoArgument<Transform>::vector_constant(
78 context, sources->get_data(),
79 delta_const_column->get_field().get<String>(), col_res->get_chars(),
80 col_res->get_offsets(), vec_null_map_to);
81 } else {
82 return Status::InternalError(
83 "Illegal column {} is not const {}",
84 block.get_by_position(arguments[1]).column->get_name(), name);
85 }
86 } else {
87 TransformerToStringTwoArgument<Transform>::vector_constant(
88 context, sources->get_data(), "%Y-%m-%d %H:%i:%s", col_res->get_chars(),
89 col_res->get_offsets(), vec_null_map_to);
90 }
91
92 if (nullable_column) {
93 const auto& origin_null_map = nullable_column->get_null_map_column().get_data();
94 for (int i = 0; i < origin_null_map.size(); ++i) {
95 vec_null_map_to[i] |= origin_null_map[i];
96 }
97 }
98 block.get_by_position(result).column =
99 ColumnNullable::create(std::move(col_res), std::move(col_null_map_to));
100 } else {
101 return Status::InternalError("Illegal column {} of first argument of function {}",
102 block.get_by_position(arguments[0]).column->get_name(),
103 name);
104 }
105 return Status::OK();
106 }
107};
