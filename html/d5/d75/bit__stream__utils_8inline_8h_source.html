35inline void BitWriter::PutValue(uint64_t v, int num_bits) {
36 DCHECK_LE(num_bits, 64);
37 // Truncate the higher-order bits. This is necessary to
38 // support signed values.
39 v &= ~0ULL >> (64 - num_bits);
40
41 buffered_values_ |= v << bit_offset_;
42 bit_offset_ += num_bits;
43
44 if (PREDICT_FALSE(bit_offset_ >= 64)) {
45 // Flush buffered_values_ and write out bits of v that did not fit
46 buffer_->reserve(ALIGN_UP(byte_offset_ + 8, 8));
47 buffer_->resize(byte_offset_ + 8);
48 DCHECK_LE(byte_offset_ + 8, buffer_->capacity());
49 memcpy(buffer_->data() + byte_offset_, &buffered_values_, 8);
50 buffered_values_ = 0;
51 byte_offset_ += 8;
52 bit_offset_ -= 64;
53 buffered_values_ = BitUtil::ShiftRightZeroOnOverflow(v, (num_bits - bit_offset_));
54 }
55 DCHECK_LT(bit_offset_, 64);
56}

58inline void BitWriter::Flush(bool align) {
59 int num_bytes = BitUtil::Ceil(bit_offset_, 8);
60 buffer_->reserve(ALIGN_UP(byte_offset_ + num_bytes, 8));
61 buffer_->resize(byte_offset_ + num_bytes);
62 DCHECK_LE(byte_offset_ + num_bytes, buffer_->capacity());
63 memcpy(buffer_->data() + byte_offset_, &buffered_values_, num_bytes);
64
65 if (align) {
66 buffered_values_ = 0;
67 byte_offset_ += num_bytes;
68 bit_offset_ = 0;
69 }
70}

72inline uint8_t* BitWriter::GetNextBytePtr(int num_bytes) {
73 Flush(/* align */ true);
74 buffer_->reserve(ALIGN_UP(byte_offset_ + num_bytes, 8));
75 buffer_->resize(byte_offset_ + num_bytes);
76 uint8_t* ptr = buffer_->data() + byte_offset_;
77 byte_offset_ += num_bytes;
78 DCHECK_LE(byte_offset_, buffer_->capacity());
79 return ptr;

89inline void BitWriter::PutVlqInt(int32_t v) {
90 while ((v & 0xFFFFFF80) != 0L) {
91 PutAligned<uint8_t>((v & 0x7F) | 0x80, 1);
92 v >>= 7;
93 }
94 PutAligned<uint8_t>(v & 0x7F, 1);
95}

107inline void BitReader::BufferValues() {
108 int bytes_remaining = max_bytes_ - byte_offset_;
109 if (PREDICT_TRUE(bytes_remaining >= 8)) {
110 memcpy(&buffered_values_, buffer_ + byte_offset_, 8);
111 } else {

116template <typename T>
117bool BitReader::GetValue(int num_bits, T* v) {
118 DCHECK_LE(num_bits, 64);
119 DCHECK_LE(num_bits, sizeof(T) * 8);
120
121 if (PREDICT_FALSE(byte_offset_ * 8 + bit_offset_ + num_bits > max_bytes_ * 8)) return false;
122
123 *v = BitUtil::TrailingBits(buffered_values_, bit_offset_ + num_bits) >> bit_offset_;
124
125 bit_offset_ += num_bits;
126 if (bit_offset_ >= 64) {
127 byte_offset_ += 8;
128 bit_offset_ -= 64;
129 BufferValues();
130 // Read bits of v that crossed into new buffered_values_
131 *v |= BitUtil::ShiftLeftZeroOnOverflow(BitUtil::TrailingBits(buffered_values_, bit_offset_),
132 (num_bits - bit_offset_));
133 }
134 DCHECK_LE(bit_offset_, 64);
135 return true;
136}

138inline void BitReader::Rewind(int num_bits) {
139 bit_offset_ -= num_bits;
140 if (bit_offset_ >= 0) {
141 return;
142 }
143 while (bit_offset_ < 0) {
144 int seek_back = std::min(byte_offset_, 8);
145 byte_offset_ -= seek_back;
146 bit_offset_ += seek_back * 8;
147 }
148 // This should only be executed *if* rewinding by 'num_bits'
149 // make the existing buffered_values_ invalid
150 DCHECK_GE(byte_offset_, 0); // Check for underflow
151 memcpy(&buffered_values_, buffer_ + byte_offset_, 8);
152}

154inline bool BitReader::Advance(int64_t num_bits) {
155 int64_t bits_required = bit_offset_ + num_bits;
156 int64_t bytes_required = (bits_required >> 3) + ((bits_required & 7) != 0);
157 if (bytes_required > max_bytes_ - byte_offset_) {
158 return false;
159 }
160 byte_offset_ += static_cast<int>(bits_required >> 3);
161 bit_offset_ = static_cast<int>(bits_required & 7);
162 BufferValues();
163 return true;
164}

166inline void BitReader::SeekToBit(unsigned int stream_position) {
167 DCHECK_LE(stream_position, max_bytes_ * 8);
168
169 int delta = static_cast<int>(stream_position) - position();
170 if (delta == 0) {
171 return;
172 } else if (delta < 0) {
173 Rewind(position() - stream_position);
174 } else {
175 bit_offset_ += delta;
176 while (bit_offset_ >= 64) {
177 byte_offset_ += 8;
178 bit_offset_ -= 64;
179 if (bit_offset_ < 64) {
180 // This should only be executed if seeking to
181 // 'stream_position' makes the existing buffered_values_
182 // invalid.
183 BufferValues();
184 }
185 }
186 }

189template <typename T>
190bool BitReader::GetAligned(int num_bytes, T* v) {
191 DCHECK_LE(num_bytes, sizeof(T));
192 int bytes_read = BitUtil::Ceil(bit_offset_, 8);
193 if (PREDICT_FALSE(byte_offset_ + bytes_read + num_bytes > max_bytes_)) return false;
194
195 // Advance byte_offset to next unread byte and read num_bytes
196 byte_offset_ += bytes_read;
197 memcpy(v, buffer_ + byte_offset_, num_bytes);
198 byte_offset_ += num_bytes;
199
200 // Reset buffered_values_
201 bit_offset_ = 0;
202 int bytes_remaining = max_bytes_ - byte_offset_;
203 if (PREDICT_TRUE(bytes_remaining >= 8)) {
204 memcpy(&buffered_values_, buffer_ + byte_offset_, 8);
205 } else {
206 memcpy(&buffered_values_, buffer_ + byte_offset_, bytes_remaining);
207 }
208 return true;
209}

211inline bool BitReader::GetVlqInt(uint32_t* v) {
212 uint32_t tmp = 0;
213 for (int num_bytes = 0; num_bytes < MAX_VLQ_BYTE_LEN; num_bytes++) {
214 uint8_t byte = 0;
215 if (!GetAligned<uint8_t>(1, &byte)) return false;
216 tmp |= static_cast<uint32_t>(byte & 0x7F) << (7 * num_bytes);
217 if ((byte & 0x80) == 0) {
218 *v = tmp;
219 return true;
220 }
221 }
222 return false;
223}

260template <typename T>
261int BatchedBitReader::UnpackBatch(int bit_width, int num_values, T* v) {
262 DCHECK(buffer_pos_ != nullptr);
263 DCHECK_GE(bit_width, 0);
264 DCHECK_LE(bit_width, MAX_BITWIDTH);
265 DCHECK_LE(bit_width, sizeof(T) * 8);
266 DCHECK_GE(num_values, 0);
267
268 int64_t num_read;
269 std::tie(buffer_pos_, num_read) =
270 BitPacking::UnpackValues(bit_width, buffer_pos_, bytes_left(), num_values, v);
271 DCHECK_LE(buffer_pos_, buffer_end_);
272 DCHECK_LE(num_read, num_values);
273 return static_cast<int>(num_read);
274}

276inline bool BatchedBitReader::SkipBatch(int bit_width, int num_values_to_skip) {
277 DCHECK(buffer_pos_ != nullptr);
278 DCHECK_GE(bit_width, 0);
279 DCHECK_LE(bit_width, MAX_BITWIDTH);
280 DCHECK_GE(num_values_to_skip, 0);
281
282 int skip_bytes = BitUtil::RoundUpNumBytes(bit_width * num_values_to_skip);
283 if (skip_bytes > buffer_end_ - buffer_pos_) {
284 return false;
285 }
286 buffer_pos_ += skip_bytes;
287 return true;
288}

290template <typename T>
291int BatchedBitReader::UnpackAndDecodeBatch(int bit_width, T* dict, int64_t dict_len, int num_values,
292 T* v, int64_t stride) {
293 DCHECK(buffer_pos_ != nullptr);
294 DCHECK_GE(bit_width, 0);
295 DCHECK_LE(bit_width, MAX_BITWIDTH);
296 DCHECK_GE(num_values, 0);
297
298 const uint8_t* new_buffer_pos;
299 int64_t num_read;
300 bool decode_error = false;
301 std::tie(new_buffer_pos, num_read) =
302 BitPacking::UnpackAndDecodeValues(bit_width, buffer_pos_, bytes_left(), dict, dict_len,
303 num_values, v, stride, &decode_error);
304 if (UNLIKELY(decode_error)) return -1;
305 buffer_pos_ = new_buffer_pos;
306 DCHECK_LE(buffer_pos_, buffer_end_);
307 DCHECK_LE(num_read, num_values);
308 return static_cast<int>(num_read);
309}

311template <typename T>
312bool BatchedBitReader::GetBytes(int num_bytes, T* v) {
313 DCHECK(buffer_pos_ != nullptr);
314 DCHECK_GE(num_bytes, 0);
315 DCHECK_LE(num_bytes, sizeof(T));
316 if (UNLIKELY(buffer_pos_ + num_bytes > buffer_end_)) return false;
317 *v = 0; // Ensure unset bytes are initialized to zero.
318 memcpy(v, buffer_pos_, num_bytes);
319 buffer_pos_ += num_bytes;
320 return true;
321}

323template <typename UINT_T>
324bool BatchedBitReader::GetUleb128(UINT_T* v) {
325 static_assert(std::is_integral<UINT_T>::value, "Integral type required.");
326 static_assert(std::is_unsigned<UINT_T>::value, "Unsigned type required.");
327 static_assert(!std::is_same<UINT_T, bool>::value, "Bools are not supported.");
328
329 *v = 0;
330 int shift = 0;
331 uint8_t byte = 0;
332 do {
333 if (UNLIKELY(shift >= max_vlq_byte_len<UINT_T>() * 7)) return false;
334 if (!GetBytes(1, &byte)) return false;
335
339 const UINT_T byte_as_UINT_T = byte;
340 *v |= (byte_as_UINT_T & 0x7Fu) << shift;
341 shift += 7;
342 } while ((byte & 0x80u) != 0);
343 return true;
344}

346template <typename INT_T>
347bool BatchedBitReader::GetZigZagInteger(INT_T* v) {
348 static_assert(std::is_integral<INT_T>::value, "Integral type required.");
349 static_assert(std::is_signed<INT_T>::value, "Signed type required.");
350
351 using UINT_T = std::make_unsigned_t<INT_T>;
352
353 UINT_T v_unsigned;
354 if (UNLIKELY(!GetUleb128<UINT_T>(&v_unsigned))) return false;
355
357 *v = (v_unsigned >> 1) ^ -(v_unsigned & 1u);
358
359 return true;
360}
