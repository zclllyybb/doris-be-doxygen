80 Status init_udaf(const TFunction& fn, const std::string& local_location) {
81 JNIEnv* env = nullptr;
82 RETURN_NOT_OK_STATUS_WITH_WARN(JniUtil::GetJNIEnv(&env), "Java-Udaf init_udaf function");
83 RETURN_IF_ERROR(JniUtil::GetGlobalClassRef(env, UDAF_EXECUTOR_CLASS, &executor_cl));
84 RETURN_NOT_OK_STATUS_WITH_WARN(register_func_id(env),
85 "Java-Udaf register_func_id function");
86
87 // Add a scoped cleanup jni reference object. This cleans up local refs made below.
88 JniLocalFrame jni_frame;
89 {
90 TJavaUdfExecutorCtorParams ctor_params;
91 ctor_params.__set_fn(fn);
92 ctor_params.__set_location(local_location);
93 ctor_params.__set_input_offsets_ptrs((int64_t)input_offsets_ptrs.get());
94 ctor_params.__set_input_buffer_ptrs((int64_t)input_values_buffer_ptr.get());
95 ctor_params.__set_input_nulls_ptrs((int64_t)input_nulls_buffer_ptr.get());
96 ctor_params.__set_input_array_nulls_buffer_ptr(
97 (int64_t)input_array_nulls_buffer_ptr.get());
98 ctor_params.__set_input_array_string_offsets_ptrs(
99 (int64_t)input_array_string_offsets_ptrs.get());
100
101 ctor_params.__set_output_buffer_ptr((int64_t)output_value_buffer.get());
102 ctor_params.__set_input_places_ptr((int64_t)input_place_ptrs.get());
103
104 ctor_params.__set_output_null_ptr((int64_t)output_null_value.get());
105 ctor_params.__set_output_offsets_ptr((int64_t)output_offsets_ptr.get());
106 ctor_params.__set_output_intermediate_state_ptr(
107 (int64_t)output_intermediate_state_ptr.get());
108 ctor_params.__set_output_array_null_ptr((int64_t)output_array_null_ptr.get());
109 ctor_params.__set_output_array_string_offsets_ptr(
110 (int64_t)output_array_string_offsets_ptr.get());
111
112 jbyteArray ctor_params_bytes;
113
114 // Pushed frame will be popped when jni_frame goes out-of-scope.
115 RETURN_IF_ERROR(jni_frame.push(env));
116 RETURN_IF_ERROR(SerializeThriftMsg(env, &ctor_params, &ctor_params_bytes));
117 executor_obj = env->NewObject(executor_cl, executor_ctor_id, ctor_params_bytes);
118
119 jbyte* pBytes = env->GetByteArrayElements(ctor_params_bytes, nullptr);
120 env->ReleaseByteArrayElements(ctor_params_bytes, pBytes, JNI_ABORT);
121 env->DeleteLocalRef(ctor_params_bytes);
122 }
123 RETURN_ERROR_IF_EXC(env);
124 RETURN_IF_ERROR(JniUtil::LocalToGlobalRef(env, executor_obj, &executor_obj));
125 return Status::OK();
126 }

128 Status add(const int64_t places_address[], bool is_single_place, const IColumn** columns,
129 size_t row_num_start, size_t row_num_end, const DataTypes& argument_types) {
130 JNIEnv* env = nullptr;
131 RETURN_NOT_OK_STATUS_WITH_WARN(JniUtil::GetJNIEnv(&env), "Java-Udaf add function");
132 for (int arg_idx = 0; arg_idx < argument_size; ++arg_idx) {
133 auto data_col = columns[arg_idx];
134 if (auto* nullable = check_and_get_column<const ColumnNullable>(*columns[arg_idx])) {
135 data_col = nullable->get_nested_column_ptr();
136 auto null_col = check_and_get_column<ColumnVector<UInt8>>(
137 nullable->get_null_map_column_ptr());
138 input_nulls_buffer_ptr.get()[arg_idx] =
139 reinterpret_cast<int64_t>(null_col->get_data().data());
140 } else {
141 input_nulls_buffer_ptr.get()[arg_idx] = -1;
142 }
143 if (data_col->is_column_string()) {
144 const ColumnString* str_col = check_and_get_column<ColumnString>(data_col);
145 input_values_buffer_ptr.get()[arg_idx] =
146 reinterpret_cast<int64_t>(str_col->get_chars().data());
147 input_offsets_ptrs.get()[arg_idx] =
148 reinterpret_cast<int64_t>(str_col->get_offsets().data());
149 } else if (data_col->is_numeric() || data_col->is_column_decimal()) {
150 input_values_buffer_ptr.get()[arg_idx] =
151 reinterpret_cast<int64_t>(data_col->get_raw_data().data);
152 } else if (data_col->is_column_array()) {
153 const ColumnArray* array_col = assert_cast<const ColumnArray*>(data_col);
154 input_offsets_ptrs.get()[arg_idx] = reinterpret_cast<int64_t>(
155 array_col->get_offsets_column().get_raw_data().data);
156 const ColumnNullable& array_nested_nullable =
157 assert_cast<const ColumnNullable&>(array_col->get_data());
158 auto data_column_null_map = array_nested_nullable.get_null_map_column_ptr();
159 auto data_column = array_nested_nullable.get_nested_column_ptr();
160 input_array_nulls_buffer_ptr.get()[arg_idx] = reinterpret_cast<int64_t>(
161 check_and_get_column<ColumnVector<UInt8>>(data_column_null_map)
162 ->get_data()
163 .data());
164
165 //need pass FE, nullamp and offset, chars
166 if (data_column->is_column_string()) {
167 const ColumnString* col = assert_cast<const ColumnString*>(data_column.get());
168 input_values_buffer_ptr.get()[arg_idx] =
169 reinterpret_cast<int64_t>(col->get_chars().data());
170 input_array_string_offsets_ptrs.get()[arg_idx] =
171 reinterpret_cast<int64_t>(col->get_offsets().data());
172 } else {
173 input_values_buffer_ptr.get()[arg_idx] =
174 reinterpret_cast<int64_t>(data_column->get_raw_data().data);
175 }
176 } else {
177 return Status::InvalidArgument(
178 strings::Substitute("Java UDAF doesn't support type is $0 now !",
179 argument_types[arg_idx]->get_name()));
180 }
181 }
182 *input_place_ptrs = reinterpret_cast<int64_t>(places_address);
183 env->CallNonvirtualVoidMethod(executor_obj, executor_cl, executor_add_id, is_single_place,
184 row_num_start, row_num_end);
185 return JniUtil::GetJniExceptionMsg(env);
186 }

188 Status merge(const AggregateJavaUdafData& rhs, int64_t place) {
189 JNIEnv* env = nullptr;
190 RETURN_NOT_OK_STATUS_WITH_WARN(JniUtil::GetJNIEnv(&env), "Java-Udaf merge function");
191 serialize_data = rhs.serialize_data;
192 long len = serialize_data.length();
193 jbyteArray arr = env->NewByteArray(len);
194 env->SetByteArrayRegion(arr, 0, len, reinterpret_cast<jbyte*>(serialize_data.data()));
195 env->CallNonvirtualVoidMethod(executor_obj, executor_cl, executor_merge_id, place, arr);
196 jbyte* pBytes = env->GetByteArrayElements(arr, nullptr);
197 env->ReleaseByteArrayElements(arr, pBytes, JNI_ABORT);
198 env->DeleteLocalRef(arr);
199 return JniUtil::GetJniExceptionMsg(env);
200 }

202 Status write(BufferWritable& buf, int64_t place) {
203 JNIEnv* env = nullptr;
204 RETURN_NOT_OK_STATUS_WITH_WARN(JniUtil::GetJNIEnv(&env), "Java-Udaf write function");
205 // TODO: Here get a byte[] from FE serialize, and then allocate the same length bytes to
206 // save it in BE, Because i'm not sure there is a way to use the byte[] not allocate again.
207 jbyteArray arr = (jbyteArray)(env->CallNonvirtualObjectMethod(

209 int len = env->GetArrayLength(arr);
210 serialize_data.resize(len);

212 write_binary(serialize_data, buf);

216 return JniUtil::GetJniExceptionMsg(env);

228 Status get(IColumn& to, const DataTypePtr& result_type, int64_t place) const {
229 to.insert_default();
230 JNIEnv* env = nullptr;
231 RETURN_NOT_OK_STATUS_WITH_WARN(JniUtil::GetJNIEnv(&env), "Java-Udaf get value function");
232 if (result_type->is_nullable()) {
233 auto& nullable = assert_cast<ColumnNullable&>(to);
234 *output_null_value =
235 reinterpret_cast<int64_t>(nullable.get_null_map_column().get_raw_data().data);
236 auto& data_col = nullable.get_nested_column();
237
238#ifndef EVALUATE_JAVA_UDAF
239#define EVALUATE_JAVA_UDAF \
240 if (data_col.is_column_string()) { \
241 const ColumnString* str_col = check_and_get_column<ColumnString>(data_col); \
242 ColumnString::Chars& chars = const_cast<ColumnString::Chars&>(str_col->get_chars()); \
243 ColumnString::Offsets& offsets = \
244 const_cast<ColumnString::Offsets&>(str_col->get_offsets()); \
245 int increase_buffer_size = 0; \
246 int64_t buffer_size = JniUtil::IncreaseReservedBufferSize(increase_buffer_size); \
247 chars.resize(buffer_size); \
248 *output_value_buffer = reinterpret_cast<int64_t>(chars.data()); \
249 *output_offsets_ptr = reinterpret_cast<int64_t>(offsets.data()); \
250 *output_intermediate_state_ptr = chars.size(); \
251 jboolean res = env->CallNonvirtualBooleanMethod(executor_obj, executor_cl, \
252 executor_result_id, to.size() - 1, place); \
253 while (res != JNI_TRUE) { \
254 /*Add this check is now, the agg function can't deal with the return status, */ \
255 /*even we return a bad status, nobody could deal with it,*/ \
256 /*so add this limit avoid std::bad_alloc, (1024<<10) is enough*/ \
257 /*but this maybe get a mistake of result,when could handle exception need removethis*/ \
258 if (increase_buffer_size == 10) { \
259 return Status::MemoryAllocFailed("memory allocate failed, buffer:{},size:{}", \
260 increase_buffer_size, buffer_size); \
261 } \
262 increase_buffer_size++; \
263 buffer_size = JniUtil::IncreaseReservedBufferSize(increase_buffer_size); \
264 chars.resize(buffer_size); \
265 *output_value_buffer = reinterpret_cast<int64_t>(chars.data()); \
266 *output_intermediate_state_ptr = chars.size(); \
267 res = env->CallNonvirtualBooleanMethod(executor_obj, executor_cl, executor_result_id, \
268 to.size() - 1, place); \
269 } \
270 } else if (data_col.is_numeric() || data_col.is_column_decimal()) { \
271 *output_value_buffer = reinterpret_cast<int64_t>(data_col.get_raw_data().data); \
272 env->CallNonvirtualBooleanMethod(executor_obj, executor_cl, executor_result_id, \
273 to.size() - 1, place); \
274 } else if (data_col.is_column_array()) { \
275 ColumnArray& array_col = assert_cast<ColumnArray&>(data_col); \
276 ColumnNullable& array_nested_nullable = \
277 assert_cast<ColumnNullable&>(array_col.get_data()); \
278 auto data_column_null_map = array_nested_nullable.get_null_map_column_ptr(); \
279 auto data_column = array_nested_nullable.get_nested_column_ptr(); \
280 auto& offset_column = array_col.get_offsets_column(); \
281 int increase_buffer_size = 0; \
282 int64_t buffer_size = JniUtil::IncreaseReservedBufferSize(increase_buffer_size); \
283 *output_offsets_ptr = reinterpret_cast<int64_t>(offset_column.get_raw_data().data); \
284 data_column_null_map->resize(buffer_size); \
285 auto& null_map_data = \
286 assert_cast<ColumnVector<UInt8>*>(data_column_null_map.get())->get_data(); \
287 *output_array_null_ptr = reinterpret_cast<int64_t>(null_map_data.data()); \
288 *output_intermediate_state_ptr = buffer_size; \
289 if (data_column->is_column_string()) { \
290 ColumnString* str_col = assert_cast<ColumnString*>(data_column.get()); \
291 ColumnString::Chars& chars = assert_cast<ColumnString::Chars&>(str_col->get_chars()); \
292 ColumnString::Offsets& offsets = \
293 assert_cast<ColumnString::Offsets&>(str_col->get_offsets()); \
294 chars.resize(buffer_size); \
295 offsets.resize(buffer_size); \
296 *output_value_buffer = reinterpret_cast<int64_t>(chars.data()); \
297 *output_array_string_offsets_ptr = reinterpret_cast<int64_t>(offsets.data()); \
298 jboolean res = env->CallNonvirtualBooleanMethod( \
299 executor_obj, executor_cl, executor_result_id, to.size() - 1, place); \
300 while (res != JNI_TRUE) { \
301 if (increase_buffer_size == 10) { \
302 return Status::MemoryAllocFailed("memory allocate failed, buffer:{},size:{}", \
303 increase_buffer_size, buffer_size); \
304 } \
305 increase_buffer_size++; \
306 buffer_size = JniUtil::IncreaseReservedBufferSize(increase_buffer_size); \
307 null_map_data.resize(buffer_size); \
308 chars.resize(buffer_size); \
309 offsets.resize(buffer_size); \
310 *output_array_null_ptr = reinterpret_cast<int64_t>(null_map_data.data()); \
311 *output_value_buffer = reinterpret_cast<int64_t>(chars.data()); \
312 *output_array_string_offsets_ptr = reinterpret_cast<int64_t>(offsets.data()); \
313 *output_intermediate_state_ptr = buffer_size; \
314 res = env->CallNonvirtualBooleanMethod(executor_obj, executor_cl, \
315 executor_result_id, to.size() - 1, place); \
316 } \
317 } else { \
318 data_column->resize(buffer_size); \
319 *output_value_buffer = reinterpret_cast<int64_t>(data_column->get_raw_data().data); \
320 jboolean res = env->CallNonvirtualBooleanMethod( \
321 executor_obj, executor_cl, executor_result_id, to.size() - 1, place); \
322 while (res != JNI_TRUE) { \
323 if (increase_buffer_size == 10) { \
324 return Status::MemoryAllocFailed("memory allocate failed, buffer:{},size:{}", \
325 increase_buffer_size, buffer_size); \
326 } \
327 increase_buffer_size++; \
328 buffer_size = JniUtil::IncreaseReservedBufferSize(increase_buffer_size); \
329 null_map_data.resize(buffer_size); \
330 data_column->resize(buffer_size); \
331 *output_array_null_ptr = reinterpret_cast<int64_t>(null_map_data.data()); \
332 *output_value_buffer = \
333 reinterpret_cast<int64_t>(data_column->get_raw_data().data); \
334 *output_intermediate_state_ptr = buffer_size; \
335 res = env->CallNonvirtualBooleanMethod(executor_obj, executor_cl, \
336 executor_result_id, to.size() - 1, place); \
337 } \
338 } \
339 } else { \
340 return Status::InvalidArgument(strings::Substitute( \
341 "Java UDAF doesn't support return type is $0 now !", result_type->get_name())); \
342 }
343#endif
344 EVALUATE_JAVA_UDAF;
345 } else {
346 *output_null_value = -1;
347 auto& data_col = to;
348 EVALUATE_JAVA_UDAF;
349 }
350 return JniUtil::GetJniExceptionMsg(env);
351 }

355 auto register_id = [&](const char* func_name, const char* func_sign, jmethodID& func_id) {
356 func_id = env->GetMethodID(executor_cl, func_name, func_sign);

362 return s;
363 };

469 // TODO: Here we calling method by jni, And if we get a thrown from FE,

492 }

495 // will call create --- deserialize --- merge --- destory for each rows ,
496 // so need doing new (place), to create Data and read to buf, then call merge ,

503 }

507 }
