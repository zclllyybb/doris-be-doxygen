45// This class splits a string using the given delimiter, returning the split
46// substrings via an iterator interface. An optional Predicate functor may be
47// supplied, which will be used to filter the split strings: strings for which
48// the predicate returns false will be skipped. A Predicate object is any
49// functor that takes a StringPiece and returns bool. By default, the NoFilter
50// Predicate is used, which does not filter out anything.
51//
52// This class is NOT part of the public splitting API.
53//
54// Usage:
55//
56// using strings::delimiter::Literal;
57// Literal d(",");
58// for (SplitIterator<Literal> it("a,b,c", d), end(d); it != end; ++it) {
59// StringPiece substring = *it;
60// DoWork(substring);
61// }
62//
63// The explicit single-argument constructor is used to create an "end" iterator.
64// The two-argument constructor is used to split the given text using the given
65// delimiter.
66template <typename Delimiter, typename Predicate = NoFilter>
67class SplitIterator {
68public:
69 using iterator_category = std::input_iterator_tag;
70 using value_type = StringPiece;
71 using difference_type = ptrdiff_t;
72 using pointer = StringPiece*;
73 using reference = StringPiece&;
74
75 // Two constructors for "end" iterators.
76 explicit SplitIterator(Delimiter d) : delimiter_(std::move(d)), predicate_(), is_end_(true) {}
77 SplitIterator(Delimiter d, Predicate p)
78 : delimiter_(std::move(d)), predicate_(std::move(p)), is_end_(true) {}
79 // Two constructors taking the text to iterator.
80 SplitIterator(StringPiece text, Delimiter d)
81 : text_(std::move(text)), delimiter_(std::move(d)), predicate_(), is_end_(false) {
82 ++(*this);
83 }
84 SplitIterator(StringPiece text, Delimiter d, Predicate p)
85 : text_(std::move(text)),
86 delimiter_(std::move(d)),
87 predicate_(std::move(p)),
88 is_end_(false) {
89 ++(*this);
90 }
91
92 StringPiece operator*() { return curr_piece_; }
93 StringPiece* operator->() { return &curr_piece_; }
94
95 SplitIterator& operator++() {
96 do {
97 if (text_.end() == curr_piece_.end()) {
98 // Already consumed all of text_, so we're done.
99 is_end_ = true;
100 return *this;
101 }
102 StringPiece found_delimiter = delimiter_.Find(text_);
103 assert(found_delimiter.data() != NULL);
104 assert(text_.begin() <= found_delimiter.begin());
105 assert(found_delimiter.end() <= text_.end());
106 // found_delimiter is allowed to be empty.
107 // Sets curr_piece_ to all text up to but excluding the delimiter itself.
108 // Sets text_ to remaining data after the delimiter.
109 curr_piece_.set(text_.begin(), found_delimiter.begin() - text_.begin());
110 text_.remove_prefix(found_delimiter.end() - text_.begin());
111 } while (!predicate_(curr_piece_));
112 return *this;
113 }
114
115 SplitIterator operator++(int /* postincrement */) {
116 SplitIterator old(*this);
117 ++(*this);
118 return old;
119 }
120
121 bool operator==(const SplitIterator& other) const {
122 // Two "end" iterators are always equal. If the two iterators being compared
123 // aren't both end iterators, then we fallback to comparing their fields.
124 // Importantly, the text being split must be equal and the current piece
125 // within the text being split must also be equal. The delimiter_ and
126 // predicate_ fields need not be checked here because they're template
127 // parameters that are already part of the SplitIterator's type.
128 return (is_end_ && other.is_end_) ||
129 (is_end_ == other.is_end_ && text_ == other.text_ &&
130 text_.data() == other.text_.data() && curr_piece_ == other.curr_piece_ &&
131 curr_piece_.data() == other.curr_piece_.data());
132 }
133
134 bool operator!=(const SplitIterator& other) const { return !(*this == other); }
135
136private:
137 // The text being split. Modified as delimited pieces are consumed.
138 StringPiece text_;
139 Delimiter delimiter_;
140 Predicate predicate_;
141 bool is_end_;
142 // Holds the currently split piece of text. Will always refer to string data
143 // within text_. This value is returned when the iterator is dereferenced.
144 StringPiece curr_piece_;
145};

147// Declares a functor that can convert a StringPiece to another type. This works
148// for any type that has a constructor (explicit or not) taking a single
149// StringPiece argument. A specialization exists for converting to string
150// because the underlying data needs to be copied. In theory, these
151// specializations could be extended to work with other types (e.g., int32), but
152// then a solution for error reporting would need to be devised.
153template <typename To>
154struct StringPieceTo {
155 To operator()(StringPiece from) const { return To(from); }
156};

158// Specialization for converting to string.
159template <>
160struct StringPieceTo<string> {
161 string operator()(StringPiece from) const { return from.ToString(); }
162};

165template <>
166struct StringPieceTo<const string> {
167 string operator()(StringPiece from) const { return from.ToString(); }

181// This class implements the behavior of the split API by giving callers access
182// to the underlying split substrings in various convenient ways, such as
183// through iterators or implicit conversion functions. Do not construct this
184// class directly, rather use the Split() function instead.
185//
186// Output containers can be collections of either StringPiece or string objects.
187// StringPiece is more efficient because the underlying data will not need to be
188// copied; the returned StringPieces will all refer to the data within the
189// original input string. If a collection of string objects is used, then each
190// substring will be copied.
191//
192// An optional Predicate functor may be supplied. This predicate will be used to
193// filter the split strings: only strings for which the predicate returns true
194// will be kept. A Predicate object is any unary functor that takes a
195// StringPiece and returns bool. By default, the NoFilter predicate is used,
196// which does not filter out anything.
197template <typename Delimiter, typename Predicate = NoFilter>
198class Splitter {
199public:
200 typedef internal::SplitIterator<Delimiter, Predicate> Iterator;
201
202 Splitter(StringPiece text, Delimiter d) : begin_(text, d), end_(d) {}
203
204 Splitter(StringPiece text, Delimiter d, Predicate p) : begin_(text, d, p), end_(d, p) {}

232 // Uses SFINAE to restrict conversion to container-like types (by testing for
233 // the presence of a const_iterator member type) and also to disable

293 // Inserts split results into the container. To do this the results are first
294 // stored in a vector<StringPiece>. This is where the input text is actually
295 // "parsed". This vector is then used to possibly reserve space in the output
296 // container, and the StringPieces in "v" are converted as necessary to the
297 // output container's value type.
298 //
299 // The reason to use an intermediate vector of StringPiece is so we can learn
300 // the needed capacity of the output container. This is needed when the output
301 // container is a vector<string> in which case resizes can be expensive due to
302 // copying of the ::string objects.
303 //
304 // At some point in the future we might add a C++11 move constructor to
305 // ::string, in which case the vector resizes are much less expensive and the
306 // use of this intermediate vector "v" can be removed.
307 template <typename Container>
308 Container ToContainer() {
309 vector<StringPiece> v;
310 for (Iterator it = begin(); it != end_; ++it) {
311 v.push_back(*it);
312 }
313 typedef typename Container::value_type ToType;
314 internal::StringPieceTo<ToType> converter;
315 Container c;
316 ReserveCapacity(&c, v.size());
317 std::insert_iterator<Container> inserter(c, c.begin());
318 for (const auto& sp : v) {
319 *inserter++ = converter(sp);
320 }
321 return c;
322 }

324 // The algorithm is to insert a new pair into the map for each even-numbered
325 // item, with the even-numbered item as the key with a default-constructed
326 // value. Each odd-numbered item will then be assigned to the last pair's
327 // value.
328 template <typename Map>
329 Map ToMap() {
330 typedef typename Map::key_type Key;
331 typedef typename Map::mapped_type Data;
332 Map m;
333 StringPieceTo<Key> key_converter;
334 StringPieceTo<Data> val_converter;
335 typename Map::iterator curr_pair;
336 bool is_even = true;
337 for (Iterator it = begin(); it != end_; ++it) {
338 if (is_even) {
339 curr_pair = InsertInMap(std::make_pair(key_converter(*it), Data()), &m);
340 } else {
341 curr_pair->second = val_converter(*it);
342 }
343 is_even = !is_even;
344 }
345 return m;
346 }

348 // Returns a pair with its .first and .second members set to the first two
349 // strings returned by the begin() iterator. Either/both of .first and .second
350 // will be empty strings if the iterator doesn't have a corresponding value.
351 template <typename First, typename Second>
352 std::pair<First, Second> ToPair() {
353 StringPieceTo<First> first_converter;
354 StringPieceTo<Second> second_converter;
355 StringPiece first, second;
356 Iterator it = begin();
357 if (it != end()) {
358 first = *it;
359 if (++it != end()) {
360 second = *it;
361 }
362 }
363 return std::make_pair(first_converter(first), second_converter(second));
364 }

366 // Overloaded InsertInMap() function. The first overload is the commonly used
367 // one for most map-like objects. The second overload is a special case for
368 // multimap, because multimap's insert() member function directly returns an
369 // iterator, rather than a pair<iterator, bool> like map's.
370 template <typename Map>
371 typename Map::iterator InsertInMap(const typename Map::value_type& value, Map* map) {
