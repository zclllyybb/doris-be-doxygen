43class KafkaLoadInfo {
44public:
45 KafkaLoadInfo(const TKafkaLoadInfo& t_info)
46 : brokers(t_info.brokers),
47 topic(t_info.topic),
48 begin_offset(t_info.partition_begin_offset),
49 properties(t_info.properties) {
50 // The offset(begin_offset) sent from FE is the starting offset,
51 // and the offset(cmt_offset) reported by BE to FE is the consumed offset,
52 // so we need to minus 1 here.
53 for (auto& p : t_info.partition_begin_offset) {
54 cmt_offset[p.first] = p.second - 1;
55 }
56 }
57
58 void reset_offset() {
59 // reset the commit offset
60 for (auto& p : begin_offset) {
61 cmt_offset[p.first] = p.second - 1;
62 }
63 }
64
65public:
66 std::string brokers;
67 std::string topic;
68
69 // the following members control the max progress of a consuming
70 // process. if any of them reach, the consuming will finish.
71 int64_t max_interval_s = 5;
72 int64_t max_batch_rows = 1024;
73 int64_t max_batch_size = 100 * 1024 * 1024; // 100MB
74
75 // partition -> begin offset, inclusive.
76 std::map<int32_t, int64_t> begin_offset;
77 // partition -> commit offset, inclusive.
78 std::map<int32_t, int64_t> cmt_offset;
79 //custom kafka property key -> value
80 std::map<std::string, std::string> properties;
81};
