35// Fixed capacity FIFO queue, where both blocking_get and blocking_put operations block
36// if the queue is empty or full, respectively.
37template <typename T>
38class BlockingPriorityQueue {
39public:
40 BlockingPriorityQueue(size_t max_elements)
41 : _shutdown(false),
42 _max_element(max_elements),
43 _upgrade_counter(0),
44 _total_get_wait_time(0),
45 _total_put_wait_time(0) {}
46
47 // Get an element from the queue, waiting indefinitely (or until timeout) for one to become available.
48 // Returns false if we were shut down prior to getting the element, and there
49 // are no more elements available.
50 // -- timeout_ms: 0 means wait indefinitely
51 bool blocking_get(T* out, uint32_t timeout_ms = 0) {
52 MonotonicStopWatch timer;
53 timer.start();
54 std::unique_lock unique_lock(_lock);
55 bool wait_successful = false;
56#if !defined(USE_BTHREAD_SCANNER)
57 if (timeout_ms > 0) {
58 wait_successful = _get_cv.wait_for(unique_lock, std::chrono::milliseconds(timeout_ms),
59 [this] { return _shutdown || !_queue.empty(); });
60 } else {
61 _get_cv.wait(unique_lock, [this] { return _shutdown || !_queue.empty(); });
62 wait_successful = true;
63 }
64#else
65 if (timeout_ms > 0) {
66 wait_successful = true;
67 while (!(_shutdown || !_queue.empty())) {
68 if (_get_cv.wait_for(unique_lock, timeout_ms * 1000) != 0) {
69 // timeout
70 wait_successful = _shutdown || !_queue.empty();
71 break;
72 }
73 }
74 } else {
75 while (!(_shutdown || !_queue.empty())) {
76 _get_cv.wait(unique_lock);
77 }
78 wait_successful = true;
79 }
80#endif
81 _total_get_wait_time += timer.elapsed_time();
82 if (wait_successful) {
83 if (_upgrade_counter > config::priority_queue_remaining_tasks_increased_frequency) {
84 std::priority_queue<T> tmp_queue;
85 while (!_queue.empty()) {
86 T v = _queue.top();
87 _queue.pop();
88 ++v;
89 tmp_queue.push(v);
90 }
91 swap(_queue, tmp_queue);
92 _upgrade_counter = 0;
93 }
94 if (!_queue.empty()) {
95 *out = _queue.top();
96 _queue.pop();
97 ++_upgrade_counter;
98 _put_cv.notify_one();
99 return true;
100 } else {
101 assert(_shutdown);
102 return false;
103 }
104 } else {
105 //time out
106 assert(!_shutdown);
107 return false;
108 }
109 }
110
111 bool non_blocking_get(T* out) {
112 MonotonicStopWatch timer;
113 timer.start();
114 std::unique_lock unique_lock(_lock);
115
116 if (!_queue.empty()) {
117 // 定期提高队列中残留的任务优先级
118 // 保证优先级较低的大查询不至于完全饿死
119 if (_upgrade_counter > config::priority_queue_remaining_tasks_increased_frequency) {
120 std::priority_queue<T> tmp_queue;
121 while (!_queue.empty()) {
122 T v = _queue.top();
123 _queue.pop();
124 ++v;
125 tmp_queue.push(v);
126 }
127 swap(_queue, tmp_queue);
128 _upgrade_counter = 0;
129 }
130 *out = _queue.top();
131 _queue.pop();
132 ++_upgrade_counter;
133 _total_get_wait_time += timer.elapsed_time();
134 _put_cv.notify_one();
135 return true;
136 }
137
138 return false;
139 }
140
141 // Puts an element into the queue, waiting indefinitely until there is space.
142 // If the queue is shut down, returns false.
143 bool blocking_put(const T& val) {
144 MonotonicStopWatch timer;
145 timer.start();
146 std::unique_lock unique_lock(_lock);
147 while (!(_shutdown || _queue.size() < _max_element)) {
148 _put_cv.wait(unique_lock);
149 }
150 _total_put_wait_time += timer.elapsed_time();
151
152 if (_shutdown) {
153 return false;
154 }
155
156 _queue.push(val);
157 _get_cv.notify_one();
158 return true;
159 }
160
161 // Return false if queue full or has been shutdown.
162 bool try_put(const T& val) {
163 std::unique_lock<std::mutex> unique_lock(_lock);
164 if (_queue.size() < _max_element && !_shutdown) {
165 _queue.push(val);
166 unique_lock.unlock();
167 _get_cv.notify_one();
168 return true;
169 }
170 return false;
171 }
172
173 // Shut down the queue. Wakes up all threads waiting on blocking_get or blocking_put.
174 void shutdown() {
175 {
176 std::lock_guard l(_lock);
177 _shutdown = true;
178 }
179 _get_cv.notify_all();
180 _put_cv.notify_all();
181 }
182
183 uint32_t get_size() const {
184 std::lock_guard l(_lock);
185 return _queue.size();
186 }
187
188 // Returns the total amount of time threads have blocked in blocking_get.
189 uint64_t total_get_wait_time() const { return _total_get_wait_time; }
190
191 // Returns the total amount of time threads have blocked in blocking_put.
192 uint64_t total_put_wait_time() const { return _total_put_wait_time; }
193
194private:
195 bool _shutdown;
196 const int _max_element;
197 doris::ConditionVariable _get_cv; // 'get' callers wait on this
198 doris::ConditionVariable _put_cv; // 'put' callers wait on this
199 // _lock guards access to _queue, total_get_wait_time, and total_put_wait_time
200 mutable doris::Mutex _lock;
201 std::priority_queue<T> _queue;
202 int _upgrade_counter;
203 std::atomic<uint64_t> _total_get_wait_time;
204 std::atomic<uint64_t> _total_put_wait_time;
205};
206
