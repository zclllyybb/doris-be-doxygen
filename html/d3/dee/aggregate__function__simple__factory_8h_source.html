33namespace doris::vectorized {
34using DataTypePtr = std::shared_ptr<const IDataType>;
35using DataTypes = std::vector<DataTypePtr>;
36using AggregateFunctionCreator =
37 std::function<AggregateFunctionPtr(const std::string&, const DataTypes&, const bool)>;
38
39class AggregateFunctionSimpleFactory {
40public:
41 using Creator = AggregateFunctionCreator;
42
43private:
44 using AggregateFunctions = std::unordered_map<std::string, Creator>;
45
46 AggregateFunctions aggregate_functions;
47 AggregateFunctions nullable_aggregate_functions;
48 std::unordered_map<std::string, std::string> function_alias;
49
50public:
51 void register_nullable_function_combinator(const Creator& creator) {
52 for (const auto& entity : aggregate_functions) {
53 if (nullable_aggregate_functions.find(entity.first) ==
54 nullable_aggregate_functions.end()) {
55 nullable_aggregate_functions[entity.first] = creator;
56 }
57 }
58 }
59
60 void register_distinct_function_combinator(const Creator& creator, const std::string& prefix,
61 bool nullable = false) {
62 auto& functions = nullable ? nullable_aggregate_functions : aggregate_functions;
63 std::vector<std::string> need_insert;
64 for (const auto& entity : aggregate_functions) {
65 std::string target_value = prefix + entity.first;
66 if (functions.find(target_value) == functions.end()) {
67 need_insert.emplace_back(std::move(target_value));
68 }
69 }
70 for (const auto& function_name : need_insert) {
71 register_function(function_name, creator, nullable);
72 }
73 }
74
75 AggregateFunctionPtr get(const std::string& name, const DataTypes& argument_types,
76 const bool result_is_nullable = false) {
77 bool nullable = false;
78 for (const auto& type : argument_types) {
79 if (type->is_nullable()) {
80 nullable = true;
81 }
82 }
83
84 std::string name_str = name;
85 if (function_alias.count(name)) {
86 name_str = function_alias[name];
87 }
88
89 if (nullable) {
90 return nullable_aggregate_functions.find(name_str) == nullable_aggregate_functions.end()
91 ? nullptr
92 : nullable_aggregate_functions[name_str](name_str, argument_types,
93 result_is_nullable);
94 } else {
95 return aggregate_functions.find(name_str) == aggregate_functions.end()
96 ? nullptr
97 : aggregate_functions[name_str](name_str, argument_types,
98 result_is_nullable);
99 }
100 }
101
102 void register_function(const std::string& name, const Creator& creator, bool nullable = false) {
103 if (nullable) {
104 nullable_aggregate_functions[name] = creator;
105 } else {
106 aggregate_functions[name] = creator;
107 }
108 }
109
110 void register_function_both(const std::string& name, const Creator& creator) {
111 register_function(name, creator, false);
112 register_function(name, creator, true);
113 }
114
115 void register_alias(const std::string& name, const std::string& alias) {
116 function_alias[alias] = name;
117 }
118
119public:
120 static AggregateFunctionSimpleFactory& instance();
121};
122}; // namespace doris::vectorized
