32class FunctionArrayDifference : public IFunction {
33public:
34 static constexpr auto name = "array_difference";
35
36 static FunctionPtr create() { return std::make_shared<FunctionArrayDifference>(); }
37
38 String get_name() const override { return name; }
39
40 bool is_variadic() const override { return false; }
41
42 size_t get_number_of_arguments() const override { return 1; }
43
44 bool use_default_implementation_for_nulls() const override { return true; }

46 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
47 DCHECK(is_array(arguments[0]))
48 << "argument for function: " << name << " should be DataTypeArray but it has type "
49 << arguments[0]->get_name() << ".";
50 auto nested_type = assert_cast<const DataTypeArray&>(*(arguments[0])).get_nested_type();
51 bool is_nullable = nested_type->is_nullable();
52
53 WhichDataType which(remove_nullable(nested_type));
54 //return type is promoted to prevent result overflow
55 //like: input is int32 ---> return type will be int64
56 DataTypePtr return_type = nullptr;
57 if (which.is_uint8() || which.is_int8()) {
58 return_type = std::make_shared<DataTypeInt16>();
59 } else if (which.is_uint16() || which.is_int16()) {
60 return_type = std::make_shared<DataTypeInt32>();
61 } else if (which.is_uint32() || which.is_uint64() || which.is_int32()) {
62 return_type = std::make_shared<DataTypeInt64>();
63 } else if (which.is_int64() || which.is_int128()) {
64 return_type = std::make_shared<DataTypeInt128>();
65 } else if (which.is_float32() || which.is_float64()) {
66 return_type = std::make_shared<DataTypeFloat64>();
67 } else if (which.is_decimal()) {
68 return arguments[0];
69 }
70 if (return_type) {
71 return std::make_shared<DataTypeArray>(is_nullable ? make_nullable(return_type)
72 : return_type);
73 } else {
74 LOG(FATAL) << "Function of " << name
75 << " return type get wrong: and input argument is: "
76 << arguments[0]->get_name();
77 }
78 }

80 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
81 size_t result, size_t input_rows_count) override {
82 const ColumnWithTypeAndName& arg = block.get_by_position(arguments[0]);
83 auto res_column = _execute_non_nullable(arg, input_rows_count);
84 if (!res_column) {
85 return Status::RuntimeError(
86 fmt::format("unsupported types for function {}({})", get_name(),
87 block.get_by_position(arguments[0]).type->get_name()));
88 }
89 DCHECK_EQ(arg.column->size(), res_column->size());
90 block.replace_by_position(result, std::move(res_column));
91 return Status::OK();
92 }
93
94private:
95 template <typename Element, typename Result>
96 static void impl(const Element* __restrict src, Result* __restrict dst, size_t begin,
97 size_t end) {
98 size_t curr_pos = begin;
99 if (curr_pos < end) {
100 Element prev_element = src[curr_pos];
101 dst[curr_pos] = {};
102 curr_pos++;
103 Element curr_element = src[curr_pos];
104 for (; curr_pos < end; ++curr_pos) {
105 curr_element = src[curr_pos];
106 dst[curr_pos] =
107 static_cast<Result>(curr_element) - static_cast<Result>(prev_element);
108 prev_element = curr_element;
109 }
110 }
111 }

113 template <typename Element, typename Result>
114 ColumnPtr _execute_number_expanded(const ColumnArray::Offsets64& offsets,
115 const IColumn& nested_column, ColumnPtr nested_null_map) {
116 using ColVecType = ColumnVectorOrDecimal<Element>;
117 using ColVecResult = ColumnVectorOrDecimal<Result>;
118 typename ColVecResult::MutablePtr res_nested = nullptr;
119
120 const auto& src_data = reinterpret_cast<const ColVecType&>(nested_column).get_data();
121 if constexpr (IsDecimalNumber<Result>) {
122 res_nested = ColVecResult::create(0, src_data.get_scale());
123 } else {
124 res_nested = ColVecResult::create();
125 }
126 auto size = nested_column.size();
127 typename ColVecResult::Container& res_values = res_nested->get_data();
128 res_values.resize(size);
129
130 size_t pos = 0;
131 for (auto offset : offsets) {
132 impl(src_data.data(), res_values.data(), pos, offset);
133 pos = offset;
134 }
135 if (nested_null_map) {
136 auto null_map_col = ColumnUInt8::create(size, 0);
137 auto& null_map_col_data = null_map_col->get_data();
138 auto nested_colum_data = static_cast<const ColumnVector<UInt8>*>(nested_null_map.get());
139 VectorizedUtils::update_null_map(null_map_col_data, nested_colum_data->get_data());
140 for (size_t row = 0; row < offsets.size(); ++row) {
141 auto off = offsets[row - 1];
142 auto len = offsets[row] - off;
143 auto pos = len ? len - 1 : 0;
144 for (; pos > 0; --pos) {
145 if (null_map_col_data[pos + off - 1]) {
146 null_map_col_data[pos + off] = 1;
147 }
148 }
149 }
150 return ColumnNullable::create(std::move(res_nested), std::move(null_map_col));
151 } else {
152 return res_nested;
153 }
154 }

156 ColumnPtr _execute_non_nullable(const ColumnWithTypeAndName& arg, size_t input_rows_count) {
157 // check array nested column type and get data
158 auto left_column = arg.column->convert_to_full_column_if_const();
159 const auto& array_column = reinterpret_cast<const ColumnArray&>(*left_column);
160 const auto& offsets = array_column.get_offsets();
161 DCHECK(offsets.size() == input_rows_count);
162
163 ColumnPtr nested_column = nullptr;
164 ColumnPtr nested_null_map = nullptr;
165 if (is_column_nullable(array_column.get_data())) {
166 const auto& nested_null_column =
167 reinterpret_cast<const ColumnNullable&>(array_column.get_data());
168 nested_column = nested_null_column.get_nested_column_ptr();
169 nested_null_map = nested_null_column.get_null_map_column_ptr();
170 } else {
171 nested_column = array_column.get_data_ptr();
172 }
173
174 ColumnPtr res = nullptr;
175 auto left_element_type =
176 remove_nullable(assert_cast<const DataTypeArray&>(*arg.type).get_nested_type());
177 if (check_column<ColumnUInt8>(*nested_column)) {
178 res = _execute_number_expanded<UInt8, Int16>(offsets, *nested_column, nested_null_map);
179 } else if (check_column<ColumnInt8>(*nested_column)) {
180 res = _execute_number_expanded<Int8, Int16>(offsets, *nested_column, nested_null_map);
181 } else if (check_column<ColumnInt16>(*nested_column)) {
182 res = _execute_number_expanded<Int16, Int32>(offsets, *nested_column, nested_null_map);
183 } else if (check_column<ColumnInt32>(*nested_column)) {
184 res = _execute_number_expanded<Int32, Int64>(offsets, *nested_column, nested_null_map);
185 } else if (check_column<ColumnInt64>(*nested_column)) {
186 res = _execute_number_expanded<Int64, Int128>(offsets, *nested_column, nested_null_map);
187 } else if (check_column<ColumnInt128>(*nested_column)) {
188 res = _execute_number_expanded<Int128, Int128>(offsets, *nested_column,
189 nested_null_map);
190 } else if (check_column<ColumnFloat32>(*nested_column)) {
191 res = _execute_number_expanded<Float32, Float64>(offsets, *nested_column,
192 nested_null_map);
193 } else if (check_column<ColumnFloat64>(*nested_column)) {
194 res = _execute_number_expanded<Float64, Float64>(offsets, *nested_column,
195 nested_null_map);
196 } else if (check_column<ColumnDecimal32>(*nested_column)) {
197 res = _execute_number_expanded<Decimal32, Decimal32>(offsets, *nested_column,
198 nested_null_map);
199 } else if (check_column<ColumnDecimal64>(*nested_column)) {
200 res = _execute_number_expanded<Decimal64, Decimal64>(offsets, *nested_column,
201 nested_null_map);
202 } else if (check_column<ColumnDecimal128>(*nested_column)) {
203 res = _execute_number_expanded<Decimal128, Decimal128>(offsets, *nested_column,
204 nested_null_map);
205 }
206 return ColumnArray::create(std::move(res), array_column.get_offsets_ptr());
207 }
