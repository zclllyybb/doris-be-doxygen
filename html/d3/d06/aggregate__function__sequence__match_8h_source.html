37template <template <typename> class Comparator>
38struct ComparePairFirst final {
39 template <typename T1, typename T2>
40 bool operator()(const std::pair<T1, T2>& lhs, const std::pair<T1, T2>& rhs) const {
41 return Comparator<T1> {}(lhs.first, rhs.first);
42 }
43};

50template <typename DateValueType, typename NativeType, typename Derived>
51struct AggregateFunctionSequenceMatchData final {
52 using Timestamp = DateValueType;
53 using Events = std::bitset<MAX_EVENTS>;
54 using TimestampEvents = std::pair<Timestamp, Events>;
55 using Comparator = ComparePairFirst<std::less>;
56
57 AggregateFunctionSequenceMatchData() { reset(); }
58
59public:
60 const std::string get_pattern() const { return pattern; }
61
62 size_t get_arg_count() const { return arg_count; }
63
64 void init(const std::string pattern, size_t arg_count) {
65 if (!init_flag) {
66 this->pattern = pattern;
67 this->arg_count = arg_count;
68 parse_pattern();
69 init_flag = true;
70 }
71 }
72
73 void reset() {
74 sorted = true;
75 init_flag = false;
76 pattern_has_time = false;
77 pattern = "";
78 arg_count = 0;
79 conditions_met.reset();
80 conditions_in_pattern.reset();
81
82 events_list.clear();
83 actions.clear();
84 dfa_states.clear();
85 }
86
87 void add(const Timestamp& timestamp, const Events& events) {
89 if (events.any()) {
90 events_list.emplace_back(timestamp, events);
91 sorted = false;
92 conditions_met |= events;
93 }
94 }
95
96 void merge(const AggregateFunctionSequenceMatchData& other) {
97 if (other.events_list.empty()) return;
98
99 events_list.insert(std::begin(other.events_list), std::end(other.events_list));
100 sorted = false;
101 conditions_met |= other.conditions_met;
102 }
103
104 void sort() {
105 if (sorted) return;
106
107 std::sort(std::begin(events_list), std::end(events_list), Comparator {});
108 sorted = true;
109 }
110
111 void write(BufferWritable& buf) const {
112 write_binary(sorted, buf);
113 write_binary(events_list.size(), buf);
114
115 for (const auto& events : events_list) {
116 write_binary(events.first, buf);
117 write_binary(events.second.to_ulong(), buf);
118 }
119
120 UInt32 conditions_met_value = conditions_met.to_ulong();
121 write_binary(conditions_met_value, buf);
122
123 write_binary(pattern, buf);
124 write_binary(arg_count, buf);
125 }
126
127 void read(BufferReadable& buf) {
128 read_binary(sorted, buf);
129
130 size_t events_list_size;
131 read_binary(events_list_size, buf);
132
133 events_list.clear();
134 events_list.reserve(events_list_size);
135
136 for (size_t i = 0; i < events_list_size; ++i) {
137 Timestamp timestamp;
138 read_binary(timestamp, buf);
139
140 UInt64 events;
141 read_binary(events, buf);
142
143 events_list.emplace_back(timestamp, Events {events});
144 }
145
146 UInt32 conditions_met_value;
147 read_binary(conditions_met_value, buf);
148 conditions_met = conditions_met_value;
149
150 read_binary(pattern, buf);
151 read_binary(arg_count, buf);
152 }

166 struct PatternAction final {
167 PatternActionType type;
168 std::uint64_t extra;
169
170 PatternAction() = default;
171 explicit PatternAction(const PatternActionType type_, const std::uint64_t extra_ = 0)
172 : type {type_}, extra {extra_} {}
173 };
174
175 using PatternActions = PODArrayWithStackMemory<PatternAction, 64>;
176
177 Derived& derived() { return static_cast<Derived&>(*this); }
178
179 void parse_pattern() {
180 actions.clear();
181 actions.emplace_back(PatternActionType::KleeneStar);
182
183 dfa_states.clear();
184 dfa_states.emplace_back(true);
185
186 pattern_has_time = false;
187
188 const char* pos = pattern.data();
189 const char* begin = pos;
190 const char* end = pos + pattern.size();
191
192 // Pattern is checked in fe, so pattern should be valid here, we check it and if pattern is invalid, we return.
193 auto throw_exception = [&](const std::string& msg) {
194 LOG(WARNING) << msg + " '" + std::string(pos, end) + "' at position " +
195 std::to_string(pos - begin);
196 };
197
198 auto match = [&pos, end](const char* str) mutable {
199 size_t length = strlen(str);
200 if (pos + length <= end && 0 == memcmp(pos, str, length)) {
201 pos += length;
202 return true;
203 }
204 return false;
205 };
206
207 while (pos < end) {
208 if (match("(?")) {
209 if (match("t")) {
210 PatternActionType type;
211
212 if (match("<="))
213 type = PatternActionType::TimeLessOrEqual;
214 else if (match("<"))
215 type = PatternActionType::TimeLess;
216 else if (match(">="))
217 type = PatternActionType::TimeGreaterOrEqual;
218 else if (match(">"))
219 type = PatternActionType::TimeGreater;
220 else if (match("=="))
221 type = PatternActionType::TimeEqual;
222 else {
223 throw_exception("Unknown time condition");
224 return;
225 }
226
227 NativeType duration = 0;
228 const auto* prev_pos = pos;
229 pos = try_read_first_int_text(duration, pos, end);
230 if (pos == prev_pos) {
231 throw_exception("Could not parse number");
232 return;
233 }
234
235 if (actions.back().type != PatternActionType::SpecificEvent &&
236 actions.back().type != PatternActionType::AnyEvent &&
237 actions.back().type != PatternActionType::KleeneStar) {
238 throw_exception(
239 "Temporal condition should be preceded by an event condition");
240 return;
241 }
242
243 pattern_has_time = true;
244 actions.emplace_back(type, duration);
245 } else {
246 UInt64 event_number = 0;
247 const auto* prev_pos = pos;
248 pos = try_read_first_int_text(event_number, pos, end);
249 if (pos == prev_pos) throw_exception("Could not parse number");
250
251 if (event_number > arg_count - 1) {
252 throw_exception("Event number " + std::to_string(event_number) +
253 " is out of range");
254 return;
255 }
256
257 actions.emplace_back(PatternActionType::SpecificEvent, event_number - 1);
258 dfa_states.back().transition = DFATransition::SpecificEvent;
259 dfa_states.back().event = static_cast<uint32_t>(event_number - 1);
260 dfa_states.emplace_back();
261 conditions_in_pattern.set(event_number - 1);
262 }
263
264 if (!match(")")) {
265 throw_exception("Expected closing parenthesis, found");
266 return;
267 }
268
269 } else if (match(".*")) {
270 actions.emplace_back(PatternActionType::KleeneStar);
271 dfa_states.back().has_kleene = true;
272 } else if (match(".")) {
273 actions.emplace_back(PatternActionType::AnyEvent);
274 dfa_states.back().transition = DFATransition::AnyEvent;
275 dfa_states.emplace_back();
276 } else {
277 throw_exception("Could not parse pattern, unexpected starting symbol");
278 return;
279 }
280 }
281 }

292 template <typename EventEntry>
293 bool dfa_match(EventEntry& events_it, const EventEntry events_end) const {
294 using ActiveStates = std::vector<bool>;
297 ActiveStates active_states(dfa_states.size(), false);
298 ActiveStates next_active_states(dfa_states.size(), false);
299 active_states[0] = true;
300
303 size_t n_active = 1;
304
305 for (/* empty */; events_it != events_end && n_active > 0 && !active_states.back();
306 ++events_it) {
307 n_active = 0;
308 next_active_states.assign(dfa_states.size(), false);
309
310 for (size_t state = 0; state < dfa_states.size(); ++state) {
311 if (!active_states[state]) {
312 continue;
313 }
314
315 switch (dfa_states[state].transition) {
316 case DFATransition::None:
317 break;
318 case DFATransition::AnyEvent:
319 next_active_states[state + 1] = true;
320 ++n_active;
321 break;
322 case DFATransition::SpecificEvent:
323 if (events_it->second.test(dfa_states[state].event)) {
324 next_active_states[state + 1] = true;
325 ++n_active;
326 }
327 break;
328 }
329
330 if (dfa_states[state].has_kleene) {
331 next_active_states[state] = true;
332 ++n_active;
333 }
334 }
335 swap(active_states, next_active_states);
336 }
337
338 return active_states.back();
339 }

341 template <typename EventEntry>
342 bool backtracking_match(EventEntry& events_it, const EventEntry events_end) const {
343 const auto action_begin = std::begin(actions);
344 const auto action_end = std::end(actions);
345 auto action_it = action_begin;
346
347 const auto events_begin = events_it;
348 auto base_it = events_it;
349
351 using backtrack_info = std::tuple<decltype(action_it), EventEntry, EventEntry>;
352 std::stack<backtrack_info> back_stack;
353
355 const auto do_backtrack = [&] {
356 while (!back_stack.empty()) {
357 auto& top = back_stack.top();
358
359 action_it = std::get<0>(top);
360 events_it = std::next(std::get<1>(top));
361 base_it = std::get<2>(top);
362
363 back_stack.pop();
364
365 if (events_it != events_end) return true;
366 }
367
368 return false;
369 };
370
371 size_t i = 0;
372 while (action_it != action_end && events_it != events_end) {
373 if (action_it->type == PatternActionType::SpecificEvent) {
374 if (events_it->second.test(action_it->extra)) {
376 base_it = events_it;
377 ++action_it, ++events_it;
378 } else if (!do_backtrack())
380 break;
381 } else if (action_it->type == PatternActionType::AnyEvent) {
382 base_it = events_it;
383 ++action_it, ++events_it;
384 } else if (action_it->type == PatternActionType::KleeneStar) {
385 back_stack.emplace(action_it, events_it, base_it);
386 base_it = events_it;
387 ++action_it;
388 } else if (action_it->type == PatternActionType::TimeLessOrEqual) {
389 if (events_it->first.second_diff(base_it->first) <= action_it->extra) {
391 back_stack.emplace(action_it, events_it, base_it);
392 base_it = events_it;
393 ++action_it;
394 } else if (!do_backtrack())
395 break;
396 } else if (action_it->type == PatternActionType::TimeLess) {
397 if (events_it->first.second_diff(base_it->first) < action_it->extra) {
398 back_stack.emplace(action_it, events_it, base_it);
399 base_it = events_it;
400 ++action_it;
401 } else if (!do_backtrack())
402 break;
403 } else if (action_it->type == PatternActionType::TimeGreaterOrEqual) {
404 if (events_it->first.second_diff(base_it->first) >= action_it->extra) {
405 back_stack.emplace(action_it, events_it, base_it);
406 base_it = events_it;
407 ++action_it;
408 } else if (++events_it == events_end && !do_backtrack())
409 break;
410 } else if (action_it->type == PatternActionType::TimeGreater) {
411 if (events_it->first.second_diff(base_it->first) > action_it->extra) {
412 back_stack.emplace(action_it, events_it, base_it);
413 base_it = events_it;
414 ++action_it;
415 } else if (++events_it == events_end && !do_backtrack())
416 break;
417 } else if (action_it->type == PatternActionType::TimeEqual) {
418 if (events_it->first.second_diff(base_it->first) == action_it->extra) {
419 back_stack.emplace(action_it, events_it, base_it);
420 base_it = events_it;
421 ++action_it;
422 } else if (++events_it == events_end && !do_backtrack())
423 break;
424 } else {
425 LOG(WARNING) << "Unknown PatternActionType";
426 return false;
427 }
428
429 if (++i > sequence_match_max_iterations) {
430 LOG(WARNING)
431 << "Pattern application proves too difficult, exceeding max iterations (" +
432 std::to_string(sequence_match_max_iterations) + ")";
433 return false;
434 }
435 }
436
438 if (action_it != action_end) {
440 while (action_it->type == PatternActionType::KleeneStar ||
441 action_it->type == PatternActionType::TimeLessOrEqual ||
442 action_it->type == PatternActionType::TimeLess ||
443 (action_it->type == PatternActionType::TimeGreaterOrEqual &&
444 action_it->extra == 0))
445 ++action_it;
446 }
447
448 if (events_it == events_begin) ++events_it;
449
450 return action_it == action_end;
451 }

457 template <typename EventEntry>
458 bool could_match_deterministic_parts(const EventEntry events_begin, const EventEntry events_end,
459 bool limit_iterations = true) const {
460 size_t events_processed = 0;
461 auto events_it = events_begin;
462
463 const auto actions_end = std::end(actions);
464 auto actions_it = std::begin(actions);
465 auto det_part_begin = actions_it;
466
467 auto match_deterministic_part = [&events_it, events_end, &events_processed, det_part_begin,
468 actions_it, limit_iterations]() {
469 auto events_it_init = events_it;
470 auto det_part_it = det_part_begin;
471
472 while (det_part_it != actions_it && events_it != events_end) {
474 if (det_part_it->type == PatternActionType::AnyEvent) ++events_it, ++det_part_it;
475
477 else {
478 if (events_it->second.test(det_part_it->extra)) ++events_it, ++det_part_it;
479
481 else {
482 events_it = ++events_it_init;
483 det_part_it = det_part_begin;
484 }
485 }
486
487 if (limit_iterations && ++events_processed > sequence_match_max_iterations) {
488 LOG(WARNING) << "Pattern application proves too difficult, exceeding max "
489 "iterations are " +
490 std::to_string(sequence_match_max_iterations);
491 return false;
492 }
493 }
494
495 return det_part_it == actions_it;
496 };
497
498 for (; actions_it != actions_end; ++actions_it)
499 if (actions_it->type != PatternActionType::SpecificEvent &&
500 actions_it->type != PatternActionType::AnyEvent) {
501 if (!match_deterministic_part()) return false;
502 det_part_begin = std::next(actions_it);
503 }
504
505 return match_deterministic_part();
506 }

619class AggregateFunctionSequenceMatch final
620 : public AggregateFunctionSequenceBase<
621 DateValueType, NativeType,
622 AggregateFunctionSequenceMatch<DateValueType, NativeType>> {
623public:
624 AggregateFunctionSequenceMatch(const DataTypes& arguments, const String& pattern_)
625 : AggregateFunctionSequenceBase<
626 DateValueType, NativeType,
627 AggregateFunctionSequenceMatch<DateValueType, NativeType>>(arguments,
628 pattern_) {}
629
630 using AggregateFunctionSequenceBase<DateValueType, NativeType,
631 AggregateFunctionSequenceMatch<DateValueType, NativeType>>::
632 AggregateFunctionSequenceBase;
633
634 String get_name() const override { return "sequence_match"; }
635
636 DataTypePtr get_return_type() const override { return std::make_shared<DataTypeUInt8>(); }
637
638 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
639 auto& output = assert_cast<ColumnUInt8&>(to).get_data();
640 if (!this->data(place).conditions_in_pattern.any()) {
641 output.push_back(false);
642 return;
643 }
644
645 if ((this->data(place).conditions_in_pattern & this->data(place).conditions_met) !=
646 this->data(place).conditions_in_pattern) {
647 output.push_back(false);
648 return;
649 }
650 this->data(const_cast<AggregateDataPtr>(place)).sort();
651
652 const auto& data_ref = this->data(place);
653
654 const auto events_begin = std::begin(data_ref.events_list);
655 const auto events_end = std::end(data_ref.events_list);
656 auto events_it = events_begin;
657
658 bool match = (this->data(place).pattern_has_time
659 ? (this->data(place).could_match_deterministic_parts(events_begin,
660 events_end) &&
661 this->data(place).backtracking_match(events_it, events_end))
662 : this->data(place).dfa_match(events_it, events_end));
663 output.push_back(match);
664 }

687 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
688 auto& output = assert_cast<ColumnInt64&>(to).get_data();
689 if (!this->data(place).conditions_in_pattern.any()) {
690 output.push_back(0);
691 return;
692 }
693
694 if ((this->data(place).conditions_in_pattern & this->data(place).conditions_met) !=
695 this->data(place).conditions_in_pattern) {
696 output.push_back(0);
697 return;
698 }
699 this->data(const_cast<AggregateDataPtr>(place)).sort();
700 output.push_back(count(place));
701 }
702
703private:
704 UInt64 count(ConstAggregateDataPtr __restrict place) const {
705 const auto& data_ref = this->data(place);
706
707 const auto events_begin = std::begin(data_ref.events_list);
708 const auto events_end = std::end(data_ref.events_list);
709 auto events_it = events_begin;
710
711 size_t count = 0;
712 // check if there is a chance of matching the sequence at least once
713 if (this->data(place).could_match_deterministic_parts(events_begin, events_end)) {
714 while (events_it != events_end &&
715 this->data(place).backtracking_match(events_it, events_end))
716 ++count;
717 }
718
719 return count;
720 }
