56template <typename T>
57constexpr auto get_bitmap_size() {
58 return (sizeof(T) == 32)
59 ? 4: (sizeof(T) == 16)
60 ? 2: ((sizeof(T) == 8)
61 ? 1: ((sizeof(T) == 4)
62 ? 1: ((sizeof(T) == 2)

74template <typename T, bool has_low_cardinality = false>
75T pack_fixed(size_t i, size_t keys_size, const ColumnRawPtrs& key_columns, const Sizes& key_sizes,
76 const ColumnRawPtrs* low_cardinality_positions [[maybe_unused]] = nullptr,
77 const Sizes* low_cardinality_sizes [[maybe_unused]] = nullptr) {
78 union {
79 T key;
80 char bytes[sizeof(key)] = {};
81 };
82
83 size_t offset = 0;
84
85 for (size_t j = 0; j < keys_size; ++j) {
86 size_t index = i;
87 const IColumn* column = key_columns[j];
88 if constexpr (has_low_cardinality) {
89 if (const IColumn* positions = (*low_cardinality_positions)[j]) {
90 switch ((*low_cardinality_sizes)[j]) {
91 case sizeof(UInt8):
92 index = assert_cast<const ColumnUInt8*>(positions)->get_element(i);
93 break;
94 case sizeof(UInt16):
95 index = assert_cast<const ColumnUInt16*>(positions)->get_element(i);
96 break;
97 case sizeof(UInt32):
98 index = assert_cast<const ColumnUInt32*>(positions)->get_element(i);
99 break;
100 case sizeof(UInt64):
101 index = assert_cast<const ColumnUInt64*>(positions)->get_element(i);
102 break;
103 default:
104 LOG(FATAL) << "Unexpected size of index type for low cardinality column.";
105 }
106 }
107 }
108
109 switch (key_sizes[j]) {
110 case 1:
111 memcpy(bytes + offset,
112 static_cast<const ColumnVectorHelper*>(column)->get_raw_data_begin<1>() + index,
113 1);
114 offset += 1;
115 break;
116 case 2:
117 memcpy(bytes + offset,
118 static_cast<const ColumnVectorHelper*>(column)->get_raw_data_begin<2>() +
119 index * 2,
120 2);
121 offset += 2;
122 break;
123 case 4:
124 memcpy(bytes + offset,
125 static_cast<const ColumnVectorHelper*>(column)->get_raw_data_begin<4>() +
126 index * 4,
127 4);
128 offset += 4;
129 break;
130 case 8:
131 memcpy(bytes + offset,
132 static_cast<const ColumnVectorHelper*>(column)->get_raw_data_begin<8>() +
133 index * 8,
134 8);
135 offset += 8;
136 break;
137 default:
138 memcpy(bytes + offset,
139 static_cast<const ColumnVectorHelper*>(column)->get_raw_data_begin<1>() +
140 index * key_sizes[j],
141 key_sizes[j]);
142 offset += key_sizes[j];
143 }
144 }
145
146 return key;
147}

150template <typename T>
151T pack_fixed(size_t i, size_t keys_size, const ColumnRawPtrs& key_columns, const Sizes& key_sizes,
152 const KeysNullMap<T>& bitmap) {
153 union {
154 T key;
155 char bytes[sizeof(key)] = {};
156 };
157
158 size_t offset = 0;
159
160 static constexpr auto bitmap_size = std::tuple_size<KeysNullMap<T>>::value;
161 static constexpr bool has_bitmap = bitmap_size > 0;
162
163 if (has_bitmap) {
164 memcpy(bytes + offset, bitmap.data(), bitmap_size * sizeof(UInt8));
165 offset += bitmap_size;
166 }
167
168 for (size_t j = 0; j < keys_size; ++j) {
169 bool is_null;
170
171 if (!has_bitmap)
172 is_null = false;
173 else {
174 size_t bucket = j / 8;
175 size_t off = j % 8;
176 is_null = ((bitmap[bucket] >> off) & 1) == 1;
177 }
178
179 if (is_null) continue;
180
181 switch (key_sizes[j]) {
182 case 1:
183 memcpy(bytes + offset,
184 static_cast<const ColumnVectorHelper*>(key_columns[j])->get_raw_data_begin<1>() +
185 i,
186 1);
187 offset += 1;
188 break;
189 case 2:
190 memcpy(bytes + offset,
191 static_cast<const ColumnVectorHelper*>(key_columns[j])->get_raw_data_begin<2>() +
192 i * 2,
193 2);
194 offset += 2;
195 break;
196 case 4:
197 memcpy(bytes + offset,
198 static_cast<const ColumnVectorHelper*>(key_columns[j])->get_raw_data_begin<4>() +
199 i * 4,
200 4);
201 offset += 4;
202 break;
203 case 8:
204 memcpy(bytes + offset,
205 static_cast<const ColumnVectorHelper*>(key_columns[j])->get_raw_data_begin<8>() +
206 i * 8,
207 8);
208 offset += 8;
209 break;
210 default:
211 memcpy(bytes + offset,
212 static_cast<const ColumnVectorHelper*>(key_columns[j])->get_raw_data_begin<1>() +
213 i * key_sizes[j],
214 key_sizes[j]);
215 offset += key_sizes[j];
216 }
217 }
218
219 return key;
220}

235inline StringRef* place_keys_in_pool(size_t keys_size, StringRefs& keys, Arena& pool) {
236 for (size_t j = 0; j < keys_size; ++j) {
237 char* place = pool.alloc(keys[j].size);
238 memcpy_small_allow_read_write_overflow15(place, keys[j].data, keys[j].size);
239 keys[j].data = place;
240 }
241
243 char* res = pool.aligned_alloc(keys_size * sizeof(StringRef), alignof(StringRef));
244 memcpy_small_allow_read_write_overflow15(res, keys.data(), keys_size * sizeof(StringRef));
245
246 return reinterpret_cast<StringRef*>(res);
247}
