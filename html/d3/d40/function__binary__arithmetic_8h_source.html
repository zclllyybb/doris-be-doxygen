64template <template <typename, typename> typename Operation>
65struct OperationTraits {
66 using T = UInt8;
67 using Op = Operation<T, T>;
68 static constexpr bool is_plus_minus =
69 std::is_same_v<Op, PlusImpl<T, T>> || std::is_same_v<Op, MinusImpl<T, T>>;
70 static constexpr bool is_multiply = std::is_same_v<Op, MultiplyImpl<T, T>>;
71 static constexpr bool is_division = std::is_same_v<Op, DivideFloatingImpl<T, T>> ||
72 std::is_same_v<Op, DivideIntegralImpl<T, T>>;
73 static constexpr bool is_mod = std::is_same_v<Op, ModuloImpl<T, T>>;
74 static constexpr bool allow_decimal =
75 std::is_same_v<Op, PlusImpl<T, T>> || std::is_same_v<Op, MinusImpl<T, T>> ||
76 std::is_same_v<Op, MultiplyImpl<T, T>> || std::is_same_v<Op, ModuloImpl<T, T>> ||
77 std::is_same_v<Op, DivideFloatingImpl<T, T>> ||
78 std::is_same_v<Op, DivideIntegralImpl<T, T>>;
79 static constexpr bool can_overflow = is_plus_minus || is_multiply;
80 static constexpr bool has_variadic_argument =
81 !std::is_void_v<decltype(has_variadic_argument_types(std::declval<Op>()))>;
82};

219template <typename A, typename B, template <typename, typename> typename Operation,
220 typename ResultType, bool is_to_null_type, bool return_nullable_type,
221 bool check_overflow = true>
222struct DecimalBinaryOperation {
223 using OpTraits = OperationTraits<Operation>;
224
225 using NativeResultType = typename NativeType<ResultType>::Type;
226 using Op = Operation<NativeResultType, NativeResultType>;
227
228 using Traits = NumberTraits::BinaryOperatorTraits<A, B>;
229 using ArrayC = typename ColumnDecimal<ResultType>::Container;
230
231 static void vector_vector(const typename Traits::ArrayA& a, const typename Traits::ArrayB& b,
232 ArrayC& c) {
233 size_t size = a.size();
234
235 if constexpr (OpTraits::is_multiply && IsDecimalV2<A> && IsDecimalV2<B> &&
236 IsDecimalV2<ResultType>) {
237 Op::vector_vector(a, b, c);
238 } else {
239 for (size_t i = 0; i < size; i++) {
240 c[i] = apply(a[i], b[i]);
241 }
242 }
243 }
244
246 static void vector_vector(const typename Traits::ArrayA& a, const typename Traits::ArrayB& b,
247 ArrayC& c, NullMap& null_map) {
248 size_t size = a.size();
249 if constexpr (IsDecimalV2<B> || IsDecimalV2<A>) {
251 for (size_t i = 0; i < size; ++i) {
252 c[i] = apply(a[i], b[i], null_map[i]);
253 }
254 } else if constexpr (OpTraits::is_division && (IsDecimalNumber<B> || IsDecimalNumber<A>)) {
255 for (size_t i = 0; i < size; ++i) {
256 c[i] = apply_scaled_div(a[i], b[i], null_map[i]);
257 }
258 } else if constexpr ((OpTraits::is_multiply || OpTraits::is_plus_minus) &&
259 (IsDecimalNumber<B> || IsDecimalNumber<A>)) {
260 for (size_t i = 0; i < size; ++i) {
261 null_map[i] = apply_op_safely(a[i], b[i], c[i].value);
262 }
263 }
264 }
265
266 static void vector_constant(const typename Traits::ArrayA& a, B b, ArrayC& c) {
267 size_t size = a.size();
268 if constexpr (OpTraits::is_division && IsDecimalNumber<B>) {
269 for (size_t i = 0; i < size; ++i) {
270 c[i] = apply_scaled_div(a[i], b);
271 }
272 return;
273 }
274
276 for (size_t i = 0; i < size; ++i) {
277 c[i] = apply(a[i], b);
278 }
279 }
280
281 static void vector_constant(const typename Traits::ArrayA& a, B b, ArrayC& c,
282 NullMap& null_map) {
283 size_t size = a.size();
284 if constexpr (OpTraits::is_division && IsDecimalNumber<B>) {
285 for (size_t i = 0; i < size; ++i) {
286 c[i] = apply_scaled_div(a[i], b, null_map[i]);
287 }
288 } else if constexpr ((OpTraits::is_multiply || OpTraits::is_plus_minus) &&
289 (IsDecimalNumber<B> || IsDecimalNumber<A>)) {
290 for (size_t i = 0; i < size; ++i) {
291 null_map[i] = apply_op_safely(a[i], b, c[i].value);
292 }
293 } else {
294 for (size_t i = 0; i < size; ++i) {
295 c[i] = apply(a[i], b, null_map[i]);
296 }
297 }
298 }
299
300 static void constant_vector(A a, const typename Traits::ArrayB& b, ArrayC& c) {
301 size_t size = b.size();
302 if constexpr (IsDecimalV2<A> || IsDecimalV2<B>) {
303 DecimalV2Value da(a);
304 for (size_t i = 0; i < size; ++i) {
305 c[i] = Op::template apply(da, DecimalV2Value(b[i])).value();
306 }
307 } else {
308 for (size_t i = 0; i < size; ++i) {
309 c[i] = apply(a, b[i]);
310 }
311 }
312 }
313
314 static void constant_vector(A a, const typename Traits::ArrayB& b, ArrayC& c,
315 NullMap& null_map) {
316 size_t size = b.size();
317 if constexpr (OpTraits::is_division && IsDecimalNumber<B>) {
318 for (size_t i = 0; i < size; ++i) {
319 c[i] = apply_scaled_div(a, b[i], null_map[i]);
320 }
321 } else if constexpr ((OpTraits::is_multiply || OpTraits::is_plus_minus) &&
322 (IsDecimalNumber<B> || IsDecimalNumber<A>)) {
323 for (size_t i = 0; i < size; ++i) {
324 null_map[i] = apply_op_safely(a, b[i], c[i].value);
325 }
326 } else {
327 for (size_t i = 0; i < size; ++i) {
328 c[i] = apply(a, b[i], null_map[i]);
329 }
330 }
331 }
332
333 static ResultType constant_constant(A a, B b) { return apply(a, b); }
334
335 static ResultType constant_constant(A a, B b, UInt8& is_null) {
336 if constexpr (OpTraits::is_division && IsDecimalNumber<B>) {
337 return apply_scaled_div(a, b, is_null);
338 } else if constexpr ((OpTraits::is_multiply || OpTraits::is_plus_minus) &&
339 (IsDecimalNumber<B> || IsDecimalNumber<A>)) {
340 NativeResultType res;
341 is_null = apply_op_safely(a, b, res);
342 return res;
343 } else {
344 return apply(a, b, is_null);
345 }
346 }

442private:
444 static NativeResultType apply(NativeResultType a, NativeResultType b) {
445 if constexpr (IsDecimalV2<B> || IsDecimalV2<A>) {
446 // Now, Doris only support decimal +-*/ decimal.
447 // overflow in consider in operator
448 return Op::template apply(DecimalV2Value(a), DecimalV2Value(b)).value();
449 } else {
450 if constexpr (OpTraits::can_overflow && check_overflow) {
451 NativeResultType res;
452 // TODO handle overflow gracefully
453 if (Op::template apply<NativeResultType>(a, b, res)) {
454 LOG(WARNING) << "Decimal math overflow";
455 res = max_decimal_value<ResultType>();
456 }
457 return res;
458 } else {
459 return Op::template apply<NativeResultType>(a, b);
460 }
461 }
462 }
463
465 static NativeResultType apply(NativeResultType a, NativeResultType b, UInt8& is_null) {
466 if constexpr (IsDecimalV2<B> || IsDecimalV2<A>) {
467 DecimalV2Value l(a);
468 DecimalV2Value r(b);
469 auto ans = Op::template apply(l, r, is_null);
470 NativeResultType result;
471 memcpy(&result, &ans, std::min(sizeof(result), sizeof(ans)));
472 return result;
473 } else {
474 return Op::template apply<NativeResultType>(a, b, is_null);
475 }
476 }
477
478 static NativeResultType apply_scaled(NativeResultType a, NativeResultType b) {
479 if constexpr (OpTraits::is_plus_minus) {
480 NativeResultType res;
481
482 if constexpr (check_overflow) {
483 bool overflow = false;
484
485 if constexpr (OpTraits::can_overflow) {
486 overflow |= Op::template apply<NativeResultType>(a, b, res);
487 } else {
488 res = Op::template apply<NativeResultType>(a, b);
489 }
490
491 // TODO handle overflow gracefully
492 if (overflow) {
493 LOG(WARNING) << "Decimal math overflow";
494 res = max_decimal_value<ResultType>();
495 }
496 } else {
497 res = apply(a, b);
498 }
499
500 return res;
501 }
502 }
503
504 static NativeResultType apply_scaled_div(NativeResultType a, NativeResultType b,
505 UInt8& is_null) {
506 if constexpr (OpTraits::is_division) {
507 return apply(a, b, is_null);
508 }
509 }
510
511 static NativeResultType apply_scaled_mod(NativeResultType a, NativeResultType b,
512 UInt8& is_null) {
513 return apply(a, b, is_null);
514 }
515
516 static UInt8 apply_op_safely(NativeResultType a, NativeResultType b, NativeResultType& c) {
517 if constexpr (OpTraits::is_multiply || OpTraits::is_plus_minus) {
518 return Op::template apply(a, b, c);
519 }
520 }

565template <template <typename, typename> class Operation, typename LeftDataType,
566 typename RightDataType>
567struct BinaryOperationTraits {
568 using Op = Operation<typename LeftDataType::FieldType, typename RightDataType::FieldType>;
569 using OpTraits = OperationTraits<Operation>;
572 using ResultDataType = Switch<
574 Case<!OpTraits::allow_decimal &&
575 (IsDataTypeDecimal<LeftDataType> || IsDataTypeDecimal<RightDataType>),
576 InvalidType>,
577 Case<(IsDataTypeDecimalV2<LeftDataType> && IsDataTypeDecimal<RightDataType> &&
578 !IsDataTypeDecimalV2<RightDataType>) ||
579 (IsDataTypeDecimalV2<RightDataType> && IsDataTypeDecimal<LeftDataType> &&
580 !IsDataTypeDecimalV2<LeftDataType>),
581 InvalidType>,
582 Case<IsDataTypeDecimal<LeftDataType> && IsDataTypeDecimal<RightDataType> &&
583 UseLeftDecimal<LeftDataType, RightDataType>,
584 LeftDataType>,
585 Case<IsDataTypeDecimal<LeftDataType> && IsDataTypeDecimal<RightDataType>,
586 RightDataType>,
587 Case<IsDataTypeDecimal<LeftDataType> && !IsDataTypeDecimal<RightDataType> &&
588 IsIntegral<RightDataType>,
589 LeftDataType>,
590 Case<!IsDataTypeDecimal<LeftDataType> && IsDataTypeDecimal<RightDataType> &&

605template <typename LeftDataType, typename RightDataType, typename ExpectedResultDataType,
606 template <typename, typename> class Operation, bool is_to_null_type,
607 bool return_nullable_type>
608struct ConstOrVectorAdapter {
609 static constexpr bool result_is_decimal =
610 IsDataTypeDecimal<LeftDataType> || IsDataTypeDecimal<RightDataType>;
611
612 using ResultDataType = ExpectedResultDataType;
613 using ResultType = typename ResultDataType::FieldType;
614 using A = typename LeftDataType::FieldType;
615 using B = typename RightDataType::FieldType;
616
617 using OperationImpl = std::conditional_t<
618 IsDataTypeDecimal<ResultDataType>,
619 DecimalBinaryOperation<A, B, Operation, ResultType, is_to_null_type,
620 return_nullable_type>,
621 BinaryOperationImpl<A, B, Operation<A, B>, is_to_null_type, ResultType>>;
622
623 static ColumnPtr execute(ColumnPtr column_left, ColumnPtr column_right,
624 const LeftDataType& type_left, const RightDataType& type_right,
625 DataTypePtr res_data_type) {
626 bool is_const_left = is_column_const(*column_left);
627 bool is_const_right = is_column_const(*column_right);
628
629 if (is_const_left && is_const_right) {
630 return constant_constant(column_left, column_right, type_left, type_right,
631 res_data_type);
632 } else if (is_const_left) {
633 return constant_vector(column_left, column_right, type_left, type_right, res_data_type);
634 } else if (is_const_right) {
635 return vector_constant(column_left, column_right, type_left, type_right, res_data_type);
636 } else {
637 return vector_vector(column_left, column_right, type_left, type_right, res_data_type);
638 }
639 }
640
641private:
642 static ColumnPtr constant_constant(ColumnPtr column_left, ColumnPtr column_right,
643 const LeftDataType& type_left,
644 const RightDataType& type_right, DataTypePtr res_data_type) {
645 auto column_left_ptr = check_and_get_column<ColumnConst>(column_left);
646 auto column_right_ptr = check_and_get_column<ColumnConst>(column_right);
647 DCHECK(column_left_ptr != nullptr && column_right_ptr != nullptr);
648
649 ColumnPtr column_result = nullptr;
650
651 if constexpr (result_is_decimal) {
652 column_result = OperationImpl::adapt_decimal_constant_constant(
653 column_left_ptr->template get_value<A>(),
654 column_right_ptr->template get_value<B>(), res_data_type);
655
656 } else {
657 column_result = OperationImpl::adapt_normal_constant_constant(
658 column_left_ptr->template get_value<A>(),
659 column_right_ptr->template get_value<B>());
660 }
661
662 return ColumnConst::create(std::move(column_result), column_left->size());
663 }
664
665 static ColumnPtr vector_constant(ColumnPtr column_left, ColumnPtr column_right,
666 const LeftDataType& type_left, const RightDataType& type_right,
667 DataTypePtr res_data_type) {
668 auto column_right_ptr = check_and_get_column<ColumnConst>(column_right);
669 DCHECK(column_right_ptr != nullptr);
670
671 if constexpr (result_is_decimal) {
672 return OperationImpl::adapt_decimal_vector_constant(
673 column_left->get_ptr(), column_right_ptr->template get_value<B>(),
674 res_data_type);
675 } else {
676 return OperationImpl::adapt_normal_vector_constant(
677 column_left->get_ptr(), column_right_ptr->template get_value<B>());
678 }
679 }
680
681 static ColumnPtr constant_vector(ColumnPtr column_left, ColumnPtr column_right,
682 const LeftDataType& type_left, const RightDataType& type_right,
683 DataTypePtr res_data_type) {
684 auto column_left_ptr = check_and_get_column<ColumnConst>(column_left);
685 DCHECK(column_left_ptr != nullptr);
686
687 if constexpr (result_is_decimal) {
688 return OperationImpl::adapt_decimal_constant_vector(
689 column_left_ptr->template get_value<A>(), column_right->get_ptr(),
690 res_data_type);
691 } else {
692 return OperationImpl::adapt_normal_constant_vector(
693 column_left_ptr->template get_value<A>(), column_right->get_ptr());
694 }
695 }
696
697 static ColumnPtr vector_vector(ColumnPtr column_left, ColumnPtr column_right,
698 const LeftDataType& type_left, const RightDataType& type_right,
699 DataTypePtr res_data_type) {
700 if constexpr (result_is_decimal) {
701 return OperationImpl::adapt_decimal_vector_vector(
702 column_left->get_ptr(), column_right->get_ptr(), res_data_type);
703 } else {
704 return OperationImpl::adapt_normal_vector_vector(column_left->get_ptr(),
705 column_right->get_ptr());
706 }
707 }
