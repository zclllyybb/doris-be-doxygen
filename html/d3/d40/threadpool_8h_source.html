48// ThreadPool takes a lot of arguments. We provide sane defaults with a builder.
49//
50// name: Used for debugging output and default names of the worker threads.
51// Since thread names are limited to 16 characters on Linux, it's good to
52// choose a short name here.
53// Required.
54//
55// trace_metric_prefix: used to prefix the names of TraceMetric counters.
56// When a task on a thread pool has an associated trace, the thread pool
57// implementation will increment TraceMetric counters to indicate the
58// amount of time spent waiting in the queue as well as the amount of wall
59// and CPU time spent executing. By default, these counters are prefixed
60// with the name of the thread pool. For example, if the pool is named
61// 'apply', then counters such as 'apply.queue_time_us' will be
62// incremented.
63//
64// The TraceMetrics implementation relies on the number of distinct counter
65// names being small. Thus, if the thread pool name itself is dynamically
66// generated, the default behavior described above would result in an
67// unbounded number of distinct counter names. The 'trace_metric_prefix'
68// setting can be used to override the prefix used in generating the trace
69// metric names.
70//
71// For example, the Raft thread pools are named "<tablet id>-raft" which
72// has unbounded cardinality (a server may have thousands of different
73// tablet IDs over its lifetime). In that case, setting the prefix to
74// "raft" will avoid any issues.
75//
76// min_threads: Minimum number of threads we'll have at any time.
77// Default: 0.
78//
79// max_threads: Maximum number of threads we'll have at any time.
80// Default: Number of CPUs detected on the system.
81//
82// max_queue_size: Maximum number of items to enqueue before returning a
83// Status::ServiceUnavailable message from Submit().
84// Default: INT_MAX.
85//
86// idle_timeout: How long we'll keep around an idle thread before timing it out.
87// We always keep at least min_threads.
88// Default: 500 milliseconds.
89//
90// metrics: Histograms, counters, etc. to update on various threadpool events.
91// Default: not set.
92//
93class ThreadPoolBuilder {
94public:
95 explicit ThreadPoolBuilder(std::string name);
96
97 // Note: We violate the style guide by returning mutable references here
98 // in order to provide traditional Builder pattern conveniences.
99 ThreadPoolBuilder& set_min_threads(int min_threads);
100 ThreadPoolBuilder& set_max_threads(int max_threads);
101 ThreadPoolBuilder& set_max_queue_size(int max_queue_size);
102 template <class Rep, class Period>
103 ThreadPoolBuilder& set_idle_timeout(const std::chrono::duration<Rep, Period>& idle_timeout) {
104 _idle_timeout = std::chrono::duration_cast<std::chrono::milliseconds>(idle_timeout);
105 return *this;
106 }
107 // Instantiate a new ThreadPool with the existing builder arguments.
108 Status build(std::unique_ptr<ThreadPool>* pool) const;
109
110private:
111 friend class ThreadPool;
112 const std::string _name;
113 int _min_threads;
114 int _max_threads;
115 int _max_queue_size;
116 std::chrono::milliseconds _idle_timeout;

118 ThreadPoolBuilder(const ThreadPoolBuilder&) = delete;
119 void operator=(const ThreadPoolBuilder&) = delete;

122// Thread pool with a variable number of threads.
123//
124// Tasks submitted directly to the thread pool enter a FIFO queue and are
125// dispatched to a worker thread when one becomes free. Tasks may also be
126// submitted via ThreadPoolTokens. The token Wait() and Shutdown() functions
127// can then be used to block on logical groups of tasks.
128//
129// A token operates in one of two ExecutionModes, determined at token
130// construction time:
131// 1. SERIAL: submitted tasks are run one at a time.
132// 2. CONCURRENT: submitted tasks may be run in parallel. This isn't unlike
133// tasks submitted without a token, but the logical grouping that tokens
134// impart can be useful when a pool is shared by many contexts (e.g. to
135// safely shut down one context, to derive context-specific metrics, etc.).
136//
137// Tasks submitted without a token or via ExecutionMode::CONCURRENT tokens are
138// processed in FIFO order. On the other hand, ExecutionMode::SERIAL tokens are
139// processed in a round-robin fashion, one task at a time. This prevents them
140// from starving one another. However, tokenless (and CONCURRENT token-based)
141// tasks can starve SERIAL token-based tasks.
142//
143// Usage Example:
144// static void Func(int n) { ... }
145// class Task : public Runnable { ... }
146//
147// std::unique_ptr<ThreadPool> thread_pool;
148// CHECK_OK(
149// ThreadPoolBuilder("my_pool")
150// .set_min_threads(0)
151// .set_max_threads(5)
152// .set_max_queue_size(10)
153// .set_idle_timeout(2000ms))
154// .Build(&thread_pool));
155// thread_pool->Submit(shared_ptr<Runnable>(new Task()));
156// thread_pool->SubmitFunc(std::bind(&Func, 10));

359// Entry point for token-based task submission and blocking for a particular
360// thread pool. Tokens can only be created via ThreadPool::new_token().
361//
362// All functions are thread-safe. Mutable members are protected via the
363// ThreadPool's lock.
364class ThreadPoolToken {
365public:
366 // Destroys the token.
367 //
368 // May be called on a token with outstanding tasks, as Shutdown() will be
369 // called first to take care of them.
370 ~ThreadPoolToken();
371
372 // Submits a Runnable class.
373 Status submit(std::shared_ptr<Runnable> r);
374
375 // Submits a function bound using std::bind(&FuncName, args...).
376 Status submit_func(std::function<void()> f);
377
378 // Marks the token as unusable for future submissions. Any queued tasks not
379 // yet running are destroyed. If tasks are in flight, Shutdown() will wait
380 // on their completion before returning.
381 void shutdown();
382
383 // Waits until all the tasks submitted via this token are completed.
384 void wait();
385
386 // Waits for all submissions using this token are complete, or until 'delta'
387 // time elapses.
388 //
389 // Returns true if all submissions are complete, false otherwise.
390 template <class Rep, class Period>
391 bool wait_for(const std::chrono::duration<Rep, Period>& delta) {
392 std::unique_lock<std::mutex> l(_pool->_lock);
393 _pool->check_not_pool_thread_unlocked();
394 return _not_running_cond.wait_for(l, delta, [&]() { return !is_active(); });
395 }
396
397 bool need_dispatch();
398
399 size_t num_tasks() {
400 std::lock_guard<std::mutex> l(_pool->_lock);
401 return _entries.size();
402 }

404private:
405 // All possible token states. Legal state transitions:
406 // IDLE -> RUNNING: task is submitted via token
407 // IDLE -> QUIESCED: token or pool is shut down
408 // RUNNING -> IDLE: worker thread finishes executing a task and
409 // there are no more tasks queued to the token
410 // RUNNING -> QUIESCING: token or pool is shut down while worker thread
411 // is executing a task
412 // RUNNING -> QUIESCED: token or pool is shut down
413 // QUIESCING -> QUIESCED: worker thread finishes executing a task
414 // belonging to a shut down token or pool
415 enum class State {
416 // Token has no queued tasks.
417 IDLE,
418
419 // A worker thread is running one of the token's previously queued tasks.
420 RUNNING,
421
422 // No new tasks may be submitted to the token. A worker thread is still
423 // running a previously queued task.
424 QUIESCING,
425
426 // No new tasks may be submitted to the token. There are no active tasks
427 // either. At this state, the token may only be destroyed.
428 QUIESCED,
429 };
430
431 // Writes a textual representation of the token state in 's' to 'o'.
432 friend std::ostream& operator<<(std::ostream& o, ThreadPoolToken::State s);
433
434 friend class ThreadPool;
435
436 // Returns a textual representation of 's' suitable for debugging.
437 static const char* state_to_string(State s);
438
439 // Constructs a new token.
440 //
441 // The token may not outlive its thread pool ('pool').
442 ThreadPoolToken(ThreadPool* pool, ThreadPool::ExecutionMode mode,
443 int max_concurrency = INT_MAX);
444
445 // Changes this token's state to 'new_state' taking actions as needed.
446 void transition(State new_state);
447
448 // Returns true if this token has a task queued and ready to run, or if a
449 // task belonging to this token is already running.
450 bool is_active() const { return _state == State::RUNNING || _state == State::QUIESCING; }
451
452 // Returns true if new tasks may be submitted to this token.
453 bool may_submit_new_tasks() const {
454 return _state != State::QUIESCING && _state != State::QUIESCED;
455 }
456
457 State state() const { return _state; }
458 ThreadPool::ExecutionMode mode() const { return _mode; }
459
460 // Token's configured execution mode.
461 ThreadPool::ExecutionMode _mode;
462
463 // Pointer to the token's thread pool.
464 ThreadPool* _pool;
465
466 // Token state machine.
467 State _state;
468
469 // Queued client tasks.
470 std::deque<ThreadPool::Task> _entries;
471
472 // Condition variable for "token is idle". Waiters wake up when the token
473 // transitions to IDLE or QUIESCED.
474 std::condition_variable _not_running_cond;
475
476 // Number of worker threads currently executing tasks belonging to this
477 // token.
478 int _active_threads;
479 // The max number of tasks that can be ran concurrenlty. This is to limit
480 // the concurrency of a thread pool token, and default is INT_MAX(no limited)
481 int _max_concurrency;
482 // Number of tasks which has been submitted to the thread pool's queue.
483 int _num_submitted_tasks;
484 // Number of tasks which has not been submitted to the thread pool's queue.
485 int _num_unsubmitted_tasks;
486
487 ThreadPoolToken(const ThreadPoolToken&) = delete;
488 void operator=(const ThreadPoolToken&) = delete;
489};
