78template <typename LoopType, LoopType start, LoopType end, template <LoopType> typename Reducer>
79struct constexpr_loop_match {
80 template <typename... TArgs>
81 static void run(LoopType target, TArgs&&... args) {
82 if constexpr (start <= end) {
83 if (start == target) {
84 Reducer<start>::run(std::forward<TArgs>(args)...);
85 } else {
86 if constexpr (start < std::numeric_limits<LoopType>::max()) {
87 constexpr_loop_match<LoopType, start + 1, end, Reducer>::run(
88 target, std::forward<TArgs>(args)...);
89 }
90 }
91 }
92 }
93};

101// we can't use variadic-parameters, because it will reject alias-templates.

103template <typename LoopType, LoopType start, LoopType end,
104 template <LoopType, LoopType> typename Reducer,
105 template <template <LoopType> typename> typename InnerMatch>
106struct constexpr_2_loop_match {
107 template <LoopType matched>
108 using InnerReducer = Reducer<start, matched>;
109
110 template <typename... TArgs>
111 static void run(LoopType target, TArgs&&... args) {
112 if constexpr (start <= end) {
113 if (start == target) {
114 InnerMatch<InnerReducer>::run(std::forward<TArgs>(args)...);
115 } else {
116 if constexpr (start < std::numeric_limits<LoopType>::max()) {
117 constexpr_2_loop_match<LoopType, start + 1, end, Reducer, InnerMatch>::run(
118 target, std::forward<TArgs>(args)...);
119 }
120 }
121 }
122 }
