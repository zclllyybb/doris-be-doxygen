83template <typename LoopType, LoopType start, LoopType end, template <LoopType> typename Reducer>
84struct constexpr_loop_match {
85 template <typename... TArgs>
86 static void run(LoopType target, TArgs&&... args) {
87 if constexpr (start <= end) {
88 if (start == target) {
89 Reducer<start>::run(std::forward<TArgs>(args)...);
90 } else {
91 if constexpr (start < std::numeric_limits<LoopType>::max()) {
92 constexpr_loop_match<LoopType, start + 1, end, Reducer>::run(
93 target, std::forward<TArgs>(args)...);
94 }
95 }
96 }
97 }
98};

106// we can't use variadic-parameters, because it will reject alias-templates.
107// https://stackoverflow.com/questions/30707011/pack-expansion-for-alias-template
108template <typename LoopType, LoopType start, LoopType end,
109 template <LoopType, LoopType> typename Reducer,
110 template <template <LoopType> typename> typename InnerMatch>
111struct constexpr_2_loop_match {
112 template <LoopType matched>
113 using InnerReducer = Reducer<start, matched>;
114
115 template <typename... TArgs>
116 static void run(LoopType target, TArgs&&... args) {
117 if constexpr (start <= end) {
118 if (start == target) {
119 InnerMatch<InnerReducer>::run(std::forward<TArgs>(args)...);
120 } else {
121 if constexpr (start < std::numeric_limits<LoopType>::max()) {
122 constexpr_2_loop_match<LoopType, start + 1, end, Reducer, InnerMatch>::run(
123 target, std::forward<TArgs>(args)...);
124 }
125 }
126 }
127 }
128};
