48inline OpentelemetrySpan start_rpc_server_span(std::string span_name,
49 google::protobuf::RpcController* cntl_base) {
50 RpcServerCarrier carrier(static_cast<brpc::Controller*>(cntl_base));
51 auto current_ctx = opentelemetry::context::RuntimeContext::GetCurrent();
52 auto prop = opentelemetry::context::propagation::GlobalTextMapPropagator::GetGlobalPropagator();
53 auto parent_context = prop->Extract(carrier, current_ctx);
54
55 if (opentelemetry::trace::GetSpan(parent_context)->GetContext().IsValid()) {
56 opentelemetry::trace::StartSpanOptions options;
57 options.kind = opentelemetry::trace::SpanKind::kServer;
58 options.parent = parent_context;
59 return telemetry::get_tracer("tracer")->StartSpan(std::move(span_name), options);
60 } else {
61 return telemetry::get_noop_tracer()->StartSpan("");
62 }
63}
