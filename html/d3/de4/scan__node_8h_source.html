33// Abstract base class of all scan nodes; introduces set_scan_range().
34//
35// Includes ScanNode common counters:
36// BytesRead - total bytes read by this scan node
37//
38// TotalRawHdfsReadTime - it measures the total time spent in the disk-io-mgr's reading
39// threads for this node. For example, if we have 3 reading threads and each spent
40// 1 sec, this counter will report 3 sec.
41//
42// TotalReadThroughput - BytesRead divided by the total time spent in this node
43// (from Open to Close). For IO bounded queries, this should be very close to the
44// total throughput of all the disks.
45//
46// PerDiskRawHdfsThroughput - the read throughput for each disk. If all the data reside
47// on disk, this should be the read throughput the disk, regardless of whether the
48// query is IO bounded or not.
49//
50// NumDisksAccessed - number of disks accessed.
51//
52// AverageIoMgrQueueCapcity - the average queue capacity in the io mgr for this node.
53// AverageIoMgrQueueSize - the average queue size (for ready buffers) in the io mgr
54// for this node.
55//
56// AverageScannerThreadConcurrency - the average number of active scanner threads. A
57// scanner thread is considered active if it is not blocked by IO. This number would
58// be low (less than 1) for IO bounded queries. For cpu bounded queries, this number
59// would be close to the max scanner threads allowed.
60//
61// AverageHdfsReadThreadConcurrency - the average number of active hdfs reading threads
62// reading for this scan node. For IO bound queries, this should be close to the
63// number of disk.
64//
65// HdfsReadThreadConcurrencyCount=<i> - the number of samples taken when the hdfs read
66// thread concurrency is <i>.
67//
68// ScanRangesComplete - number of scan ranges completed
69//

86 RuntimeProfile::Counter* bytes_read_counter() const { return _bytes_read_counter; }
87 RuntimeProfile::Counter* rows_read_counter() const { return _rows_read_counter; }
88 RuntimeProfile::Counter* total_throughput_counter() const { return _total_throughput_counter; }

107protected:

109};
