59template <typename A, typename B, typename Op>
60struct NumComparisonImpl {
62 static void NO_INLINE vector_vector(const PaddedPODArray<A>& a, const PaddedPODArray<B>& b,
63 PaddedPODArray<UInt8>& c) {
69 size_t size = a.size();
70 const A* a_pos = a.data();
71 const B* b_pos = b.data();
72 UInt8* c_pos = c.data();
73 const A* a_end = a_pos + size;
74
75 while (a_pos < a_end) {
76 *c_pos = Op::apply(*a_pos, *b_pos);
77 ++a_pos;
78 ++b_pos;
79 ++c_pos;
80 }
81 }
82
83 static void NO_INLINE vector_constant(const PaddedPODArray<A>& a, B b,
84 PaddedPODArray<UInt8>& c) {
85 size_t size = a.size();
86 const A* a_pos = a.data();
87 UInt8* c_pos = c.data();
88 const A* a_end = a_pos + size;
89
90 while (a_pos < a_end) {
91 *c_pos = Op::apply(*a_pos, b);
92 ++a_pos;
93 ++c_pos;
94 }
95 }
96
97 static void constant_vector(A a, const PaddedPODArray<B>& b, PaddedPODArray<UInt8>& c) {
98 NumComparisonImpl<B, A, typename Op::SymmetricOp>::vector_constant(b, a, c);
99 }
100
101 static void constant_constant(A a, B b, UInt8& c) { c = Op::apply(a, b); }
102};

105template <typename Op>
106struct GenericComparisonImpl {
107 static void NO_INLINE vector_vector(const IColumn& a, const IColumn& b,
108 PaddedPODArray<UInt8>& c) {
109 for (size_t i = 0, size = a.size(); i < size; ++i)
110 c[i] = Op::apply(a.compare_at(i, i, b, 1), 0);
111 }
112
113 static void NO_INLINE vector_constant(const IColumn& a, const IColumn& b,
114 PaddedPODArray<UInt8>& c) {
115 auto b_materialized = b.clone_resized(1)->convert_to_full_column_if_const();
116 for (size_t i = 0, size = a.size(); i < size; ++i)
117 c[i] = Op::apply(a.compare_at(i, 0, *b_materialized, 1), 0);
118 }
119
120 static void constant_vector(const IColumn& a, const IColumn& b, PaddedPODArray<UInt8>& c) {
121 GenericComparisonImpl<typename Op::SymmetricOp>::vector_constant(b, a, c);
122 }
123
124 static void constant_constant(const IColumn& a, const IColumn& b, UInt8& c) {
125 c = Op::apply(a.compare_at(0, 0, b, 1), 0);
126 }
127};

157 template <typename T0, typename T1>
158 bool execute_num_right_type(Block& block, size_t result, const ColumnVector<T0>* col_left,
159 const IColumn* col_right_untyped) {
160 if (const ColumnVector<T1>* col_right =
161 check_and_get_column<ColumnVector<T1>>(col_right_untyped)) {
162 auto col_res = ColumnUInt8::create();
163
164 ColumnUInt8::Container& vec_res = col_res->get_data();
165 vec_res.resize(col_left->get_data().size());
166 NumComparisonImpl<T0, T1, Op<T0, T1>>::vector_vector(col_left->get_data(),
167 col_right->get_data(), vec_res);
168
169 block.replace_by_position(result, std::move(col_res));
170 return true;
171 } else if (auto col_right_const =
172 check_and_get_column_const<ColumnVector<T1>>(col_right_untyped)) {
173 auto col_res = ColumnUInt8::create();
174
175 ColumnUInt8::Container& vec_res = col_res->get_data();
176 vec_res.resize(col_left->size());
177 NumComparisonImpl<T0, T1, Op<T0, T1>>::vector_constant(
178 col_left->get_data(), col_right_const->template get_value<T1>(), vec_res);
179
180 block.replace_by_position(result, std::move(col_res));
181 return true;
182 }
183
184 return false;
185 }

272 Status execute_decimal(Block& block, size_t result, const ColumnWithTypeAndName& col_left,
273 const ColumnWithTypeAndName& col_right) {
274 TypeIndex left_number = col_left.type->get_type_id();
275 TypeIndex right_number = col_right.type->get_type_id();
276
277 auto call = [&](const auto& types) -> bool {
278 using Types = std::decay_t<decltype(types)>;
279 using LeftDataType = typename Types::LeftType;
280 using RightDataType = typename Types::RightType;
281
282 DecimalComparison<LeftDataType, RightDataType, Op, false>(block, result, col_left,
283 col_right);
284 return true;
285 };
286
287 if (!call_on_basic_types<true, false, true, false>(left_number, right_number, call)) {
288 return Status::RuntimeError("Wrong call for {} with {} and {}", get_name(),
289 col_left.type->get_name(), col_right.type->get_name());
290 }
291 return Status::OK();
292 }

294 void execute_generic_identical_types(Block& block, size_t result, const IColumn* c0,
295 const IColumn* c1) {
296 bool c0_const = is_column_const(*c0);
297 bool c1_const = is_column_const(*c1);
298
299 if (c0_const && c1_const) {
300 UInt8 res = 0;
301 GenericComparisonImpl<Op<int, int>>::constant_constant(*c0, *c1, res);
302 block.replace_by_position(
303 result, DataTypeUInt8().create_column_const(c0->size(), to_field(res)));
304 } else {
305 auto c_res = ColumnUInt8::create();
306 ColumnUInt8::Container& vec_res = c_res->get_data();
307 vec_res.resize(c0->size());
308
309 if (c0_const)
310 GenericComparisonImpl<Op<int, int>>::constant_vector(*c0, *c1, vec_res);
311 else if (c1_const)
312 GenericComparisonImpl<Op<int, int>>::vector_constant(*c0, *c1, vec_res);
313 else
314 GenericComparisonImpl<Op<int, int>>::vector_vector(*c0, *c1, vec_res);
315
316 block.replace_by_position(result, std::move(c_res));
317 }
318 }

320 Status execute_generic(Block& block, size_t result, const ColumnWithTypeAndName& c0,
321 const ColumnWithTypeAndName& c1) {
322 execute_generic_identical_types(block, result, c0.column.get(), c1.column.get());
323 return Status::OK();
324 }

336 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
337 size_t result, size_t input_rows_count) override {
338 const auto& col_with_type_and_name_left = block.get_by_position(arguments[0]);
339 const auto& col_with_type_and_name_right = block.get_by_position(arguments[1]);
340 const IColumn* col_left_untyped = col_with_type_and_name_left.column.get();
341 const IColumn* col_right_untyped = col_with_type_and_name_right.column.get();
342
343 const DataTypePtr& left_type = col_with_type_and_name_left.type;
344 const DataTypePtr& right_type = col_with_type_and_name_right.type;
345
349 if (left_type->equals(*right_type) && !left_type->is_nullable() &&
350 col_left_untyped == col_right_untyped) {
352 // TODO: Return const column in the future
353 if constexpr (std::is_same_v<Op<int, int>, EqualsOp<int, int>> ||
354 std::is_same_v<Op<int, int>, LessOrEqualsOp<int, int>> ||
355 std::is_same_v<Op<int, int>, GreaterOrEqualsOp<int, int>>) {
356 block.get_by_position(result).column =
357 DataTypeUInt8()
358 .create_column_const(input_rows_count, 1u)
359 ->convert_to_full_column_if_const();
360 return Status::OK();
361 } else {
362 block.get_by_position(result).column =
363 DataTypeUInt8()
364 .create_column_const(input_rows_count, 0u)
365 ->convert_to_full_column_if_const();
366 return Status::OK();
367 }
368 }
369
370 WhichDataType which_left {left_type};
371 WhichDataType which_right {right_type};
372
373 const bool left_is_num = col_left_untyped->is_numeric();
374 const bool right_is_num = col_right_untyped->is_numeric();
375
376 // Compare date and datetime direct use the Int64 compare. Keep the comment
377 // may we should refactor the code.
378 // bool date_and_datetime = (left_type != right_type) && which_left.is_date_or_datetime() &&
379 // which_right.is_date_or_datetime();
380
381 if (left_is_num && right_is_num) {
382 if (!(execute_num_left_type<UInt8>(block, result, col_left_untyped,
383 col_right_untyped) ||
384 execute_num_left_type<UInt16>(block, result, col_left_untyped,
385 col_right_untyped) ||
386 execute_num_left_type<UInt32>(block, result, col_left_untyped,
387 col_right_untyped) ||
388 execute_num_left_type<UInt64>(block, result, col_left_untyped,
389 col_right_untyped) ||
390 execute_num_left_type<Int8>(block, result, col_left_untyped, col_right_untyped) ||
391 execute_num_left_type<Int16>(block, result, col_left_untyped,
392 col_right_untyped) ||
393 execute_num_left_type<Int32>(block, result, col_left_untyped,
394 col_right_untyped) ||
395 execute_num_left_type<Int64>(block, result, col_left_untyped,
396 col_right_untyped) ||
397 execute_num_left_type<Int128>(block, result, col_left_untyped,
398 col_right_untyped) ||
399 execute_num_left_type<Float32>(block, result, col_left_untyped,
400 col_right_untyped) ||
401 execute_num_left_type<Float64>(block, result, col_left_untyped,
402 col_right_untyped)))
403
404 return Status::RuntimeError("Illegal column {} of first argument of function {}",
405 col_left_untyped->get_name(), get_name());
406 } else if (is_decimal_v2(left_type) || is_decimal_v2(right_type)) {
407 if (!allow_decimal_comparison(left_type, right_type)) {
408 return Status::RuntimeError("No operation {} between {} and {}", get_name(),
409 left_type->get_name(), right_type->get_name());
410 }
411 return execute_decimal(block, result, col_with_type_and_name_left,
412 col_with_type_and_name_right);
413 } else if (is_decimal(left_type) || is_decimal(right_type)) {
414 if (!allow_decimal_comparison(left_type, right_type)) {
415 return Status::RuntimeError("No operation {} between {} and {}", get_name(),
416 left_type->get_name(), right_type->get_name());
417 }
418 return execute_decimal(block, result, col_with_type_and_name_left,
419 col_with_type_and_name_right);
420 } else {
421 // TODO: varchar and string maybe need a quickly way
422 return execute_generic(block, result, col_with_type_and_name_left,
423 col_with_type_and_name_right);
424 }
425 return Status::OK();
426 }
