34template <typename ColVecType, bool arg_is_nullable>
35struct Value {
36public:
37 bool is_null() const {
38 if (_ptr == nullptr) {
39 return true;
40 }
41 if constexpr (arg_is_nullable) {
42 return assert_cast<const ColumnNullable*>(_ptr)->is_null_at(_offset);
43 }
44 return false;
45 }
46
47 void insert_into(IColumn& to) const {
48 if constexpr (arg_is_nullable) {
49 auto* col = assert_cast<const ColumnNullable*>(_ptr);
50 assert_cast<ColVecType&>(to).insert_from(col->get_nested_column(), _offset);
51 } else {
52 assert_cast<ColVecType&>(to).insert_from(*_ptr, _offset);
53 }
54 }
55
56 void set_value(const IColumn* column, size_t row) {
57 _ptr = column;
58 _offset = row;
59 }
60
61 void reset() {
62 _ptr = nullptr;
63 _offset = 0;
64 }
65
66protected:
67 const IColumn* _ptr = nullptr;
68 size_t _offset = 0;
69};

71template <typename ColVecType, bool arg_is_nullable>
72struct CopiedValue : public Value<ColVecType, arg_is_nullable> {
73public:
74 void insert_into(IColumn& to) const { assert_cast<ColVecType&>(to).insert(_copied_value); }
75
76 bool is_null() const { return this->_ptr == nullptr; }
77
78 void set_value(const IColumn* column, size_t row) {
79 // here _ptr, maybe null at row, so call reset to set nullptr
80 // But we will use is_null() check first, others have set _ptr column to a meaningless address
81 // because the address have meaningless, only need it to check is nullptr
82 this->_ptr = (IColumn*)0x00000001;
83 if constexpr (arg_is_nullable) {
84 auto* col = assert_cast<const ColumnNullable*>(column);
85 if (col->is_null_at(row)) {
86 this->reset();
87 return;
88 } else {
89 auto& nested_col = assert_cast<const ColVecType&>(col->get_nested_column());
90 nested_col.get(row, _copied_value);
91 }
92 } else {
93 column->get(row, _copied_value);
94 }
95 }
96
97private:
98 Field _copied_value;
99};

101template <typename ColVecType, bool result_is_nullable, bool arg_is_nullable, bool is_copy>
102struct ReaderFirstAndLastData {
103public:
104 using StoreType = std::conditional_t<is_copy, CopiedValue<ColVecType, arg_is_nullable>,
105 Value<ColVecType, arg_is_nullable>>;
106 static constexpr bool nullable = arg_is_nullable;
107
108 void reset() {
109 _data_value.reset();
110 _has_value = false;
111 }
112
113 void insert_result_into(IColumn& to) const {
114 if constexpr (result_is_nullable) {
115 if (_data_value.is_null()) { //_ptr == nullptr || null data at row
116 auto& col = assert_cast<ColumnNullable&>(to);
117 col.insert_default();
118 } else {
119 auto& col = assert_cast<ColumnNullable&>(to);
120 col.get_null_map_data().push_back(0);
121 _data_value.insert_into(col.get_nested_column());
122 }
123 } else {
124 _data_value.insert_into(to);
125 }
126 }
127
128 // here not check the columns[0] is null at the row,
129 // but it is need to check in other
130 void set_value(const IColumn** columns, size_t pos) {
131 _data_value.set_value(columns[0], pos);
132 _has_value = true;
133 }
134
135 bool has_set_value() { return _has_value; }
136
137protected:
138 StoreType _data_value;
139 bool _has_value = false;
140};

233template <template <typename> class AggregateFunctionTemplate, template <typename> class Impl,
234 bool result_is_nullable, bool arg_is_nullable, bool is_copy = false>
235IAggregateFunction* create_function_single_value(const String& name,
236 const DataTypes& argument_types) {
237 auto type = remove_nullable(argument_types[0]);
238 WhichDataType which(*type);
239
240#define DISPATCH(TYPE, COLUMN_TYPE) \
241 if (which.idx == TypeIndex::TYPE) \
242 return new AggregateFunctionTemplate<Impl<ReaderFirstAndLastData< \
243 COLUMN_TYPE, result_is_nullable, arg_is_nullable, is_copy>>>(argument_types);
244 TYPE_TO_COLUMN_TYPE(DISPATCH)
245#undef DISPATCH
246
247 LOG(FATAL) << "with unknowed type, failed in create_aggregate_function_" << name
248 << " and type is: " << argument_types[0]->get_name();
249 return nullptr;
250}

252#define CREATE_READER_FUNCTION_WITH_NAME_AND_DATA(CREATE_FUNCTION_NAME, FUNCTION_DATA) \
253 template <bool is_copy> \
254 AggregateFunctionPtr CREATE_FUNCTION_NAME( \
255 const std::string& name, const DataTypes& argument_types, bool result_is_nullable) { \
256 const bool arg_is_nullable = argument_types[0]->is_nullable(); \
257 AggregateFunctionPtr res = nullptr; \
258 std::visit( \
259 [&](auto result_is_nullable, auto arg_is_nullable) { \
260 res = AggregateFunctionPtr( \
261 create_function_single_value<ReaderFunctionData, FUNCTION_DATA, \
262 result_is_nullable, arg_is_nullable, \
263 is_copy>(name, argument_types)); \
264 }, \
265 make_bool_variant(result_is_nullable), make_bool_variant(arg_is_nullable)); \
266 if (!res) { \
267 LOG(WARNING) << " failed in create_aggregate_function_" << name \
268 << " and type is: " << argument_types[0]->get_name(); \
269 } \
270 return res; \
271 }
