38template <PrimitiveType primitive_type, class T>
39std::string cast_to_string(T value, int scale) {
40 if constexpr (primitive_type == TYPE_DECIMAL32) {
41 std::stringstream ss;
42 vectorized::write_text<int32_t>((int32_t)value, scale, ss);
43 return ss.str();
44 } else if constexpr (primitive_type == TYPE_DECIMAL64) {
45 std::stringstream ss;
46 vectorized::write_text<int64_t>((int64_t)value, scale, ss);
47 return ss.str();
48 } else if constexpr (primitive_type == TYPE_DECIMAL128I) {
49 std::stringstream ss;
50 vectorized::write_text<int128_t>((int128_t)value, scale, ss);
51 return ss.str();
52 } else if constexpr (primitive_type == TYPE_TINYINT) {
53 return std::to_string(static_cast<int>(value));
54 } else if constexpr (primitive_type == TYPE_LARGEINT) {
55 return vectorized::int128_to_string(value);
56 } else if constexpr (primitive_type == TYPE_DATETIMEV2) {
57 doris::vectorized::DateV2Value<doris::vectorized::DateTimeV2ValueType> datetimev2_val =
58 static_cast<doris::vectorized::DateV2Value<doris::vectorized::DateTimeV2ValueType>>(
59 value);
60 char buf[30];
61 datetimev2_val.to_string(buf);
62 std::stringstream ss;
63 ss << buf;
64 return ss.str();
65 } else {
66 return boost::lexical_cast<std::string>(value);
67 }
68}

178 void to_olap_filter(std::vector<TCondition>& filters) {
179 if (is_fixed_value_range() || is_match_value_range()) {
180 // 1. convert to in filter condition
181 if (is_fixed_value_range()) {
182 to_in_condition(filters, true);
183 }
184
185 if (is_match_value_range()) {
186 to_match_condition(filters);
187 }
188 } else if (_low_value < _high_value) {
189 // 2. convert to min max filter condition
190 TCondition null_pred;
191 if (TYPE_MAX == _high_value && _high_op == FILTER_LESS_OR_EQUAL &&
192 TYPE_MIN == _low_value && _low_op == FILTER_LARGER_OR_EQUAL && !contain_null()) {
193 null_pred.__set_column_name(_column_name);
194 null_pred.__set_condition_op("is");
195 null_pred.condition_values.emplace_back("not null");
196 }
197
198 if (null_pred.condition_values.size() != 0) {
199 filters.push_back(null_pred);
200 return;
201 }
202
203 TCondition low;
204 if (TYPE_MIN != _low_value || FILTER_LARGER_OR_EQUAL != _low_op) {
205 low.__set_column_name(_column_name);
206 low.__set_condition_op((_low_op == FILTER_LARGER_OR_EQUAL ? ">=" : ">>"));
207 low.condition_values.push_back(
208 cast_to_string<primitive_type, CppType>(_low_value, _scale));
209 }
210
211 if (low.condition_values.size() != 0) {
212 filters.push_back(low);
213 }
214
215 TCondition high;
216 if (TYPE_MAX != _high_value || FILTER_LESS_OR_EQUAL != _high_op) {
217 high.__set_column_name(_column_name);
218 high.__set_condition_op((_high_op == FILTER_LESS_OR_EQUAL ? "<=" : "<<"));
219 high.condition_values.push_back(
220 cast_to_string<primitive_type, CppType>(_high_value, _scale));
221 }
222
223 if (high.condition_values.size() != 0) {
224 filters.push_back(high);
225 }
226 } else {
227 // 3. convert to is null and is not null filter condition
228 TCondition null_pred;
229 if (TYPE_MAX == _low_value && TYPE_MIN == _high_value && contain_null()) {
230 null_pred.__set_column_name(_column_name);
231 null_pred.__set_condition_op("is");
232 null_pred.condition_values.emplace_back("null");
233 }
234
235 if (null_pred.condition_values.size() != 0) {
236 filters.push_back(null_pred);
237 }
238 }
239 }

241 void to_in_condition(std::vector<TCondition>& filters, bool is_in = true) {
242 TCondition condition;
243 condition.__set_column_name(_column_name);
244 condition.__set_condition_op(is_in ? "*=" : "!*=");
245
246 for (const auto& value : _fixed_values) {
247 condition.condition_values.push_back(
248 cast_to_string<primitive_type, CppType>(value, _scale));
249 }
250
251 if (condition.condition_values.size() != 0) {
252 filters.push_back(condition);
253 }
254 }
255
256 void to_condition_in_compound(std::vector<TCondition>& filters) {
257 for (const auto& value : _compound_values) {
258 TCondition condition;
259 condition.__set_column_name(_column_name);
260 if (value.first == FILTER_LARGER) {
261 condition.__set_condition_op(">>");
262 } else if (value.first == FILTER_LARGER_OR_EQUAL) {
263 condition.__set_condition_op(">=");
264 } else if (value.first == FILTER_LESS) {
265 condition.__set_condition_op("<<");
266 } else if (value.first == FILTER_LESS_OR_EQUAL) {
267 condition.__set_condition_op("<=");
268 } else if (value.first == FILTER_IN) {
269 condition.__set_condition_op("*=");
270 } else if (value.first == FILTER_NOT_IN) {
271 condition.__set_condition_op("!*=");
272 }
273 condition.condition_values.push_back(
274 cast_to_string<primitive_type, CppType>(value.second, _scale));
275 if (condition.condition_values.size() != 0) {
276 filters.push_back(condition);
277 }
278 }
279 }

281 void to_match_condition(std::vector<TCondition>& filters) {
282 for (const auto& value : _match_values) {
283 TCondition condition;
284 condition.__set_column_name(_column_name);
285
286 if (value.first == MatchType::MATCH_ANY) {
287 condition.__set_condition_op("match_any");
288 } else if (value.first == MatchType::MATCH_ALL) {
289 condition.__set_condition_op("match_all");
290 } else if (value.first == MatchType::MATCH_PHRASE) {
291 condition.__set_condition_op("match_phrase");
292 } else if (value.first == MatchType::MATCH_ELEMENT_EQ) {
293 condition.__set_condition_op("match_element_eq");
294 } else if (value.first == MatchType::MATCH_ELEMENT_LT) {
295 condition.__set_condition_op("match_element_lt");
296 } else if (value.first == MatchType::MATCH_ELEMENT_GT) {
297 condition.__set_condition_op("match_element_gt");
298 } else if (value.first == MatchType::MATCH_ELEMENT_LE) {
299 condition.__set_condition_op("match_element_le");
300 } else if (value.first == MatchType::MATCH_ELEMENT_GE) {
301 condition.__set_condition_op("match_element_ge");
302 }
303 condition.condition_values.push_back(
304 cast_to_string<primitive_type, CppType>(value.second, 0));
305 if (condition.condition_values.size() != 0) {
306 filters.push_back(condition);
307 }
308 }

311 void set_whole_value_range() {
312 _fixed_values.clear();
313 _low_value = TYPE_MIN;
314 _high_value = TYPE_MAX;
315 _low_op = FILTER_LARGER_OR_EQUAL;
316 _high_op = FILTER_LESS_OR_EQUAL;
317 _contain_null = true;
318 }
319
320 bool is_whole_value_range() const {
321 return _fixed_values.empty() && _low_value == TYPE_MIN && _high_value == TYPE_MAX &&
322 _low_op == FILTER_LARGER_OR_EQUAL && _high_op == FILTER_LESS_OR_EQUAL &&
323 contain_null();
324 }
325
326 // only two case will set range contain null, call by temp_range in olap scan node
327 // 'is null' and 'is not null'
328 // 1. if the pred is 'is null' means the range should be
329 // empty in fixed_range and _high_value < _low_value
330 // 2. if the pred is 'is not null' means the range should be whole range and
331 // 'is not null' be effective
332 void set_contain_null(bool contain_null) {
333 if (contain_null) {
334 set_empty_value_range();
335 } else {
336 set_whole_value_range();
337 }
338 _contain_null = contain_null;
339 }

694// The return value indicates whether the split result is range or fixed value.
695template <PrimitiveType primitive_type>
696bool ColumnValueRange<primitive_type>::convert_to_avg_range_value(
697 std::vector<OlapTuple>& begin_scan_keys, std::vector<OlapTuple>& end_scan_keys,
698 bool& begin_include, bool& end_include, int32_t max_scan_key_num) {
699 if constexpr (!_is_reject_split_type) {
700 auto no_split = [&]() -> bool {
701 begin_scan_keys.emplace_back();
702 begin_scan_keys.back().add_value(
703 cast_to_string<primitive_type, CppType>(get_range_min_value(), scale()),
704 contain_null());
705 end_scan_keys.emplace_back();
706 end_scan_keys.back().add_value(
707 cast_to_string<primitive_type, CppType>(get_range_max_value(), scale()));
708 return true;
709 };
710
711 CppType min_value = get_range_min_value();
712 CppType max_value = get_range_max_value();
713 if constexpr (primitive_type == PrimitiveType::TYPE_DATE) {
714 min_value.set_type(TimeType::TIME_DATE);
715 max_value.set_type(TimeType::TIME_DATE);
716 }
717
718 if (contain_null()) {
719 begin_scan_keys.emplace_back();
720 begin_scan_keys.back().add_null();
721 end_scan_keys.emplace_back();
722 end_scan_keys.back().add_null();
723 }
724
725 if (min_value > max_value || max_scan_key_num == 1) {
726 return no_split();
727 }
728
729 auto cast = [](const CppType& value) {
730 if constexpr (primitive_type == PrimitiveType::TYPE_DATE ||
731 primitive_type == PrimitiveType::TYPE_DATEV2) {
732 return value;
733 } else {
734 return (int128_t)value;
735 }
736 };
737
738 // When CppType is date, we can not convert it to integer number and calculate distance.
739 // In other case, we convert element to int128 to avoit overflow.
740 size_t step_size = (cast(max_value) - min_value) / max_scan_key_num;
741
742 constexpr size_t MAX_STEP_SIZE = 1 << 20;
743 // When the step size is too large, the range is easy to not really contain data.
744 if (step_size > MAX_STEP_SIZE) {
745 return no_split();
746 }
747
748 while (true) {
749 begin_scan_keys.emplace_back();
750 begin_scan_keys.back().add_value(
751 cast_to_string<primitive_type, CppType>(min_value, scale()));
752
753 if (cast(max_value) - min_value < step_size) {
754 min_value = max_value;
755 } else {
756 min_value += step_size;
757 }
758
759 end_scan_keys.emplace_back();
760 end_scan_keys.back().add_value(
761 cast_to_string<primitive_type, CppType>(min_value, scale()));
762
763 if (min_value == max_value) {
764 break;
765 }
766 ++min_value;
767 }
768
769 return step_size != 0;
770 }
771 return false;
772}

774template <PrimitiveType primitive_type>
775void ColumnValueRange<primitive_type>::convert_to_range_value() {
776 if (!is_range_value_convertible()) {
777 return;
778 }
779
780 if (!_fixed_values.empty()) {
781 _low_value = *_fixed_values.begin();
782 _low_op = FILTER_LARGER_OR_EQUAL;
783 _high_value = *_fixed_values.rbegin();
784 _high_op = FILTER_LESS_OR_EQUAL;
785 _fixed_values.clear();
786 }

789template <PrimitiveType primitive_type>
790Status ColumnValueRange<primitive_type>::add_range(SQLFilterOp op, CppType value) {
791 if (INVALID_TYPE == _column_type) {
792 return Status::InternalError("AddRange failed, Invalid type");
793 }
794
795 // add range means range should not contain null
796 _contain_null = false;
797
798 if (is_fixed_value_range()) {
799 std::pair<IteratorType, IteratorType> bound_pair = _fixed_values.equal_range(value);
800
801 switch (op) {
802 case FILTER_LARGER: {
803 _fixed_values.erase(_fixed_values.begin(), bound_pair.second);
804 break;
805 }
806
807 case FILTER_LARGER_OR_EQUAL: {
808 _fixed_values.erase(_fixed_values.begin(), bound_pair.first);
809 break;
810 }
811
812 case FILTER_LESS: {
813 if (bound_pair.first == _fixed_values.find(value)) {
814 _fixed_values.erase(bound_pair.first, _fixed_values.end());
815 } else {
816 _fixed_values.erase(bound_pair.second, _fixed_values.end());
817 }
818
819 break;
820 }
821
822 case FILTER_LESS_OR_EQUAL: {
823 _fixed_values.erase(bound_pair.second, _fixed_values.end());
824 break;
825 }
826
827 default: {
828 return Status::InternalError("Add Range fail! Unsupported SQLFilterOp.");
829 }
830 }
831
832 _high_value = TYPE_MIN;
833 _low_value = TYPE_MAX;
834 } else {
835 if (_high_value > _low_value) {
836 switch (op) {
837 case FILTER_LARGER: {
838 if (value >= _low_value) {
839 _low_value = value;
840 _low_op = op;
841 }
842
843 break;
844 }
845
846 case FILTER_LARGER_OR_EQUAL: {
847 if (value > _low_value) {
848 _low_value = value;
849 _low_op = op;
850 }
851
852 break;
853 }
854
855 case FILTER_LESS: {
856 if (value <= _high_value) {
857 _high_value = value;
858 _high_op = op;
859 }
860
861 break;
862 }
863
864 case FILTER_LESS_OR_EQUAL: {
865 if (value < _high_value) {
866 _high_value = value;
867 _high_op = op;
868 }
869
870 break;
871 }
872
873 default: {
874 return Status::InternalError("Add Range fail! Unsupported SQLFilterOp.");
875 }
876 }
877 }
878
879 if (FILTER_LARGER_OR_EQUAL == _low_op && FILTER_LESS_OR_EQUAL == _high_op &&
880 _high_value == _low_value) {
881 add_fixed_value(_high_value);
882 _high_value = TYPE_MIN;
883 _low_value = TYPE_MAX;
884 }
885 }
886
887 return Status::OK();
888}

890template <PrimitiveType primitive_type>
891bool ColumnValueRange<primitive_type>::is_in_range(const CppType& value) {
892 switch (_high_op) {
893 case FILTER_LESS: {
894 switch (_low_op) {
895 case FILTER_LARGER: {
896 return value < _high_value && value > _low_value;
897 }
898
899 case FILTER_LARGER_OR_EQUAL: {
900 return value < _high_value && value >= _low_value;
901 }
902
903 default: {
904 DCHECK(false);
905 }
906 }
907
908 break;
909 }
910
911 case FILTER_LESS_OR_EQUAL: {
912 switch (_low_op) {
913 case FILTER_LARGER: {
914 return value <= _high_value && value > _low_value;
915 }
916
917 case FILTER_LARGER_OR_EQUAL: {
918 return value <= _high_value && value >= _low_value;
919 }
920
921 default: {
922 DCHECK(false);
923 }
924 }
925 }
926
927 default: {
928 DCHECK(false);
929 }
930 }
931
932 return false;

935template <PrimitiveType primitive_type>
936void ColumnValueRange<primitive_type>::intersection(ColumnValueRange<primitive_type>& range) {
937 // 1. clear if column type not match
938 if (_column_type != range._column_type) {
939 set_empty_value_range();
940 }
941
942 // 2. clear if any range is empty
943 if (is_empty_value_range() || range.is_empty_value_range()) {
944 set_empty_value_range();
945 }
946
947 std::set<CppType> result_values;
948 // 3. fixed_value intersection, fixed value range do not contain null
949 if (is_fixed_value_range() || range.is_fixed_value_range() || range.is_match_value_range()) {
950 if (is_fixed_value_range() && range.is_fixed_value_range()) {
951 set_intersection(_fixed_values.begin(), _fixed_values.end(),
952 range._fixed_values.begin(), range._fixed_values.end(),
953 std::inserter(result_values, result_values.begin()));
954 } else if (is_fixed_value_range() && !range.is_fixed_value_range()) {
955 IteratorType iter = _fixed_values.begin();
956
957 while (iter != _fixed_values.end()) {
958 if (range.is_in_range(*iter)) {
959 result_values.insert(*iter);
960 }
961 ++iter;
962 }
963 } else if (!is_fixed_value_range() && range.is_fixed_value_range()) {
964 IteratorType iter = range._fixed_values.begin();
965 while (iter != range._fixed_values.end()) {
966 if (this->is_in_range(*iter)) {
967 result_values.insert(*iter);
968 }
969 ++iter;
970 }
971 }
972
973 if (!result_values.empty()) {
974 _fixed_values = std::move(result_values);
975 _contain_null = false;
976 _high_value = TYPE_MIN;
977 _low_value = TYPE_MAX;
978 } else if (range.is_match_value_range()) {
979 for (auto& value : range._match_values) {
980 add_match_value(value.first, value.second);
981 }
982 } else {
983 set_empty_value_range();
984 }
985 } else {
986 if (contain_null() && range.contain_null()) {
987 // if both is_whole_range to keep the same, else set_contain_null
988 if (!is_whole_value_range() || !range.is_whole_value_range()) {
989 set_contain_null(true);
990 }
991 } else {
992 add_range(range._high_op, range._high_value);
993 add_range(range._low_op, range._low_value);
994 }
995 }

999bool ColumnValueRange<primitive_type>::has_intersection(ColumnValueRange<primitive_type>& range) {
1000 // 1. return false if column type not match
1001 if (_column_type != range._column_type) {
1002 return false;
1003 }
1004
1005 // 2. return false if any range is empty
1006 if (is_empty_value_range() || range.is_empty_value_range()) {
1007 return false;
1008 }
1009
1010 // 3.1 return false if two int fixedRange has no intersection
1011 if (is_fixed_value_range() && range.is_fixed_value_range()) {
1012 std::set<CppType> result_values;
1013 set_intersection(_fixed_values.begin(), _fixed_values.end(), range._fixed_values.begin(),
1014 range._fixed_values.end(),
1015 std::inserter(result_values, result_values.begin()));
1016
1017 if (result_values.size() != 0) {
1018 return true;
1019 } else {
1020 return false;
1021 }
1022 } // 3.2
1023 else if (is_fixed_value_range() && !range.is_fixed_value_range()) {
1024 IteratorType iter = _fixed_values.begin();
1025
1026 while (iter != _fixed_values.end()) {
1027 if (range.is_in_range(*iter)) {
1028 return true;
1029 }
1030
1031 ++iter;
1032 }
1033
1034 return false;
1035 } else if (!is_fixed_value_range() && range.is_fixed_value_range()) {
1036 IteratorType iter = range._fixed_values.begin();
1037
1038 while (iter != range._fixed_values.end()) {
1039 if (this->is_in_range(*iter)) {
1040 return true;
1041 }
1042
1043 ++iter;
1044 }
1045
1046 return false;
1047 } else {
1048 if (_low_value > range._high_value || range._low_value > _high_value) {
1049 return false;
1050 } else if (_low_value == range._high_value) {
1051 if (FILTER_LARGER_OR_EQUAL == _low_op && FILTER_LESS_OR_EQUAL == range._high_op) {
1052 return true;
1053 } else {
1054 return false;
1055 }
1056 } else if (range._low_value == _high_value) {
1057 if (FILTER_LARGER_OR_EQUAL == range._low_op && FILTER_LESS_OR_EQUAL == _high_op) {
1058 return true;
1059 } else {
1060 return false;
1061 }
1062 } else {
1063 return true;
1064 }

1068template <PrimitiveType primitive_type>
1069Status OlapScanKeys::extend_scan_key(ColumnValueRange<primitive_type>& range,
1070 int32_t max_scan_key_num, bool* exact_value, bool* eos) {
1071 using CppType = typename PrimitiveTypeTraits<primitive_type>::CppType;
1072 using ConstIterator = typename std::set<CppType>::const_iterator;
1073
1074 // 1. clear ScanKey if some column range is empty
1075 if (range.is_empty_value_range()) {
1076 _begin_scan_keys.clear();
1077 _end_scan_keys.clear();
1078 return Status::OK();
1079 }
1080
1081 // 2. stop extend ScanKey when it's already extend a range value
1082 if (_has_range_value) {
1083 return Status::OK();
1084 }
1085
1086 //if a column doesn't have any predicate, we will try converting the range to fixed values
1087 auto scan_keys_size = _begin_scan_keys.empty() ? 1 : _begin_scan_keys.size();
1088 if (range.is_fixed_value_range()) {
1089 if (range.get_fixed_value_size() > max_scan_key_num / scan_keys_size) {
1090 if (range.is_range_value_convertible()) {
1091 range.convert_to_range_value();
1092 *exact_value = false;
1093 } else {
1094 return Status::OK();
1095 }
1096 }
1097 } else {
1098 if (_begin_scan_keys.empty() && range.is_fixed_value_convertible() && _is_convertible &&
1099 !range.is_reject_split_type()) {
1100 *eos |= range.convert_to_close_range(_begin_scan_keys, _end_scan_keys, _begin_include,
1101 _end_include);
1102
1103 if (range.convert_to_avg_range_value(_begin_scan_keys, _end_scan_keys, _begin_include,
1104 _end_include, max_scan_key_num)) {
1105 _has_range_value = true;
1106 }
1107 return Status::OK();
1108 }
1109 }
1110
1111 // extend ScanKey with MatchValueRange
1112 if (range.is_match_value_range() && _begin_scan_keys.empty()) {
1113 _begin_scan_keys.emplace_back();
1114 _begin_scan_keys.back().add_value(
1115 cast_to_string<primitive_type, CppType>(type_limit<CppType>::min(), 0));
1116 _end_scan_keys.emplace_back();
1117 _end_scan_keys.back().add_value(
1118 cast_to_string<primitive_type, CppType>(type_limit<CppType>::max(), 0));
1119 _begin_include = true;
1120 _end_include = true;
1121 *exact_value = false;
1122 // not empty, do nothing
1123 }
1124
1125 // 3.1 extend ScanKey with FixedValueRange
1126 if (range.is_fixed_value_range()) {
1127 // 3.1.1 construct num of fixed value ScanKey (begin_key == end_key)
1128 if (_begin_scan_keys.empty()) {
1129 auto fixed_value_set = range.get_fixed_value_set();
1130 ConstIterator iter = fixed_value_set.begin();
1131
1132 for (; iter != fixed_value_set.end(); ++iter) {
1133 _begin_scan_keys.emplace_back();
1134 _begin_scan_keys.back().add_value(
1135 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1136 _end_scan_keys.emplace_back();
1137 _end_scan_keys.back().add_value(
1138 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1139 }
1140
1141 if (range.contain_null()) {
1142 _begin_scan_keys.emplace_back();
1143 _begin_scan_keys.back().add_null();
1144 _end_scan_keys.emplace_back();
1145 _end_scan_keys.back().add_null();
1146 }
1147 } // 3.1.2 produces the Cartesian product of ScanKey and fixed_value
1148 else {
1149 auto fixed_value_set = range.get_fixed_value_set();
1150 int original_key_range_size = _begin_scan_keys.size();
1151
1152 for (int i = 0; i < original_key_range_size; ++i) {
1153 OlapTuple start_base_key_range = _begin_scan_keys[i];
1154 OlapTuple end_base_key_range = _end_scan_keys[i];
1155
1156 ConstIterator iter = fixed_value_set.begin();
1157
1158 for (; iter != fixed_value_set.end(); ++iter) {
1159 // alter the first ScanKey in original place
1160 if (iter == fixed_value_set.begin()) {
1161 _begin_scan_keys[i].add_value(
1162 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1163 _end_scan_keys[i].add_value(
1164 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1165 } // append follow ScanKey
1166 else {
1167 _begin_scan_keys.push_back(start_base_key_range);
1168 _begin_scan_keys.back().add_value(
1169 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1170 _end_scan_keys.push_back(end_base_key_range);
1171 _end_scan_keys.back().add_value(
1172 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1173 }
1174 }
1175
1176 if (range.contain_null()) {
1177 _begin_scan_keys.push_back(start_base_key_range);
1178 _begin_scan_keys.back().add_null();
1179 _end_scan_keys.push_back(end_base_key_range);
1180 _end_scan_keys.back().add_null();
1181 }
1182 }
1183 }
1184
1185 _begin_include = true;
1186 _end_include = true;
1187 } // Extend ScanKey with range value
1188 else {
1189 _has_range_value = true;
1190
1191 if (_begin_scan_keys.empty()) {
1192 _begin_scan_keys.emplace_back();
1193 _begin_scan_keys.back().add_value(cast_to_string<primitive_type, CppType>(
1194 range.get_range_min_value(), range.scale()),
1195 range.contain_null());
1196 _end_scan_keys.emplace_back();
1197 _end_scan_keys.back().add_value(cast_to_string<primitive_type, CppType>(
1198 range.get_range_max_value(), range.scale()));
1199 } else {
1200 for (int i = 0; i < _begin_scan_keys.size(); ++i) {
1201 _begin_scan_keys[i].add_value(cast_to_string<primitive_type, CppType>(
1202 range.get_range_min_value(), range.scale()),
1203 range.contain_null());
1204 }
1205
1206 for (int i = 0; i < _end_scan_keys.size(); ++i) {
1207 _end_scan_keys[i].add_value(cast_to_string<primitive_type, CppType>(
1208 range.get_range_max_value(), range.scale()));
1209 }
1210 }
1211 _begin_include = range.is_begin_include();
1212 _end_include = range.is_end_include();
1213 }
1214
1215 return Status::OK();
1216}
