39template <PrimitiveType primitive_type, class T>
40std::string cast_to_string(T value, int scale) {
41 if constexpr (primitive_type == TYPE_DECIMAL32) {
42 return ((vectorized::Decimal<int32_t>)value).to_string(scale);
43 } else if constexpr (primitive_type == TYPE_DECIMAL64) {
44 return ((vectorized::Decimal<int64_t>)value).to_string(scale);
45 } else if constexpr (primitive_type == TYPE_DECIMAL128I) {
46 return ((vectorized::Decimal<int128_t>)value).to_string(scale);
47 } else if constexpr (primitive_type == TYPE_TINYINT) {
48 return std::to_string(static_cast<int>(value));
49 } else if constexpr (primitive_type == TYPE_LARGEINT) {
50 return vectorized::int128_to_string(value);
51 } else if constexpr (primitive_type == TYPE_DATETIMEV2) {
52 doris::vectorized::DateV2Value<doris::vectorized::DateTimeV2ValueType> datetimev2_val =
53 static_cast<doris::vectorized::DateV2Value<doris::vectorized::DateTimeV2ValueType>>(
54 value);
55 char buf[30];
56 datetimev2_val.to_string(buf);
57 std::stringstream ss;
58 ss << buf;
59 return ss.str();
60 } else {
61 return boost::lexical_cast<std::string>(value);
62 }

177 void to_olap_filter(std::vector<TCondition>& filters) {
178 if (is_fixed_value_range() || is_match_value_range()) {
179 // 1. convert to in filter condition
180 if (is_fixed_value_range()) {
181 to_in_condition(filters, true);
182 }
183
184 if (is_match_value_range()) {
185 to_match_condition(filters);
186 }
187 } else if (_low_value < _high_value) {
188 // 2. convert to min max filter condition
189 TCondition null_pred;
190 if (TYPE_MAX == _high_value && _high_op == FILTER_LESS_OR_EQUAL &&
191 TYPE_MIN == _low_value && _low_op == FILTER_LARGER_OR_EQUAL && _is_nullable_col &&
192 !contain_null()) {
193 null_pred.__set_column_name(_column_name);
194 null_pred.__set_condition_op("is");
195 null_pred.condition_values.emplace_back("not null");
196 }
197
198 if (null_pred.condition_values.size() != 0) {
199 filters.push_back(null_pred);
200 return;
201 }
202
203 TCondition low;
204 if (TYPE_MIN != _low_value || FILTER_LARGER_OR_EQUAL != _low_op) {
205 low.__set_column_name(_column_name);
206 low.__set_condition_op((_low_op == FILTER_LARGER_OR_EQUAL ? ">=" : ">>"));
207 low.__set_marked_by_runtime_filter(_marked_runtime_filter_predicate);
208 low.condition_values.push_back(
209 cast_to_string<primitive_type, CppType>(_low_value, _scale));
210 }
211
212 if (low.condition_values.size() != 0) {
213 filters.push_back(low);
214 }
215
216 TCondition high;
217 if (TYPE_MAX != _high_value || FILTER_LESS_OR_EQUAL != _high_op) {
218 high.__set_column_name(_column_name);
219 high.__set_condition_op((_high_op == FILTER_LESS_OR_EQUAL ? "<=" : "<<"));
220 high.__set_marked_by_runtime_filter(_marked_runtime_filter_predicate);
221 high.condition_values.push_back(
222 cast_to_string<primitive_type, CppType>(_high_value, _scale));
223 }
224
225 if (high.condition_values.size() != 0) {
226 filters.push_back(high);
227 }
228 } else {
229 // 3. convert to is null and is not null filter condition
230 TCondition null_pred;
231 if (TYPE_MAX == _low_value && TYPE_MIN == _high_value && _is_nullable_col &&
232 contain_null()) {
233 null_pred.__set_column_name(_column_name);
234 null_pred.__set_condition_op("is");
235 null_pred.condition_values.emplace_back("null");
236 }
237
238 if (null_pred.condition_values.size() != 0) {
239 filters.push_back(null_pred);
240 }
241 }

244 void to_in_condition(std::vector<TCondition>& filters, bool is_in = true) {
245 TCondition condition;
246 condition.__set_column_name(_column_name);
247 condition.__set_condition_op(is_in ? "*=" : "!*=");
248 condition.__set_marked_by_runtime_filter(_marked_runtime_filter_predicate);
249
250 for (const auto& value : _fixed_values) {
251 condition.condition_values.push_back(
252 cast_to_string<primitive_type, CppType>(value, _scale));
253 }
254
255 if (condition.condition_values.size() != 0) {
256 filters.push_back(condition);
257 }

260 void to_condition_in_compound(std::vector<TCondition>& filters) {
261 for (const auto& value : _compound_values) {
262 TCondition condition;
263 condition.__set_column_name(_column_name);
264 if (value.first == FILTER_LARGER) {
265 condition.__set_condition_op(">>");
266 } else if (value.first == FILTER_LARGER_OR_EQUAL) {
267 condition.__set_condition_op(">=");
268 } else if (value.first == FILTER_LESS) {
269 condition.__set_condition_op("<<");
270 } else if (value.first == FILTER_LESS_OR_EQUAL) {
271 condition.__set_condition_op("<=");
272 } else if (value.first == FILTER_IN) {
273 condition.__set_condition_op("*=");
274 } else if (value.first == FILTER_NOT_IN) {
275 condition.__set_condition_op("!*=");
276 }
277 condition.condition_values.push_back(
278 cast_to_string<primitive_type, CppType>(value.second, _scale));
279 if (condition.condition_values.size() != 0) {
280 filters.push_back(condition);
281 }
282 }

285 void to_match_condition(std::vector<TCondition>& filters) {
286 for (const auto& value : _match_values) {
287 TCondition condition;
288 condition.__set_column_name(_column_name);
289
290 if (value.first == MatchType::MATCH_ANY) {
291 condition.__set_condition_op("match_any");
292 } else if (value.first == MatchType::MATCH_ALL) {
293 condition.__set_condition_op("match_all");
294 } else if (value.first == MatchType::MATCH_PHRASE) {
295 condition.__set_condition_op("match_phrase");
296 } else if (value.first == MatchType::MATCH_ELEMENT_EQ) {
297 condition.__set_condition_op("match_element_eq");
298 } else if (value.first == MatchType::MATCH_ELEMENT_LT) {
299 condition.__set_condition_op("match_element_lt");
300 } else if (value.first == MatchType::MATCH_ELEMENT_GT) {
301 condition.__set_condition_op("match_element_gt");
302 } else if (value.first == MatchType::MATCH_ELEMENT_LE) {
303 condition.__set_condition_op("match_element_le");
304 } else if (value.first == MatchType::MATCH_ELEMENT_GE) {
305 condition.__set_condition_op("match_element_ge");
306 }
307 condition.condition_values.push_back(
308 cast_to_string<primitive_type, CppType>(value.second, _scale));
309 if (condition.condition_values.size() != 0) {
310 filters.push_back(condition);
311 }

315 void set_whole_value_range() {
316 _fixed_values.clear();
317 _low_value = TYPE_MIN;
318 _high_value = TYPE_MAX;
319 _low_op = FILTER_LARGER_OR_EQUAL;
320 _high_op = FILTER_LESS_OR_EQUAL;

324 bool is_whole_value_range() const {
325 DCHECK(_is_nullable_col || !contain_null())
326 << "Non-nullable column cannot contains null value";
327
328 return _fixed_values.empty() && _low_value == TYPE_MIN && _high_value == TYPE_MAX &&
329 _low_op == FILTER_LARGER_OR_EQUAL && _high_op == FILTER_LESS_OR_EQUAL &&
330 _is_nullable_col == contain_null();
331 }
332
333 // only two case will set range contain null, call by temp_range in olap scan node
334 // 'is null' and 'is not null'
335 // 1. if the pred is 'is null' means the range should be
336 // empty in fixed_range and _high_value < _low_value
337 // 2. if the pred is 'is not null' means the range should be whole range and
338 // 'is not null' be effective
339 void set_contain_null(bool contain_null) {
340 if (contain_null) {
341 set_empty_value_range();
342 } else {
343 set_whole_value_range();
344 }
345 _contain_null = _is_nullable_col && contain_null;
346 }

710template <PrimitiveType primitive_type>
711bool ColumnValueRange<primitive_type>::convert_to_avg_range_value(
712 std::vector<OlapTuple>& begin_scan_keys, std::vector<OlapTuple>& end_scan_keys,
713 bool& begin_include, bool& end_include, int32_t max_scan_key_num) {
714 if constexpr (!_is_reject_split_type) {
715 auto no_split = [&]() -> bool {
716 begin_scan_keys.emplace_back();
717 begin_scan_keys.back().add_value(
718 cast_to_string<primitive_type, CppType>(get_range_min_value(), scale()),
719 contain_null());
720 end_scan_keys.emplace_back();
721 end_scan_keys.back().add_value(
722 cast_to_string<primitive_type, CppType>(get_range_max_value(), scale()));
723 return true;
724 };
725
726 CppType min_value = get_range_min_value();
727 CppType max_value = get_range_max_value();
728 if constexpr (primitive_type == PrimitiveType::TYPE_DATE) {
729 min_value.set_type(TimeType::TIME_DATE);
730 max_value.set_type(TimeType::TIME_DATE);
731 }
732
733 if (min_value > max_value || max_scan_key_num == 1) {
734 return no_split();
735 }
736
737 auto cast = [](const CppType& value) {
738 if constexpr (primitive_type == PrimitiveType::TYPE_DATE ||
739 primitive_type == PrimitiveType::TYPE_DATEV2) {
740 return value;
741 } else {
742 return (int128_t)value;
743 }
744 };
745
746 // When CppType is date, we can not convert it to integer number and calculate distance.
747 // In other case, we convert element to int128 to avoit overflow.
748 size_t step_size = (cast(max_value) - min_value) / max_scan_key_num;
749
750 constexpr size_t MAX_STEP_SIZE = 1 << 20;
751 // When the step size is too large, the range is easy to not really contain data.
752 if (step_size > MAX_STEP_SIZE) {
753 return no_split();
754 }
755
756 // Add null key if contain null, must do after no_split check
757 if (contain_null()) {
758 begin_scan_keys.emplace_back();
759 begin_scan_keys.back().add_null();
760 end_scan_keys.emplace_back();
761 end_scan_keys.back().add_null();
762 }
763 while (true) {
764 begin_scan_keys.emplace_back();
765 begin_scan_keys.back().add_value(
766 cast_to_string<primitive_type, CppType>(min_value, scale()));
767
768 if (cast(max_value) - min_value < step_size) {
769 min_value = max_value;
770 } else {
771 min_value += step_size;
772 }
773
774 end_scan_keys.emplace_back();
775 end_scan_keys.back().add_value(
776 cast_to_string<primitive_type, CppType>(min_value, scale()));
777
778 if (min_value == max_value) {
779 break;
780 }
781 ++min_value;
782 }
783
784 return step_size != 0;
785 }
786 return false;

804template <PrimitiveType primitive_type>
805Status ColumnValueRange<primitive_type>::add_range(SQLFilterOp op, CppType value) {
806 if (INVALID_TYPE == _column_type) {
807 return Status::InternalError("AddRange failed, Invalid type");
808 }
809
810 // add range means range should not contain null
811 _contain_null = false;
812
813 if (is_fixed_value_range()) {
814 std::pair<IteratorType, IteratorType> bound_pair = _fixed_values.equal_range(value);
815
816 switch (op) {
817 case FILTER_LARGER: {
818 _fixed_values.erase(_fixed_values.begin(), bound_pair.second);
819 break;
820 }
821
822 case FILTER_LARGER_OR_EQUAL: {
823 _fixed_values.erase(_fixed_values.begin(), bound_pair.first);
824 break;
825 }
826
827 case FILTER_LESS: {
828 if (bound_pair.first == _fixed_values.find(value)) {
829 _fixed_values.erase(bound_pair.first, _fixed_values.end());
830 } else {
831 _fixed_values.erase(bound_pair.second, _fixed_values.end());
832 }
833
834 break;
835 }
836
837 case FILTER_LESS_OR_EQUAL: {
838 _fixed_values.erase(bound_pair.second, _fixed_values.end());
839 break;
840 }
841
842 default: {
843 return Status::InternalError("Add Range fail! Unsupported SQLFilterOp.");
844 }
845 }
846
847 _high_value = TYPE_MIN;
848 _low_value = TYPE_MAX;
849 } else {
850 if (_high_value > _low_value) {
851 switch (op) {
852 case FILTER_LARGER: {
853 if (value >= _low_value) {
854 _low_value = value;
855 _low_op = op;
856 }
857
858 break;
859 }
860
861 case FILTER_LARGER_OR_EQUAL: {
862 if (value > _low_value) {
863 _low_value = value;
864 _low_op = op;
865 }
866
867 break;
868 }
869
870 case FILTER_LESS: {
871 if (value <= _high_value) {
872 _high_value = value;
873 _high_op = op;
874 }
875
876 break;
877 }
878
879 case FILTER_LESS_OR_EQUAL: {
880 if (value < _high_value) {
881 _high_value = value;
882 _high_op = op;
883 }
884
885 break;
886 }
887
888 default: {
889 return Status::InternalError("Add Range fail! Unsupported SQLFilterOp.");
890 }
891 }
892 }
893
894 if (FILTER_LARGER_OR_EQUAL == _low_op && FILTER_LESS_OR_EQUAL == _high_op &&
895 _high_value == _low_value) {
896 add_fixed_value(_high_value);
897 _high_value = TYPE_MIN;
898 _low_value = TYPE_MAX;
899 }
900 }
901
902 return Status::OK();
903}

1083template <PrimitiveType primitive_type>
1084Status OlapScanKeys::extend_scan_key(ColumnValueRange<primitive_type>& range,
1085 int32_t max_scan_key_num, bool* exact_value, bool* eos) {
1086 using CppType = typename PrimitiveTypeTraits<primitive_type>::CppType;
1087 using ConstIterator = typename std::set<CppType>::const_iterator;
1088
1089 // 1. clear ScanKey if some column range is empty
1090 if (range.is_empty_value_range()) {
1091 _begin_scan_keys.clear();
1092 _end_scan_keys.clear();
1093 return Status::OK();
1094 }
1095
1096 // 2. stop extend ScanKey when it's already extend a range value
1097 if (_has_range_value) {
1098 return Status::OK();
1099 }
1100
1101 //if a column doesn't have any predicate, we will try converting the range to fixed values
1102 auto scan_keys_size = _begin_scan_keys.empty() ? 1 : _begin_scan_keys.size();
1103 if (range.is_fixed_value_range()) {
1104 if (range.get_fixed_value_size() > max_scan_key_num / scan_keys_size) {
1105 if (range.is_range_value_convertible()) {
1106 range.convert_to_range_value();
1107 *exact_value = false;
1108 } else {
1109 return Status::OK();
1110 }
1111 }
1112 } else {
1113 if (_begin_scan_keys.empty() && range.is_fixed_value_convertible() && _is_convertible &&
1114 !range.is_reject_split_type()) {
1115 *eos |= range.convert_to_close_range(_begin_scan_keys, _end_scan_keys, _begin_include,
1116 _end_include);
1117
1118 if (range.convert_to_avg_range_value(_begin_scan_keys, _end_scan_keys, _begin_include,
1119 _end_include, max_scan_key_num)) {
1120 _has_range_value = true;
1121 }
1122 return Status::OK();
1123 }
1124 }
1125
1126 // extend ScanKey with MatchValueRange
1127 if (range.is_match_value_range() && _begin_scan_keys.empty()) {
1128 _begin_scan_keys.emplace_back();
1129 _begin_scan_keys.back().add_value(
1130 cast_to_string<primitive_type, CppType>(type_limit<CppType>::min(), 0));
1131 _end_scan_keys.emplace_back();
1132 _end_scan_keys.back().add_value(
1133 cast_to_string<primitive_type, CppType>(type_limit<CppType>::max(), 0));
1134 _begin_include = true;
1135 _end_include = true;
1136 *exact_value = false;
1137 // not empty, do nothing
1138 return Status::OK();
1139 }
1140
1141 // 3.1 extend ScanKey with FixedValueRange
1142 if (range.is_fixed_value_range()) {
1143 // 3.1.1 construct num of fixed value ScanKey (begin_key == end_key)
1144 if (_begin_scan_keys.empty()) {
1145 auto fixed_value_set = range.get_fixed_value_set();
1146 ConstIterator iter = fixed_value_set.begin();
1147
1148 for (; iter != fixed_value_set.end(); ++iter) {
1149 _begin_scan_keys.emplace_back();
1150 _begin_scan_keys.back().add_value(
1151 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1152 _end_scan_keys.emplace_back();
1153 _end_scan_keys.back().add_value(
1154 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1155 }
1156
1157 if (range.contain_null()) {
1158 _begin_scan_keys.emplace_back();
1159 _begin_scan_keys.back().add_null();
1160 _end_scan_keys.emplace_back();
1161 _end_scan_keys.back().add_null();
1162 }
1163 } // 3.1.2 produces the Cartesian product of ScanKey and fixed_value
1164 else {
1165 auto fixed_value_set = range.get_fixed_value_set();
1166 int original_key_range_size = _begin_scan_keys.size();
1167
1168 for (int i = 0; i < original_key_range_size; ++i) {
1169 OlapTuple start_base_key_range = _begin_scan_keys[i];
1170 OlapTuple end_base_key_range = _end_scan_keys[i];
1171
1172 ConstIterator iter = fixed_value_set.begin();
1173
1174 for (; iter != fixed_value_set.end(); ++iter) {
1175 // alter the first ScanKey in original place
1176 if (iter == fixed_value_set.begin()) {
1177 _begin_scan_keys[i].add_value(
1178 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1179 _end_scan_keys[i].add_value(
1180 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1181 } // append follow ScanKey
1182 else {
1183 _begin_scan_keys.push_back(start_base_key_range);
1184 _begin_scan_keys.back().add_value(
1185 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1186 _end_scan_keys.push_back(end_base_key_range);
1187 _end_scan_keys.back().add_value(
1188 cast_to_string<primitive_type, CppType>(*iter, range.scale()));
1189 }
1190 }
1191
1192 if (range.contain_null()) {
1193 _begin_scan_keys.push_back(start_base_key_range);
1194 _begin_scan_keys.back().add_null();
1195 _end_scan_keys.push_back(end_base_key_range);
1196 _end_scan_keys.back().add_null();
1197 }
1198 }
1199 }
1200
1201 _begin_include = true;
1202 _end_include = true;
1203 } // Extend ScanKey with range value
1204 else {
1205 _has_range_value = true;
1206
1208 if (range.get_range_max_value() < range.get_range_min_value()) {
1209 CHECK(range.contain_null());
1210 if (_begin_scan_keys.empty()) {
1211 _begin_scan_keys.emplace_back();
1212 _begin_scan_keys.back().add_null();
1213 _end_scan_keys.emplace_back();
1214 _end_scan_keys.back().add_null();
1215 } else {
1216 for (int i = 0; i < _begin_scan_keys.size(); ++i) {
1217 _begin_scan_keys[i].add_null();
1218 _end_scan_keys[i].add_null();
1219 }
1220 }
1221 } else if (_begin_scan_keys.empty()) {
1222 _begin_scan_keys.emplace_back();
1223 _begin_scan_keys.back().add_value(cast_to_string<primitive_type, CppType>(
1224 range.get_range_min_value(), range.scale()),
1225 range.contain_null());
1226 _end_scan_keys.emplace_back();
1227 _end_scan_keys.back().add_value(cast_to_string<primitive_type, CppType>(
1228 range.get_range_max_value(), range.scale()));
1229 } else {
1230 for (int i = 0; i < _begin_scan_keys.size(); ++i) {
1231 _begin_scan_keys[i].add_value(cast_to_string<primitive_type, CppType>(
1232 range.get_range_min_value(), range.scale()),
1233 range.contain_null());
1234 }
1235
1236 for (int i = 0; i < _end_scan_keys.size(); ++i) {
1237 _end_scan_keys[i].add_value(cast_to_string<primitive_type, CppType>(
1238 range.get_range_max_value(), range.scale()));
1239 }
1240 }
1241 _begin_include = range.is_begin_include();
1242 _end_include = range.is_end_include();
1243 }
1244
1245 return Status::OK();
