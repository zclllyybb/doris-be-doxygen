71// @brief 切分字符串
72// @param base 原串
73// @param separator 分隔符
74// @param result 切分结果
75template <typename T>
76Status split_string(const std::string& base, const T separator, std::vector<std::string>* result) {
77 if (!result) {
78 return Status::Error<ErrorCode::INVALID_ARGUMENT>();
79 }
80
81 // 处理base为空的情况
82 // 在删除功能中，当varchar类型列的过滤条件为空时，会出现这种情况
83 if (base.size() == 0) {
84 result->push_back("");
85 return Status::OK();
86 }
87
88 size_t offset = 0;
89 while (offset < base.length()) {
90 size_t next = base.find(separator, offset);
91 if (next == std::string::npos) {
92 result->push_back(base.substr(offset));
93 break;
94 } else {
95 result->push_back(base.substr(offset, next - offset));
96 offset = next + 1;
97 }
98 }
99
100 return Status::OK();
101}

187// 检查int8_t, int16_t, int32_t, int64_t的值是否溢出
188template <typename T>
189bool valid_signed_number(const std::string& value_str) {
190 char* endptr = nullptr;
191 errno = 0;
192 int64_t value = strtol(value_str.c_str(), &endptr, 10);
193
194 if ((errno == ERANGE && (value == LONG_MAX || value == LONG_MIN)) ||
195 (errno != 0 && value == 0) || endptr == value_str || *endptr != '\0') {
196 return false;
197 }
198
199 if (value < std::numeric_limits<T>::min() || value > std::numeric_limits<T>::max()) {
200 return false;
201 }
202
203 return true;
204}

210template <typename T>
211bool valid_unsigned_number(const std::string& value_str) {
212 if (value_str[0] == '-') {
213 return false;
214 }
215
216 char* endptr = nullptr;
217 errno = 0;
218 uint64_t value = strtoul(value_str.c_str(), &endptr, 10);
219
220 if ((errno == ERANGE && (value == ULONG_MAX)) || (errno != 0 && value == 0) ||
221 endptr == value_str || *endptr != '\0') {
222 return false;
223 }
224
225 if (value < std::numeric_limits<T>::min() || value > std::numeric_limits<T>::max()) {
226 return false;
227 }

229 return true;
