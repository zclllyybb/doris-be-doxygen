69template <class Traits>
70class PredicateFunctionCreator {
71public:
72 template <PrimitiveType type>
73 static typename Traits::BasePtr create() {
74 return Traits::template get_function<type>();

97template <class Traits>
98typename Traits::BasePtr create_predicate_function(PrimitiveType type) {
99 using Creator = PredicateFunctionCreator<Traits>;
100
101 switch (type) {
102 case TYPE_BOOLEAN: {
103 return Creator::template create<TYPE_BOOLEAN>();
104 }
105 case TYPE_DECIMALV2: {
106 return Creator::template create<TYPE_DECIMALV2>();
107 }
108#define M(NAME) \
109 case NAME: { \
110 return Creator::template create<NAME>(); \
111 }
112 APPLY_FOR_PRIMTYPE(M)
113#undef M
114 default:
115 DCHECK(false) << "Invalid type.";
116 }
117
118 return nullptr;
119}

136 }

139}

189template <typename T>
190inline ColumnPredicate* create_column_predicate(uint32_t column_id,
191 const std::shared_ptr<T>& filter, FieldType type,
192 int be_exec_version,
193 const TabletColumn* column = nullptr) {
194 switch (type) {
195#define M(NAME) \
196 case OLAP_FIELD_##NAME: { \
197 return create_olap_column_predicate<NAME>(column_id, filter, be_exec_version, column); \
198 }
199 APPLY_FOR_PRIMTYPE(M)
200#undef M
201 case OLAP_FIELD_TYPE_DECIMAL: {
202 return create_olap_column_predicate<TYPE_DECIMALV2>(column_id, filter, be_exec_version,
203 column);
204 }
205 case OLAP_FIELD_TYPE_BOOL: {
206 return create_olap_column_predicate<TYPE_BOOLEAN>(column_id, filter, be_exec_version,
207 column);
208 }
209 default:
210 return nullptr;
211 }
