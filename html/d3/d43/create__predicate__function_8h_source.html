107template <class Traits, size_t N = 0>
108typename Traits::BasePtr create_predicate_function(PrimitiveType type) {
109 using Creator = PredicateFunctionCreator<Traits>;
110
111 switch (type) {
112 case TYPE_BOOLEAN: {
113 return Creator::template create<TYPE_BOOLEAN, N>();
114 }
115 case TYPE_DECIMALV2: {
116 return Creator::template create<TYPE_DECIMALV2, N>();
117 }
118#define M(NAME) \
119 case NAME: { \
120 return Creator::template create<NAME, N>(); \
121 }
122 APPLY_FOR_PRIMTYPE(M)
123#undef M
124 default:
125 DCHECK(false) << "Invalid type.";
126 }
127
128 return nullptr;
129}

270template <typename T>
271ColumnPredicate* create_column_predicate(uint32_t column_id, const std::shared_ptr<T>& filter,
272 FieldType type, int be_exec_version,
273 const TabletColumn* column = nullptr) {
274 switch (type) {
275#define M(NAME) \
276 case OLAP_FIELD_##NAME: { \
277 return create_olap_column_predicate<NAME>(column_id, filter, be_exec_version, column); \
278 }
279 APPLY_FOR_PRIMTYPE(M)
280#undef M
281 case OLAP_FIELD_TYPE_DECIMAL: {
282 return create_olap_column_predicate<TYPE_DECIMALV2>(column_id, filter, be_exec_version,
283 column);
284 }
285 case OLAP_FIELD_TYPE_BOOL: {
286 return create_olap_column_predicate<TYPE_BOOLEAN>(column_id, filter, be_exec_version,
287 column);
288 }
289 default:
290 return nullptr;
291 }
