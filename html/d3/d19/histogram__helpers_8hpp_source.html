55template <typename T>
56bool can_assign_into_buckets(const std::map<T, size_t>& value_map, const size_t max_bucket_size,
57 const size_t num_buckets) {
58 if (value_map.empty()) {
59 return false;
60 };
61
62 size_t used_buckets = 1;
63 size_t current_bucket_size = 0;
64
65 for (const auto& [value, count] : value_map) {
66 current_bucket_size += count;
67
68 // If adding the current value to the current bucket would exceed max_bucket_size,
69 // then we start a new bucket.
70 if (current_bucket_size > max_bucket_size) {
71 ++used_buckets;
72 current_bucket_size = count;
73 }
74
75 // If we have used more buckets than num_buckets, we cannot assign the values to buckets.
76 if (used_buckets > num_buckets) {
77 return false;
78 }
79 }
80
81 return true;
82}

93template <typename T>
94size_t calculate_bucket_max_values(const std::map<T, size_t>& value_map, const size_t num_buckets) {
95 // Ensure that the value map is not empty
96 assert(!value_map.empty());
97
98 // Calculate the total number of values in the map using std::accumulate()
99 size_t total_values = 0;
100 for (const auto& [value, count] : value_map) {
101 total_values += count;
102 }
103
104 // If there is only one bucket, then all values will be assigned to that bucket
105 if (num_buckets == 1) {
106 return total_values;
107 }
108
109 // To calculate the maximum value count in each bucket, we first calculate a conservative upper
110 // bound, which is equal to 2 * total_values / (max_buckets - 1) + 1. This upper bound may exceed
111 // the actual maximum value count, but it does not underestimate it. The subsequent binary search
112 // algorithm will approach the actual maximum value count.
113 size_t upper_bucket_values = 2 * total_values / (num_buckets - 1) + 1;
114
115 // Initialize the lower bound to 0
116 size_t lower_bucket_values = 0;
117
118 // Perform a binary search to find the maximum number of values that can fit into each bucket
119 int search_step = 0;
120 const int max_search_steps =
121 10; // Limit the number of search steps to avoid excessive iteration
122
123 while (upper_bucket_values > lower_bucket_values + 1 && search_step < max_search_steps) {
124 // Calculate the midpoint of the upper and lower bounds
125 const size_t bucket_values = (upper_bucket_values + lower_bucket_values) / 2;
126
127 // Check if the given number of values can be assigned to the desired number of buckets
128 if (can_assign_into_buckets(value_map, bucket_values, num_buckets)) {
129 // If it can, then set the upper bound to the midpoint
130 upper_bucket_values = bucket_values;
131 } else {
132 // If it can't, then set the lower bound to the midpoint
133 lower_bucket_values = bucket_values;
134 }
135 // Increment the search step counter
136 ++search_step;
137 }
138
139 return upper_bucket_values;
140}

173template <typename T>
174bool build_histogram(std::vector<Bucket<T>>& buckets, const std::map<T, size_t>& ordered_map,
175 const size_t max_num_buckets) {
176 // If the input map is empty, there is nothing to build.
177 if (ordered_map.empty()) {
178 return false;
179 }
180
181 // Calculate the maximum number of values that can be assigned to each bucket.
182 auto bucket_max_values = calculate_bucket_max_values(ordered_map, max_num_buckets);
183
184 // Ensure that the capacity is at least max_num_buckets in order to avoid the overhead of additional
185 // allocations when inserting buckets.
186 buckets.clear();
187 buckets.reserve(max_num_buckets);
188
189 // Initialize bucket variables.
190 size_t distinct_values_count = 0;
191 size_t values_count = 0;
192 size_t cumulative_values = 0;
193
194 // Record how many values still need to be assigned.
195 auto remaining_distinct_values = ordered_map.size();
196
197 auto it = ordered_map.begin();
198
199 // Lower value of the current bucket.
200 const T* lower_value = &it->first;
201
202 // Iterate over the ordered map of distinct values and their counts.
203 for (; it != ordered_map.end(); ++it) {
204 const auto count = it->second;
205 const auto current_value = it->first;
206
207 // Update the bucket counts and track the number of distinct values assigned.
208 distinct_values_count++;
209 remaining_distinct_values--;
210 values_count += count;
211 cumulative_values += count;
212
213 // Check whether the current value should be added to the current bucket.
214 auto next = std::next(it);
215 size_t remaining_empty_buckets = max_num_buckets - buckets.size() - 1;
216
217 if (next != ordered_map.end() && remaining_distinct_values > remaining_empty_buckets &&
218 values_count + next->second <= bucket_max_values) {
219 // If the current value is the last in the input map and there are more remaining
220 // distinct values than empty buckets and adding the value does not cause the bucket
221 // to exceed its max size, skip adding the value to the current bucket.
222 continue;
223 }
224
225 // Finalize the current bucket and add it to our collection of buckets.
226 auto pre_sum = cumulative_values - values_count;
227
228 Bucket<T> new_bucket(*lower_value, current_value, distinct_values_count, values_count,
229 pre_sum);
230 buckets.push_back(new_bucket);
231
232 // Reset variables for the next bucket.
233 if (next != ordered_map.end()) {
234 lower_value = &next->first;
235 }
236 values_count = 0;
237 distinct_values_count = 0;
238 }
239
240 return true;
241}

243template <typename T>
244bool histogram_to_json(rapidjson::StringBuffer& buffer, const std::vector<Bucket<T>>& buckets,
245 const DataTypePtr& data_type) {
246 rapidjson::Document doc;
247 doc.SetObject();
248 rapidjson::Document::AllocatorType& allocator = doc.GetAllocator();
249
250 int num_buckets = buckets.size();
251 doc.AddMember("num_buckets", num_buckets, allocator);
252
253 rapidjson::Value bucket_arr(rapidjson::kArrayType);
254 bucket_arr.Reserve(num_buckets, allocator);
255
256 std::stringstream ss1;
257 std::stringstream ss2;
258
259 rapidjson::Value lower_val;
260 rapidjson::Value upper_val;
261
262 // Convert bucket's lower and upper to 2 columns
263 MutableColumnPtr lower_column = data_type->create_column();
264 MutableColumnPtr upper_column = data_type->create_column();
265 for (const auto& bucket : buckets) {
266 // String type is different, it has to pass in length
267 if constexpr (std::is_same_v<T, std::string>) {
268 lower_column->insert_data(bucket.lower.c_str(), bucket.lower.length());
269 upper_column->insert_data(bucket.upper.c_str(), bucket.upper.length());
270 } else {
271 lower_column->insert_data(reinterpret_cast<const char*>(&bucket.lower), 0);
272 upper_column->insert_data(reinterpret_cast<const char*>(&bucket.upper), 0);
273 }
274 }
275 size_t row_num = 0;
276 for (const auto& bucket : buckets) {
277 std::string lower_str = data_type->to_string(*lower_column, row_num);
278 std::string upper_str = data_type->to_string(*upper_column, row_num);
279 ++row_num;
280 lower_val.SetString(lower_str.data(), static_cast<rapidjson::SizeType>(lower_str.size()),
281 allocator);
282 upper_val.SetString(upper_str.data(), static_cast<rapidjson::SizeType>(upper_str.size()),
283 allocator);
284
285 rapidjson::Value bucket_json(rapidjson::kObjectType);
286 bucket_json.AddMember("lower", lower_val, allocator);
287 bucket_json.AddMember("upper", upper_val, allocator);
288 bucket_json.AddMember("ndv", static_cast<int64_t>(bucket.ndv), allocator);
289 bucket_json.AddMember("count", static_cast<int64_t>(bucket.count), allocator);
290 bucket_json.AddMember("pre_sum", static_cast<int64_t>(bucket.pre_sum), allocator);
291
292 bucket_arr.PushBack(bucket_json, allocator);
293 }
294
295 doc.AddMember("buckets", bucket_arr, allocator);
296 rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
297 doc.Accept(writer);
298
299 return !buckets.empty() && buffer.GetSize() > 0;
300}
