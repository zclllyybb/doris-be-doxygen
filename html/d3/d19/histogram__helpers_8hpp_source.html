47template <typename T>
48bool build_bucket_from_data(std::vector<Bucket<T>>& buckets,
49 const std::map<T, uint64_t>& ordered_map, double sample_rate,
50 uint32_t max_bucket_num) {
51 if (ordered_map.size() == 0) {
52 return false;
53 }
54
55 uint64_t element_number = 0;
56 for (auto it : ordered_map) {
57 element_number += it.second;
58 }
59
60 auto sample_number = (uint64_t)std::ceil(element_number * sample_rate);
61 auto num_per_bucket = (uint64_t)std::ceil((Float64)sample_number / max_bucket_num);
62
63 LOG(INFO) << fmt::format(
64 "histogram bucket info: element number {}, sample number:{}, num per bucket:{}",
65 element_number, sample_number, num_per_bucket);
66
67 if (sample_rate == 1) {
68 for (auto it : ordered_map) {
69 for (auto i = it.second; i > 0; --i) {
70 auto v = it.first;
71 value_to_bucket(buckets, v, num_per_bucket);
72 }
73 }
74 return true;
75 }
76
77 // if sampling is required (0<sample_rate<1),
78 // we need to build the sampled data index
79 boost::dynamic_bitset<> sample_index(element_number);
80
81 // use a same seed value so that we get
82 // same result each time we run this function
83 srand(element_number * sample_rate * max_bucket_num);
84
85 while (sample_index.count() < sample_number) {
86 uint64_t num = (rand() % (element_number));
87 sample_index[num] = true;
88 }
89
90 uint64_t element_cnt = 0;
91 uint64_t sample_cnt = 0;
92 bool break_flag = false;
93
94 for (auto it : ordered_map) {
95 if (break_flag) {
96 break;
97 }
98 for (auto i = it.second; i > 0; --i) {
99 if (sample_cnt >= sample_number) {
100 break_flag = true;
101 break;
102 }
103 if (sample_index[element_cnt]) {
104 sample_cnt += 1;
105 auto v = it.first;
106 value_to_bucket(buckets, v, num_per_bucket);
107 }
108 element_cnt += 1;
109 }
110 }
111
112 return true;
113}

115bool inline bucket_to_json(rapidjson::Document::AllocatorType& allocator,
116 rapidjson::Value& bucket_arr, std::string lower, std::string upper,
117 int64 count, int64 pre_sum, int64 ndv) {
118 rapidjson::Value bucket(rapidjson::kObjectType);
119
120 rapidjson::Value lower_val(lower.c_str(), allocator);
121 bucket.AddMember("lower", lower_val, allocator);
122
123 rapidjson::Value upper_val(upper.c_str(), allocator);
124 bucket.AddMember("upper", upper_val, allocator);
125
126 rapidjson::Value count_val(count);
127 bucket.AddMember("count", count_val, allocator);
128
129 rapidjson::Value pre_sum_val(pre_sum);
130 bucket.AddMember("pre_sum", pre_sum_val, allocator);
131
132 rapidjson::Value ndv_val(ndv);
133 bucket.AddMember("ndv", ndv_val, allocator);
134
135 bucket_arr.PushBack(bucket, allocator);
136
137 return bucket_arr.Size() > 0;
138}

140template <typename T>
141bool value_to_bucket(std::vector<Bucket<T>>& buckets, T v, size_t num_per_bucket) {
142 if (buckets.empty()) {
143 Bucket<T> bucket(v, 0);
144 buckets.emplace_back(bucket);
145 } else {
146 Bucket<T>* bucket = &buckets.back();
147 T upper = bucket->upper;
148 if (upper == v) {
149 bucket->count++;
150 } else if (bucket->count < num_per_bucket) {
151 bucket->count++;
152 bucket->ndv++;
153 bucket->upper = v;
154 } else {
155 uint64_t pre_sum = bucket->pre_sum + bucket->count;
156 Bucket<T> new_bucket(v, pre_sum);
157 buckets.emplace_back(new_bucket);
158 }
159 }
160
161 return buckets.size() > 0;
162}

164template <typename T>
165bool value_to_string(std::stringstream& ss, T input, const DataTypePtr& data_type) {
166 fmt::memory_buffer _insert_stmt_buffer;
167 switch (data_type->get_type_id()) {
168 case TypeIndex::Int8:
169 case TypeIndex::UInt8:
170 case TypeIndex::Int16:
171 case TypeIndex::UInt16:
172 case TypeIndex::Int32:
173 case TypeIndex::UInt32:
174 case TypeIndex::Int64:
175 case TypeIndex::UInt64:
176 case TypeIndex::Int128:
177 case TypeIndex::UInt128:
178 case TypeIndex::Float32:
179 case TypeIndex::Float64:
180 case TypeIndex::String: {
181 fmt::memory_buffer buffer;
182 fmt::format_to(buffer, "{}", input);
183 ss << std::string(buffer.data(), buffer.size());
184 break;
185 }
186 case TypeIndex::Decimal32: {
187 auto scale = get_decimal_scale(*data_type);
188 auto decimal_val = reinterpret_cast<const Decimal32*>(&input);
189 write_text(*decimal_val, scale, ss);
190 break;
191 }
192 case TypeIndex::Decimal64: {
193 auto scale = get_decimal_scale(*data_type);
194 auto decimal_val = reinterpret_cast<const Decimal64*>(&input);
195 write_text(*decimal_val, scale, ss);
196 break;
197 }
198 case TypeIndex::Decimal128:
199 case TypeIndex::Decimal128I: {
200 auto scale = get_decimal_scale(*data_type);
201 auto decimal_val = reinterpret_cast<const Decimal128*>(&input);
202 write_text(*decimal_val, scale, ss);
203 break;
204 }
205 case TypeIndex::Date:
206 case TypeIndex::DateTime: {
207 auto* date_int = reinterpret_cast<Int64*>(&input);
208 auto date_value = binary_cast<Int64, VecDateTimeValue>(*date_int);
209 char buf[32] = {};
210 date_value.to_string(buf);
211 ss << std::string(buf, strlen(buf));
212 break;
213 }
214 case TypeIndex::DateV2: {
215 auto* value = (DateV2Value<DateV2ValueType>*)(&input);
216 ss << *value;
217 break;
218 }
219 case TypeIndex::DateTimeV2: {
220 auto* value = (DateV2Value<DateTimeV2ValueType>*)(&input);
221 ss << *value;
222 break;
223 }
224 default:
225 LOG(WARNING) << fmt::format("unable to convert histogram data of type {}",
226 data_type->get_name());
227 return false;
228 }
229
230 return true;
231}

233template <typename T>
234bool build_json_from_bucket(rapidjson::StringBuffer& buffer, const std::vector<Bucket<T>>& buckets,
235 const DataTypePtr& data_type, double sample_rate,
236 uint32_t max_bucket_num) {
237 rapidjson::Document doc;
238 doc.SetObject();
239 rapidjson::Document::AllocatorType& allocator = doc.GetAllocator();
240
241 rapidjson::Value sample_rate_val(sample_rate);
242 doc.AddMember("sample_rate", sample_rate_val, allocator);
243
244 rapidjson::Value max_bucket_num_val(max_bucket_num);
245 doc.AddMember("max_bucket_num", max_bucket_num_val, allocator);
246
247 int bucket_num = buckets.size();
248 rapidjson::Value bucket_num_val(bucket_num);
249 doc.AddMember("bucket_num", bucket_num_val, allocator);
250
251 rapidjson::Value bucket_arr(rapidjson::kArrayType);
252 for (auto i = 0; i < bucket_num; ++i) {
253 std::stringstream ss1;
254 std::stringstream ss2;
255 value_to_string(ss1, buckets[i].lower, data_type);
256 value_to_string(ss2, buckets[i].upper, data_type);
257 std::string lower_str = ss1.str();
258 std::string upper_str = ss2.str();
259 bucket_to_json(allocator, bucket_arr, lower_str, upper_str, (uint64_t)(buckets[i].count),
260 (uint64_t)(buckets[i].pre_sum), (uint64_t)(buckets[i].ndv));
261 }
262 doc.AddMember("buckets", bucket_arr, allocator);
263
264 rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
265 doc.Accept(writer);
266
267 return buffer.GetSize() > 0;
268}
