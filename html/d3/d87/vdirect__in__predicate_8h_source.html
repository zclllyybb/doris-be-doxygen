23namespace doris::vectorized {
24class VDirectInPredicate final : public VExpr {
25public:
26 VDirectInPredicate(const TExprNode& node)
27 : VExpr(node), _filter(nullptr), _expr_name("direct_in_predicate") {}
28 ~VDirectInPredicate() override = default;
29
30 Status execute(VExprContext* context, doris::vectorized::Block* block,
31 int* result_column_id) override {
32 doris::vectorized::ColumnNumbers arguments(_children.size());
33 for (int i = 0; i < _children.size(); ++i) {
34 int column_id = -1;
35 RETURN_IF_ERROR(_children[i]->execute(context, block, &column_id));
36 arguments[i] = column_id;
37 }
38
39 size_t num_columns_without_result = block->columns();
40 auto res_data_column = ColumnVector<UInt8>::create(block->rows());
41 ColumnPtr argument_column =
42 block->get_by_position(arguments[0]).column->convert_to_full_column_if_const();
43 size_t sz = argument_column->size();
44 res_data_column->resize(sz);
45
46 if (argument_column->is_nullable()) {
47 auto column_nested = static_cast<const ColumnNullable*>(argument_column.get())
48 ->get_nested_column_ptr();
49 auto& null_map =
50 static_cast<const ColumnNullable*>(argument_column.get())->get_null_map_data();
51 _filter->find_batch_nullable(*column_nested, sz, null_map, res_data_column->get_data());
52 } else {
53 _filter->find_batch(*argument_column, sz, res_data_column->get_data());
54 }
55
56 DCHECK(!_data_type->is_nullable());
57
58 block->insert({std::move(res_data_column), _data_type, _expr_name});
59
60 *result_column_id = num_columns_without_result;
61 return Status::OK();
62 }
63
64 VExpr* clone(doris::ObjectPool* pool) const override {
65 return pool->add(new VDirectInPredicate(*this));
66 }
67
68 const std::string& expr_name() const override { return _expr_name; }
69
70 void set_filter(std::shared_ptr<HybridSetBase>& filter) { _filter = filter; }
71
72 std::shared_ptr<HybridSetBase> get_set_func() const override { return _filter; }
73
74private:
75 std::shared_ptr<HybridSetBase> _filter;
76 std::string _expr_name;
77};
78} // namespace doris::vectorized
