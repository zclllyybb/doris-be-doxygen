24class VDirectInPredicate final : public VExpr {
25public:
26 VDirectInPredicate(const TExprNode& node)
27 : VExpr(node), _filter(nullptr), _expr_name("direct_in_predicate") {}
28 ~VDirectInPredicate() override = default;
29
30 Status execute(VExprContext* context, doris::vectorized::Block* block,
31 int* result_column_id) override {
32 doris::vectorized::ColumnNumbers arguments(_children.size());
33 for (int i = 0; i < _children.size(); ++i) {
34 int column_id = -1;
35 RETURN_IF_ERROR(_children[i]->execute(context, block, &column_id));
36 arguments[i] = column_id;
37 }
38
39 size_t num_columns_without_result = block->columns();
40 auto res_data_column = ColumnVector<UInt8>::create(block->rows());
41 ColumnPtr argument_column =
42 block->get_by_position(arguments[0]).column->convert_to_full_column_if_const();
43 size_t sz = argument_column->size();
44 res_data_column->resize(sz);
45
46 auto ptr = ((ColumnVector<UInt8>*)res_data_column.get())->get_data().data();
47 auto type = WhichDataType(remove_nullable(block->get_by_position(arguments[0]).type));
48 if (type.is_string_or_fixed_string()) {
49 for (size_t i = 0; i < sz; i++) {
50 auto ele = argument_column->get_data_at(i);
51 StringRef v(ele.data, ele.size);
52 ptr[i] = _filter->find(reinterpret_cast<const void*>(&v));
53 }
54 } else if (type.is_int_or_uint() || type.is_float()) {
55 if (argument_column->is_nullable()) {
56 auto column_nested = reinterpret_cast<const ColumnNullable*>(argument_column.get())
57 ->get_nested_column_ptr();
58 auto column_nullmap = reinterpret_cast<const ColumnNullable*>(argument_column.get())
59 ->get_null_map_column_ptr();
60 _filter->find_fixed_len(column_nested->get_raw_data().data,
61 (uint8*)column_nullmap->get_raw_data().data, sz, ptr);
62 } else {
63 _filter->find_fixed_len(argument_column->get_raw_data().data, nullptr, sz, ptr);
64 }
65 } else {
66 for (size_t i = 0; i < sz; i++) {
67 ptr[i] = _filter->find(
68 reinterpret_cast<const void*>(argument_column->get_data_at(i).data));
69 }
70 }
71
72 DCHECK(!_data_type->is_nullable());
73
74 block->insert({std::move(res_data_column), _data_type, _expr_name});
75
76 *result_column_id = num_columns_without_result;
77 return Status::OK();
78 }
79
80 VExpr* clone(doris::ObjectPool* pool) const override {
81 return pool->add(new VDirectInPredicate(*this));
82 }
83
84 const std::string& expr_name() const override { return _expr_name; }
85
86 void set_filter(std::shared_ptr<HybridSetBase>& filter) { _filter = filter; }
87
88 std::shared_ptr<HybridSetBase> get_set_func() const override { return _filter; }
89
90private:
91 std::shared_ptr<HybridSetBase> _filter;
92 std::string _expr_name;
93};
