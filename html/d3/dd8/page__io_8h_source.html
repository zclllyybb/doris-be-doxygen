82// Utility class for read and write page. All types of page share the same general layout:
83// Page := PageBody, PageFooter, FooterSize(4), Checksum(4)
84// - PageBody is defined by page type and may be compressed
85// - PageFooter is serialized PageFooterPB. It contains page_type, uncompressed_body_size,
86// and other custom metadata. PageBody is not compressed when its size is equal to
87// uncompressed_body_size
88// - FooterSize stores the size of PageFooter
89// - Checksum is the crc32c checksum of all previous part
90class PageIO {
91public:
92 // Compress `body' using `codec' into `compressed_body'.
93 // The size of returned `compressed_body' is 0 when the body is not compressed, this
94 // could happen when `codec' is null or space saving is less than `min_space_saving'.
95 static Status compress_page_body(BlockCompressionCodec* codec, double min_space_saving,
96 const std::vector<Slice>& body, OwnedSlice* compressed_body);
97
98 // Encode page from `body' and `footer' and write to `file'.
99 // `body' could be either uncompressed or compressed.
100 // On success, the file pointer to the written page is stored in `result'.
101 static Status write_page(io::FileWriter* writer, const std::vector<Slice>& body,
102 const PageFooterPB& footer, PagePointer* result);
103
104 // Convenient function to compress page body and write page in one go.
105 static Status compress_and_write_page(BlockCompressionCodec* codec, double min_space_saving,
106 io::FileWriter* writer, const std::vector<Slice>& body,
107 const PageFooterPB& footer, PagePointer* result) {
108 DCHECK_EQ(footer.uncompressed_size(), Slice::compute_total_size(body));
109 OwnedSlice compressed_body;
110 RETURN_IF_ERROR(compress_page_body(codec, min_space_saving, body, &compressed_body));
111 if (compressed_body.slice().empty()) { // uncompressed
112 return write_page(writer, body, footer, result);
113 }
114 return write_page(writer, {compressed_body.slice()}, footer, result);
115 }
116
117 // Read and parse a page according to `opts'.
118 // On success
119 // `handle' holds the memory of page data,
120 // `body' points to page body,
121 // `footer' stores the page footer.
122 static Status read_and_decompress_page(const PageReadOptions& opts, PageHandle* handle,
123 Slice* body, PageFooterPB* footer);
124};
