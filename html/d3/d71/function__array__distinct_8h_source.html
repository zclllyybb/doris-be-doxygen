50 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
51 size_t result, size_t input_rows_count) override {
52 ColumnPtr src_column =
53 block.get_by_position(arguments[0]).column->convert_to_full_column_if_const();
54 const auto& src_column_array = check_and_get_column<ColumnArray>(*src_column);
55 if (!src_column_array) {
56 return Status::RuntimeError(
57 fmt::format("unsupported types for function {}({})", get_name(),
58 block.get_by_position(arguments[0]).type->get_name()));
59 }
60 const auto& src_offsets = src_column_array->get_offsets();
61 const auto* src_nested_column = &src_column_array->get_data();
62 DCHECK(src_nested_column != nullptr);
63
64 DataTypePtr src_column_type = block.get_by_position(arguments[0]).type;
65 auto nested_type = assert_cast<const DataTypeArray&>(*src_column_type).get_nested_type();
66 auto dest_column_ptr = ColumnArray::create(nested_type->create_column(),
67 ColumnArray::ColumnOffsets::create());
68 IColumn* dest_nested_column = &dest_column_ptr->get_data();
69 auto& dest_offsets = dest_column_ptr->get_offsets();
70 DCHECK(dest_nested_column != nullptr);
71 dest_nested_column->reserve(src_nested_column->size());
72 dest_offsets.reserve(input_rows_count);
73
74 const NullMapType* src_null_map = nullptr;
75 if (src_nested_column->is_nullable()) {
76 const ColumnNullable* src_nested_nullable_col =
77 check_and_get_column<ColumnNullable>(*src_nested_column);
78 src_nested_column = src_nested_nullable_col->get_nested_column_ptr();
79 src_null_map = &src_nested_nullable_col->get_null_map_column().get_data();
80 }
81
82 NullMapType* dest_null_map = nullptr;
83 if (dest_nested_column->is_nullable()) {
84 ColumnNullable* dest_nested_nullable_col =
85 reinterpret_cast<ColumnNullable*>(dest_nested_column);
86 dest_nested_column = dest_nested_nullable_col->get_nested_column_ptr();
87 dest_null_map = &dest_nested_nullable_col->get_null_map_column().get_data();
88 }
89
90 auto res_val = _execute_by_type(*src_nested_column, src_offsets, *dest_nested_column,
91 dest_offsets, src_null_map, dest_null_map, nested_type);
92 if (!res_val) {
93 return Status::RuntimeError(
94 fmt::format("execute failed or unsupported types for function {}({})",
95 get_name(), block.get_by_position(arguments[0]).type->get_name()));
96 }
97
98 block.replace_by_position(result, std::move(dest_column_ptr));
99 return Status::OK();
100 }

102private:
103 // Note: Here initially allocate a piece of memory for 2^5 = 32 elements.
104 static constexpr size_t INITIAL_SIZE_DEGREE = 5;
105
106 template <typename ColumnType>
107 bool _execute_number(const IColumn& src_column, const ColumnArray::Offsets64& src_offsets,
108 IColumn& dest_column, ColumnArray::Offsets64& dest_offsets,
109 const NullMapType* src_null_map, NullMapType* dest_null_map) {
110 using NestType = typename ColumnType::value_type;
111 using ElementNativeType = typename NativeType<NestType>::Type;
112
113 const ColumnType* src_data_concrete = reinterpret_cast<const ColumnType*>(&src_column);
114 if (!src_data_concrete) {
115 return false;
116 }
117 const PaddedPODArray<NestType>& src_datas = src_data_concrete->get_data();
118
119 ColumnType& dest_data_concrete = reinterpret_cast<ColumnType&>(dest_column);
120 PaddedPODArray<NestType>& dest_datas = dest_data_concrete.get_data();
121
122 using Set = HashSetWithStackMemory<ElementNativeType, DefaultHash<ElementNativeType>,
123 INITIAL_SIZE_DEGREE>;
124 Set set;
125
126 size_t prev_src_offset = 0;
127 size_t res_offset = 0;
128
129 for (auto curr_src_offset : src_offsets) {
130 set.clear();
131 size_t null_size = 0;
132 for (size_t j = prev_src_offset; j < curr_src_offset; ++j) {
133 if (null_size != 0 && src_null_map && (*src_null_map)[j]) {
134 // ignore duplicated nulls
135 continue;
136 }
137 if (src_null_map && (*src_null_map)[j]) {
138 DCHECK(dest_null_map != nullptr);
139 (*dest_null_map).push_back(true);
140 // Note: here we need to add an element which will not use for output
141 // because we expand the value of each offset
142 dest_datas.push_back(NestType());
143 null_size++;
144 continue;
145 }
146
147 if (!set.find(src_datas[j])) {
148 set.insert(src_datas[j]);
149 dest_datas.push_back(src_datas[j]);
150 if (dest_null_map) {
151 (*dest_null_map).push_back(false);
152 }
153 }
154 }
155
156 res_offset += set.size() + null_size;
157 dest_offsets.push_back(res_offset);
158 prev_src_offset = curr_src_offset;
159 }
160
161 return true;
162 }

164 bool _execute_string(const IColumn& src_column, const ColumnArray::Offsets64& src_offsets,
165 IColumn& dest_column, ColumnArray::Offsets64& dest_offsets,
166 const NullMapType* src_null_map, NullMapType* dest_null_map) {
167 const ColumnString* src_data_concrete = reinterpret_cast<const ColumnString*>(&src_column);
168 if (!src_data_concrete) {
169 return false;
170 }
171
172 ColumnString& dest_column_string = reinterpret_cast<ColumnString&>(dest_column);
173 ColumnString::Chars& column_string_chars = dest_column_string.get_chars();
174 ColumnString::Offsets& column_string_offsets = dest_column_string.get_offsets();
175 column_string_chars.reserve(src_column.size());
176
177 using Set = HashSetWithStackMemory<StringRef, DefaultHash<StringRef>, INITIAL_SIZE_DEGREE>;
178 Set set;
179
180 size_t prev_src_offset = 0;
181 size_t res_offset = 0;
182
183 for (auto curr_src_offset : src_offsets) {
184 set.clear();
185 size_t null_size = 0;
186 for (size_t j = prev_src_offset; j < curr_src_offset; ++j) {
187 if (null_size != 0 && src_null_map && (*src_null_map)[j]) {
188 // ignore duplicated nulls
189 continue;
190 }
191 if (src_null_map && (*src_null_map)[j]) {
192 DCHECK(dest_null_map != nullptr);
193 // Note: here we need to update the offset of ColumnString
194 column_string_offsets.push_back(column_string_offsets.back());
195 (*dest_null_map).push_back(true);
196 null_size++;
197 continue;
198 }
199
200 StringRef src_str_ref = src_data_concrete->get_data_at(j);
201 if (!set.find(src_str_ref)) {
202 set.insert(src_str_ref);
203 // copy the src data to column_string_chars
204 const size_t old_size = column_string_chars.size();
205 const size_t new_size = old_size + src_str_ref.size;
206 column_string_chars.resize(new_size);
207 if (src_str_ref.size > 0) {
208 memcpy(column_string_chars.data() + old_size, src_str_ref.data,
209 src_str_ref.size);
210 }
211 column_string_offsets.push_back(new_size);
212
213 if (dest_null_map) {
214 (*dest_null_map).push_back(false);
215 }
216 }
217 }
218
219 res_offset += set.size() + null_size;
220 dest_offsets.push_back(res_offset);
221 prev_src_offset = curr_src_offset;
222 }
223 return true;
224 }

226 bool _execute_by_type(const IColumn& src_column, const ColumnArray::Offsets64& src_offsets,
227 IColumn& dest_column, ColumnArray::Offsets64& dest_offsets,
228 const NullMapType* src_null_map, NullMapType* dest_null_map,
229 DataTypePtr& nested_type) {
230 bool res = false;
231 WhichDataType which(remove_nullable(nested_type));
232 if (which.is_uint8()) {
233 res = _execute_number<ColumnUInt8>(src_column, src_offsets, dest_column, dest_offsets,
234 src_null_map, dest_null_map);

237 src_null_map, dest_null_map);
238 } else if (which.is_int16()) {

240 src_null_map, dest_null_map);
241 } else if (which.is_int32()) {
242 res = _execute_number<ColumnInt32>(src_column, src_offsets, dest_column, dest_offsets,
243 src_null_map, dest_null_map);
244 } else if (which.is_int64()) {
245 res = _execute_number<ColumnInt64>(src_column, src_offsets, dest_column, dest_offsets,
246 src_null_map, dest_null_map);
247 } else if (which.is_int128()) {
248 res = _execute_number<ColumnInt128>(src_column, src_offsets, dest_column, dest_offsets,
249 src_null_map, dest_null_map);
250 } else if (which.is_float32()) {
251 res = _execute_number<ColumnFloat32>(src_column, src_offsets, dest_column, dest_offsets,
252 src_null_map, dest_null_map);
253 } else if (which.is_float64()) {
254 res = _execute_number<ColumnFloat64>(src_column, src_offsets, dest_column, dest_offsets,
255 src_null_map, dest_null_map);
256 } else if (which.is_date()) {
257 res = _execute_number<ColumnDate>(src_column, src_offsets, dest_column, dest_offsets,
258 src_null_map, dest_null_map);
259 } else if (which.is_date_time()) {
260 res = _execute_number<ColumnDateTime>(src_column, src_offsets, dest_column,
261 dest_offsets, src_null_map, dest_null_map);
262 } else if (which.is_date_v2()) {
263 res = _execute_number<ColumnDateV2>(src_column, src_offsets, dest_column, dest_offsets,
264 src_null_map, dest_null_map);
265 } else if (which.is_date_time_v2()) {
266 res = _execute_number<ColumnDateTimeV2>(src_column, src_offsets, dest_column,
267 dest_offsets, src_null_map, dest_null_map);
268 } else if (which.is_decimal32()) {
269 res = _execute_number<ColumnDecimal32>(src_column, src_offsets, dest_column,
270 dest_offsets, src_null_map, dest_null_map);
271 } else if (which.is_decimal64()) {
272 res = _execute_number<ColumnDecimal64>(src_column, src_offsets, dest_column,
273 dest_offsets, src_null_map, dest_null_map);
274 } else if (which.is_decimal128i()) {
275 res = _execute_number<ColumnDecimal128I>(src_column, src_offsets, dest_column,
276 dest_offsets, src_null_map, dest_null_map);
277 } else if (which.is_decimal128()) {
278 res = _execute_number<ColumnDecimal128>(src_column, src_offsets, dest_column,
279 dest_offsets, src_null_map, dest_null_map);
280 } else if (which.is_string()) {
281 res = _execute_string(src_column, src_offsets, dest_column, dest_offsets, src_null_map,
282 dest_null_map);
283 }
284 return res;
285 }
