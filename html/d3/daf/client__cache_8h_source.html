33// Helper class which implements the majority of the caching
34// functionality without using templates (i.e. pointers to the
35// superclass of all ThriftClients and a void* for the key).
36//
37// The user of this class only sees RPC proxy classes, but we have
38// to track the ThriftClient to manipulate the underlying
39// transport. To do this, we maintain a map from an opaque 'key'
40// pointer type to the client implementation. We actually know the
41// type of the pointer (it's the type parameter to ClientCache), but
42// we deliberately avoid using it so that this entire class doesn't
43// get inlined every time it gets used.
44//
45// This class is thread-safe.
46//
47// TODO: shut down clients in the background if they don't get used for a period of time
48// TODO: in order to reduce locking overhead when getting/releasing clients,
49// add call to hand back pointer to list stored in ClientCache and add separate lock
50// to list (or change to lock-free list)
51// TODO: reduce locking overhead and by adding per-address client caches, each with its
52// own lock.
53// TODO: More graceful handling of clients that have failed (maybe better
54// handled by a smart-wrapper of the interface object).
55// TODO: limits on total number of clients, and clients per-backend
56class ClientCacheHelper {
57public:
58 ~ClientCacheHelper();
59 // Callback method which produces a client object when one cannot be
60 // found in the cache. Supplied by the ClientCache wrapper.
61 using ClientFactory =
62 std::function<ThriftClientImpl*(const TNetworkAddress& hostport, void** client_key)>;
63
64 // Return client for specific host/port in 'client'. If a client
65 // is not available, the client parameter is set to nullptr.
66 Status get_client(const TNetworkAddress& hostport, ClientFactory& factory_method,
67 void** client_key, int timeout_ms);
68
69 // Close and delete the underlying transport and remove the client from _client_map.
70 // Return a new client connecting to the same host/port.
71 // Return an error status and set client_key to nullptr if a new client cannot
72 // created.
73 Status reopen_client(ClientFactory& factory_method, void** client_key, int timeout_ms);
74
75 // Return a client to the cache, without closing it, and set *client_key to nullptr.
76 void release_client(void** client_key);
77
78 // Close all connections to a host (e.g., in case of failure) so that on their
79 // next use they will have to be Reopen'ed.
80 void close_connections(const TNetworkAddress& address);
81
82 std::string debug_string();
83
84 void test_shutdown();
85
86 void init_metrics(const std::string& name);
87
88private:
89 template <class T>
90 friend class ClientCache;
91 // Private constructor so that only ClientCache can instantiate this class.
92 ClientCacheHelper() : _metrics_enabled(false), _max_cache_size_per_host(-1) {}

134// A scoped client connection to help manage clients from a client cache.
135//
136// Example:
137// {
138// DorisInternalServiceConnection client(cache, address, &status);
139// try {
140// client->TransmitData(...);
141// } catch (TTransportException& e) {
142// // Retry
143// RETURN_IF_ERROR(client.Reopen());
144// client->TransmitData(...);
145// }
146// }
147// ('client' is released back to cache upon destruction.)
148template <class T>
149class ClientConnection {
150public:
151 ClientConnection(ClientCache<T>* client_cache, const TNetworkAddress& address, Status* status)
152 : ClientConnection(client_cache, address, 0, status, 3) {}
153
154 ClientConnection(ClientCache<T>* client_cache, const TNetworkAddress& address, int timeout_ms,
155 Status* status, int max_retries = 3)
156 : _client_cache(client_cache), _client(nullptr) {
157 int num_retries = 0;
158 do {
159 *status = _client_cache->get_client(address, &_client, timeout_ms);
160 if (status->ok()) {
161 DCHECK(_client != nullptr);
162 break;
163 }
164 if (num_retries++ < max_retries) {
165 // exponential backoff retry with starting delay of 500ms
166 usleep(500000 * (1 << num_retries));
167 LOG(INFO) << "Failed to get client from cache: " << status->to_string()
168 << ", retrying[" << num_retries << "]...";
169 }
170 } while (num_retries < max_retries);
171 }
172
173 ~ClientConnection() {
174 if (_client != nullptr) {
175 _client_cache->release_client(&_client);
176 }
177 }
178
179 Status reopen(int timeout_ms) { return _client_cache->reopen_client(&_client, timeout_ms); }
180
181 Status reopen() { return _client_cache->reopen_client(&_client, 0); }
182
183 T* operator->() const { return _client; }
184
185private:
186 ClientCache<T>* _client_cache;
187 T* _client;
188};

190// Generic cache of Thrift clients for a given service type.
191// This class is thread-safe.
192template <class T>
193class ClientCache {
194public:
195 using Client = ThriftClient<T>;
196
197 ClientCache() : _client_cache_helper() {
198 _client_factory =
199 std::bind<ThriftClientImpl*>(std::mem_fn(&ClientCache::make_client), this,
200 std::placeholders::_1, std::placeholders::_2);
201 }
202
203 ClientCache(int max_cache_size) : _client_cache_helper(max_cache_size) {
204 _client_factory =
205 std::bind<ThriftClientImpl*>(std::mem_fn(&ClientCache::make_client), this,
206 std::placeholders::_1, std::placeholders::_2);
207 }
208
209 // Close all clients connected to the supplied address, (e.g., in
210 // case of failure) so that on their next use they will have to be
211 // Reopen'ed.
212 void close_connections(const TNetworkAddress& hostport) {
213 return _client_cache_helper.close_connections(hostport);
214 }
215
216 // Helper method which returns a debug string
217 std::string debug_string() { return _client_cache_helper.debug_string(); }
218
219 // For testing only: shutdown all clients
220 void test_shutdown() { return _client_cache_helper.test_shutdown(); }
221
222 // Adds metrics for this cache.
223 // The metrics have an identification by the 'name' argument
224 // (which should not end in a period).
225 // Must be called before the cache is used, otherwise the metrics might be wrong
226 void init_metrics(const std::string& name) { _client_cache_helper.init_metrics(name); }
227
228private:
229 friend class ClientConnection<T>;
230
231 // Most operations in this class are thin wrappers around the
232 // equivalent in ClientCacheHelper, which is a non-templated cache
233 // to avoid inlining lots of code wherever this cache is used.
234 ClientCacheHelper _client_cache_helper;
235
236 // Function pointer, bound to make_client, which produces clients when the cache is empty
237 using ClientFactory = ClientCacheHelper::ClientFactory;
238 ClientFactory _client_factory;
239
240 // Obtains a pointer to a Thrift interface object (of type T),
241 // backed by a live transport which is already open. Returns
242 // Status::OK() unless there was an error opening the transport.
243 Status get_client(const TNetworkAddress& hostport, T** iface, int timeout_ms) {
244 return _client_cache_helper.get_client(hostport, _client_factory,
245 reinterpret_cast<void**>(iface), timeout_ms);
246 }
247
248 // Close and delete the underlying transport. Return a new client connecting to the
249 // same host/port.
250 // Return an error status if a new connection cannot be established and *client will be
251 // nullptr in that case.
252 Status reopen_client(T** client, int timeout_ms) {
253 return _client_cache_helper.reopen_client(_client_factory, reinterpret_cast<void**>(client),
254 timeout_ms);
255 }
256
257 // Return the client to the cache and set *client to nullptr.
258 void release_client(T** client) {
259 return _client_cache_helper.release_client(reinterpret_cast<void**>(client));
260 }
261
262 // Factory method to produce a new ThriftClient<T> for the wrapped cache
263 ThriftClientImpl* make_client(const TNetworkAddress& hostport, void** client_key) {
264 static ThriftServer::ServerType server_type = get_thrift_server_type();
265 Client* client = new Client(hostport.hostname, hostport.port, server_type);
266 *client_key = reinterpret_cast<void*>(client->iface());
267 return client;
268 }
269
270 // since service type is multiple, we should set thrift server type here for be thrift client
271 ThriftServer::ServerType get_thrift_server_type() {
272 auto& thrift_server_type = config::thrift_server_type_of_fe;
273 std::transform(thrift_server_type.begin(), thrift_server_type.end(),
274 thrift_server_type.begin(), [](auto c) { return std::toupper(c); });
275 if (strcmp(typeid(T).name(), "N5doris21FrontendServiceClientE") == 0 &&
276 thrift_server_type == "THREADED_SELECTOR") {
277 return ThriftServer::ServerType::NON_BLOCKING;
278 } else {
279 return ThriftServer::ServerType::THREADED;
280 }
281 }
282};
