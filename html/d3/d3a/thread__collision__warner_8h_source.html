16// A helper class alongside macros to be used to verify assumptions about thread
17// safety of a class.
18//
19// Example: Queue implementation non thread-safe but still usable if clients
20// are synchronized somehow.
21//
22// In this case the macro DFAKE_SCOPED_LOCK has to be
23// used, it checks that if a thread is inside the push/pop then
24// noone else is still inside the pop/push
25//
26// class NonThreadSafeQueue {
27// public:
28// ...
29// void push(int) { DFAKE_SCOPED_LOCK(push_pop_); ... }
30// int pop() { DFAKE_SCOPED_LOCK(push_pop_); ... }
31// ...
32// private:
33// DFAKE_MUTEX(push_pop_);
34// };
35//
36//
37// Example: Queue implementation non thread-safe but still usable if clients
38// are synchronized somehow, it calls a method to "protect" from
39// a "protected" method
40//
41// In this case the macro DFAKE_SCOPED_RECURSIVE_LOCK
42// has to be used, it checks that if a thread is inside the push/pop
43// then noone else is still inside the pop/push
44//
45// class NonThreadSafeQueue {
46// public:
47// void push(int) {
48// DFAKE_SCOPED_LOCK(push_pop_);
49// ...
50// }
51// int pop() {
52// DFAKE_SCOPED_RECURSIVE_LOCK(push_pop_);
53// bar();
54// ...
55// }
56// void bar() { DFAKE_SCOPED_RECURSIVE_LOCK(push_pop_); ... }
57// ...
58// private:
59// DFAKE_MUTEX(push_pop_);
60// };
61//
62//
63// Example: Queue implementation not usable even if clients are synchronized,
64// so only one thread in the class life cycle can use the two members
65// push/pop.
66//
67// In this case the macro DFAKE_SCOPED_LOCK_THREAD_LOCKED pins the
68// specified
69// critical section the first time a thread enters push or pop, from
70// that time on only that thread is allowed to execute push or pop.
71//
72// class NonThreadSafeQueue {
73// public:
74// ...
75// void push(int) { DFAKE_SCOPED_LOCK_THREAD_LOCKED(push_pop_); ... }
76// int pop() { DFAKE_SCOPED_LOCK_THREAD_LOCKED(push_pop_); ... }
77// ...
78// private:
79// DFAKE_MUTEX(push_pop_);
80// };
81//
82//
83// Example: Class that has to be contructed/destroyed on same thread, it has
84// a "shareable" method (with external synchronization) and a not
85// shareable method (even with external synchronization).
86//
87// In this case 3 Critical sections have to be defined
88//
89// class ExoticClass {
90// public:
91// ExoticClass() { DFAKE_SCOPED_LOCK_THREAD_LOCKED(ctor_dtor_); ... }
92// ~ExoticClass() { DFAKE_SCOPED_LOCK_THREAD_LOCKED(ctor_dtor_); ... }
93//
94// void Shareable() { DFAKE_SCOPED_LOCK(shareable_section_); ... }
95// void NotShareable() { DFAKE_SCOPED_LOCK_THREAD_LOCKED(ctor_dtor_); ... }
96// ...
97// private:
98// DFAKE_MUTEX(ctor_dtor_);
99// DFAKE_MUTEX(shareable_section_);
100// };
101
102#if !defined(NDEBUG)

104// Defines a class member that acts like a mutex. It is used only as a
105// verification tool.
106#define DFAKE_MUTEX(obj) mutable base::ThreadCollisionWarner obj
107// Asserts the call is never called simultaneously in two threads. Used at
108// member function scope.
109#define DFAKE_SCOPED_LOCK(obj) base::ThreadCollisionWarner::ScopedCheck s_check_##obj(&obj)
110// Asserts the call is never called simultaneously in two threads. Used at
111// member function scope. Same as DFAKE_SCOPED_LOCK but allows recursive locks.
112#define DFAKE_SCOPED_RECURSIVE_LOCK(obj) \
113 base::ThreadCollisionWarner::ScopedRecursiveCheck sr_check_##obj(&obj)
114// Asserts the code is always executed in the same thread.
115#define DFAKE_SCOPED_LOCK_THREAD_LOCKED(obj) base::ThreadCollisionWarner::Check check_##obj(&obj)
116
117#else
118
119#define DFAKE_MUTEX(obj) typedef void InternalFakeMutexType##obj
120#define DFAKE_SCOPED_LOCK(obj) ((void)0)
121#define DFAKE_SCOPED_RECURSIVE_LOCK(obj) ((void)0)
122#define DFAKE_SCOPED_LOCK_THREAD_LOCKED(obj) ((void)0)
123
124#endif

128// The class ThreadCollisionWarner uses an Asserter to notify the collision
129// AsserterBase is the interfaces and DCheckAsserter is the default asserter
130// used. During the unit tests is used another class that doesn't "DCHECK"
131// in case of collision (check thread_collision_warner_unittests.cc)
132struct BASE_EXPORT AsserterBase {
133 virtual ~AsserterBase() {}
134 virtual void warn(int64_t previous_thread_id, int64_t current_thread_id) = 0;
135};
136
137struct BASE_EXPORT DCheckAsserter : public AsserterBase {
138 virtual ~DCheckAsserter() {}
139 void warn(int64_t previous_thread_id, int64_t current_thread_id) override;
140};
141
142class BASE_EXPORT ThreadCollisionWarner {
143public:
144 // The parameter asserter is there only for test purpose
145 explicit ThreadCollisionWarner(AsserterBase* asserter = new DCheckAsserter())
146 : valid_thread_id_(0), counter_(0), asserter_(asserter) {}
147
148 ~ThreadCollisionWarner() { delete asserter_; }
149
150 // This class is meant to be used through the macro
151 // DFAKE_SCOPED_LOCK_THREAD_LOCKED
152 // it doesn't leave the critical section, as opposed to ScopedCheck,
153 // because the critical section being pinned is allowed to be used only
154 // from one thread
155 class BASE_EXPORT Check {
156 public:
157 explicit Check(ThreadCollisionWarner* warner) : warner_(warner) { warner_->EnterSelf(); }
158
159 ~Check() {}
160
161 private:
162 ThreadCollisionWarner* warner_;
163
164 DISALLOW_COPY_AND_ASSIGN(Check);
165 };
166
167 // This class is meant to be used through the macro
168 // DFAKE_SCOPED_LOCK
169 class BASE_EXPORT ScopedCheck {
170 public:
171 explicit ScopedCheck(ThreadCollisionWarner* warner) : warner_(warner) { warner_->Enter(); }
172
173 ~ScopedCheck() { warner_->Leave(); }
174
175 private:
176 ThreadCollisionWarner* warner_;
177
178 DISALLOW_COPY_AND_ASSIGN(ScopedCheck);
179 };
180
181 // This class is meant to be used through the macro
182 // DFAKE_SCOPED_RECURSIVE_LOCK
183 class BASE_EXPORT ScopedRecursiveCheck {
184 public:
185 explicit ScopedRecursiveCheck(ThreadCollisionWarner* warner) : warner_(warner) {
186 warner_->EnterSelf();
187 }
188
189 ~ScopedRecursiveCheck() { warner_->Leave(); }
190
191 private:
192 ThreadCollisionWarner* warner_;
193
194 DISALLOW_COPY_AND_ASSIGN(ScopedRecursiveCheck);
195 };
196
197private:
198 // This method stores the current thread identifier and does a DCHECK
199 // if a another thread has already done it, it is safe if same thread
200 // calls this multiple time (recursion allowed).
201 void EnterSelf();
202
203 // Same as EnterSelf but recursion is not allowed.
204 void Enter();
205
206 // Removes the thread_id stored in order to allow other threads to
207 // call EnterSelf or Enter.
208 void Leave();
209
210 // This stores the thread id that is inside the critical section, if the
211 // value is 0 then no thread is inside.
212 volatile subtle::Atomic64 valid_thread_id_;
213
214 // Counter to trace how many time a critical section was "pinned"
215 // (when allowed) in order to unpin it when counter_ reaches 0.
216 volatile subtle::Atomic64 counter_;
217
218 // Here only for class unit tests purpose, during the test I need to not
219 // DCHECK but notify the collision with something else.
220 AsserterBase* asserter_;
221
222 DISALLOW_COPY_AND_ASSIGN(ThreadCollisionWarner);
223};
