45// Issue a trace message, if tracing is enabled in the current thread.
46// See Trace::SubstituteAndTrace for arguments.
47// Example:
48// TRACE("Acquired timestamp $0", timestamp);
49#define TRACE(format, substitutions...) \
50 do { \
51 doris::Trace* _trace = doris::Trace::CurrentTrace(); \
52 if (_trace) { \
53 _trace->SubstituteAndTrace(__FILE__, __LINE__, (format), ##substitutions); \
54 } \
55 } while (0)

58#define TRACE_TO(trace, format, substitutions...) \

61// Increment a counter associated with the current trace.
62//
63// Each trace contains a map of counters which can be used to keep
64// request-specific statistics. It is significantly faster to increment
65// a trace counter compared to logging a message. Additionally, having
66// slightly more structured information makes it easier to aggregate
67// and show information back to operators.
68//
69// NOTE: the 'counter_name' MUST be a string which stays alive forever.
70// Typically, this is a compile-time constant. If something other than
71// a constant is required, use TraceMetric::InternName() in order to
72// create a string which will last for the process lifetime. Of course,
73// these strings will never be cleaned up, so it's important to use this
74// judiciously.
75//
76// If no trace is active, this does nothing and does not evaluate its
77// parameters.
78#define TRACE_COUNTER_INCREMENT(counter_name, val) \
79 do { \
80 doris::Trace* _trace = doris::Trace::CurrentTrace(); \
81 if (_trace) { \
82 _trace->metrics()->Increment(counter_name, val); \
83 } \
84 } while (0)

86// Increment a counter for the amount of wall time spent in the current
87// scope. For example:
88//
89// void DoFoo() {
90// TRACE_COUNTER_SCOPE_LATENCY_US("foo_us");
91// ... do expensive Foo thing
92// }
93//
94// will result in a trace metric indicating the number of microseconds spent
95// in invocations of DoFoo().
96#define TRACE_COUNTER_SCOPE_LATENCY_US(counter_name) \

99// Construct a constant C string counter name which acts as a sort of
100// coarse-grained histogram for trace metrics.
101#define BUCKETED_COUNTER_NAME(prefix, duration_us) \
102 []() { \
103 if (duration_us >= 100 * 1000) { \
104 return prefix "_gt_100_ms"; \
105 } else if (duration_us >= 10 * 1000) { \
106 return prefix "_10-100_ms"; \
107 } else if (duration_us >= 1000) { \
108 return prefix "_1-10_ms"; \
109 } else { \
110 return prefix "_lt_1ms"; \
111 } \
112 }()

225// Adopt a Trace object into the current thread for the duration
226// of this object.
227// This should only be used on the stack (and thus created and destroyed
228// on the same thread)
229class ScopedAdoptTrace {
230public:
231 explicit ScopedAdoptTrace(Trace* t) : old_trace_(Trace::threadlocal_trace_) {
232 Trace::threadlocal_trace_ = t;
233 if (t) {
234 t->AddRef();
235 }
236 DFAKE_SCOPED_LOCK_THREAD_LOCKED(ctor_dtor_);
237 }
238
239 ~ScopedAdoptTrace() {
240 if (Trace::threadlocal_trace_) {
241 Trace::threadlocal_trace_->Release();
242 }
243 Trace::threadlocal_trace_ = old_trace_;
244 DFAKE_SCOPED_LOCK_THREAD_LOCKED(ctor_dtor_);
245 }
246
247private:
248 DFAKE_MUTEX(ctor_dtor_);
249 Trace* old_trace_;
250
251 DISALLOW_COPY_AND_ASSIGN(ScopedAdoptTrace);
252};
