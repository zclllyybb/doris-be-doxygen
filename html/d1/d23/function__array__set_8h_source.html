58template <SetOperation operation, typename ColumnType>
59struct OpenSetImpl {
60 using Element = typename ColumnType::value_type;
61 using ElementNativeType = typename NativeType<Element>::Type;
62 using Set = HashSetWithStackMemory<ElementNativeType, DefaultHash<ElementNativeType>, 4>;
63 using Action = typename ActionImpl<Set, Element, operation>::Action;
64 Action action;
65 Set set;
66 Set result_set;
67 void reset() {
68 set.clear();
69 result_set.clear();
70 action.reset();
71 }
72 template <bool is_left>
73 void apply(const ColumnArrayExecutionData& src, size_t off, size_t len,
74 ColumnArrayMutableData& dst, size_t* count) {
75 const auto& src_data = assert_cast<const ColumnType&>(*src.nested_col).get_data();
76 auto& dst_data = assert_cast<ColumnType&>(*dst.nested_col).get_data();
77 for (size_t i = off; i < off + len; ++i) {
78 if (src.nested_nullmap_data && src.nested_nullmap_data[i]) {
79 if (action.template apply_null<is_left>()) {
80 dst_data.push_back(Element());
81 dst.nested_nullmap_data->push_back(1);
82 ++(*count);
83 }
84 } else {
85 if (action.template apply<is_left>(set, result_set, src_data[i])) {
86 dst_data.push_back(src_data[i]);
87 if (dst.nested_nullmap_data) {
88 dst.nested_nullmap_data->push_back(0);
89 }
90 ++(*count);
91 }
92 }
93 }
94 }
95};

98struct OpenSetImpl<operation, ColumnString> {
99 using Set = HashSetWithStackMemory<StringRef, DefaultHash<StringRef>, 4>;
100 using Action = typename ActionImpl<Set, StringRef, operation>::Action;
101 Action action;

109 template <bool is_left>
110 void apply(const ColumnArrayExecutionData& src, size_t off, size_t len,
111 ColumnArrayMutableData& dst, size_t* count) {
112 const auto& src_column = assert_cast<const ColumnString&>(*src.nested_col);
113 auto& dst_column = assert_cast<ColumnString&>(*dst.nested_col);
114 for (size_t i = off; i < off + len; ++i) {
115 if (src.nested_nullmap_data && src.nested_nullmap_data[i]) {
116 if (action.template apply_null<is_left>()) {
117 dst_column.insert_default();
118 dst.nested_nullmap_data->push_back(1);
119 ++(*count);
120 }
121 } else {
122 if (action.template apply<is_left>(set, result_set, src_column.get_data_at(i))) {
123 dst_column.insert_from(src_column, i);
124 if (dst.nested_nullmap_data) {
125 dst.nested_nullmap_data->push_back(0);
126 }
127 ++(*count);
128 }
129 }
130 }
131 }

134template <SetOperation operation>
135struct ArraySetImpl {
136public:
137 static DataTypePtr get_return_type(const DataTypes& arguments) {
138 DataTypePtr res;
139 // if any nested type of array arguments is nullable then return array with
140 // nullable nested type.
141 for (const auto& arg : arguments) {
142 const DataTypeArray* array_type = check_and_get_data_type<DataTypeArray>(arg.get());
143 if (array_type->get_nested_type()->is_nullable()) {
144 res = arg;
145 break;
146 }
147 }
148 res = res ? res : arguments[0];
149 return res;
150 }
151
152 static Status execute(ColumnPtr& res_ptr, const ColumnArrayExecutionData& left_data,
153 const ColumnArrayExecutionData& right_data) {
154 ColumnArrayMutableData dst;
155 if (left_data.nested_nullmap_data || right_data.nested_nullmap_data) {
156 dst = create_mutable_data(left_data.nested_col, true);
157 } else {
158 dst = create_mutable_data(left_data.nested_col, false);
159 }
160 ColumnPtr res_column;
161 if (_execute_internal<ColumnString>(dst, left_data, right_data) ||
162 _execute_internal<ColumnDate>(dst, left_data, right_data) ||
163 _execute_internal<ColumnDateTime>(dst, left_data, right_data) ||
164 _execute_internal<ColumnUInt8>(dst, left_data, right_data) ||
165 _execute_internal<ColumnInt8>(dst, left_data, right_data) ||
166 _execute_internal<ColumnInt16>(dst, left_data, right_data) ||
167 _execute_internal<ColumnInt32>(dst, left_data, right_data) ||
168 _execute_internal<ColumnInt64>(dst, left_data, right_data) ||
169 _execute_internal<ColumnInt128>(dst, left_data, right_data) ||
170 _execute_internal<ColumnFloat32>(dst, left_data, right_data) ||
171 _execute_internal<ColumnFloat64>(dst, left_data, right_data) ||
172 _execute_internal<ColumnDecimal128>(dst, left_data, right_data)) {
173 res_column = assemble_column_array(dst);
174 if (res_column) {
175 res_ptr = std::move(res_column);
176 return Status::OK();
177 }
178 }
179 return Status::RuntimeError("Unexpected columns: {}, {}", left_data.nested_col->get_name(),
180 right_data.nested_col->get_name());
181 }
182
183private:
184 template <typename ColumnType>
185 static bool _execute_internal(ColumnArrayMutableData& dst,
186 const ColumnArrayExecutionData& left_data,
187 const ColumnArrayExecutionData& right_data) {
188 using Impl = OpenSetImpl<operation, ColumnType>;
189 if (!check_column<ColumnType>(*left_data.nested_col)) {
190 return false;
191 }
192 constexpr auto execute_left_column_first = Impl::Action::execute_left_column_first;
193 size_t current = 0;
194 Impl impl;
195 for (size_t row = 0; row < left_data.offsets_ptr->size(); ++row) {
196 size_t count = 0;
197 size_t left_off = (*left_data.offsets_ptr)[row - 1];
198 size_t left_len = (*left_data.offsets_ptr)[row] - left_off;
199 size_t right_off = (*right_data.offsets_ptr)[row - 1];
200 size_t right_len = (*right_data.offsets_ptr)[row] - right_off;
201 if constexpr (execute_left_column_first) {
202 impl.template apply<true>(left_data, left_off, left_len, dst, &count);
203 impl.template apply<false>(right_data, right_off, right_len, dst, &count);
204 } else {
205 impl.template apply<false>(right_data, right_off, right_len, dst, &count);
206 impl.template apply<true>(left_data, left_off, left_len, dst, &count);
207 }
208 current += count;
209 dst.offsets_ptr->push_back(current);
210 impl.reset();
211 }
212 return true;
213 }
214};
