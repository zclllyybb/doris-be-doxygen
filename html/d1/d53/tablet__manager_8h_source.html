85 [](Tablet* t) { return t->is_used(); }) {
86 std::vector<TabletSharedPtr> res;
87 for (const auto& tablets_shard : _tablets_shards) {
88 std::shared_lock rdlock(tablets_shard.lock);
89 for (auto& [id, tablet] : tablets_shard.tablet_map) {
90 if (filter(tablet.get())) {

92 }
93 }
94 }
