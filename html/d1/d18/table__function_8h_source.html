57 virtual int get_value(MutableColumnPtr& column, int max_step) {
58 max_step = std::max(1, std::min(max_step, (int)(_cur_size - _cur_offset)));
59 int i = 0;
60 for (; i < max_step && !eos(); i++) {
61 get_value(column);
62 forward();
63 }
64 return i;
65 }

69 virtual Status forward(int step = 1) {
70 if (current_empty()) {
71 _eos = true;
72 } else {
73 _cur_offset += step;
74 if (_cur_offset >= _cur_size) {
75 _eos = true;
76 }
77 }
78 return Status::OK();
79 }

85 void set_nullable() { _is_nullable = true; }
86
87 bool is_outer() const { return _is_outer; }
88 void set_outer() {
89 if (is_outer()) {
90 return;
91 }
92 _is_outer = true;
93 _fn_name += COMBINATOR_SUFFIX_OUTER;
94 }
95
96 bool current_empty() const { return _cur_size == 0; }
97
98protected:
99 std::string _fn_name;
100 VExprContext* _vexpr_context = nullptr;
101 // true if there is no more data can be read from this function.
102 bool _eos = false;
103 // the position of current cursor
104 int64_t _cur_offset = 0;
105 // the size of current result
106 int64_t _cur_size = 0;
107 // set _is_outer to false for explode function, and should not return tuple while array is null or empty
108 bool _is_outer = false;
109
110 bool _is_nullable = false;
111 bool _is_const = false;
112};
