124 void insert(const Field& x) override {
125 const String& s = doris::vectorized::get<const String&>(x);
126 const size_t old_size = chars.size();
127 const size_t size_to_append = s.size();
128 const size_t new_size = old_size + size_to_append;
129
130 check_chars_length(new_size, old_size + 1);
131
132 chars.resize(new_size);
133 memcpy(chars.data() + old_size, s.c_str(), size_to_append);
134 offsets.push_back(new_size);
135 }

141 void insert_from(const IColumn& src_, size_t n) override {
142 const ColumnString& src = assert_cast<const ColumnString&>(src_);
143 const size_t size_to_append =

146 if (!size_to_append) {

189 void insert_many_strings_without_reserve(const StringRef* strings, size_t num) {
190 Char* data = chars.data();
191 size_t offset = chars.size();
192 size_t length = 0;
193
194 const char* ptr = strings[0].data;
195 for (size_t i = 0; i != num; i++) {
196 uint32_t len = strings[i].size;
197 length += len;
198 offset += len;
199 offsets.push_back(offset);
200
201 if (i != num - 1 && strings[i].data + len == strings[i + 1].data) {
202 continue;
203 }
204 memcpy(data, ptr, length);
205 data += length;
206 if (LIKELY(i != num - 1)) {
207 ptr = strings[i + 1].data;
208 length = 0;
209 }
210 }
211 check_chars_length(offset, offsets.size());
212 chars.resize(offset);
213 }

215 void insert_many_continuous_binary_data(const char* data, const uint32_t* offsets_,
216 const size_t num) override {
217 static_assert(sizeof(offsets_[0]) == sizeof(*offsets.data()));
218 if (UNLIKELY(num == 0)) {
219 return;
220 }
221 const auto old_size = chars.size();
222 const auto begin_offset = offsets_[0];
223 const size_t total_mem_size = offsets_[num] - begin_offset;
224 if (LIKELY(total_mem_size > 0)) {
225 check_chars_length(total_mem_size + old_size, offsets.size() + num);
226 chars.resize(total_mem_size + old_size);
227 memcpy(chars.data() + old_size, data + begin_offset, total_mem_size);
228 }
229 const auto old_rows = offsets.size();
230 auto tail_offset = offsets.back();
231 DCHECK(tail_offset == old_size);
232 offsets.resize(old_rows + num);
233 auto* offsets_ptr = &offsets[old_rows];
234
235 for (size_t i = 0; i < num; ++i) {
236 offsets_ptr[i] = tail_offset + offsets_[i + 1] - begin_offset;
237 }
238 DCHECK(chars.size() == offsets.back());
239 }

241 void insert_many_binary_data(char* data_array, uint32_t* len_array,
242 uint32_t* start_offset_array, size_t num) override {
243 size_t new_size = 0;
244 for (size_t i = 0; i < num; i++) {
245 new_size += len_array[i];
246 }
247
248 const size_t old_size = chars.size();
249 check_chars_length(old_size + new_size, offsets.size() + num);
250 chars.resize(old_size + new_size);
251
252 Char* data = chars.data();
253 size_t offset = old_size;
254 for (size_t i = 0; i < num; i++) {
255 uint32_t len = len_array[i];
256 uint32_t start_offset = start_offset_array[i];
257 // memcpy will deal len == 0, not do it here
258 memcpy(data + offset, data_array + start_offset, len);
259 offset += len;
260 offsets.push_back(offset);
261 }
262 }

315 void insert_many_strings_overflow(const StringRef* strings, size_t num,
316 size_t max_length) override {
317 if (max_length <= 8) {
318 insert_many_strings_fixed_length<8>(strings, num);
319 } else if (max_length <= 16) {
320 insert_many_strings_fixed_length<16>(strings, num);

322 insert_many_strings_fixed_length<32>(strings, num);

325 } else if (max_length <= 128) {

332 void insert_many_dict_data(const int32_t* data_array, size_t start_index, const StringRef* dict,
333 size_t num, uint32_t /*dict_num*/) override {
334 size_t offset_size = offsets.size();
335 size_t old_size = chars.size();
336 size_t new_size = old_size;
337 offsets.resize(offsets.size() + num);
338
339 for (size_t i = 0; i < num; i++) {
340 int32_t codeword = data_array[i + start_index];
341 new_size += dict[codeword].size;
342 offsets[offset_size + i] = new_size;
343 }
344
345 check_chars_length(new_size, offsets.size());
346 chars.resize(new_size);
347
348 for (size_t i = start_index; i < start_index + num; i++) {
349 int32_t codeword = data_array[i];
350 auto& src = dict[codeword];
351 memcpy(chars.data() + old_size, src.data, src.size);
352 old_size += src.size;
353 }
