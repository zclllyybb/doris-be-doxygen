44template <>
45inline bool add_overflow(__int128 x, __int128 y, __int128& res) {
46 static constexpr __int128 min_int128 = __int128(0x8000000000000000ll) << 64;
47 static constexpr __int128 max_int128 =
48 (__int128(0x7fffffffffffffffll) << 64) + 0xffffffffffffffffll;
49 res = x + y;
50 return (y > 0 && x > max_int128 - y) || (y < 0 && x < min_int128 - y);
51}

73template <>
74inline bool sub_overflow(__int128 x, __int128 y, __int128& res) {
75 static constexpr __int128 min_int128 = __int128(0x8000000000000000ll) << 64;
76 static constexpr __int128 max_int128 =
77 (__int128(0x7fffffffffffffffll) << 64) + 0xffffffffffffffffll;
78 res = x - y;
79 return (y < 0 && x > max_int128 + y) || (y > 0 && x < min_int128 + y);
80}

102template <>
103inline bool mul_overflow(__int128 x, __int128 y, __int128& res) {
104 res = static_cast<unsigned __int128>(x) *
105 static_cast<unsigned __int128>(y);
106 if (!x || !y) return false;
107
108 unsigned __int128 a = (x > 0) ? x : -x;
109 unsigned __int128 b = (y > 0) ? y : -y;
110 return (a * b) / b != a;
111}
