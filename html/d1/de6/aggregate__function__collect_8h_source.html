34template <typename T>
35struct AggregateFunctionCollectSetData {
36 using ElementType = T;
37 using ColVecType = ColumnVectorOrDecimal<ElementType>;
38 using ElementNativeType = typename NativeType<T>::Type;
39 using Set = HashSetWithStackMemory<ElementNativeType, DefaultHash<ElementNativeType>, 4>;
40 Set set;
41
42 void add(const IColumn& column, size_t row_num) {
43 const auto& vec = assert_cast<const ColVecType&>(column).get_data();
44 set.insert(vec[row_num]);
45 }
46 void merge(const AggregateFunctionCollectSetData& rhs) { set.merge(rhs.set); }
47 void write(BufferWritable& buf) const { set.write(buf); }
48 void read(BufferReadable& buf) { set.read(buf); }
49 void reset() { set.clear(); }
50 void insert_result_into(IColumn& to) const {
51 auto& vec = assert_cast<ColVecType&>(to).get_data();
52 vec.reserve(set.size());
53 for (auto item : set) {
54 vec.push_back(item.key);
55 }
56 }
57};

59template <>
60struct AggregateFunctionCollectSetData<StringRef> {
61 using ElementType = StringRef;
62 using ColVecType = ColumnString;
63 using Set = HashSetWithSavedHashWithStackMemory<ElementType, DefaultHash<ElementType>, 4>;
64 Set set;
65
66 void add(const IColumn& column, size_t row_num, Arena* arena) {
67 Set::LookupResult it;
68 bool inserted;
69 auto key_holder = get_key_holder<true>(column, row_num, *arena);
70 set.emplace(key_holder, it, inserted);
71 }
72
73 void merge(const AggregateFunctionCollectSetData& rhs, Arena* arena) {
74 Set::LookupResult it;
75 bool inserted;
76 for (const auto& elem : rhs.set) {
77 set.emplace(ArenaKeyHolder {elem.get_value(), *arena}, it, inserted);
78 }
79 }
80 void write(BufferWritable& buf) const {
81 write_var_uint(set.size(), buf);
82 for (const auto& elem : set) {
83 write_string_binary(elem.get_value(), buf);
84 }
85 }
86 void read(BufferReadable& buf) {
87 UInt64 rows;
88 read_var_uint(rows, buf);
89
90 StringRef ref;
91 for (size_t i = 0; i < rows; ++i) {
92 read_string_binary(ref, buf);
93 set.insert(ref);
94 }
95 }
96 void reset() { set.clear(); }
97 void insert_result_into(IColumn& to) const {
98 auto& vec = assert_cast<ColVecType&>(to);
99 vec.reserve(set.size());
100 for (const auto& item : set) {
101 vec.insert_data(item.key.data, item.key.size);
102 }
103 }
104};

106template <typename T>
107struct AggregateFunctionCollectListData {
108 using ElementType = T;
109 using ColVecType = ColumnVectorOrDecimal<ElementType>;
110 PaddedPODArray<ElementType> data;
111
112 void add(const IColumn& column, size_t row_num) {
113 const auto& vec = assert_cast<const ColVecType&>(column).get_data();
114 data.push_back(vec[row_num]);
115 }
116 void merge(const AggregateFunctionCollectListData& rhs) {
117 data.insert(rhs.data.begin(), rhs.data.end());
118 }
119 void write(BufferWritable& buf) const {
120 write_var_uint(data.size(), buf);
121 buf.write(data.raw_data(), data.size() * sizeof(ElementType));
122 }
123 void read(BufferReadable& buf) {
124 UInt64 rows = 0;
125 read_var_uint(rows, buf);
126 data.resize(rows);
127 buf.read(reinterpret_cast<char*>(data.data()), rows * sizeof(ElementType));
128 }
129 void reset() { data.clear(); }
130 void insert_result_into(IColumn& to) const {
131 auto& vec = assert_cast<ColVecType&>(to).get_data();
132 size_t old_size = vec.size();
133 vec.resize(old_size + data.size());
134 memcpy(vec.data() + old_size, data.data(), data.size() * sizeof(ElementType));
135 }
136};

138template <>
139struct AggregateFunctionCollectListData<StringRef> {
140 using ElementType = StringRef;
141 using ColVecType = ColumnString;
142 MutableColumnPtr data;
143
144 AggregateFunctionCollectListData() { data = ColVecType::create(); }
145
146 void add(const IColumn& column, size_t row_num) { data->insert_from(column, row_num); }
147
148 void merge(const AggregateFunctionCollectListData& rhs) {
149 data->insert_range_from(*rhs.data, 0, rhs.data->size());
150 }
151
152 void write(BufferWritable& buf) const {
153 auto& col = assert_cast<ColVecType&>(*data);
154
155 write_var_uint(col.size(), buf);
156 buf.write(col.get_offsets().raw_data(), col.size() * sizeof(IColumn::Offset));
157
158 write_var_uint(col.get_chars().size(), buf);
159 buf.write(col.get_chars().raw_data(), col.get_chars().size());
160 }
161
162 void read(BufferReadable& buf) {
163 auto& col = assert_cast<ColVecType&>(*data);
164 UInt64 offs_size = 0;
165 read_var_uint(offs_size, buf);
166 col.get_offsets().resize(offs_size);
167 buf.read(reinterpret_cast<char*>(col.get_offsets().data()),
168 offs_size * sizeof(IColumn::Offset));
169
170 UInt64 chars_size = 0;
171 read_var_uint(chars_size, buf);
172 col.get_chars().resize(chars_size);
173 buf.read(reinterpret_cast<char*>(col.get_chars().data()), chars_size);
174 }
175
176 void reset() { data->clear(); }
177
178 void insert_result_into(IColumn& to) const {
179 auto& to_str = assert_cast<ColVecType&>(to);
180 to_str.insert_range_from(*data, 0, data->size());
181 }
182};

184template <typename Data>
185class AggregateFunctionCollect final
186 : public IAggregateFunctionDataHelper<Data, AggregateFunctionCollect<Data>> {
187public:
188 static constexpr bool alloc_memory_in_arena =
189 std::is_same_v<Data, AggregateFunctionCollectSetData<StringRef>>;
190
191 AggregateFunctionCollect(const DataTypes& argument_types_)
192 : IAggregateFunctionDataHelper<Data, AggregateFunctionCollect<Data>>(argument_types_,
193 {}),
194 _argument_type(argument_types_[0]) {}
195
196 std::string get_name() const override {
197 if constexpr (std::is_same_v<AggregateFunctionCollectListData<typename Data::ElementType>,
198 Data>) {
199 return "collect_list";
200 } else {
201 return "collect_set";
202 }
203 }
204
205 DataTypePtr get_return_type() const override {
206 return std::make_shared<DataTypeArray>(make_nullable(_argument_type));
207 }
208
209 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
210 Arena* arena) const override {
211 assert(!columns[0]->is_null_at(row_num));
212 if constexpr (alloc_memory_in_arena) {
213 this->data(place).add(*columns[0], row_num, arena);
214 } else {
215 this->data(place).add(*columns[0], row_num);
216 }
217 }
218
219 void reset(AggregateDataPtr place) const override { this->data(place).reset(); }
220
221 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
222 Arena* arena) const override {
223 if constexpr (alloc_memory_in_arena) {
224 this->data(place).merge(this->data(rhs), arena);
225 } else {
226 this->data(place).merge(this->data(rhs));
227 }
228 }
229
230 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
231 this->data(place).write(buf);
232 }
233
234 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
235 Arena*) const override {
236 this->data(place).read(buf);
237 }
238
239 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
240 auto& to_arr = assert_cast<ColumnArray&>(to);
241 auto& to_nested_col = to_arr.get_data();
242 if (to_nested_col.is_nullable()) {
243 auto col_null = reinterpret_cast<ColumnNullable*>(&to_nested_col);
244 this->data(place).insert_result_into(col_null->get_nested_column());
245 col_null->get_null_map_data().resize_fill(col_null->get_nested_column().size(), 0);
246 } else {
247 this->data(place).insert_result_into(to_nested_col);
248 }
249 to_arr.get_offsets().push_back(to_nested_col.size());
250 }
251
252 bool allocates_memory_in_arena() const override { return alloc_memory_in_arena; }
253
254private:
255 DataTypePtr _argument_type;
256};
