36template <typename T, typename HasLimit>
37struct AggregateFunctionCollectSetData {
38 using ElementType = T;
39 using ColVecType = ColumnVectorOrDecimal<ElementType>;
40 using ElementNativeType = typename NativeType<T>::Type;
41 using SelfType = AggregateFunctionCollectSetData;
42 using Set = HashSetWithStackMemory<ElementNativeType, DefaultHash<ElementNativeType>, 4>;
43 Set data_set;
44 Int64 max_size = -1;
45
46 size_t size() const { return data_set.size(); }
47
48 void add(const IColumn& column, size_t row_num) {
49 data_set.insert(assert_cast<const ColVecType&>(column).get_data()[row_num]);
50 }
51
52 void merge(const SelfType& rhs) {
53 if constexpr (HasLimit::value) {
54 DCHECK(max_size == -1 || max_size == rhs.max_size);
55 max_size = rhs.max_size;
56
57 for (auto& rhs_elem : rhs.data_set) {
58 if (size() >= max_size) {
59 return;
60 }
61 data_set.insert(rhs_elem.get_value());
62 }
63 } else {
64 data_set.merge(rhs.data_set);
65 }
66 }
67
68 void write(BufferWritable& buf) const {
69 data_set.write(buf);
70 write_var_int(max_size, buf);
71 }
72
73 void read(BufferReadable& buf) {
74 data_set.read(buf);
75 read_var_int(max_size, buf);
76 }
77
78 void insert_result_into(IColumn& to) const {
79 auto& vec = assert_cast<ColVecType&>(to).get_data();
80 vec.reserve(size());
81 for (const auto& item : data_set) {
82 vec.push_back(item.key);
83 }
84 }
85
86 void reset() { data_set.clear(); }
87};

89template <typename HasLimit>
90struct AggregateFunctionCollectSetData<StringRef, HasLimit> {
91 using ElementType = StringRef;
92 using ColVecType = ColumnString;
93 using SelfType = AggregateFunctionCollectSetData<ElementType, HasLimit>;
94 using Set = HashSetWithSavedHashWithStackMemory<ElementType, DefaultHash<ElementType>, 4>;
95 Set data_set;
96 Int64 max_size = -1;
97
98 size_t size() const { return data_set.size(); }
99
100 void add(const IColumn& column, size_t row_num, Arena* arena) {
101 Set::LookupResult it;
102 bool inserted;
103 auto key_holder = get_key_holder<true>(column, row_num, *arena);
104 data_set.emplace(key_holder, it, inserted);
105 }
106
107 void merge(const SelfType& rhs, Arena* arena) {
108 bool inserted;
109 Set::LookupResult it;
110 DCHECK(max_size == -1 || max_size == rhs.max_size);
111 max_size = rhs.max_size;
112
113 for (auto& rhs_elem : rhs.data_set) {
114 if constexpr (HasLimit::value) {
115 if (size() >= max_size) {
116 return;
117 }
118 }
119 assert(arena != nullptr);
120 data_set.emplace(ArenaKeyHolder {rhs_elem.get_value(), *arena}, it, inserted);
121 }
122 }
123
124 void write(BufferWritable& buf) const {
125 write_var_uint(size(), buf);
126 for (const auto& elem : data_set) {
127 write_string_binary(elem.get_value(), buf);
128 }
129 write_var_int(max_size, buf);
130 }
131
132 void read(BufferReadable& buf) {
133 UInt64 size;
134 read_var_uint(size, buf);
135 StringRef ref;
136 for (size_t i = 0; i < size; ++i) {
137 read_string_binary(ref, buf);
138 data_set.insert(ref);
139 }
140 read_var_int(max_size, buf);
141 }
142
143 void insert_result_into(IColumn& to) const {
144 auto& vec = assert_cast<ColVecType&>(to);
145 vec.reserve(size());
146 for (const auto& item : data_set) {
147 vec.insert_data(item.key.data, item.key.size);
148 }
149 }
150
151 void reset() { data_set.clear(); }
152};

154template <typename T, typename HasLimit>
155struct AggregateFunctionCollectListData {
156 using ElementType = T;
157 using ColVecType = ColumnVectorOrDecimal<ElementType>;
158 using SelfType = AggregateFunctionCollectListData<ElementType, HasLimit>;
159 PaddedPODArray<ElementType> data;
160 Int64 max_size = -1;
161
162 size_t size() const { return data.size(); }
163
164 void add(const IColumn& column, size_t row_num) {
165 const auto& vec = assert_cast<const ColVecType&>(column).get_data();
166 data.push_back(vec[row_num]);
167 }
168
169 void merge(const SelfType& rhs) {
170 if constexpr (HasLimit::value) {
171 DCHECK(max_size == -1 || max_size == rhs.max_size);
172 max_size = rhs.max_size;
173 for (auto& rhs_elem : rhs.data) {
174 if (size() >= max_size) {
175 return;
176 }
177 data.push_back(rhs_elem);
178 }
179 } else {
180 data.insert(rhs.data.begin(), rhs.data.end());
181 }
182 }
183
184 void write(BufferWritable& buf) const {
185 write_var_uint(size(), buf);
186 buf.write(data.raw_data(), size() * sizeof(ElementType));
187 write_var_int(max_size, buf);
188 }
189
190 void read(BufferReadable& buf) {
191 UInt64 rows = 0;
192 read_var_uint(rows, buf);
193 data.resize(rows);
194 buf.read(reinterpret_cast<char*>(data.data()), rows * sizeof(ElementType));
195 read_var_int(max_size, buf);
196 }
197
198 void reset() { data.clear(); }
199
200 void insert_result_into(IColumn& to) const {
201 auto& vec = assert_cast<ColVecType&>(to).get_data();
202 size_t old_size = vec.size();
203 vec.resize(old_size + size());
204 memcpy(vec.data() + old_size, data.data(), size() * sizeof(ElementType));
205 }
206};

208template <typename HasLimit>
209struct AggregateFunctionCollectListData<StringRef, HasLimit> {
210 using ElementType = StringRef;
211 using ColVecType = ColumnString;
212 MutableColumnPtr data;
213 Int64 max_size = -1;
214
215 AggregateFunctionCollectListData() { data = ColVecType::create(); }
216
217 size_t size() const { return data->size(); }
218
219 void add(const IColumn& column, size_t row_num) { data->insert_from(column, row_num); }
220
221 void merge(const AggregateFunctionCollectListData& rhs) {
222 if constexpr (HasLimit::value) {
223 DCHECK(max_size == -1 || max_size == rhs.max_size);
224 max_size = rhs.max_size;
225
226 data->insert_range_from(*rhs.data, 0,
227 std::min(static_cast<size_t>(max_size - size()), rhs.size()));
228 } else {
229 data->insert_range_from(*rhs.data, 0, rhs.size());
230 }
231 }
232
233 void write(BufferWritable& buf) const {
234 auto& col = assert_cast<ColVecType&>(*data);
235
236 write_var_uint(col.size(), buf);
237 buf.write(col.get_offsets().raw_data(), col.size() * sizeof(IColumn::Offset));
238
239 write_var_uint(col.get_chars().size(), buf);
240 buf.write(col.get_chars().raw_data(), col.get_chars().size());
241 write_var_int(max_size, buf);
242 }
243
244 void read(BufferReadable& buf) {
245 auto& col = assert_cast<ColVecType&>(*data);
246 UInt64 offs_size = 0;
247 read_var_uint(offs_size, buf);
248 col.get_offsets().resize(offs_size);
249 buf.read(reinterpret_cast<char*>(col.get_offsets().data()),
250 offs_size * sizeof(IColumn::Offset));
251
252 UInt64 chars_size = 0;
253 read_var_uint(chars_size, buf);
254 col.get_chars().resize(chars_size);
255 buf.read(reinterpret_cast<char*>(col.get_chars().data()), chars_size);
256 read_var_int(max_size, buf);
257 }
258
259 void reset() { data->clear(); }
260
261 void insert_result_into(IColumn& to) const {
262 auto& to_str = assert_cast<ColVecType&>(to);
263 to_str.insert_range_from(*data, 0, size());
264 }
265};

267template <typename Data, typename HasLimit>
268class AggregateFunctionCollect
269 : public IAggregateFunctionDataHelper<Data, AggregateFunctionCollect<Data, HasLimit>> {
270 using GenericType = AggregateFunctionCollectSetData<StringRef, HasLimit>;
271
272 static constexpr bool ENABLE_ARENA = std::is_same_v<Data, GenericType>;
273
274public:
275 AggregateFunctionCollect(const DataTypes& argument_types,
276 UInt64 max_size_ = std::numeric_limits<UInt64>::max())
277 : IAggregateFunctionDataHelper<Data, AggregateFunctionCollect<Data, HasLimit>>(
278 {argument_types}),
279 return_type(argument_types[0]) {}
280
281 std::string get_name() const override {
282 if constexpr (std::is_same_v<AggregateFunctionCollectListData<typename Data::ElementType,
283 HasLimit>,
284 Data>) {
285 return "collect_list";
286 } else {
287 return "collect_set";
288 }
289 }
290
291 DataTypePtr get_return_type() const override {
292 return std::make_shared<DataTypeArray>(make_nullable(return_type));
293 }
294
295 bool allocates_memory_in_arena() const override { return ENABLE_ARENA; }
296
297 void add(AggregateDataPtr __restrict place, const IColumn** columns, size_t row_num,
298 Arena* arena) const override {
299 auto& data = this->data(place);
300 if constexpr (HasLimit::value) {
301 if (data.max_size == -1) {
302 data.max_size =
303 (UInt64) static_cast<const ColumnInt32*>(columns[1])->get_element(row_num);
304 }
305 if (data.size() >= data.max_size) {
306 return;
307 }
308 }
309 if constexpr (ENABLE_ARENA) {
310 data.add(*columns[0], row_num, arena);
311 } else {
312 data.add(*columns[0], row_num);
313 }
314 }
315
316 void merge(AggregateDataPtr __restrict place, ConstAggregateDataPtr rhs,
317 Arena* arena) const override {
318 auto& data = this->data(place);
319 auto& rhs_data = this->data(rhs);
320 if constexpr (ENABLE_ARENA) {
321 data.merge(rhs_data, arena);
322 } else {
323 data.merge(rhs_data);
324 }
325 }
326
327 void serialize(ConstAggregateDataPtr __restrict place, BufferWritable& buf) const override {
328 this->data(place).write(buf);
329 }
330
331 void deserialize(AggregateDataPtr __restrict place, BufferReadable& buf,
332 Arena*) const override {
333 this->data(place).read(buf);
334 }
335
336 void insert_result_into(ConstAggregateDataPtr __restrict place, IColumn& to) const override {
337 auto& to_arr = assert_cast<ColumnArray&>(to);
338 auto& to_nested_col = to_arr.get_data();
339 if (to_nested_col.is_nullable()) {
340 auto col_null = reinterpret_cast<ColumnNullable*>(&to_nested_col);
341 this->data(place).insert_result_into(col_null->get_nested_column());
342 col_null->get_null_map_data().resize_fill(col_null->get_nested_column().size(), 0);
343 } else {
344 this->data(place).insert_result_into(to_nested_col);
345 }
346 to_arr.get_offsets().push_back(to_nested_col.size());
347 }
348
349private:
350 DataTypePtr return_type;
351};
