29// convert int to a hex format string, buf must enough to hold converted hex string
30template <typename T>
31void to_hex(T val, char* buf) {
32 static const char* digits = "0123456789abcdef";
33 for (int i = 0; i < 2 * sizeof(T); ++i) {
34 buf[2 * sizeof(T) - 1 - i] = digits[val & 0x0F];
35 val >>= 4;
36 }
37}

39template <typename T>
40void from_hex(T* ret, const std::string& buf) {
41 T val = 0;
42 for (int i = 0; i < buf.length(); ++i) {
43 int buf_val = 0;
44 if (buf.c_str()[i] >= '0' && buf.c_str()[i] <= '9') {
45 buf_val = buf.c_str()[i] - '0';
46 } else {
47 buf_val = buf.c_str()[i] - 'a' + 10;
48 }
49 val <<= 4;
50 val = val | buf_val;
51 }
52 *ret = val;
53}
