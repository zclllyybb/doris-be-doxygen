77 // Note that, If call the memory allocation operation in Memory Hook,
78 // such as calling LOG/iostream/sstream/stringstream/etc. related methods,
79 // must increase the control to avoid entering infinite recursion, otherwise it may cause crash or stuck,
80 // Returns whether the memory exceeds limit, and will consume mem trcker no matter whether the limit is exceeded.
81 void consume(int64_t size);
82 // If the memory exceeds the limit, return false, and will not consume mem tracker.
83 bool try_consume(int64_t size);
84
85 // Force is equal to false. When the memory exceeds the limit,this alloc will be terminated and false
86 // will be returned.
87 // Force is equal to true, even if the memory is found to be overrun, continue to consume mem tracker,
88 // because this time alloc will still actually allocate memory, and always return true.
89 template <bool CheckLimit, bool Force>
90 bool flush_untracked_mem();

110 fmt::memory_buffer consumer_tracker_buf;

121private:
122 void cancel_fragment();
123 void exceeded(int64_t size);
124

126 _exceed_mem_limit_msg = _limiter_tracker_raw->mem_limit_exceeded(

128 }

130private:
131 // is false: ExecEnv::GetInstance()->initialized() = false when thread local is initialized
132 bool _init = false;
133 // Cache untracked mem.
134 int64_t _untracked_mem = 0;
135 int64_t old_untracked_mem = 0;
136
137 bool _count_scope_mem = false;
138 int64_t _scope_mem = 0;
139
140 std::string _failed_consume_msg = std::string();
141 std::string _exceed_mem_limit_msg = std::string();
142 bool _is_process_exceed = false;
143 bool _wait_gc = true;
144
145 std::shared_ptr<MemTrackerLimiter> _limiter_tracker;
146 MemTrackerLimiter* _limiter_tracker_raw = nullptr;
147 std::vector<MemTracker*> _consumer_tracker_stack;
148
149 // If true, call memtracker try_consume, otherwise call consume.
150 bool _check_limit = false;
151 // If there is a memory new/delete operation in the consume method, it may enter infinite recursion.
152 bool _stop_consume = false;
153 TUniqueId _fragment_instance_id = TUniqueId();
154 ExceedCallBack _cb_func = nullptr;

157inline void ThreadMemTrackerMgr::init() {
158 DCHECK(_limiter_tracker == nullptr);
159 _limiter_tracker = ExecEnv::GetInstance()->orphan_mem_tracker();
160 _limiter_tracker_raw = ExecEnv::GetInstance()->orphan_mem_tracker_raw();
161 _check_limit = true;
162 _init = true;
163}

165inline bool ThreadMemTrackerMgr::push_consumer_tracker(MemTracker* tracker) {
166 DCHECK(tracker) << print_debug_string();
167 if (std::count(_consumer_tracker_stack.begin(), _consumer_tracker_stack.end(), tracker)) {
168 return false;
169 }
170 _consumer_tracker_stack.push_back(tracker);
171 tracker->release(_untracked_mem);
172 return true;
173}
174
175inline void ThreadMemTrackerMgr::pop_consumer_tracker() {
176 DCHECK(!_consumer_tracker_stack.empty());
177 _consumer_tracker_stack.back()->consume(_untracked_mem);
178 _consumer_tracker_stack.pop_back();
179}

181inline void ThreadMemTrackerMgr::consume(int64_t size) {
182 _untracked_mem += size;
183 // When some threads `0 < _untracked_mem < config::mem_tracker_consume_min_size_bytes`
184 // and some threads `_untracked_mem <= -config::mem_tracker_consume_min_size_bytes` trigger consumption(),
185 // it will cause tracker->consumption to be temporarily less than 0.
186 // After the jemalloc hook is loaded, before ExecEnv init, _limiter_tracker=nullptr.
187 if ((_untracked_mem >= config::mem_tracker_consume_min_size_bytes ||
188 _untracked_mem <= -config::mem_tracker_consume_min_size_bytes) &&
189 !_stop_consume && ExecEnv::GetInstance()->initialized()) {
190 if (_check_limit) {
191 flush_untracked_mem<true, true>();
192 } else {
193 flush_untracked_mem<false, true>();
194 }
195 }
196}
197
198inline bool ThreadMemTrackerMgr::try_consume(int64_t size) {
199 _untracked_mem += size;
200 if ((_untracked_mem >= config::mem_tracker_consume_min_size_bytes ||
201 _untracked_mem <= -config::mem_tracker_consume_min_size_bytes) &&
202 !_stop_consume && ExecEnv::GetInstance()->initialized()) {
203 if (_check_limit) {
204 return flush_untracked_mem<true, false>();
205 } else {
206 return flush_untracked_mem<false, true>();
207 }
208 }
209 return true;
210}

212template <bool CheckLimit, bool Force>
213inline bool ThreadMemTrackerMgr::flush_untracked_mem() {
214 // Temporary memory may be allocated during the consumption of the mem tracker, which will lead to entering
215 // the Memory Hook again, so suspend consumption to avoid falling into an infinite loop.
216 _stop_consume = true;
217 if (!_init) init(); // ExecEnv not initialized when thread is created.
218 DCHECK(_limiter_tracker_raw);
219 old_untracked_mem = _untracked_mem;
220 if (_count_scope_mem) _scope_mem += _untracked_mem;
221 if (CheckLimit) {
222 if (!_limiter_tracker_raw->try_consume(old_untracked_mem, _failed_consume_msg,
223 _is_process_exceed)) {
224 if (Force) _limiter_tracker_raw->consume(old_untracked_mem);
225 save_exceed_mem_limit_msg();
226 exceeded(old_untracked_mem);
227 if (!Force) return false;
228 }
229 } else {
230 _limiter_tracker_raw->consume(old_untracked_mem);
231 }
232 for (auto tracker : _consumer_tracker_stack) {
233 tracker->consume(old_untracked_mem);
234 }
235 _untracked_mem -= old_untracked_mem;
236 _stop_consume = false;
237 return true;
238}
