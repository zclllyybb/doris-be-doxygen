35template <typename A, typename B>
36struct DivideIntegralImpl {
37 using ResultType = typename NumberTraits::ResultOfIntegerDivision<A, B>::Type;
38 using Traits = NumberTraits::BinaryOperatorTraits<A, B>;
39
40 template <typename Result = ResultType>
41 static void apply(const typename Traits::ArrayA& a, B b,
42 typename ColumnVector<Result>::Container& c,
43 typename Traits::ArrayNull& null_map) {
44 size_t size = c.size();
45 UInt8 is_null = b == 0;
46 memset(null_map.data(), is_null, size);
47
48 if (!is_null) {
49 if constexpr (!std::is_floating_point_v<A> && !std::is_same_v<A, Int128> &&
50 !std::is_same_v<A, Int8> && !std::is_same_v<A, UInt8>) {
51 const auto divider = libdivide::divider<A>(b);
52 for (size_t i = 0; i < size; i++) {
53 c[i] = a[i] / divider;
54 }
55 } else {
56 for (size_t i = 0; i < size; i++) {
57 c[i] = a[i] / b;
58 }
59 }
60 }
61 }
62
63 template <typename Result = ResultType>
64 static inline Result apply(A a, B b, UInt8& is_null) {
65 is_null = b == 0;
66 return a / (b + is_null);
67 }
68};
