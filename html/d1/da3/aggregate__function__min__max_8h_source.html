272struct SingleValueDataString {
273private:
274 using Self = SingleValueDataString;
275
276 Int32 size = -1;
277 Int32 capacity = 0;
278 char* large_data = nullptr;
279
280public:
281 static constexpr Int32 AUTOMATIC_STORAGE_SIZE = 64;
282 static constexpr Int32 MAX_SMALL_STRING_SIZE =
283 AUTOMATIC_STORAGE_SIZE - sizeof(size) - sizeof(capacity) - sizeof(large_data);
284
285private:
286 char small_data[MAX_SMALL_STRING_SIZE];
287
288public:
289 ~SingleValueDataString() { delete[] large_data; }
290
291 constexpr static bool IsFixedLength = false;
292
293 bool has() const { return size >= 0; }
294
295 const char* get_data() const { return size <= MAX_SMALL_STRING_SIZE ? small_data : large_data; }
296
297 void insert_result_into(IColumn& to) const {
298 if (has()) {
299 assert_cast<ColumnString&>(to).insert_data(get_data(), size);
300 } else {
301 assert_cast<ColumnString&>(to).insert_default();
302 }
303 }
304
305 void reset() {
306 if (size != -1) {
307 size = -1;
308 capacity = 0;
309 delete[] large_data;
310 large_data = nullptr;
311 }
312 }
313
314 void write(BufferWritable& buf) const {
315 write_binary(size, buf);
316 if (has()) {
317 buf.write(get_data(), size);
318 }
319 }
320
321 void read(BufferReadable& buf) {
322 Int32 rhs_size;
323 read_binary(rhs_size, buf);
324
325 if (rhs_size >= 0) {
326 if (rhs_size <= MAX_SMALL_STRING_SIZE) {
328
329 size = rhs_size;
330
331 if (size > 0) {
332 buf.read(small_data, size);
333 }
334 } else {
335 if (capacity < rhs_size) {
336 capacity = static_cast<UInt32>(round_up_to_power_of_two_or_zero(rhs_size));
337 delete[] large_data;
338 large_data = new char[capacity];
339 }
340
341 size = rhs_size;
342 buf.read(large_data, size);
343 }
344 } else {
346 size = rhs_size;
347 }
348 }
349
350 StringRef get_string_ref() const { return StringRef(get_data(), size); }
351
353 void change_impl(StringRef value, Arena* arena) {
354 Int32 value_size = value.size;
355
356 if (value_size <= MAX_SMALL_STRING_SIZE) {
358 size = value_size;
359
360 if (size > 0) {
361 memcpy(small_data, value.data, size);
362 }
363 } else {
364 if (capacity < value_size) {
366 capacity = round_up_to_power_of_two_or_zero(value_size);
367 delete[] large_data;
368 large_data = new char[capacity];
369 }
370
371 size = value_size;
372 memcpy(large_data, value.data, size);
373 }
374 }
375
376 void change(const IColumn& column, size_t row_num, Arena* arena) {
377 change_impl(assert_cast<const ColumnString&>(column).get_data_at(row_num), arena);
378 }
379
380 void change(const Self& to, Arena* arena) { change_impl(to.get_string_ref(), arena); }
381
382 bool change_if_less(const IColumn& column, size_t row_num, Arena* arena) {
383 if (!has() ||
384 assert_cast<const ColumnString&>(column).get_data_at(row_num) < get_string_ref()) {
385 change(column, row_num, arena);
386 return true;
387 } else {
388 return false;
389 }
390 }
391
392 bool change_if_greater(const IColumn& column, size_t row_num, Arena* arena) {
393 if (!has() ||
394 assert_cast<const ColumnString&>(column).get_data_at(row_num) > get_string_ref()) {
395 change(column, row_num, arena);
396 return true;
397 } else {
398 return false;
399 }
400 }
401
402 bool change_if_less(const Self& to, Arena* arena) {
403 if (to.has() && (!has() || to.get_string_ref() < get_string_ref())) {
404 change(to, arena);
405 return true;
406 } else {
407 return false;
408 }
409 }
410
411 bool change_if_greater(const Self& to, Arena* arena) {
412 if (to.has() && (!has() || to.get_string_ref() > get_string_ref())) {
413 change(to, arena);
414 return true;
415 } else {
416 return false;
417 }
418 }
419
420 bool change_first_time(const IColumn& column, size_t row_num, Arena* arena) {
421 if (!has()) {
422 change(column, row_num, arena);
423 return true;
424 } else {
425 return false;
426 }
427 }
428
429 bool change_first_time(const Self& to, Arena* arena) {
430 if (!has() && to.has()) {
431 change(to, arena);
432 return true;
433 } else {
434 return false;
435 }
436 }
437
438 bool is_equal_to(const Self& to) const {
439 return has() && to.get_string_ref() == get_string_ref();
440 }
441
442 bool is_equal_to(const IColumn& column, size_t row_num) const { return false; }
