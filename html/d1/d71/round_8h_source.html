38enum class ScaleMode {

63template <typename T, RoundingMode rounding_mode, ScaleMode scale_mode,
64 TieBreakingMode tie_breaking_mode>
65struct IntegerRoundingComputation {
66 static const size_t data_count = 1;
67
68 static size_t prepare(size_t scale) { return scale; }
69
71 static ALWAYS_INLINE T compute_impl(T x, T scale, T target_scale) {
72 switch (rounding_mode) {
73 case RoundingMode::Trunc: {
74 return target_scale > 1 ? x / scale * target_scale : x / scale;
75 }
76 case RoundingMode::Floor: {
77 if (x < 0) {
78 x -= scale - 1;
79 }
80 return target_scale > 1 ? x / scale * target_scale : x / scale;
81 }
82 case RoundingMode::Ceil: {
83 if (x >= 0) {
84 x += scale - 1;
85 }
86 return target_scale > 1 ? x / scale * target_scale : x / scale;
87 }
88 case RoundingMode::Round: {
89 if (x < 0) {
90 x -= scale;
91 }
92 switch (tie_breaking_mode) {
93 case TieBreakingMode::Auto: {
94 x = (x + scale / 2) / scale;
95 break;
96 }
97 case TieBreakingMode::Bankers: {
98 T quotient = (x + scale / 2) / scale;
99 if (quotient * scale == x + scale / 2) {
100 // round half to even
101 x = (quotient + (x < 0)) & ~1;
102 } else {
103 // round the others as usual
104 x = quotient;
105 }
106 break;
107 }
108 }
109 return target_scale > 1 ? x * target_scale : x;
110 }
111 }
112
113 __builtin_unreachable();
114 }
115
116 static ALWAYS_INLINE T compute(T x, T scale, size_t target_scale) {
117 switch (scale_mode) {
118 case ScaleMode::Zero:
119 case ScaleMode::Positive:
120 return x;
121 case ScaleMode::Negative:
122 return compute_impl(x, scale, target_scale);
123 }
124
125 __builtin_unreachable();
126 }
127
128 static ALWAYS_INLINE void compute(const T* __restrict in, size_t scale, T* __restrict out,
129 size_t target_scale) {
130 if constexpr (sizeof(T) <= sizeof(scale) && scale_mode == ScaleMode::Negative) {
131 if (scale > size_t(std::numeric_limits<T>::max())) {
132 *out = 0;
133 return;
134 }
135 }
136 *out = compute(*in, scale, target_scale);
137 }
138};

140template <typename T, RoundingMode rounding_mode, TieBreakingMode tie_breaking_mode>
141class DecimalRoundingImpl {
142private:
143 using NativeType = typename T::NativeType;
144 using Op = IntegerRoundingComputation<NativeType, rounding_mode, ScaleMode::Negative,
145 tie_breaking_mode>;
146 using Container = typename ColumnDecimal<T>::Container;
147
148public:
149 static NO_INLINE void apply(const Container& in, UInt32 in_scale, Container& out,
150 Int16 out_scale) {
151 Int16 scale_arg = in_scale - out_scale;
152 if (scale_arg > 0) {
153 size_t scale = int_exp10(scale_arg);
154
155 const NativeType* __restrict p_in = reinterpret_cast<const NativeType*>(in.data());
156 const NativeType* end_in = reinterpret_cast<const NativeType*>(in.data()) + in.size();
157 NativeType* __restrict p_out = reinterpret_cast<NativeType*>(out.data());
158
159 if (out_scale < 0) {
160 size_t target_scale = int_exp10(-out_scale);
161 while (p_in < end_in) {
162 Op::compute(p_in, scale, p_out, target_scale);
163 ++p_in;
164 ++p_out;
165 }
166 } else {
167 while (p_in < end_in) {
168 Op::compute(p_in, scale, p_out, 1);
169 ++p_in;
170 ++p_out;
171 }
172 }
173 } else {
174 memcpy(out.data(), in.data(), in.size() * sizeof(T));
175 }
176 }
177};

282template <typename T, RoundingMode rounding_mode, ScaleMode scale_mode>
283class FloatRoundingComputation : public BaseFloatRoundingComputation<T> {
284 using Base = BaseFloatRoundingComputation<T>;
285
286public:
287 static inline void compute(const T* __restrict in, const typename Base::VectorType& scale,
288 T* __restrict out) {
289 auto val = Base::load(in);
290
291 if (scale_mode == ScaleMode::Positive) {
292 val = Base::multiply(val, scale);
293 } else if (scale_mode == ScaleMode::Negative) {
294 val = Base::divide(val, scale);
295 }
296
297 val = Base::template apply<rounding_mode>(val);
298
299 if (scale_mode == ScaleMode::Positive) {
300 val = Base::divide(val, scale);
301 } else if (scale_mode == ScaleMode::Negative) {
302 val = Base::multiply(val, scale);
303 }
304
305 Base::store(out, val);
306 }
307};

311template <typename T, RoundingMode rounding_mode, ScaleMode scale_mode>
312struct FloatRoundingImpl {
313private:
314 static_assert(!IsDecimalNumber<T>);
315
316 using Op = FloatRoundingComputation<T, rounding_mode, scale_mode>;
317 using Data = std::array<T, Op::data_count>;
318 using ColumnType = ColumnVector<T>;
319 using Container = typename ColumnType::Container;
320
321public:
322 static NO_INLINE void apply(const Container& in, size_t scale, Container& out) {
323 auto mm_scale = Op::prepare(scale);
324
325 const size_t data_count = std::tuple_size<Data>();
326
327 const T* end_in = in.data() + in.size();
328 const T* limit = in.data() + in.size() / data_count * data_count;
329
330 const T* __restrict p_in = in.data();
331 T* __restrict p_out = out.data();
332
333 while (p_in < limit) {
334 Op::compute(p_in, mm_scale, p_out);
335 p_in += data_count;
336 p_out += data_count;
337 }
338
339 if (p_in < end_in) {
340 Data tmp_src {{}};
341 Data tmp_dst;
342
343 size_t tail_size_bytes = (end_in - p_in) * sizeof(*p_in);
344
345 memcpy(&tmp_src, p_in, tail_size_bytes);
346 Op::compute(reinterpret_cast<T*>(&tmp_src), mm_scale, reinterpret_cast<T*>(&tmp_dst));
347 memcpy(p_out, &tmp_dst, tail_size_bytes);
348 }
349 }
350};

425template <typename T, RoundingMode rounding_mode, TieBreakingMode tie_breaking_mode>
426struct Dispatcher {
427 template <ScaleMode scale_mode>
428 using FunctionRoundingImpl = std::conditional_t<
429 IsDecimalNumber<T>, DecimalRoundingImpl<T, rounding_mode, tie_breaking_mode>,
430 std::conditional_t<
431 std::is_floating_point_v<T>, FloatRoundingImpl<T, rounding_mode, scale_mode>,
432 IntegerRoundingImpl<T, rounding_mode, scale_mode, tie_breaking_mode>>>;
433
434 static ColumnPtr apply(const IColumn* col_general, Int16 scale_arg) {
435 if constexpr (IsNumber<T>) {
436 const auto* const col = check_and_get_column<ColumnVector<T>>(col_general);
437 auto col_res = ColumnVector<T>::create();
438
439 typename ColumnVector<T>::Container& vec_res = col_res->get_data();
440 vec_res.resize(col->get_data().size());
441
442 if (!vec_res.empty()) {
443 if (scale_arg == 0) {
444 size_t scale = 1;
445 FunctionRoundingImpl<ScaleMode::Zero>::apply(col->get_data(), scale, vec_res);
446 } else if (scale_arg > 0) {
447 size_t scale = int_exp10(scale_arg);
448 FunctionRoundingImpl<ScaleMode::Positive>::apply(col->get_data(), scale,
449 vec_res);
450 } else {
451 size_t scale = int_exp10(-scale_arg);
452 FunctionRoundingImpl<ScaleMode::Negative>::apply(col->get_data(), scale,
453 vec_res);
454 }
455 }
456
457 return col_res;
458 } else if constexpr (IsDecimalNumber<T>) {
459 const auto* const decimal_col = check_and_get_column<ColumnDecimal<T>>(col_general);
460 const auto& vec_src = decimal_col->get_data();
461
462 auto col_res = ColumnDecimal<T>::create(vec_src.size(), scale_arg);
463 auto& vec_res = col_res->get_data();
464
465 if (!vec_res.empty()) {
466 FunctionRoundingImpl<ScaleMode::Negative>::apply(
467 decimal_col->get_data(), decimal_col->get_scale(), vec_res, scale_arg);
468 }
469
470 return col_res;
471 } else {
472 __builtin_unreachable();
473 return nullptr;
474 }
475 }
476};

478template <typename Impl, RoundingMode rounding_mode, TieBreakingMode tie_breaking_mode>
479class FunctionRounding : public IFunction {
480public:
481 static constexpr auto name = Impl::name;
482 static FunctionPtr create() { return std::make_shared<FunctionRounding>(); }
483
484 String get_name() const override { return name; }
485
486 bool is_variadic() const override { return true; }
487 size_t get_number_of_arguments() const override { return 0; }
488
489 DataTypes get_variadic_argument_types_impl() const override {
490 return Impl::get_variadic_argument_types();
491 }
492
494 DataTypePtr get_return_type_impl(const DataTypes& arguments) const override {
495 if ((arguments.empty()) || (arguments.size() > 2)) {
496 LOG(FATAL) << "Number of arguments for function " + get_name() +
497 " doesn't match: should be 1 or 2. ";
498 }
499
500 return arguments[0];
501 }
502
503 static Status get_scale_arg(const ColumnWithTypeAndName& arguments, Int16* scale) {
504 const IColumn& scale_column = *arguments.column;
505
506 Int32 scale64 = static_cast<const ColumnInt32*>(
507 &(is_column_const(scale_column)
508 ? static_cast<const ColumnConst*>(&scale_column)
509 ->get_data_column()
510 : scale_column))
511 ->get_element(0);
512
513 if (scale64 > std::numeric_limits<Int16>::max() ||
514 scale64 < std::numeric_limits<Int16>::min()) {
515 return Status::InvalidArgument("Scale argument for function {} is out of bound: {}",
516 name, scale64);
517 }
518
519 *scale = scale64;
520 return Status::OK();
521 }
522
523 bool use_default_implementation_for_constants() const override { return true; }
524 ColumnNumbers get_arguments_that_are_always_constant() const override { return {1}; }
525
526 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
527 size_t result, size_t /*input_rows_count*/) override {
528 const ColumnWithTypeAndName& column = block.get_by_position(arguments[0]);
529 Int16 scale_arg = 0;
530 if (arguments.size() == 2) {
531 RETURN_IF_ERROR(get_scale_arg(block.get_by_position(arguments[1]), &scale_arg));
532 }
533
534 ColumnPtr res;
535 auto call = [&](const auto& types) -> bool {
536 using Types = std::decay_t<decltype(types)>;
537 using DataType = typename Types::LeftType;
538
539 if constexpr (IsDataTypeNumber<DataType> || IsDataTypeDecimal<DataType>) {
540 using FieldType = typename DataType::FieldType;
541 res = Dispatcher<FieldType, rounding_mode, tie_breaking_mode>::apply(
542 column.column.get(), scale_arg);
543 return true;
544 }
545 return false;
546 };
547
548#if !defined(__SSE4_1__)
551
552 if constexpr (rounding_mode == RoundingMode::Round) {
553 if (0 != fesetround(FE_TONEAREST)) {
554 return Status::InvalidArgument("Cannot set floating point rounding mode");
555 }
556 }
557#endif
558
559 if (!call_on_index_and_data_type<void>(column.type->get_type_id(), call)) {
560 return Status::InvalidArgument("Invalid argument type {} for function {}",
561 column.type->get_name(), name);
562 }
563
564 block.replace_by_position(result, std::move(res));
565 return Status::OK();
566 }
567};
