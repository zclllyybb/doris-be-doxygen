24// as well as "Google" versions of all the standards: ghtonl, ghtons, and
25// so on. These functions do exactly the same as their standard variants,

39inline uint64 gbswap_64(uint64 host_int) {
40#if defined(__GNUC__) && defined(__x86_64__) && !defined(__APPLE__)
41 // Adapted from /usr/include/byteswap.h. Not available on Mac.
42 if (__builtin_constant_p(host_int)) {
43 return __bswap_constant_64(host_int);
44 } else {
45 uint64 result;
46 __asm__("bswap %0" : "=r"(result) : "0"(host_int));
47 return result;
48 }
49#elif defined(bswap_64)
50 return bswap_64(host_int);
51#else
52 return static_cast<uint64>(bswap_32(static_cast<uint32>(host_int >> 32))) |
53 (static_cast<uint64>(bswap_32(static_cast<uint32>(host_int))) << 32);
54#endif // bswap_64
55}

57inline unsigned __int128 gbswap_128(unsigned __int128 host_int) {
58 return static_cast<unsigned __int128>(bswap_64(static_cast<uint64>(host_int >> 64))) |
59 (static_cast<unsigned __int128>(bswap_64(static_cast<uint64>(host_int))) << 64);
60}

62// Swap bytes of a 24-bit value.
63inline uint32_t bswap_24(uint32_t x) {
64 return ((x & 0x0000ffULL) << 16) | ((x & 0x00ff00ULL)) | ((x & 0xff0000ULL) >> 16);
65}

69// Definitions for ntohl etc. that don't require us to include
70// netinet/in.h. We wrap bswap_32 and bswap_16 in functions rather
71// than just #defining them because in debug mode, gcc doesn't
72// correctly handle the (rather involved) definitions of bswap_32.
73// gcc guarantees that inline functions are as fast as macros, so
74// this isn't a performance hit.

77}

79 return bswap_32(x);
80}

161 // Build a uint64 from 1-8 bytes.
162 // 8 * len least significant bits are loaded from the memory with
163 // LittleEndian order. The 64 - 8 * len most significant bits are
164 // set all to 0.
165 // In latex-friendly words, this function returns:
166 // $\sum_{i=0}^{len-1} p[i] 256^{i}$, where p[i] is unsigned.
167 //
168 // This function is equivalent with:
169 // uint64 val = 0;
170 // memcpy(&val, p, len);
171 // return ToHost64(val);
172 // TODO(user): write a small benchmark and benchmark the speed
173 // of a memcpy based approach.
174 //
175 // For speed reasons this function does not work for len == 0.
176 // The caller needs to guarantee that 1 <= len <= 8.
177 static uint64 Load64VariableLength(const void* const p, int len) {
178 assert(len >= 1 && len <= 8);
179 const char* const buf = static_cast<const char*>(p);
180 uint64 val = 0;
181 --len;
182 do {
183 val = (val << 8) | buf[len];
184 // (--len >= 0) is about 10 % faster than (len--) in some benchmarks.
185 } while (--len >= 0);
186 // No ToHost64(...) needed. The bytes are accessed in little-endian manner
187 // on every architecture.
188 return val;
189 }
190
191 static void Store64(void* p, uint64 v) { UNALIGNED_STORE64(p, FromHost64(v)); }
192
193 static uint128 Load128(const void* p) {
194 return uint128(ToHost64(UNALIGNED_LOAD64(reinterpret_cast<const uint64*>(p) + 1)),
195 ToHost64(UNALIGNED_LOAD64(p)));
196 }
197
198 static void Store128(void* p, const uint128 v) {
199 UNALIGNED_STORE64(p, FromHost64(Uint128Low64(v)));
200 UNALIGNED_STORE64(reinterpret_cast<uint64*>(p) + 1, FromHost64(Uint128High64(v)));
201 }
202
203 // Build a uint128 from 1-16 bytes.
204 // 8 * len least significant bits are loaded from the memory with
205 // LittleEndian order. The 128 - 8 * len most significant bits are
206 // set all to 0.
207 static uint128 Load128VariableLength(const void* p, int len) {
208 if (len <= 8) {
209 return uint128(Load64VariableLength(p, len));
210 } else {
211 return uint128(Load64VariableLength(static_cast<const char*>(p) + 8, len - 8),
212 Load64(p));
213 }
214 }
215
216 // Load & Store in machine's word size.
217 static uword_t LoadUnsignedWord(const void* p) {
218 if (sizeof(uword_t) == 8)
219 return Load64(p);
220 else
221 return Load32(p);
222 }
223
224 static void StoreUnsignedWord(void* p, uword_t v) {
225 if (sizeof(v) == 8)
226 Store64(p, v);
227 else
228 Store32(p, v);
229 }
230};

288 // Build a uint64 from 1-8 bytes.
289 // 8 * len least significant bits are loaded from the memory with
290 // BigEndian order. The 64 - 8 * len most significant bits are
291 // set all to 0.
292 // In latex-friendly words, this function returns:
293 // $\sum_{i=0}^{len-1} p[i] 256^{i}$, where p[i] is unsigned.
294 //
295 // This function is equivalent with:
296 // uint64 val = 0;
297 // memcpy(&val, p, len);
298 // return ToHost64(val);
299 // TODO(user): write a small benchmark and benchmark the speed
300 // of a memcpy based approach.
301 //
302 // For speed reasons this function does not work for len == 0.
303 // The caller needs to guarantee that 1 <= len <= 8.
304 static uint64 Load64VariableLength(const void* const p, int len) {
305 assert(len >= 1 && len <= 8);
306 uint64 val = Load64(p);
307 uint64 mask = 0;
308 --len;
309 do {
310 mask = (mask << 8) | 0xff;
311 // (--len >= 0) is about 10 % faster than (len--) in some benchmarks.
312 } while (--len >= 0);
313 return val & mask;
314 }
