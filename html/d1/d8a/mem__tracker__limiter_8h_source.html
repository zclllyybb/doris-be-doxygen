34// Track and limit the memory usage of process and query.
35// Contains an limit, arranged into a tree structure.
36//
37// Automatically track every once malloc/free of the system memory allocator (Currently, based on TCMlloc hook).
38// Put Query MemTrackerLimiter into SCOPED_ATTACH_TASK when the thread starts,all memory used by this thread
39// will be recorded on this Query, otherwise it will be recorded in Orphan Tracker by default.

41public:
42 enum Type {
43 GLOBAL = 0, // Life cycle is the same as the process, e.g. Cache and default Orphan
44 QUERY = 1, // Count the memory consumption of all Query tasks.
45 LOAD = 2, // Count the memory consumption of all Load tasks.
46 COMPACTION = 3, // Count the memory consumption of all Base and Cumulative tasks.
47 SCHEMA_CHANGE = 4, // Count the memory consumption of all SchemaChange tasks.
48 CLONE = 5, // Count the memory consumption of all EngineCloneTask. Note: Memory that does not contain make/release snapshots.
49 EXPERIMENTAL =
50 6 // Experimental memory statistics, usually inaccurate, used for debugging, and expect to add other types in the future.
51 };

80 static bool sys_mem_exceed_limit_check(int64_t bytes) {
81 if (!_oom_avoidance) {
82 return false;
83 }
84 // Limit process memory usage using the actual physical memory of the process in `/proc/self/status`.
85 // This is independent of the consumption value of the mem tracker, which counts the virtual memory
86 // of the process malloc.
87 // for fast, expect MemInfo::initialized() to be true.
88 //
89 // tcmalloc/jemalloc allocator cache does not participate in the mem check as part of the process physical memory.
90 // because `new/malloc` will trigger mem hook when using tcmalloc/jemalloc allocator cache,
91 // but it may not actually alloc physical memory, which is not expected in mem hook fail.
92 if (MemInfo::proc_mem_no_allocator_cache() + bytes >= MemInfo::mem_limit() ||
93 MemInfo::sys_mem_available() < MemInfo::sys_mem_available_low_water_mark()) {
94 print_log_process_usage(
95 fmt::format("System Mem Exceed Limit Check Faild, Try Alloc: {}", bytes));
96 return true;
97 }
98 return false;
99 }

256inline int64_t MemTrackerLimiter::add_untracked_mem(int64_t bytes) {
257 _untracked_mem += bytes;
258 if (std::abs(_untracked_mem) >= config::mem_tracker_consume_min_size_bytes) {
259 return _untracked_mem.exchange(0);
260 }
261 return 0;
262}

264inline void MemTrackerLimiter::cache_consume(int64_t bytes) {
265 if (bytes == 0) return;
266 int64_t consume_bytes = add_untracked_mem(bytes);
267 consume(consume_bytes);
268}

270inline bool MemTrackerLimiter::try_consume(int64_t bytes, std::string& failed_msg,
271 bool& is_process_exceed) {
272 if (bytes <= 0) {
273 release(-bytes);
274 failed_msg = std::string();
275 return true;
276 }
277
278 if (config::memory_debug && bytes > 1073741824) { // 1G
279 print_log_process_usage(fmt::format("Alloc Large Memory, Try Alloc: {}", bytes));
280 }
281
282 if (sys_mem_exceed_limit_check(bytes)) {
283 failed_msg = process_limit_exceeded_errmsg_str(bytes);
284 is_process_exceed = true;
285 return false;
286 }
287
288 if (_limit < 0 || (is_overcommit_tracker() && config::enable_query_memroy_overcommit)) {
289 _consumption->add(bytes); // No limit at this tracker.
290 } else {
291 if (!_consumption->try_add(bytes, _limit)) {
292 failed_msg = tracker_limit_exceeded_errmsg_str(bytes, this);
293 is_process_exceed = false;
294 return false;
295 }
296 }
297 failed_msg = std::string();
298 return true;
299}
300
301inline Status MemTrackerLimiter::check_limit(int64_t bytes) {
302 if (sys_mem_exceed_limit_check(bytes)) {
303 return Status::MemoryLimitExceeded(process_limit_exceeded_errmsg_str(bytes));
304 }
305 if (bytes <= 0 || (is_overcommit_tracker() && config::enable_query_memroy_overcommit)) {
306 return Status::OK();
307 }
308 if (_limit > 0 && _consumption->current_value() + bytes > _limit) {
309 return Status::MemoryLimitExceeded(tracker_limit_exceeded_errmsg_str(bytes, this));
310 }
311 return Status::OK();
312}
