36template <typename T>
37struct AggOrthBitmapBaseData {
38public:
39 using ColVecData = std::conditional_t<IsNumber<T>, ColumnVector<T>, ColumnString>;
40
41 void add(const IColumn** columns, size_t row_num) {
42 const auto& bitmap_col = static_cast<const ColumnBitmap&>(*columns[0]);
43 const auto& data_col = static_cast<const ColVecData&>(*columns[1]);
44 const auto& bitmap_value = bitmap_col.get_element(row_num);
45
46 if constexpr (IsNumber<T>) {
47 bitmap.update(data_col.get_element(row_num), bitmap_value);
48 }
49 if constexpr (std::is_same_v<T, std::string_view>) {
50 // TODO: rethink here we really need to do a virtual function call
51 auto sr = data_col.get_data_at(row_num);
52 bitmap.update(std::string_view {sr.data, sr.size}, bitmap_value);
53 }
54 }
55
56 void init_add_key(const IColumn** columns, size_t row_num, int argument_size) {
57 if (first_init) {
58 DCHECK(argument_size > 1);
59 for (int idx = 2; idx < argument_size; ++idx) {
60 const auto& col = static_cast<const ColVecData&>(*columns[idx]);
61 if constexpr (IsNumber<T>) {
62 bitmap.add_key(col.get_element(row_num));
63 }
64 if constexpr (std::is_same_v<T, std::string_view>) {
65 auto sr = col.get_data_at(row_num);
66 bitmap.add_key(std::string_view {sr.data, sr.size});
67 }
68 }
69 first_init = false;
70 }
71 }
72
73protected:
74 doris::BitmapIntersect<T> bitmap;
75 bool first_init = true;
76};
