35template <typename T>
36struct AggOrthBitmapBaseData {
37public:
38 using ColVecData = std::conditional_t<IsNumber<T>, ColumnVector<T>, ColumnString>;
39
40 void add(const IColumn** columns, size_t row_num) {
41 const auto& bitmap_col = static_cast<const ColumnBitmap&>(*columns[0]);
42 const auto& data_col = static_cast<const ColVecData&>(*columns[1]);
43 const auto& bitmap_value = bitmap_col.get_element(row_num);
44
45 if constexpr (IsNumber<T>) {
46 bitmap.update(data_col.get_element(row_num), bitmap_value);
47 }
48 if constexpr (std::is_same_v<T, std::string_view>) {
49 // TODO: rethink here we really need to do a virtual function call
50 auto sr = data_col.get_data_at(row_num);
51 bitmap.update(std::string_view {sr.data, sr.size}, bitmap_value);
52 }
53 }
54
55 void init_add_key(const IColumn** columns, size_t row_num, int argument_size) {
56 if (first_init) {
57 DCHECK(argument_size > 1);
58 for (int idx = 2; idx < argument_size; ++idx) {
59 const auto& col = static_cast<const ColVecData&>(*columns[idx]);
60 if constexpr (IsNumber<T>) {
61 bitmap.add_key(col.get_element(row_num));
62 }
63 if constexpr (std::is_same_v<T, std::string_view>) {
64 auto sr = col.get_data_at(row_num);
65 bitmap.add_key(std::string_view {sr.data, sr.size});
66 }
67 }
68 first_init = false;
69 }
70 }
71
72protected:
73 doris::BitmapIntersect<T> bitmap;
74 bool first_init = true;
75};

77template <typename T>
78struct AggOrthBitMapIntersect : public AggOrthBitmapBaseData<T> {
79public:
80 static constexpr auto name = "orthogonal_bitmap_intersect";
81
82 static DataTypePtr get_return_type() { return std::make_shared<DataTypeBitMap>(); }
83
84 void merge(const AggOrthBitMapIntersect& rhs) {
85 if (rhs.first_init) {
86 return;
87 }
88 result |= rhs.result;
89 }
90
91 void write(BufferWritable& buf) {
92 write_binary(AggOrthBitmapBaseData<T>::first_init, buf);
93 result = AggOrthBitmapBaseData<T>::bitmap.intersect();
94 DataTypeBitMap::serialize_as_stream(result, buf);
95 }
96
97 void read(BufferReadable& buf) {
98 read_binary(AggOrthBitmapBaseData<T>::first_init, buf);
99 DataTypeBitMap::deserialize_as_stream(result, buf);
100 }
101
102 void get(IColumn& to) const {
103 auto& column = static_cast<ColumnBitmap&>(to);
104 column.get_data().emplace_back(result);
105 }
106
107private:
108 BitmapValue result;
109};

180template <typename T>
181struct OrthBitmapUnionCountData {
182 static constexpr auto name = "orthogonal_bitmap_union_count";
183
184 static DataTypePtr get_return_type() { return std::make_shared<DataTypeInt64>(); }
185 // Here no need doing anything, so only given an function declaration
186 void init_add_key(const IColumn** columns, size_t row_num, int argument_size) {}
187
188 void add(const IColumn** columns, size_t row_num) {
189 const auto& column = static_cast<const ColumnBitmap&>(*columns[0]);
190 value |= column.get_data()[row_num];
191 }
192 void merge(const OrthBitmapUnionCountData& rhs) { result += rhs.result; }
193
194 void write(BufferWritable& buf) {
195 result = value.cardinality();
196 write_binary(result, buf);
197 }
198
199 void read(BufferReadable& buf) { read_binary(result, buf); }
200
201 void get(IColumn& to) const {
202 auto& column = static_cast<ColumnVector<Int64>&>(to);
203 column.get_data().emplace_back(result ? result : value.cardinality());
204 }
205
206private:
207 BitmapValue value;
208 int64_t result = 0;
209};

217 AggFunctionOrthBitmapFunc(const DataTypes& argument_types_)

221
222 DataTypePtr get_return_type() const override { return Impl::get_return_type(); }

240 Arena*) const override {

245 this->data(place).get(to);
246 }

248private:
