52struct creator_without_type {
53 template <bool multi_arguments, bool f, typename T>
54 using NullableT = std::conditional_t<multi_arguments, AggregateFunctionNullVariadicInline<T, f>,
55 AggregateFunctionNullUnaryInline<T, f>>;
56
57 template <typename AggregateFunctionTemplate>
58 static AggregateFunctionPtr creator(const std::string& name, const DataTypes& argument_types,
59 const bool result_is_nullable) {
60 return create<AggregateFunctionTemplate>(argument_types, result_is_nullable);
61 }
62
63 template <typename AggregateFunctionTemplate, typename... TArgs>
64 static AggregateFunctionPtr create(const DataTypes& argument_types,
65 const bool result_is_nullable, TArgs&&... args) {
66 IAggregateFunction* result(new AggregateFunctionTemplate(std::forward<TArgs>(args)...,
67 remove_nullable(argument_types)));
68 if (have_nullable(argument_types)) {
69 std::visit(
70 [&](auto multi_arguments, auto result_is_nullable) {
71 result = new NullableT<multi_arguments, result_is_nullable,
72 AggregateFunctionTemplate>(result, argument_types);
73 },
74 make_bool_variant(argument_types.size() > 1),
75 make_bool_variant(result_is_nullable));
76 }
77 return AggregateFunctionPtr(result);
78 }
79};

104template <bool allow_integer, bool allow_float, bool allow_decimal, int define_index = 0>
105struct creator_with_type_base {
106 template <typename Class, typename... TArgs>
107 static AggregateFunctionPtr create_base(const DataTypes& argument_types,
108 const bool result_is_nullable, TArgs&&... args) {
109 WhichDataType which(remove_nullable(argument_types[define_index]));
110#define DISPATCH(TYPE) \
111 if (which.idx == TypeIndex::TYPE) { \
112 return creator_without_type::create<typename Class::template T<TYPE>>( \
113 argument_types, result_is_nullable, std::forward<TArgs>(args)...); \
114 }
115
116 if constexpr (allow_integer) {
117 FOR_INTEGER_TYPES(DISPATCH);
118 }
119 if constexpr (allow_float) {
120 FOR_FLOAT_TYPES(DISPATCH);
121 }
122 if constexpr (allow_decimal) {
123 FOR_DECIMAL_TYPES(DISPATCH);
124 }
125#undef DISPATCH
126 return nullptr;
127 }
