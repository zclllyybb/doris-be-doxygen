43template <template <typename> class AggregateFunctionTemplate, typename... TArgs>
44static IAggregateFunction* create_with_numeric_type(const IDataType& argument_type,
45 TArgs&&... args) {
46 WhichDataType which(argument_type);
47#define DISPATCH(TYPE) \
48 if (which.idx == TypeIndex::TYPE) \
49 return new AggregateFunctionTemplate<TYPE>(std::forward<TArgs>(args)...);
50 FOR_NUMERIC_TYPES(DISPATCH)
51#undef DISPATCH
52 if (which.idx == TypeIndex::Enum8) {
53 return new AggregateFunctionTemplate<Int8>(std::forward<TArgs>(args)...);
54 }
55 if (which.idx == TypeIndex::Enum16) {
56 return new AggregateFunctionTemplate<Int16>(std::forward<TArgs>(args)...);
57 }
58 return nullptr;
59}

153template <template <typename> class AggregateFunctionTemplate, typename... TArgs>
154static IAggregateFunction* create_with_numeric_based_type(const IDataType& argument_type,
155 TArgs&&... args) {
156 IAggregateFunction* f = create_with_numeric_type<AggregateFunctionTemplate>(
157 argument_type, std::forward<TArgs>(args)...);
158 if (f) {
159 return f;
160 }
161
163 WhichDataType which(argument_type);
164 if (which.idx == TypeIndex::Date) {
165 return new AggregateFunctionTemplate<UInt16>(std::forward<TArgs>(args)...);
166 }
167 if (which.idx == TypeIndex::DateTime) {
168 return new AggregateFunctionTemplate<UInt32>(std::forward<TArgs>(args)...);
169 }
170 if (which.idx == TypeIndex::UUID) {
171 return new AggregateFunctionTemplate<UInt128>(std::forward<TArgs>(args)...);
172 }
173 return nullptr;
174}
