52template <typename Derived>
53class ITokenExtractorHelper : public ITokenExtractor {
54public:
55 void string_to_bloom_filter(const char* data, size_t length,
56 segment_v2::BloomFilter& bloom_filter) const override {
57 size_t cur = 0;
58 size_t token_start = 0;
59 size_t token_len = 0;
60
61 while (cur < length && static_cast<const Derived*>(this)->next_in_string(
62 data, length, &cur, &token_start, &token_len))
63 bloom_filter.add_bytes(data + token_start, token_len);
64 }
65
66 bool string_like_to_bloom_filter(const char* data, size_t length,
67 segment_v2::BloomFilter& bloom_filter) const override {
68 size_t cur = 0;
69 bool added = false;
70 std::string token;
71 while (cur < length &&
72 static_cast<const Derived*>(this)->next_in_string_like(data, length, &cur, token)) {
73 bloom_filter.add_bytes(token.data(), token.size());
74 added = true;
75 }
76
77 return added;
78 }
79};
