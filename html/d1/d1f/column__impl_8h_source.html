35template <typename Derived>
36std::vector<IColumn::MutablePtr> IColumn::scatter_impl(ColumnIndex num_columns,
37 const Selector& selector) const {
38 size_t num_rows = size();
39
40 if (num_rows != selector.size()) {
41 LOG(FATAL) << fmt::format("Size of selector: {}, doesn't match size of column:{}",
42 selector.size(), num_rows);
43 }
44
45 std::vector<MutablePtr> columns(num_columns);
46 for (auto& column : columns) column = clone_empty();
47
48 {
49 size_t reserve_size =
50 num_rows * 1.1 / num_columns;
51
52 if (reserve_size > 1)
53 for (auto& column : columns) column->reserve(reserve_size);
54 }
55
56 for (size_t i = 0; i < num_rows; ++i)
57 static_cast<Derived&>(*columns[selector[i]]).insert_from(*this, i);
58
59 return columns;
60}

62template <typename Derived>
63void IColumn::append_data_by_selector_impl(MutablePtr& res, const Selector& selector) const {
64 size_t num_rows = size();
65
66 if (num_rows < selector.size()) {
67 LOG(FATAL) << fmt::format("Size of selector: {}, is larger than size of column:{}",
68 selector.size(), num_rows);
69 }
70
71 res->reserve(num_rows);
72
73 for (size_t i = 0; i < selector.size(); ++i)
74 static_cast<Derived&>(*res).insert_from(*this, selector[i]);
75}

77template <typename Derived>
78void IColumn::get_indices_of_non_default_rows_impl(IColumn::Offsets64& indices, size_t from,
79 size_t limit) const {
80 size_t to = limit && from + limit < size() ? from + limit : size();
81 indices.reserve(indices.size() + to - from);
82 for (size_t i = from; i < to; ++i) {
83 if (!static_cast<const Derived&>(*this).is_default_at(i)) {
84 indices.push_back(i);
85 }
86 }
87}
