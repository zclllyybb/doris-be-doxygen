265 && code != ErrorCode::END_OF_FILE

267 && code != ErrorCode::TRY_LOCK_FAILED
268 && code != ErrorCode::TOO_MANY_SEGMENTS
269 && code != ErrorCode::TOO_MANY_VERSION
270 && code != ErrorCode::ALREADY_CANCELLED
271 && code != ErrorCode::ALREADY_CLOSED
272 && code != ErrorCode::PUSH_TRANSACTION_ALREADY_EXIST
273 && code != ErrorCode::BE_NO_SUITABLE_VERSION
274 && code != ErrorCode::CUMULATIVE_NO_SUITABLE_VERSION
275 && code != ErrorCode::PUBLISH_VERSION_NOT_CONTINUOUS
276 && code != ErrorCode::ROWSET_RENAME_FILE_FAILED
277 && code != ErrorCode::SEGCOMPACTION_INIT_READER
278 && code != ErrorCode::SEGCOMPACTION_INIT_WRITER
279 && code != ErrorCode::SEGCOMPACTION_FAILED
280 && code != ErrorCode::INVERTED_INDEX_INVALID_PARAMETERS
281 && code != ErrorCode::INVERTED_INDEX_NOT_SUPPORTED
