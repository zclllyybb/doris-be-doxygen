33struct InState {
34 bool use_set = true;
35
36 // only use in null in set
37 bool null_in_set = false;
38 std::unique_ptr<HybridSetBase> hybrid_set;

41template <bool negative>
42class FunctionIn : public IFunction {
43public:
44 static constexpr auto name = negative ? "not_in" : "in";
45
46 static FunctionPtr create() { return std::make_shared<FunctionIn>(); }
47
48 String get_name() const override { return name; }
49
50 bool is_variadic() const override { return true; }
51
52 size_t get_number_of_arguments() const override { return 0; }
53
54 DataTypePtr get_return_type_impl(const DataTypes& args) const override {
55 for (const auto& arg : args) {
56 if (arg->is_nullable()) {
57 return make_nullable(std::make_shared<DataTypeUInt8>());
58 }
59 }
60 return std::make_shared<DataTypeUInt8>();
61 }
62
63 bool use_default_implementation_for_nulls() const override { return false; }
64
65 Status prepare(FunctionContext* context, FunctionContext::FunctionStateScope scope) override {
66 if (scope == FunctionContext::THREAD_LOCAL) {
67 return Status::OK();
68 }
69 auto* state = new InState();
70 context->set_function_state(scope, state);
71 if (context->get_arg_type(0)->type == FunctionContext::Type::TYPE_CHAR ||
72 context->get_arg_type(0)->type == FunctionContext::Type::TYPE_VARCHAR ||
73 context->get_arg_type(0)->type == FunctionContext::Type::TYPE_STRING) {
74 // the StringValue's memory is held by FunctionContext, so we can use StringValueSet here directly
75 state->hybrid_set.reset(new StringValueSet());
76 } else {
77 state->hybrid_set.reset(
78 create_set(convert_type_to_primitive(context->get_arg_type(0)->type)));
79 }
80
81 DCHECK(context->get_num_args() >= 1);
82 for (int i = 1; i < context->get_num_args(); ++i) {
83 const auto& const_column_ptr = context->get_constant_col(i);
84 if (const_column_ptr != nullptr) {
85 auto const_data = const_column_ptr->column_ptr->get_data_at(0);
86 if (const_data.data == nullptr) {
87 state->null_in_set = true;
88 } else {
89 state->hybrid_set->insert((void*)const_data.data, const_data.size);
90 }
91 } else {
92 state->use_set = false;
93 break;
94 }
95 }
96 return Status::OK();
97 }
98
99 Status execute_impl(FunctionContext* context, Block& block, const ColumnNumbers& arguments,
100 size_t result, size_t input_rows_count) override {
101 auto in_state = reinterpret_cast<InState*>(
102 context->get_function_state(FunctionContext::FRAGMENT_LOCAL));
103 if (!in_state) {
104 return Status::RuntimeError("funciton context for function '{}' must have Set;",
105 get_name());
106 }
107 auto res = ColumnUInt8::create();
108 ColumnUInt8::Container& vec_res = res->get_data();
109 vec_res.resize(input_rows_count);
110
111 ColumnUInt8::MutablePtr col_null_map_to;
112 col_null_map_to = ColumnUInt8::create(input_rows_count, false);
113 auto& vec_null_map_to = col_null_map_to->get_data();
114
116 const ColumnWithTypeAndName& left_arg = block.get_by_position(arguments[0]);
117 auto materialized_column = left_arg.column->convert_to_full_column_if_const();
118
119 if (in_state->use_set) {
120 if (materialized_column->is_nullable()) {
121 auto* null_col_ptr = vectorized::check_and_get_column<vectorized::ColumnNullable>(
122 materialized_column);
123 auto& null_bitmap = reinterpret_cast<const vectorized::ColumnUInt8&>(
124 null_col_ptr->get_null_map_column())
125 .get_data();
126 auto* nested_col_ptr = null_col_ptr->get_nested_column_ptr().get();
127 auto search_hash_set = [&](auto* col_ptr) {
128 for (size_t i = 0; i < input_rows_count; ++i) {
129 const auto& ref_data = col_ptr->get_data_at(i);
130 vec_res[i] =
131 !null_bitmap[i] &&
132 in_state->hybrid_set->find((void*)ref_data.data, ref_data.size);
133 if constexpr (negative) {
134 vec_res[i] = !vec_res[i];
135 }
136 }
137 };
138
139 if (nested_col_ptr->is_column_string()) {
140 const auto* column_string_ptr =
141 reinterpret_cast<const vectorized::ColumnString*>(nested_col_ptr);
142 search_hash_set(column_string_ptr);
143 } else {
144 // todo support other column type
145 search_hash_set(nested_col_ptr);
146 }
147
148 if (!in_state->null_in_set) {
149 for (size_t i = 0; i < input_rows_count; ++i) {
150 vec_null_map_to[i] = null_bitmap[i];
151 }
152 } else {
153 for (size_t i = 0; i < input_rows_count; ++i) {
154 vec_null_map_to[i] = null_bitmap[i] || negative == vec_res[i];
155 }
156 }
157
158 } else { // non-nullable
159 DCHECK(!in_state->null_in_set);
160
161 auto search_hash_set = [&](auto* col_ptr) {
162 for (size_t i = 0; i < input_rows_count; ++i) {
163 const auto& ref_data = col_ptr->get_data_at(i);
164 vec_res[i] =
165 in_state->hybrid_set->find((void*)ref_data.data, ref_data.size);
166 if constexpr (negative) {
167 vec_res[i] = !vec_res[i];
168 }
169 }
170 };
171
172 if (materialized_column->is_column_string()) {
173 const auto* column_string_ptr =
174 reinterpret_cast<const vectorized::ColumnString*>(
175 materialized_column.get());
176 search_hash_set(column_string_ptr);
177 } else {
178 search_hash_set(materialized_column.get());
179 }
180 }
181 } else {
182 std::vector<ColumnPtr> set_columns;
183 for (int i = 1; i < arguments.size(); ++i) {
184 set_columns.emplace_back(block.get_by_position(arguments[i]).column);
185 }
186
187 for (size_t i = 0; i < input_rows_count; ++i) {
188 const auto& ref_data = materialized_column->get_data_at(i);
189 if (ref_data.data == nullptr) {
190 vec_null_map_to[i] = true;
191 continue;
192 }
193
194 std::unique_ptr<HybridSetBase> hybrid_set(
195 create_set(convert_type_to_primitive(context->get_arg_type(0)->type)));
196 bool null_in_set = false;
197
198 for (const auto& set_column : set_columns) {
199 auto set_data = set_column->get_data_at(i);
200 if (set_data.data == nullptr) {
201 null_in_set = true;
202 } else {
203 hybrid_set->insert((void*)(set_data.data), set_data.size);
204 }
205 }
206 vec_res[i] = negative ^ hybrid_set->find((void*)ref_data.data, ref_data.size);
207 if (null_in_set) {
208 vec_null_map_to[i] = negative == vec_res[i];
209 } else {
210 vec_null_map_to[i] = false;
211 }
212 }
213 }
214
215 if (block.get_by_position(result).type->is_nullable()) {
216 block.replace_by_position(
217 result, ColumnNullable::create(std::move(res), std::move(col_null_map_to)));
218 } else {
219 block.replace_by_position(result, std::move(res));
220 }
221
222 return Status::OK();
223 }
224
225 Status close(FunctionContext* context, FunctionContext::FunctionStateScope scope) override {
226 if (scope == FunctionContext::FRAGMENT_LOCAL) {
227 delete reinterpret_cast<InState*>(
228 context->get_function_state(FunctionContext::FRAGMENT_LOCAL));
229 }
230 return Status::OK();
231 }
232};
