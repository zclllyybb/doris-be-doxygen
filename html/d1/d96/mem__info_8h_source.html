38// Provides the amount of physical memory available.
39// Populated from /proc/meminfo.
40// TODO: Combine mem-info, cpu-info and disk-info into hardware-info/perf_counters ?
41class MemInfo {
42public:
43 // Initialize MemInfo.
44 static void init();
45
46 static inline bool initialized() { return _s_initialized; }
47
48 // Get total physical memory in bytes (if has cgroups memory limits, return the limits).
49 static inline int64_t physical_mem() {
50 DCHECK(_s_initialized);
51 return _s_physical_mem;
52 }
53
54 static void refresh_proc_meminfo();
55
56 static inline int64_t sys_mem_available() {
57 return _s_sys_mem_available - refresh_interval_memory_growth;
58 }
59 static inline std::string sys_mem_available_str() { return _s_sys_mem_available_str; }
60 static inline int64_t sys_mem_available_low_water_mark() {
61 return _s_sys_mem_available_low_water_mark;
62 }
63 static inline int64_t sys_mem_available_warning_water_mark() {
64 return _s_sys_mem_available_warning_water_mark;
65 }
66
67 static inline int64_t get_tc_metrics(const std::string& name) {
68#ifndef USE_JEMALLOC
69 size_t value = 0;
70 MallocExtension::instance()->GetNumericProperty(name.c_str(), &value);
71 return value;
72#endif
73 return 0;
74 }
75 static inline int64_t get_je_metrics(const std::string& name) {
76#ifdef USE_JEMALLOC
77 size_t value = 0;
78 size_t sz = sizeof(value);
79 if (jemallctl(name.c_str(), &value, &sz, nullptr, 0) == 0) {
80 return value;
81 }
82#endif
83 return 0;
84 }
85 static inline size_t allocator_virtual_mem() { return _s_virtual_memory_used; }
86 static inline size_t allocator_cache_mem() { return _s_allocator_cache_mem; }
87 static inline std::string allocator_cache_mem_str() { return _s_allocator_cache_mem_str; }
88 static inline int64_t proc_mem_no_allocator_cache() {
89 return _s_proc_mem_no_allocator_cache + refresh_interval_memory_growth;
90 }
91
92 // Tcmalloc property `generic.total_physical_bytes` records the total length of the virtual memory
93 // obtained by the process malloc, not the physical memory actually used by the process in the OS.
94 static void refresh_allocator_mem();
95
96 static inline void refresh_proc_mem_no_allocator_cache() {
97 _s_proc_mem_no_allocator_cache =
98 PerfCounters::get_vm_rss() - static_cast<int64_t>(_s_allocator_cache_mem);
99 refresh_interval_memory_growth = 0;
100 }
101
102 static inline int64_t mem_limit() {
103 DCHECK(_s_initialized);
104 return _s_mem_limit;
105 }
106 static inline std::string mem_limit_str() {
107 DCHECK(_s_initialized);
108 return _s_mem_limit_str;
109 }
110 static inline int64_t soft_mem_limit() {
111 DCHECK(_s_initialized);
112 return _s_soft_mem_limit;
113 }
114 static inline std::string soft_mem_limit_str() {
115 DCHECK(_s_initialized);
116 return _s_soft_mem_limit_str;
117 }
118
119 static std::string debug_string();
120
121 static void process_cache_gc(int64_t& freed_mem);
122 static bool process_minor_gc();
123 static bool process_full_gc();
124
125 // It is only used after the memory limit is exceeded. When multiple threads are waiting for the available memory of the process,
126 // avoid multiple threads starting at the same time and causing OOM.
127 static std::atomic<int64_t> refresh_interval_memory_growth;
128
129private:
130 static bool _s_initialized;
131 static int64_t _s_physical_mem;
132 static int64_t _s_mem_limit;
133 static std::string _s_mem_limit_str;
134 static int64_t _s_soft_mem_limit;
135 static std::string _s_soft_mem_limit_str;
136
137 static int64_t _s_allocator_cache_mem;
138 static std::string _s_allocator_cache_mem_str;
139 static int64_t _s_virtual_memory_used;
140 static int64_t _s_proc_mem_no_allocator_cache;
141
142 static int64_t _s_sys_mem_available;
143 static std::string _s_sys_mem_available_str;
144 static int64_t _s_sys_mem_available_low_water_mark;
145 static int64_t _s_sys_mem_available_warning_water_mark;
146 static int64_t _s_process_minor_gc_size;
147 static int64_t _s_process_full_gc_size;
148};
