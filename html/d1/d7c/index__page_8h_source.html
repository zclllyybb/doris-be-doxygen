34// IndexPage is the building block for IndexedColumn's ordinal index and value index.
35// It is used to guide searching for a particular key to the data page containing it.
36// We use the same general format for all index pages, regardless of the data type and node type (leaf or internal)
37// IndexPageBody := IndexEntry^NumEntry
38// IndexEntry := KeyLength(vint), Byte^KeyLength, PageOffset(vlong), PageSize(vint)
39//
40// IndexPageFooterPB records NumEntry and type (leaf/internal) of the index page.
41// For leaf, IndexKey records the first/smallest key of the data page PagePointer points to.
42// For internal, IndexKey records the first/smallest key of the next-level index page PagePointer points to.
43//
44// All keys are treated as binary string and compared with memcpy. Keys of other data type are encoded first by
45// KeyCoder, e.g., ordinal index's original key type is uint64_t but is encoded to binary string.
46class IndexPageBuilder {
47public:
48 explicit IndexPageBuilder(size_t index_page_size, bool is_leaf)
49 : _index_page_size(index_page_size), _is_leaf(is_leaf) {}
50
51 void add(const Slice& key, const PagePointer& ptr);
52
53 bool is_full() const;
54
55 size_t count() const { return _count; }
56
57 void finish(OwnedSlice* body, PageFooterPB* footer);
58
59 uint64_t size() { return _buffer.size(); }
60
61 // Return the key of the first entry in this index block.
62 // The pointed-to data is only valid until the next call to this builder.
63 Status get_first_key(Slice* key) const;
64
65 void reset() {
66 _finished = false;
67 _buffer.clear();
68 _count = 0;
69 }
70
71private:
72 DISALLOW_COPY_AND_ASSIGN(IndexPageBuilder);
73 const size_t _index_page_size;
74 const bool _is_leaf;
75 bool _finished = false;
76 faststring _buffer;
77 uint32_t _count = 0;
78};

119class IndexPageIterator {
120public:
121 explicit IndexPageIterator(const IndexPageReader* reader) : _reader(reader), _pos(0) {}
122
123 // Find the largest index entry whose key is <= search_key.
124 // Return OK status when such entry exists.
125 // Return NotFound when no such entry is found (all keys > search_key).
126 // Return other error status otherwise.
127 Status seek_at_or_before(const Slice& search_key);
128
129 void seek_to_first() { _pos = 0; }
130
131 // Move to the next index entry.
132 // Return true on success, false when no more entries can be read.
133 bool move_next() {
134 _pos++;
135 if (_pos >= _reader->count()) {
136 return false;
137 }
138 return true;
139 }
140
141 // Return true when has next page.
142 bool has_next() { return (_pos + 1) < _reader->count(); }
143
144 const Slice& current_key() const { return _reader->get_key(_pos); }
145
146 const PagePointer& current_page_pointer() const { return _reader->get_value(_pos); }
147
148private:
149 const IndexPageReader* _reader;
150
151 size_t _pos;
152};
