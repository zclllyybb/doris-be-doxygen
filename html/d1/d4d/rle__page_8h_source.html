32// RLE builder for generic integer and bool types. What is missing is some way
33// to enforce that this can only be instantiated for INT and BOOL types.
34//
35// The page format is as follows:
36//
37// 1. Header: (4 bytes total)
38//
39// <num_elements> [32-bit]
40// The number of elements encoded in the page.
41//
42// NOTE: all on-disk ints are encoded little-endian
43//
44// 2. Element data
45//
46// The header is followed by the rle-encoded element data.
47//
48// This Rle encoding algorithm is only effective for repeated INT type and bool type,
49// It is not good for sequence number or random number. BitshufflePage is recommended
50// for these case.
51//
52// TODO(hkp): optimize rle algorithm
53template <FieldType Type>
54class RlePageBuilder : public PageBuilder {
55public:
56 RlePageBuilder(const PageBuilderOptions& options)
57 : _options(options), _count(0), _finished(false), _bit_width(0), _rle_encoder(nullptr) {
58 switch (Type) {
59 case OLAP_FIELD_TYPE_BOOL: {
60 _bit_width = 1;
61 break;
62 }
63 default: {
64 _bit_width = SIZE_OF_TYPE * 8;
65 break;
66 }
67 }
68 _rle_encoder = new RleEncoder<CppType>(&_buf, _bit_width);
69 reset();
70 }
71
72 ~RlePageBuilder() { delete _rle_encoder; }
73
74 bool is_page_full() override { return _rle_encoder->len() >= _options.data_page_size; }
75
76 Status add(const uint8_t* vals, size_t* count) override {
77 DCHECK(!_finished);
78 auto new_vals = reinterpret_cast<const CppType*>(vals);
79 for (int i = 0; i < *count; ++i) {
80 // note: vals is not guaranteed to be aligned for now, thus memcpy here
81 CppType value;
82 memcpy(&value, &new_vals[i], SIZE_OF_TYPE);
83 _rle_encoder->Put(value);
84 }
85
86 if (_count == 0) {
87 memcpy(&_first_value, new_vals, SIZE_OF_TYPE);
88 }
89 memcpy(&_last_value, &new_vals[*count - 1], SIZE_OF_TYPE);
90
91 _count += *count;
92 return Status::OK();
93 }
94
95 OwnedSlice finish() override {
96 DCHECK(!_finished);
97 _finished = true;
98 // here should Flush first and then encode the count header
99 // or it will lead to a bug if the header is less than 8 byte and the data is small
100 _rle_encoder->Flush();
101 encode_fixed32_le(&_buf[0], _count);
102 return _buf.build();
103 }
104
105 void reset() override {
106 _count = 0;
107 _finished = false;
108 _rle_encoder->Clear();
109 _rle_encoder->Reserve(RLE_PAGE_HEADER_SIZE, 0);
110 }
111
112 size_t count() const override { return _count; }
113
114 uint64_t size() const override { return _rle_encoder->len(); }
115
116 Status get_first_value(void* value) const override {
117 DCHECK(_finished);
118 if (_count == 0) {
119 return Status::NotFound("page is empty");
120 }
121 memcpy(value, &_first_value, SIZE_OF_TYPE);
122 return Status::OK();
123 }
124
125 Status get_last_value(void* value) const override {
126 DCHECK(_finished);
127 if (_count == 0) {
128 return Status::NotFound("page is empty");
129 }
130 memcpy(value, &_last_value, SIZE_OF_TYPE);
131 return Status::OK();
132 }
133
134private:
135 typedef typename TypeTraits<Type>::CppType CppType;
136 enum { SIZE_OF_TYPE = TypeTraits<Type>::size };
137
138 PageBuilderOptions _options;
139 size_t _count;
140 bool _finished;
141 int _bit_width;
142 RleEncoder<CppType>* _rle_encoder;
143 faststring _buf;
144 CppType _first_value;
145 CppType _last_value;
146};

148template <FieldType Type>
149class RlePageDecoder : public PageDecoder {
150public:
151 RlePageDecoder(Slice slice, const PageDecoderOptions& options)
152 : _data(slice),
153 _options(options),
154 _parsed(false),
155 _num_elements(0),
156 _cur_index(0),
157 _bit_width(0) {}
158
159 Status init() override {
160 CHECK(!_parsed);
161
162 if (_data.size < RLE_PAGE_HEADER_SIZE) {
163 return Status::Corruption("not enough bytes for header in RleBitMapBlockDecoder");
164 }
165 _num_elements = decode_fixed32_le((const uint8_t*)&_data[0]);
166
167 _parsed = true;
168
169 switch (Type) {
170 case OLAP_FIELD_TYPE_BOOL: {
171 _bit_width = 1;
172 break;
173 }
174 default: {
175 _bit_width = SIZE_OF_TYPE * 8;
176 break;
177 }
178 }
179
180 _rle_decoder = RleDecoder<CppType>((uint8_t*)_data.data + RLE_PAGE_HEADER_SIZE,
181 _data.size - RLE_PAGE_HEADER_SIZE, _bit_width);
182
183 seek_to_position_in_page(0);
184 return Status::OK();
185 }
186
187 Status seek_to_position_in_page(size_t pos) override {
188 DCHECK(_parsed) << "Must call init()";
189 DCHECK_LE(pos, _num_elements)
190 << "Tried to seek to " << pos << " which is > number of elements (" << _num_elements
191 << ") in the block!";
192 // If the block is empty (e.g. the column is filled with nulls), there is no data to seek.
193 if (PREDICT_FALSE(_num_elements == 0)) {
194 return Status::OK();
195 }
196 if (_cur_index == pos) {
197 // No need to seek.
198 return Status::OK();
199 } else if (_cur_index < pos) {
200 uint nskip = pos - _cur_index;
201 _rle_decoder.Skip(nskip);
202 } else {
203 _rle_decoder = RleDecoder<CppType>((uint8_t*)_data.data + RLE_PAGE_HEADER_SIZE,
204 _data.size - RLE_PAGE_HEADER_SIZE, _bit_width);
205 _rle_decoder.Skip(pos);
206 }
207 _cur_index = pos;
208 return Status::OK();
209 }
210
211 Status next_batch(size_t* n, vectorized::MutableColumnPtr& dst) override {
212 DCHECK(_parsed);
213 if (PREDICT_FALSE(*n == 0 || _cur_index >= _num_elements)) {
214 *n = 0;
215 return Status::OK();
216 }
217
218 size_t to_fetch = std::min(*n, static_cast<size_t>(_num_elements - _cur_index));
219 size_t remaining = to_fetch;
220 bool result = false;
221 CppType value;
222 while (remaining > 0) {
223 result = _rle_decoder.Get(&value);
224 DCHECK(result);
225 dst->insert_data((char*)(&value), SIZE_OF_TYPE);
226 remaining--;
227 }
228
229 _cur_index += to_fetch;
230 *n = to_fetch;
231 return Status::OK();
232 }
233
234 Status read_by_rowids(const rowid_t* rowids, ordinal_t page_first_ordinal, size_t* n,
235 vectorized::MutableColumnPtr& dst) override {
236 DCHECK(_parsed);
237 if (PREDICT_FALSE(*n == 0 || _cur_index >= _num_elements)) {
238 *n = 0;
239 return Status::OK();
240 }
241
242 auto total = *n;
243 bool result = false;
244 size_t read_count = 0;
245 CppType value;
246 for (size_t i = 0; i < total; ++i) {
247 ordinal_t ord = rowids[i] - page_first_ordinal;
248 if (UNLIKELY(ord >= _num_elements)) {
249 *n = read_count;
250 return Status::OK();
251 }
252
253 _rle_decoder.Skip(ord - _cur_index);
254 _cur_index = ord;
255
256 result = _rle_decoder.Get(&value);
257 _cur_index++;
258 DCHECK(result);
259 dst->insert_data((char*)(&value), SIZE_OF_TYPE);
260 read_count++;
261 }
262 *n = read_count;
263 return Status::OK();
264 }
265
266 size_t count() const override { return _num_elements; }
267
268 size_t current_index() const override { return _cur_index; }
269
270private:
271 typedef typename TypeTraits<Type>::CppType CppType;
272 enum { SIZE_OF_TYPE = TypeTraits<Type>::size };
273
274 Slice _data;
275 PageDecoderOptions _options;
276 bool _parsed;
277 uint32_t _num_elements;
278 size_t _cur_index;
279 int _bit_width;
280 RleDecoder<CppType> _rle_decoder;
281};
