39template <typename RESULT, typename INTERMEDIATE>
40class UdaTestHarnessBase {
41public:
42 typedef void (*InitFn)(FunctionContext* context, INTERMEDIATE* result);
43
44 typedef void (*MergeFn)(FunctionContext* context, const INTERMEDIATE& src, INTERMEDIATE* dst);
45
46 typedef const INTERMEDIATE (*SerializeFn)(FunctionContext* context, const INTERMEDIATE& type);
47
48 typedef RESULT (*FinalizeFn)(FunctionContext* context, const INTERMEDIATE& value);
49
50 // UDA test harness allows for custom comparator to validate results. UDAs
51 // can specify a custom comparator to, for example, tolerate numerical imprecision.
52 // Returns true if x and y should be treated as equal.
53 typedef bool (*ResultComparator)(const RESULT& x, const RESULT& y);
54
55 void set_result_comparator(ResultComparator fn) { _result_comparator_fn = fn; }
56
57 // This must be called if the INTERMEDIATE is TYPE_FIXED_BUFFER
58 void set_intermediate_size(int byte_size) { _fixed_buffer_byte_size = byte_size; }
59
60 // Returns the failure string if any.
61 const std::string& get_error_msg() const { return _error_msg; }
62
63protected:
64 UdaTestHarnessBase(InitFn init_fn, MergeFn merge_fn, SerializeFn serialize_fn,
65 FinalizeFn finalize_fn)
66 : _init_fn(init_fn),
67 _merge_fn(merge_fn),
68 _serialize_fn(serialize_fn),
69 _finalize_fn(finalize_fn),
70 _result_comparator_fn(nullptr),
71 _num_input_values(0) {}
72
73 // Runs the UDA in all the modes, validating the result is 'expected' each time.
74 bool execute(const RESULT& expected, UdaExecutionMode mode);
75
76 // Returns false if there is an error set in the context.
77 bool check_context(FunctionContext* context);
78
79 // Verifies x == y, using the custom comparator if set.
80 bool check_result(const RESULT& x, const RESULT& y);
81
82 // Runs the UDA on a single node. The entire execution happens in 1 context.
83 // The UDA does a update on all the input values and then a finalize.
84 RESULT execute_single_node();
85
86 // Runs the UDA, simulating a single level aggregation. The values are processed
87 // on num_nodes + 1 contexts. There are num_nodes that do update and serialize.
88 // There is a final context that does merge and finalize.
89 RESULT execute_one_level(int num_nodes);
90
91 // Runs the UDA, simulating a two level aggregation with num1 in the first level and
92 // num2 in the second. The values are processed in num1 + num2 contexts.
93 RESULT execute_two_level(int num1, int num2);
94
95 virtual void update(int idx, FunctionContext* context, INTERMEDIATE* dst) = 0;
96
97private:
98 // UDA functions
99 InitFn _init_fn;
100 MergeFn _merge_fn;
101 SerializeFn _serialize_fn;
102 FinalizeFn _finalize_fn;
103
104 // Customer comparator, nullptr if default == should be used.
105 ResultComparator _result_comparator_fn;
106
107 // Set during execute() by subclass
108 int _num_input_values;
109
110 // Buffer len for intermediate results if the type is TYPE_FIXED_BUFFER
111 int _fixed_buffer_byte_size;
112
113 // Error message if anything went wrong during the execution.
114 std::string _error_msg;
115};

117template <typename RESULT, typename INTERMEDIATE, typename INPUT>
118class UdaTestHarness : public UdaTestHarnessBase<RESULT, INTERMEDIATE> {
119public:
120 typedef void (*UpdateFn)(FunctionContext* context, const INPUT& input, INTERMEDIATE* result);
121
122 typedef UdaTestHarnessBase<RESULT, INTERMEDIATE> BaseClass;
123
124 UdaTestHarness(typename BaseClass::InitFn init_fn, UpdateFn update_fn,
125 typename BaseClass::MergeFn merge_fn,
126 typename BaseClass::SerializeFn serialize_fn,
127 typename BaseClass::FinalizeFn finalize_fn)
128 : BaseClass(init_fn, merge_fn, serialize_fn, finalize_fn), _update_fn(update_fn) {}
129
130 bool execute(const std::vector<INPUT>& values, const RESULT& expected) {
131 return execute(values, expected, ALL);
132 }
133 // Runs the UDA in all the modes, validating the result is 'expected' each time.
134 bool execute(const std::vector<INPUT>& values, const RESULT& expected, UdaExecutionMode mode);
135
136 // Runs the UDA in all the modes, validating the result is 'expected' each time.
137 // T needs to be compatible (i.e. castable to) with INPUT
138 template <typename T>
139 bool execute(const std::vector<T>& values, const RESULT& expected) {
140 return execute(values, expected, ALL);
141 }
142 template <typename T>
143 bool execute(const std::vector<T>& values, const RESULT& expected, UdaExecutionMode mode) {
144 _input.resize(values.size());
145 BaseClass::_num_input_values = _input.size();
146
147 for (int i = 0; i < values.size(); ++i) {
148 _input[i] = &values[i];
149 }
150
151 return BaseClass::execute(expected, mode);
152 }
153
154protected:
155 virtual void update(int idx, FunctionContext* context, INTERMEDIATE* dst);
156
157private:
158 UpdateFn _update_fn;
159 // Set during execute()
160 std::vector<const INPUT*> _input;
161};
