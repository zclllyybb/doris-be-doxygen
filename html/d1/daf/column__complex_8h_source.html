68 void insert_binary_data(const char* pos, size_t length) {
69 insert_default();
70 T* pvalue = &get_element(size() - 1);
71 if (!length) {
72 *pvalue = *reinterpret_cast<const T*>(pos);
73 return;
74 }
75
76 if constexpr (std::is_same_v<T, BitmapValue>) {
77 pvalue->deserialize(pos);
78 } else if constexpr (std::is_same_v<T, HyperLogLog>) {
79 pvalue->deserialize(Slice(pos, length));
80 } else if constexpr (std::is_same_v<T, QuantileStateDouble>) {
81 pvalue->deserialize(Slice(pos, length));

83 LOG(FATAL) << "Unexpected type in column complex";
84 }

312template <typename T>
313ColumnPtr ColumnComplexType<T>::filter(const IColumn::Filter& filt,
314 ssize_t result_size_hint) const {
315 size_t size = data.size();
316 if (size != filt.size()) {
317 LOG(FATAL) << "Size of filter doesn't match size of column.";
318 }
319
320 if (data.size() == 0) return this->create();
321 auto res = this->create();
322 Container& res_data = res->get_data();
323
324 if (result_size_hint) res_data.reserve(result_size_hint > 0 ? result_size_hint : size);
325
326 const UInt8* filt_pos = filt.data();
327 const UInt8* filt_end = filt_pos + size;
328 const T* data_pos = data.data();
329
330 while (filt_pos < filt_end) {
331 if (*filt_pos) res_data.push_back(*data_pos);
332
333 ++filt_pos;
334 ++data_pos;
335 }
336
337 return res;
338}

340template <typename T>
341size_t ColumnComplexType<T>::filter(const IColumn::Filter& filter) {
342 size_t size = data.size();
343 if (size != filter.size()) {
344 LOG(FATAL) << "Size of filter doesn't match size of column.";
345 }
346
347 if (data.size() == 0) {
348 return 0;
349 }
350
351 T* res_data = data.data();
352
353 const UInt8* filter_pos = filter.data();
354 const UInt8* filter_end = filter_pos + size;
355 const T* data_pos = data.data();
356
357 while (filter_pos < filter_end) {
358 if (*filter_pos) {
359 *res_data = std::move(*data_pos);
360 ++res_data;
361 }
362
363 ++filter_pos;
364 ++data_pos;
365 }
366
367 data.resize(res_data - data.data());
368
369 return res_data - data.data();
370}

372template <typename T>
373ColumnPtr ColumnComplexType<T>::permute(const IColumn::Permutation& perm, size_t limit) const {
374 size_t size = data.size();
375
376 limit = limit ? std::min(size, limit) : size;
377
378 if (perm.size() < limit) {
379 LOG(FATAL) << "Size of permutation is less than required.";
380 }
381
382 auto res = this->create(limit);
383 typename Self::Container& res_data = res->get_data();
384 for (size_t i = 0; i < limit; ++i) {
385 res_data[i] = data[perm[i]];
386 }
387
388 return res;
389}

391template <typename T>
392ColumnPtr ColumnComplexType<T>::replicate(const IColumn::Offsets& offsets) const {
393 size_t size = data.size();
394 if (size != offsets.size()) {
395 LOG(FATAL) << "Size of offsets doesn't match size of column.";
396 }
397
398 if (0 == size) return this->create();
399
400 auto res = this->create();
401 typename Self::Container& res_data = res->get_data();
402 res_data.reserve(offsets.back());
403
404 IColumn::Offset prev_offset = 0;
405 for (size_t i = 0; i < size; ++i) {
406 size_t size_to_replicate = offsets[i] - prev_offset;
407 prev_offset = offsets[i];
408
409 for (size_t j = 0; j < size_to_replicate; ++j) {
410 res_data.push_back(data[i]);
411 }
412 }
413
414 return res;
415}

417template <typename T>
418void ColumnComplexType<T>::replicate(const uint32_t* counts, size_t target_size, IColumn& column,
419 size_t begin, int count_sz) const {
420 size_t size = count_sz < 0 ? data.size() : count_sz;
421 if (0 == size) return;
422
423 auto& res = reinterpret_cast<ColumnComplexType<T>&>(column);
424 typename Self::Container& res_data = res.get_data();
425 res_data.reserve(target_size);
426
427 size_t end = size + begin;
428 for (size_t i = begin; i < end; ++i) {
429 size_t size_to_replicate = counts[i];
430 for (size_t j = 0; j < size_to_replicate; ++j) {
431 res_data.push_back(data[i]);
432 }
433 }
