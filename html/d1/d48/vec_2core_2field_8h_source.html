277 }

654 template <typename T>
655 void create_concrete(T&& x) {
656 using UnqualifiedType = std::decay_t<T>;
657
658 // In both Field and PODArray, small types may be stored as wider types,
659 // e.g. char is stored as UInt64. Field can return this extended value
660 // with get<StorageType>(). To avoid uninitialized results from get(),
661 // we must initialize the entire wide stored type, and not just the
662 // nominal type.
663 using StorageType = NearestFieldType<UnqualifiedType>;
664 new (&storage) StorageType(std::forward<T>(x));
665 which = TypeToEnum<UnqualifiedType>::value;
666 }

674 *ptr = std::forward<T>(x);
