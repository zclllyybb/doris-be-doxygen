212template <typename T>
213class DecimalField {
214public:
215 DecimalField(T value, UInt32 scale_) : dec(value), scale(scale_) {}
216
217 operator T() const { return dec; }
218 T get_value() const { return dec; }
219 T get_scale_multiplier() const;
220 UInt32 get_scale() const { return scale; }
221
222 template <typename U>
223 bool operator<(const DecimalField<U>& r) const {
224 using MaxType = std::conditional_t<(sizeof(T) > sizeof(U)), T, U>;
225 return decimal_less<MaxType>(dec, r.get_value(), scale, r.get_scale());
226 }
227
228 template <typename U>
229 bool operator<=(const DecimalField<U>& r) const {
230 using MaxType = std::conditional_t<(sizeof(T) > sizeof(U)), T, U>;
231 return decimal_less_or_equal<MaxType>(dec, r.get_value(), scale, r.get_scale());
232 }
233
234 template <typename U>
235 bool operator==(const DecimalField<U>& r) const {
236 using MaxType = std::conditional_t<(sizeof(T) > sizeof(U)), T, U>;
237 return decimal_equal<MaxType>(dec, r.get_value(), scale, r.get_scale());
238 }
239
240 template <typename U>
241 bool operator>(const DecimalField<U>& r) const {
242 return r < *this;
243 }
244 template <typename U>
245 bool operator>=(const DecimalField<U>& r) const {
246 return r <= *this;
247 }
248 template <typename U>
249 bool operator!=(const DecimalField<U>& r) const {
250 return !(*this == r);
251 }
252
253 const DecimalField<T>& operator+=(const DecimalField<T>& r) {
254 if (scale != r.get_scale()) {
255 LOG(FATAL) << "Add different decimal fields";
256 }
257 dec += r.get_value();
258 return *this;
259 }
260
261 const DecimalField<T>& operator-=(const DecimalField<T>& r) {
262 if (scale != r.get_scale()) {
263 LOG(FATAL) << "Sub different decimal fields";
264 }
265 dec -= r.get_value();
266 return *this;
267 }
268
269private:
270 T dec;
271 UInt32 scale;
272};

626 template <typename T>
627 void create_concrete(T&& x) {
628 using UnqualifiedType = std::decay_t<T>;
629
630 // In both Field and PODArray, small types may be stored as wider types,
631 // e.g. char is stored as UInt64. Field can return this extended value
632 // with get<StorageType>(). To avoid uninitialized results from get(),
633 // we must initialize the entire wide stored type, and not just the
634 // nominal type.
635 using StorageType = NearestFieldType<UnqualifiedType>;
636 new (&storage) StorageType(std::forward<T>(x));
637 which = TypeToEnum<UnqualifiedType>::value;
638 }

641 template <typename T>
642 void assign_concrete(T&& x) {
643 using JustT = std::decay_t<T>;
644 assert(which == TypeToEnum<JustT>::value);
645 JustT* MAY_ALIAS ptr = reinterpret_cast<JustT*>(&storage);
646 *ptr = std::forward<T>(x);
647 }

1076template <typename T>
1077decltype(auto) cast_to_nearest_field_type(T&& x) {
1078 using U = NearestFieldType<std::decay_t<T>>;
1079 if constexpr (std::is_same_v<PackedInt128, std::decay_t<T>>) {
1080 return U(x.value);
1081 } else if constexpr (std::is_same_v<std::decay_t<T>, U>) {
1082 return std::forward<T>(x);
1083 } else {
1084 return U(x);
1085 }
1086}

1095template <typename T>
1096Field::Field(T&& rhs, std::enable_if_t<!std::is_same_v<std::decay_t<T>, Field>, void*>) {
1097 auto&& val = cast_to_nearest_field_type(std::forward<T>(rhs));
1098 create_concrete(std::forward<decltype(val)>(val));
1099}

1101template <typename T>
1102std::enable_if_t<!std::is_same_v<std::decay_t<T>, Field>, Field&> Field::operator=(T&& rhs) {
1103 auto&& val = cast_to_nearest_field_type(std::forward<T>(rhs));
1104 using U = decltype(val);
1105 if (which != TypeToEnum<std::decay_t<U>>::value) {
1106 destroy();
1107 create_concrete(std::forward<U>(val));
1108 } else
1109 assign_concrete(std::forward<U>(val));
1110
1111 return *this;
1112}
